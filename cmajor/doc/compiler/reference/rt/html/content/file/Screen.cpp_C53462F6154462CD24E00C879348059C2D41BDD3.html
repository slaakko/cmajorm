240 extern "C" int RtTranslateCh(int ch)
241 {
242     using cmajor::rt::keyMap;
243     auto it = keyMap.find(ch);
244     if (it != keyMap.cend())
245     {
246         ch = it->second;
247     }
248     return ch;
249 }
