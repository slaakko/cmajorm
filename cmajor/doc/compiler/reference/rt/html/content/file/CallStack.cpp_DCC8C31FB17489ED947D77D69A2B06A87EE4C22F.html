 65 extern "C" void RtEnterFunction(const char* functionName, const char* sourceFilePath)
 66 {
 67     cmajor::rt::CallStack* callStack = cmajor::rt::callStack;
 68     if (!callStack)
 69     {
 70         callStack = new cmajor::rt::CallStack();
 71         cmajor::rt::callStack = callStack;
 72     }
 73     callStack->Locations().push_back(cmajor::rt::SourceLocation(functionName, sourceFilePath));
 74 }

 76 extern "C" void RtSetLineNumber(int32_t lineNumber)
 77 {
 78     try
 79     {
 80         cmajor::rt::CallStack* callStack = cmajor::rt::callStack;
 81         Assert(callStack && !callStack->Locations().empty(), "call stack is empty");
 82         callStack->Locations().back().lineNumber = lineNumber;
 83     }
 84     catch (const std::exception& ex)
 85     {
 86         std::stringstream s;
 87         s << "internal error: " << ex.what() << "\n";
 88         std::string str = s.str();
 89         RtWrite(stdErrFileHandle, reinterpret_cast<const uint8_t*>(str.c_str()), str.length());
 90         exit(exitCodeInternalError);
 91     }
 92 }

 94 extern "C" void RtExitFunction()
 95 {

 98         cmajor::rt::CallStack* callStack = cmajor::rt::callStack;
 99         Assert(callStack && !callStack->Locations().empty(), "call stack is empty");

102     catch (const std::exception& ex)

106         std::string str = s.str();
107         RtWrite(stdErrFileHandle, reinterpret_cast<const uint8_t*>(str.c_str()), str.length());
108         exit(exitCodeInternalError);
109     }

112 extern "C" void RtPrintCallStack(int fileHandle)
113 {
114     std::stringstream s;
115     cmajor::rt::CallStack* callStack = cmajor::rt::callStack;
116     if (!callStack)
117     {
118         callStack = new cmajor::rt::CallStack();
119         cmajor::rt::callStack = callStack;
120     }
121     s << "CALL STACK:\n";
122     int n = callStack->Locations().size();
123     for (int i = n - 1; i >= 0; --i)
124     {
125         const cmajor::rt::SourceLocation& location = callStack->Locations()[i];
126         s << location.functionName << " " << location.sourceFilePath << ":" << location.lineNumber << "\n";
127     }
128     std::string str = s.str();
129     RtWrite(stdErrFileHandle, reinterpret_cast<const uint8_t*>(str.c_str()), str.length());
130 }

132 extern "C" const char* RtGetStackTrace()
133 {
134     std::stringstream s;
135     cmajor::rt::CallStack* callStack = cmajor::rt::callStack;
136     if (!callStack)
137     {
138         callStack = new cmajor::rt::CallStack();
139         cmajor::rt::callStack = callStack;
140     }
141     int n = callStack->Locations().size();
142     for (int i = n - 1; i >= 0; --i)
143     {
144         const cmajor::rt::SourceLocation& location = callStack->Locations()[i];
145         s << location.functionName << " " << location.sourceFilePath << ":" << location.lineNumber << "\n";
146     }
147     cmajor::rt::stackTrace = new std::string(s.str());
148     return cmajor::rt::stackTrace->c_str();
149 }
