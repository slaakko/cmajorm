15 enum class OpenMode : uint8_t 
16 {
17     none=  0, 
18     read=  1 << 0, 
19     write=  1 << 1, 
20     append=  1 << 2, 
21     binary=  1 << 3
22 };
23 
24 enum class Origin : uint8_t 
25 {
26     seekSet, seekCur, seekEnd
27 };
28 
29 inline OpenMode operator&(OpenMode left, OpenMode right)
30 {
31     return OpenMode(uint8_t(left) & uint8_t(right));
32 }
33 
34 inline OpenMode operator|(OpenMode left, OpenMode right)
35 {
36     return OpenMode(uint8_t(left) | uint8_t(right));
37 }
