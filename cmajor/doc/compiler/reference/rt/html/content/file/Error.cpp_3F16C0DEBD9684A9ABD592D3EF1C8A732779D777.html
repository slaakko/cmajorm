 19 class ErrorTable 
 20 {
 21 public:
 22     static void Init();
 23     static void Done();
 24     static ErrorTable& Instance() { Assert(instance, "error table not initialized"); return *instance; }
 25     int32_t InstallError(const std::string& errorMessage);
 26     const char* GetError(int32_t errorId);
 27     void DisposeError(int32_t errorId);
 28 private:
 29     static std::unique_ptr<ErrorTable> instance;
 30     std::mutex mtx;
 31     std::unordered_map<int32_t, std::string> errorMap;
 32     int32_t nextErrorId;
 33     ErrorTable();
 34 };
 35 
 36 std::unique_ptr<ErrorTable> ErrorTable::instance;
 37 
 38 void ErrorTable::Init()
 39 {
 40     instance.reset(new ErrorTable());
 41 }
 42 
 43 void ErrorTable::Done()
 44 {
 45     instance.reset();
 46 }
 47 
 48 ErrorTable::ErrorTable() : nextErrorId(1)
 49 {
 50 }
 51 
 52 int32_t ErrorTable::InstallError(const std::string& errorMessage)
 53 {
 54     std::lock_guard<std::mutex> lock(mtx);
 55     int32_t errorId = nextErrorId++;
 56     errorMap[errorId] = errorMessage;
 57     return errorId;
 58 }
 59 
 60 const char* ErrorTable::GetError(int32_t errorId)
 61 {
 62     std::lock_guard<std::mutex> lock(mtx);
 63     auto it = errorMap.find(errorId);
 64     if (it != errorMap.cend())
 65     {
 66         return it->second.c_str();
 67     }
 68     else
 69     {
 70         return "<error message not found>";
 71     }
 72 }
 73 
 74 void ErrorTable::DisposeError(int32_t errorId)
 75 {
 76     std::lock_guard<std::mutex> lock(mtx);
 77     errorMap.erase(errorId);
 78 }

119 extern "C" void RtFailAssertion(const char* assertion, const char* function, const char* sourceFilePath, int lineNumber)
120 {
121     std::stringstream s;
122     s << "assertion '" << assertion << "' failed in function '" << function << "' at " << sourceFilePath << ":" << lineNumber << "\n";
123     std::string str = s.str();
124     RtWrite(stdErrFileHandle, reinterpret_cast<const uint8_t*>(str.c_str()), str.length());
125     RtPrintCallStack(stdErrFileHandle);
126     exit(exitCodeAssertionFailed);
127 }

139 extern "C" void RtThrowException(void* exception, void* exceptionTypeId)
140 {
141     cmajor::rt::currentException = exception;
142     boost::uuids::uuid* exTypeId = reinterpret_cast<boost::uuids::uuid*>(exceptionTypeId);
143     uint64_t exceptionClassId = cmajor::rt::GetClassId(*exTypeId);
144     cmajor::rt::currentExceptionClassId = exceptionClassId;
145     throw cmajor::eh::Exception();
146 }

148 extern "C" void RtCaptureException(void** exception, uint64_t* exceptionClassId)
149 {
150     *exception = cmajor::rt::currentException;
151     cmajor::rt::currentException = nullptr;
152     *exceptionClassId = cmajor::rt::currentExceptionClassId;
153     cmajor::rt::currentExceptionClassId = 0;
154 }

157 {
158     cmajor::rt::currentException = exception;
159     cmajor::rt::currentExceptionClassId = exceptionClassId;
160     throw cmajor::eh::Exception();
161 }

163 extern "C" bool RtHandleException(void* exceptionTypeId)
164 {
165     uint64_t currentExceptionClassId = cmajor::rt::currentExceptionClassId;
166     boost::uuids::uuid* exTypeId = reinterpret_cast<boost::uuids::uuid*>(exceptionTypeId);
167     uint64_t exceptionClassId = cmajor::rt::GetClassId(*exTypeId);
168     bool handle = currentExceptionClassId % exceptionClassId == 0;
169     return handle;
170 }
