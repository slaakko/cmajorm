 16 class MutexTable 
 17 {
 18 public:
 19     static void Init();
 20     static void Done();
 21     static MutexTable& Instance() { Assert(instance, "mutex table not initialized"); return *instance; }
 22     int32_t AllocateMutex();
 23     void FreeMutex(int32_t mutexId);
 24     void LockMutex(int32_t mutexId);
 25     void UnlockMutex(int32_t mutexId);
 26 private:
 27     MutexTable();
 28     static std::unique_ptr<MutexTable> instance;
 29     const int32_t numNoLockMutexes = 256;
 30     std::atomic<int32_t> nextMutexId;
 31     std::vector<std::std::unique_ptr<std::mutex>>noLockMutexes;
 32     std::unordered_map<int32_t, std::std::unique_ptr<std::mutex>>mutexMap;
 33     std::mutex mtx;
 34 };

 39 {
 40     instance.reset(new MutexTable());
 41 }

 44 {
 45     instance.reset();
 46 }

 48 MutexTable::MutexTable() : nextMutexId(1), noLockMutexes()
 49 {
 50     noLockMutexes.resize(256);
 51 }
 52 
 53 int32_t MutexTable::AllocateMutex()
 54 {
 55     int32_t mutexId = nextMutexId++;
 56     if (mutexId < numNoLockMutexes)
 57     {
 58         noLockMutexes[mutexId].reset(new std::mutex());
 59         return mutexId;
 60     }
 61     else
 62     {
 63         std::lock_guard<std::mutex> lock(mtx);
 64         mutexMap[mutexId].reset(new std::mutex());
 65         return mutexId;
 66     }
 67 }
 68 
 69 void MutexTable::FreeMutex(int32_t mutexId)
 70 {
 71     if (mutexId < numNoLockMutexes)
 72     {
 73         noLockMutexes[mutexId].reset();
 74     }
 75     else
 76     {
 77         std::lock_guard<std::mutex> lock(mtx);
 78         mutexMap[mutexId].reset();
 79     }
 80 }
 81 
 82 void MutexTable::LockMutex(int32_t mutexId)
 83 {
 84     if (mutexId < numNoLockMutexes)
 85     {
 86         noLockMutexes[mutexId]->lock();
 87     }
 88     else
 89     {
 90         std::lock_guard<std::mutex> lock(mtx);
 91         mutexMap[mutexId]->lock();
 92     }
 93 }
 94 
 95 void MutexTable::UnlockMutex(int32_t mutexId)
 96 {
 97     if (mutexId < numNoLockMutexes)
 98     {
 99         noLockMutexes[mutexId]->unlock();
100     }
101     else
102     {
103         std::lock_guard<std::mutex> lock(mtx);
104         mutexMap[mutexId]->unlock();;
105     }
106 }

201 std::recursive_mutex& RecursiveMutexTable::GetRecursiveMutex(int32_t mutexId)
202 {
203     if (mutexId < numNoLockMutexes)
204     {
205         std::recursive_mutex* mtx = noLockMutexes[mutexId].get();
206         if (!mtx)
207         {
208             throw std::runtime_error("invalid recursive mutex id " + std::to_string(mutexId));
209         }
210         return *mtx;
211     }
212     else
213     {
214         std::lock_guard<std::mutex> lock(mtx);
215         auto it = mutexMap.find(mutexId);
216         if (it != mutexMap.cend())
217         {
218             std::recursive_mutex* mtx = it->second.get();
219             if (!mtx)
220             {
221                 throw std::runtime_error("invalid recursive mutex id " + std::to_string(mutexId));
222             }
223             return *mtx;
224         }
225         else
226         {
227             throw std::runtime_error("invalid recursive mutex id " + std::to_string(mutexId));
228         }
229     }
230 }
