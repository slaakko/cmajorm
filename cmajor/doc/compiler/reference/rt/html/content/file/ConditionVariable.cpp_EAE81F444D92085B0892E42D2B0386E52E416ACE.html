 21 class ConditionVariableTable 
 22 {
 23 public:
 24     static void Init();
 25     static void Done();
 26     static ConditionVariableTable& Instance() { Assert(instance, "condition variable table not initialized"); return *instance; }
 27     int32_t AllocateConditionVariable();
 28     void FreeConditionVariable(int32_t conditionVariableId);
 29     std::condition_variable_any& GetConditionVariable(int32_t conditionVariableId);
 30 private:
 31     const int32_t maxNoLockConditionVariables = 256;
 32     static std::unique_ptr<ConditionVariableTable> instance;
 33     std::atomic<int32_t> nextConditionVariableId;
 34     std::vector<std::std::unique_ptr<std::condition_variable_any>>conditionVariables;
 35     std::unordered_map<int32_t, std::std::unique_ptr<std::condition_variable_any>>conditionVariableMap;
 36     std::mutex mtx;
 37     ConditionVariableTable();
 38 };
 39 
 40 std::unique_ptr<ConditionVariableTable> ConditionVariableTable::instance;
 41 
 42 void ConditionVariableTable::Init()
 43 {
 44     instance.reset(new ConditionVariableTable());
 45 }
 46 
 47 void ConditionVariableTable::Done()
 48 {
 49     instance.reset();
 50 }
 51 
 52 ConditionVariableTable::ConditionVariableTable() : nextConditionVariableId(1)
 53 {
 54     conditionVariables.resize(maxNoLockConditionVariables);
 55 }
 56 
 57 int32_t ConditionVariableTable::AllocateConditionVariable()
 58 {
 59     int32_t conditionVariableId = nextConditionVariableId++;
 60     Assert(conditionVariableId > 0, "invalid condition variable id");
 61     if (conditionVariableId < maxNoLockConditionVariables)
 62     {
 63         conditionVariables[conditionVariableId].reset(new std::condition_variable_any());
 64     }
 65     else
 66     {
 67         std::lock_guard<std::mutex> lock(mtx);
 68         conditionVariableMap[conditionVariableId].reset(new std::condition_variable_any());
 69     }
 70     return conditionVariableId;
 71 }
 72 
 73 void ConditionVariableTable::FreeConditionVariable(int32_t conditionVariableId)
 74 {
 75     if (conditionVariableId <= 0)
 76     {
 77         throw std::runtime_error("invalid condition variable id " + std::to_string(conditionVariableId));
 78     }
 79     if (conditionVariableId < maxNoLockConditionVariables)
 80     {
 81         conditionVariables[conditionVariableId].reset();
 82     }
 83     else
 84     {
 85         std::lock_guard<std::mutex> lock(mtx);
 86         conditionVariableMap.erase(conditionVariableId);
 87     }
 88 }
 89 
 90 std::condition_variable_any& ConditionVariableTable::GetConditionVariable(int32_t conditionVariableId)
 91 {
 92     if (conditionVariableId <= 0)
 93     {
 94         throw std::runtime_error("invalid condition variable id " + std::to_string(conditionVariableId));
 95     }
 96     else if (conditionVariableId < maxNoLockConditionVariables)
 97     {
 98         std::condition_variable_any* c = conditionVariables[conditionVariableId].get();
 99         if (c)
100         {
101             return *c;
102         }
103         else
104         {
105             throw std::runtime_error("invalid condition variable id " + std::to_string(conditionVariableId));
106         }
107     }
108     else
109     {
110         std::lock_guard<std::mutex> lock(mtx);
111         auto it = conditionVariableMap.find(conditionVariableId);
112         if (it != conditionVariableMap.cend())
113         {
114             std::condition_variable_any* c = it->second.get();
115             if (c)
116             {
117                 return *c;
118             }
119             else
120             {
121                 throw std::runtime_error("invalid condition variable id " + std::to_string(conditionVariableId));
122             }
123         }
124         else
125         {
126             throw std::runtime_error("invalid condition variable id " + std::to_string(conditionVariableId));
127         }
128     }
129 }

189 struct RecursiveMutex 
190 {
191     int32_t id;
192 };

194 extern "C" int32_t RtWaitConditionVariable(int32_t conditionVariableId, void* recursiveMutex)
195 {
196     try
197     {
198         RecursiveMutex* m = reinterpret_cast<RecursiveMutex*>(recursiveMutex);
199         int32_t mutexId = m->id;
200         std::recursive_mutex& mtx = cmajor::rt::GetRecursiveMutex(mutexId);
201         std::condition_variable_any& conditionVariable = cmajor::rt::ConditionVariableTable::Instance().GetConditionVariable(conditionVariableId);
202         conditionVariable.wait(mtx);
203         return 0;
204     }
205     catch (const std::exception& ex)
206     {
207         return cmajor::rt::InstallError(ex.what());
208     }

211 enum class CondVarStatus : int32_t 
212 {
213     timeout=  0, no_timeout=  1
214 };
215 
216 extern "C" int32_t RtWaitConditionVariableDuration(int32_t conditionVariableId, void* recursiveMutex, int64_t nanoseconds)
217 {
218     try
219     {
220         RecursiveMutex* m = reinterpret_cast<RecursiveMutex*>(recursiveMutex);
221         int32_t mutexId = m->id;
222         std::recursive_mutex& mtx = cmajor::rt::GetRecursiveMutex(mutexId);
223         std::condition_variable_any& conditionVariable = cmajor::rt::ConditionVariableTable::Instance().GetConditionVariable(conditionVariableId);
224         CondVarStatus status = CondVarStatus::timeout;
225         std::chrono::nanoseconds duration(nanoseconds);
226         if (conditionVariable.wait_for(mtx, duration) == std::cv_status::no_timeout)
227         {
228             status = CondVarStatus::no_timeout;
229         }
230         return static_cast<int32_t>(status);
231     }
232     catch (const std::exception& ex)
233     {
234         return cmajor::rt::InstallError(ex.what());
235     }
