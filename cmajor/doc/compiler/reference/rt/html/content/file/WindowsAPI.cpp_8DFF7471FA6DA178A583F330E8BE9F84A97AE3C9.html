 10 struct CodeMapping 
 11 {
 12     CodeMapping(long extCode_, long winCode_) : extCode(extCode_), winCode(winCode_)
 13     {
 14     }
 15     long extCode;
 16     long winCode;
 17 };
 18 
 19 long ERROR_OUT_OF_RESOURCES = 0;
 20 
 21 CodeMapping code_mapping[] =
 22 {
 23     CodeMapping(0, ERROR_OUT_OF_RESOURCES), 
 24     CodeMapping(1, ERROR_FILE_NOT_FOUND), 
 25     CodeMapping(2, ERROR_PATH_NOT_FOUND), 
 26     CodeMapping(3, ERROR_BAD_FORMAT), 
 27     CodeMapping(4, SE_ERR_ACCESSDENIED), 
 28     CodeMapping(5, SE_ERR_ASSOCINCOMPLETE), 
 29     CodeMapping(6, SE_ERR_DDEBUSY), 
 30     CodeMapping(7, SE_ERR_DDEFAIL), 
 31     CodeMapping(8, SE_ERR_DDETIMEOUT), 
 32     CodeMapping(9, SE_ERR_DLLNOTFOUND), 
 33     CodeMapping(10, SE_ERR_FNF), 
 34     CodeMapping(11, SE_ERR_NOASSOC), 
 35     CodeMapping(12, SE_ERR_OOM), 
 36     CodeMapping(13, SE_ERR_PNF), 
 37     CodeMapping(14, SE_ERR_SHARE)
 38 };
 39 
 40 bool codeMappingInitialized = false;
 41 
 42 std::map<long, long> winToExtCodeMap;
 43 std::map<long, long> extToWinCodeMap;
 44 
 45 bool InitCodeMapping()
 46 {
 47     for (const CodeMapping& mapping : code_mapping)
 48     {
 49         auto it1 = winToExtCodeMap.find(mapping.winCode);
 50         if (it1 != winToExtCodeMap.cend())
 51         {
 52             return false;
 53         }
 54         winToExtCodeMap[mapping.winCode] = mapping.extCode;
 55         auto it2 = extToWinCodeMap.find(mapping.extCode);
 56         if (it2 != extToWinCodeMap.cend())
 57         {
 58             return false;
 59         }
 60         extToWinCodeMap[mapping.extCode] = mapping.winCode;
 61     }
 62     codeMappingInitialized = true;
 63     return true;
 64 }
 65 
 66 long MapWinCodeToExtCode(long winCode)
 67 {
 68     if (!codeMappingInitialized)
 69     {
 70         if (!InitCodeMapping())
 71         {
 72             return -2;
 73         }
 74     }
 75     auto it = winToExtCodeMap.find(winCode);
 76     if (it != winToExtCodeMap.cend())
 77     {
 78         return it->second;
 79     }
 80     return -1;
 81 }
 82 
 83 long MapExtCodeToWinCode(long extCode)
 84 {
 85     if (!codeMappingInitialized)
 86     {
 87         if (!InitCodeMapping())
 88         {
 89             return -2;
 90         }
 91     }
 92     auto it = extToWinCodeMap.find(extCode);
 93     if (it != extToWinCodeMap.cend())
 94     {
 95         return it->second;
 96     }
 97     return -1;
 98 }

140 int32_t WinWaitForMultipleObjects(uint32_t count, void** handles)
141 {
142     uint32_t retval = WaitForMultipleObjects(count, handles, false, INFINITE);
143     if (retval == WAIT_FAILED)
144     {
145         return -1;
146     }
147     else if (retval == WAIT_TIMEOUT)
148     {
149         return -2;
150     }
151     else if (retval >= WAIT_OBJECT_0 && retval < WAIT_OBJECT_0 + count)
152     {
153         return retval - WAIT_OBJECT_0;
154     }
155     else if (retval >= WAIT_ABANDONED_0 && retval < WAIT_ABANDONED_0 + count)
156     {
157         return -3;
158     }
159     else
160     {
161         return -4;
162     }
163 }

187 bool WinShellExecute(const char* filePath, int64_t& errorCode)
188 {
189     int code = (int)ShellExecuteA(NULL, "open", filePath, NULL, NULL, SW_SHOWNORMAL);
190     if (code > 32)
191     {
192         errorCode = 0;
193         return true;
194     }
195     errorCode = MapWinCodeToExtCode(code);
196     return false;
