11 std::u32string ParseDQString(const std::string& fileName, const soulng::lexer::Token& token)
12 {
13     std::u32string s;
14     const char32_t* p = token.match.begin;
15     const char32_t* e = token.match.end;
16     if (p != e && *p == '"')
17     {
18         ++p;
19     }
20     while (p != e && *p != '"')
21     {
22         s.append(1, *p);
23         ++p;
24     }
25     if (p != e && *p == '"')
26     {
27         ++p;
28     }
29     if (p != e)
30     {
31         throw std::runtime_error("invalid string literal at " + fileName + ":" + std::to_string(token.line) + ": " + soulng::unicode::ToUtf8(token.match.ToString()));
32     }
33     return s;
34 }
