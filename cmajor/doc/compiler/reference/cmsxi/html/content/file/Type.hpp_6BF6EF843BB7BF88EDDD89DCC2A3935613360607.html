 35 class Type 
 36 {
 37 public:
 38     Type(int id_);
 39     virtual ~Type();
 40     virtual std::string Name() const;
 41     virtual ConstantValue* DefaultValue();
 42     int Id() const { return id; }
 43     void Write(CodeFormatter& formatter);
 44     virtual void WriteDeclaration(CodeFormatter& formatter);
 45     bool IsPtrType() const { return id == ptrTypeId; }
 46     bool IsVoidType() const { return id == voidTypeId; }
 47     virtual bool IsStructureType() const { return false; }
 48     virtual bool IsArrayType() const { return false; }
 49     virtual bool IsFunctionType() const { return false; }
 50 private:
 51     int id;
 52 };
