 25 void GlobalVariable::Write(Context& context, CodeFormatter& formatter)
 26 {
 27     formatter.Write(type->Name());
 28     if (linkOnce)
 29     {
 30         formatter.Write(" once");
 31     }
 32     formatter.Write(" ");
 33     formatter.Write(name);
 34     if (initializer)
 35     {
 36         formatter.Write(" = ");
 37         if (initializer->IsAggregateValue() || initializer->IsStringValue())
 38         {
 39             formatter.Write(initializer->Name(context));
 40         }
 41         else
 42         {
 43             formatter.Write(initializer->GetType(context)->Name());
 44             formatter.Write(" ");
 45             formatter.Write(initializer->Name(context));
 46         }
 47     }
 48     else
 49     {
 50         formatter.Write(";");
 51     }
 52 }

 58 GlobalVariable* DataRepository::GetOrInsertGlobal(const std::string& name, Type* type)
 59 {
 60     auto it = globalVariableMap.find(name);
 61     if (it != globalVariableMap.cend())
 62     {
 63         return it->second;
 64     }
 65     GlobalVariable* globalVariable = new GlobalVariable(type, name);
 66     globalVariableDefinitions.push_back(std::unique_ptr<GlobalVariable>(globalVariable));
 67     globalVariableMap[name] = globalVariable;
 68     return globalVariable;
 69 }
 70 
 71 GlobalVariable* DataRepository::CreateGlobalStringPtr(Context& context, const std::string& stringValue)
 72 {
 73     GlobalVariable* globalVariable = new GlobalVariable(context.GetByteType(), "string" + std::to_string(nextStringId++) + "_" + compileUnitId);
 74     globalVariable->SetInitializer(context.GetStringValue(globalVariable->GetType(context), stringValue));
 75     globalVariableDefinitions.push_back(std::unique_ptr<GlobalVariable>(globalVariable));
 76     return globalVariable;
 77 }
 78 
 79 void DataRepository::Write(Context& context, CodeFormatter& formatter)
 80 {
 81     if (globalVariableDefinitions.empty()) return;
 82     formatter.WriteLine("data");
 83     formatter.WriteLine("{");
 84     formatter.IncIndent();
 85     for (const auto& globalVariable : globalVariableDefinitions)
 86     {
 87         globalVariable->Write(context, formatter);
 88         formatter.WriteLine();
 89     }
 90     formatter.DecIndent();
 91     formatter.WriteLine("}");
 92     formatter.WriteLine();
 93 }
