224 size_t FunctionTypeKeyHash::operator()(const FunctionTypeKey& key) const
225 {
226     size_t h = std::hash<Type*>()(key.returnType);
227     for (Type* paramType : key.paramTypes)
228     {
229         h = h ^ std::hash<Type*>()(paramType);
230     }
231     return h;
232 }

244 void TypeRepository::Write(CodeFormatter& formatter)
245 {
246     if (types.empty()) return;
247     formatter.WriteLine("types");
248     formatter.WriteLine("{");
249     formatter.IncIndent();
250     for (const auto& t : types)
251     {
252         t->WriteDeclaration(formatter);
253         formatter.WriteLine();
254     }
255     formatter.DecIndent();
256     formatter.WriteLine("}");
257     formatter.WriteLine();
258 }

260 Type* TypeRepository::GetPtrType(Type* baseType)
261 {
262     auto it = ptrTypeMap.find(baseType);
263     if (it != ptrTypeMap.cend())
264     {
265         return it->second;
266     }
267     else
268     {
269         PtrType* ptrType = new PtrType(baseType);
270         ptrTypeMap[baseType] = ptrType;
271         ptrTypes.push_back(std::unique_ptr<PtrType>(ptrType));
272         return ptrType;
273     }
274 }

276 Type* TypeRepository::GetStructureType(const std::std::vector<Type*>&memberTypes)
277 {
278     auto it = structureTypeMap.find(memberTypes);

280     {
