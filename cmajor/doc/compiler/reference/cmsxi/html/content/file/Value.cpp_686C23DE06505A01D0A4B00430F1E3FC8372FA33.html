232 ArrayValue::ArrayValue(Type* type_, const std::std::vector<ConstantValue*>&elements_, conststd::string&prefix_):ConstantValue(), type(type_), elements(elements_), prefix(prefix_)
233 {
234 }
235 
236 std::string ArrayValue::Name(Context& context)
237 {
238     std::string name = prefix + "[ ";
239     bool first = true;
240     for (ConstantValue* element : elements)
241     {
242         if (first)
243         {
244             first = false;
245         }
246         else
247         {
248             name.append(", ");
249         }
250         if (element->IsAggregateValue())
251         {
252             name.append(element->Name(context));
253         }
254         else
255         {
256             name.append(element->GetType(context)->Name()).append(" ").append(element->Name(context));
257         }
258     }
259     name.append(" ]");
260     return name;
261 }

290         }

294         }

297             name.append(member->GetType(context)->Name()).append(" ").append(member->Name(context));
298         }

313 std::string StringValue::Name(Context& context)
314 {
315     std::string name("\"");
316     for (char c : value)
317     {
318         if (c == '"')
319         {
320             name.append("\\").append(soulng::util::ToHexString(static_cast<uint8_t>(c)));
321         }
322         else if (c == '\\')
323         {
324             name.append("\\").append(soulng::util::ToHexString(static_cast<uint8_t>(c)));
325         }
326         else if (c >= 32 && c < 127)
327         {
328             name.append(1, c);
329         }
330         else
331         {
332             name.append("\\").append(soulng::util::ToHexString(static_cast<uint8_t>(c)));
333         }
334     }
335     name.append("\\").append(soulng::util::ToHexString(static_cast<uint8_t>(0)));
336     name.append("\"");
337     return name;
338 }
