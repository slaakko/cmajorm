 22 bool operator==(const FunctionTemplateKey& left, const FunctionTemplateKey& right)
 23 {
 24     if (left.functionTemplate != right.functionTemplate) return false;
 25     if (left.templateArgumentTypes.size() != right.templateArgumentTypes.size()) return false;
 26     int n = left.templateArgumentTypes.size();
 27     for (int i = 0; i < n; ++i)
 28     {
 29         if (!TypesEqual(left.templateArgumentTypes[i], right.templateArgumentTypes[i])) return false;
 30     }
 31     return true;
 32 }
 33 
 34 bool operator!=(const FunctionTemplateKey& left, const FunctionTemplateKey& right)
 35 {
 36     return !(left == right);
 37 }

 43 FunctionSymbol* FunctionTemplateRepository::Instantiate(FunctionSymbol* functionTemplate, const std::std::unordered_map<TemplateParameterSymbol*, TypeSymbol*>&templateParameterMapping, constSpan&span)
 44 {
 45     std::vector<TypeSymbol*> templateArgumentTypes;
 46     for (TemplateParameterSymbol* templateParameter : functionTemplate->TemplateParameters())
 47     {
 48         auto it = templateParameterMapping.find(templateParameter);
 49         if (it != templateParameterMapping.cend())
 50         {
 51             TypeSymbol* templateArgumentType = it->second;
 52             templateArgumentTypes.push_back(templateArgumentType);
 53         }
 54         else
 55         {
 56             throw Exception(&boundCompileUnit.GetModule(), "template parameter type not found", span);
 57         }
 58     }
 59     FunctionTemplateKey key(functionTemplate, templateArgumentTypes);
 60     auto it = functionTemplateMap.find(key);
 61     if (it != functionTemplateMap.cend())
 62     {
 63         return it->second;
 64     }
 65     SymbolTable& symbolTable = boundCompileUnit.GetSymbolTable();
 66     Node* node = symbolTable.GetNodeNoThrow(functionTemplate);
 67     if (!node)
 68     {
 69         node = functionTemplate->GetFunctionNode();
 70         symbolTable.MapNode(node, functionTemplate);
 71         Assert(node, "function node not read");
 72     }
 73     Assert(node->GetNodeType() == NodeType::functionNode, "function node expected");
 74     FunctionNode* functionNode = static_cast<FunctionNode*>(node);
 75     SpanMapper spanMapper;
 76     std::unique_ptr<NamespaceNode> globalNs(new NamespaceNode(spanMapper.MapSpan(functionNode->GetSpan(), functionNode->RootModuleId()), 
 77         new IdentifierNode(spanMapper.MapSpan(functionNode->GetSpan(), functionNode->RootModuleId()), U"")));
 78     NamespaceNode* currentNs = globalNs.get();
 79     CloneContext cloneContext;
 80     cloneContext.SetSpanMapper(&spanMapper);
 81     cloneContext.SetInstantiateFunctionNode();
 82     int n = functionTemplate->UsingNodes().Count();
 83     for (int i = 0; i < n; ++i)
 84     {
 85         Node* usingNode = functionTemplate->UsingNodes()[i];
 86         globalNs->AddMember(usingNode->Clone(cloneContext));
 87     }
 88     bool fileScopeAdded = false;
 89     if (!functionTemplate->Ns()->IsGlobalNamespace())
 90     {
 91         FileScope* primaryFileScope = new FileScope(&boundCompileUnit.GetModule());
 92         primaryFileScope->AddContainerScope(functionTemplate->Ns()->GetContainerScope());
 93         boundCompileUnit.AddFileScope(primaryFileScope);
 94         fileScopeAdded = true;
 95         std::u32string fullNsName = functionTemplate->Ns()->FullName();
 96         std::vector<std::u32string> nsComponents = Split(fullNsName, '.');
 97         for (const std::u32string& nsComponent : nsComponents)
 98         {
 99             NamespaceNode* nsNode = new NamespaceNode(spanMapper.MapSpan(functionNode->GetSpan(), functionNode->RootModuleId()), 
100                 new IdentifierNode(spanMapper.MapSpan(functionNode->GetSpan(), functionNode->RootModuleId()), nsComponent));
101             currentNs->AddMember(nsNode);
102             currentNs = nsNode;
103         }
104     }
105     FunctionNode* functionInstanceNode = static_cast<FunctionNode*>(functionNode->Clone(cloneContext));
106     currentNs->AddMember(functionInstanceNode);
107     symbolTable.SetCurrentCompileUnit(boundCompileUnit.GetCompileUnitNode());
108     SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
109     globalNs->Accept(symbolCreatorVisitor);
110     Symbol* symbol = symbolTable.GetSymbol(functionInstanceNode);
111     Assert(symbol->GetSymbolType() == SymbolType::functionSymbol, "function symbol expected");
112     FunctionSymbol* functionSymbol = static_cast<FunctionSymbol*>(symbol);
113     functionSymbol->SetLinkOnceOdrLinkage();
114     functionSymbol->SetTemplateSpecialization();
115     functionSymbol->SetFunctionTemplate(functionTemplate);
116     functionSymbol->SetTemplateArgumentTypes(templateArgumentTypes);
117     functionTemplateMap[key] = functionSymbol;
118     for (TemplateParameterSymbol* templateParameter : functionTemplate->TemplateParameters())
119     {
120         auto it = templateParameterMapping.find(templateParameter);
121         if (it != templateParameterMapping.cend())
122         {
123             TypeSymbol* boundType = it->second;
124             BoundTemplateParameterSymbol* boundTemplateParameter = new BoundTemplateParameterSymbol(span, templateParameter->Name());
125             boundTemplateParameter->SetType(boundType);
126             functionSymbol->AddMember(boundTemplateParameter);
127         }
128         else
129         {
130             throw Exception(&boundCompileUnit.GetModule(), "template parameter type not found", span);
131         }
132     }
133     TypeBinder typeBinder(boundCompileUnit);
134     globalNs->Accept(typeBinder);
135     StatementBinder statementBinder(boundCompileUnit);
136     globalNs->Accept(statementBinder);
137     if (fileScopeAdded)
138     {
139         boundCompileUnit.RemoveLastFileScope();
140     }
141     boundCompileUnit.AddGlobalNs(std::move(globalNs));
142     functionSymbol->SetFlag(FunctionSymbolFlags::dontReuse);
143     if (functionTemplate->IsSystemDefault())
144     {
145         functionSymbol->SetSystemDefault();
146     }
147     boundCompileUnit.SetCanReuse(functionSymbol);
148     return functionSymbol;
149 }
