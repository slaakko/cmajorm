360 FunctionSymbol* BoundCompileUnit::GetConversion(TypeSymbol* sourceType, TypeSymbol* targetType, ContainerScope* containerScope, BoundFunction* currentFunction, const Span& span, ArgumentMatch& argumentMatch)
361 {
362     FunctionSymbol* conversion = symbolTable.GetConversion(sourceType, targetType, span);
363     if (conversion && conversion->GetSymbolType() == SymbolType::conversionFunctionSymbol)
364     {
365         argumentMatch.preReferenceConversionFlags = OperationFlags::addr;
366     }
367     if (!conversion)
368     {
369         conversion = conversionTable.GetConversion(sourceType, targetType, span);
370         if (!conversion)
371         {
372             if (sourceType->IsNullPtrType() && targetType->IsPointerType() && !targetType->IsReferenceType())
373             {
374                 std::unique_ptr<FunctionSymbol> nullPtrToPtrConversion(new NullPtrToPtrConversion(symbolTable.GetTypeByName(U"@nullptr_type"), targetType));
375                 nullPtrToPtrConversion->SetParent(&symbolTable.GlobalNs());
376                 conversion = nullPtrToPtrConversion.get();
377                 conversionTable.AddConversion(conversion);
378                 conversionTable.AddGeneratedConversion(std::move(nullPtrToPtrConversion));
379                 return conversion;
380             }
381             else if (sourceType->IsVoidPtrType() && targetType->IsPointerType() && !targetType->IsReferenceType())
382             {
383                 std::unique_ptr<FunctionSymbol> voidPtrToPtrConversion(new VoidPtrToPtrConversion(symbolTable.GetTypeByName(U"void")->AddPointer(span), targetType));
384                 voidPtrToPtrConversion->SetParent(&symbolTable.GlobalNs());
385                 conversion = voidPtrToPtrConversion.get();
386                 conversionTable.AddConversion(conversion);
387                 conversionTable.AddGeneratedConversion(std::move(voidPtrToPtrConversion));
388                 return conversion;
389             }
390             else if (sourceType->PlainType(span)->IsPointerType() && targetType == symbolTable.GetTypeByName(U"ulong"))
391             {
392                 std::unique_ptr<FunctionSymbol> ptrToULongConversion(new PtrToULongConversion(sourceType->PlainType(span), symbolTable.GetTypeByName(U"ulong")));
393                 ptrToULongConversion->SetParent(&symbolTable.GlobalNs());
394                 conversion = ptrToULongConversion.get();
395                 conversionTable.AddConversion(conversion);
396                 conversionTable.AddGeneratedConversion(std::move(ptrToULongConversion));
397                 return conversion;
398             }
399             else if (sourceType == symbolTable.GetTypeByName(U"ulong") && targetType->IsVoidPtrType())
400             {
401                 std::unique_ptr<FunctionSymbol> ulongToVoidPtrConversion(new ULongToVoidPtrConversion(sourceType, targetType));
402                 ulongToVoidPtrConversion->SetParent(&symbolTable.GlobalNs());
403                 conversion = ulongToVoidPtrConversion.get();
404                 conversionTable.AddConversion(conversion);
405                 conversionTable.AddGeneratedConversion(std::move(ulongToVoidPtrConversion));
406                 return conversion;
407             }
408             else if (sourceType->IsDelegateType() && targetType->IsVoidPtrType())
409             {
410                 std::unique_ptr<FunctionSymbol> dlgToVoidPtrConversion(new DelegateToVoidPtrConversion(sourceType, targetType));
411                 dlgToVoidPtrConversion->SetParent(&symbolTable.GlobalNs());
412                 conversion = dlgToVoidPtrConversion.get();
413                 conversionTable.AddConversion(conversion);
414                 conversionTable.AddGeneratedConversion(std::move(dlgToVoidPtrConversion));
415                 return conversion;
416             }
417             else if (sourceType->IsVoidPtrType() && targetType->IsDelegateType())
418             {
419                 std::unique_ptr<FunctionSymbol> voidPtrToDlgConversion(new VoidPtrToDelegateConversion(sourceType, targetType, symbolTable.GetTypeByName(U"ulong")));
420                 voidPtrToDlgConversion->SetParent(&symbolTable.GlobalNs());
421                 conversion = voidPtrToDlgConversion.get();
422                 conversionTable.AddConversion(conversion);
423                 conversionTable.AddGeneratedConversion(std::move(voidPtrToDlgConversion));
424                 return conversion;
425             }
426             else if (sourceType->PlainType(span)->IsPointerType() && targetType->RemoveConst(span)->IsVoidPtrType())
427             {
428                 std::unique_ptr<FunctionSymbol> ptrToVoidPtrConversion(new PtrToVoidPtrConversion(sourceType->PlainType(span), symbolTable.GetTypeByName(U"void")->AddPointer(span)));
429                 ptrToVoidPtrConversion->SetParent(&symbolTable.GlobalNs());
430                 conversion = ptrToVoidPtrConversion.get();
431                 conversionTable.AddConversion(conversion);
432                 conversionTable.AddGeneratedConversion(std::move(ptrToVoidPtrConversion));
433                 return conversion;
434             }
435             else if (sourceType->PlainType(span)->IsCharacterPointerType() && targetType->IsStringFunctionContainer())
436             {
437                 std::unique_ptr<FunctionSymbol> charPtr2StringFunctionsConversion(
438                     new CharacterPointerLiteralToStringFunctionContainerConversion(sourceType->PlainType(span), symbolTable.GetTypeByName(U"@string_functions")));
439                 charPtr2StringFunctionsConversion->SetParent(&symbolTable.GlobalNs());
440                 conversion = charPtr2StringFunctionsConversion.get();
441                 conversionTable.AddConversion(conversion);
442                 conversionTable.AddGeneratedConversion(std::move(charPtr2StringFunctionsConversion));
443                 return conversion;
444             }
445             else if (sourceType->BaseType()->IsClassTypeSymbol() && targetType->BaseType()->IsClassTypeSymbol())
446             {
447                 if (sourceType->PointerCount() == targetType->PointerCount() && 
448                     (!sourceType->IsArrayType() && !targetType->IsArrayType()))
449                 {
450                     ClassTypeSymbol* sourceClassType = static_cast<ClassTypeSymbol*>(sourceType->BaseType());
451                     ClassTypeSymbol* targetClassType = static_cast<ClassTypeSymbol*>(targetType->BaseType());
452                     uint8_t conversionDistance = 0;
453                     if (sourceClassType->HasBaseClass(targetClassType, conversionDistance))
454                     {
455                         //if (targetType->IsLvalueReferenceType() && !sourceType->IsReferenceType()) // bug???
456                         if (targetType->IsReferenceType() && !sourceType->IsReferenceType())
457                         {
458                             argumentMatch.preReferenceConversionFlags = OperationFlags::addr;
459                             sourceType = sourceType->AddLvalueReference(span);
460                             if (targetType->IsConstType())
461                             {
462                                 sourceType = sourceType->AddConst(span);
463                             }
464                         }
465                         std::u32string conversionName = sourceType->FullName() + U"2" + targetType->FullName();
466                         std::unique_ptr<FunctionSymbol> implicitClassTypeConversion(new ClassTypeConversion(conversionName, ConversionType::implicit_, conversionDistance, sourceType, targetType));
467                         conversion = implicitClassTypeConversion.get();
468                         // do not add entry to the conversion table
469                         conversionTable.AddGeneratedConversion(std::move(implicitClassTypeConversion));
470                         return conversion;
471                     }
472                     else
473                     {
474                         uint8_t conversionDistance = 0;
475                         if (targetClassType->HasBaseClass(sourceClassType, conversionDistance))
476                         {
477                             //if (targetType->IsLvalueReferenceType() && !sourceType->IsReferenceType()) // bug???
478                             if (targetType->IsReferenceType() && !sourceType->IsReferenceType())
479                             {
480                                 argumentMatch.preReferenceConversionFlags = OperationFlags::addr;
481                                 sourceType = sourceType->AddLvalueReference(span);
482                                 if (targetType->IsConstType())
483                                 {
484                                     sourceType = sourceType->AddConst(span);
485                                 }
486                             }
487                             std::u32string conversionName = sourceType->FullName() + U"2" + targetType->FullName();
488                             std::unique_ptr<FunctionSymbol> explicitClassTypeConversion(new ClassTypeConversion(conversionName, ConversionType::explicit_, conversionDistance, sourceType, targetType));
489                             conversion = explicitClassTypeConversion.get();
490                             // do not add entry to the conversion table
491                             conversionTable.AddGeneratedConversion(std::move(explicitClassTypeConversion));
492                             return conversion;
493                         }
494                     }
495                 }
496             }
497             else if ((sourceType->GetSymbolType() == SymbolType::functionGroupTypeSymbol || sourceType->GetSymbolType() == SymbolType::memberExpressionTypeSymbol) && 
498                 targetType->GetSymbolType() == SymbolType::delegateTypeSymbol)
499             {
500                 FunctionGroupSymbol* functionGroupSymbol = nullptr;
501                 BoundMemberExpression* boundMemberExpression = nullptr;
502                 if (sourceType->GetSymbolType() == SymbolType::functionGroupTypeSymbol)
503                 {
504                     FunctionGroupTypeSymbol* functionGroupTypeSymbol = static_cast<FunctionGroupTypeSymbol*>(sourceType);
505                     functionGroupSymbol = functionGroupTypeSymbol->FunctionGroup();
506                 }
507                 else if (sourceType->GetSymbolType() == SymbolType::memberExpressionTypeSymbol)
508                 {
509                     MemberExpressionTypeSymbol* memberExpressionTypeSymbol = static_cast<MemberExpressionTypeSymbol*>(sourceType);
510                     boundMemberExpression = static_cast<BoundMemberExpression*>(memberExpressionTypeSymbol->BoundMemberExpression());
511                     if (boundMemberExpression->Member()->GetBoundNodeType() == BoundNodeType::boundFunctionGroupExpression)
512                     {
513                         BoundFunctionGroupExpression* boundFunctionGroupExpression = static_cast<BoundFunctionGroupExpression*>(boundMemberExpression->Member());
514                         functionGroupSymbol = boundFunctionGroupExpression->FunctionGroup();
515                     }
516                 }
517                 if (functionGroupSymbol)
518                 {
519                     DelegateTypeSymbol* delegateTypeSymbol = static_cast<DelegateTypeSymbol*>(targetType);
520                     int arity = delegateTypeSymbol->Arity();
521                     ViableFunctionSet viableFunctions;
522                     functionGroupSymbol->CollectViableFunctions(arity, viableFunctions, &GetModule());
523                     for (FunctionSymbol* viableFunction : viableFunctions.Get())
524                     {
525                         if (viableFunction->GetSymbolType() == SymbolType::memberFunctionSymbol && !viableFunction->IsStatic()) continue;
526                         bool found = true;
527                         for (int i = 0; i < arity; ++i)
528                         {
529                             ParameterSymbol* sourceParam = viableFunction->Parameters()[i];
530                             ParameterSymbol* targetParam = delegateTypeSymbol->Parameters()[i];
531                             if (!TypesEqual(sourceParam->GetType(), targetParam->GetType()))
532                             {
533                                 found = false;
534                                 break;
535                             }
536                         }
537                         if (found)
538                         {
539                             found = TypesEqual(viableFunction->ReturnType(), delegateTypeSymbol->ReturnType());
540                         }
541                         if (found)
542                         {
543                             if (boundMemberExpression)
544                             {
545                                 boundMemberExpression->ResetClassPtr();
546                             }
547                             if (viableFunction->IsFunctionTemplate())
548                             {
549                                 if (sourceType->GetSymbolType() == SymbolType::functionGroupTypeSymbol)
550                                 {
551                                     FunctionGroupTypeSymbol* functionGroupTypeSymbol = static_cast<FunctionGroupTypeSymbol*>(sourceType);
552                                     BoundFunctionGroupExpression* boundFunctionGroupExpression = static_cast<BoundFunctionGroupExpression*>(functionGroupTypeSymbol->BoundFunctionGroup());
553                                     std::unordered_map<TemplateParameterSymbol*, TypeSymbol*> templateParameterMap;
554                                     int n = viableFunction->TemplateParameters().size();
555                                     if (boundFunctionGroupExpression->TemplateArgumentTypes().size() == n)
556                                     {
557                                         for (int i = 0; i < n; ++i)
558                                         {
559                                             TemplateParameterSymbol* templateParameterSymbol = viableFunction->TemplateParameters()[i];
560                                             TypeSymbol* templateArgumentType = boundFunctionGroupExpression->TemplateArgumentTypes()[i];
561                                             templateParameterMap[templateParameterSymbol] = templateArgumentType;
562                                         }
563                                         viableFunction = InstantiateFunctionTemplate(viableFunction, templateParameterMap, span);
564                                     }
565                                     else
566                                     {
567                                         return nullptr;
568                                     }
569                                 }
570                                 else
571                                 {
572                                     return nullptr;
573                                 }
574                             }
575                             std::unique_ptr<FunctionSymbol> functionToDelegateConversion(new FunctionToDelegateConversion(sourceType, delegateTypeSymbol, viableFunction));
576                             functionToDelegateConversion->SetParent(&symbolTable.GlobalNs());
577                             conversion = functionToDelegateConversion.get();
578                             conversionTable.AddConversion(conversion);
579                             conversionTable.AddGeneratedConversion(std::move(functionToDelegateConversion));
580                             return conversion;
581                         }
582                     }
583                 }
584             }
585             else if ((sourceType->GetSymbolType() == SymbolType::functionGroupTypeSymbol || sourceType->GetSymbolType() == SymbolType::memberExpressionTypeSymbol) && 
586                 targetType->PlainType(span)->GetSymbolType() == SymbolType::classDelegateTypeSymbol && currentFunction)
587             {
588                 ClassDelegateTypeSymbol* classDelegateType = static_cast<ClassDelegateTypeSymbol*>(targetType->PlainType(span));
589                 FunctionGroupSymbol* functionGroup = nullptr;
590                 if (sourceType->GetSymbolType() == SymbolType::functionGroupTypeSymbol)
591                 {
592                     FunctionGroupTypeSymbol* functionGroupTypeSymbol = static_cast<FunctionGroupTypeSymbol*>(sourceType);
593                     BoundFunctionGroupExpression* boundFunctionGroup = static_cast<BoundFunctionGroupExpression*>(functionGroupTypeSymbol->BoundFunctionGroup());
594                     functionGroup = functionGroupTypeSymbol->FunctionGroup();
595                 }
596                 else if (sourceType->GetSymbolType() == SymbolType::memberExpressionTypeSymbol)
597                 {
598                     MemberExpressionTypeSymbol* memberExpressionType = static_cast<MemberExpressionTypeSymbol*>(sourceType);
599                     BoundMemberExpression* boundMemberExpr = static_cast<BoundMemberExpression*>(memberExpressionType->BoundMemberExpression());
600                     if (boundMemberExpr->Member()->GetBoundNodeType() == BoundNodeType::boundFunctionGroupExpression)
601                     {
602                         BoundFunctionGroupExpression* boundFunctionGroup = static_cast<BoundFunctionGroupExpression*>(boundMemberExpr->Member());
603                         functionGroup = boundFunctionGroup->FunctionGroup();
604                     }
605                 }
606                 if (functionGroup)
607                 {
608                     int arity = classDelegateType->Arity();
609                     ViableFunctionSet viableFunctions;
610                     functionGroup->CollectViableFunctions(arity + 1, viableFunctions, &GetModule());
611                     for (FunctionSymbol* viableFunction : viableFunctions.Get())
612                     {
613                         bool found = true;
614                         for (int i = 1; i < arity + 1; ++i)
615                         {
616                             ParameterSymbol* sourceParam = viableFunction->Parameters()[i];
617                             ParameterSymbol* targetParam = classDelegateType->Parameters()[i - 1];
618                             if (!TypesEqual(sourceParam->GetType(), targetParam->GetType()))
619                             {
620                                 found = false;
621                                 break;
622                             }
623                         }
624                         if (found)
625                         {
626                             found = TypesEqual(viableFunction->ReturnType(), classDelegateType->ReturnType());
627                         }
628                         if (found)
629                         {
630                             //LocalVariableSymbol* objectDelegatePairVariable = currentFunction->GetFunctionSymbol()->CreateTemporary(classDelegateType->ObjectDelegatePairType(), span);
631                             std::unique_ptr<FunctionSymbol> memberFunctionToClassDelegateConversion(new MemberFunctionToClassDelegateConversion(span, sourceType, classDelegateType, viableFunction));
632                             memberFunctionToClassDelegateConversion->SetParent(&symbolTable.GlobalNs());
633                             conversion = memberFunctionToClassDelegateConversion.get();
634                             conversionTable.AddConversion(conversion);
635                             conversionTable.AddGeneratedConversion(std::move(memberFunctionToClassDelegateConversion));
636                             return conversion;
637                         }
638                     }
639                 }
640             }
641             else if (targetType->PlainType(span)->GetSymbolType() == SymbolType::interfaceTypeSymbol && currentFunction)
642             {
643                 InterfaceTypeSymbol* targetInterfaceType = static_cast<InterfaceTypeSymbol*>(targetType->PlainType(span));
644                 if (sourceType->IsClassTypeSymbol())
645                 {
646                     ClassTypeSymbol* sourceClassType = static_cast<ClassTypeSymbol*>(sourceType);
647                     int32_t n = sourceClassType->ImplementedInterfaces().size();
648                     for (int32_t i = 0; i < n; ++i)
649                     {
650                         InterfaceTypeSymbol* sourceInterfaceType = sourceClassType->ImplementedInterfaces()[i];
651                         if (TypesEqual(targetInterfaceType, sourceInterfaceType))
652                         {
653                             //LocalVariableSymbol* temporaryInterfaceObjectVar = currentFunction->GetFunctionSymbol()->CreateTemporary(targetInterfaceType, span);
654                             std::unique_ptr<FunctionSymbol> classToInterfaceConversion(new ClassToInterfaceConversion(sourceClassType, targetInterfaceType, i, span));
655                             classToInterfaceConversion->SetParent(&symbolTable.GlobalNs());
656                             classToInterfaceConversion->SetModule(&GetModule());
657                             conversion = classToInterfaceConversion.get();
658                             conversionTable.AddConversion(conversion);
659                             conversionTable.AddGeneratedConversion(std::move(classToInterfaceConversion));
660                             return conversion;
661                         }
662                     }
663                 }
664             }
665         }
666     }
667     if (conversion)
668     {
669         if (conversion->Parent() && !conversion->IsGeneratedFunction() && conversion->Parent()->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
670         {
671             ClassTemplateSpecializationSymbol* specialization = static_cast<ClassTemplateSpecializationSymbol*>(conversion->Parent());
672             if (specialization->GetModule() != &GetModule())
673             {
674                 specialization = symbolTable.GetCurrentClassTemplateSpecialization(specialization);
675                 int index = conversion->GetIndex();
676                 conversion = specialization->GetFunctionByIndex(index);
677             }
678             bool firstTry = InstantiateClassTemplateMemberFunction(conversion, containerScope, currentFunction, span);
679             if (!firstTry)
680             {
681                 ClassTemplateSpecializationSymbol* specialization = static_cast<ClassTemplateSpecializationSymbol*>(conversion->Parent());
682                 ClassTemplateSpecializationSymbol* copy = symbolTable.CopyClassTemplateSpecialization(specialization);
683                 classTemplateRepository.BindClassTemplateSpecialization(copy, symbolTable.GlobalNs().GetContainerScope(), span);
684                 int index = conversion->GetIndex();
685                 conversion = copy->GetFunctionByIndex(index);
686                 bool secondTry = InstantiateClassTemplateMemberFunction(conversion, containerScope, currentFunction, span);
687                 if (!secondTry)
688                 {
689                     throw Exception(GetRootModuleForCurrentThread(), 
690                         "internal error: could not instantiate member function of a class template specialization '" + ToUtf8(specialization->FullName()) + "'", 
691                         specialization->GetSpan());
692                 }
693             }
694         }
695         else if (GetGlobalFlag(GlobalFlags::release) && conversion->IsInline())
696         {
697             conversion = InstantiateInlineFunction(conversion, containerScope, span);
698         }
699     }
700     return conversion;
701 }
