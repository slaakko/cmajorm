  61 void BoundParameter::Load(Emitter& emitter, OperationFlags flags)
  62 {
  63     emitter.SetCurrentDebugLocation(GetSpan());
  64     if ((flags & OperationFlags::addr) != OperationFlags::none)
  65     {
  66         emitter.Stack().Push(parameterSymbol->IrObject(emitter));

  69     {

  73         {
  74             value = emitter.CreateLoad(value);
  75         }
  76         emitter.Stack().Push(value);
  77     }
  78     else
  79     {
  80         emitter.Stack().Push(emitter.CreateLoad(parameterSymbol->IrObject(emitter)));
  81     }
  82     DestroyTemporaries(emitter);

 198 void BoundMemberVariable::Load(Emitter& emitter, OperationFlags flags)
 199 {
 200     emitter.SetCurrentDebugLocation(GetSpan());
 201     Assert(memberVariableSymbol->LayoutIndex() != -1, "layout index of the member variable not set");
 202     if (memberVariableSymbol->IsStatic())
 203     {
 204         ClassTypeSymbol* classType = static_cast<ClassTypeSymbol*>(memberVariableSymbol->Parent());
 205         if (staticInitNeeded)
 206         {
 207             if (classType->StaticConstructor())
 208             {
 209                 BoundFunctionCall staticConstructorCall(GetModule(), GetSpan(), classType->StaticConstructor());
 210                 staticConstructorCall.Load(emitter, OperationFlags::none);
 211             }
 212         }
 213         emitter.Stack().Push(classType->StaticObject(emitter, false));
 214     }
 215     else
 216     {
 217         if (!classPtr)
 218         {
 219             throw Exception(GetModule(), "class pointer of the member variable not set", GetSpan());
 220         }
 221         classPtr->Load(emitter, OperationFlags::none);
 222     }
 223     void* ptr = emitter.Stack().Pop();
 224     void* memberVariablePtr = emitter.GetMemberVariablePtr(ptr, memberVariableSymbol->LayoutIndex());
 225     if ((flags & OperationFlags::addr) != OperationFlags::none)
 226     {
 227         emitter.Stack().Push(memberVariablePtr);
 228     }
 229     else if ((flags & OperationFlags::deref) != OperationFlags::none)
 230     {
 231         void* value = emitter.CreateLoad(memberVariablePtr);
 232         uint8_t n = GetDerefCount(flags);
 233         for (uint8_t i = 0; i < n; ++i)
 234         {
 235             value = emitter.CreateLoad(value);
 236         }
 237         emitter.Stack().Push(value);

 838     {

 840         {
 841             std::vector<GenObject*> genObjects;
 842             for (const std::std::unique_ptr<BoundExpression>&argument : arguments)
 843             {
 844                 genObjects.push_back(argument.get());

 846             }
 847             functionSymbol->GenerateCall(emitter, genObjects, flags, GetSpan());
 848         }
 849         else

 852         }
 853     }
 854     else

 856         std::vector<GenObject*> genObjects;
 857         for (const std::std::unique_ptr<BoundExpression>&argument : arguments)
 858         {

 868         if (GetFlag(BoundExpressionFlags::virtualCall))

 882             interfaceType->GenerateCall(emitter, genObjects, callFlags, interfaceMemberFunction, GetSpan());
 883         }
 884         else
 885         {
 886             functionSymbol->GenerateCall(emitter, genObjects, callFlags, GetSpan());
 887         }

 889         {
 890             void* value = emitter.Stack().Pop();

 893             {
 894                 value = emitter.CreateLoad(value);
