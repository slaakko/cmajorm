33 enum class TypeResolverFlags : uint8_t 
34 {
35     none=  0, 
36     resolveClassGroup=  1 << 0, 
37     createMemberSymbols=  1 << 1
38 };
39 
40 inline TypeResolverFlags operator&(TypeResolverFlags left, TypeResolverFlags right)
41 {
42     return TypeResolverFlags(uint8_t(left) & uint8_t(right));
43 }
44 
45 inline TypeResolverFlags operator|(TypeResolverFlags left, TypeResolverFlags right)
46 {
47     return TypeResolverFlags(uint8_t(left) | uint8_t(right));
48 }
