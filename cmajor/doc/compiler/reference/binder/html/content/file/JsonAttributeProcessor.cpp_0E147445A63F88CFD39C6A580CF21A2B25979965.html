 34 void JsonAttributeProcessor::TypeCheck(Attribute* attribute, Symbol* symbol)
 35 {
 36     switch (symbol->GetSymbolType())
 37     {
 38         case SymbolType::classTypeSymbol: case SymbolType::classTemplateSpecializationSymbol: case SymbolType::memberVariableSymbol:
 39         {
 40             if (attribute->Value() == U"true" || attribute->Value() == U"false")
 41             {
 42                 if (symbol->IsClassTypeSymbol())
 43                 {
 44                     ClassTypeSymbol* classTypeSymbol = static_cast<ClassTypeSymbol*>(symbol);
 45                     ClassTypeSymbol* baseClass = classTypeSymbol->BaseClass();
 46                     if (baseClass)
 47                     {
 48                         bool baseClassHasJsonAttribute = false;
 49                         Attributes* baseClassAttributes = baseClass->GetAttributes();
 50                         if (baseClassAttributes)
 51                         {
 52                             Attribute* jsonAttribute = baseClassAttributes->GetAttribute(U"json");
 53                             if (jsonAttribute)
 54                             {
 55                                 baseClassHasJsonAttribute = true;
 56                             }
 57                         }
 58                         if (!baseClassHasJsonAttribute)
 59                         {
 60                             Warning warning(module->GetCurrentProjectName(), "base class '" + ToUtf8(baseClass->FullName()) + "' of json-attributed class '" + 
 61                                 ToUtf8(classTypeSymbol->FullName()) + "' does not explicitly declare 'json' attribute value to \"true\" or \"false\"");
 62                             warning.SetModule(GetRootModuleForCurrentThread());
 63                             warning.SetDefined(classTypeSymbol->GetSpan());
 64                             std::vector<Span> references;
 65                             references.push_back(baseClass->GetSpan());
 66                             warning.SetReferences(references);
 67                             module->WarningCollection().AddWarning(warning);
 68                         }
 69                     }
 70                 }
 71                 return;
 72             }
 73             else
 74             {
 75                 throw Exception(module, "unknown attribute value '" + ToUtf8(attribute->Value()) + "' for attribute '" + ToUtf8(attribute->Name()) + "'", attribute->GetSpan());
 76             }
 77         }
 78     }
 79     AttributeProcessor::TypeCheck(attribute, symbol);
 80 }

 82 void JsonAttributeProcessor::GenerateSymbols(Attribute* attribute, Symbol* symbol, BoundCompileUnit& boundCompileUnit, ContainerScope* containerScope)
 83 {
 84     if (symbol->IsClassTypeSymbol())
 85     {
 86         ClassTypeSymbol* classTypeSymbol = static_cast<ClassTypeSymbol*>(symbol);
 87         if (attribute->Value() == U"true")
 88         {
 89             GenerateMemberVariableJsonFieldNames(classTypeSymbol);
 90             std::map<std::u32string, Attribute*> memberVariableFieldNames;
 91             CheckMemberVariableJsonFieldNames(classTypeSymbol, memberVariableFieldNames);
 92             GenerateJsonCreatorFunctionSymbol(attribute, classTypeSymbol);
 93             GenerateJsonConstructorSymbol(attribute, classTypeSymbol);
 94             GenerateToJsonJsonObjectSymbol(attribute, classTypeSymbol);
 95             GenerateToJsonSymbol(attribute, classTypeSymbol, boundCompileUnit, containerScope);
 96         }
 97     }

100 void JsonAttributeProcessor::CheckMemberVariableJsonFieldNames(ClassTypeSymbol* classTypeSymbol, std::std::map<std::u32string, Attribute*>&memberVariableFieldNames)
101 {
102     if (classTypeSymbol->BaseClass())
103     {
104         CheckMemberVariableJsonFieldNames(classTypeSymbol->BaseClass(), memberVariableFieldNames);
105     }
106     for (MemberVariableSymbol* memberVariableSymbol : classTypeSymbol->MemberVariables())
107     {
108         Attributes* attributes = memberVariableSymbol->GetAttributes();
109         if (attributes)
110         {
111             Attribute* jsonAttribute = attributes->GetAttribute(U"json");
112             if (jsonAttribute)
113             {
114                 if (jsonAttribute->Value() == U"false")
115                 {
116                     continue;
117                 }
118             }
119             Attribute* jsonFieldNameAttribute = attributes->GetAttribute(U"jsonFieldName");
120             if (jsonFieldNameAttribute)
121             {
122                 auto it = memberVariableFieldNames.find(jsonFieldNameAttribute->Value());
123                 if (it != memberVariableFieldNames.cend())
124                 {
125                     Attribute* prev = it->second;
126                     throw Exception(module, "error in JSON field name generation: 'jsonFieldName' attribute not unique among member variable names of the current class and its base classes", 
127                         jsonFieldNameAttribute->GetSpan(), prev->GetSpan());
128                 }
129                 memberVariableFieldNames[jsonFieldNameAttribute->Value()] = jsonFieldNameAttribute;
130             }
131             else
132             {
133                 throw Exception(module, "internal error in JSON field name generation: 'jsonFieldName' attribute not found", memberVariableSymbol->GetSpan());
134             }
135         }
136         else
137         {
138             throw Exception(module, "internal error in JSON field name generation: attributes not found", memberVariableSymbol->GetSpan());
139         }
140     }
141 }

143 void JsonAttributeProcessor::GenerateMemberVariableJsonFieldNames(ClassTypeSymbol* classTypeSymbol)
144 {
145     for (MemberVariableSymbol* memberVariableSymbol : classTypeSymbol->MemberVariables())
146     {
147         bool hasJsonFieldAttribute = false;
148         Attributes* attributes = memberVariableSymbol->GetAttributes();
149         if (attributes)
150         {
151             Attribute* jsonAttribute = attributes->GetAttribute(U"json");
152             if (jsonAttribute)
153             {
154                 if (jsonAttribute->Value() == U"false")
155                 {
156                     continue;
157                 }
158             }
159             Attribute* jsonFieldNameAttribute = attributes->GetAttribute(U"jsonFieldName");
160             if (jsonFieldNameAttribute)
161             {
162                 hasJsonFieldAttribute = true;
163             }
164         }
165         else
166         {
167             memberVariableSymbol->SetAttributes(std::unique_ptr<Attributes>(new Attributes()));
168             attributes = memberVariableSymbol->GetAttributes();
169         }
170         if (!hasJsonFieldAttribute)
171         {
172             attributes->AddAttribute(memberVariableSymbol->GetSpan(), U"jsonFieldName", memberVariableSymbol->Name());
173         }
174     }

177 void JsonAttributeProcessor::GenerateJsonCreatorFunctionSymbol(Attribute* attribute, ClassTypeSymbol* classTypeSymbol)
178 {
179     MemberFunctionSymbol* jsonCreatorFunctionSymbol = new MemberFunctionSymbol(attribute->GetSpan(), U"Create");
180     jsonCreatorFunctionSymbol->SetGroupName(U"Create");
181     jsonCreatorFunctionSymbol->SetModule(module);
182     module->GetSymbolTable().SetFunctionIdFor(jsonCreatorFunctionSymbol);
183     jsonCreatorFunctionSymbol->SetAccess(SymbolAccess::public_);
184     jsonCreatorFunctionSymbol->SetStatic();
185     Symbol* jsonValue = classTypeSymbol->GetModule()->GetSymbolTable().GlobalNs().GetContainerScope()->Lookup(U"System.Json.JsonValue");
186     if (!jsonValue || jsonValue->GetSymbolType() != SymbolType::classGroupTypeSymbol)
187     {
188         throw Exception(module, "System.Json.JsonValue class not found from the symbol table", attribute->GetSpan(), classTypeSymbol->GetSpan());
189     }
190     ClassGroupTypeSymbol* jsonValueGroup = static_cast<ClassGroupTypeSymbol*>(jsonValue);
191     ClassTypeSymbol* jsonValueClass = jsonValueGroup->GetClass(0);
192     if (!jsonValueClass)
193     {
194         throw Exception(module, "System.Json.JsonValue class not found from the symbol table", attribute->GetSpan(), classTypeSymbol->GetSpan());
195     }
196     ParameterSymbol* jsonValueParam = new ParameterSymbol(attribute->GetSpan(), U"@value");
197     jsonValueParam->SetType(jsonValueClass->AddPointer(attribute->GetSpan()));
198     jsonCreatorFunctionSymbol->AddMember(jsonValueParam);
199     jsonCreatorFunctionSymbol->SetReturnType(module->GetSymbolTable().GetTypeByName(U"void")->AddPointer(attribute->GetSpan()));
200     classTypeSymbol->AddMember(jsonCreatorFunctionSymbol);
201     jsonCreatorFunctionSymbol->ComputeName();
202     jsonCreatorMap[classTypeSymbol] = jsonCreatorFunctionSymbol;
203 }

217     }
218     ClassGroupTypeSymbol* jsonValueGroup = static_cast<ClassGroupTypeSymbol*>(jsonValue);
219     ClassTypeSymbol* jsonValueClass = jsonValueGroup->GetClass(0);
220     if (!jsonValueClass)
221     {
222         throw Exception(module, "System.Json.JsonValue class not found from the symbol table", attribute->GetSpan(), classTypeSymbol->GetSpan());

236     MemberFunctionSymbol* toJsonJsonObjectMemberFunctionSymbol = new MemberFunctionSymbol(attribute->GetSpan(), U"ToJson");

240     bool jsonBase = false;
241     if (baseClass)
242     {
243         Attributes* attributes = baseClass->GetAttributes();

246             Attribute* jsonAttribute = attributes->GetAttribute(U"json");

251                     jsonBase = true;
252                 }
253             }
254         }
255     }
256     if (!jsonBase)
257     {
258         toJsonJsonObjectMemberFunctionSymbol->SetVirtual();

266     thisParam->SetType(classTypeSymbol->AddPointer(attribute->GetSpan()));

268     Symbol* jsonObject = GetRootModuleForCurrentThread()->GetSymbolTable().GlobalNs().GetContainerScope()->Lookup(U"System.Json.JsonObject");
269     if (!jsonObject || jsonObject->GetSymbolType() != SymbolType::classGroupTypeSymbol)
270     {
271         throw Exception(module, "System.Json.JsonObject class not found from the symbol table", attribute->GetSpan(), classTypeSymbol->GetSpan());
272     }

274     ClassTypeSymbol* jsonObjectClass = jsonObjectGroup->GetClass(0);
275     if (!jsonObjectClass)
276     {
277         throw Exception(module, "System.Json.JsonObject class not found from the symbol table", attribute->GetSpan(), classTypeSymbol->GetSpan());

281     toJsonJsonObjectMemberFunctionSymbol->SetAccess(SymbolAccess::public_);
282     toJsonJsonObjectMemberFunctionSymbol->AddMember(thisParam);
283     toJsonJsonObjectMemberFunctionSymbol->AddMember(jsonObjectParam);
284     classTypeSymbol->AddMember(toJsonJsonObjectMemberFunctionSymbol);
285     toJsonJsonObjectMemberFunctionSymbol->ComputeName();
286     toJsonJsonObjectMemberFunctionSymbolMap[classTypeSymbol] = toJsonJsonObjectMemberFunctionSymbol;
287 }

290 {

292     toJsonMemberFunctionSymbol->SetModule(module);
293     toJsonMemberFunctionSymbol->SetGroupName(U"ToJson");
294     ClassTypeSymbol* baseClass = classTypeSymbol->BaseClass();
295     bool jsonBase = false;
296     if (baseClass)
297     {
298         Attributes* attributes = baseClass->GetAttributes();
299         if (attributes)
300         {
301             Attribute* jsonAttribute = attributes->GetAttribute(U"json");
302             if (jsonAttribute)
303             {
304                 if (jsonAttribute->Value() == U"true")
305                 {
306                     jsonBase = true;
307                 }
308             }
309         }
310     }
311     if (!jsonBase)
312     {
313         toJsonMemberFunctionSymbol->SetVirtual();
314     }
315     else
316     {
317         toJsonMemberFunctionSymbol->SetOverride();
318     }
319     GetRootModuleForCurrentThread()->GetSymbolTable().SetFunctionIdFor(toJsonMemberFunctionSymbol);
320     ParameterSymbol* thisParam = new ParameterSymbol(attribute->GetSpan(), U"this");
321     thisParam->SetType(classTypeSymbol->AddPointer(attribute->GetSpan()));

330     toJsonMemberFunctionSymbol->SetAccess(SymbolAccess::public_);
331     toJsonMemberFunctionSymbol->AddMember(thisParam);
332     classTypeSymbol->AddMember(toJsonMemberFunctionSymbol);
333     toJsonMemberFunctionSymbol->ComputeName();
334     toJsonObjectMemberFunctionSymbolMap[classTypeSymbol] = toJsonMemberFunctionSymbol;
335 }

337 void JsonAttributeProcessor::GenerateImplementation(Attribute* attribute, Symbol* symbol, StatementBinder* statementBinder)
338 {
339     if (symbol->IsClassTypeSymbol())
340     {
341         ClassTypeSymbol* classTypeSymbol = static_cast<ClassTypeSymbol*>(symbol);
342         auto it = jsonConstructorMap.find(symbol);
343         if (it != jsonConstructorMap.cend())
344         {
345             ConstructorSymbol* jsonConstructorSymbol = it->second;
346             GenerateJsonConstructorImplementation(attribute, classTypeSymbol, jsonConstructorSymbol, statementBinder);
347         }
348         else
349         {
350             throw Exception(module, "internal error in JSON attribute implementation: constructor symbol for symbol '" + ToUtf8(symbol->FullName()) + "' not found", attribute->GetSpan());
351         }
352         auto it1 = jsonCreatorMap.find(symbol);
353         if (it1 != jsonCreatorMap.cend())
354         {
355             MemberFunctionSymbol* jsonCreatorSymbol = it1->second;
356             GenerateJsonCreatorImplementation(attribute, classTypeSymbol, jsonCreatorSymbol, statementBinder);
357         }
358         else
359         {
360             throw Exception(module, "internal error in JSON attribute implementation: Creator function symbol for symbol '" + ToUtf8(symbol->FullName()) + "' not found", attribute->GetSpan());
361         }
362         auto it2 = toJsonJsonObjectMemberFunctionSymbolMap.find(symbol);
363         if (it2 != toJsonJsonObjectMemberFunctionSymbolMap.cend())
364         {
365             MemberFunctionSymbol* toJsonJsonObjectMemberFunctionSymbol = it2->second;
366             GenerateToJsonJsonObjectImplementation(attribute, classTypeSymbol, toJsonJsonObjectMemberFunctionSymbol, statementBinder);
367         }
368         else
369         {
370             throw Exception(module, "internal error in JSON attribute implementation: member function 'ToJson' symbol for symbol '" + ToUtf8(symbol->FullName()) + "' not found", attribute->GetSpan());
371         }
372         auto it3 = toJsonObjectMemberFunctionSymbolMap.find(symbol);
373         if (it3 != toJsonObjectMemberFunctionSymbolMap.cend())
374         {
375             MemberFunctionSymbol* toJsonMemberFunctionSymbol = it3->second;
376             GenerateToJsonImplementation(attribute, classTypeSymbol, toJsonMemberFunctionSymbol, statementBinder);
377         }
378         else
379         {
380             throw Exception(module, "internal error in JSON attribute implementation: member function 'ToJson' symbol for symbol '" + ToUtf8(symbol->FullName()) + "' not found", attribute->GetSpan());
381         }
382         SymbolTable& symbolTable = module->GetSymbolTable();
383         symbolTable.AddJsonClass(classTypeSymbol->FullName());
384     }
385 }

387 void JsonAttributeProcessor::GenerateJsonCreatorImplementation(Attribute* attribute, ClassTypeSymbol* classTypeSymbol, MemberFunctionSymbol* jsonCreatorFunctionSymbol, StatementBinder* statementBinder)
388 {
389     try
390     {
391         FileScope* fileScope = new FileScope(&statementBinder->GetBoundCompileUnit().GetModule());
392         Symbol* jsonValue = classTypeSymbol->GetModule()->GetSymbolTable().GlobalNs().GetContainerScope()->Lookup(U"System.Json.JsonValue");
393         if (jsonValue)
394         {
395             fileScope->AddContainerScope(jsonValue->Ns()->GetContainerScope());
396         }
397         statementBinder->GetBoundCompileUnit().AddFileScope(fileScope);
398         std::unique_ptr<BoundFunction> boundFunction(new BoundFunction(module, jsonCreatorFunctionSymbol));
399         const Span& span = attribute->GetSpan();
400         CompoundStatementNode compoundStatementNode(span);
401         compoundStatementNode.SetEndBraceSpan(span);
402         NewNode* newNode = new NewNode(span, new IdentifierNode(span, classTypeSymbol->FullName()));
403         newNode->AddArgument(new IdentifierNode(span, U"@value"));
404         ReturnStatementNode* returnStatementNode = new ReturnStatementNode(span, newNode);
405         compoundStatementNode.AddStatement(returnStatementNode);
406         SymbolTable& symbolTable = statementBinder->GetBoundCompileUnit().GetSymbolTable();
407         symbolTable.BeginContainer(jsonCreatorFunctionSymbol);
408         SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
409         compoundStatementNode.Accept(symbolCreatorVisitor);
410         symbolTable.EndContainer();
411         TypeBinder typeBinder(statementBinder->GetBoundCompileUnit());
412         ContainerScope* containerScope = statementBinder->GetContainerScope();
413         typeBinder.SetContainerScope(containerScope);
414         statementBinder->SetContainerScope(jsonCreatorFunctionSymbol->GetContainerScope());
415         typeBinder.SetCurrentFunctionSymbol(jsonCreatorFunctionSymbol);
416         compoundStatementNode.Accept(typeBinder);
417         BoundFunction* prevFunction = statementBinder->CurrentFunction();
418         statementBinder->SetCurrentFunction(boundFunction.get());
419         compoundStatementNode.Accept(*statementBinder);
420         statementBinder->SetContainerScope(containerScope);
421         BoundStatement* boundStatement = statementBinder->ReleaseStatement();
422         Assert(boundStatement->GetBoundNodeType() == BoundNodeType::boundCompoundStatement, "bound compound statement expected");
423         BoundCompoundStatement* compoundStatement = static_cast<BoundCompoundStatement*>(boundStatement);
424         boundFunction->SetBody(std::unique_ptr<BoundCompoundStatement>(compoundStatement));
425         statementBinder->CurrentClass()->AddMember(std::move(boundFunction));
426         statementBinder->SetCurrentFunction(prevFunction);
427         statementBinder->GetBoundCompileUnit().RemoveLastFileScope();
428     }
429     catch (const Exception& ex;)
430     {
431         std::vector<Span> references;
432         references.push_back(ex.Defined());
433         references.insert(references.end(), ex.References().begin(), ex.References().end());
434         throw Exception(module, "error in JSON attribute generation: could not create JSON Create() function for class '" + ToUtf8(classTypeSymbol->FullName()) + "': " + ex.Message(), 
435             classTypeSymbol->GetSpan(), references);
436     }
437 }

439 void JsonAttributeProcessor::GenerateJsonConstructorImplementation(Attribute* attribute, ClassTypeSymbol* classTypeSymbol, ConstructorSymbol* jsonConstructorSymbol, StatementBinder* statementBinder)
440 {
441     try
442     {
443         FileScope* fileScope = new FileScope(&statementBinder->GetBoundCompileUnit().GetModule());
444         Symbol* jsonValue = classTypeSymbol->GetModule()->GetSymbolTable().GlobalNs().GetContainerScope()->Lookup(U"System.Json.JsonValue");
445         if (jsonValue)
446         {
447             fileScope->AddContainerScope(jsonValue->Ns()->GetContainerScope());
448         }
449         statementBinder->GetBoundCompileUnit().AddFileScope(fileScope);
450         std::unique_ptr<BoundFunction> boundFunction(new BoundFunction(module, jsonConstructorSymbol));
451         const Span& span = attribute->GetSpan();
452         ConstructorNode constructorNode(span);
453         CompoundStatementNode compoundStatementNode(span);
454         compoundStatementNode.SetEndBraceSpan(span);
455         ClassTypeSymbol* baseClass = classTypeSymbol->BaseClass();
456         if (baseClass)
457         {
458             Attributes* attributes = baseClass->GetAttributes();
459             if (attributes)
460             {
461                 Attribute* jsonAttribute = attributes->GetAttribute(U"json");
462                 if (jsonAttribute)
463                 {
464                     if (jsonAttribute->Value() == U"true")
465                     {
466                         BaseInitializerNode* baseInitializer = new BaseInitializerNode(span);
467                         baseInitializer->AddArgument(new IdentifierNode(span, U"@value"));
468                         constructorNode.AddInitializer(baseInitializer);
469                     }
470                 }
471             }
472         }
473         for (MemberVariableSymbol* memberVariableSymbol : classTypeSymbol->MemberVariables())
474         {
475             std::u32string jsonFieldName = memberVariableSymbol->Name();
476             Attributes* attributes = memberVariableSymbol->GetAttributes();
477             if (attributes)
478             {
479                 Attribute* jsonAttribute = attributes->GetAttribute(U"json");
480                 if (jsonAttribute)
481                 {
482                     if (jsonAttribute->Value() == U"false")
483                     {
484                         continue;
485                     }
486                 }
487                 Attribute* jsonFieldNameAttribute = attributes->GetAttribute(U"jsonFieldName");
488                 if (jsonFieldNameAttribute)
489                 {
490                     jsonFieldName = jsonFieldNameAttribute->Value();
491                 }
492             }
493             InvokeNode* invokeNode = new InvokeNode(memberVariableSymbol->GetSpan(), new IdentifierNode(span, U"FromJson"));
494             invokeNode->AddArgument(new IdentifierNode(memberVariableSymbol->GetSpan(), U"@value"));
495             invokeNode->AddArgument(new UStringLiteralNode(memberVariableSymbol->GetSpan(), jsonFieldName));
496             invokeNode->AddArgument(new IdentifierNode(memberVariableSymbol->GetSpan(), memberVariableSymbol->Name()));
497             ExpressionStatementNode* fromJsonStatement = new ExpressionStatementNode(memberVariableSymbol->GetSpan(), invokeNode);
498             compoundStatementNode.AddStatement(fromJsonStatement);
499         }
500         SymbolTable& symbolTable = statementBinder->GetBoundCompileUnit().GetSymbolTable();
501         symbolTable.BeginContainer(jsonConstructorSymbol);
502         SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
503         compoundStatementNode.Accept(symbolCreatorVisitor);
504         symbolTable.EndContainer();
505         TypeBinder typeBinder(statementBinder->GetBoundCompileUnit());
506         ContainerScope* containerScope = statementBinder->GetContainerScope();
507         typeBinder.SetContainerScope(containerScope);
508         statementBinder->SetContainerScope(jsonConstructorSymbol->GetContainerScope());
509         typeBinder.SetCurrentFunctionSymbol(jsonConstructorSymbol);
510         compoundStatementNode.Accept(typeBinder);
511         BoundFunction* prevFunction = statementBinder->CurrentFunction();
512         statementBinder->SetCurrentFunction(boundFunction.get());
513         ConstructorSymbol* prevConstructorSymbol = statementBinder->CurrentConstructorSymbol();
514         ConstructorNode* prevConstructorNode = statementBinder->CurrentConstructorNode();
515         statementBinder->SetCurrentConstructor(jsonConstructorSymbol, &constructorNode);
516         compoundStatementNode.Accept(*statementBinder);
517         statementBinder->SetContainerScope(containerScope);
518         BoundStatement* boundStatement = statementBinder->ReleaseStatement();
519         Assert(boundStatement->GetBoundNodeType() == BoundNodeType::boundCompoundStatement, "bound compound statement expected");
520         BoundCompoundStatement* compoundStatement = static_cast<BoundCompoundStatement*>(boundStatement);
521         boundFunction->SetBody(std::unique_ptr<BoundCompoundStatement>(compoundStatement));
522         statementBinder->CurrentClass()->AddMember(std::move(boundFunction));
523         statementBinder->SetCurrentConstructor(prevConstructorSymbol, prevConstructorNode);
524         statementBinder->SetCurrentFunction(prevFunction);
525         statementBinder->GetBoundCompileUnit().RemoveLastFileScope();
526     }
527     catch (const Exception& ex;)
528     {
529         std::vector<Span> references;
530         references.push_back(ex.Defined());
531         references.insert(references.end(), ex.References().begin(), ex.References().end());
532         throw Exception(module, "error in JSON attribute generation: could not create JSON constructor for class '" + ToUtf8(classTypeSymbol->FullName()) + "': " + ex.Message(), 
533             classTypeSymbol->GetSpan(), references);
534     }
