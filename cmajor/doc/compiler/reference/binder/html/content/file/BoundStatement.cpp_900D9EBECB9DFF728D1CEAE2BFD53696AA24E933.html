171 void BoundCaseStatement::AddStatement(std::std::unique_ptr<BoundStatement>&&statement)
172 {
173     if (compoundStatement)
174     {
175         compoundStatement->AddStatement(std::move(statement));
176     }
177     else if (statement->GetBoundNodeType() == BoundNodeType::boundCompoundStatement)
178     {
179         compoundStatement.reset(static_cast<BoundCompoundStatement*>(statement.release()));
180         compoundStatement->SetParent(this);
181     }
182     else
183     {
184         compoundStatement.reset(new BoundCompoundStatement(GetModule(), GetSpan()));
185         compoundStatement->SetParent(this);
186         compoundStatement->AddStatement(std::move(statement));
187     }
188 }
