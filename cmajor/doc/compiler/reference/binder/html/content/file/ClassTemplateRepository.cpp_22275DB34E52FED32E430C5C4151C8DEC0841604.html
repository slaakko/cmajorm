 35 void ClassTemplateRepository::ResolveDefaultTemplateArguments(std::std::vector<TypeSymbol*>&templateArgumentTypes, ClassTypeSymbol*classTemplate, ContainerScope*containerScope, constSpan&span)
 36 {
 37     int n = classTemplate->TemplateParameters().size();
 38     int m = templateArgumentTypes.size();
 39     if (m == n) return;
 40     SymbolTable& symbolTable = boundCompileUnit.GetSymbolTable();
 41     Node* node = symbolTable.GetNodeNoThrow(classTemplate);
 42     if (!node)
 43     {
 44         node = classTemplate->GetClassNode();
 45         Assert(node, "class node not read");
 46     }
 47     Assert(node->GetNodeType() == NodeType::classNode, "class node expected");
 48     ClassNode* classNode = static_cast<ClassNode*>(node);
 49     int numFileScopeAdded = 0;
 50     int nu = classTemplate->UsingNodes().Count();
 51     if (nu > 0)
 52     {
 53         FileScope* fileScope = new FileScope(&boundCompileUnit.GetModule());
 54         for (int i = 0; i < nu; ++i)
 55         {
 56             Node* usingNode = classTemplate->UsingNodes()[i];
 57             if (usingNode->GetNodeType() == NodeType::namespaceImportNode)
 58             {
 59                 NamespaceImportNode* namespaceImportNode = static_cast<NamespaceImportNode*>(usingNode);
 60                 fileScope->InstallNamespaceImport(containerScope, namespaceImportNode);
 61             }
 62             else if (usingNode->GetNodeType() == NodeType::aliasNode)
 63             {
 64                 AliasNode* aliasNode = static_cast<AliasNode*>(usingNode);
 65                 fileScope->InstallAlias(containerScope, aliasNode);
 66             }
 67         }
 68         boundCompileUnit.AddFileScope(fileScope);
 69         ++numFileScopeAdded;
 70     }
 71     if (!classTemplate->Ns()->IsGlobalNamespace())
 72     {
 73         FileScope* primaryFileScope = new FileScope(&boundCompileUnit.GetModule());
 74         primaryFileScope->AddContainerScope(classTemplate->Ns()->GetContainerScope());
 75         boundCompileUnit.AddFileScope(primaryFileScope);
 76         ++numFileScopeAdded;
 77     }
 78     ContainerScope resolveScope;
 79     resolveScope.SetParentScope(containerScope);
 80     std::vector<std::std::unique_ptr<BoundTemplateParameterSymbol>>boundTemplateParameters;
 81     for (int i = 0; i < n; ++i)
 82     {
 83         TemplateParameterSymbol* templateParameterSymbol = classTemplate->TemplateParameters()[i];
 84         BoundTemplateParameterSymbol* boundTemplateParameter = new BoundTemplateParameterSymbol(span, templateParameterSymbol->Name());
 85         boundTemplateParameters.push_back(std::unique_ptr<BoundTemplateParameterSymbol>(boundTemplateParameter));
 86         if (i < m)
 87         {
 88             boundTemplateParameter->SetType(templateArgumentTypes[i]);
 89             resolveScope.Install(boundTemplateParameter);
 90         }
 91         else
 92         {
 93             if (i >= classNode->TemplateParameters().Count())
 94             {
 95                 throw Exception(&boundCompileUnit.GetModule(), "too few template arguments", span);
 96             }
 97             Node* defaultTemplateArgumentNode = classNode->TemplateParameters()[i]->DefaultTemplateArgument();
 98             if (!defaultTemplateArgumentNode)
 99             {
100                 throw Exception(&boundCompileUnit.GetModule(), "too few template arguments", span);
101             }
102             TypeSymbol* templateArgumentType = ResolveType(defaultTemplateArgumentNode, boundCompileUnit, &resolveScope);
103             templateArgumentTypes.push_back(templateArgumentType);
104         }
105     }
106     for (int i = 0; i < numFileScopeAdded; ++i)
107     {
108         boundCompileUnit.RemoveLastFileScope();
109     }
110 }

112 void ClassTemplateRepository::BindClassTemplateSpecialization(ClassTemplateSpecializationSymbol* classTemplateSpecialization, ContainerScope* containerScope, const Span& span)
113 {
114     if (classTemplateSpecialization->IsBound()) return;
115     SymbolTable& symbolTable = boundCompileUnit.GetSymbolTable();
116     ClassTypeSymbol* classTemplate = classTemplateSpecialization->GetClassTemplate();
117     Node* node = symbolTable.GetNodeNoThrow(classTemplate);
118     if (!node)
119     {
120         node = classTemplate->GetClassNode();
121         Assert(node, "class node not read");
122     }
123     Assert(node->GetNodeType() == NodeType::classNode, "class node expected");
124     ClassNode* classNode = static_cast<ClassNode*>(node);
125     std::unique_ptr<NamespaceNode> globalNs(new NamespaceNode(classTemplate->GetSpan(), new IdentifierNode(classTemplate->GetSpan(), U"")));
126     NamespaceNode* currentNs = globalNs.get();
127     CloneContext cloneContext;
128     cloneContext.SetInstantiateClassNode();
129     int nu = classTemplate->UsingNodes().Count();
130     for (int i = 0; i < nu; ++i)
131     {
132         Node* usingNode = classTemplate->UsingNodes()[i];
133         globalNs->AddMember(usingNode->Clone(cloneContext));
134     }
135     bool fileScopeAdded = false;
136     if (!classTemplate->Ns()->IsGlobalNamespace())
137     {
138         FileScope* primaryFileScope = new FileScope(&boundCompileUnit.GetModule());
139         primaryFileScope->AddContainerScope(classTemplate->Ns()->GetContainerScope());
140         boundCompileUnit.AddFileScope(primaryFileScope);
141         fileScopeAdded = true;
142         std::u32string fullNsName = classTemplate->Ns()->FullName();
143         std::vector<std::u32string> nsComponents = Split(fullNsName, '.');
144         for (const std::u32string& nsComponent : nsComponents)
145         {
146             NamespaceNode* nsNode = new NamespaceNode(classTemplate->GetSpan(), new IdentifierNode(classTemplate->GetSpan(), nsComponent));
147             currentNs->AddMember(nsNode);
148             currentNs = nsNode;
149         }
150     }
151     ClassNode* classInstanceNode = static_cast<ClassNode*>(classNode->Clone(cloneContext));
152     currentNs->AddMember(classInstanceNode);
153     int n = classTemplate->TemplateParameters().size();
154     int m = classTemplateSpecialization->TemplateArgumentTypes().size();
155     if (n != m)
156     {
157         throw Exception(&boundCompileUnit.GetModule(), "wrong number of template arguments", span);
158     }
159     bool templateParameterBinding = false;
160     ContainerScope resolveScope;
161     resolveScope.SetParentScope(containerScope);
162     for (int i = 0; i < n; ++i)
163     {
164         TemplateParameterSymbol* templateParameter = classTemplate->TemplateParameters()[i];
165         BoundTemplateParameterSymbol* boundTemplateParameter = new BoundTemplateParameterSymbol(span, templateParameter->Name());
166         boundTemplateParameter->SetParent(classTemplateSpecialization);
167         TypeSymbol* templateArgumentType = classTemplateSpecialization->TemplateArgumentTypes()[i];
168         boundTemplateParameter->SetType(templateArgumentType);
169         if (templateArgumentType->GetSymbolType() == SymbolType::templateParameterSymbol)
170         {
171             templateParameterBinding = true;
172             if (classTemplateSpecialization->IsPrototype())
173             {
174                 if (classTemplateSpecialization->IsProject())
175                 {
176                     resolveScope.Install(boundTemplateParameter);
177                     TemplateParameterNode* templateParameterNode = classNode->TemplateParameters()[i];
178                     Node* defaultTemplateArgumentNode = templateParameterNode->DefaultTemplateArgument();
179                     if (defaultTemplateArgumentNode)
180                     {
181                         TypeSymbol* templateArgumentType = ResolveType(defaultTemplateArgumentNode, boundCompileUnit, &resolveScope);
182                         templateParameter->SetDefaultType(templateArgumentType);
183                     }
184                 }
185             }
186         }
187         classTemplateSpecialization->AddMember(boundTemplateParameter);
188     }
189     symbolTable.SetCurrentCompileUnit(boundCompileUnit.GetCompileUnitNode());
190     SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
191     symbolCreatorVisitor.SetClassInstanceNode(classInstanceNode);
192     symbolCreatorVisitor.SetClassTemplateSpecialization(classTemplateSpecialization);
193     globalNs->Accept(symbolCreatorVisitor);
194     TypeBinder typeBinder(boundCompileUnit);
195     if (templateParameterBinding)
196     {
197         typeBinder.CreateMemberSymbols();
198     }
199     globalNs->Accept(typeBinder);
200     if (templateParameterBinding)
201     {
202         classTemplateSpecialization->SetGlobalNs(std::move(globalNs));
203         if (fileScopeAdded)
204         {
205             boundCompileUnit.RemoveLastFileScope();
206         }
207     }
208     else if (boundCompileUnit.BindingTypes())
209     {
210         classTemplateSpecialization->SetGlobalNs(std::move(globalNs));
211         classTemplateSpecialization->SetStatementsNotBound();
212         if (fileScopeAdded)
213         {
214             FileScope* fileScope = boundCompileUnit.ReleaseLastFileScope();
215             classTemplateSpecialization->SetFileScope(fileScope);
216         }
217     }
218     else
219     {
220         StatementBinder statementBinder(boundCompileUnit);
221         globalNs->Accept(statementBinder);
222         classTemplateSpecialization->SetGlobalNs(std::move(globalNs));
223         if (fileScopeAdded)
224         {
225             boundCompileUnit.RemoveLastFileScope();
226         }
227     }
228 }

230 bool ClassTemplateRepository::Instantiate(FunctionSymbol* memberFunction, ContainerScope* containerScope, BoundFunction* currentFunction, const Span& span)
231 {
232     if (instantiatedMemberFunctions.find(memberFunction) != instantiatedMemberFunctions.cend()) return true;
233     instantiatedMemberFunctions.insert(memberFunction);
234     try
235     {
236         SymbolTable& symbolTable = boundCompileUnit.GetSymbolTable();
237         Symbol* parent = memberFunction->Parent();
238         Assert(parent->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol, "class template specialization expected");
239         ClassTemplateSpecializationSymbol* classTemplateSpecialization = static_cast<ClassTemplateSpecializationSymbol*>(parent);
240         std::pair<boost::uuids::uuid, int> classIdMemFunIndexPair = std::make_pair(classTemplateSpecialization->TypeId(), memberFunction->GetIndex());
241         if (classIdMemberFunctionIndexSet.find(classIdMemFunIndexPair) != classIdMemberFunctionIndexSet.cend())
242         {
243 //          If <parent class id, member function index> pair is found from the classIdMemberFunctionIndexSet, the member function is already instantiated 
244 //          for this compile unit, so return true.
245             instantiatedMemberFunctions.insert(memberFunction);
246             return true;
247         }
248         Assert(classTemplateSpecialization->IsBound(), "class template specialization not bound");
249         Node* node = symbolTable.GetNodeNoThrow(memberFunction);
250         if (!node)
251         {
252             return false;
253         }
254         boundCompileUnit.FinalizeBinding(classTemplateSpecialization);
255         ClassTypeSymbol* classTemplate = classTemplateSpecialization->GetClassTemplate();
256         std::unordered_map<TemplateParameterSymbol*, TypeSymbol*> templateParameterMap;
257         int n = classTemplateSpecialization->TemplateArgumentTypes().size();
258         for (int i = 0; i < n; ++i)
259         {
260             TemplateParameterSymbol* templateParameter = classTemplate->TemplateParameters()[i];
261             TypeSymbol* templateArgument = classTemplateSpecialization->TemplateArgumentTypes()[i];
262             templateParameterMap[templateParameter] = templateArgument;
263         }
264         if (!classTemplateSpecialization->IsConstraintChecked())
265         {
266             classTemplateSpecialization->SetConstraintChecked();
267             if (classTemplate->Constraint())
268             {
269                 std::unique_ptr<BoundConstraint> boundConstraint;
270                 std::unique_ptr<Exception> conceptCheckException;
271                 if (!CheckConstraint(classTemplate->Constraint(), classTemplate->UsingNodes(), boundCompileUnit, containerScope, currentFunction, classTemplate->TemplateParameters(), 
272                     templateParameterMap, boundConstraint, span, memberFunction, conceptCheckException))
273                 {
274                     if (conceptCheckException)
275                     {
276                         throw Exception(&boundCompileUnit.GetModule(), "concept check of class template specialization '" + ToUtf8(classTemplateSpecialization->FullName()) + "' failed: " + conceptCheckException->Message(), span, 
277                             conceptCheckException->References());
278                     }
279                     else
280                     {
281                         throw Exception(&boundCompileUnit.GetModule(), "concept check of class template specialization '" + ToUtf8(classTemplateSpecialization->FullName()) + "' failed.", span);
282                     }
283                 }
284             }
285         }
286         FileScope* fileScope = new FileScope(&boundCompileUnit.GetModule());
287         int nu = classTemplate->UsingNodes().Count();
288         for (int i = 0; i < nu; ++i)
289         {
290             Node* usingNode = classTemplate->UsingNodes()[i];
291             if (usingNode->GetNodeType() == NodeType::namespaceImportNode)
292             {
293                 NamespaceImportNode* namespaceImportNode = static_cast<NamespaceImportNode*>(usingNode);
294                 fileScope->InstallNamespaceImport(containerScope, namespaceImportNode);
295             }
296             else if (usingNode->GetNodeType() == NodeType::aliasNode)
297             {
298                 AliasNode* aliasNode = static_cast<AliasNode*>(usingNode);
299                 fileScope->InstallAlias(containerScope, aliasNode);
300             }
301         }
302         if (!classTemplate->Ns()->IsGlobalNamespace())
303         {
304             fileScope->AddContainerScope(classTemplate->Ns()->GetContainerScope());
305         }
306         boundCompileUnit.AddFileScope(fileScope);
307         Assert(node->IsFunctionNode(), "function node expected");
308         FunctionNode* functionInstanceNode = static_cast<FunctionNode*>(node);
309         Assert(functionInstanceNode->BodySource(), "body source expected");
310         CloneContext cloneContext;
311         functionInstanceNode->SetBody(static_cast<CompoundStatementNode*>(functionInstanceNode->BodySource()->Clone(cloneContext)));
312         if (functionInstanceNode->WhereConstraint())
313         {
314             std::unique_ptr<BoundConstraint> boundConstraint;
315             std::unique_ptr<Exception> conceptCheckException;
316             FileScope* classTemplateScope = new FileScope(&boundCompileUnit.GetModule());
317             classTemplateScope->AddContainerScope(classTemplateSpecialization->GetContainerScope());
318             boundCompileUnit.AddFileScope(classTemplateScope);
319             if (!CheckConstraint(functionInstanceNode->WhereConstraint(), classTemplate->UsingNodes(), boundCompileUnit, containerScope, currentFunction, classTemplate->TemplateParameters(), 
320                 templateParameterMap, boundConstraint, span, memberFunction, conceptCheckException))
321             {
322                 boundCompileUnit.RemoveLastFileScope();
323                 if (conceptCheckException)
324                 {
325                     std::vector<Span> references;
326                     references.push_back(conceptCheckException->Defined());
327                     references.insert(references.end(), conceptCheckException->References().begin(), conceptCheckException->References().end());
328                     throw Exception(&boundCompileUnit.GetModule(), "concept check of class template member function '" + ToUtf8(memberFunction->FullName()) + "' failed: " + conceptCheckException->Message(), span, references);
329                 }
330                 else
331                 {
332                     throw Exception(&boundCompileUnit.GetModule(), "concept check of class template template member function '" + ToUtf8(memberFunction->FullName()) + "' failed.", span);
333                 }
334             }
335             else
336             {
337                 boundCompileUnit.RemoveLastFileScope();
338             }
339         }
340         symbolTable.SetCurrentCompileUnit(boundCompileUnit.GetCompileUnitNode());
341         SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
342         symbolTable.BeginContainer(memberFunction);
343         functionInstanceNode->Body()->Accept(symbolCreatorVisitor);
344         symbolTable.EndContainer();
345         TypeBinder typeBinder(boundCompileUnit);
346         typeBinder.SetContainerScope(memberFunction->GetContainerScope());
347         functionInstanceNode->Body()->Accept(typeBinder);
348         StatementBinder statementBinder(boundCompileUnit);
349         std::unique_ptr<BoundClass> boundClass(new BoundClass(&boundCompileUnit.GetModule(), classTemplateSpecialization));
350         statementBinder.SetCurrentClass(boundClass.get());
351         std::unique_ptr<BoundFunction> boundFunction(new BoundFunction(&boundCompileUnit.GetModule(), memberFunction));
352         statementBinder.SetCurrentFunction(boundFunction.get());
353         statementBinder.SetContainerScope(memberFunction->GetContainerScope());
354         if (memberFunction->GetSymbolType() == SymbolType::constructorSymbol)
355         {
356             ConstructorSymbol* constructorSymbol = static_cast<ConstructorSymbol*>(memberFunction);
357             Node* node = symbolTable.GetNode(memberFunction);
358             Assert(node->GetNodeType() == NodeType::constructorNode, "constructor node expected");
359             ConstructorNode* constructorNode = static_cast<ConstructorNode*>(node);
360             statementBinder.SetCurrentConstructor(constructorSymbol, constructorNode);
361         }
362         else if (memberFunction->GetSymbolType() == SymbolType::destructorSymbol)
363         {
364             DestructorSymbol* destructorSymbol = static_cast<DestructorSymbol*>(memberFunction);
365             Node* node = symbolTable.GetNode(memberFunction);
366             Assert(node->GetNodeType() == NodeType::destructorNode, "destructor node expected");
367             DestructorNode* destructorNode = static_cast<DestructorNode*>(node);
368             statementBinder.SetCurrentDestructor(destructorSymbol, destructorNode);
369         }
370         else if (memberFunction->GetSymbolType() == SymbolType::memberFunctionSymbol)
371         {
372             MemberFunctionSymbol* memberFunctionSymbol = static_cast<MemberFunctionSymbol*>(memberFunction);
373             Node* node = symbolTable.GetNode(memberFunction);
374             Assert(node->GetNodeType() == NodeType::memberFunctionNode, "member function node expected");
375             MemberFunctionNode* memberFunctionNode = static_cast<MemberFunctionNode*>(node);
376             statementBinder.SetCurrentMemberFunction(memberFunctionSymbol, memberFunctionNode);
377         }
378         functionInstanceNode->Body()->Accept(statementBinder);
379         BoundStatement* boundStatement = statementBinder.ReleaseStatement();
380         Assert(boundStatement->GetBoundNodeType() == BoundNodeType::boundCompoundStatement, "bound compound statement expected");
381         BoundCompoundStatement* compoundStatement = static_cast<BoundCompoundStatement*>(boundStatement);
382         boundFunction->SetBody(std::unique_ptr<BoundCompoundStatement>(compoundStatement));
383         std::u32string instantiatedMemberFunctionMangledName = boundFunction->GetFunctionSymbol()->MangledName();
384         boundClass->AddMember(std::move(boundFunction));
385         classIdMemberFunctionIndexSet.insert(classIdMemFunIndexPair);
386         boundCompileUnit.AddBoundNode(std::move(boundClass));
387         boundCompileUnit.RemoveLastFileScope();
388         return InstantiateDestructorAndVirtualFunctions(classTemplateSpecialization, containerScope, currentFunction, span);
389     }
390     catch (const Exception& ex;)
391     {
392         std::vector<Span> references;
393         references.push_back(memberFunction->GetSpan());
394         references.push_back(ex.Defined());
395         references.insert(references.end(), ex.References().begin(), ex.References().end());
396         throw Exception(&boundCompileUnit.GetModule(), "could not instantiate member function '" + ToUtf8(memberFunction->FullName()) + "'. Reason: " + ex.Message(), span, references);
397     }
398 }
399 
400 bool ClassTemplateRepository::InstantiateDestructorAndVirtualFunctions(ClassTemplateSpecializationSymbol* classTemplateSpecialization, ContainerScope* containerScope, BoundFunction* currentFunction, const Span& span)
401 {
402     for (FunctionSymbol* virtualMemberFunction : classTemplateSpecialization->Vmt())
403     {
404         if (virtualMemberFunction->Parent() == classTemplateSpecialization && !virtualMemberFunction->IsGeneratedFunction())
405         {
406             if (!Instantiate(virtualMemberFunction, containerScope, currentFunction, span))
407             {
408                 return false;
409             }
410         }
411     }
412     if (classTemplateSpecialization->Destructor())
413     {
414         if (!classTemplateSpecialization->Destructor()->IsGeneratedFunction())
415         {
416             if (!Instantiate(classTemplateSpecialization->Destructor(), containerScope, currentFunction, span))
417             {
418                 return false;
419             }
420         }
421     }
422     return true;
423 }
