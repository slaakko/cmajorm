10 BoundConcept* ConceptRepository::GetBoundConcept(const BoundConceptKey& key) const
11 {
12     auto it = boundConceptMap.find(key);
13     if (it != boundConceptMap.cend())
14     {
15         return it->second;
16     }
17     else
18     {
19         return nullptr;
20     }
21 }
