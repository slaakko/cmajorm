193 void TypeResolver::Visit(ArrayNode& arrayNode)
194 {
195     type = ResolveType(arrayNode.Subject(), boundCompileUnit, containerScope, currentClass);
196     if (type->IsReferenceType())
197     {
198         throw Exception(module, "cannot have array of reference type", arrayNode.GetSpan());
199     }
200     int64_t size = -1;
201     if (arrayNode.Size())
202     {
203         std::unique_ptr<Value> sizeValue = Evaluate(arrayNode.Size(), boundCompileUnit.GetSymbolTable().GetTypeByName(U"long"), containerScope, boundCompileUnit, false, nullptr, arrayNode.GetSpan());
204         if (sizeValue->GetValueType() == ValueType::longValue)
205         {
206             LongValue* longSizeValue = static_cast<LongValue*>(sizeValue.get());
207             size = longSizeValue->GetValue();
208         }
209         else
210         {
211             throw Exception(module, "long type value expected ", arrayNode.Size()->GetSpan());
212         }
213     }
214     type = symbolTable.MakeArrayType(type, size, arrayNode.GetSpan());
215 }

217 void TypeResolver::ResolveSymbol(Node& node, IdentifierNode* idNode, Symbol* symbol)
218 {
219     if (symbol->IsTypeSymbol())
220     {
221         type = static_cast<TypeSymbol*>(symbol);
222         if (GetGlobalFlag(GlobalFlags::cmdoc))
223         {
224             TypeSymbol* cmdocType = type;
225             if (type->GetSymbolType() == SymbolType::classGroupTypeSymbol)
226             {
227                 ClassGroupTypeSymbol* group = static_cast<ClassGroupTypeSymbol*>(type);
228                 cmdocType = group->GetClass(0);
229             }
230             symbolTable.MapSymbol(idNode, cmdocType);
231         }
232     }
233     else
234     {
235         switch (symbol->GetSymbolType())
236         {
237             case SymbolType::typedefSymbol:
238             {
239                 TypedefSymbol* typedefSymbol = static_cast<TypedefSymbol*>(symbol);
240                 if (typedefSymbol->IsProject() && !typedefSymbol->IsBound())
241                 {
242                     TypeBinder typeBinder(boundCompileUnit);
243                     typeBinder.SetContainerScope(typedefSymbol->Parent()->GetContainerScope());
244                     Node* node = symbolTable.GetNode(typedefSymbol);
245                     Assert(node->GetNodeType() == NodeType::typedefNode, "typedef node expected");
246                     TypedefNode* typedefNode = static_cast<TypedefNode*>(node);
247                     typeBinder.BindTypedef(typedefSymbol, typedefNode, false);
248                 }
249                 type = typedefSymbol->GetType();
250                 if (GetGlobalFlag(GlobalFlags::cmdoc))
251                 {
252                     symbolTable.MapSymbol(idNode, typedefSymbol);
253                 }
254                 break;
255             }
256             case SymbolType::boundTemplateParameterSymbol:
257             {
258                 BoundTemplateParameterSymbol* boundTemplateParameterSymbol = static_cast<BoundTemplateParameterSymbol*>(symbol);
259                 type = boundTemplateParameterSymbol->GetType();
260                 break;
261             }
262             case SymbolType::namespaceSymbol:
263             {
264                 NamespaceSymbol* ns = static_cast<NamespaceSymbol*>(symbol);
265                 nsTypeSymbol.reset(new NamespaceTypeSymbol(ns));
266                 symbolTable.SetTypeIdFor(nsTypeSymbol.get());
267                 type = nsTypeSymbol.get();
268                 break;
269             }
270             default:
271             {
272                 throw Exception(module, "symbol '" + ToUtf8(symbol->FullName()) + "' does not denote a type", node.GetSpan(), symbol->GetSpan());
273             }
274         }
275     }
276 }

278 void TypeResolver::Visit(IdentifierNode& identifierNode)
279 {
280     symbolTable.SetLatestIdentifier(&identifierNode);
281     std::u32string name = identifierNode.Str();
282     Symbol* symbol = containerScope->Lookup(name, ScopeLookup::this_and_base_and_parent);
283     if (!symbol)
284     {
285         for (const std::std::unique_ptr<FileScope>&fileScope : boundCompileUnit.FileScopes())
286         {
287             symbol = fileScope->Lookup(name);
288             if (symbol)
289             {
290                 break;
291             }
292         }
293     }
294     if (symbol)
295     {
296         ResolveSymbol(identifierNode, &identifierNode, symbol);
297     }
298     else
299     {
300         throw Exception(module, "type symbol '" + ToUtf8(name) + "' not found", identifierNode.GetSpan());
301     }
302 }

304 void TypeResolver::Visit(TemplateIdNode& templateIdNode)
305 {
306     IdentifierNode* prevId = symbolTable.GetLatestIdentifier();
307     int arity = templateIdNode.TemplateArguments().Count();
308     TypeSymbol* primaryTemplateType = ResolveType(templateIdNode.Primary(), boundCompileUnit, containerScope, TypeResolverFlags::resolveClassGroup, currentClass);
309     if (primaryTemplateType->GetSymbolType() == SymbolType::classGroupTypeSymbol)
310     {
311         ClassGroupTypeSymbol* classGroup = static_cast<ClassGroupTypeSymbol*>(primaryTemplateType);
312         ClassTypeSymbol* classType = classGroup->GetClass(arity);
313         if (classType)
314         {
315             primaryTemplateType = classType;
316         }
317         else
318         {
319             throw Exception(classGroup->GetModule(), "primary class template with arity '" + std::to_string(arity) + "' not found", classGroup->GetSpan());
320         }
321     }
322     if (!primaryTemplateType->IsClassTypeSymbol())
323     {
324         throw Exception(module, "class type symbol expected", templateIdNode.Primary()->GetSpan());
325     }
326     ClassTypeSymbol* classTemplate = static_cast<ClassTypeSymbol*>(primaryTemplateType);
327     if (!classTemplate->IsClassTemplate())
328     {
329         throw Exception(module, "class template expected", templateIdNode.Primary()->GetSpan());
330     }
331     IdentifierNode* idNode = symbolTable.GetLatestIdentifier();
332     symbolTable.SetLatestIdentifier(prevId);
333     if (GetGlobalFlag(GlobalFlags::cmdoc))
334     {
335         symbolTable.MapSymbol(idNode, classTemplate);
336     }
337     if (classTemplate->IsProject() && !classTemplate->IsBound())
338     {
339         TypeBinder typeBinder(boundCompileUnit);
340         Node* node = boundCompileUnit.GetSymbolTable().GetNodeNoThrow(classTemplate);
341         if (node && node->GetNodeType() == NodeType::classNode)
342         {
343             typeBinder.AddUsingNodesToCurrentCompileUnit(static_cast<ClassNode*>(node));
344         }
345     }
346     std::vector<TypeSymbol*> templateArgumentTypes;
347     int n = arity;
348     for (int i = 0; i < n; ++i)
349     {
350         IdentifierNode* prevId = symbolTable.GetLatestIdentifier();
351         TypeSymbol* templateArgumentType = ResolveType(templateIdNode.TemplateArguments()[i], boundCompileUnit, containerScope, currentClass);
352         templateArgumentTypes.push_back(templateArgumentType);
353         IdentifierNode* idNode = symbolTable.GetLatestIdentifier();
354         if (idNode && GetGlobalFlag(GlobalFlags::cmdoc))
355         {
356             symbolTable.MapSymbol(idNode, templateArgumentType);
357         }
358         symbolTable.SetLatestIdentifier(prevId);
359     }
360     int m = classTemplate->TemplateParameters().size();
361     if (n < m)
362     {
363         classTemplateRepository.ResolveDefaultTemplateArguments(templateArgumentTypes, classTemplate, containerScope, templateIdNode.GetSpan());
364     }
365     ClassTemplateSpecializationSymbol* classTemplateSpecialization = symbolTable.MakeClassTemplateSpecialization(classTemplate, templateArgumentTypes, templateIdNode.GetSpan());
366     if (!classTemplateSpecialization->IsBound())
367     {
368         classTemplateRepository.BindClassTemplateSpecialization(classTemplateSpecialization, containerScope, templateIdNode.GetSpan());
369     }
370     type = classTemplateSpecialization;
371 }

373 void TypeResolver::Visit(DotNode& dotNode)
374 {
375     dotNode.Subject()->Accept(*this);
376     Scope* scope = nullptr;
377     if (type->GetSymbolType() == SymbolType::classGroupTypeSymbol)
378     {
379         ClassGroupTypeSymbol* classGroup = static_cast<ClassGroupTypeSymbol*>(type);
380         if (currentClass && classGroup->Name() == currentClass->Name())
381         {
382             type = currentClass;
383         }
384         else
385         {
386             type = classGroup->GetClass(0);
387         }
388         if (!type)
389         {
390             throw Exception(module, "symbol '" + ToUtf8(type->FullName()) + "' does not denote a class type, an array type or a namespace", dotNode.GetSpan(), type->GetSpan());
391         }
392     }
393     if (type->GetSymbolType() == SymbolType::namespaceTypeSymbol)
394     {
395         NamespaceTypeSymbol* nsType = static_cast<NamespaceTypeSymbol*>(type);
396         scope = nsType->Ns()->GetContainerScope();
397     }
398     else if (type->IsClassTypeSymbol() || 
399         type->IsArrayType() || 
400         type->IsEnumeratedType() || 
401         (flags & TypeResolverFlags::createMemberSymbols) != TypeResolverFlags::none && type->GetSymbolType() == SymbolType::templateParameterSymbol)
402     {
403         scope = type->GetContainerScope();
404     }
405     else
406     {
407         throw Exception(module, "symbol '" + ToUtf8(type->FullName()) + "' does not denote a class type, an array type or a namespace", dotNode.GetSpan(), type->GetSpan());
408     }
409     std::u32string name = dotNode.MemberId()->Str();
410     Symbol* symbol = scope->Lookup(name, ScopeLookup::this_and_base);
411     if (symbol)
412     {
413         ResolveSymbol(dotNode, dotNode.MemberId(), symbol);
414     }
415     else
416     {
417         if ((flags & TypeResolverFlags::createMemberSymbols) != TypeResolverFlags::none && type->GetSymbolType() == SymbolType::templateParameterSymbol)
418         {
419             TemplateParameterSymbol* templateParameterSymbol = new TemplateParameterSymbol(dotNode.GetSpan(), name);
420             templateParameterSymbol->SetModule(module);
421             symbolTable.SetTypeIdFor(templateParameterSymbol);
422             type->AddMember(templateParameterSymbol);
423             ResolveSymbol(dotNode, dotNode.MemberId(), templateParameterSymbol);
424         }
425         else
426         {
427             throw Exception(module, "type symbol '" + ToUtf8(name) + "' not found", dotNode.GetSpan());
428         }
429     }
430 }

447 TypeSymbol* ResolveType(Node* typeExprNode, BoundCompileUnit& boundCompileUnit, ContainerScope* containerScope, TypeResolverFlags flags, ClassTypeSymbol* currentClass)
448 {
449     Module* module = &boundCompileUnit.GetModule();
450     bool resolveClassGroup = (flags & TypeResolverFlags::resolveClassGroup) != TypeResolverFlags::none;
451     TypeResolver typeResolver(boundCompileUnit, containerScope, flags, currentClass);
452     typeExprNode->Accept(typeResolver);
453     TypeSymbol* type = typeResolver.GetType();
454     if (resolveClassGroup && type && type->GetSymbolType() == SymbolType::classGroupTypeSymbol)
455     {
456         return type;
457     }
458     if (type && type->GetSymbolType() == SymbolType::classGroupTypeSymbol)
459     {
460         ClassGroupTypeSymbol* classGroup = static_cast<ClassGroupTypeSymbol*>(type);
461         if (currentClass && classGroup->Name() == currentClass->GroupName())
462         {
463             type = currentClass;
464         }
465         else
466         {
467             type = classGroup->GetClass(0);
468         }
469     }
470     if (!type || type->IsInComplete())
471     {
472         throw Exception(module, "incomplete type expression", typeExprNode->GetSpan());
473     }
474     TypeDerivationRec derivationRec = UnifyDerivations(typeResolver.DerivationRec(), type->DerivationRec());
475     if (!derivationRec.derivations.empty())
476     {
477         return boundCompileUnit.GetSymbolTable().MakeDerivedType(type->BaseType(), derivationRec, typeExprNode->GetSpan());
478     }
479     return type;
480 }
