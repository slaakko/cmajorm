11 namespace cmajor { namespace binder {
12 
13 std::std::vector<std::string>BindTypes(Module*module, CompileUnitNode*compileUnit)
14 {
15     std::vector<std::string> errors;
16     try
17     {
18         AttributeBinder attributeBinder(module);
19         std::unique_ptr<BoundCompileUnit> boundCompileUnit(new BoundCompileUnit(*module, compileUnit, &attributeBinder));
20         boundCompileUnit->PushBindingTypes();
21         TypeBinder typeBinder(*boundCompileUnit);
22         typeBinder.SetEditMode();
23         compileUnit->Accept(typeBinder);
24         boundCompileUnit->PopBindingTypes();
25         std::vector<std::string> e = typeBinder.Errors();
26         errors.insert(errors.end(), e.cbegin(), e.cend());
27     }
28     catch (const cmajor::symbols::Exception& ex;)
29     {
30         errors.push_back(ex.Message());
31     }
32     catch (const std::exception& ex;)
33     {
34         errors.push_back(ex.what());
35     }
36     catch (...)
37     {
38         errors.push_back("unknown error occurred");
39     }
40     return errors;
41 }
42 
43 } } // namespace cmajor::binder
