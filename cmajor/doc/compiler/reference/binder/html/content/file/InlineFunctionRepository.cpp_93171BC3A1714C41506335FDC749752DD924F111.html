 21 InlineFunctionRepository::InlineFunctionRepository(BoundCompileUnit& boundCompileUnit_) : boundCompileUnit(boundCompileUnit_)
 22 {
 23 }
 24 
 25 FunctionSymbol* InlineFunctionRepository::Instantiate(FunctionSymbol* inlineFunction, ContainerScope* containerScope, const Span& span)
 26 {
 27     if (inlineFunction->GetCompileUnit() == boundCompileUnit.GetCompileUnitNode()) return inlineFunction;
 28     while (inlineFunction->Master())
 29     {
 30         inlineFunction = inlineFunction->Master();
 31     }
 32     auto it = inlineFunctionMap.find(inlineFunction);
 33     if (it != inlineFunctionMap.cend())
 34     {
 35         return it->second;
 36     }
 37     SymbolTable& symbolTable = boundCompileUnit.GetSymbolTable();
 38     Node* node = symbolTable.GetNodeNoThrow(inlineFunction);
 39     if (!node)
 40     {
 41         node = inlineFunction->GetFunctionNode();
 42         symbolTable.MapNode(node, inlineFunction);
 43         Assert(node, "function node not read");
 44     }
 45     FunctionNode* functionNode = static_cast<FunctionNode*>(node);
 46     SpanMapper spanMapper;
 47     std::unique_ptr<NamespaceNode> globalNs(new NamespaceNode(spanMapper.MapSpan(functionNode->GetSpan(), functionNode->RootModuleId()), 
 48         new IdentifierNode(spanMapper.MapSpan(functionNode->GetSpan(), functionNode->RootModuleId()), U"")));
 49     NamespaceNode* currentNs = globalNs.get();
 50     CloneContext cloneContext;
 51     cloneContext.SetSpanMapper(&spanMapper);
 52     cloneContext.SetInstantiateFunctionNode();
 53     bool fileScopeAdded = false;
 54     int n = inlineFunction->UsingNodes().Count();
 55     if (!inlineFunction->Ns()->IsGlobalNamespace() || n > 0)
 56     {
 57         FileScope* primaryFileScope = new FileScope(&boundCompileUnit.GetModule());
 58         if (!inlineFunction->Ns()->IsGlobalNamespace())
 59         {
 60             primaryFileScope->AddContainerScope(inlineFunction->Ns()->GetContainerScope());
 61         }
 62         for (int i = 0; i < n; ++i)
 63         {
 64             Node* usingNode = inlineFunction->UsingNodes()[i];
 65             if (usingNode->GetNodeType() == NodeType::namespaceImportNode)
 66             {
 67                 primaryFileScope->InstallNamespaceImport(containerScope, static_cast<NamespaceImportNode*>(usingNode));
 68             }
 69             else if (usingNode->GetNodeType() == NodeType::aliasNode)
 70             {
 71                 primaryFileScope->InstallAlias(containerScope, static_cast<AliasNode*>(usingNode));
 72             }
 73         }
 74         boundCompileUnit.AddFileScope(primaryFileScope);
 75         fileScopeAdded = true;
 76         std::u32string fullNsName = inlineFunction->Ns()->FullName();
 77         std::vector<std::u32string> nsComponents = Split(fullNsName, '.');
 78         for (const std::u32string& nsComponent : nsComponents)
 79         {
 80             NamespaceNode* nsNode = new NamespaceNode(spanMapper.MapSpan(functionNode->GetSpan(), functionNode->RootModuleId()), 
 81                 new IdentifierNode(spanMapper.MapSpan(functionNode->GetSpan(), functionNode->RootModuleId()), nsComponent));
 82             currentNs->AddMember(nsNode);
 83             currentNs = nsNode;
 84         }
 85     }
 86     FunctionNode* functionInstanceNode = static_cast<FunctionNode*>(functionNode->Clone(cloneContext));
 87     if (inlineFunction->IsDefault())
 88     {
 89         functionInstanceNode->SetBody(new sngcm::ast::CompoundStatementNode(span));
 90         inlineFunction->SetHasArtificialBody();
 91     }
 92     currentNs->AddMember(functionInstanceNode);
 93     symbolTable.SetCurrentCompileUnit(boundCompileUnit.GetCompileUnitNode());
 94     if (!inlineFunction->Parent()->IsClassTypeSymbol())
 95     {
 96         SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
 97         symbolCreatorVisitor.SetLeaveFunction();
 98         globalNs->Accept(symbolCreatorVisitor);
 99         std::unique_ptr<FunctionSymbol> functionSymbol(symbolTable.GetCreatedFunctionSymbol());
100         if (inlineFunction->IsDefault())
101         {
102             functionSymbol->SetHasArtificialBody();
103         }
104         functionSymbol->SetParent(inlineFunction->Parent());
105         functionSymbol->SetLinkOnceOdrLinkage();
106         if (inlineFunction->IsSystemDefault())
107         {
108             functionSymbol->SetSystemDefault();
109         }
110         TypeBinder typeBinder(boundCompileUnit);
111         typeBinder.SetContainerScope(functionSymbol->GetContainerScope());
112         typeBinder.SetCurrentFunctionSymbol(functionSymbol.get());
113         functionInstanceNode->Accept(typeBinder);
114         StatementBinder statementBinder(boundCompileUnit);
115         std::unique_ptr<BoundFunction> boundFunction(new BoundFunction(&boundCompileUnit, functionSymbol.get()));
116         statementBinder.SetCurrentFunction(boundFunction.get());
117         statementBinder.SetContainerScope(functionSymbol->GetContainerScope());
118         functionInstanceNode->Body()->Accept(statementBinder);
119         BoundStatement* boundStatement = statementBinder.ReleaseStatement();
120         Assert(boundStatement->GetBoundNodeType() == BoundNodeType::boundCompoundStatement, "bound compound statement expected");
121         BoundCompoundStatement* compoundStatement = static_cast<BoundCompoundStatement*>(boundStatement);
122         boundFunction->SetBody(std::unique_ptr<BoundCompoundStatement>(compoundStatement));
123         boundCompileUnit.AddBoundNode(std::move(boundFunction));
124         if (fileScopeAdded)
125         {
126             boundCompileUnit.RemoveLastFileScope();
127         }
128         FunctionSymbol* result = functionSymbol.get();
129         boundCompileUnit.AddFunctionSymbol(std::move(functionSymbol));
130         boundCompileUnit.AddGlobalNs(std::move(globalNs));
131         inlineFunctionMap[inlineFunction] = result;
132         result->SetFunctionId(inlineFunction->FunctionId());
133         result->SetMaster(inlineFunction);
134         result->SetCopy();
135         return result;
136     }
137     else
138     {
139         ClassTypeSymbol* classTypeSymbol = static_cast<ClassTypeSymbol*>(inlineFunction->Parent());
140         symbolTable.SetCurrentClass(classTypeSymbol);
141         SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
142         symbolCreatorVisitor.SetLeaveFunction();
143         globalNs->Accept(symbolCreatorVisitor);
144         std::unique_ptr<FunctionSymbol> functionSymbol(symbolTable.GetCreatedFunctionSymbol());
145         functionSymbol->SetVmtIndex(inlineFunction->VmtIndex());
146         functionSymbol->SetImtIndex(inlineFunction->ImtIndex());
147         if (inlineFunction->IsDefault())
148         {
149             functionSymbol->SetHasArtificialBody();
150         }
151         functionSymbol->SetParent(classTypeSymbol);
152         functionSymbol->SetLinkOnceOdrLinkage();
153         if (inlineFunction->IsSystemDefault())
154         {
155             functionSymbol->SetSystemDefault();
156         }
157         TypeBinder typeBinder(boundCompileUnit);
158         typeBinder.SetContainerScope(functionSymbol->GetContainerScope());
159         functionInstanceNode->Accept(typeBinder);
160         StatementBinder statementBinder(boundCompileUnit);
161         std::unique_ptr<BoundClass> boundClass(new BoundClass(&boundCompileUnit.GetModule(), classTypeSymbol));
162         boundClass->SetInlineFunctionContainer();
163         statementBinder.SetCurrentClass(boundClass.get());
164         std::unique_ptr<BoundFunction> boundFunction(new BoundFunction(&boundCompileUnit, functionSymbol.get()));
165         statementBinder.SetCurrentFunction(boundFunction.get());
166         statementBinder.SetContainerScope(functionSymbol->GetContainerScope());
167         if (functionSymbol->GetSymbolType() == SymbolType::constructorSymbol)
168         {
169             ConstructorSymbol* constructorSymbol = static_cast<ConstructorSymbol*>(functionSymbol.get());
170             Node* node = symbolTable.GetNode(functionSymbol.get());
171             Assert(node->GetNodeType() == NodeType::constructorNode, "constructor node expected");
172             ConstructorNode* constructorNode = static_cast<ConstructorNode*>(node);
173             statementBinder.SetCurrentConstructor(constructorSymbol, constructorNode);
174         }
175         else if (functionSymbol->GetSymbolType() == SymbolType::destructorSymbol)
176         {
177             DestructorSymbol* destructorSymbol = static_cast<DestructorSymbol*>(functionSymbol.get());
178             Node* node = symbolTable.GetNode(functionSymbol.get());
179             Assert(node->GetNodeType() == NodeType::destructorNode, "destructor node expected");
180             DestructorNode* destructorNode = static_cast<DestructorNode*>(node);
181             statementBinder.SetCurrentDestructor(destructorSymbol, destructorNode);
182         }
183         else if (functionSymbol->GetSymbolType() == SymbolType::memberFunctionSymbol)
184         {
185             MemberFunctionSymbol* memberFunctionSymbol = static_cast<MemberFunctionSymbol*>(functionSymbol.get());
186             Node* node = symbolTable.GetNode(functionSymbol.get());
187             Assert(node->GetNodeType() == NodeType::memberFunctionNode, "member function node expected");
188             MemberFunctionNode* memberFunctionNode = static_cast<MemberFunctionNode*>(node);
189             statementBinder.SetCurrentMemberFunction(memberFunctionSymbol, memberFunctionNode);
190         }
191         functionInstanceNode->Body()->Accept(statementBinder);
192         BoundStatement* boundStatement = statementBinder.ReleaseStatement();
193         Assert(boundStatement->GetBoundNodeType() == BoundNodeType::boundCompoundStatement, "bound compound statement expected");
194         BoundCompoundStatement* compoundStatement = static_cast<BoundCompoundStatement*>(boundStatement);
195         boundFunction->SetBody(std::unique_ptr<BoundCompoundStatement>(compoundStatement));
196         boundClass->AddMember(std::move(boundFunction));
197         boundCompileUnit.AddBoundNode(std::move(boundClass));
198         FunctionSymbol* result = functionSymbol.get();
199         boundCompileUnit.AddGlobalNs(std::move(globalNs));
200         boundCompileUnit.AddFunctionSymbol(std::move(functionSymbol));
201         if (fileScopeAdded)
202         {
203             boundCompileUnit.RemoveLastFileScope();
204         }
205         inlineFunctionMap[inlineFunction] = result;
206         result->SetFunctionId(inlineFunction->FunctionId());
207         result->SetMaster(inlineFunction);
208         result->SetCopy();
209         return result;
210     }
211 }
