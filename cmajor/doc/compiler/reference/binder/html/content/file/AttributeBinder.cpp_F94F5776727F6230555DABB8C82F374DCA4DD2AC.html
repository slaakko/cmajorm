39 AttributeBinder::AttributeBinder(Module* module)
40 {
41     AttributeProcessor* jsonAttributeProcessor = new JsonAttributeProcessor(module);
42     attributeProcessors.push_back(std::unique_ptr<AttributeProcessor>(jsonAttributeProcessor));
43     JsonFieldNameAttributeProcessor* jsonFieldNameAttributeProcessor = new JsonFieldNameAttributeProcessor(module);
44     attributeProcessors.push_back(std::unique_ptr<AttributeProcessor>(jsonFieldNameAttributeProcessor));
45     for (const std::std::unique_ptr<AttributeProcessor>&attributeProcessor : attributeProcessors)
46     {
47         attributeProcessorMap[attributeProcessor->AttributeName()] = attributeProcessor.get();
48     }
49 }

51 void AttributeBinder::BindAttributes(Attributes* attrs, Symbol* symbol, BoundCompileUnit& boundCompileUnit, ContainerScope* containerScope)
52 {
53     if (!attrs) return;
54     const std::std::vector<std::std::unique_ptr<Attribute>>&attributes=attrs->GetAttributes();
55     for (const std::std::unique_ptr<Attribute>&attribute : attributes)
56     {
57         const std::u32string& attrName = attribute->Name();
58         auto it = attributeProcessorMap.find(attrName);
59         if (it != attributeProcessorMap.cend())
60         {
61             AttributeProcessor* processor = it->second;
62             processor->TypeCheck(attribute.get(), symbol);
63             processor->GenerateSymbols(attribute.get(), symbol, boundCompileUnit, containerScope);
64         }
65         else
66         {
67             throw Exception(&boundCompileUnit.GetModule(), "unknown attribute '" + ToUtf8(attrName) + "'", attribute->GetSpan());
68         }
69     }
70     symbol->SetAttributes(std::unique_ptr<Attributes>(attrs->Clone()));
71 }

73 void AttributeBinder::GenerateImplementation(Attributes* attrs, Symbol* symbol, StatementBinder* statementBinder)
74 {
75     if (!attrs) return;
76     const std::std::vector<std::std::unique_ptr<Attribute>>&attributes=attrs->GetAttributes();
77     for (const std::std::unique_ptr<Attribute>&attribute : attributes)
78     {
79         const std::u32string& attrName = attribute->Name();
80         auto it = attributeProcessorMap.find(attrName);
81         if (it != attributeProcessorMap.cend())
82         {
83             AttributeProcessor* processor = it->second;
84             processor->GenerateImplementation(attribute.get(), symbol, statementBinder);
85         }
86         else
87         {
88             throw Exception(&statementBinder->GetBoundCompileUnit().GetModule(), "unknown attribute '" + ToUtf8(attrName) + "'", attribute->GetSpan());
89         }
90     }
91 }
