  56 void LlvmCodeGenerator::Visit(BoundCompileUnit& boundCompileUnit)
  57 {
  58     boundCompileUnit.ResetCodeGenerated();
  59     symbolTable = &boundCompileUnit.GetSymbolTable();
  60     symbolsModule = &boundCompileUnit.GetModule();
  61     NativeModule nativeModule(emitter, boundCompileUnit.GetCompileUnitNode()->FilePath());
  62     module = nativeModule.module;
  63     emitter->SetTargetTriple(emittingContext->TargetTriple());
  64     emitter->SetDataLayout(emittingContext->DataLayout());
  65     emitter->SetSourceFileName(boundCompileUnit.GetCompileUnitNode()->FilePath());
  66     emitter->ResetCurrentDebugLocation();
  67     debugInfo = false;
  68     if (GetGlobalFlag(GlobalFlags::generateDebugInfo) && boundCompileUnit.GetCompileUnitNode() && !boundCompileUnit.GetCompileUnitNode()->IsSynthesizedUnit())
  69     {
  70         emitter->StartDebugInfo(boundCompileUnit.GetCompileUnitNode()->FilePath(), GetCompilerVersion(), GetGlobalFlag(GlobalFlags::release), boundCompileUnit.GetCompileUnitNode());
  71         debugInfo = true;
  72     }
  73     compileUnit = &boundCompileUnit;
  74     ConstantArrayRepository& constantArrayRepository = boundCompileUnit.GetConstantArrayRepository();
  75     for (ConstantSymbol* constantSymbol : constantArrayRepository.ConstantArrays())
  76     {
  77         constantSymbol->ArrayIrObject(*emitter, true);
  78     }
  79     ConstantStructureRepository& constantStructureRepository = boundCompileUnit.GetConstantStructureRepository();
  80     for (ConstantSymbol* constantSymbol : constantStructureRepository.ConstantStructures())
  81     {
  82         constantSymbol->StructureIrObject(*emitter, true);
  83     }
  84     int n = boundCompileUnit.BoundNodes().size();
  85     for (int i = 0; i < n; ++i)
  86     {
  87         BoundNode* boundNode = boundCompileUnit.BoundNodes()[i].get();
  88         boundNode->Accept(*this);
  89     }
  90     if (debugInfo)
  91     {
  92         emitter->FinalizeDebugInfo();
  93     }
  94     if (GetGlobalFlag(GlobalFlags::emitLlvm))
  95     {
  96         emitter->EmitIrText(boundCompileUnit.LLFilePath());
  97     }
  98     emitter->VerifyModule();
  99     if (GetGlobalFlag(GlobalFlags::release))
 100     {
 101         emitter->EmitIrFile(boundCompileUnit.BCFilePath());
 102         Optimize(GetOptimizationLevel(), boundCompileUnit.BCFilePath(), boundCompileUnit.OptBCFilePath());
 103         if (GetGlobalFlag(GlobalFlags::emitOptLlvm))
 104         {
 105             emitter->Disassemble(boundCompileUnit.OptBCFilePath(), boundCompileUnit.OptLLFilePath());
 106         }
 107         emitter->Compile(boundCompileUnit.OptBCFilePath(), boundCompileUnit.ObjectFilePath(), GetOptimizationLevel());
 108     }
 109     else
 110     {
 111         emitter->EmitObjectCodeFile(boundCompileUnit.ObjectFilePath());
 112     }
 113     if (debugInfo)
 114     {
 115         emitter->EndDebugInfo();
 116     }

 147         if (!boundNamespace.GetNamespaceNode().Id()->Str().empty())
 148         {
 149             for (int i = 0; i < numComponents; ++i)

 161 void LlvmCodeGenerator::Visit(BoundClass& boundClass)
 162 {
 163     classStack.push(currentClass);
 164     currentClass = &boundClass;
 165     bool prevDebugInfo = debugInfo;
 166     void* prevDIBuilder = emitter->DIBuilder();
 167     if (!boundClass.ContainsSourceFunctions())
 168     {
 169         emitter->ResetCurrentDebugLocation();
 170         debugInfo = false;
 171         emitter->SetCurrentDIBuilder(nullptr);
 172     }
 173     if (debugInfo)
 174     {
 175         emitter->MapClassPtr(currentClass->GetClassTypeSymbol()->TypeId(), currentClass->GetClassTypeSymbol(), ToUtf8(currentClass->GetClassTypeSymbol()->FullName()));
 176         void* diType = emitter->GetDITypeByTypeId(currentClass->GetClassTypeSymbol()->TypeId());
 177         if (diType)
 178         {
 179             emitter->PushScope(diType);
 180         }
 181         else
 182         {
 183             void* baseClassDIType = nullptr;
 184             if (currentClass->GetClassTypeSymbol()->BaseClass())
 185             {
 186                 baseClassDIType = currentClass->GetClassTypeSymbol()->BaseClass()->GetDIType(*emitter);
 187             }
 188             void* vtableHolderClassDIType = nullptr;
 189             if (currentClass->GetClassTypeSymbol()->IsPolymorphic() && currentClass->GetClassTypeSymbol()->VmtPtrHolderClass())
 190             {
 191                 vtableHolderClassDIType = currentClass->GetClassTypeSymbol()->VmtPtrHolderClass()->CreateDIForwardDeclaration(*emitter);
 192                 emitter->MapFwdDeclaration(vtableHolderClassDIType, currentClass->GetClassTypeSymbol()->VmtPtrHolderClass()->TypeId());
 193             }
 194             void* classIrType = currentClass->GetClassTypeSymbol()->IrType(*emitter);
 195             Span classSpan = currentClass->GetClassTypeSymbol()->GetSpan();
 196             if (currentClass->GetClassTypeSymbol()->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
 197             {
 198                 ClassTemplateSpecializationSymbol* specialization = static_cast<ClassTemplateSpecializationSymbol*>(currentClass->GetClassTypeSymbol());
 199                 classSpan = specialization->GetClassTemplate()->GetSpan();
 200             }
 201             void* forwardDeclaration = emitter->CreateIrDIForwardDeclaration(classIrType, ToUtf8(currentClass->GetClassTypeSymbol()->Name()), ToUtf8(currentClass->GetClassTypeSymbol()->MangledName()), 
 202                 classSpan);
 203             emitter->SetDITypeByTypeId(currentClass->GetClassTypeSymbol()->TypeId(), forwardDeclaration, ToUtf8(currentClass->GetClassTypeSymbol()->FullName()));
 204             std::vector<void*> memberVariableElements;
 205             for (MemberVariableSymbol* memberVariable : currentClass->GetClassTypeSymbol()->MemberVariables())
 206             {
 207                 int memberVariableLayoutIndex = memberVariable->LayoutIndex();
 208                 uint64_t offsetInBits = emitter->GetOffsetInBits(classIrType, memberVariableLayoutIndex);
 209                 memberVariableElements.push_back(memberVariable->GetDIMemberType(*emitter, offsetInBits));
 210             }
 211             void* clsDIType = emitter->CreateDITypeForClassType(classIrType, memberVariableElements, classSpan, ToUtf8(currentClass->GetClassTypeSymbol()->Name()), vtableHolderClassDIType, 
 212                 ToUtf8(currentClass->GetClassTypeSymbol()->MangledName()), baseClassDIType);
 213             emitter->MapFwdDeclaration(forwardDeclaration, currentClass->GetClassTypeSymbol()->TypeId());
 214             emitter->SetDITypeByTypeId(currentClass->GetClassTypeSymbol()->TypeId(), clsDIType, ToUtf8(currentClass->GetClassTypeSymbol()->FullName()));
 215             emitter->PushScope(clsDIType);
 216         }
 217     }
 218     int n = boundClass.Members().size();
 219     for (int i = 0; i < n; ++i)
 220     {
 221         BoundNode* boundNode = boundClass.Members()[i].get();
 222         boundNode->Accept(*this);
 223     }
 224     if (debugInfo)
 225     {
 226         emitter->PopScope();
 227     }
 228     currentClass = classStack.top();
 229     classStack.pop();
 230     debugInfo = prevDebugInfo;
 231     emitter->SetCurrentDIBuilder(prevDIBuilder);
 232 }

 234 void LlvmCodeGenerator::Visit(BoundEnumTypeDefinition& boundEnumTypeDefinition)

 244         {

 250                 {
 251                     ULongValue* ulongValue = static_cast<ULongValue*>(val);
 252                     value = static_cast<int64_t>(ulongValue->GetValue());
 253                 }
 254             }
 255             else
 256             {
 257                 Value* val = enumConstant->GetValue()->As(symbolTable->GetTypeByName(U"long"), false, enumTypeSymbol->GetSpan(), true);
 258                 if (val)
 259                 {
 260                     LongValue* longValue = static_cast<LongValue*>(val);
 261                     value = longValue->GetValue();
 262                 }
 263             }
 264             elements.push_back(emitter->CreateDITypeForEnumConstant(ToUtf8(enumConstant->Name()), value));
 265         }
 266         void* enumTypeDI = emitter->CreateDITypeForEnumType(ToUtf8(enumTypeSymbol->Name()), ToUtf8(enumTypeSymbol->MangledName()), enumTypeSymbol->GetSpan(), elements, 
 267             sizeInBits, alignInBits, enumTypeSymbol->UnderlyingType()->GetDIType(*emitter));
 268         emitter->SetDITypeByTypeId(enumTypeSymbol->TypeId(), enumTypeDI, ToUtf8(enumTypeSymbol->FullName()));
 269     }

 272 void LlvmCodeGenerator::Visit(BoundFunction& boundFunction)
 273 {
 274     if (!boundFunction.Body()) return;
 275     currentFunction = &boundFunction;
 276     handlerBlock = nullptr;
 277     cleanupBlock = nullptr;
 278     newCleanupNeeded = false;
 279     currentPad = nullptr;
 280     prevLineNumber = 0;
 281     destructorCallGenerated = false;
 282     lastInstructionWasRet = false;
 283     basicBlockOpen = false;
 284     lastAlloca = nullptr;
 285     compoundLevel = 0;
 286     cleanups.clear();
 287     pads.clear();
 288     labeledStatementMap.clear();
 289     FunctionSymbol* functionSymbol = boundFunction.GetFunctionSymbol();
 290     if (functionSymbol->MangledName() == U"destructor_Edge_int_E57D66B72F5D4567500CC822E3AB34B236FB95FB")
 291     {
 292         int x = 0;
 293     }
 294     if (functionSymbol->CodeGenerated()) return;
 295     functionSymbol->SetCodeGenerated();
 296     void* functionType = functionSymbol->IrType(*emitter);
 297     function = emitter->GetOrInsertFunction(ToUtf8(functionSymbol->MangledName()), functionType);
 298     bool setInline = false;
 299     if (GetGlobalFlag(GlobalFlags::release) && functionSymbol->IsInline())
 300     {
 301         emitter->AddInlineFunctionAttribute(function);
 302         functionSymbol->SetLinkOnceOdrLinkage();
 303     }
 304     if (functionSymbol->HasLinkOnceOdrLinkage())
 305     {
 306         setInline = true;
 307         void* comdat = emitter->GetOrInsertAnyFunctionComdat(ToUtf8(functionSymbol->MangledName()), function);
 308     }
 309     if (functionSymbol->GetFlag(FunctionSymbolFlags::winapi))
 310     {
 311         emitter->SetFunctionCallConventionToStdCall(function);
 312     }
 313     emitter->SetFunctionLinkage(function, setInline);
 314     emitter->SetFunction(function);
 315     bool hasSource = functionSymbol->HasSource();
 316     bool prevDebugInfo = debugInfo;
 317     void* prevDIBuilder = emitter->DIBuilder();
 318     if (!hasSource)
 319     {
 320         emitter->ResetCurrentDebugLocation();
 321         debugInfo = false;
 322         emitter->SetCurrentDIBuilder(nullptr);
 323     }
 324     if (debugInfo)
 325     {
 326         emitter->SetInPrologue(true);
 327         emitter->SetCurrentDebugLocation(Span());
 328         std::vector<void*> elementTypes;
 329         if (functionSymbol->ReturnType())
 330         {
 331             elementTypes.push_back(functionSymbol->ReturnType()->GetDIType(*emitter));
 332         }
 333         else
 334         {
 335             elementTypes.push_back(symbolTable->GetTypeByName(U"void")->GetDIType(*emitter));
 336         }
 337         for (ParameterSymbol* parameter : functionSymbol->Parameters())
 338         {
 339             TypeSymbol* parameterType = parameter->GetType();
 340             elementTypes.push_back(parameterType->GetDIType(*emitter));
 341         }
 342         void* subroutineType = emitter->CreateSubroutineType(elementTypes);
 343         void* subprogram = nullptr;
 344         if (functionSymbol->IsConstructorDestructorOrNonstaticMemberFunction())
 345         {
 346             unsigned virtuality = 0;
 347             if (functionSymbol->IsVirtualAbstractOrOverride())
 348             {
 349                 if (functionSymbol->IsAbstract())
 350                 {
 351                     virtuality = emitter->GetPureVirtualVirtuality();
 352                 }
 353                 else
 354                 {
 355                     virtuality = emitter->GetVirtualVirtuality();
 356                 }
 357             }
 358             void* vtableHolder = nullptr;
 359             Symbol* parent = functionSymbol->Parent();
 360             if (parent && parent->IsClassTypeSymbol())
 361             {
 362                 ClassTypeSymbol* cls = static_cast<ClassTypeSymbol*>(parent);
 363                 if (cls->IsPolymorphic() && cls->VmtPtrHolderClass())
 364                 {
 365                     vtableHolder = cls->VmtPtrHolderClass()->GetDIType(*emitter);
 366                 }
 367             }
 368             unsigned flags = emitter->GetFunctionFlags(functionSymbol->IsStatic(), AccessFlag(*emitter, functionSymbol->Access()), functionSymbol->IsExplicit());
 369             unsigned vtableIndex = 0;
 370             if (functionSymbol->VmtIndex() != -1)
 371             {
 372                 vtableIndex = functionSymbol->VmtIndex();
 373             }
 374             subprogram = emitter->CreateDIMethod(ToUtf8(functionSymbol->Name()), ToUtf8(functionSymbol->MangledName()), functionSymbol->GetSpan(), subroutineType, virtuality, vtableIndex, 
 375                 vtableHolder, flags);
 376         }
 377         else
 378         {
 379             unsigned flags = AccessFlag(*emitter, functionSymbol->Access());
 380             subprogram = emitter->CreateDIFunction(ToUtf8(functionSymbol->Name()), ToUtf8(functionSymbol->MangledName()), functionSymbol->GetSpan(), subroutineType, flags);
 381         }
 382         emitter->SetDISubprogram(function, subprogram);
 383         emitter->PushScope(subprogram);
 384     }
 385     void* entryBlock = emitter->CreateBasicBlock("entry");
 386     entryBasicBlock = entryBlock;
 387     emitter->SetCurrentBasicBlock(entryBlock);
 388     if (currentClass && !currentClass->IsInlineFunctionContainer())
 389     {
 390         ClassTypeSymbol* classTypeSymbol = currentClass->GetClassTypeSymbol();
 391         if (!emitter->IsVmtObjectCreated(classTypeSymbol))
 392         {
 393             classTypeSymbol->VmtObject(*emitter, true);
 394         }
 395         if (!emitter->IsStaticObjectCreated(classTypeSymbol))
 396         {
 397             classTypeSymbol->StaticObject(*emitter, true);
 398         }
 399     }
 400     int np = functionSymbol->Parameters().size();
 401     for (int i = 0; i < np; ++i)
 402     {
 403         ParameterSymbol* parameter = functionSymbol->Parameters()[i];
 404         void* allocaInst = emitter->CreateAlloca(parameter->GetType()->IrType(*emitter));
 405         emitter->SetIrObject(parameter, allocaInst);
 406         lastAlloca = allocaInst;
 407         if (debugInfo)
 408         {
 409             void* paramVar = emitter->CreateDIParameterVariable(ToUtf8(parameter->Name()), i + 1, parameter->GetSpan(), parameter->GetType()->GetDIType(*emitter), allocaInst);
 410         }
 411     }
 412     if (functionSymbol->ReturnParam())
 413     {
 414         ParameterSymbol* parameter = functionSymbol->ReturnParam();
 415         void* allocaInst = emitter->CreateAlloca(parameter->GetType()->IrType(*emitter));
 416         emitter->SetIrObject(parameter, allocaInst);
 417         lastAlloca = allocaInst;
 418     }
 419     int nlv = functionSymbol->LocalVariables().size();
 420     for (int i = 0; i < nlv; ++i)
 421     {
 422         LocalVariableSymbol* localVariable = functionSymbol->LocalVariables()[i];
 423         void* allocaInst = emitter->CreateAlloca(localVariable->GetType()->IrType(*emitter));
 424         emitter->SetIrObject(localVariable, allocaInst);
 425         lastAlloca = allocaInst;
 426         if (debugInfo && localVariable->GetSpan().Valid())
 427         {
 428             void* localVar = emitter->CreateDIAutoVariable(ToUtf8(localVariable->Name()), localVariable->GetSpan(), localVariable->GetType()->GetDIType(*emitter), allocaInst);
 429         }
 430     }
 431     if (!functionSymbol->DontThrow())
 432     {
 433         int funId = compileUnit->Install(ToUtf8(functionSymbol->FullName()));
 434         int sfpId = compileUnit->Install(compileUnit->SourceFilePath());
 435         void* funValue = GetGlobalStringPtr(funId);
 436         void* sfpValue = GetGlobalStringPtr(sfpId);
 437         std::vector<void*> enterFunctionParamTypes;
 438         enterFunctionParamTypes.push_back(emitter->GetIrTypeForVoidPtrType());
 439         enterFunctionParamTypes.push_back(emitter->GetIrTypeForVoidPtrType());
 440         void* enterFunctionType = emitter->GetIrTypeForFunction(emitter->GetIrTypeForVoid(), enterFunctionParamTypes);
 441         void* enterFunction = emitter->GetOrInsertFunction("RtEnterFunction", enterFunctionType);
 442         std::vector<void*> enterFunctionArgs;
 443         enterFunctionArgs.push_back(funValue);
 444         enterFunctionArgs.push_back(sfpValue);
 445         emitter->CreateCall(enterFunction, enterFunctionArgs);
 446     }
 447     for (int i = 0; i < np; ++i)
 448     {
 449         void* arg = emitter->GetFunctionArgument(function, i);
 450         ParameterSymbol* parameter = functionSymbol->Parameters()[i];
 451         if (parameter->GetType()->IsClassTypeSymbol())
 452         {
 453             ClassTypeSymbol* classType = static_cast<ClassTypeSymbol*>(parameter->GetType());
 454             FunctionSymbol* copyConstructor = classType->CopyConstructor();
 455             if (!copyConstructor)
 456             {
 457                 copyConstructor = compileUnit->GetCopyConstructorFor(classType->TypeId());
 458             }
 459             void* copyCtorType = copyConstructor->IrType(*emitter);
 460             void* callee = emitter->GetOrInsertFunction(ToUtf8(copyConstructor->MangledName()), copyCtorType);
 461             std::vector<void*> args;
 462             args.push_back(parameter->IrObject(*emitter));
 463             args.push_back(arg);
 464             if (debugInfo)
 465             {
 466                 emitter->SetInPrologue(false);
 467                 emitter->SetCurrentDebugLocation(boundFunction.Body()->GetSpan());
 468                 emitter->CreateCall(callee, args);
 469                 emitter->SetInPrologue(true);
 470                 emitter->SetCurrentDebugLocation(Span());
 471             }
 472             else
 473             {
