17     void Write(SymbolWriter& writer) override;

29     std::u32string Info() const override { return Name(); }
30     const char* ClassName() const override { return "TemplateParameterSymbol"; }
31 private:
32     bool hasDefault;
33     TypeSymbol* defaultType;
34 };
