 84 bool TypeSymbol::IsRecursive(TypeSymbol* type, std::std::unordered_set<boost::uuids::uuid, boost::boost::hash<boost::uuids::uuid>>&tested)
 85 {
 86     if (tested.find(TypeId()) != tested.cend()) return TypesEqual(type, this);
 87     tested.insert(TypeId());
 88     return TypesEqual(type, this);
 89 }

101 void* TypeSymbol::GetDIType(Emitter& emitter)
102 {
103     void* localDiType = emitter.GetDITypeByTypeId(TypeId());
104     if (!localDiType)
105     {
106         if (IsClassTypeSymbol())
107         {
108             ClassTypeSymbol* classTypeSymbol = static_cast<ClassTypeSymbol*>(this);
109             emitter.MapClassPtr(classTypeSymbol->TypeId(), classTypeSymbol);
110             localDiType = classTypeSymbol->CreateDIForwardDeclaration(emitter);
111             emitter.MapFwdDeclaration(localDiType, classTypeSymbol->TypeId());
112             emitter.SetDITypeByTypeId(classTypeSymbol->TypeId(), localDiType);
113         }
114         localDiType = CreateDIType(emitter);
115         emitter.SetDITypeByTypeId(TypeId(), localDiType);
116     }
117     return localDiType;
118 }

139 bool CompareTypesForEquality(const TypeSymbol* left, const TypeSymbol* right)
140 {
141     if (left->GetSymbolType() == SymbolType::derivedTypeSymbol && right->GetSymbolType() == SymbolType::derivedTypeSymbol)
142     {
143         const DerivedTypeSymbol* derivedLeft = static_cast<const DerivedTypeSymbol*>(left);
144         const DerivedTypeSymbol* derivedRight = static_cast<const DerivedTypeSymbol*>(right);
145         if (TypesEqual(derivedLeft->BaseType(), derivedRight->BaseType()) && derivedLeft->DerivationRec() == derivedRight->DerivationRec())
146         {
147             return true;
148         }
149     }
150     return false;
151 }
