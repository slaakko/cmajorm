 253 void Visit(std::std::vector<Module*>&finishReadOrder, Module*module, std::std::unordered_set<Module*>&visited, std::std::unordered_set<Module*>&tempVisit, 
 254     std::std::unordered_map<Module*, ModuleDependency*>&dependencyMap, constModule*rootModule)
 255 {
 256     if (tempVisit.find(module) == tempVisit.cend())
 257     {
 258         if (visited.find(module) == visited.cend())
 259         {
 260             tempVisit.insert(module);
 261             auto i = dependencyMap.find(module);
 262             if (i != dependencyMap.cend())
 263             {
 264                 ModuleDependency* dependency = i->second;
 265                 for (Module* dependentAssembly : dependency->ReferencedModules())
 266                 {
 267                     Visit(finishReadOrder, dependentAssembly, visited, tempVisit, dependencyMap, rootModule);
 268                 }
 269                 tempVisit.erase(module);
 270                 visited.insert(module);
 271                 finishReadOrder.push_back(module);
 272             }
 273             else
 274             {
 275                 throw std::runtime_error("module '" + ToUtf8(module->Name()) + "' not found in dependencies of module '" + ToUtf8(rootModule->Name()) + "'");
 276             }
 277         }
 278     }
 279     else
 280     {
 281         throw std::runtime_error("circular module dependency '" + ToUtf8(module->Name()) + "' detected in dependencies of module '" + ToUtf8(rootModule->Name()) + "'");
 282     }
 283 }

 285 std::std::vector<Module*>CreateFinishReadOrder(std::std::vector<Module*>&modules, std::std::unordered_map<Module*, ModuleDependency*>&dependencyMap, constModule*rootModule)
 286 {
 287     std::vector<Module*> finishReadOrder;
 288     std::unordered_set<Module*> visited;
 289     std::unordered_set<Module*> tempVisit;
 290     for (Module* module : modules)
 291     {
 292         if (visited.find(module) == visited.cend())
 293         {
 294             Visit(finishReadOrder, module, visited, tempVisit, dependencyMap, rootModule);
 295         }
 296     }
 297     return finishReadOrder;
 298 }

 300 void FinishReads(Module* rootModule, std::std::vector<Module*>&finishReadOrder, boolall)
 301 {
 302 #ifdef MODULE_READING_DEBUG
 303 
 304 
 305 #endif 
 306     int n = finishReadOrder.size() - 1;
 307     if (all)
 308     {
 309         n = finishReadOrder.size();
 310     }
 311     for (int i = 0; i < n; ++i)
 312     {
 313         Module* module = finishReadOrder[i];
 314         if (!module->HasSymbolTable())
 315         {
 316 #ifdef MODULE_READING_DEBUG
 317 
 318 #endif 
 319             module->CreateSymbolTable();
 320             std::vector<TypeOrConceptRequest> typeAndConceptRequests;
 321             std::vector<FunctionRequest> functionRequests;
 322             std::vector<ClassTemplateSpecializationSymbol*> classTemplateSpecializations;
 323             std::vector<FunctionSymbol*> conversions;
 324             std::vector<ArrayTypeSymbol*> arrayTypes;
 325             std::vector<DerivedTypeSymbol*> derivedTypes;
 326             SymbolReader reader(module->FilePathReadFrom());
 327             reader.SetModule(module);
 328             reader.SetArrayTypesTarget(&arrayTypes);
 329             reader.SetDerivedTypesTarget(&derivedTypes);
 330             reader.SetClassTemplateSpecializationTarget(&classTemplateSpecializations);
 331             reader.SetTypeAndConceptRequestTarget(&typeAndConceptRequests);
 332             reader.SetFunctionRequestTarget(&functionRequests);
 333             reader.SetConversionsTarget(&conversions);
 334             reader.GetBinaryReader().Skip(module->SymbolTablePos());
 335             reader.SetRootModule(rootModule);
 336             module->GetSymbolTable().Read(reader);
 337             for (Module* referencedModule : module->ReferencedModules())
 338             {
 339                 module->GetSymbolTable().Import(referencedModule->GetSymbolTable());
 340             }
 341             module->GetSymbolTable().FinishRead(arrayTypes, derivedTypes, classTemplateSpecializations, typeAndConceptRequests, functionRequests, conversions);
 342             module->SetImmutable();
 343 #ifdef MODULE_CHECKING
 344 
 345 #endif
 346             rootModule->GetSymbolTable().Import(module->GetSymbolTable());
 347         }
 348         else
 349         {
 350 #ifdef MODULE_READING_DEBUG
 351 
 352 #endif 
 353             rootModule->GetSymbolTable().Import(module->GetSymbolTable());
 354         }
 355     }
