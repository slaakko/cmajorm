  67 std::string DelegateTypeSymbol::Syntax() const
  68 {
  69     std::string syntax = GetSpecifierStr();
  70     if (!syntax.empty())
  71     {
  72         syntax.append(1, ' ');
  73     }
  74     syntax.append("delegate ");
  75     syntax.append(ToUtf8(ReturnType()->DocName()));
  76     syntax.append(1, ' ');
  77     syntax.append(ToUtf8(DocName()));
  78     syntax.append(1, '(');
  79     bool first = true;
  80     for (ParameterSymbol* param : parameters)
  81     {
  82         if (first)
  83         {
  84             first = false;
  85         }
  86         else
  87         {
  88             syntax.append(", ");
  89         }
  90         syntax.append(ToUtf8(param->GetType()->DocName()));
  91         syntax.append(1, ' ');
  92         syntax.append(ToUtf8(param->DocName()));
  93     }
  94     syntax.append(");");
  95     return syntax;
  96 }

 118 void* DelegateTypeSymbol::IrType(Emitter& emitter)
 119 {
 120     void* localIrType = emitter.GetIrTypeByTypeId(TypeId());
 121     if (!localIrType)
 122     {
 123         void* retType = emitter.GetIrTypeForVoid();
 124         if (!returnType->IsVoidType() && !ReturnsClassInterfaceOrClassDelegateByValue())
 125         {
 126             retType = returnType->IrType(emitter);
 127         }
 128         std::vector<void*> paramTypes;
 129         int np = parameters.size();
 130         for (int i = 0; i < np; ++i)
 131         {
 132             ParameterSymbol* parameter = parameters[i];
 133             paramTypes.push_back(parameter->GetType()->IrType(emitter));
 134         }
 135         if (returnParam)
 136         {
 137             paramTypes.push_back(returnParam->GetType()->IrType(emitter));
 138         }
 139         localIrType = emitter.GetIrTypeForDelegateType(retType, paramTypes);
 140         emitter.SetIrTypeByTypeId(TypeId(), localIrType);
 141     }
 142     return localIrType;
 143 }

 150 void DelegateTypeSymbol::SetSpecifiers(Specifiers specifiers)
 151 {
 152     Specifiers accessSpecifiers = specifiers & Specifiers::access_;
 153     SetAccess(accessSpecifiers);
 154     if ((specifiers & Specifiers::static_) != Specifiers::none)
 155     {
 156         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be static", GetSpan());
 157     }
 158     if ((specifiers & Specifiers::virtual_) != Specifiers::none)
 159     {
 160         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be virtual", GetSpan());
 161     }
 162     if ((specifiers & Specifiers::override_) != Specifiers::none)
 163     {
 164         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be override", GetSpan());
 165     }
 166     if ((specifiers & Specifiers::abstract_) != Specifiers::none)
 167     {
 168         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be abstract", GetSpan());
 169     }
 170     if ((specifiers & Specifiers::inline_) != Specifiers::none)
 171     {
 172         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be inline", GetSpan());
 173     }
 174     if ((specifiers & Specifiers::explicit_) != Specifiers::none)
 175     {
 176         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be explicit", GetSpan());
 177     }
 178     if ((specifiers & Specifiers::external_) != Specifiers::none)
 179     {
 180         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be external", GetSpan());
 181     }
 182     if ((specifiers & Specifiers::suppress_) != Specifiers::none)
 183     {
 184         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be suppressed", GetSpan());
 185     }
 186     if ((specifiers & Specifiers::default_) != Specifiers::none)
 187     {
 188         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be default", GetSpan());
 189     }
 190     if ((specifiers & Specifiers::constexpr_) != Specifiers::none)
 191     {
 192         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be constexpr", GetSpan());
 193     }
 194     if ((specifiers & Specifiers::cdecl_) != Specifiers::none)
 195     {
 196         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be cdecl", GetSpan());
 197     }
 198     if ((specifiers & Specifiers::nothrow_) != Specifiers::none)
 199     {
 200         SetNothrow();
 201     }
 202     if ((specifiers & Specifiers::throw_) != Specifiers::none)
 203     {
 204         if (IsNothrow())
 205         {
 206             throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be throw and nothrow at the same time", GetSpan());
 207         }
 208     }
 209     if ((specifiers & Specifiers::new_) != Specifiers::none)
 210     {
 211         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be new", GetSpan());
 212     }
 213     if ((specifiers & Specifiers::const_) != Specifiers::none)
 214     {
 215         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be const", GetSpan());
 216     }
 217     if ((specifiers & Specifiers::unit_test_) != Specifiers::none)
 218     {
 219         throw Exception(GetRootModuleForCurrentThread(), "delegate cannot be unit_test", GetSpan());
 220     }
 221 }

 233 void DelegateTypeSymbol::GenerateCall(Emitter& emitter, std::std::vector<GenObject*>&genObjects, OperationFlagsflags, constSpan&span)
 234 {
 235     void* callee = nullptr;
 236     int na = genObjects.size();
 237     for (int i = 0; i < na; ++i)
 238     {
 239         GenObject* genObject = genObjects[i];
 240         genObject->Load(emitter, flags & OperationFlags::functionCallFlags);
 241         if (i == 0)
 242         {
 243             callee = emitter.Stack().Pop();
 244         }
 245     }
 246     std::vector<void*> args;
 247     int n = parameters.size();
 248     if (ReturnsClassInterfaceOrClassDelegateByValue())
 249     {
 250         ++n;
 251     }
 252     args.resize(n);
 253     for (int i = 0; i < n; ++i)
 254     {
 255         void* arg = emitter.Stack().Pop();
 256         args[n - i - 1] = arg;
 257     }
 258     emitter.SetCurrentDebugLocation(span);
 259     void* handlerBlock = emitter.HandlerBlock();
 260     void* cleanupBlock = emitter.CleanupBlock();
 261     bool newCleanupNeeded = emitter.NewCleanupNeeded();
 262     Pad* currentPad = emitter.CurrentPad();
 263     std::vector<void*> bundles;
 264     if (currentPad != nullptr)
 265     {
 266         bundles.push_back(currentPad->value);
 267     }
 268     if (returnType->GetSymbolType() != SymbolType::voidTypeSymbol && !ReturnsClassInterfaceOrClassDelegateByValue())
 269     {
 270         if (IsNothrow() || (!handlerBlock && !cleanupBlock && !newCleanupNeeded))
 271         {
 272             if (currentPad == nullptr)
 273             {
 274                 emitter.Stack().Push(emitter.CreateCall(callee, args));
 275             }
 276             else
 277             {
 278                 void* callInst = emitter.CreateCallInst(callee, args, bundles, span);
 279                 emitter.Stack().Push(callInst);
 280             }
 281         }
 282         else
 283         {
 284             void* nextBlock = nullptr;
 285             if (GetBackEnd() == BackEnd::llvm)
 286             {
 287                 nextBlock = emitter.CreateBasicBlock("next");
 288             }
 289             if (newCleanupNeeded)
 290             {
 291                 emitter.CreateCleanup();
 292                 cleanupBlock = emitter.CleanupBlock();
 293             }
 294             void* unwindBlock = cleanupBlock;
 295             if (unwindBlock == nullptr)
 296             {
 297                 unwindBlock = handlerBlock;
 298                 Assert(unwindBlock, "no unwind block");
 299             }
 300             if (currentPad == nullptr)
 301             {
 302                 emitter.Stack().Push(emitter.CreateInvoke(callee, nextBlock, unwindBlock, args));
 303             }
 304             else
 305             {
 306                 void* invokeInst = emitter.CreateInvokeInst(callee, nextBlock, unwindBlock, args, bundles, span);
 307                 emitter.Stack().Push(invokeInst);
 308             }
 309             if (GetBackEnd() == BackEnd::llvm)
 310             {
 311                 emitter.SetCurrentBasicBlock(nextBlock);
 312             }
 313         }
 314     }
 315     else
 316     {
 317         if (IsNothrow() || (!handlerBlock && !cleanupBlock && !newCleanupNeeded))
 318         {
 319             if (currentPad == nullptr)
 320             {
 321                 emitter.CreateCall(callee, args);
 322             }
 323             else
 324             {
 325                 emitter.CreateCallInst(callee, args, bundles, span);
 326             }
 327         }
 328         else
 329         {
 330             void* nextBlock = nullptr;
 331             if (GetBackEnd() == BackEnd::llvm)
 332             {
 333                 nextBlock = emitter.CreateBasicBlock("next");
 334             }
 335             if (newCleanupNeeded)
 336             {
 337                 emitter.CreateCleanup();
 338                 cleanupBlock = emitter.CleanupBlock();
 339             }
 340             void* unwindBlock = cleanupBlock;
 341             if (unwindBlock == nullptr)
 342             {
 343                 unwindBlock = handlerBlock;
 344                 Assert(unwindBlock, "no unwind block");
 345             }
 346             if (currentPad == nullptr)
 347             {
 348                 emitter.CreateInvoke(callee, nextBlock, unwindBlock, args);
 349             }
 350             else
 351             {
 352                 emitter.CreateInvokeInst(callee, nextBlock, unwindBlock, args, bundles, span);
 353             }
 354             if (GetBackEnd() == BackEnd::llvm)
 355             {
 356                 emitter.SetCurrentBasicBlock(nextBlock);
 357             }
 358         }
 359     }
 360 }
