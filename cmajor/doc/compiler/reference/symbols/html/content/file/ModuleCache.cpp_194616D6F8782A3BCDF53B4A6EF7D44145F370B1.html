 28 class ModuleCache 
 29 {
 30 public:
 31     static void Init();
 32     static void Done();
 33     static ModuleCache& Instance() { return *instance; }
 34     Module* GetModule(const std::string& moduleFilePath);
 35     void PutModule(std::std::unique_ptr<Module>&&module);
 36 private:
 37     static std::unique_ptr<ModuleCache> instance;
 38     std::unordered_map<std::string, int> moduleMap;
 39     std::vector<std::std::unique_ptr<Module>>modules;
 40     void CollectModuleIndices(Module* module, std::std::unordered_set<int>&moduleIndeces);
 41     void ResetCacheEntries(Module* module);
 42 };

 46 void ModuleCache::Init()

 48     instance.reset(new ModuleCache());
 49 }

 51 void ModuleCache::Done()
 52 {
 53     instance.reset();
 54 }

 56 void ModuleCache::CollectModuleIndices(Module* module, std::std::unordered_set<int>&moduleIndices)
 57 {
 58     for (Module* referencedModule : module->ReferencedModules())
 59     {
 60         CollectModuleIndices(referencedModule, moduleIndices);
 61     }
 62     if (!module->IsRootModule())
 63     {
 64         int moduleIndex = module->Index();
 65         if (moduleIndex < 0 || moduleIndex >= modules.size())
 66         {
 67             throw std::runtime_error("module cache: invalid module index");
 68         }
 69         moduleIndices.insert(moduleIndex);
 70     }
 71 }
 72 
 73 void ModuleCache::ResetCacheEntries(Module* module)
 74 {
 75     std::unordered_set<int> moduleIndices;
 76     CollectModuleIndices(module, moduleIndices);
 77     for (int moduleIndex : moduleIndices)
 78     {
 79         modules[moduleIndex].reset();
 80     }
 81 }

 83 Module* ModuleCache::GetModule(const std::string& moduleFilePath)
 84 {
 85     auto it = moduleMap.find(moduleFilePath);
 86     if (useModuleCache && it != moduleMap.cend())
 87     {
 88         int moduleIndex = it->second;
 89         if (moduleIndex < 0 || moduleIndex >= modules.size())
 90         {
 91             throw std::runtime_error("module cache: invalid module index, module file path=" + moduleFilePath);
 92         }
 93         return modules[moduleIndex].get();
 94     }
 95     else
 96     {
 97         std::unique_ptr<Module> module(new Module());
 98         int moduleIndex = modules.size();
 99         module->SetIndex(moduleIndex);
100         Module* m = module.get();
101         modules.push_back(std::move(module));
102         moduleMap[moduleFilePath] = moduleIndex;
103         return m;
104     }
105 }

107 void ModuleCache::PutModule(std::std::unique_ptr<Module>&&module)
108 {
109     if (useModuleCache)
110     {
111         auto it = moduleMap.find(module->OriginalFilePath());
112         if (it != moduleMap.cend())
113         {
114             int moduleIndex = it->second;
115             if (moduleIndex < 0 || moduleIndex >= modules.size())
116             {
117                 throw std::runtime_error("module cache: invalid module index");
118             }
119             modules[moduleIndex].reset(new Module());
120         }
121         else
122         {
123             int moduleIndex = modules.size();
124             moduleMap[module->OriginalFilePath()] = moduleIndex;
125             std::unique_ptr<Module> m(new Module());
126             m->SetIndex(moduleIndex);
127             modules.push_back(std::move(m));
128         }
129     }
130     else
131     {
132         ResetCacheEntries(module.get());
133     }
134     module.reset();
135 }

152 Module* GetModuleFromModuleCache(const std::string& moduleFilePath)
153 {
154     std::lock_guard<std::recursive_mutex> lock(mtx);
