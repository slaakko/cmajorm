 36 void SymbolCreatorVisitor::Visit(NamespaceNode& namespaceNode)
 37 {
 38     symbolTable.BeginNamespace(namespaceNode);
 39     NodeList<Node>& members = namespaceNode.Members();
 40     int n = members.Count();
 41     for (int i = 0; i < n; ++i)
 42     {
 43         Node* member = members[i];
 44         member->Accept(*this);
 45     }
 46     symbolTable.EndNamespace();
 47 }

 49 void SymbolCreatorVisitor::Visit(FunctionNode& functionNode)
 50 {
 51     symbolTable.BeginFunction(functionNode, functionIndex++);
 52     int nt = functionNode.TemplateParameters().Count();
 53     for (int i = 0; i < nt; ++i)
 54     {
 55         symbolTable.AddTemplateParameter(*functionNode.TemplateParameters()[i]);
 56     }
 57     int n = functionNode.Parameters().Count();
 58     for (int i = 0; i < n; ++i)
 59     {
 60         ParameterNode* parameterNode = functionNode.Parameters()[i];
 61         parameterNode->Accept(*this);
 62     }
 63     if (nt == 0)
 64     {
 65         if (functionNode.Body())
 66         {
 67             functionNode.Body()->Accept(*this);
 68         }
 69     }
 70     symbolTable.EndFunction(!leaveFunction);
 71 }

 78 void SymbolCreatorVisitor::Visit(ClassNode& classNode)
 79 {
 80     if (&classNode == classInstanceNode)
 81     {
 82         symbolTable.BeginClassTemplateSpecialization(*classInstanceNode, classTemplateSpecialization);
 83     }
 84     else
 85     {
 86         symbolTable.BeginClass(classNode);
 87     }
 88     int nt = classNode.TemplateParameters().Count();
 89     for (int i = 0; i < nt; ++i)
 90     {
 91         symbolTable.AddTemplateParameter(*classNode.TemplateParameters()[i]);
 92     }
 93     if (nt == 0)
 94     {
 95         int n = classNode.Members().Count();
 96         for (int i = 0; i < n; ++i)
 97         {
 98             Node* member = classNode.Members()[i];
 99             member->Accept(*this);
100         }
101     }
102     if (&classNode == classInstanceNode)
103     {
104         symbolTable.EndClassTemplateSpecialization();
105     }
106     else
107     {
108         symbolTable.EndClass();
109     }
110 }

332 void SymbolCreatorVisitor::Visit(ConditionalCompilationDisjunctionNode& conditionalCompilationDisjunctionNode)
333 {
334     conditionalCompilationDisjunctionNode.Left()->Accept(*this);
335     bool left = conditionalCompilationStack.top();
336     conditionalCompilationStack.pop();
337     conditionalCompilationDisjunctionNode.Right()->Accept(*this);
338     bool right = conditionalCompilationStack.top();
339     conditionalCompilationStack.pop();
340     conditionalCompilationStack.push(left || right);
341 }

368 void SymbolCreatorVisitor::Visit(ConditionalCompilationStatementNode& conditionalCompilationStatementNode)
369 {
370     conditionalCompilationStatementNode.IfPart()->Accept(*this);
371     bool defined = conditionalCompilationStack.top();
372     conditionalCompilationStack.pop();
373     if (defined)
374     {
375         int n = conditionalCompilationStatementNode.IfPart()->Statements().Count();
376         for (int i = 0; i < n; ++i)
377         {
378             StatementNode* statement = conditionalCompilationStatementNode.IfPart()->Statements()[i];
379             statement->Accept(*this);
380         }
381     }
382     else
383     {
384         bool executed = false;
385         int n = conditionalCompilationStatementNode.ElifParts().Count();
386         for (int i = 0; i < n; ++i)
387         {
388             ConditionalCompilationPartNode* elifPart = conditionalCompilationStatementNode.ElifParts()[i];
389             elifPart->Accept(*this);
390             bool defined = conditionalCompilationStack.top();
391             conditionalCompilationStack.pop();
392             if (defined)
393             {
394                 int n = elifPart->Statements().Count();
395                 for (int i = 0; i < n; ++i)
396                 {
397                     StatementNode* statement = elifPart->Statements()[i];
398                     statement->Accept(*this);
399                 }
400                 executed = true;
401                 break;
402             }
403         }
404         if (!executed)
405         {
406             ConditionalCompilationPartNode* elsePart = conditionalCompilationStatementNode.ElsePart();
407             if (elsePart)
408             {
409                 int n = elsePart->Statements().Count();
410                 for (int i = 0; i < n; ++i)
411                 {
412                     StatementNode* statement = elsePart->Statements()[i];
413                     statement->Accept(*this);
414                 }
415             }
416         }
417     }
418 }
