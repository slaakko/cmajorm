 44 enum class ClassTypeSymbolFlags : uint16_t 
 45 {
 46     none=  0, 
 47     abstract_=  1 << 0, 
 48     polymorphic=  1 << 1, 
 49     vmtInitialized=  1 << 2, 
 50     imtsInitialized=  1 << 3, 
 51     layoutsComputed=  1 << 4, 
 52     statementsNotBound=  1 << 5, 
 53     recursiveComputed=  1 << 6, 
 54     recursive=  1 << 7, 
 55     vmtEmitted=  1 << 8
 56 };
 57 
 58 inline ClassTypeSymbolFlags operator|(ClassTypeSymbolFlags left, ClassTypeSymbolFlags right)
 59 {
 60     return ClassTypeSymbolFlags(uint16_t(left) | uint16_t(right));
 61 }
 62 
 63 inline ClassTypeSymbolFlags operator&(ClassTypeSymbolFlags left, ClassTypeSymbolFlags right)
 64 {
 65     return ClassTypeSymbolFlags(uint16_t(left) & uint16_t(right));
 66 }
 67 
 68 inline ClassTypeSymbolFlags operator~(ClassTypeSymbolFlags operand)
 69 {
 70     return ClassTypeSymbolFlags(~uint16_t(operand));
 71 }
