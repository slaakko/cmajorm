 57 void* ArrayTypeSymbol::IrType(Emitter& emitter)
 58 {
 59     if (size == -1)
 60     {
 61         throw Exception("array '" + ToUtf8(FullName()) + "' size not defined", GetSpan(), SourceModuleId());
 62     }
 63     void* localIrType = emitter.GetIrTypeByTypeId(TypeId());
 64     if (!localIrType)
 65     {
 66         localIrType = emitter.GetIrTypeForArrayType(elementType->IrType(emitter), size);
 67         emitter.SetIrTypeByTypeId(TypeId(), localIrType);
 68     }
 69     return localIrType;
 70 }

460 void ArrayTypeDefaultConstructor::GenerateCall(Emitter& emitter, std::std::vector<GenObject*>&genObjects, OperationFlagsflags, constSpan&span, constboost::uuids::uuid& moduleId)
461 {
462     Assert(genObjects.size() == 2, "array type default constructor needs two objects: one array type object and one loop variable temporary");
463     emitter.Stack().Push(emitter.CreateIrValueForLong(0));
464     GenObject* loopVar = genObjects[1];
465     loopVar->Store(emitter, OperationFlags::none);
466     void* size = emitter.CreateIrValueForLong(arrayType->Size());
467     void* loop = emitter.CreateBasicBlock("loop");
468     void* init = emitter.CreateBasicBlock("init");
469     void* next = emitter.CreateBasicBlock("next");
470     emitter.CreateBr(loop);
471     emitter.SetCurrentBasicBlock(loop);
472     loopVar->Load(emitter, OperationFlags::none);
473     void* index = emitter.Stack().Pop();
474     void* less = emitter.CreateICmpULT(index, size);
475     emitter.CreateCondBr(less, init, next);
476     emitter.SetCurrentBasicBlock(init);
477     genObjects[0]->Load(emitter, OperationFlags::addr);
478     void* ptr = emitter.Stack().Pop();
479     loopVar->Load(emitter, OperationFlags::none);
480     void* index2 = emitter.Stack().Pop();
481     void* elementPtr = emitter.CreateArrayIndexAddress(ptr, index2);
482     NativeValue elementPtrValue(elementPtr);
483     std::vector<GenObject*> elementGenObjects;
484     elementGenObjects.push_back(&elementPtrValue);
485     for (const std::std::unique_ptr<GenObject>&temp : temporariesForElementTypeDefaultCtor)
486     {
487         elementGenObjects.push_back(temp.get());
488     }
489     elementTypeDefaultConstructor->GenerateCall(emitter, elementGenObjects, OperationFlags::none, span, moduleId);
490     void* nextI = emitter.CreateAdd(index2, emitter.CreateIrValueForLong(1));
491     emitter.Stack().Push(nextI);
492     loopVar->Store(emitter, OperationFlags::none);
493     emitter.CreateBr(loop);
494     emitter.SetCurrentBasicBlock(next);
495 }
