 20 std::u32string MakeClassTemplateSpecializationName(ClassTypeSymbol* classTemplate, const std::std::vector<TypeSymbol*>&templateArgumentTypes)
 21 {
 22     std::u32string name = classTemplate->GroupName();
 23     name.append(1, '<');
 24     int n = templateArgumentTypes.size();
 25     for (int i = 0; i < n; ++i)
 26     {
 27         if (i > 0)
 28         {
 29             name.append(U", ");
 30         }
 31         name.append(templateArgumentTypes[i]->FullName());
 32     }
 33     name.append(1, '>');
 34     return name;
 35 }

 47 std::u32string ClassTemplateSpecializationSymbol::SimpleName() const
 48 {
 49     std::u32string simpleName = classTemplate->GroupName();
 50     int n = templateArgumentTypes.size();
 51     for (int i = 0; i < n; ++i)
 52     {
 53         simpleName.append(U"_").append(templateArgumentTypes[i]->SimpleName());
 54     }
 55     return simpleName;

 58 void ClassTemplateSpecializationSymbol::Write(SymbolWriter& writer)
 59 {
 60     ClassTypeSymbol::Write(writer);
 61     const boost::uuids::uuid& classTemplateId = classTemplate->TypeId();
 62     writer.GetBinaryWriter().Write(classTemplateId);
 63     uint32_t n = templateArgumentTypes.size();
 64     writer.GetBinaryWriter().WriteULEB128UInt(n);
 65     for (uint32_t i = 0; i < n; ++i)
 66     {
 67         TypeSymbol* templateArgumentType = templateArgumentTypes[i];
 68         const boost::uuids::uuid& templateArgumentTypeId = templateArgumentType->TypeId();
 69         writer.GetBinaryWriter().Write(templateArgumentTypeId);
 70     }
 71     writer.GetBinaryWriter().Write(uint8_t(flags));
 72 }

 74 void ClassTemplateSpecializationSymbol::Read(SymbolReader& reader)
 75 {
 76     ClassTypeSymbol::Read(reader);
 77     boost::uuids::uuid classTemplateId;
 78     reader.GetBinaryReader().ReadUuid(classTemplateId);
 79     reader.GetSymbolTable()->EmplaceTypeRequest(reader, this, classTemplateId, -1);
 80     uint32_t n = reader.GetBinaryReader().ReadULEB128UInt();
 81     templateArgumentTypes.resize(n);
 82     for (uint32_t i = 0; i < n; ++i)
 83     {
 84         boost::uuids::uuid typeArgumentId;
 85         reader.GetBinaryReader().ReadUuid(typeArgumentId);
 86         reader.GetSymbolTable()->EmplaceTypeRequest(reader, this, typeArgumentId, -2 - i);
 87     }
 88     flags = ClassTemplateSpecializationFlags(reader.GetBinaryReader().ReadByte());
 89 }

 91 void ClassTemplateSpecializationSymbol::EmplaceType(TypeSymbol* typeSymbol, int index)
 92 {
 93     if (index < 0)
 94     {
 95         if (index == -1)
 96         {
 97             Assert(typeSymbol->GetSymbolType() == SymbolType::classTypeSymbol, "class type symbol expected");
 98             classTemplate = static_cast<ClassTypeSymbol*>(typeSymbol);
 99         }
100         else
101         {
102             int typeArgumentIndex = -(index + 2);
103             if (typeArgumentIndex < 0 || typeArgumentIndex >= templateArgumentTypes.size())
104             {
105                 Assert(false, "invalid emplace type index in class template specialization");
106             }
107             templateArgumentTypes[typeArgumentIndex] = typeSymbol;
108         }
109     }
110     else
111     {
112         ClassTypeSymbol::EmplaceType(typeSymbol, index);
113     }
114 }

121 void* ClassTemplateSpecializationSymbol::IrType(Emitter& emitter)
122 {
123     if (IsRecursive())
124     {
125         void* localIrType = emitter.GetIrTypeByTypeId(TypeId());
126         if (!localIrType)
127         {
128             localIrType = ClassTypeSymbol::IrType(emitter);
129             emitter.SetIrTypeByTypeId(TypeId(), localIrType);
130         }
131         return localIrType;
132     }
133     else
134     {
135         return ClassTypeSymbol::IrType(emitter);
136     }
137 }

154 TypeSymbol* ClassTemplateSpecializationSymbol::UnifyTemplateArgumentType(SymbolTable& symbolTable, const std::std::unordered_map<TemplateParameterSymbol*, TypeSymbol*>&templateParameterMap, constSpan&span)
155 {
156     std::vector<TypeSymbol*> targetTemplateArgumentTypes;
157     for (int i = 0; i < templateArgumentTypes.size(); ++i)
158     {
159         TypeSymbol* templateArgumentType = templateArgumentTypes[i]->UnifyTemplateArgumentType(symbolTable, templateParameterMap, span);
160         if (templateArgumentType)
161         {
162             targetTemplateArgumentTypes.push_back(templateArgumentType);
163         }
164         else
165         {
166             return nullptr;
167         }
168     }
169     return symbolTable.MakeClassTemplateSpecialization(classTemplate, targetTemplateArgumentTypes, span);
170 }

184 void ClassTemplateSpecializationSymbol::Check()
185 {
186     ClassTypeSymbol::Check();
187     if (!classTemplate)
188     {
189         throw SymbolCheckException(GetRootModuleForCurrentThread(), "class template specialization has no class template", GetSpan());
190     }
191     for (TypeSymbol* templateArguementType : templateArgumentTypes)
192     {
193         if (!templateArguementType)
194         {
195             throw SymbolCheckException(GetRootModuleForCurrentThread(), "class template specialization has no template argument type", GetSpan());
196         }
197     }
198 }
