 433 template<typename BinOp>
 434 void BasicTypeBinaryOperation<BinOp>::GenerateCall(Emitter& emitter, std::std::vector<GenObject*>&genObjects, OperationFlagsflags, constSpan&span)
 435 {
 436     Assert(genObjects.size() == 2, "binary operation needs two objects");
 437     genObjects[0]->Load(emitter, OperationFlags::none);
 438     void* left = emitter.Stack().Pop();
 439     genObjects[1]->Load(emitter, OperationFlags::none);
 440     void* right = emitter.Stack().Pop();

1028 BasicTypeComparisonOperation<ComparisonOp>::BasicTypeComparisonOperation(SymbolType symbolType, TypeSymbol* type, TypeSymbol* boolType) : FunctionSymbol(symbolType, Span(), ComparisonOp::GroupName())
1029 {
1030     SetGroupName(ComparisonOp::GroupName());
1031     SetAccess(SymbolAccess::public_);
1032     ParameterSymbol* leftParam = new ParameterSymbol(Span(), U"left");
1033     leftParam->SetType(type);
1034     AddMember(leftParam);
1035     ParameterSymbol* rightParam = new ParameterSymbol(Span(), U"right");
1036     rightParam->SetType(type);
1037     AddMember(rightParam);
1038     SetReturnType(boolType);
1039     ComputeName();
1040     if (ComparisonOp::IsIntegerOpEqual())
1041     {
1042         if (type->IsBasicTypeSymbol())
1043         {
1044             BasicTypeSymbol* basicTypeSymbol = static_cast<BasicTypeSymbol*>(type);
1045             basicTypeSymbol->SetEqualityOp(this);
1046         }
1047     }
1048     else if (ComparisonOp::IsFloatingOpEqual())
1049     {
1050         BasicTypeSymbol* basicTypeSymbol = static_cast<BasicTypeSymbol*>(type);
1051         basicTypeSymbol->SetEqualityOp(this);
1052     }
1053 }

1055 template<typename ComparisonOp>
1056 void BasicTypeComparisonOperation<ComparisonOp>::GenerateCall(Emitter& emitter, std::std::vector<GenObject*>&genObjects, OperationFlagsflags, constSpan&span)
1057 {
1058     Assert(genObjects.size() == 2, "comparison operation needs two objects");
1059     genObjects[0]->Load(emitter, OperationFlags::none);
1060     void* left = emitter.Stack().Pop();
1061     genObjects[1]->Load(emitter, OperationFlags::none);
1062     void* right = emitter.Stack().Pop();
1063     emitter.Stack().Push(ComparisonOp::Generate(emitter, left, right));
1064 }
