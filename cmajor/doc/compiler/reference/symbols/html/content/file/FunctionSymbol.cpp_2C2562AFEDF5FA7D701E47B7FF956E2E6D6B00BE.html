  25 bool FunctionSymbolsEqual::operator()(FunctionSymbol* left, FunctionSymbol* right) const
  26 {
  27     if (left->Parent()->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol && right->Parent()->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
  28     {
  29         ClassTemplateSpecializationSymbol* leftSpecialization = static_cast<ClassTemplateSpecializationSymbol*>(left->Parent());
  30         ClassTemplateSpecializationSymbol* rightSpecialization = static_cast<ClassTemplateSpecializationSymbol*>(right->Parent());
  31         if (TypesEqual(leftSpecialization, rightSpecialization))
  32         {
  33             int leftIndex = left->GetIndex();
  34             int rightIndex = right->GetIndex();
  35             if (leftIndex == -1 || rightIndex == -1)
  36             {
  37                 return left == right;
  38             }
  39             else
  40             {
  41                 return leftIndex == rightIndex;
  42             }
  43         }
  44     }
  45     else if (left->IsTemplateSpecialization() && right->IsTemplateSpecialization())
  46     {
  47         if (left->FunctionTemplate() == right->FunctionTemplate())
  48         {
  49             int n = left->TemplateArgumentTypes().size();
  50             if (right->TemplateArgumentTypes().size() == n)
  51             {
  52                 bool equal = true;
  53                 for (int i = 0; i < n; ++i)
  54                 {
  55                     if (!TypesEqual(left->TemplateArgumentTypes()[i], right->TemplateArgumentTypes()[i]))
  56                     {
  57                         equal = false;
  58                         break;
  59                     }
  60                 }
  61                 return equal;
  62             }
  63         }
  64     }
  65     else if (GetGlobalFlag(GlobalFlags::release) && !left->IsGeneratedFunction() && left->IsInline() && !right->IsGeneratedFunction() && right->IsInline())
  66     {
  67         return left->FunctionId() == right->FunctionId();
  68     }
  69     return left == right;
  70 }

  72 size_t FunctionSymbolHash::operator()(FunctionSymbol* fun) const
  73 {
  74     if (fun->Parent()->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
  75     {
  76         ClassTemplateSpecializationSymbol* specialization = static_cast<ClassTemplateSpecializationSymbol*>(fun->Parent());
  77         return boost::hash<boost::uuids::uuid>()(specialization->TypeId()) ^ fun->GetIndex();
  78     }
  79     else if (fun->IsTemplateSpecialization())
  80     {
  81         size_t hashValue = boost::hash<boost::uuids::uuid>()(fun->FunctionTemplate()->FunctionId());
  82         int n = fun->TemplateArgumentTypes().size();
  83         for (int i = 0; i < n; ++i)
  84         {
  85             hashValue ^= boost::hash<boost::uuids::uuid>()(fun->TemplateArgumentTypes()[i]->TypeId());
  86         }
  87         return hashValue;
  88     }
  89     else if (GetGlobalFlag(GlobalFlags::release) && !fun->IsGeneratedFunction() && fun->IsInline())
  90     {
  91         return boost::hash<boost::uuids::uuid>()(fun->FunctionId());
  92     }
  93     return std::hash<FunctionSymbol*>()(fun);

 391 {

 402     }

 405     {
 406         usingNodes.Write(writer.GetAstWriter());
 407         Node* node = GetRootModuleForCurrentThread()->GetSymbolTable().GetNode(this);
 408         writer.GetAstWriter().Write(node);
 409     }
 410     boost::uuids::uuid returnTypeId = boost::uuids::nil_generator()();

 413         returnTypeId = returnType->TypeId();
 414     }
 415     writer.GetBinaryWriter().Write(returnTypeId);
 416     writer.GetBinaryWriter().Write(vmtIndex);
 417     writer.GetBinaryWriter().Write(imtIndex);
 418     bool hasReturnParam = returnParam != nullptr;
 419     writer.GetBinaryWriter().Write(hasReturnParam);
 420     if (hasReturnParam)
 421     {
 422         writer.Write(returnParam.get());

 426         writer.GetBinaryWriter().Write(true);
 427         writer.GetAstWriter().Write(constraint.get());
 428     }

 431         writer.GetBinaryWriter().Write(false);
 432     }

 439     }

 442 void FunctionSymbol::Read(SymbolReader& reader)
 443 {
 444     ContainerSymbol::Read(reader);
 445     reader.GetBinaryReader().ReadUuid(functionId);
 446     index = reader.GetBinaryReader().ReadInt();
 447     reader.GetSymbolTable()->AddFunctionSymbolToFunctionIdMap(this);
 448     groupName = reader.GetBinaryReader().ReadUtf32String();
 449     flags = static_cast<FunctionSymbolFlags>(reader.GetBinaryReader().ReadUInt());
 450     boost::uuids::uuid functionTemplateId;
 451     reader.GetBinaryReader().ReadUuid(functionTemplateId);
 452     if (!functionTemplateId.is_nil())
 453     {

 455     }
 456     if (IsFunctionTemplate() || (GetGlobalFlag(GlobalFlags::release) && IsInline()) || IsConstExpr())
 457     {
 458         usingNodes.Read(reader.GetAstReader());
 459         Node* node = reader.GetAstReader().ReadNode();
 460         FunctionNode* funNode = static_cast<FunctionNode*>(node);
 461         functionNode.reset(funNode);
 462     }
 463     boost::uuids::uuid returnTypeId;
 464     reader.GetBinaryReader().ReadUuid(returnTypeId);
 465     if (!returnTypeId.is_nil())
 466     {
 467         reader.GetSymbolTable()->EmplaceTypeRequest(reader, this, returnTypeId, 0);
 468     }
 469     vmtIndex = reader.GetBinaryReader().ReadInt();
 470     imtIndex = reader.GetBinaryReader().ReadInt();
 471     bool hasReturnParam = reader.GetBinaryReader().ReadBool();

 473     {
 474         returnParam.reset(reader.ReadParameterSymbol(this));
 475     }
 476     bool hasConstraint = reader.GetBinaryReader().ReadBool();
 477     if (hasConstraint)
 478     {
 479         constraint.reset(reader.GetAstReader().ReadConstraintNode());
 480     }
 481     if (IsConversion())
 482     {
 483         reader.AddConversion(this);
 484     }
 485     uint32_t n = reader.GetBinaryReader().ReadULEB128UInt();
 486     templateArgumentTypes.resize(n);
 487     for (uint32_t i = 0; i < n; ++i)
 488     {
 489         boost::uuids::uuid templateArgumentId;
 490         reader.GetBinaryReader().ReadUuid(templateArgumentId);
 491         reader.GetSymbolTable()->EmplaceTypeRequest(reader, this, templateArgumentId, -1 - i);
 492     }
 493 }
