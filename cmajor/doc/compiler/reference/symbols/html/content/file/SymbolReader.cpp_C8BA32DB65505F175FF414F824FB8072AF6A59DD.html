 20 Symbol* SymbolReader::ReadSymbol(Symbol* parent)
 21 {
 22     SymbolType symbolType = static_cast<SymbolType>(astReader.GetBinaryReader().ReadByte());
 23     Span span = astReader.ReadSpan();
 24     std::u32string name = astReader.GetBinaryReader().ReadUtf32String();
 25     Symbol* symbol = SymbolFactory::Instance().CreateSymbol(symbolType, span, name);
 26     symbol->SetModule(module);
 27     symbol->SetParent(parent);
 28     symbol->Read(*this);
 29     return symbol;
 30 }

 32 ArrayTypeSymbol* SymbolReader::ReadArrayTypeSymbol(Symbol* parent)
 33 {
 34     Symbol* symbol = ReadSymbol(parent);
 35     if (symbol->GetSymbolType() == SymbolType::arrayTypeSymbol)
 36     {
 37         return static_cast<ArrayTypeSymbol*>(symbol);
 38     }
 39     else
 40     {
 41         throw std::runtime_error("internal error: array type symbol expected");
 42     }
 43 }
