117     {
118         throw Exception(GetRootModuleForCurrentThread(), "enumerated type cannot be external", GetSpan());

180 void* EnumTypeSymbol::CreateDIType(Emitter& emitter)
181 {
182     uint64_t sizeInBits = SizeInBits(emitter);
183     uint32_t alignInBits = AlignmentInBits(emitter);
184     std::vector<void*> elements;
185     std::vector<EnumConstantSymbol*> enumConstants = GetEnumConstants();
186     for (EnumConstantSymbol* enumConstant : enumConstants)
187     {
188         int64_t value = 0;
189         if (underlyingType->IsUnsignedType())
190         {
191             Value* val = enumConstant->GetValue()->As(GetRootModuleForCurrentThread()->GetSymbolTable().GetTypeByName(U"ulong"), false, GetSpan(), true);
192             if (val)
193             {
194                 ULongValue* ulongValue = static_cast<ULongValue*>(val);
195                 value = static_cast<int64_t>(ulongValue->GetValue());
196             }
197         }
198         else
199         {
200             Value* val = enumConstant->GetValue()->As(GetRootModuleForCurrentThread()->GetSymbolTable().GetTypeByName(U"long"), false, GetSpan(), true);
201             if (val)
202             {
203                 LongValue* longValue = static_cast<LongValue*>(val);
204                 value = longValue->GetValue();
205             }
206         }
207         elements.push_back(emitter.CreateDITypeForEnumConstant(ToUtf8(enumConstant->Name()), value));
208     }
209     return emitter.CreateDITypeForEnumType(ToUtf8(Name()), ToUtf8(MangledName()), GetSpan(), elements, sizeInBits, alignInBits, underlyingType->GetDIType(emitter));
210 }
