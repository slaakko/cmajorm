 42 std::string Expand(Module* module, const std::string& errorMessage, const Span& span, const std::std::vector<Span>&references, conststd::string&title)
 43 {
 44     std::vector<Span> referenceSpans = references;
 45     referenceSpans.erase(std::unique(referenceSpans.begin(), referenceSpans.end()), referenceSpans.end());
 46     std::string expandedMessage = title + ": " + errorMessage;
 47     if (span.Valid())
 48     {
 49         if (!module)
 50         {
 51             throw std::runtime_error("module not set");
 52         }
 53         std::string fileName = module->GetFilePath(span.fileIndex);
 54         if (!fileName.empty())
 55         {
 56             expandedMessage.append(" (file '" + fileName + "', line " + std::to_string(span.line) + ")");
 57             expandedMessage.append(":\n").append(ToUtf8(module->GetErrorLines(span)));
 58         }
 59     }
 60     for (const Span& referenceSpan : referenceSpans)
 61     {
 62         if (!referenceSpan.Valid()) continue;
 63         if (referenceSpan == span) continue;
 64         if (!module)
 65         {
 66             throw std::runtime_error("module not set");
 67         }
 68         std::string fileName = module->GetFilePath(referenceSpan.fileIndex);
 69         if (!fileName.empty())
 70         {
 71             expandedMessage.append("\nsee reference to file '" + fileName + "', line " + std::to_string(referenceSpan.line));
 72             expandedMessage.append(":\n").append(ToUtf8(module->GetErrorLines(referenceSpan)));
 73         }
 74     }
 75     return expandedMessage;
 76 }

 78 std::std::unique_ptr<JsonObject>SpanToJson(Module*module, constSpan&span)
 79 {
 80     if (!span.Valid()) return std::unique_ptr<JsonObject>();
 81     if (!module)
 82     {
 83         throw std::runtime_error("module not set");
 84     }
 85     const std::string& fileName = module->GetFilePath(span.fileIndex);
 86     if (fileName.empty()) return std::unique_ptr<JsonObject>();
 87     std::unique_ptr<JsonObject> json(new JsonObject());
 88     json->AddField(U"file", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(fileName))));
 89     json->AddField(U"line", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(std::to_string(span.line)))));
 90     std::u32string text = module->GetErrorLines(span);
 91     int32_t startCol = 0;
 92     int32_t endCol = 0;
 93     module->GetColumns(span, startCol, endCol);
 94     json->AddField(U"startCol", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(std::to_string(startCol)))));
 95     json->AddField(U"endCol", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(std::to_string(endCol)))));
 96     json->AddField(U"text", std::unique_ptr<JsonValue>(new JsonString(text)));
 97     return json;
 98 }

100 std::std::unique_ptr<sngxml::dom::Element>SpanToDomElement(Module*module, constSpan&span)
101 {
102     if (!span.Valid()) return std::unique_ptr<sngxml::dom::Element>();
103     if (!module)
104     {
105         throw std::runtime_error("module not set");
106     }
107     const std::string& fileName = module->GetFilePath(span.fileIndex);
108     if (fileName.empty()) return std::unique_ptr<sngxml::dom::Element>();
109     std::unique_ptr<sngxml::dom::Element> spanElement(new sngxml::dom::Element(U"span"));
110     std::unique_ptr<sngxml::dom::Element> fileElement(new sngxml::dom::Element(U"file"));
111     std::unique_ptr<sngxml::dom::Text> fileText(new sngxml::dom::Text(ToUtf32(fileName)));
112     fileElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(fileText.release()));
113     std::unique_ptr<sngxml::dom::Element> lineElement(new sngxml::dom::Element(U"line"));
114     std::unique_ptr<sngxml::dom::Text> lineText(new sngxml::dom::Text(ToUtf32(std::to_string(span.line))));
115     lineElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(lineText.release()));
116     std::u32string text = module->GetErrorLines(span);
117     int32_t startCol = 0;
118     int32_t endCol = 0;
119     module->GetColumns(span, startCol, endCol);
120     spanElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(fileElement.release()));
121     spanElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(lineElement.release()));
122     std::unique_ptr<sngxml::dom::Element> startColElement(new sngxml::dom::Element(U"startCol"));
123     std::unique_ptr<sngxml::dom::Text> startColText(new sngxml::dom::Text(ToUtf32(std::to_string(startCol))));
124     startColElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(startColText.release()));
125     spanElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(startColElement.release()));
126     std::unique_ptr<sngxml::dom::Element> endColElement(new sngxml::dom::Element(U"endCol"));
127     std::unique_ptr<sngxml::dom::Text> endColText(new sngxml::dom::Text(ToUtf32(std::to_string(endCol))));
128     endColElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(endColText.release()));
129     spanElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(endColElement.release()));
130     std::unique_ptr<sngxml::dom::Element> textElement(new sngxml::dom::Element(U"text"));
131     std::unique_ptr<sngxml::dom::Text> textText(new sngxml::dom::Text(text));
132     textElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(textText.release()));
133     spanElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(textElement.release()));
134     return spanElement;
135 }

155 std::std::unique_ptr<JsonValue>Exception::ToJson() const
156 {
157     std::unique_ptr<JsonObject> json(new JsonObject());
158     json->AddField(U"tool", std::unique_ptr<JsonValue>(new JsonString(module->GetCurrentToolName())));
159     json->AddField(U"kind", std::unique_ptr<JsonValue>(new JsonString(U"error")));
160     json->AddField(U"project", std::unique_ptr<JsonValue>(new JsonString(module->GetCurrentProjectName())));
161     json->AddField(U"message", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(message))));
162     std::unique_ptr<JsonArray> refs(new JsonArray());
163     std::unique_ptr<JsonObject> ref = SpanToJson(module, defined);
164     if (ref)
165     {
166         refs->AddItem(std::move(ref));
167     }
168     std::vector<Span> referenceSpans = references;
169     referenceSpans.erase(std::unique(referenceSpans.begin(), referenceSpans.end()), referenceSpans.end());
170     for (const Span& referenceSpan : referenceSpans)
171     {
172         if (!referenceSpan.Valid()) continue;
173         if (referenceSpan == defined) continue;
174         std::unique_ptr<JsonObject> ref = SpanToJson(module, referenceSpan);
175         if (ref)
176         {
177             refs->AddItem(std::move(ref));
178         }
179     }
180     json->AddField(U"references", std::move(refs));
181     return std::unique_ptr<JsonValue>(json.release());
182 }

184 void Exception::AddToDiagnosticsElement(sngxml::dom::Element* diagnosticsElement) const
185 {
186     std::unique_ptr<sngxml::dom::Element> diagnosticElement(new sngxml::dom::Element(U"diagnostic"));
187     std::unique_ptr<sngxml::dom::Element> categoryElement(new sngxml::dom::Element(U"category"));
188     std::unique_ptr<sngxml::dom::Text> categoryText(new sngxml::dom::Text(U"error"));
189     categoryElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(categoryText.release()));
190     std::unique_ptr<sngxml::dom::Element> subcategoryElement(new sngxml::dom::Element(U"subcategory"));
191     std::unique_ptr<sngxml::dom::Text> subcategoryText(new sngxml::dom::Text(U"error"));
192     subcategoryElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(subcategoryText.release()));
193     std::unique_ptr<sngxml::dom::Element> messageElement(new sngxml::dom::Element(U"message"));
194     std::unique_ptr<sngxml::dom::Text> messageText(new sngxml::dom::Text(ToUtf32(message)));
195     messageElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(messageText.release()));
196     diagnosticElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(categoryElement.release()));
197     diagnosticElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(subcategoryElement.release()));
198     diagnosticElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(messageElement.release()));
199     std::unique_ptr<sngxml::dom::Element> spanElement = SpanToDomElement(module, defined);
200     if (spanElement)
201     {
202         diagnosticElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(spanElement.release()));
203     }
204     diagnosticsElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(diagnosticElement.release()));
205     for (const Span& span : references)
206     {
207         if (!span.Valid()) continue;
208         std::unique_ptr<sngxml::dom::Element> diagnosticElement(new sngxml::dom::Element(U"diagnostic"));
209         std::unique_ptr<sngxml::dom::Element> categoryElement(new sngxml::dom::Element(U"category"));
210         std::unique_ptr<sngxml::dom::Text> categoryText(new sngxml::dom::Text(U"info"));
211         categoryElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(categoryText.release()));
212         std::unique_ptr<sngxml::dom::Element> messageElement(new sngxml::dom::Element(U"message"));
213         std::unique_ptr<sngxml::dom::Text> messageText(new sngxml::dom::Text(ToUtf32("see reference to")));
214         messageElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(messageText.release()));
215         diagnosticElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(categoryElement.release()));
216         diagnosticElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(messageElement.release()));
217         std::unique_ptr<sngxml::dom::Element> spanElement = SpanToDomElement(module, span);
218         if (spanElement)
219         {
220             diagnosticElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(spanElement.release()));
221             diagnosticsElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(diagnosticElement.release()));
222         }
223     }
