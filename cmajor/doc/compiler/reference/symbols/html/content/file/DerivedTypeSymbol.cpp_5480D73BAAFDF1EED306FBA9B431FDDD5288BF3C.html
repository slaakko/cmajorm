 22 std::u32string DerivationStr(Derivation derivation)
 23 {

142 TypeDerivationRec RemovePointerDerivation(const TypeDerivationRec& typeDerivationRec)
143 {
144     TypeDerivationRec pointerRemovedDerivationRec;
145     int numPointers = CountPointerDerivations(typeDerivationRec.derivations);
146     if (numPointers > 0)
147     {
148         --numPointers;
149     }
150     int ip = 0;
151     int n = typeDerivationRec.derivations.size();
152     for (int i = 0; i < n; ++i)
153     {
154         Derivation derivation = typeDerivationRec.derivations[i];
155         if (derivation == Derivation::pointerDerivation)
156         {
157             if (ip < numPointers)
158             {
159                 pointerRemovedDerivationRec.derivations.push_back(derivation);
160                 ++ip;
161             }
162         }
163         else
164         {
165             pointerRemovedDerivationRec.derivations.push_back(derivation);
166         }
167     }
168     return pointerRemovedDerivationRec;
169 }

226 TypeDerivationRec AddPointerDerivation(const TypeDerivationRec& typeDerivationRec)
227 {
228     TypeDerivationRec pointerAddedDerivationRec;
229     int numPointers = CountPointerDerivations(typeDerivationRec.derivations);
230     int np = 0;
231     for (Derivation derivation : typeDerivationRec.derivations)
232     {
233         pointerAddedDerivationRec.derivations.push_back(derivation);
234         if (derivation == Derivation::pointerDerivation)
235         {
236             ++np;
237         }
238         if (np == numPointers)
239         {
240             pointerAddedDerivationRec.derivations.push_back(Derivation::pointerDerivation);
241             ++np;
242         }
243     }
244     return pointerAddedDerivationRec;

247 TypeDerivationRec UnifyDerivations(const TypeDerivationRec& left, const TypeDerivationRec& right)
248 {
249     TypeDerivationRec result;
250     if (HasFrontConstDerivation(left.derivations) || HasFrontConstDerivation(right.derivations))
251     {
252         result.derivations.push_back(Derivation::constDerivation);
253     }
254     int pointerCount = CountPointerDerivations(left.derivations) + CountPointerDerivations(right.derivations);
255     for (int i = 0; i < pointerCount; ++i)
256     {
257         result.derivations.push_back(Derivation::pointerDerivation);
258     }
259     if (HasLvalueReferenceDerivation(left.derivations))
260     {
261         result.derivations.push_back(Derivation::lvalueRefDerivation);
262     }
263     else if (HasRvalueReferenceDerivation(left.derivations))
264     {
265         result.derivations.push_back(Derivation::rvalueRefDerivation);
266     }
267     else if (HasLvalueReferenceDerivation(right.derivations))
268     {
269         result.derivations.push_back(Derivation::lvalueRefDerivation);
270     }
271     else if (HasRvalueReferenceDerivation(right.derivations))
272     {
273         result.derivations.push_back(Derivation::rvalueRefDerivation);
274     }
275     return result;
276 }

278 std::u32string MakeDerivedTypeName(TypeSymbol* baseType, const TypeDerivationRec& derivationRec)
279 {
280     std::u32string derivedTypeName;
281     bool constAdded = false;
282     if (HasFrontConstDerivation(derivationRec.derivations))
283     {
284         derivedTypeName.append(U"const");
285         constAdded = true;
286     }
287     if (!derivedTypeName.empty())
288     {
289         derivedTypeName.append(1, U' ');
290     }
291     derivedTypeName.append(baseType->Name());
292     for (Derivation derivation : derivationRec.derivations)
293     {
294         switch (derivation)
295         {
296             case Derivation::constDerivation:
297             {
298                 if (constAdded)
299                 {
300                     constAdded = false;
301                 }
302                 else
303                 {
304                     if (!derivedTypeName.empty())
305                     {
306                         derivedTypeName.append(1, U' ');
307                     }
308                     derivedTypeName.append(U"const ");
309                 }
310                 break;
311             }
312             case Derivation::lvalueRefDerivation:
313             {
314                 derivedTypeName.append(U"&");
315                 break;
316             }
317             case Derivation::rvalueRefDerivation:
318             {
319                 derivedTypeName.append(U"&&");
320                 break;
321             }
322             case Derivation::pointerDerivation:
323             {
324                 derivedTypeName.append(U"*");
325                 break;
326             }
327         }
328     }
329     return derivedTypeName;
330 }

392 void DerivedTypeSymbol::ComputeTypeId()
393 {
394     Module* systemCoreModule = GetModule()->GetSystemCoreModule();
395     if (!systemCoreModule)
396     {
397         throw std::runtime_error("internal error: system core module not found");
398     }
399     SymbolTable& systemCoreSymbolTable = systemCoreModule->GetSymbolTable();
400     boost::uuids::uuid typeId = baseType->TypeId();
401     int n = derivationRec.derivations.size();
402     for (int i = 0; i < n; ++i)
403     {
404         const boost::uuids::uuid& derivationId = systemCoreSymbolTable.GetDerivationId(derivationRec.derivations[i]);
405         const boost::uuids::uuid& positionId = systemCoreSymbolTable.GetPositionId(i);
406         for (int k = 0; k < boost::uuids::uuid::static_size(); ++k)
407         {
408             typeId.data[k] = typeId.data[k] ^ derivationId.data[k] ^ positionId.data[k];
409         }
410     }
411     SetTypeId(typeId);
412 }

525 TypeSymbol* DerivedTypeSymbol::RemoveDerivations(const TypeDerivationRec& sourceDerivationRec, const Span& span, const boost::uuids::uuid& moduleId)
526 {
527     TypeDerivationRec result;
528     const std::std::vector<Derivation>&sourceDerivations=sourceDerivationRec.derivations;
529     if (!HasFrontConstDerivation(sourceDerivations) && HasFrontConstDerivation(derivationRec.derivations))
530     {
531         result.derivations.push_back(Derivation::constDerivation);
532     }
533     int pointerDiff = CountPointerDerivations(derivationRec.derivations) - CountPointerDerivations(sourceDerivations);
534     if (pointerDiff != 0)
535     {
536         for (int i = 0; i < pointerDiff; ++i)
537         {
538             result.derivations.push_back(Derivation::pointerDerivation);
539         }
540     }
541     if (!HasLvalueReferenceDerivation(sourceDerivations) && HasLvalueReferenceDerivation(derivationRec.derivations))
542     {
543         result.derivations.push_back(Derivation::lvalueRefDerivation);
544     }
545     else if (!HasRvalueReferenceDerivation(sourceDerivations) && HasRvalueReferenceDerivation(derivationRec.derivations))
546     {
547         result.derivations.push_back(Derivation::rvalueRefDerivation);
548     }
549     return GetRootModuleForCurrentThread()->GetSymbolTable().MakeDerivedType(baseType, result, span, moduleId);
550 }

565 void* DerivedTypeSymbol::IrType(Emitter& emitter)
566 {
567     void* localIrType = emitter.GetIrTypeByTypeId(TypeId());
568     if (!localIrType)
569     {
570         if (baseType->IsVoidType() && GetBackEnd() == BackEnd::llvm)
571         {
572             localIrType = emitter.GetIrTypeForByte();
573         }
574         else
575         {
576             localIrType = baseType->IrType(emitter);
577         }
578         for (Derivation derivation : derivationRec.derivations)
579         {
580             switch (derivation)
581             {
582                 case Derivation::lvalueRefDerivation:
583                 case Derivation::rvalueRefDerivation:
584                 case Derivation::pointerDerivation:
585                 {
586                     localIrType = emitter.GetIrTypeForPtrType(localIrType);
587                     break;
588                 }
589                 default:
590                 {
591                     // todo
592                     break;
593                 }
594             }
595         }
596         emitter.SetIrTypeByTypeId(TypeId(), localIrType);
597     }
598     return localIrType;
599 }

603     if (HasFrontConstDerivation(derivationRec.derivations) && !HasPointerDerivation(derivationRec.derivations) && !HasReferenceDerivation(derivationRec.derivations))
604     {
605         return baseType->CreateDefaultIrValue(emitter);
606     }
607     else
608     {
609         return emitter.CreateDefaultIrValueForDerivedType(IrType(emitter));
610     }
611 }

613 void* DerivedTypeSymbol::CreateDIType(Emitter& emitter)
614 {
615     void* diType = baseType->GetDIType(emitter);
616     for (Derivation derivation : derivationRec.derivations)
617     {
618         switch (derivation)
619         {
620             case Derivation::constDerivation:
621             {
622                 return emitter.CreateConstDIType(diType);
623             }
624             case Derivation::lvalueRefDerivation:
625             {
626                 return emitter.CreateLValueRefDIType(diType);
627             }
628             case Derivation::rvalueRefDerivation:
629             {
630                 return emitter.CreateRValueRefDIType(diType);
631             }
632             case Derivation::pointerDerivation:
633             {
634                 return emitter.CreatePointerDIType(diType);
635             }
636         }
637     }
638     return diType;
639 }

641 ValueType DerivedTypeSymbol::GetValueType() const
642 {
643     if (HasPointerDerivation(derivationRec.derivations))
644     {
645         if (PointerCount() == 1)
646         {
647             switch (baseType->GetSymbolType())
648             {
649                 case SymbolType::charTypeSymbol:
650                 {
651                     return ValueType::stringValue;
652                 }
653                 case SymbolType::wcharTypeSymbol:
654                 {
655                     return ValueType::wstringValue;
656                 }
657                 case SymbolType::ucharTypeSymbol:
658                 {
659                     return ValueType::ustringValue;
660                 }
661                 default:
662                 {
663                     return ValueType::pointerValue;
664                 }
665             }
666         }
667         else
668         {
669             return ValueType::pointerValue;
670         }
671     }
672     return ValueType::none;
673 }

675 Value* DerivedTypeSymbol::MakeValue() const
676 {
677     if (HasPointerDerivation(derivationRec.derivations))
678     {
679         if (PointerCount() == 1)
680         {
681             switch (baseType->GetSymbolType())
682             {
683                 case SymbolType::charTypeSymbol:
684                 {
685                     return new StringValue(Span(), boost::uuids::nil_uuid(), -1, "");
686                 }
687                 case SymbolType::wcharTypeSymbol:
688                 {
689                     return new WStringValue(Span(), boost::uuids::nil_uuid(), -1, u"");
690                 }
691                 case SymbolType::ucharTypeSymbol:
692                 {
693                     return new UStringValue(Span(), boost::uuids::nil_uuid(), -1, U"");
694                 }
695             }
696         }
697     }
698     return TypeSymbol::MakeValue();
699 }
