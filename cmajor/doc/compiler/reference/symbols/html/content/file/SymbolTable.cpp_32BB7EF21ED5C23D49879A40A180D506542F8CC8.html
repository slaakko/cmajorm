  74 void SymbolTable::Write(SymbolWriter& writer)
  75 {
  76     if (module->Name() == U"System.Core")
  77     {
  78         uint32_t numDerivationIds = derivationIds.size();
  79         writer.GetBinaryWriter().WriteULEB128UInt(numDerivationIds);
  80         for (uint32_t i = 0; i < numDerivationIds; ++i)
  81         {
  82             writer.GetBinaryWriter().Write(derivationIds[i]);
  83         }
  84         uint32_t numPositionIds = positionIds.size();
  85         writer.GetBinaryWriter().WriteULEB128UInt(numPositionIds);
  86         for (uint32_t i = 0; i < numPositionIds; ++i)
  87         {
  88             writer.GetBinaryWriter().Write(positionIds[i]);
  89         }
  90     }
  91     globalNs.Write(writer);
  92     std::vector<ArrayTypeSymbol*> exportedArrayTypes;
  93     for (const std::std::unique_ptr<ArrayTypeSymbol>&arrayType : arrayTypes)
  94     {
  95         if (arrayType->IsProject())
  96         {
  97             exportedArrayTypes.push_back(arrayType.get());
  98         }
  99     }
 100     uint32_t na = exportedArrayTypes.size();
 101     writer.GetBinaryWriter().WriteULEB128UInt(na);
 102     for (ArrayTypeSymbol* exportedArrayType : exportedArrayTypes)
 103     {
 104         writer.Write(exportedArrayType);
 105     }
 106     std::vector<TypeSymbol*> exportedDerivedTypes;
 107     std::unordered_map<boost::uuids::uuid, TypeSymbol*, boost::boost::hash<boost::uuids::uuid>>derivedTypeMap;
 108     for (const auto& derivedType : derivedTypes)
 109     {
 110         if (derivedType->IsProject())
 111         {
 112             derivedTypeMap[derivedType->TypeId()] = derivedType.get();
 113         }
 114     }
 115     for (const auto& p : derivedTypeMap)
 116     {
 117         exportedDerivedTypes.push_back(p.second);
 118     }
 119     uint32_t ned = exportedDerivedTypes.size();
 120     writer.GetBinaryWriter().WriteULEB128UInt(ned);
 121     for (TypeSymbol* exportedDerivedType : exportedDerivedTypes)
 122     {
 123         writer.Write(exportedDerivedType);
 124     }
 125     std::vector<TypeSymbol*> exportedClassTemplateSpecializations;
 126     std::unordered_map<boost::uuids::uuid, TypeSymbol*, boost::boost::hash<boost::uuids::uuid>>specializationMap;
 127     for (const auto& classTemplateSpecialization : classTemplateSpecializations)
 128     {
 129         if (classTemplateSpecialization->IsProject())
 130         {
 131             specializationMap[classTemplateSpecialization->TypeId()] = classTemplateSpecialization.get();
 132         }
 133     }
 134     for (const auto& p : specializationMap)
 135     {
 136         exportedClassTemplateSpecializations.push_back(p.second);
 137     }
 138     uint32_t nec = exportedClassTemplateSpecializations.size();
 139     writer.GetBinaryWriter().WriteULEB128UInt(nec);
 140     for (TypeSymbol* classTemplateSpecialization : exportedClassTemplateSpecializations)
 141     {
 142         writer.Write(classTemplateSpecialization);
 143     }
 144     uint32_t nj = jsonClasses.size();
 145     writer.GetBinaryWriter().WriteULEB128UInt(nj);
 146     for (const std::u32string& jsonClass : jsonClasses)
 147     {
 148         writer.GetBinaryWriter().Write(jsonClass);
 149     }
 150     if (GetGlobalFlag(GlobalFlags::profile))
 151     {
 152         uint32_t n = profiledFunctionNameMap.size();
 153         writer.GetBinaryWriter().WriteULEB128UInt(n);
 154         for (const auto& p : profiledFunctionNameMap)
 155         {
 156             writer.GetBinaryWriter().Write(p.first);
 157             writer.GetBinaryWriter().Write(p.second);
 158         }
 159     }

 162 void SymbolTable::Read(SymbolReader& reader)
 163 {
 164     reader.SetSymbolTable(this);
 165     if (module->Name() == U"System.Core")
 166     {
 167         uint32_t numDerivationIds = reader.GetBinaryReader().ReadULEB128UInt();
 168         for (uint32_t i = 0; i < numDerivationIds; ++i)
 169         {
 170             boost::uuids::uuid derivationId;
 171             reader.GetBinaryReader().ReadUuid(derivationId);
 172             derivationIds.push_back(derivationId);
 173         }
 174         uint32_t numPositionIds = reader.GetBinaryReader().ReadULEB128UInt();
 175         for (uint32_t i = 0; i < numPositionIds; ++i)
 176         {
 177             boost::uuids::uuid positionId;
 178             reader.GetBinaryReader().ReadUuid(positionId);
 179             positionIds.push_back(positionId);
 180         }
 181     }
 182     reader.GetAstReader().SetModuleId(reader.RootModule()->GetModuleId(reader.GetModule()));
 183     globalNs.Read(reader);
 184     uint32_t na = reader.GetBinaryReader().ReadULEB128UInt();
 185     for (uint32_t i = 0; i < na; ++i)
 186     {
 187         ArrayTypeSymbol* arrayTypeSymbol = reader.ReadArrayTypeSymbol(&globalNs);
 188         arrayTypes.push_back(std::unique_ptr<ArrayTypeSymbol>(arrayTypeSymbol));
 189         reader.AddArrayType(arrayTypeSymbol);
 190     }
 191     uint32_t nd = reader.GetBinaryReader().ReadULEB128UInt();
 192     for (uint32_t i = 0; i < nd; ++i)
 193     {
 194         DerivedTypeSymbol* derivedTypeSymbol = reader.ReadDerivedTypeSymbol(&globalNs);
 195         derivedTypes.push_back(std::unique_ptr<DerivedTypeSymbol>(derivedTypeSymbol));
 196         reader.AddDerivedType(derivedTypeSymbol);
 197     }
 198     uint32_t nc = reader.GetBinaryReader().ReadULEB128UInt();
 199     for (uint32_t i = 0; i < nc; ++i)
 200     {
 201         ClassTemplateSpecializationSymbol* classTemplateSpecialization = reader.ReadClassTemplateSpecializationSymbol(&globalNs);
 202         classTemplateSpecializations.push_back(std::unique_ptr<ClassTemplateSpecializationSymbol>(classTemplateSpecialization));
 203         reader.AddClassTemplateSpecialization(classTemplateSpecialization);
 204     }
 205     uint32_t nj = reader.GetBinaryReader().ReadULEB128UInt();
 206     for (uint32_t i = 0; i < nj; ++i)
 207     {
 208         std::u32string jsonClass = reader.GetBinaryReader().ReadUtf32String();
 209         jsonClasses.insert(jsonClass);
 210     }
 211     if (GetGlobalFlag(GlobalFlags::profile))
 212     {
 213         uint32_t n = reader.GetBinaryReader().ReadULEB128UInt();
 214         for (uint32_t i = 0; i < n; ++i)
 215         {
 216             boost::uuids::uuid functionId;
 217             reader.GetBinaryReader().ReadUuid(functionId);
 218             std::u32string profiledFunctionName = reader.GetBinaryReader().ReadUtf32String();
 219             MapProfiledFunction(functionId, profiledFunctionName);
 220         }
 221     }

 224 void SymbolTable::Import(const SymbolTable& symbolTable)
 225 {
 226     globalNs.Import(const_cast<NamespaceSymbol*>(&symbolTable.globalNs), *this);
 227     for (const auto& pair : symbolTable.typeIdMap)
 228     {
 229         Symbol* typeOrConcept = pair.second;
 230         if (typeOrConcept->IsTypeSymbol())
 231         {
 232             TypeSymbol* type = static_cast<TypeSymbol*>(typeOrConcept);
 233             typeIdMap[type->TypeId()] = type;
 234             typeNameMap[type->FullName()] = type;
 235         }
 236         else if (typeOrConcept->GetSymbolType() == SymbolType::conceptSymbol)
 237         {
 238             ConceptSymbol* conceptSymbol = static_cast<ConceptSymbol*>(typeOrConcept);
 239             typeIdMap[conceptSymbol->TypeId()] = conceptSymbol;
 240         }
 241         else
 242         {
 243             Assert(false, "type or concept symbol expected");
 244         }
 245     }
 246     for (const auto& pair : symbolTable.functionIdMap)
 247     {
 248         FunctionSymbol* function = pair.second;
 249         functionIdMap[function->FunctionId()] = function;
 250     }
 251     for (const auto& p : symbolTable.derivedTypeMap)
 252     {
 253         derivedTypeMap[p.first] = p.second;
 254     }
 255     for (const auto& p : symbolTable.classTemplateSpecializationMap)
 256     {
 257         classTemplateSpecializationMap[p.first] = p.second;
 258     }
 259     for (const auto& p : symbolTable.arrayTypeMap)
 260     {
 261         arrayTypeMap[p.first] = p.second;
 262     }
 263     conversionTable.Add(symbolTable.conversionTable);
 264     for (ClassTypeSymbol* polymorphicClass : symbolTable.PolymorphicClasses())
 265     {
 266         AddPolymorphicClass(polymorphicClass);
 267     }
 268     for (ClassTypeSymbol* classHavingStaticConstructor : symbolTable.ClassesHavingStaticConstructor())
 269     {
 270         AddClassHavingStaticConstructor(classHavingStaticConstructor);
 271     }
 272     for (const std::u32string& jsonClass : symbolTable.JsonClasses())
 273     {
 274         AddJsonClass(jsonClass);
 275     }
 276     if (GetGlobalFlag(GlobalFlags::profile))
 277     {
 278         for (const auto& p : symbolTable.profiledFunctionNameMap)
 279         {
 280             const boost::uuids::uuid& functionId = p.first;
 281             const std::u32string& profiledFunctionName = p.second;
 282             MapProfiledFunction(functionId, profiledFunctionName);
 283         }
 284     }
 285 }

 287 void SymbolTable::FinishRead(const std::std::vector<ArrayTypeSymbol*>&arrayTypes, conststd::std::vector<DerivedTypeSymbol*>&derivedTypes, 
 288     const std::std::vector<ClassTemplateSpecializationSymbol*>&classTemplateSpecializations, conststd::std::vector<TypeOrConceptRequest>&typeAndConceptRequests, 
 289     const std::std::vector<FunctionRequest>&functionRequests, std::std::vector<FunctionSymbol*>&conversions)
 290 {
 291     ProcessTypeConceptAndFunctionRequests(typeAndConceptRequests, functionRequests);
 292     for (FunctionSymbol* conversion : conversions)
 293     {
 294         AddConversion(conversion, module);
 295     }
 296     for (ArrayTypeSymbol* arrayTypeSymbol : arrayTypes)
 297     {
 298         ArrayKey key(arrayTypeSymbol->ElementType(), arrayTypeSymbol->Size());
 299         arrayTypeMap[key] = arrayTypeSymbol;
 300     }
 301     for (DerivedTypeSymbol* derivedTypeSymbol : derivedTypes)
 302     {
 303         std::std::vector<DerivedTypeSymbol*>&derivedTypeVec=derivedTypeMap[derivedTypeSymbol->BaseType()->TypeId()];

 305         bool found = false;
 306         for (int i = 0; i < n; ++i)
 307         {
 308             DerivedTypeSymbol* prevDerivedTypeSymbol = derivedTypeVec[i];
 309             if (prevDerivedTypeSymbol->DerivationRec() == derivedTypeSymbol->DerivationRec())
 310             {
 311                 found = true;
 312                 break;
 313             }
 314         }
 315         if (!found)
 316         {
 317             derivedTypeVec.push_back(derivedTypeSymbol);
 318         }
 319     }
 320     AddClassTemplateSpecializationsToClassTemplateSpecializationMap(classTemplateSpecializations);
 321 }

 414 void SymbolTable::BeginFunction(FunctionNode& functionNode, int32_t functionIndex)
 415 {
 416     FunctionSymbol* functionSymbol = new FunctionSymbol(functionNode.GetSpan(), functionNode.GroupId());
 417     SetFunctionIdFor(functionSymbol);
 418     functionSymbol->SetIndex(functionIndex);
 419     if ((functionNode.GetSpecifiers() & Specifiers::constexpr_) != Specifiers::none)
 420     {
 421         functionSymbol->SetConstExpr();
 422     }
 423     functionSymbol->SetHasSource();
 424     functionSymbol->SetCompileUnit(currentCompileUnit);
 425     functionSymbol->SetModule(module);
 426     functionSymbol->SetGroupName(functionNode.GroupId());
 427     if (functionNode.WhereConstraint())
 428     {
 429         CloneContext cloneContext;
 430         functionSymbol->SetConstraint(static_cast<WhereConstraintNode*>(functionNode.WhereConstraint()->Clone(cloneContext)));
 431     }
 432     if (functionSymbol->GroupName() == U"main")
 433     {
 434         if (functionNode.IsProgramMain())
 435         {
 436             functionSymbol->SetCDecl();
 437             functionSymbol->SetProgramMain();
 438         }
 439         else
 440         {
 441             if (mainFunctionSymbol)
 442             {
 443                 throw Exception(module, "already has main function", functionNode.GetSpan(), mainFunctionSymbol->GetSpan());
 444             }
 445             else
 446             {
 447                 mainFunctionSymbol = functionSymbol;
 448             }
 449         }
 450     }
 451     MapNode(&functionNode, functionSymbol);
 452     BeginContainer(functionSymbol);
 453     parameterIndex = 0;
 454     ResetDeclarationBlockIndex();
 455 }
