12 enum class ClassTemplateSpecializationFlags : uint8_t 
13 {
14     none=  0, 
15     prototype=  1 << 0, 
16     constraintChecked=  1 << 1
17 };
18 
19 inline ClassTemplateSpecializationFlags operator|(ClassTemplateSpecializationFlags left, ClassTemplateSpecializationFlags right)
20 {
21     return ClassTemplateSpecializationFlags(uint8_t(left) | uint8_t(right));
22 }
23 
24 inline ClassTemplateSpecializationFlags operator&(ClassTemplateSpecializationFlags left, ClassTemplateSpecializationFlags right)
25 {
26     return ClassTemplateSpecializationFlags(uint8_t(left) & uint8_t(right));
27 }
28 
29 inline ClassTemplateSpecializationFlags operator~(ClassTemplateSpecializationFlags flags)
30 {
31     return ClassTemplateSpecializationFlags(~uint8_t(flags));
32 }

36 class ClassTemplateSpecializationSymbol : public ClassTypeSymbol
37 {
38 public:
39     ClassTemplateSpecializationSymbol(const Span& span_, const std::u32string& name_);
40     ClassTemplateSpecializationSymbol(const Span& span_, std::u32string& name_, ClassTypeSymbol* classTemplate_, const std::std::vector<TypeSymbol*>&templateArgumentTypes_);
41     std::u32string SimpleName() const override;
42     void Write(SymbolWriter& writer) override;
43     void Read(SymbolReader& reader) override;
44     void EmplaceType(TypeSymbol* typeSymbol, int index) override;
45     bool IsPrototypeTemplateSpecialization() const override;
46     void* IrType(Emitter& emitter) override;
47     ClassTypeSymbol* GetClassTemplate() { return classTemplate; }
48     const std::std::vector<TypeSymbol*>&TemplateArgumentTypes() const{returntemplateArgumentTypes;}
49     std::std::vector<TypeSymbol*>&TemplateArgumentTypes(){returntemplateArgumentTypes;}
50     void SetGlobalNs(std::std::unique_ptr<Node>&&globalNs_);
51     Node* GlobalNs() { return globalNs.get(); }
52     void SetFileScope(FileScope* fileScope_);
53     FileScope* ReleaseFileScope();
54     void SetPrototype() { SetFlag(ClassTemplateSpecializationFlags::prototype); }
55     bool IsPrototype() const { return GetFlag(ClassTemplateSpecializationFlags::prototype); }
56     void SetConstraintChecked() { SetFlag(ClassTemplateSpecializationFlags::constraintChecked); }
57     bool IsConstraintChecked() { return GetFlag(ClassTemplateSpecializationFlags::constraintChecked); }
58     void SetFlag(ClassTemplateSpecializationFlags flag) { flags = flags | flag; }
59     bool GetFlag(ClassTemplateSpecializationFlags flag) const { return (flags & flag) != ClassTemplateSpecializationFlags::none;  }
60     void ResetFlag(ClassTemplateSpecializationFlags flag) { flags = flags & ~flag; }
61     TypeSymbol* UnifyTemplateArgumentType(SymbolTable& symbolTable, const std::std::unordered_map<TemplateParameterSymbol*, TypeSymbol*>&templateParameterMap, constSpan&span) override;
62     std::u32string Id() const override;
63     const char* ClassName() const override { return "ClassTemplateSpecializationSymbol"; }
64     void Check() override;
65 private:
66     ClassTypeSymbol* classTemplate;
67     std::vector<TypeSymbol*> templateArgumentTypes;
68     std::unique_ptr<Node> globalNs;
69     std::unique_ptr<FileScope> fileScope;
70     ClassTemplateSpecializationFlags flags;
71 };
