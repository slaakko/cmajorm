 24 cmajor::debug::DIPrimitiveType::Kind ToPrimitiveTypeKind(SymbolType symbolType)
 25 {
 26     switch (symbolType)
 27     {
 28         case SymbolType::boolTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::boolType;
 29         case SymbolType::sbyteTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::sbyteType;
 30         case SymbolType::byteTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::byteType;
 31         case SymbolType::shortTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::shortType;
 32         case SymbolType::ushortTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::ushortType;
 33         case SymbolType::intTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::intType;
 34         case SymbolType::uintTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::uintType;
 35         case SymbolType::longTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::longType;
 36         case SymbolType::ulongTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::ulongType;
 37         case SymbolType::floatTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::floatType;
 38         case SymbolType::doubleTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::doubleType;
 39         case SymbolType::charTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::charType;
 40         case SymbolType::wcharTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::wcharType;
 41         case SymbolType::ucharTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::ucharType;
 42         case SymbolType::voidTypeSymbol: return cmajor::debug::DIPrimitiveType::Kind::voidType;
 43         default: return cmajor::debug::DIPrimitiveType::Kind::none;
 44     }
 45 }

 47 void TypeIndex::AddType(const boost::uuids::uuid& typeId, TypeSymbol* typeSymbol, cmajor::ir::Emitter& emitter)
 48 {
 49     std::lock_guard<std::recursive_mutex> lock(mtx);
 50     auto it = typeMap.find(typeId);
 51     if (it == typeMap.cend())
 52     {
 53         if (typeSymbol->IsBasicTypeSymbol())
 54         {
 55             cmajor::debug::DIPrimitiveType* type = new cmajor::debug::DIPrimitiveType();
 56             type->SetPrimitiveTypeKind(ToPrimitiveTypeKind(typeSymbol->GetSymbolType()));
 57             type->SetId(typeSymbol->TypeId());
 58             type->SetName(ToUtf8(typeSymbol->Name()));
 59             type->SetIrName(emitter.GetIrTypeName(typeSymbol->IrType(emitter)));
 60             typeMap[type->Id()] = type;
 61             diTypes.push_back(std::unique_ptr<cmajor::debug::DIType>(type));
 62         }
 63         else
 64         {
 65             switch (typeSymbol->GetSymbolType())
 66             {
 67                 case SymbolType::enumTypeSymbol:
 68                 {
 69                     EnumTypeSymbol* enumType = static_cast<EnumTypeSymbol*>(typeSymbol);
 70                     cmajor::debug::DIEnumType* type = new cmajor::debug::DIEnumType();
 71                     type->SetId(typeSymbol->TypeId());
 72                     type->SetName(ToUtf8(typeSymbol->FullName()));
 73                     type->SetIrName(emitter.GetIrTypeName(typeSymbol->IrType(emitter)));
 74                     typeMap[type->Id()] = type;
 75                     type->SetUnderlyingTypeId(enumType->UnderlyingType()->TypeId());
 76                     diTypes.push_back(std::unique_ptr<cmajor::debug::DIType>(type));
 77                     AddType(enumType->UnderlyingType()->TypeId(), enumType->UnderlyingType(), emitter);
 78                     std::vector<EnumConstantSymbol*> enumConstants = enumType->GetEnumConstants();
 79                     for (EnumConstantSymbol* enumConstant : enumConstants)
 80                     {
 81                         cmajor::debug::DIEnumConstant ec;
 82                         ec.SetName(ToUtf8(enumConstant->Name()));
 83                         ec.SetValue(enumConstant->GetValue()->ToString());
 84                         ec.SetStrValue(ToUtf8(enumConstant->StrValue()));
 85                         type->AddEnumConstant(std::move(ec));
 86                     }
 87                     break;
 88                 }
 89                 case SymbolType::classTypeSymbol:
 90                 {
 91                     ClassTypeSymbol* classType = static_cast<ClassTypeSymbol*>(typeSymbol);
 92                     cmajor::debug::DIClassType* type = new cmajor::debug::DIClassType();
 93                     type->SetId(typeSymbol->TypeId());
 94                     type->SetName(ToUtf8(typeSymbol->FullName()));
 95                     type->SetIrName(emitter.GetIrTypeName(typeSymbol->IrType(emitter)));
 96                     typeMap[type->Id()] = type;
 97                     diTypes.push_back(std::unique_ptr<cmajor::debug::DIType>(type));
 98                     if (classType->IsPolymorphic())
 99                     {
100                         type->SetPolymorphic();
101                         type->SetVmtPtrIndex(classType->VmtPtrIndex());
102                         type->SetVmtVariableName(emitter.MakeVmtVariableName(classType->VmtObjectName(emitter)));
103                     }
104                     if (classType->BaseClass())
105                     {
106                         type->SetBaseClassId(classType->BaseClass()->TypeId());
107                         AddType(classType->BaseClass()->TypeId(), classType->BaseClass(), emitter);
108                     }
109                     for (TemplateParameterSymbol* templateParameter : classType->TemplateParameters())
110                     {
111                         cmajor::debug::DITemplateParameter* tp = new cmajor::debug::DITemplateParameter();
112                         tp->SetId(templateParameter->TypeId());
113                         tp->SetName(ToUtf8(templateParameter->Name()));
114                         typeMap[tp->Id()] = type;
115                         type->AddTemplateParameter(tp);
116                     }
117                     for (MemberVariableSymbol* memberVariable : classType->MemberVariables())
118                     {
119                         cmajor::debug::DIVariable* variable = new cmajor::debug::DIVariable(cmajor::debug::DIVariable::Kind::memberVariable);
120                         variable->SetName(ToUtf8(memberVariable->Name()));
121                         variable->SetIrName("m" + std::to_string(memberVariable->LayoutIndex()));
122                         variable->SetTypeId(memberVariable->GetType()->TypeId());
123                         AddType(memberVariable->GetType()->TypeId(), memberVariable->GetType(), emitter);
124                         type->AddMemberVariable(variable);
125                     }
126                     break;
127                 }
128                 case SymbolType::classTemplateSpecializationSymbol:
129                 {
130                     ClassTemplateSpecializationSymbol* specialization = static_cast<ClassTemplateSpecializationSymbol*>(typeSymbol);
131                     cmajor::debug::DIClassTemplateSpecializationType* type = new cmajor::debug::DIClassTemplateSpecializationType();
132                     ClassTypeSymbol* classTemplateType = specialization->GetClassTemplate();
133                     AddType(classTemplateType->TypeId(), classTemplateType, emitter);
134                     type->SetPrimaryTypeId(classTemplateType->TypeId());
135                     for (TypeSymbol* templateArgumentType : specialization->TemplateArgumentTypes())
136                     {
137                         type->AddTemplateArgumentTypeId(templateArgumentType->TypeId());
138                         AddType(templateArgumentType->TypeId(), templateArgumentType, emitter);
139                     }
140                     cmajor::debug::ContainerClassTemplateKind containerKind = GetContainerClassTemplateKind(classTemplateType->FullName());
141                     type->SetContainerClassTemplateKind(containerKind);
142                     if (containerKind != cmajor::debug::ContainerClassTemplateKind::notContainerClassTemplate)
143                     {
144                         Symbol* valueTypeSymbol = specialization->GetContainerScope()->Lookup(U"ValueType", ScopeLookup::this_and_base_and_parent);
145                         if (valueTypeSymbol && valueTypeSymbol->GetSymbolType() == SymbolType::typedefSymbol)
146                         {
147                             TypedefSymbol* valueTypeTypedef = static_cast<TypedefSymbol*>(valueTypeSymbol);
148                             TypeSymbol* valueType = valueTypeTypedef->GetType();
149                             type->SetValueTypeId(valueType->TypeId());
150                             AddType(valueType->TypeId(), valueType, emitter);
151                         }
152                     }
153                     type->SetId(typeSymbol->TypeId());
154                     type->SetName(ToUtf8(typeSymbol->FullName()));
155                     type->SetIrName(emitter.GetIrTypeName(typeSymbol->IrType(emitter)));
156                     typeMap[type->Id()] = type;
157                     diTypes.push_back(std::unique_ptr<cmajor::debug::DIType>(type));
158                     if (specialization->IsPolymorphic())
159                     {
160                         type->SetPolymorphic();
161                         type->SetVmtPtrIndex(specialization->VmtPtrIndex());
162                         type->SetIrName(emitter.GetIrTypeName(typeSymbol->IrType(emitter)));
163                         type->SetVmtVariableName(emitter.MakeVmtVariableName(specialization->VmtObjectName(emitter)));
164                     }
165                     if (specialization->BaseClass())
166                     {
167                         type->SetBaseClassId(specialization->BaseClass()->TypeId());
168                         AddType(specialization->BaseClass()->TypeId(), specialization->BaseClass(), emitter);
169                     }
170                     for (MemberVariableSymbol* memberVariable : specialization->MemberVariables())
171                     {
172                         cmajor::debug::DIVariable* variable = new cmajor::debug::DIVariable(cmajor::debug::DIVariable::Kind::memberVariable);
173                         variable->SetName(ToUtf8(memberVariable->Name()));
174                         variable->SetIrName("m" + std::to_string(memberVariable->LayoutIndex()));
175                         variable->SetTypeId(memberVariable->GetType()->TypeId());
176                         AddType(memberVariable->GetType()->TypeId(), memberVariable->GetType(), emitter);
177                         type->AddMemberVariable(variable);
178                     }
179                     break;
180                 }
181                 case SymbolType::delegateTypeSymbol:
182                 {
183                     DelegateTypeSymbol* delegateTypeSymbol = static_cast<DelegateTypeSymbol*>(typeSymbol);
184                     cmajor::debug::DIDelegateType* type = new cmajor::debug::DIDelegateType();
185                     type->SetId(typeSymbol->TypeId());
186                     type->SetName(ToUtf8(typeSymbol->FullName()));
187                     type->SetIrName(emitter.GetIrTypeName(typeSymbol->IrType(emitter)));
188                     typeMap[type->Id()] = type;
189                     diTypes.push_back(std::unique_ptr<cmajor::debug::DIType>(type));
190                     break;
191                 }
192                 case SymbolType::classDelegateTypeSymbol:
193                 {
194                     ClassDelegateTypeSymbol* classDelegateTypeSymbol = static_cast<ClassDelegateTypeSymbol*>(typeSymbol);
195                     cmajor::debug::DIClassDelegateType* type = new cmajor::debug::DIClassDelegateType();
196                     type->SetId(typeSymbol->TypeId());
197                     type->SetName(ToUtf8(typeSymbol->FullName()));
198                     type->SetIrName(emitter.GetIrTypeName(typeSymbol->IrType(emitter)));
199                     typeMap[type->Id()] = type;
200                     diTypes.push_back(std::unique_ptr<cmajor::debug::DIType>(type));
201                     break;
202                 }
203                 case SymbolType::interfaceTypeSymbol:
204                 {
205                     InterfaceTypeSymbol* interfaceTypeSymbol = static_cast<InterfaceTypeSymbol*>(typeSymbol);
206                     cmajor::debug::DIInterfaceType* type = new cmajor::debug::DIInterfaceType();
207                     type->SetId(typeSymbol->TypeId());
208                     type->SetName(ToUtf8(typeSymbol->FullName()));
209                     type->SetIrName(emitter.GetIrTypeName(typeSymbol->IrType(emitter)));
210                     typeMap[type->Id()] = type;
211                     diTypes.push_back(std::unique_ptr<cmajor::debug::DIType>(type));
212                     break;
213                 }
214                 case SymbolType::derivedTypeSymbol:
215                 {
216                     DerivedTypeSymbol* derivedType = static_cast<DerivedTypeSymbol*>(typeSymbol);
217                     if (derivedType->IsConstType())
218                     {
219                         TypeSymbol* nonConstType = derivedType->RemoveConst(derivedType->GetSpan());
220                         cmajor::debug::DIConstType* type = new cmajor::debug::DIConstType();
221                         type->SetId(typeSymbol->TypeId());
222                         type->SetName(ToUtf8(typeSymbol->FullName()));
223                         type->SetIrName(emitter.GetIrTypeName(typeSymbol->IrType(emitter)));
224                         type->SetBaseTypeId(nonConstType->TypeId());
225                         typeMap[type->Id()] = type;
226                         diTypes.push_back(std::unique_ptr<cmajor::debug::DIType>(type));
227                         AddType(nonConstType->TypeId(), nonConstType, emitter);
228                     }
229                     else if (derivedType->IsReferenceType())
230                     {
231                         TypeSymbol* nonReferenceType = derivedType->RemoveReference(derivedType->GetSpan());
232                         cmajor::debug::DIReferenceType* type = new cmajor::debug::DIReferenceType();
233                         type->SetId(typeSymbol->TypeId());
234                         type->SetName(ToUtf8(typeSymbol->FullName()));
235                         type->SetIrName(emitter.GetIrTypeName(typeSymbol->IrType(emitter)));
236                         type->SetBaseTypeId(nonReferenceType->TypeId());
237                         typeMap[type->Id()] = type;
238                         diTypes.push_back(std::unique_ptr<cmajor::debug::DIType>(type));
239                         AddType(nonReferenceType->TypeId(), nonReferenceType, emitter);
240                     }
241                     else if (derivedType->IsPointerType())
242                     {
243                         TypeSymbol* pointedToType = derivedType->RemovePointer(derivedType->GetSpan());
244                         cmajor::debug::DIPointerType* pointerType = new cmajor::debug::DIPointerType();
245                         pointerType->SetPointedTypeId(pointedToType->TypeId());
246                         pointerType->SetId(typeSymbol->TypeId());
247                         pointerType->SetName(ToUtf8(pointedToType->FullName() + U"*"));
248                         pointerType->SetIrName(emitter.GetIrTypeName(typeSymbol->IrType(emitter)));
249                         typeMap[pointerType->Id()] = pointerType;
250                         diTypes.push_back(std::unique_ptr<cmajor::debug::DIType>(pointerType));
251                         AddType(pointedToType->TypeId(), pointedToType, emitter);
252                     }
253                     break;
254                 }
255                 case SymbolType::arrayTypeSymbol:
256                 {
257                     ArrayTypeSymbol* arrayTypeSymbol = static_cast<ArrayTypeSymbol*>(typeSymbol);
258                     cmajor::debug::DIArrayType* type = new cmajor::debug::DIArrayType();
259                     type->SetId(typeSymbol->TypeId());
260                     type->SetName(ToUtf8(typeSymbol->FullName()));
261                     type->SetIrName(emitter.GetIrTypeName(typeSymbol->IrType(emitter)));
262                     typeMap[type->Id()] = type;
263                     TypeSymbol* elementType = arrayTypeSymbol->ElementType();
264                     type->SetElementTypeId(elementType->TypeId());
265                     type->SetSize(arrayTypeSymbol->Size());
266                     diTypes.push_back(std::unique_ptr<cmajor::debug::DIType>(type));
267                     AddType(elementType->TypeId(), elementType, emitter);
268                     break;
269                 }
270             }
271         }
272     }

275 void TypeIndex::Write(soulng::util::BinaryWriter& writer)
276 {
277     int32_t n = diTypes.size();
278     writer.Write(n);
279     for (int32_t i = 0; i < n; ++i)
280     {
281         cmajor::debug::DIType* type = diTypes[i].get();
282         cmajor::debug::WriteType(writer, type);
283     }
284 }
