 25 void ConstantSymbol::Write(SymbolWriter& writer)
 26 {
 27     Symbol::Write(writer);
 28     writer.GetBinaryWriter().Write(type->TypeId());
 29     bool hasComplexValue = value->IsComplexValue();
 30     writer.GetBinaryWriter().Write(hasComplexValue);
 31     if (hasComplexValue)
 32     {
 33         uint32_t sizePos = writer.GetBinaryWriter().Pos();
 34         uint32_t sizeOfValue = 0;
 35         writer.GetBinaryWriter().Write(sizeOfValue);
 36         uint32_t startPos = writer.GetBinaryWriter().Pos();
 37         value->Write(writer.GetBinaryWriter());
 38         uint32_t endPos = writer.GetBinaryWriter().Pos();
 39         sizeOfValue = endPos - startPos;
 40         writer.GetBinaryWriter().Seek(sizePos);
 41         writer.GetBinaryWriter().Write(sizeOfValue);
 42         writer.GetBinaryWriter().Seek(endPos);
 43     }
 44     else
 45     {
 46         WriteValue(value.get(), writer.GetBinaryWriter());
 47     }
 48     writer.GetBinaryWriter().Write(strValue);
 49 }

 51 void ConstantSymbol::Read(SymbolReader& reader)
 52 {
 53     Symbol::Read(reader);
 54     boost::uuids::uuid typeId;
 55     reader.GetBinaryReader().ReadUuid(typeId);
 56     reader.GetSymbolTable()->EmplaceTypeRequest(reader, this, typeId, 0);
 57     bool hasComplexValue = reader.GetBinaryReader().ReadBool();
 58     if (hasComplexValue)
 59     {
 60         sizeOfValue = reader.GetBinaryReader().ReadUInt();
 61         valuePos = reader.GetBinaryReader().Pos();
 62         reader.GetBinaryReader().Skip(sizeOfValue);
 63         filePathReadFrom = reader.GetBinaryReader().FileName();
 64     }
 65     else
 66     {
 67         value = ReadValue(reader.GetBinaryReader(), GetSpan());
 68     }
 69     strValue = reader.GetBinaryReader().ReadUtf32String();
 70 }

 72 Value* ConstantSymbol::GetValue()
 73 {
 74     if (!value)
 75     {
 76         if (filePathReadFrom.empty())
 77         {
 78             throw Exception(GetRootModuleForCurrentThread(), "internal error: could not read value: value file name not set", GetSpan());
 79         }
 80         BinaryReader reader(filePathReadFrom);
 81         reader.Skip(valuePos);
 82         value.reset(type->MakeValue());
 83         if (!value)
 84         {
 85             throw Exception(GetRootModuleForCurrentThread(), "internal error: could not read value because could not create value of type '" + ToUtf8(type->FullName()) + "'", GetSpan());
 86         }
 87         value->Read(reader);
 88     }
 89     return value.get();
 90 }

115 void ConstantSymbol::SetSpecifiers(Specifiers specifiers)
116 {
117     Specifiers accessSpecifiers = specifiers & Specifiers::access_;
118     SetAccess(accessSpecifiers);
119     if ((specifiers & Specifiers::static_) != Specifiers::none)
120     {
121         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be static", GetSpan());
122     }
123     if ((specifiers & Specifiers::virtual_) != Specifiers::none)
124     {
125         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be virtual", GetSpan());
126     }
127     if ((specifiers & Specifiers::override_) != Specifiers::none)
128     {
129         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be override", GetSpan());
130     }
131     if ((specifiers & Specifiers::abstract_) != Specifiers::none)
132     {
133         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be abstract", GetSpan());
134     }
135     if ((specifiers & Specifiers::inline_) != Specifiers::none)
136     {
137         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be inline", GetSpan());
138     }
139     if ((specifiers & Specifiers::explicit_) != Specifiers::none)
140     {
141         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be explicit", GetSpan());
142     }
143     if ((specifiers & Specifiers::external_) != Specifiers::none)
144     {
145         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be external", GetSpan());
146     }
147     if ((specifiers & Specifiers::suppress_) != Specifiers::none)
148     {
149         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be suppressed", GetSpan());
150     }
151     if ((specifiers & Specifiers::default_) != Specifiers::none)
152     {
153         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be default", GetSpan());
154     }
155     if ((specifiers & Specifiers::constexpr_) != Specifiers::none)
156     {
157         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be constexpr", GetSpan());
158     }
159     if ((specifiers & Specifiers::cdecl_) != Specifiers::none)
160     {
161         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be cdecl", GetSpan());
162     }
163     if ((specifiers & Specifiers::nothrow_) != Specifiers::none)
164     {
165         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be nothrow", GetSpan());
166     }
167     if ((specifiers & Specifiers::throw_) != Specifiers::none)
168     {
169         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be throw", GetSpan());
170     }
171     if ((specifiers & Specifiers::new_) != Specifiers::none)
172     {
173         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be new", GetSpan());
174     }
175     if ((specifiers & Specifiers::const_) != Specifiers::none)
176     {
177         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be const", GetSpan());
178     }
179     if ((specifiers & Specifiers::unit_test_) != Specifiers::none)
180     {
181         throw Exception(GetRootModuleForCurrentThread(), "constant cannot be unit_test", GetSpan());
182     }
183 }

191 std::string ConstantSymbol::Syntax() const
192 {
193     std::string syntax = GetSpecifierStr();
194     if (!syntax.empty())
195     {
196         syntax.append(1, ' ');
197     }
198     syntax.append("const ");
199     syntax.append(ToUtf8(GetType()->DocName()));
200     syntax.append(1, ' ');
201     syntax.append(ToUtf8(DocName()));
202     if (value)
203     {
204         syntax.append(" = ");
205         std::string valueStr = value->ToString();
206         if (GetType()->IsUnsignedType())
207         {
208             valueStr.append(1, 'u');
209         }
210         syntax.append(valueStr);
211     }
212     syntax.append(1, ';');
213     return syntax;
214 }

221 void* ConstantSymbol::ArrayIrObject(Emitter& emitter, bool create)
222 {
223     if (!type->IsArrayType())
224     {
225         throw Exception(GetRootModuleForCurrentThread(), "internal error: array object expected", GetSpan());
226     }
227     if (!value)
228     {
229         throw Exception(GetRootModuleForCurrentThread(), "internal error: array value missing", GetSpan());
230     }
231     if (value->GetValueType() != ValueType::arrayValue)
232     {
233         throw Exception(GetRootModuleForCurrentThread(), "internal error: array value expected", GetSpan());
234     }
235     ArrayValue* arrayValue = static_cast<ArrayValue*>(value.get());
236     void* irArrayType = type->IrType(emitter);
237     void* irArrayObject = emitter.GetOrInsertGlobal(ToUtf8(MangledName()), irArrayType);
238     if (create)
239     {
240         void* arrayObjectGlobal = irArrayObject;
241         emitter.SetInitializer(arrayObjectGlobal, arrayValue->IrValue(emitter));
242     }
243     return irArrayObject;
244 }

246 void* ConstantSymbol::StructureIrObject(Emitter& emitter, bool create)
247 {
248     if (!type->IsClassTypeSymbol())
249     {
250         throw Exception(GetRootModuleForCurrentThread(), "internal error: class type object expected", GetSpan());
251     }
252     if (!value)
253     {
254         throw Exception(GetRootModuleForCurrentThread(), "internal error: structured value missing", GetSpan());
255     }
256     if (value->GetValueType() != ValueType::structuredValue)
257     {
258         throw Exception(GetRootModuleForCurrentThread(), "internal error: structured value expected", GetSpan());
259     }
260     StructuredValue* structuredValue = static_cast<StructuredValue*>(value.get());
261     void* irStructureType = type->IrType(emitter);
262     void* irStructureObject = emitter.GetOrInsertGlobal(ToUtf8(MangledName()), irStructureType);
263     if (create)
264     {
265         void* structureObjectGlobal = irStructureObject;
266         emitter.SetInitializer(structureObjectGlobal, structuredValue->IrValue(emitter));
267     }
268     return irStructureObject;
269 }
