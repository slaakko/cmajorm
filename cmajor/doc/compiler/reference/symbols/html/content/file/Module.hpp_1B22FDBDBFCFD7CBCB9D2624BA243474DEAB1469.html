 39 enum class ModuleFlags : uint8_t 
 40 {
 41     none=  0, system=  1 << 0, core=  1 << 1, root=  1 << 2, immutable=  1 << 3, compiling=  1 << 4
 42 };
 43 
 44 inline ModuleFlags operator|(ModuleFlags left, ModuleFlags right)
 45 {
 46     return ModuleFlags(uint8_t(left) | uint8_t(right));
 47 }
 48 
 49 inline ModuleFlags operator&(ModuleFlags left, ModuleFlags right)
 50 {
 51     return ModuleFlags(uint8_t(left) & uint8_t(right));
 52 }
 53 
 54 inline ModuleFlags operator~(ModuleFlags flags)
 55 {
 56     return ModuleFlags(~uint8_t(flags));
