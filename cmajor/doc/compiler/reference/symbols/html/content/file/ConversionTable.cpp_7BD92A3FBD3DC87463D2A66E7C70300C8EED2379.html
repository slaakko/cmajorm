 19 void ConversionTableEntry::CheckValid() const
 20 {
 21     if (!sourceType || !targetType)
 22     {
 23         throw std::runtime_error("invalid conversion table entry: source type is null or target type is null");
 24     }
 25     if (!sourceType->BaseType() || !targetType->BaseType())
 26     {
 27         throw std::runtime_error("invalid conversion table entry: source base type is null or target base type is null");
 28     }
 29 }

 31 void ConversionTable::AddConversion(FunctionSymbol* conversion, Module* module)
 32 {
 33 #ifdef IMMUTABLE_MODULE_CHECK
 34 
 35 
 36 
 37 
 38 #endif
 39     ConversionTableEntry entry(conversion->ConversionSourceType()->PlainType(conversion->GetSpan(), module), 
 40         conversion->ConversionTargetType()->PlainType(conversion->GetSpan(), module));
 41 #ifdef VALID_CONVERSION_TABLE_ENTRY_CHECK
 42 
 43 #endif
 44     conversionMap.erase(entry);
 45     conversionMap.insert(std::make_pair(entry, conversion));
 46 }

 53 void ConversionTable::Add(const ConversionTable& that)
 54 {
 55 #ifdef IMMUTABLE_MODULE_CHECK
 56 
 57 
 58 
 59 
 60 #endif
 61     for (const auto& p : that.conversionMap)
 62     {
 63         ConversionTableEntry entry(p.first);
 64 #ifdef VALID_CONVERSION_TABLE_ENTRY_CHECK
 65 
 66 #endif
 67         conversionMap[entry] = p.second;
 68     }
 69 }

 71 void ConversionTable::Check()
 72 {
 73     for (const auto& p : conversionMap)
 74     {
 75         const ConversionTableEntry& entry = p.first;
 76         entry.sourceType->SetFlag(SymbolFlags::inConversionTable);
 77         entry.sourceType->BaseType()->SetFlag(SymbolFlags::inConversionTable);
 78         entry.targetType->SetFlag(SymbolFlags::inConversionTable);
 79         entry.targetType->BaseType()->SetFlag(SymbolFlags::inConversionTable);
 80         entry.CheckValid();
 81     }
 82 }

 84 FunctionSymbol* ConversionTable::GetConversion(TypeSymbol* sourceType, TypeSymbol* targetType, const Span& span) const
 85 {
 86     TypeSymbol* sourcePlainType = sourceType->PlainType(span);
 87     TypeSymbol* targetPlainType = targetType->PlainType(span);
 88     ConversionTableEntry entry(sourcePlainType, targetPlainType);
 89     auto it = conversionMap.find(entry);
 90     if (it != conversionMap.cend())
 91     {
 92         return it->second;
 93     }
 94     return nullptr;
 95 }
