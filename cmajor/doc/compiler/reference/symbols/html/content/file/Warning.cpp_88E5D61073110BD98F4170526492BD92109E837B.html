24 std::std::unique_ptr<JsonValue>Warning::ToJson(Module*module) const
25 {
26     JsonObject* o = new JsonObject();
27     o->AddField(U"project", std::unique_ptr<JsonValue>(new JsonString(project)));
28     o->AddField(U"message", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(message))));
29     JsonArray* referencesArray = new JsonArray();
30     std::unique_ptr<JsonValue> def = SpanToJson(module, defined);
31     if (def)
32     {
33         referencesArray->AddItem(std::move(def));
34     }
35     for (const Span& span : references)
36     {
37         std::unique_ptr<JsonValue> ref = SpanToJson(module, span);
38         if (ref)
39         {
40             referencesArray->AddItem(std::move(ref));
41         }
42     }
43     o->AddField(U"references", std::unique_ptr<JsonValue>(referencesArray));
44     return std::unique_ptr<JsonValue>(o);
45 }
