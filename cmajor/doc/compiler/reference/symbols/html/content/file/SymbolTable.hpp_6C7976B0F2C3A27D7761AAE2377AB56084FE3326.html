 61 struct ClassTemplateSpecializationKeyHash 
 62 {
 63     size_t operator()(const ClassTemplateSpecializationKey& key) const
 64     {
 65         size_t x = boost::hash<boost::uuids::uuid>()(key.classTemplate->TypeId());
 66         int n = key.templateArgumentTypes.size();
 67         for (int i = 0; i < n; ++i)
 68         {
 69             x = x ^ boost::hash<boost::uuids::uuid>()(key.templateArgumentTypes[i]->TypeId());
 70         }
 71         return x;
 72     }
 73 };

 88 struct ArrayKeyHash 
 89 {
 90     size_t operator()(const ArrayKey& key) const
 91     {
 92         size_t x = boost::hash<boost::uuids::uuid>()(key.elementType->TypeId());
 93         x = x ^ std::hash<int64_t>()(key.size);
 94         return x;
 95     }
