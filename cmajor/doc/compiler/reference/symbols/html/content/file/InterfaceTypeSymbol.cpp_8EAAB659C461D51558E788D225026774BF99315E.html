 42 void InterfaceTypeSymbol::SetSpecifiers(Specifiers specifiers)
 43 {
 44     Specifiers accessSpecifiers = specifiers & Specifiers::access_;
 45     SetAccess(accessSpecifiers);
 46     if ((specifiers & Specifiers::static_) != Specifiers::none)
 47     {
 48         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be static", GetSpan());
 49     }
 50     if ((specifiers & Specifiers::virtual_) != Specifiers::none)
 51     {
 52         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be virtual", GetSpan());
 53     }
 54     if ((specifiers & Specifiers::override_) != Specifiers::none)
 55     {
 56         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be override", GetSpan());
 57     }
 58     if ((specifiers & Specifiers::abstract_) != Specifiers::none)
 59     {
 60         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be abstract", GetSpan());
 61     }
 62     if ((specifiers & Specifiers::inline_) != Specifiers::none)
 63     {
 64         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be inline", GetSpan());
 65     }
 66     if ((specifiers & Specifiers::explicit_) != Specifiers::none)
 67     {
 68         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be explicit", GetSpan());
 69     }
 70     if ((specifiers & Specifiers::external_) != Specifiers::none)
 71     {
 72         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be external", GetSpan());
 73     }
 74     if ((specifiers & Specifiers::suppress_) != Specifiers::none)
 75     {
 76         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be suppressed", GetSpan());
 77     }
 78     if ((specifiers & Specifiers::default_) != Specifiers::none)
 79     {
 80         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be default", GetSpan());
 81     }
 82     if ((specifiers & Specifiers::constexpr_) != Specifiers::none)
 83     {
 84         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be constexpr", GetSpan());
 85     }
 86     if ((specifiers & Specifiers::cdecl_) != Specifiers::none)
 87     {
 88         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be cdecl", GetSpan());
 89     }
 90     if ((specifiers & Specifiers::nothrow_) != Specifiers::none)
 91     {
 92         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be nothrow", GetSpan());
 93     }
 94     if ((specifiers & Specifiers::throw_) != Specifiers::none)
 95     {
 96         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be throw", GetSpan());
 97     }
 98     if ((specifiers & Specifiers::new_) != Specifiers::none)
 99     {
100         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be new", GetSpan());
101     }
102     if ((specifiers & Specifiers::const_) != Specifiers::none)
103     {
104         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be const", GetSpan());
105     }
106     if ((specifiers & Specifiers::unit_test_) != Specifiers::none)
107     {
108         throw Exception(GetRootModuleForCurrentThread(), "interface cannot be unit_test", GetSpan());
109     }

135 void InterfaceTypeSymbol::GenerateCall(Emitter& emitter, std::std::vector<GenObject*>&genObjects, OperationFlagsflags, MemberFunctionSymbol*interfaceMemberFunction, constSpan&span)
136 {
137     TypeSymbol* type = static_cast<TypeSymbol*>(genObjects[0]->GetType());
138     if (type->GetSymbolType() == SymbolType::interfaceTypeSymbol)
139     {
140         genObjects[0]->Load(emitter, OperationFlags::addr);
141     }
142     else
143     {
144         genObjects[0]->Load(emitter, OperationFlags::none);
145     }
146     void* interfaceTypePtr = emitter.Stack().Pop();
147     void* objectPtr = emitter.GetObjectFromInterface(interfaceTypePtr);
148     void* imtPtr = emitter.GetImtPtrFromInterface(interfaceTypePtr);
149     void* callee = emitter.GetInterfaceMethod(imtPtr, interfaceMemberFunction->ImtIndex(), interfaceMemberFunction->IrType(emitter));
150     int na = genObjects.size();
151     for (int i = 1; i < na; ++i)
152     {
153         GenObject* genObject = genObjects[i];
154         genObject->Load(emitter, flags & OperationFlags::functionCallFlags);
155     }
156     emitter.SetCurrentDebugLocation(span);
157     std::vector<void*> args;
158     int n = interfaceMemberFunction->Parameters().size();
159     if (interfaceMemberFunction->ReturnsClassInterfaceOrClassDelegateByValue())
160     {
161         ++n;
162     }
163     args.resize(n);
164     args[0] = objectPtr;
165     for (int i = 0; i < n - 1; ++i)
166     {
167         void* arg = emitter.Stack().Pop();
168         args[n - i - 1] = arg;
169     }
170     void* handlerBlock = emitter.HandlerBlock();
171     void* cleanupBlock = emitter.CleanupBlock();
172     bool newCleanupNeeded = emitter.NewCleanupNeeded();
173     Pad* currentPad = emitter.CurrentPad();
174     std::vector<void*> bundles;
175     if (currentPad != nullptr)
176     {
177         bundles.push_back(currentPad->value);
178     }
179     if (interfaceMemberFunction->ReturnType()->GetSymbolType() != SymbolType::voidTypeSymbol && !interfaceMemberFunction->ReturnsClassInterfaceOrClassDelegateByValue())
180     {
181         if (IsNothrow() || (!handlerBlock && !cleanupBlock && !newCleanupNeeded))
182         {
183             if (currentPad == nullptr)
184             {
185                 emitter.Stack().Push(emitter.CreateCall(callee, args));
186             }
187             else
188             {
189                 emitter.Stack().Push(emitter.CreateCallInst(callee, args, bundles, span));
190             }
191         }
192         else
193         {
194             void* nextBlock = nullptr;
195             if (GetBackEnd() == BackEnd::llvm || GetBackEnd() == BackEnd::cmcpp)
196             {
197                 nextBlock = emitter.CreateBasicBlock("next");
198             }
199             if (newCleanupNeeded)
200             {
201                 emitter.CreateCleanup();
202                 cleanupBlock = emitter.CleanupBlock();
203             }
204             void* unwindBlock = cleanupBlock;
205             if (unwindBlock == nullptr)
206             {
207                 unwindBlock = handlerBlock;
208                 Assert(unwindBlock, "no unwind block");
209             }
210             if (currentPad == nullptr)
211             {
212                 emitter.Stack().Push(emitter.CreateInvoke(callee, nextBlock, unwindBlock, args));
213             }
214             else
215             {
216                 emitter.Stack().Push(emitter.CreateInvokeInst(callee, nextBlock, unwindBlock, args, bundles, span));
217             }
218             if (GetBackEnd() == BackEnd::llvm || GetBackEnd() == BackEnd::cmcpp)
219             {
220                 emitter.SetCurrentBasicBlock(nextBlock);
221             }
222         }
223     }
224     else
225     {
226         if (IsNothrow() || (!handlerBlock && !cleanupBlock && !newCleanupNeeded))
227         {
228             if (currentPad == nullptr)
229             {
230                 emitter.CreateCall(callee, args);
231             }
232             else
233             {
234                 emitter.CreateCallInst(callee, args, bundles, span);
235             }
236         }
237         else
238         {
239             void* nextBlock = emitter.CreateBasicBlock("next");
240             if (newCleanupNeeded)
241             {
242                 emitter.CreateCleanup();
243                 cleanupBlock = emitter.CleanupBlock();
244             }
245             void* unwindBlock = cleanupBlock;
246             if (unwindBlock == nullptr)
247             {
248                 unwindBlock = handlerBlock;
249                 Assert(unwindBlock, "no unwind block");
250             }
251             if (currentPad == nullptr)
252             {
253                 emitter.CreateInvoke(callee, nextBlock, unwindBlock, args);
254             }
255             else
256             {
257                 emitter.CreateInvokeInst(callee, nextBlock, unwindBlock, args, bundles, span);
258             }
259             emitter.SetCurrentBasicBlock(nextBlock);
260         }
261     }
262 }

469 void ClassToInterfaceConversion::GenerateCall(Emitter& emitter, std::std::vector<GenObject*>&genObjects, OperationFlagsflags, constSpan&span)
470 {
471     void* classPtr = emitter.Stack().Pop();
472     void* classPtrAsVoidPtr = emitter.CreateBitCast(classPtr, emitter.GetIrTypeForVoidPtrType());
473     genObjects[0]->Load(emitter, OperationFlags::addr);
474     void* temporaryInterfaceObjectVar = emitter.Stack().Pop();
475     void* objectPtr = emitter.GetObjectPtrFromInterface(temporaryInterfaceObjectVar);
476     emitter.CreateStore(classPtrAsVoidPtr, objectPtr);
477     void* vmtObjectPtr = sourceClassType->VmtObject(emitter, false);
478     void* imtArray = emitter.GetImtArray(vmtObjectPtr, GetImtsVmtIndexOffset());
479     void* imt = emitter.GetImt(imtArray, interfaceIndex);
480     void* imtPtr = emitter.GetImtPtrFromInterface(temporaryInterfaceObjectVar);
481     emitter.CreateStore(imt, imtPtr);
482     emitter.Stack().Push(temporaryInterfaceObjectVar);
483 }
