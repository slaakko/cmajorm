 30 void TypedefSymbol::Read(SymbolReader& reader)
 31 {
 32     Symbol::Read(reader);
 33     boost::uuids::uuid typeId;
 34     reader.GetBinaryReader().ReadUuid(typeId);
 35     reader.GetSymbolTable()->EmplaceTypeRequest(reader, this, typeId, 0);
 36 }

 65 std::string TypedefSymbol::Syntax() const
 66 {
 67     std::string syntax = GetSpecifierStr();
 68     if (!syntax.empty())
 69     {
 70         syntax.append(1, ' ');
 71     }
 72     syntax.append("typedef ");
 73     syntax.append(ToUtf8(GetType()->DocName()));
 74     syntax.append(1, ' ');
 75     syntax.append(ToUtf8(DocName()));
 76     syntax.append(1, ';');
 77     return syntax;
 78 }

 80 void TypedefSymbol::SetSpecifiers(Specifiers specifiers)
 81 {
 82     Specifiers accessSpecifiers = specifiers & Specifiers::access_;
 83     SetAccess(accessSpecifiers);
 84     if ((specifiers & Specifiers::static_) != Specifiers::none)
 85     {
 86         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be static", GetSpan());
 87     }
 88     if ((specifiers & Specifiers::virtual_) != Specifiers::none)
 89     {
 90         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be virtual", GetSpan());
 91     }
 92     if ((specifiers & Specifiers::override_) != Specifiers::none)
 93     {
 94         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be override", GetSpan());
 95     }
 96     if ((specifiers & Specifiers::abstract_) != Specifiers::none)
 97     {
 98         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be abstract", GetSpan());
 99     }
100     if ((specifiers & Specifiers::inline_) != Specifiers::none)
101     {
102         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be inline", GetSpan());
103     }
104     if ((specifiers & Specifiers::explicit_) != Specifiers::none)
105     {
106         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be explicit", GetSpan());
107     }
108     if ((specifiers & Specifiers::external_) != Specifiers::none)
109     {
110         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be external", GetSpan());
111     }
112     if ((specifiers & Specifiers::suppress_) != Specifiers::none)
113     {
114         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be suppressed", GetSpan());
115     }
116     if ((specifiers & Specifiers::default_) != Specifiers::none)
117     {
118         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be default", GetSpan());
119     }
120     if ((specifiers & Specifiers::constexpr_) != Specifiers::none)
121     {
122         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be constexpr", GetSpan());
123     }
124     if ((specifiers & Specifiers::cdecl_) != Specifiers::none)
125     {
126         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be cdecl", GetSpan());
127     }
128     if ((specifiers & Specifiers::nothrow_) != Specifiers::none)
129     {
130         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be nothrow", GetSpan());
131     }
132     if ((specifiers & Specifiers::throw_) != Specifiers::none)
133     {
134         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be throw", GetSpan());
135     }
136     if ((specifiers & Specifiers::new_) != Specifiers::none)
137     {
138         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be new", GetSpan());
139     }
140     if ((specifiers & Specifiers::const_) != Specifiers::none)
141     {
142         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be const", GetSpan());
143     }
144     if ((specifiers & Specifiers::unit_test_) != Specifiers::none)
145     {
146         throw Exception(GetRootModuleForCurrentThread(), "typedef cannot be unit_test", GetSpan());
147     }
148 }

150 std::std::unique_ptr<sngxml::dom::Element>TypedefSymbol::CreateDomElement(TypeMap&typeMap)
151 {
152     std::unique_ptr<sngxml::dom::Element> element(new sngxml::dom::Element(U"TypedefSymbol"));
153     if (type)
154     {
155         std::unique_ptr<sngxml::dom::Element> typeElement(new sngxml::dom::Element(U"type"));
156         int typeId = typeMap.GetOrInsertType(type);
157         typeElement->SetAttribute(U"ref", U"type_" + ToUtf32(std::to_string(typeId)));
158         element->AppendChild(std::unique_ptr<sngxml::dom::Node>(typeElement.release()));
159     }
160     return element;
161 }
