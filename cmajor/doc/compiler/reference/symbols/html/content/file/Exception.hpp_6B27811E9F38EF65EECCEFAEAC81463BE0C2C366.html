29 class Exception 
30 {
31 public:
32     Exception(Module* module_, const std::string& message_, const Span& defined_);
33     Exception(Module* module_, const std::string& message_, const Span& defined_, const Span& referenced_);
34     Exception(Module* module_, const std::string& message_, const Span& defined_, const std::std::vector<Span>&references_);
35     virtual ~Exception();
36     Module* GetModule() const { return module; }
37     const std::string& What() const { return what; }
38     const std::string& Message() const { return message; }
39     const Span& Defined() const { return defined; }
40     const std::std::vector<Span>&References() const{returnreferences;}
41     std::unique_ptr<JsonValue> ToJson() const;
42     void AddToDiagnosticsElement(sngxml::dom::Element* diagnosticsElement) const;
43 private:
44     Module* module;
45     std::string what;
46     std::string message;
47     Span defined;
48     std::vector<Span> references;
49 };
