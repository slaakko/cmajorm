 21 BasicTypeSymbol::BasicTypeSymbol(SymbolType symbolType_, const Span& span_, const std::u32string& name_) :
 22     TypeSymbol(symbolType_, span_, name_), 
 23     defaultConstructor(nullptr), copyConstructor(nullptr), moveConstructor(nullptr), copyAssignment(nullptr), moveAssignment(nullptr), returnFun(nullptr), equalityOp(nullptr)
 24 {
 25 }

 27 void BasicTypeSymbol::Write(SymbolWriter& writer)
 28 {
 29     TypeSymbol::Write(writer);
 30     boost::uuids::uuid defaultConstructorId = boost::uuids::nil_generator()();
 31     if (defaultConstructor)
 32     {
 33         defaultConstructorId = defaultConstructor->FunctionId();
 34     }
 35     writer.GetBinaryWriter().Write(defaultConstructorId);
 36     boost::uuids::uuid copyConstructorId = boost::uuids::nil_generator()();
 37     if (copyConstructor)
 38     {
 39         copyConstructorId = copyConstructor->FunctionId();
 40     }
 41     writer.GetBinaryWriter().Write(copyConstructorId);
 42     boost::uuids::uuid moveConstructorId = boost::uuids::nil_generator()();
 43     if (moveConstructor)
 44     {
 45         moveConstructorId = moveConstructor->FunctionId();
 46     }
 47     writer.GetBinaryWriter().Write(moveConstructorId);
 48     boost::uuids::uuid copyAssignmentId = boost::uuids::nil_generator()();
 49     if (copyAssignment)
 50     {
 51         copyAssignmentId = copyAssignment->FunctionId();
 52     }
 53     writer.GetBinaryWriter().Write(copyAssignmentId);
 54     boost::uuids::uuid moveAssignmentId = boost::uuids::nil_generator()();
 55     if (moveAssignment)
 56     {
 57         moveAssignmentId = moveAssignment->FunctionId();
 58     }
 59     writer.GetBinaryWriter().Write(moveAssignmentId);
 60     boost::uuids::uuid returnId = boost::uuids::nil_generator()();
 61     if (returnFun)
 62     {
 63         returnId = returnFun->FunctionId();
 64     }
 65     writer.GetBinaryWriter().Write(returnId);
 66     boost::uuids::uuid equalityOpId = boost::uuids::nil_generator()();
 67     if (equalityOp)
 68     {
 69         equalityOpId = equalityOp->FunctionId();
 70     }
 71     writer.GetBinaryWriter().Write(equalityOpId);
 72 }

 74 void BasicTypeSymbol::Read(SymbolReader& reader)
 75 {
 76     TypeSymbol::Read(reader);
 77     boost::uuids::uuid defaultConstructorId;
 78     reader.GetBinaryReader().ReadUuid(defaultConstructorId);
 79     if (!defaultConstructorId.is_nil())
 80     {
 81         reader.GetSymbolTable()->EmplaceFunctionRequest(reader, this, defaultConstructorId, 0);
 82     }
 83     boost::uuids::uuid copyConstructorId;
 84     reader.GetBinaryReader().ReadUuid(copyConstructorId);
 85     if (!copyConstructorId.is_nil())
 86     {
 87         reader.GetSymbolTable()->EmplaceFunctionRequest(reader, this, copyConstructorId, 1);
 88     }
 89     boost::uuids::uuid moveConstructorId;
 90     reader.GetBinaryReader().ReadUuid(moveConstructorId);
 91     if (!moveConstructorId.is_nil())
 92     {
 93         reader.GetSymbolTable()->EmplaceFunctionRequest(reader, this, moveConstructorId, 2);
 94     }
 95     boost::uuids::uuid copyAssignmentId;
 96     reader.GetBinaryReader().ReadUuid(copyAssignmentId);
 97     if (!copyAssignmentId.is_nil())
 98     {
 99         reader.GetSymbolTable()->EmplaceFunctionRequest(reader, this, copyAssignmentId, 3);
100     }
101     boost::uuids::uuid moveAssignmentId;
102     reader.GetBinaryReader().ReadUuid(moveAssignmentId);
103     if (!moveAssignmentId.is_nil())
104     {
105         reader.GetSymbolTable()->EmplaceFunctionRequest(reader, this, moveAssignmentId, 4);
106     }
107     boost::uuids::uuid returnId;
108     reader.GetBinaryReader().ReadUuid(returnId);
109     if (!returnId.is_nil())
110     {
111         reader.GetSymbolTable()->EmplaceFunctionRequest(reader, this, returnId, 5);
112     }
113     boost::uuids::uuid equalityOpId;
114     reader.GetBinaryReader().ReadUuid(equalityOpId);
115     if (!equalityOpId.is_nil())
116     {
117         reader.GetSymbolTable()->EmplaceFunctionRequest(reader, this, equalityOpId, 6);
118     }
119 }

121 void BasicTypeSymbol::EmplaceFunction(FunctionSymbol* functionSymbol, int index)
122 {
123     switch (index)
124     {
125         case 0: defaultConstructor = functionSymbol; break;
126         case 1: copyConstructor = functionSymbol; break;
127         case 2: moveConstructor = functionSymbol; break;
128         case 3: copyAssignment = functionSymbol; break;
129         case 4: moveAssignment = functionSymbol; break;
130         case 5: returnFun = functionSymbol; break;
131         case 6: equalityOp = functionSymbol; break;
132         default:
133         {
134             Assert(false, "invalid emplace function index");
135         }
136     }
137 }

153 void BasicTypeSymbol::Check()
154 {
155     TypeSymbol::Check();
156     if (!defaultConstructor && !IsVoidType())
157     {
158         throw SymbolCheckException(GetRootModuleForCurrentThread(), "basic type symbol has no default constructor", GetSpan());
159     }
160     if (!copyConstructor && !IsVoidType())
161     {
162         throw SymbolCheckException(GetRootModuleForCurrentThread(), "basic type symbol has no copy constructor", GetSpan());
163     }
164     if (!moveConstructor && !IsVoidType())
165     {
166         throw SymbolCheckException(GetRootModuleForCurrentThread(), "basic type symbol has no move constructor", GetSpan());
167     }
168     if (!copyAssignment && !IsVoidType())
169     {
170         throw SymbolCheckException(GetRootModuleForCurrentThread(), "basic type symbol has no copy assignment", GetSpan());
171     }
172     if (!moveAssignment && !IsVoidType())
173     {
174         throw SymbolCheckException(GetRootModuleForCurrentThread(), "basic type symbol has no move assignment", GetSpan());
175     }
176     if (!returnFun && !IsVoidType())
177     {
178         throw SymbolCheckException(GetRootModuleForCurrentThread(), "basic type symbol has no return function", GetSpan());
179     }
180     if (!equalityOp && !IsVoidType())
181     {
182         throw SymbolCheckException(GetRootModuleForCurrentThread(), "basic type symbol has no equality comparison operation", GetSpan());
183     }
184 }
