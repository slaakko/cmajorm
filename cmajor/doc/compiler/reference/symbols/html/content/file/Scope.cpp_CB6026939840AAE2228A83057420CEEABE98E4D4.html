 32 ContainerScope* ContainerScope::BaseScope() const

 34     if (container)
 35     {
 36         if (container->GetSymbolType() == SymbolType::classTypeSymbol || container->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
 37         {
 38             ClassTypeSymbol* cls = static_cast<ClassTypeSymbol*>(container);
 39             ClassTypeSymbol* baseClass = cls->BaseClass();
 40             if (baseClass)
 41             {
 42                 return baseClass->GetContainerScope();
 43             }
 44         }
 45     }

 49 ContainerScope* ContainerScope::ParentScope() const
 50 {
 51     if (parentScope)
 52     {
 53         return parentScope;
 54     }
 55     if (container)
 56     {
 57         Symbol* parent = nullptr;
 58         if (container->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
 59         {
 60             ClassTemplateSpecializationSymbol* specialization = static_cast<ClassTemplateSpecializationSymbol*>(container);
 61             ClassTypeSymbol* classTemplate = specialization->GetClassTemplate();
 62             parent = classTemplate->Parent();
 63         }
 64         else
 65         {
 66             parent = container->Parent();
 67         }
 68         if (parent)
 69         {
 70             if (!parent->GetModule()->IsRootModule() && parent->GetSymbolType() == SymbolType::namespaceSymbol)
 71             {
 72                 NamespaceSymbol* ns = static_cast<NamespaceSymbol*>(parent);
 73                 Module* rootModule = GetRootModuleForCurrentThread();
 74                 NamespaceSymbol* mappedNs = rootModule->GetSymbolTable().GetMappedNs(ns);
 75                 if (mappedNs)
 76                 {
 77                     return mappedNs->GetContainerScope();
 78                 }
 79             }
 80             return parent->GetContainerScope();
 81         }
 82     }
 83     return nullptr;
 84 }

 86 void ContainerScope::Install(Symbol* symbol)
 87 {
 88     auto it = symbolMap.find(symbol->Name());
 89     if (symbol->GetSymbolType() != SymbolType::namespaceSymbol && 
 90         symbol->GetSymbolType() != SymbolType::declarationBlock && 
 91         it != symbolMap.cend())
 92     {
 93         Symbol* prev = it->second;
 94         if (prev != symbol)
 95         {
 96             const Span& defined = symbol->GetSpan();
 97             const Span& referenced = prev->GetSpan();
 98             throw Exception(GetRootModuleForCurrentThread(), "symbol '" + ToUtf8(symbol->Name()) + "' already defined", defined, referenced);
 99         }
100     }
101     else
102     {
103         symbolMap[symbol->Name()] = symbol;
104     }
105 }

112 int CountQualifiedNameComponents(const std::u32string& qualifiedName)
113 {
114     int numComponents = 0;
115     int componentSize = 0;
116     int state = 0;
117     int angleBracketCount = 0;
118     for (char32_t c : qualifiedName)
119     {
120         switch (state)
121         {
122             case 0:
123             {
124                 if (c == '.')
125                 {
126                     ++numComponents;
127                     componentSize = 0;
128                 }
129                 else if (c == '<')
130                 {
131                     ++componentSize;
132                     angleBracketCount = 1;
133                     state = 1;
134                 }
135                 else
136                 {
137                     ++componentSize;
138                 }
139                 break;
140             }
141             case 1:
142             {
143                 ++componentSize;
144                 if (c == '<')
145                 {
146                     ++angleBracketCount;
147                 }
148                 else if (c == '>')
149                 {
150                     --angleBracketCount;
151                     if (angleBracketCount == 0)
152                     {
153                         state = 0;
154                     }
155                 }
156                 break;
157             }
158         }
159     }
160     if (componentSize > 0)
161     {
162         ++numComponents;
163     }
164     return numComponents;
165 }

167 std::std::vector<std::u32string>ParseQualifiedName(conststd::u32string&qualifiedName)
168 {
169     std::vector<std::u32string> components;
170     int state = 0;
171     std::u32string component;
172     int angleBracketCount = 0;
173     for (char32_t c : qualifiedName)
174     {
175         switch (state)
176         {
177             case 0:
178             {
179                 if (c == '.')
180                 {
181                     components.push_back(component);
182                     component.clear();
183                 }
184                 else if (c == '<')
185                 {
186                     component.append(1, c);
187                     angleBracketCount = 1;
188                     state = 1;
189                 }
190                 else
191                 {
192                     component.append(1, c);
193                 }
194                 break;
195             }
196             case 1:
197             {
198                 component.append(1, c);
199                 if (c == '<')
200                 {
201                     ++angleBracketCount;
202                 }
203                 else if (c == '>')
204                 {
205                     --angleBracketCount;
206                     if (angleBracketCount == 0)
207                     {
208                         state = 0;
209                     }
210                 }
211                 break;
212             }
213         }
214     }
215     if (!component.empty())
216     {
217         components.push_back(component);
218     }
219     return components;
220 }

222 Symbol* ContainerScope::Lookup(const std::u32string& name, ScopeLookup lookup) const
223 {
224     int numQualifiedNameComponents = CountQualifiedNameComponents(name);
225     if (numQualifiedNameComponents > 1)
226     {
227         std::vector<std::u32string> components = ParseQualifiedName(name);
228         return LookupQualified(components, lookup);
229     }
230     else
231     {
232         auto it = symbolMap.find(name);
233         if (it != symbolMap.end())
234         {
235             Symbol* s = it->second;
236             return s;
237         }
238         if ((lookup & ScopeLookup::base) != ScopeLookup::none)
239         {
240             ContainerScope* baseScope = BaseScope();
241             if (baseScope)
242             {
243                 Symbol* s = baseScope->Lookup(name, lookup);
244                 if (s)
245                 {
246                     return s;
247                 }
248             }
249         }
250         if ((lookup & ScopeLookup::parent) != ScopeLookup::none)
251         {
252             ContainerScope* parentScope = ParentScope();
253             if (parentScope)
254             {
255                 Symbol* s = parentScope->Lookup(name, lookup);
256                 if (s)
257                 {
258                     return s;
259                 }
260             }
261         }
262         return nullptr;
263     }

266 Symbol* ContainerScope::LookupQualified(const std::std::vector<std::u32string>&components, ScopeLookuplookup) const
267 {
268     const ContainerScope* scope = this;
269     Symbol* s = nullptr;
270     int n = int(components.size());
271     bool allComponentsMatched = true;
272     for (int i = 0; i < n; ++i)
273     {
274         const std::u32string& component = components[i];
275         if (scope)
276         {
277             s = scope->Lookup(component, ScopeLookup::this_);
278             if (s)
279             {
280                 scope = s->GetContainerScope();
281             }
282             else
283             {
284                 allComponentsMatched = false;
285             }
286         }
287     }
288     if (!s || !allComponentsMatched)
289     {
290         if ((lookup & ScopeLookup::parent) != ScopeLookup::none)
291         {
292             ContainerScope* parentScope = ParentScope();
293             if (parentScope)
294             {
295                 return parentScope->LookupQualified(components, lookup);
296             }
297             else
298             {
299                 return nullptr;
300             }
301         }
302     }
303     return s;
304 }

321 NamespaceSymbol* ContainerScope::CreateNamespace(const std::u32string& qualifiedNsName, const Span& span)
322 {
323     ContainerScope* scope = this;
324     NamespaceSymbol* parentNs = scope->Ns();
325     std::vector<std::u32string> components = Split(qualifiedNsName, '.');
326     for (const std::u32string& component : components)
327     {
328         Symbol* s = scope->Lookup(component);
329         if (s)
330         {
331             if (s->GetSymbolType() == SymbolType::namespaceSymbol)
332             {
333                 scope = s->GetContainerScope();
334                 parentNs = scope->Ns();
335             }
336             else
337             {
338                 throw Exception(GetRootModuleForCurrentThread(), "symbol '" + ToUtf8(s->Name()) + "' does not denote a namespace", s->GetSpan());
339             }
340         }
341         else
342         {
343             NamespaceSymbol* newNs = new NamespaceSymbol(span, component);
344             newNs->SetModule(container->GetModule());
345             scope = newNs->GetContainerScope();
346             parentNs->AddMember(newNs);
347             parentNs = newNs;
348         }
349     }
350     return parentNs;
351 }

353 void ContainerScope::CollectViableFunctions(int arity, const std::u32string& groupName, std::std::unordered_set<ContainerScope*>&scopesLookedUp, ScopeLookupscopeLookup, 
354     ViableFunctionSet& viableFunctions, Module* module)
355 {
356     if ((scopeLookup & ScopeLookup::this_) != ScopeLookup::none)
357     {
358         if (scopesLookedUp.find(this) == scopesLookedUp.end())
359         {
360             scopesLookedUp.insert(this);
361             Symbol* symbol = Lookup(groupName);
362             if (symbol)
363             {
364                 if (symbol->GetSymbolType() == SymbolType::functionGroupSymbol)
365                 {
366                     FunctionGroupSymbol* functionGroupSymbol = static_cast<FunctionGroupSymbol*>(symbol);
367                     functionGroupSymbol->CollectViableFunctions(arity, viableFunctions, module);
368                 }
369             }
370         }
371     }
372     if ((scopeLookup & ScopeLookup::base) != ScopeLookup::none)
373     {
374         ContainerScope* baseScope = BaseScope();
375         if (baseScope)
376         {
377             baseScope->CollectViableFunctions(arity, groupName, scopesLookedUp, scopeLookup, viableFunctions, module);
378         }
379     }
380     if ((scopeLookup & ScopeLookup::parent) != ScopeLookup::none)
381     {
382         ContainerScope* parentScope = ParentScope();
383         if (parentScope)
384         {
385             parentScope->CollectViableFunctions(arity, groupName, scopesLookedUp, scopeLookup, viableFunctions, module);
386         }
387     }
388 }

421     {
422         Assert(containerScope, "container scope is null");
423         std::u32string importedNamespaceName = namespaceImportNode->Ns()->Str();

438             }
439         }
440         else
441         {
442             throw Exception(module, "referred namespace symbol '" + ToUtf8(namespaceImportNode->Ns()->Str()) + "' not found", namespaceImportNode->Ns()->GetSpan());
443         }
444     }

456 Symbol* FileScope::Lookup(const std::u32string& name, ScopeLookup lookup) const
457 {
458     if (lookup != ScopeLookup::this_)
459     {
460         throw std::runtime_error("file scope supports only this scope lookup");
461     }
462     std::unordered_set<Symbol*> foundSymbols;
463     auto it = aliasSymbolMap.find(name);
464     if (it != aliasSymbolMap.cend())
465     {
466         Symbol* symbol = it->second;
467         foundSymbols.insert(symbol);
468     }
469     else
470     {
471         for (ContainerScope* containerScope : containerScopes)
472         {
473             Symbol* symbol = containerScope->Lookup(name, ScopeLookup::this_);
474             if (symbol)
475             {
476                 foundSymbols.insert(symbol);
477             }
478         }
479     }
480     if (foundSymbols.empty())
481     {
482         return nullptr;
483     }
484     else if (foundSymbols.size() > 1)
485     {
486         std::string message("reference to object '" + ToUtf8(name) + "' is ambiguous: ");
487         bool first = true;
488         Span span;
489         for (Symbol* symbol : foundSymbols)
490         {
491             if (first)
492             {
493                 first = false;
494                 span = symbol->GetSpan();
495             }
496             else
497             {
498                 message.append(" or ");
499             }
500             message.append(ToUtf8(symbol->FullName()));
501         }
502         throw Exception(module, message, span);
503     }
504     else
505     {
506         return *foundSymbols.begin();
507     }
508 }
