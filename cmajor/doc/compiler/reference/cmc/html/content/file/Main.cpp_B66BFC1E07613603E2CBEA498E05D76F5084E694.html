 46 struct BackendSelector 
 47 {
 48     BackendSelector(cmajor::symbols::BackEnd backend)
 49     {
 50         switch (backend)
 51         {
 52             case cmajor::symbols::BackEnd::llvm:
 53             {
 54                 CmmInit(cmajor::mid::BackEndKind::llvmBackEnd);
 55                 break;
 56             }
 57             case cmajor::symbols::BackEnd::cmsx:
 58             {
 59                 CmmInit(cmajor::mid::BackEndKind::cmsxBackEnd);
 60                 break;
 61             }
 62         }
 63     }
 64     ~BackendSelector()
 65     {
 66         CmmDone();
 67     }
