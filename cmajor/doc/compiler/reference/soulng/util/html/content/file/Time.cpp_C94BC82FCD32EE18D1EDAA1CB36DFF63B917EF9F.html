 17 std::string Date::ToString(bool omitDashes) const
 18 {
 19     std::string date;
 20     date.append(1, static_cast<char>(static_cast<int16_t>('0') + ((year / 1000) % 10)));
 21     date.append(1, static_cast<char>(static_cast<int16_t>('0') + ((year / 100) % 10)));
 22     date.append(1, static_cast<char>(static_cast<int16_t>('0') + ((year / 10) % 10)));
 23     date.append(1, static_cast<char>(static_cast<int16_t>('0') + (year % 10)));
 24     if (!omitDashes)
 25     {
 26         date.append(1, '-');
 27     }
 28     date.append(1, static_cast<char>(static_cast<int8_t>('0') + ((static_cast<int8_t>(month) / 10) % 10)));
 29     date.append(1, static_cast<char>(static_cast<int8_t>('0') + (static_cast<int8_t>(month) % 10)));
 30     if (!omitDashes)
 31     {
 32         date.append(1, '-');
 33     }
 34     date.append(1, static_cast<char>(static_cast<int8_t>('0') + ((day / 10) % 10)));
 35     date.append(1, static_cast<char>(static_cast<int8_t>('0') + (day % 10)));
 36     return date;
 37 }

 53 std::string DateTime::ToString(bool omitDashes, bool omitColons, bool omitMins, bool omitSecs) const
 54 {
 55     std::string dateTime;
 56     dateTime.append(date.ToString(omitDashes));
 57     dateTime.append(1, 'T');
 58     int32_t hh = Hours() % 24;
 59     int32_t mm = Minutes() % 60;
 60     int32_t ss = Seconds() % 60;
 61     dateTime.append(1, static_cast<char>(static_cast<int32_t>('0') + ((hh / 10) % 10)));
 62     dateTime.append(1, static_cast<char>(static_cast<int32_t>('0') + (hh % 10)));
 63     if (!omitMins)
 64     {
 65         if (!omitColons)
 66         {
 67             dateTime.append(1, ':');
 68         }
 69         dateTime.append(1, static_cast<char>(static_cast<int32_t>('0') + ((mm / 10) % 10)));
 70         dateTime.append(1, static_cast<char>(static_cast<int>('0') + (mm % 10)));
 71         if (!omitSecs)
 72         {
 73             if (!omitColons)
 74             {
 75                 dateTime.append(1, ':');
 76             }
 77             dateTime.append(1, static_cast<char>(static_cast<int32_t>('0') + ((ss / 10) % 10)));
 78             dateTime.append(1, static_cast<char>(static_cast<int32_t>('0') + (ss % 10)));
 79         }
 80     }
 81     return dateTime;
 82 }

 84 std::string FormatTimeMs(int32_t milliseconds)
 85 {
 86     int32_t hh = milliseconds / 3600000;
 87     int32_t mm = milliseconds / 60000 % 60;
 88     int32_t ss = milliseconds / 1000 % 60;
 89     int32_t ms = milliseconds % 1000;
 90     std::string time;
 91     time.append(1, static_cast<char>(static_cast<int32_t>('0') + ((hh / 10) % 10)));
 92     time.append(1, static_cast<char>(static_cast<int32_t>('0') + (hh % 10)));
 93     time.append(1, ':');
 94     time.append(1, static_cast<char>(static_cast<int32_t>('0') + ((mm / 10) % 10)));
 95     time.append(1, static_cast<char>(static_cast<int>('0') + (mm % 10)));
 96     time.append(1, ':');
 97     time.append(1, static_cast<char>(static_cast<int32_t>('0') + ((ss / 10) % 10)));
 98     time.append(1, static_cast<char>(static_cast<int32_t>('0') + (ss % 10)));
 99     time.append(1, '.');
100     time.append(1, static_cast<char>(static_cast<int32_t>('0') + ((ms / 100) % 10)));
101     time.append(1, static_cast<char>(static_cast<int32_t>('0') + ((ms / 10) % 10)));
102     time.append(1, static_cast<char>(static_cast<int32_t>('0') + (ms % 10)));
103     return time;
104 }

113 }
