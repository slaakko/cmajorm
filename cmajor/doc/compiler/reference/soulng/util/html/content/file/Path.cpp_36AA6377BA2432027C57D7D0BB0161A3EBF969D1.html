117 std::string Path::MakeCanonical(const std::string& path)
118 {
119     bool startsWithDriveLetter = false;
120 #ifdef _WIN32
121 
122 
123 
124 
125 #endif
126     std::string result;
127     char prev = ' ';
128     bool first = true;
129     for (char c : path)
130     {
131         if (first)
132         {
133             first = false;
134             if (startsWithDriveLetter)
135             {
136                 c = std::toupper(static_cast<unsigned char>(c));
137             }
138         }
139         if (c == '\\')
140         {
141             c = '/';
142         }
143         if (c == '/')
144         {
145             if (prev != '/')
146             {
147                 result.append(1, c);
148             }
149         }
150         else
151         {
152             result.append(1, c);
153         }
154         prev = c;
155     }
156     if (result.length() == 3 && std::isalpha(result[0]) && result[1] == ':' && result[2] == '/')
157     {
158         return result;
159     }
160     if (result == "/")
161     {
162         return result;
163     }
164     if (!result.empty())
165     {
166         if (result[result.length() - 1] == '/')
167         {
168             return result.substr(0, result.length() - 1);
169         }
170     }
171     return result;
172 }

174 std::string Path::ChangeExtension(const std::string& path, const std::string& extension)
175 {
176     std::string::size_type lastDotPos = path.rfind('.');
177     if (extension.empty())
178     {
179         if (lastDotPos != std::string::npos)
180         {
181             return path.substr(0, lastDotPos);
182         }
183         else
184         {
185             return path;
186         }
187     }
188     else
189     {
190         if (lastDotPos == std::string::npos)
191         {
192             if (extension[0] == '.')
193             {
194                 return path + extension;
195             }
196             else
197             {
198                 return path + "." + extension;
199             }
200         }
201         else
202         {
203             if (extension[0] == '.')
204             {
205                 return path.substr(0, lastDotPos) + extension;
206             }
207             else
208             {
209                 return path.substr(0, lastDotPos + 1) + extension;
210             }
211         }
212     }

215 bool Path::HasExtension(const std::string& path)
216 {
217     std::string::size_type lastDotPos = path.rfind('.');
218     if (lastDotPos != std::string::npos)
219     {
220         std::string::size_type lastColon = path.find(':', lastDotPos + 1);
221         std::string::size_type lastDirSep = path.find('/', lastDotPos + 1);
222         if (lastColon > lastDotPos || lastDirSep > lastDotPos)
223         {
224             return false;
225         }
226         else if (lastDotPos < path.length() - 1)
227         {
228             return true;
229         }
230         else
231         {
232             return false;
233         }
234     }
235     else
236     {
237         return false;
238     }
239 }

329 {

331     {

335     {

338     else

342             return path2;
343         }
344         else

346             std::string result = path1;
347             if (result[result.length() - 1] != '/')

349                 result.append(1, '/');
350             }
351             result.append(path2);
352             return result;
353         }

385 std::string GetFullPath(const std::string& path)
386 {
387     std::string p = Path::MakeCanonical(path);
388     if (Path::IsRelative(p))
389     {
390         p = GetCurrentWorkingDirectory();
391         p.append(1, '/');
392         p.append(Path::MakeCanonical(path));
393     }
394     std::vector<std::string> components = Split(p, '/');
395     int w = 0;
396     int n = int(components.size());
397     for (int i = 0; i < n; ++i)
398     {
399         const std::string& c = components[i];
400         if (i == 0 || (!c.empty() && c != "."))
401         {
402             if (c == "..")
403             {
404                 --w;
405                 if (w < 0)
406                 {
407                     throw InvalidPathException("path '" + path + "' is invalid");
408                 }
409             }
410             else
411             {
412                 if (w != i)
413                 {
414                     components[w] = components[i];
415                 }
416                 ++w;
417             }
418         }
419     }
420     if (w == 0)
421     {
422         return "/";
423     }
424     else if (w == 1)
425     {
426         const std::string& p = components[0];
427         if (p.length() == 2 && std::isalpha(p[0]) && p[1] == ':')
428         {
429             return p + "/";
430         }
431     }
432     std::string result;
433     for (int i = 0; i < w; ++i)
434     {
435         if (i != 0)
436         {
437             result.append(1, '/');
438         }
439         result.append(components[i]);
440     }
441     return result;
442 }
