10 bool IsPrime(uint64_t x)
11 {
12     uint64_t i = 3;
13     while (true)
14     {
15         uint64_t q = x / i;
16         if (q < i)
17         {
18             return true;
19         }
20         if (x == q * i)
21         {
22             return false;
23         }
24         i += 2;
25     }
26 }

28 uint64_t NextPrime(uint64_t x)
29 {
30     if (x <= 2)
31     {
32         return 2;
33     }
34     if ((x & 1) == 0)
35     {
36         ++x;
37     }
38     while (!IsPrime(x))
39     {
40         x += 2;
41     }
42     return x;
43 }
