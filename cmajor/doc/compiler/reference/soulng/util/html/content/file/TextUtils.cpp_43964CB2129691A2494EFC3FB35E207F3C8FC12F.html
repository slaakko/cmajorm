 21 std::string Trim(const std::string& s)
 22 {
 23     int b = 0;
 24     while (b < int(s.length()) && std::isspace(s[b])) ++b;
 25     int e = int(s.length()) - 1;
 26     while (e >= b && std::isspace(s[e])) --e;
 27     return s.substr(b, e - b + 1);
 28 }

 30 std::string TrimAll(const std::string& s)
 31 {
 32     std::string result;
 33     result.reserve(s.length());
 34     int state = 0;
 35     std::string::const_iterator e = s.cend();
 36     for (std::string::const_iterator i = s.cbegin(); i != e; ++i)
 37     {
 38         char c = *i;
 39         switch (state)
 40         {
 41             case 0:
 42             {
 43                 if (!std::isspace(c))
 44                 {
 45                     result.append(1, c);
 46                     state = 1;
 47                 }
 48                 break;
 49             }
 50             case 1:
 51             {
 52                 if (std::isspace(c))
 53                 {
 54                     state = 2;
 55                 }
 56                 else
 57                 {
 58                     result.append(1, c);
 59                 }
 60                 break;
 61             }
 62             case 2:
 63             {
 64                 if (!std::isspace(c))
 65                 {
 66                     result.append(1, ' ');
 67                     result.append(1, c);
 68                     state = 1;
 69                 }
 70                 break;
 71             }
 72         }
 73     }
 74     return result;
 75 }

137     int n = int(s.length());

139     {
140         if (s[i] == c)
141         {
142             v.push_back(s.substr(start, i - start));
143             start = i + 1;

167 std::string Replace(const std::string& s, const std::string& oldString, const std::string& newString)
168 {
169     std::string r;
170     std::string::size_type start = 0;
171     std::string::size_type pos = s.find(oldString.c_str(), start);
172     while (pos != std::string::npos)
173     {
174         r.append(s.substr(start, pos - start));
175         r.append(newString);
176         start = pos + oldString.length();
177         pos = s.find(oldString.c_str(), start);
178     }
179     r.append(s.substr(start, s.length() - start));
180     return r;
181 }

379 std::string ToString(double x, int minNumDecimals, int maxNumDecimals)
380 {
381     std::string result;
382     if (x < 0)
383     {
384         x = -x;
385         result.append(1, '-');
386     }
387     result.append(std::to_string(static_cast<int>(x)));
388     double d = x - static_cast<int>(x);
389     if (d > 0 || minNumDecimals > 0)
390     {
391         result.append(1, '.');
392         for (int i = 0; (d > 0 || i < minNumDecimals) && i < maxNumDecimals; ++i)
393         {
394             d = 10 * d;
395             int digit = static_cast<int>(d) % 10;
396             result.append(1, static_cast<char>(static_cast<int>('0') + digit));
397             d = d - static_cast<int>(d);
398         }
399     }
400     return result;
401 }

460 std::u32string FormatNumber(int n, int numDigits)
461 {
462     std::u32string s(numDigits, ' ');
463     int k = numDigits - 1;
464     while (n > 0)
465     {
466         if (k < 0)
467         {
468             throw std::runtime_error("invalid number of digits");

471         s[k] = digit + '0';
472         --k;
473         n = n / 10;
474     }
475     return s;

505 std::string Format(const std::string& s, int width, FormatWidth fw, FormatJustify justify, char fillChar)

508     {

519     {
