 19 void DebugExprBinder::Visit(IdentifierDebugExprNode& node)
 20 {
 21     DIVariable* variable = scope->GetVariable(node.Identifier());
 22     if (variable)
 23     {
 24         InitializationStatus varStatus = InitializationStatus::unknown;
 25         if (variable->GetKind() == DIVariable::Kind::localVariable && variable->GetInitLineNumber() != -1)
 26         {
 27             if (debugger.StoppedInstruction()->GetSourceSpan().line > variable->GetInitLineNumber())
 28             {
 29                 varStatus = InitializationStatus::initialized;
 30             }
 31             else
 32             {
 33                 varStatus = InitializationStatus::uninitialized;
 34             }
 35         }
 36         currentNode.reset(new BoundVariableReferenceNode(variable->GetType(), variable, varStatus, &node));
 37         if (status == InitializationStatus::unknown)
 38         {
 39             status = varStatus;
 40         }
 41     }
 42     else
 43     {
 44         throw std::runtime_error("variable '" + node.Identifier() + "' not found from scope '" + scope->Name() + "'");
 45     }
 46 }

 53 void DebugExprBinder::Visit(AddDebugExprNode& node)
 54 {
 55     node.Left()->Accept(*this);
 56     std::unique_ptr<BoundDebugNode> left(std::move(currentNode));
 57     node.Right()->Accept(*this);
 58     std::unique_ptr<BoundDebugNode> right(std::move(currentNode));
 59     bool valid = false;
 60     if (left->Type()->GetKind() == DIType::Kind::primitiveType)
 61     {
 62         DIPrimitiveType* primitiveType = static_cast<DIPrimitiveType*>(left->Type());
 63         if (primitiveType->IsIntegerType())
 64         {
 65             if (right->Type()->GetKind() == DIType::Kind::pointerType)
 66             {
 67                 currentNode.reset(new BoundAddNode(right->Type(), left.release(), right.release(), &node));
 68                 valid = true;
 69             }
 70         }
 71     }
 72     else if (left->Type()->GetKind() == DIType::Kind::pointerType)
 73     {
 74         if (right->Type()->GetKind() == DIType::Kind::primitiveType)
 75         {
 76             DIPrimitiveType* primitiveType = static_cast<DIPrimitiveType*>(right->Type());
 77             if (primitiveType->IsIntegerType())
 78             {
 79                 DIType* type = left->Type();
 80                 currentNode.reset(new BoundAddNode(type, left.release(), right.release(), &node));
 81                 valid = true;
 82             }
 83         }
 84     }
 85     if (!valid)
 86     {
 87         throw std::runtime_error("add expression not valid: (pointer + integer) | (integer + pointer) expected: " + node.ToString());
 88     }
 89 }

 91 void DebugExprBinder::Visit(SubDebugExprNode& node)
 92 {
 93     node.Left()->Accept(*this);
 94     std::unique_ptr<BoundDebugNode> left(std::move(currentNode));
 95     node.Right()->Accept(*this);
 96     std::unique_ptr<BoundDebugNode> right(std::move(currentNode));
 97     bool valid = false;
 98     if (left->Type()->GetKind() == DIType::Kind::pointerType)
 99     {
100         DIPointerType* leftPtrType = static_cast<DIPointerType*>(left->Type());
101         if (right->Type()->GetKind() == DIType::Kind::pointerType)
102         {
103             DIPointerType* rightPtrType = static_cast<DIPointerType*>(right->Type());
104             if (leftPtrType->Id() == rightPtrType->Id())
105             {
106                 currentNode.reset(new BoundSubNode(debugInfo->GetMainProject()->GetLongType(), left.release(), right.release(), &node));
107                 valid = true;
108             }
109         }
110         else if (right->Type()->GetKind() == DIType::Kind::primitiveType)
111         {
112             DIPrimitiveType* primitiveType = static_cast<DIPrimitiveType*>(right->Type());
113             if (primitiveType->IsIntegerType())
114             {
115                 DIType* type = left->Type();
116                 currentNode.reset(new BoundSubNode(type, left.release(), right.release(), &node));
117                 valid = true;
118             }
119         }
120     }
121     if (!valid)
122     {
123         throw std::runtime_error("subtract expression not valid: (pointer - integer) | (pointer - pointer) expected: " + node.ToString());
124     }
125 }

133 void DebugExprBinder::Visit(DerefDebugExprNode& node)
134 {
135     node.Operand()->Accept(*this);
136     std::unique_ptr<BoundDebugNode> operandNode = std::move(currentNode);
137     DIType* operandType = nullptr;
138     switch (operandNode->Type()->GetKind())
139     {
140         case DIType::Kind::pointerType:
141         {
142             DIPointerType* pointerType = static_cast<DIPointerType*>(operandNode->Type());
143             operandType = pointerType->PointedToType();
144             break;
145         }
146         default:
147         {
148             throw std::runtime_error("cannot dereference an object of a non-pointer type (" + operandNode->Type()->Name() + ")");
149         }
150     }
151     currentNode.reset(new BoundDerefNode(operandType, operandNode.release(), &node));
152 }

162 void DebugExprBinder::Visit(DotDebugExprNode& node)
163 {
164     node.Subject()->Accept(*this);
165     DIType* type = currentNode->Type();
166     Scope* scope = type->GetScope();
167     if (scope)
168     {
169         if (node.Member() == "base")
170         {
171             if (type->GetKind() == DIType::Kind::classType || type->GetKind() == DIType::Kind::specializationType)
172             {
173                 DIClassType* classType = static_cast<DIClassType*>(type);
174                 if (!classType->BaseClassId().is_nil())
175                 {
176                     DIType* dotMemberType = classType->BaseClassType();
177                     std::string gdbExprString = currentNode->GdbExprString() + ".m0";
178                     BoundDotNode* boundDotNode = new BoundDotNode(dotMemberType, currentNode.release(), node.Member(), gdbExprString, &node);
179                     currentNode.reset(boundDotNode);
180                 }
181                 else
182                 {
183                     throw std::runtime_error("class '" + classType->Name() + "' has no base class");
184                 }
185             }
186             else
187             {
188                 throw std::runtime_error("expression '" + currentNode->ToString() + "' has no class scope");
189             }
190         }
191         else
192         {
193             DIVariable* memberVariable = scope->GetVariable(node.Member());
194             if (memberVariable)
195             {
196                 DIType* dotMemberType = memberVariable->GetType();
197                 std::string gdbExprString = currentNode->GdbExprString() + "." + memberVariable->IrName();
198                 BoundDotNode* boundDotNode = new BoundDotNode(dotMemberType, currentNode.release(), node.Member(), gdbExprString, &node);
199                 currentNode.reset(boundDotNode);
200             }
201             else
202             {
203                 throw std::runtime_error("member variable '" + node.Member() + "' not found from scope '" + scope->Name() + "'");
204             }
205         }
206     }
207     else
208     {
209         throw std::runtime_error("expression '" + currentNode->ToString() + "' has no class scope");
210     }
211 }

213 void DebugExprBinder::Visit(SubscriptDebugExprNode& node)
214 {
215     DIType* type = nullptr;
216     node.Subject()->Accept(*this);
217     std::unique_ptr<BoundDebugNode> subject(currentNode.release());
218     DIType* subjectType = subject->Type();
219     node.Index()->Accept(*this);
220     std::unique_ptr<BoundDebugNode> index(currentNode.release());
221     bool indexTypeValid = false;
222     if (index->Type()->GetKind() == DIType::Kind::primitiveType)
223     {
224         DIPrimitiveType* primitiveType = static_cast<DIPrimitiveType*>(index->Type());
225         if (primitiveType->GetPrimitiveTypeKind() == DIPrimitiveType::DIPrimitiveType::Kind::longType)
226         {
227             indexTypeValid = true;
228         }
229     }
230     if (!indexTypeValid)
231     {
232         throw std::runtime_error("index type of subscript expression expected to be 'long'");
233     }
234     switch (subjectType->GetKind())
235     {
236         case DIType::Kind::pointerType:
237         {
238             DIPointerType* pointerType = static_cast<DIPointerType*>(subjectType);
239             type = pointerType->PointedToType();
240             break;
241         }
242         case DIType::Kind::arrayType:
243         {
244             DIArrayType* arrayType = static_cast<DIArrayType*>(subjectType);
245             type = arrayType->ElementType();
246             break;
247         }
248         case DIType::Kind::specializationType:
249         {
250             DIClassTemplateSpecializationType* specializationType = static_cast<DIClassTemplateSpecializationType*>(subjectType);
251             type = GetValueType(specializationType);
252             if (specializationType->GetContainerClassTemplateKind() != ContainerClassTemplateKind::notContainerClassTemplate)
253             {
254                 hasContainerSubscript = true;
255             }
256             break;
257         }
258         default:
259         {
260             throw std::runtime_error("the subject type of a subscript expression must be a pointer, array or container type");
261         }
262     }
263     currentNode.reset(new BoundSubscriptNode(type, subject.release(), index.release(), &node));
264 }

266 void DebugExprBinder::Visit(RangeDebugExprNode& node)
267 {
268     DIType* type = nullptr;
269     node.Subject()->Accept(*this);
270     std::unique_ptr<BoundDebugNode> subject(currentNode.release());
271     DIType* subjectType = subject->Type();
272     node.RangeStart()->Accept(*this);
273     std::unique_ptr<BoundDebugNode> rangeStart(currentNode.release());
274     bool startTypeValid = false;
275     if (rangeStart->Type()->GetKind() == DIType::Kind::primitiveType)
276     {
277         DIPrimitiveType* primitiveType = static_cast<DIPrimitiveType*>(rangeStart->Type());
278         if (primitiveType->GetPrimitiveTypeKind() == DIPrimitiveType::DIPrimitiveType::Kind::longType)
279         {
280             startTypeValid = true;
281         }
282     }
283     if (!startTypeValid)
284     {
285         throw std::runtime_error("start index type of subscript expression expected to be 'long'");
286     }
287     bool rangeEndSet = false;
288     std::unique_ptr<BoundDebugNode> rangeEnd;
289     if (node.RangeEnd()->GetKind() == DebugExprNode::Kind::identifier)
290     {
291         IdentifierDebugExprNode* idNode = static_cast<IdentifierDebugExprNode*>(node.RangeEnd());
292         if (idNode->Identifier() == "count")
293         {
294             rangeEnd.reset(new BoundIntegerLiteralNode(debugInfo->GetMainProject()->GetLongType(), -1, &node));
295             rangeEndSet = true;
296         }
297     }
298     if (!rangeEndSet)
299     {
300         node.RangeEnd()->Accept(*this);
301         rangeEnd.reset(currentNode.release());
302     }
303     bool endTypeValid = false;
304     if (rangeStart->Type()->GetKind() == DIType::Kind::primitiveType)
305     {
306         DIPrimitiveType* primitiveType = static_cast<DIPrimitiveType*>(rangeEnd->Type());
307         if (primitiveType->GetPrimitiveTypeKind() == DIPrimitiveType::DIPrimitiveType::Kind::longType)
308         {
309             endTypeValid = true;
310         }
311     }
312     if (!endTypeValid)
313     {
314         throw std::runtime_error("end index type of subscript expression expected to be 'long'");
315     }
316     switch (subjectType->GetKind())
317     {
318         case DIType::Kind::pointerType:
319         {
320             DIPointerType* pointerType = static_cast<DIPointerType*>(subjectType);
321             type = pointerType->PointedToType();
322             break;
323         }
324         case DIType::Kind::arrayType:
325         {
326             DIArrayType* arrayType = static_cast<DIArrayType*>(subjectType);
327             type = arrayType->ElementType();
328             break;
329         }
330         case DIType::Kind::specializationType:
331         {
332             DIClassTemplateSpecializationType* specializationType = static_cast<DIClassTemplateSpecializationType*>(subjectType);
333             type = GetValueType(specializationType);
334             break;
335         }
336         default:
337         {
338             throw std::runtime_error("the subject type of a subscript expression must be a pointer, array or container type");
339         }
340     }
341     currentNode.reset(new BoundRangeNode(type, subject.release(), rangeStart.release(), rangeEnd.release(), &node));
342 }

354 void DebugExprBinder::Visit(CastDebugExprNode& node)
355 {
356     node.TypeIdNode()->Accept(*this);
357     std::unique_ptr<BoundDebugNode> typeNode = std::move(currentNode);
358     if (!typeNode->IsTypeNode())
359     {
360         throw std::runtime_error("typeid expresssion expected");
361     }
362     node.ExprNode()->Accept(*this);
363     std::unique_ptr<BoundDebugNode> exprNode = std::move(currentNode);
364     std::string gdbExprString;
365     DIType* type = typeNode->Type();
366     gdbExprString = "(" + type->IrName() + ")" + exprNode->GdbExprString();
367     currentNode.reset(new BoundCastNode(typeNode.release(), exprNode.release(), gdbExprString, &node));
368 }

372     expression.reset(new BoundDebugExpression(currentNode.release(), sourceNode, hasContainerSubscript, status));

376 DIType* DebugExprBinder::GetValueType(DIClassTemplateSpecializationType* specializationType) const
377 {
378     if (specializationType->GetContainerClassTemplateKind() != ContainerClassTemplateKind::notContainerClassTemplate)
379     {
380         boost::uuids::uuid valueTypeId = specializationType->ValueTypeId();
381         DIType* valueType = specializationType->GetProject()->GetType(valueTypeId);
382         if (valueType)
383         {
384             return valueType;
385         }
386         else
387         {
388             throw std::runtime_error("value type for container class template specialization '" + specializationType->Name() + "' not found");
389         }
390     }
391     else
392     {
393         throw std::runtime_error("class template specialization '" + specializationType->Name() + "' is not a container class template specialization");
394     }
