 11 void WriteCmdbFileTag(BinaryWriter& writer)
 12 {
 13     writer.Write("CMDB", false);
 14     writer.Write(currentDebugInfoFormat);
 15 }

 17 void ReadCmdbFileTag(BinaryReader& reader)
 18 {
 19     char c;
 20     char m;
 21     char d;
 22     char b;
 23     uint8_t format;
 24     try
 25     {
 26         c = reader.ReadChar();
 27         m = reader.ReadChar();
 28         d = reader.ReadChar();
 29         b = reader.ReadChar();
 30         format = reader.ReadByte();
 31     }
 32     catch (const std::exception& ex;)
 33     {
 34         throw std::runtime_error("invalid CMDB file tag read from '" + reader.FileName() + "': not a CMDB debug information file: " + ex.what());
 35     }
 36     if (c != 'C' || m != 'M' || d != 'D' || b != 'B')
 37     {
 38         throw std::runtime_error("invalid CMDB file tag read from '" + reader.FileName() + "': not a CMDB debug information file");
 39     }
 40     if (format != currentDebugInfoFormat)
 41     {
 42         throw std::runtime_error("wrong CMDB file format version number: format version " + std::string(1, format) + " read, format version " + 
 43             std::string(1, currentDebugInfoFormat) + " expected. Please rebuild project(s) from sources.");
 44     }
 45 }
