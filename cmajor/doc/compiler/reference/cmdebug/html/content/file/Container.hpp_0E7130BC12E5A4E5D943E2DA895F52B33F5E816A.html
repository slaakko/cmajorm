140 class Container 
141 {
142 public:
143     Container(ContainerClassTemplateKind kind_, Debugger& debugger_, uint64_t address_);
144     virtual ~Container();
145     Container(const Container&) = delete;
146     Container(Container&&) = delete;
147     Container& operator=(const Container&) = delete;
148     Container& operator=(Container&&) = delete;
149     virtual void ClearBrowsingData();
150     virtual int64_t Count(const std::string& expr);
151     virtual Iterator* Begin(const std::string& containerVarExpr) = 0;
152     virtual Iterator* End(const std::string& containerVarExpr) = 0;
153     virtual std::unique_ptr<JsonValue> Subscript(const std::string& containerVarExpr, int64_t index) = 0;
154     virtual std::string SubscriptExpressionString(const std::string& containerVarExpr, int64_t index) = 0;
155     std::unique_ptr<JsonValue> Range(const std::string& containerVarExpr, int64_t rangeStart, int64_t rangeEnd);
156     ContainerClassTemplateKind GetKind() const { return kind; }
157     uint64_t Address() const { return address; }
158     Pointer* GetPointer(uint64_t address) const;
159     void AddPointer(Pointer* ptr);
160     Debugger& GetDebugger() { return debugger; }
161     Iterator* GetIterator(int64_t index) const;
162     void AddIterator(Iterator* iterator);
163     int64_t GetCount() { return count; }
164     void SetCount(int64_t count_) { count = count_; }
165     DIType* ValueNodePtrType() const { return valueNodePtrType; }
166     void SetValueNodePtrType(DIType* type) { valueNodePtrType = type; }
167     Pointer* EndPtr() const { return endPtr; }
168     void SetEndPtr(Pointer* endPtr_) { endPtr = endPtr_; }
169 private:
170     ContainerClassTemplateKind kind;
171     Debugger& debugger;
172     uint64_t address;
173     int64_t count;
174     DIType* valueNodePtrType;
175     Pointer* endPtr;
176     std::unordered_map<uint64_t, Pointer*> pointerMap;
177     std::vector<std::std::unique_ptr<Pointer>>pointers;

180 };
