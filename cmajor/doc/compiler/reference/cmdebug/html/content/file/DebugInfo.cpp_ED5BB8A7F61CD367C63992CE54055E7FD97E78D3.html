  91 std::string InstructionFlagsStr(InstructionFlags flags)
  92 {
  93     std::string s;
  94     if ((flags & InstructionFlags::entryCode) != InstructionFlags::none)
  95     {
  96         if (!s.empty())
  97         {
  98             s.append(" | ");
  99         }
 100         s.append("entry");
 101     }
 102     if ((flags & InstructionFlags::exitCode) != InstructionFlags::none)
 103     {
 104         if (!s.empty())
 105         {
 106             s.append(" | ");
 107         }
 108         s.append("exit");
 109     }
 110     if ((flags & InstructionFlags::beginBrace) != InstructionFlags::none)
 111     {
 112         if (!s.empty())
 113         {
 114             s.append(" | ");
 115         }
 116         s.append("beginBrace");
 117     }
 118     if ((flags & InstructionFlags::endBrace) != InstructionFlags::none)
 119     {
 120         if (!s.empty())
 121         {
 122             s.append(" | ");
 123         }
 124         s.append("endBrace");
 125     }
 126     if ((flags & InstructionFlags::startFunction) != InstructionFlags::none)
 127     {
 128         if (!s.empty())
 129         {
 130             s.append(" | ");
 131         }
 132         s.append("startFunction");
 133     }
 134     if ((flags & InstructionFlags::throwInst) != InstructionFlags::none)
 135     {
 136         if (!s.empty())
 137         {
 138             s.append(" | ");
 139         }
 140         s.append("throw");
 141     }
 142     if ((flags & InstructionFlags::catchInst) != InstructionFlags::none)
 143     {
 144         if (!s.empty())
 145         {
 146             s.append(" | ");
 147         }
 148         s.append("catch");
 149     }
 150     return s;
 151 }

 489 void Project::AddType(DIType* type)
 490 {
 491     if (type->GetKind() == DIType::Kind::primitiveType)
 492     {
 493         DIPrimitiveType* primitiveType = static_cast<DIPrimitiveType*>(type);
 494         if (primitiveType->GetPrimitiveTypeKind() == DIPrimitiveType::Kind::longType)
 495         {
 496             longType = primitiveType;
 497         }
 498     }
 499     typeMap[type->Id()] = type;
 500     types.push_back(std::unique_ptr<DIType>(type));
 501     switch (type->GetKind())
 502     {
 503         case DIType::Kind::pointerType:
 504         {
 505             DIPointerType* pointerType = static_cast<DIPointerType*>(type);
 506             pointerTypeMap[pointerType->PointedTypeId()] = pointerType;
 507             break;
 508         }
 509         case DIType::Kind::referenceType:
 510         {
 511             DIReferenceType* referenceType = static_cast<DIReferenceType*>(type);
 512             referenceTypeMap[referenceType->BaseTypeId()] = referenceType;
 513             break;
 514         }
 515         case DIType::Kind::constType:
 516         {
 517             DIConstType* constType = static_cast<DIConstType*>(type);
 518             constTypeMap[constType->BaseTypeId()] = constType;
 519             break;
 520         }
 521     }
 522     debugInfo->AddType(type);
 523 }

 809 std::string SourceFileMap::GetSourceFilePath(const SourceLocation& location) const
 810 {
 811     std::string projectName = Path::GetDirectoryName(location.path);
 812     Project* project = debugInfo->GetProjectByName(projectName);
 813     std::string locationPath;
 814     if (project)
 815     {
 816         locationPath = GetFullPath(Path::Combine(Path::Combine(Path::Combine(Path::Combine(Path::Combine(project->DirectoryPath(), ".."), ".."), ".."), ".."), 
 817             Path::GetFileName(location.path)));
 818     }
 819     else
 820     {
 821         locationPath = location.path;
 822     }
 823     auto it = sourceFileMap.find(locationPath);
 824     if (it != sourceFileMap.cend())
 825     {
 826         return it->first;
 827     }
 828     else
 829     {

 831         std::vector<std::string> locationPathComponents = Split(locationPath, '/');
 832         int nlpc = locationPathComponents.size();
 833         for (std::map<std::string, SourceLineMap*>::const_iterator it = sourceFileMap.cbegin(); it != sourceFileMap.cend(); ++it)
 834         {
 835             const std::string& sourceFilePath = it->first;
 836             bool sourceFilePathMatches = true;
 837             std::vector<std::string> sourceFileComponents = Split(sourceFilePath, '/');
 838             int nsfc = sourceFileComponents.size();
 839             int n = std::min(nlpc, nsfc);
 840             for (int i = 0; i < n; ++i)
 841             {
 842                 if (locationPathComponents[nlpc - i - 1] != sourceFileComponents[nsfc - i - 1])
 843                 {
 844                     sourceFilePathMatches = false;
 845                     break;
 846                 }
 847             }
 848             if (sourceFilePathMatches)
 849             {
 850                 matchingSourceFilePathIterators.push_back(it);
 851             }
 852         }
 853         if (matchingSourceFilePathIterators.empty())
 854         {
 855             throw std::runtime_error("no source file path matched location '" + location.ToString() + "'");
 856         }
 857         else if (matchingSourceFilePathIterators.size() == 1)
 858         {
 859             auto it = matchingSourceFilePathIterators.front();
 860             return it->first;
 861         }
 862         else
 863         {
 864             std::string matchingSourceFiles("    ");
 865             bool first = true;
 866             for (const auto it : matchingSourceFilePathIterators)
 867             {
 868                 if (first)
 869                 {
 870                     first = false;
 871                 }
 872                 else
 873                 {
 874                     matchingSourceFiles.append("\n    ");
 875                 }
 876                 std::string projectDir = Path::GetDirectoryName(it->first);
 877                 Project* project = debugInfo->GetProjectByPath(projectDir);
 878                 matchingSourceFiles.append(it->first);
 879                 if (project != nullptr)
 880                 {
 881                     matchingSourceFiles.append(", project name='").append(project->Name() + "'");
 882                 }
 883             }
 884             throw std::runtime_error("More that one source file matched source location '" + location.ToString() + "':\n" + matchingSourceFiles + "\n" + 
 885                 "Give source location with project name '/' prefix to disambiguate.");
 886         }
 887     }
