28 std::string DIVariable::KindStr(Kind kind)
29 {
30     switch (kind)
31     {
32         case Kind::localVariable: return "local";
33         case Kind::memberVariable: return "member";
34     }
35     return std::string();
36 }

53 DIType* DIVariable::GetType() const
54 {
55     if (project)
56     {
57         return project->GetType(typeId);
58     }
59     else
60     {
61         throw std::runtime_error("internal error: project of variable not set");
62     }
63 }

65 void DIVariable::Write(soulng::util::BinaryWriter& writer)
66 {
67     writer.Write(static_cast<int8_t>(kind));
68     writer.Write(initLineNumber);
69     writer.Write(name);
70     writer.Write(irName);
71     writer.Write(typeId);
72 }
73 
74 void DIVariable::Read(soulng::util::BinaryReader& reader)
75 {
76     kind = static_cast<Kind>(reader.ReadSByte());
77     initLineNumber = reader.ReadInt();
78     name = reader.ReadUtf8String();
79     irName = reader.ReadUtf8String();
80     reader.ReadUuid(typeId);
81 }
82 
83 std::std::unique_ptr<JsonValue>DIVariable::ToJson() const
84 {
85     JsonObject* jsonObject = new JsonObject();
86     jsonObject->AddField(U"name", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(name))));
87     jsonObject->AddField(U"irName", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(irName))));
88     jsonObject->AddField(U"typeId", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(boost::uuids::to_string(typeId)))));
89     return std::unique_ptr<JsonValue>(jsonObject);
90 }
