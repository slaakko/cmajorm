 15 std::string InitializationStatusStr(InitializationStatus status)
 16 {
 17     switch (status)
 18     {
 19         case InitializationStatus::unknown: return "unknown";
 20         case InitializationStatus::initialized: return "initialized";
 21         case InitializationStatus::uninitialized: return "uninitialized";
 22     }
 23     return std::string();
 24 }
