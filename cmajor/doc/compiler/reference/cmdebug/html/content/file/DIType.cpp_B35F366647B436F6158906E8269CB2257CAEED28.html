326 void DIClassType::Write(soulng::util::BinaryWriter& writer)
327 {
328     DIType::Write(writer);
329     writer.Write(baseClassId);
330     int32_t ntp = templateParameters.size();
331     writer.Write(ntp);
332     for (int32_t i = 0; i < ntp; ++i)
333     {
334         DITemplateParameter* templateParameter = templateParameters[i].get();
335         templateParameter->Write(writer);
336     }
337     int32_t nmv = memberVariables.size();
338     writer.Write(nmv);
339     for (int32_t i = 0; i < nmv; ++i)
340     {
341         DIVariable* memberVariable = memberVariables[i].get();
342         memberVariable->Write(writer);
343     }
344     writer.Write(polymorphic);
345     if (polymorphic)
346     {
347         writer.Write(vmtPtrIndex);
348         writer.Write(vmtVariableName);
349     }
350 }
351 
352 void DIClassType::Read(soulng::util::BinaryReader& reader)
353 {
354     DIType::Read(reader);
355     reader.ReadUuid(baseClassId);
356     int32_t ntp = reader.ReadInt();
357     for (int32_t i = 0; i < ntp; ++i)
358     {
359         DITemplateParameter* templateParameter = new DITemplateParameter();
360         templateParameter->SetProject(GetProject());
361         templateParameter->Read(reader);
362         templateParameters.push_back(std::unique_ptr<DITemplateParameter>(templateParameter));
363     }
364     int32_t nmv = reader.ReadInt();
365     for (int32_t i = 0; i < nmv; ++i)
366     {
367         DIVariable* memberVariable = new DIVariable(DIVariable::Kind::memberVariable);
368         memberVariable->SetProject(GetProject());
369         memberVariable->Read(reader);
370         AddMemberVariable(memberVariable);
371     }
372     polymorphic = reader.ReadBool();
373     if (polymorphic)
374     {
375         vmtPtrIndex = reader.ReadInt();
376         vmtVariableName = reader.ReadUtf8String();
377     }
378 }

380 std::std::unique_ptr<JsonValue>DIClassType::ToJson() const
381 {
382     std::unique_ptr<JsonValue> value = DIType::ToJson();
383     if (value->Type() == JsonValueType::object)
384     {
385         JsonObject* jsonObject = static_cast<JsonObject*>(value.get());
386         jsonObject->AddField(U"baseClassId", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(boost::uuids::to_string(baseClassId)))));
387         JsonArray* templateParameterArray = new JsonArray();
388         for (const auto& templateParameter : templateParameters)
389         {
390             templateParameterArray->AddItem(templateParameter->ToJson());
391         }
392         jsonObject->AddField(U"templateParameters", std::unique_ptr<JsonValue>(templateParameterArray));
393         JsonArray* memberVariableArray = new JsonArray();
394         for (const auto& memberVariable : memberVariables)
395         {
396             memberVariableArray->AddItem(memberVariable->ToJson());
397         }
398         jsonObject->AddField(U"memberVariables", std::unique_ptr<JsonValue>(memberVariableArray));
399         jsonObject->AddField(U"polymorphic", std::unique_ptr<JsonValue>(new JsonBool(polymorphic)));
400         if (polymorphic)
401         {
402             jsonObject->AddField(U"vmtPtrIndex", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(std::to_string(vmtPtrIndex)))));
403             jsonObject->AddField(U"vmtVariableName", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(vmtVariableName))));
404         }
405     }
406     return value;
407 }

430 void DIClassTemplateSpecializationType::Write(soulng::util::BinaryWriter& writer)
431 {
432     DIClassType::Write(writer);
433     writer.Write(primaryTypeId);
434     int32_t n = templateArgumentTypeIds.size();
435     writer.Write(n);
436     for (int32_t i = 0; i < n; ++i)
437     {

439     }

442     {

447 void DIClassTemplateSpecializationType::Read(soulng::util::BinaryReader& reader)
448 {
449     DIClassType::Read(reader);
450     reader.ReadUuid(primaryTypeId);
451     int32_t n = reader.ReadInt();
452     for (int32_t i = 0; i < n; ++i)
453     {
454         boost::uuids::uuid templateArgumentTypeId;

457     }
458     containerKind = static_cast<ContainerClassTemplateKind>(reader.ReadSByte());

460     {
461         reader.ReadUuid(valueTypeId);

674 std::std::unique_ptr<DIType>ReadType(soulng::util::BinaryReader&reader, Project*project)
675 {
676     std::unique_ptr<DIType> type;
677     int8_t k = reader.ReadSByte();
678     DIType::Kind kind = static_cast<DIType::Kind>(k);
679     switch (kind)
680     {
681         case DIType::Kind::primitiveType: type.reset(new DIPrimitiveType()); break;
682         case DIType::Kind::enumType: type.reset(new DIEnumType()); break;
683         case DIType::Kind::classType: type.reset(new DIClassType()); break;
684         case DIType::Kind::specializationType: type.reset(new DIClassTemplateSpecializationType()); break;
685         case DIType::Kind::delegateType: type.reset(new DIDelegateType()); break;
686         case DIType::Kind::classDelegateType: type.reset(new DIClassDelegateType()); break;
687         case DIType::Kind::interfaceType: type.reset(new DIInterfaceType()); break;
688         case DIType::Kind::templateParameter: type.reset(new DITemplateParameter()); break;
689         case DIType::Kind::constType: type.reset(new DIConstType()); break;
690         case DIType::Kind::referenceType: type.reset(new DIReferenceType()); break;
691         case DIType::Kind::pointerType: type.reset(new DIPointerType()); break;
692         case DIType::Kind::arrayType: type.reset(new DIArrayType()); break;
693         default: throw std::runtime_error("unknown DIType kind '" + std::to_string(k));
694     }
695     type->SetProject(project);
696     type->Read(reader);
697     return type;
698 }
