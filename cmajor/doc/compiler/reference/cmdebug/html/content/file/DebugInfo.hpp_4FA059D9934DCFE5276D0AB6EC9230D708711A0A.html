 69 enum class InstructionFlags : int16_t 
 70 {
 71     none=  0, 
 72     entryCode=  1 << 0, 
 73     exitCode=  1 << 1, 
 74     beginBrace=  1 << 2, 
 75     endBrace=  1 << 3, 
 76     startFunction=  1 << 4, 
 77     throwInst=  1 << 5, 
 78     catchInst=  1 << 6
 79 };
 80 
 81 constexpr, inlineInstructionFlags operator|(InstructionFlags left, InstructionFlags right)
 82 {
 83     return InstructionFlags(int16_t(left) | int16_t(right));
 84 }
 85 
 86 constexpr, inlineInstructionFlags operator&(InstructionFlags left, InstructionFlags right)
 87 {
 88     return InstructionFlags(int16_t(left) & int16_t(right));
 89 }
 90 
 91 constexpr, inlineInstructionFlags operator~(InstructionFlags operand)
 92 {
 93     return InstructionFlags(~int16_t(operand));
 94 }
