18     DebugExpr : cmajor::debug::DebugExprNode*
19         ::= AdditiveDebugExpr:expr{ return expr; }

21 
22     AdditiveDebugExpr(var std::unique_ptr expr, var cmajor::debug::Operator op) : cmajor::debug::DebugExprNode*
23         ::=
24         (   PrefixDebugExpr:left{ expr.reset(left); }
25             (   (PLUS{ op = cmajor::debug::Operator::plus; } | MINUS{ op = cmajor::debug::Operator::minus; })
26                 PrefixDebugExpr:right
27                 {
28                     switch (op)
29                     {
30                         case cmajor::debug::Operator::plus: expr.reset(new cmajor::debug::AddDebugExprNode(expr.release(), right)); break;
31                         case cmajor::debug::Operator::minus: expr.reset(new cmajor::debug::SubDebugExprNode(expr.release(), right)); break;
32                     }
33                 }
34             )*
35         )
36         {
37             return expr.release();
38         }
39         ;
40 
41     PrefixDebugExpr : cmajor::debug::DebugExprNode*
42         ::= AST PrefixDebugExpr:derefOperand{ return new cmajor::debug::DerefDebugExprNode(derefOperand); }
43         |   AMP PrefixDebugExpr:addrOfOperand{ return new cmajor::debug::AddrOfDebugExprNode(addrOfOperand); }
44         |   PostfixDebugExpr:postfix{ return postfix; }
45         ;
46 
47     PostfixDebugExpr(var std::unique_ptr expr) : cmajor::debug::DebugExprNode*
48         ::=
49         (   PrimaryDebugExpr:primary{ expr.reset(primary); }
50             (   DOT ID{ expr.reset(new cmajor::debug::DotDebugExprNode(expr.release(), ToUtf8(lexer.GetToken(pos).match.ToString()))); }
51             |   DOT BASE{ expr.reset(new cmajor::debug::DotDebugExprNode(expr.release(), "base")); }
52             |   LBRACKET DebugExpr:rangeStart COMMA DebugExpr:rangeEnd RBRACKET{ expr.reset(new cmajor::debug::RangeDebugExprNode(expr.release(), rangeStart, rangeEnd)); }
53             |   LBRACKET DebugExpr:index RBRACKET{ expr.reset(new cmajor::debug::SubscriptDebugExprNode(expr.release(), index)); }
54             )*
55         )
56         {
57             return expr.release();
58         }
59         ;
60 
61     PrimaryDebugExpr : cmajor::debug::DebugExprNode*
62         ::= ID{ return new cmajor::debug::IdentifierDebugExprNode(ToUtf8(lexer.GetToken(pos).match.ToString())); }
63         |   INTEGER{ return new cmajor::debug::IntegerDebugExprNode(cmajor::debug::ParseLong(ToUtf8(lexer.GetToken(pos).match.ToString()))); }
64         |   LPAREN DebugExpr:expr RPAREN{ return new cmajor::debug::ParenthesizedDebugExprNode(expr); }
65         |   TypeIdExpr:typeExpr{ return typeExpr; }
66         |   CastDebugExpr:castExpr{ return castExpr; }
67         ;
68 
69     CastDebugExpr : cmajor::debug::DebugExprNode*
70         ::= CAST LANGLE TypeIdExpr:typeIdExpr RANGLE LPAREN DebugExpr:expr RPAREN{ return new cmajor::debug::CastDebugExprNode(typeIdExpr, expr); }
71         ;
72 
73     TypeIdExpr(var std::string typeId) : cmajor::debug::TypeIdDebugExprNode*
74         ::= (TYPEID LPAREN STRING{ typeId = cmajor::debug::ParseString(lexer.GetToken(pos)); } RPAREN)
75         {
76             return new cmajor::debug::TypeIdDebugExprNode(typeId);
77         }
78         ;
