342 class GdbReplyRecord 
343 {
344 public:
345     enum class Kind : uint8_t 
346     {
347         result, execRunning, execStopped, notification, consoleOutput, targetOutput, logOutput, prompt, parsingError
348     };
349     const char* KindStr() const;
350     GdbReplyRecord(Kind kind_, GdbResults* results_);
351     GdbReplyRecord(const GdbReplyRecord&) = delete;
352     GdbReplyRecord(GdbReplyRecord&&) = delete;
353     GdbReplyRecord& operator=(const GdbReplyRecord&) = delete;
354     GdbReplyRecord& operator=(GdbReplyRecord&&) = delete;
355     virtual ~GdbReplyRecord();
356     Kind GetKind() const { return kind; }
357     virtual bool Stopped() const { return false; }
358     virtual bool CommandSucceeded() const { return kind != Kind::parsingError; }
359     virtual std::unique_ptr<JsonValue> ToJson() const;
360     GdbResults* Results() const { return results.get(); }
361 private:
362     Kind kind;
363     std::unique_ptr<GdbResults> results;
364 };
