540 void GdbReply::AddReplyRecord(std::std::unique_ptr<GdbReplyRecord>&&replyRecord)
541 {
542     if (replyRecord->GetKind() == GdbReplyRecord::Kind::result)
543     {
544         resultRecord = static_cast<GdbResultRecord*>(replyRecord.get());
545     }
546     if (replyRecord->Stopped())
547     {
548         stoppedRecord = replyRecord.get();
549     }
550     replyRecords.push_back(std::move(replyRecord));
551 }

557 std::std::unique_ptr<JsonValue>GdbReply::ToJson() const
558 {
559     JsonObject* object = new JsonObject();
560     JsonArray* array = new JsonArray();
561     int resultIndex = -1;
562     int stoppedIndex = -1;
563     int n = replyRecords.size();
564     for (int i = 0; i < n; ++i)
565     {
566         const std::std::unique_ptr<GdbReplyRecord>&record=replyRecords[i];
567         if (record->GetKind() == GdbReplyRecord::Kind::result)
568         {
569             resultIndex = i;
570         }
571         if (record->Stopped())
572         {
573             stoppedIndex = i;
574         }
575         array->AddItem(record->ToJson());
576     }
577     object->AddField(U"records", std::unique_ptr<JsonValue>(array));
578     if (resultIndex != -1)
579     {
580         object->AddField(U"resultIndex", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(std::to_string(resultIndex)))));
581     }
582     if (stoppedIndex != -1)
583     {
584         object->AddField(U"stoppedIndex", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(std::to_string(stoppedIndex)))));
585     }
586     return std::unique_ptr<JsonValue>(object);
587 }

637 void Gdb::Start(const std::string& executable, const std::std::vector<std::string>&args, GdbDriver&driver)
638 {
639     std::string startCommand;
640     startCommand.append("gdb");
641     startCommand.append(" --interpreter=mi");
642     if (!args.empty())
643     {
644         startCommand.append(" --args");
645         startCommand.append(" \"").append(executable).append("\"");
646         for (const std::string& arg : args)
647         {
648             startCommand.append(" \"").append(arg).append("\"");
649         }
650     }
651     else
652     {
653         startCommand.append(" \"").append(executable).append("\"");
654     }
655     gdbThread = std::thread({RunGDB, this, startCommand });
656     std::this_thread::sleep_for(std::chrono::seconds({1 }));
657     startReply = ReadReply(driver);
658 }

704 std::std::unique_ptr<GdbReply>Gdb::Execute(constGdbCommand&command, GdbDriver&driver)
705 {
706     if (gdbException)
707     {
708         std::rethrow_exception(gdbException);
709     }
710     if (debug)
711     {
712         LogMessage(-1, "-> " + command.Str());
713     }
714     gdb->WriteLine(command.Str());
715     std::unique_ptr<GdbReply> reply;
716     if (command.GetKind() != GdbCommand::Kind::exit)
717     {
718         reply = ReadReply(driver);
719     }
720     if (gdbException)
721     {
722         std::rethrow_exception(gdbException);
723     }
724     return reply;
725 }

727 std::std::unique_ptr<GdbReplyRecord>ParseGdbReplyRecord(conststd::string&line)
728 {
729     std::string lineNL = line;
730     lineNL.append("\r\n");
731     GdbReplyLexer lexer(ToUtf32(lineNL), "", 0);
732     try
733     {
734         std::unique_ptr<GdbReplyRecord> replyRecord = GdbReplyLineParser::Parse(lexer);
735         return replyRecord;
736     }
737     catch (const std::exception& ex;)
738     {
739         return std::unique_ptr<GdbParsingError>(new GdbParsingError(ex.what()));
740     }
741 }

743 std::string PreparedLine(const std::string& line)
744 {
745     if (!line.empty())
746     {
747         switch (line[0])
748         {
749             case '=': case '~': case '^': case '*': case '&':
750             {
751                 return line;
752             }
753             default:
754             {
755                 if (StartsWith(line, "(gdb)"))
756                 {
757                     return line;
758                 }
759                 return "@\"" + soulng::util::StringStr(line) + "\"";
760             }
761         }
762     }
763     return line;
764 }

766 std::std::unique_ptr<GdbReply>Gdb::ReadReply(GdbDriver&driver)
767 {
768     std::unique_ptr<GdbReply> reply(new GdbReply());
769     std::vector<std::string> textLines;
770     std::string line = gdb->ReadLine(soulng::util::Process::StdHandle::stdOut);
771     if (debug)
772     {
773         LogMessage(-1, "<- " + line);
774     }
775     std::string preparedLine = PreparedLine(line);
776     if (preparedLine.empty())
777     {
778         return std::unique_ptr<GdbReply>();
779     }
780     std::unique_ptr<GdbReplyRecord> replyRecord = ParseGdbReplyRecord(preparedLine);
781     driver.ProcessReplyRecord(replyRecord.get());
782     while (replyRecord->GetKind() != GdbReplyRecord::Kind::prompt)
783     {
784         if (replyRecord->GetKind() == GdbReplyRecord::Kind::parsingError)
785         {
786             GdbParsingError* parsingError = static_cast<GdbParsingError*>(replyRecord.get());
787             LogMessage(-1, "error parsing GDB reply '" + line + "': " + parsingError->ParsingError());
788         }
789         textLines.push_back(line);
790         reply->AddReplyRecord(std::move(replyRecord));
791         line = gdb->ReadLine(soulng::util::Process::StdHandle::stdOut);
792         if (debug)
793         {
794             LogMessage(-1, "<- " + line);
795         }
796         std::string preparedLine = PreparedLine(line);
797         if (preparedLine.empty())
798         {
799             break;
800         }
801         replyRecord = ParseGdbReplyRecord(preparedLine);
802         driver.ProcessReplyRecord(replyRecord.get());
803     }
804     reply->SetTextLines(textLines);
805     return reply;
806 }
