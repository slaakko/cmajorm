 29 class BoundDebugNode 
 30 {
 31 public:
 32     enum class Kind : int8_t 
 33     {
 34         variableReferenceNode, integerLiteralNode, addNode, subNode, derefNode, addrOfNode, dotNode, subscriptNode, rangeNode, parenExprNode, 
 35         typeNode, castNode, debugExpression
 36     };
 37     BoundDebugNode(Kind kind_, DIType* type_, DebugExprNode* sourceNode_);
 38     Kind GetKind() const { return kind; }
 39     virtual ~BoundDebugNode();
 40     virtual void Accept(BoundDebugNodeVisitor& visitor) = 0;
 41     virtual bool IsDebuggerVarNode() const { return false; }
 42     virtual BoundDebugNode* Clone() const = 0;
 43     DIType* Type() const { return type; }
 44     void SetType(DIType* type_) { type = type_; }
 45     virtual bool IsTypeNode() const { return false; }
 46     virtual std::string ToString() const = 0;
 47     virtual std::string GdbExprString() const = 0;
 48     DebugExprNode* SourceNode() const { return sourceNode; }
 49 private:
 50     Kind kind;
 51     DIType* type;
 52     DebugExprNode* sourceNode;
 53 };
