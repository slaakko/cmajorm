 21 class DebugExprNode 
 22 {
 23 public:
 24     enum class Kind : int8_t 
 25     {
 26         identifier, integer, parenExpr, add, sub, deref, addrOf, dot, subscript, range, typeId, cast_
 27     };
 28     DebugExprNode(Kind kind_);
 29     virtual ~DebugExprNode();
 30     virtual DebugExprNode* Clone() const = 0;
 31     virtual void Accept(DebugExprVisitor& visitor) = 0;
 32     virtual std::string ToString() const = 0;
 33     Kind GetKind() const { return kind; }
 34 private:
 35     Kind kind;
 36 };
