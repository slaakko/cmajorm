106 void CmdbSession::Run()
107 {
108     try
109     {
110         listenSocket.Bind(port);
111         listenSocket.Listen(10);
112         started.notify_one();
113         while (!exiting)
114         {
115             TcpSocket socket = listenSocket.Accept();
116             if (exiting) return;
117             sessionClosed = false;
118             while (!sessionClosed)
119             {
120                 Transact(socket);
121             }
122         }
123     }
124     catch (const std::exception& ex;)
125     {
126         if (!exiting)
127         {
128             std::cerr << "error from CMDB session run: " << ex.what() << std::endl;
129         }
130     }
131 }

172 void CmdbSession::OpenSession(TcpSocket& socket, sngxml::dom::Element* request)
173 {
174     std::u32string skeyValue = request->GetAttribute(U"key");
175     if (!skeyValue.empty())
176     {
177         if (GetSha1MessageDigest(skeyStr) == ToUtf8(skeyValue))
178         {
179             sngxml::dom::Document openSessionResponse;
180             sngxml::dom::Element* responseMessage = new sngxml::dom::Element(U"cmdbMessage");
181             responseMessage->SetAttribute(U"kind", U"openSessionResponse");
182             responseMessage->SetAttribute(U"key", ToUtf32(GetSha1MessageDigest(rkeyStr)));
183             openSessionResponse.AppendChild(std::unique_ptr<sngxml::dom::Node>(responseMessage));
184             sngxml::dom::SendDocument(socket, openSessionResponse);
185             return;
186         }
187     }
188     throw std::runtime_error("invalid session key");
189 }

196 void CmdbSession::Transact(TcpSocket& socket)
197 {
198     std::unique_ptr<sngxml::dom::Document> request = sngxml::dom::ReceiveDocument(socket);
199     if (request)
200     {
201         std::unique_ptr<sngxml::xpath::XPathObject> requestObject(sngxml::xpath::Evaluate(U"/cmdbMessage", request.get()));
202         if (requestObject)
203         {
204             if (requestObject->Type() == sngxml::xpath::XPathObjectType::nodeSet)
205             {
206                 sngxml::xpath::XPathNodeSet* nodeSet = static_cast<sngxml::xpath::XPathNodeSet*>(requestObject.get());
207                 if (nodeSet->Length() == 1)
208                 {
209                     sngxml::dom::Node* node = (*nodeSet)[0];
210                     if (node->GetNodeType() == sngxml::dom::NodeType::elementNode)
211                     {
212                         sngxml::dom::Element* element = static_cast<sngxml::dom::Element*>(node);
213                         std::u32string kind = element->GetAttribute(U"kind");
214                         if (kind == U"openSessionRequest")
215                         {
216                             OpenSession(socket, element);
217                         }
218                         else if (kind == U"closeSessionRequest")
219                         {
220                             CloseSession();
221                         }
222                         else if (kind == U"inputRequest")
223                         {
224                             GetInput(socket);
225                         }
226                         else if (kind == U"outputRequest")
227                         {
228                             PutOutput(socket, element);
229                         }
230                         else
231                         {
232                             throw std::runtime_error("invalid CMDB session request received: kind='" + ToUtf8(kind) + "'");
233                         }
234                     }
235                 }
236             }
237         }
238     }
239     else
240     {
241         CloseSession();
242     }
243 }

245 void CmdbSession::GetInput(TcpSocket& socket)
246 {
247     std::string bytes = client->GetTargetInputBytes();
248     sngxml::dom::Document inputResponse;
249     sngxml::dom::Element* inputResponseElement = new sngxml::dom::Element(U"cmdbMessage");
250     inputResponseElement->SetAttribute(U"kind", U"inputResponse");
251     inputResponseElement->SetAttribute(U"bytes", ToUtf32(bytes));
252     inputResponse.AppendChild(std::unique_ptr<sngxml::dom::Node>(inputResponseElement));
253     sngxml::dom::SendDocument(socket, inputResponse);
254 }

256 void CmdbSession::PutOutput(TcpSocket& socket, sngxml::dom::Element* request)
257 {
258     std::u32string handleValue = request->GetAttribute(U"handle");
259     if (!handleValue.empty())
260     {
261         std::string handleStr = ToUtf8(handleValue);
262         int handle = boost::lexical_cast<int>(handleStr);
263         if (handle == 1 || handle == 2)
264         {
265             std::u32string bytesValue = request->GetAttribute(U"bytes");
266             if (!bytesValue.empty())
267             {
268                 std::string output;
269                 std::string bytesStr = ToUtf8(bytesValue);
270                 int n = bytesStr.length() / 2;
271                 for (int i = 0; i < n; ++i)
272                 {
273                     std::string hex;
274                     hex.append(1, bytesStr[i * 2]);
275                     hex.append(1, bytesStr[i * 2 + 1]);
276                     uint8_t x = ParseHexByte(hex);
277                     output.append(1, static_cast<char>(x));
278                 }
279                 client->WriteTargetOuput(handle, output);
280                 sngxml::dom::Document outputResponse;
281                 sngxml::dom::Element* outputResponseElement = new sngxml::dom::Element(U"cmdbMessage");
282                 outputResponseElement->SetAttribute(U"kind", U"outputResponse");
283                 outputResponse.AppendChild(std::unique_ptr<sngxml::dom::Node>(outputResponseElement));
284                 sngxml::dom::SendDocument(socket, outputResponse);
285             }
286         }
287     }

295 void StartCmdbSession(const std::string& cmdbSessionFilePath, CodeFormatter& formatter, CmdbSessionClient* client, bool verbose)
296 {
297     if (verbose)
298     {
299         formatter.WriteLine("starting CMDB session...");
300     }
301     std::ofstream cmdbSessionFile(cmdbSessionFilePath);
302     CodeFormatter sessionFileFormatter(cmdbSessionFile);
303     sessionFileFormatter.SetIndentSize(1);
304     sngxml::dom::Document cmdbSessionDoc;
305     sngxml::dom::Element* sessionElement = new sngxml::dom::Element(U"cmdbSession");
306     sngxml::dom::Element* timestampElement = new sngxml::dom::Element(U"timestamp");
307     time_t t;
308     time(&t);
309     timestampElement->SetAttribute(U"value", ToUtf32(std::to_string(t)));
310     sessionElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(timestampElement));
311     sngxml::dom::Element* skeyElement = new sngxml::dom::Element(U"skey");
312     boost::uuids::uuid skey = boost::uuids::random_generator()();
313     std::string skeyStr = boost::uuids::to_string(skey);
314     skeyElement->SetAttribute(U"value", ToUtf32(skeyStr));
315     sessionElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(skeyElement));
316     sngxml::dom::Element* rkeyElement = new sngxml::dom::Element(U"rkey");
317     boost::uuids::uuid rkey = boost::uuids::random_generator()();
318     std::string rkeyStr = boost::uuids::to_string(rkey);
319     rkeyElement->SetAttribute(U"value", ToUtf32(rkeyStr));
320     sessionElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(rkeyElement));
321     sngxml::dom::Element* portElement = new sngxml::dom::Element(U"port");
322     portElement->SetAttribute(U"value", ToUtf32(std::to_string(port)));
323     sessionElement->AppendChild(std::unique_ptr<sngxml::dom::Node>(portElement));
324     cmdbSessionDoc.AppendChild(std::unique_ptr<sngxml::dom::Node>(sessionElement));
325     cmdbSessionDoc.Write(sessionFileFormatter);
326     CmdbSession::Instance().SetKeys(skeyStr, rkeyStr);
327     CmdbSession::Instance().Start(client);
328     CmdbSession::Instance().WaitForStarted();
329 }
