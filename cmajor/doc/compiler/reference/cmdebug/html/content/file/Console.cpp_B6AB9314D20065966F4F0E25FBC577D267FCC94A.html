 25 void Console::Run()
 26 {
 27     std::string line;
 28     while (true)
 29     {
 30         std::istream& s = std::getline(std::cin, line);
 31         canProceeed = false;
 32         if (needTargetInput)
 33         {
 34             std::unique_lock<std::mutex> lock(mtx);
 35             targetInputLineReceived = false;
 36             if (!s)
 37             {
 38                 targetInputEof = true;
 39             }
 40             else
 41             {
 42                 targetInputEof = false;
 43             }
 44             targetInputLine = line;
 45             targetInputLineAvailable = true;
 46             targetInputLineAvailableVar.notify_one();
 47             targetInputLineReceivedVar.wait(lock, [this]{ return targetInputLineReceived; });
 48             needTargetInput = false;
 49         }
 50         else
 51         {
 52             if (!s)
 53             {
 54                 break;
 55             }
 56             std::unique_ptr<DebuggerCommand> command;
 57             try
 58             {
 59                 command = ParseDebuggerCommand(driver->CurrentSourceFilePath(), line);
 60             }
 61             catch (const std::exception& ex;)
 62             {
 63                 driver->Error(ex.what());
 64                 driver->Prompt();
 65                 continue;
 66             }
 67             if (command->GetKind() == DebuggerCommand::Kind::exit)
 68             {
 69                 break;
 70             }
 71             else
 72             {
 73                 std::unique_lock<std::mutex> lock(mtx);
 74                 commandReceived = false;
 75                 targetRunning = false;
 76                 commands.push_back(std::move(command));
 77                 commandAvailableVar.notify_one();
 78                 commandAvailable = true;
 79                 commandReceivedVar.wait(lock, [this]{ return commandReceived; });
 80             }
 81         }
 82         targetInputLineAvailable = false;
 83         commandAvailable = false;
 84         {
 85             std::unique_lock<std::mutex> lock(mtx);
 86             proceed.wait(lock, [this]{ return canProceeed; });
 87         }
 88         if (needTargetInput)
 89         {
 90             driver->TargetInputPrompt();
 91         }
 92         else
 93         {
 94             driver->Prompt();
 95         }
 96     }
 97     terminated = true;
 98     TerminateGDB();
 99     std::unique_lock<std::mutex> lock(mtx);
100     commands.push_back(std::unique_ptr<DebuggerCommand>(new DebuggerExitCommand()));
101     commandAvailable = true;
102     commandAvailableVar.notify_one();
103 }

110 std::std::unique_ptr<DebuggerCommand>Console::GetCommand()
111 {
112     if (terminated) return std::unique_ptr<DebuggerCommand>();
113     std::unique_lock<std::mutex> lock(mtx);
114     commandAvailableVar.wait(lock, [this]{ return commandAvailable; });
115     std::unique_ptr<DebuggerCommand> command = std::move(commands.front());
116     commands.pop_front();
117     commandReceived = true;
118     commandReceivedVar.notify_one();
119     return command;
120 }

122 void Console::SetTargetRunning()
123 {
124     std::unique_lock<std::mutex> lock(mtx);
125     targetRunning = true;
126     targetRunningVar.notify_one();
127 }

129 std::string Console::GetTargetInputLine()
130 {
131     std::unique_lock<std::mutex> lock(mtx);
132     targetInputLineAvailable = false;
133     needTargetInput = true;
134     canProceeed = true;
135     proceed.notify_one();
136     targetInputLineAvailableVar.wait(lock, [this]{ return targetInputLineAvailable; });
137     std::string line = targetInputLine;
138     targetInputLineReceived = true;
139     targetInputLineReceivedVar.notify_one();
140     return line;
141 }

143 void Console::Reset()
144 {
145     std::unique_lock<std::mutex> lock(mtx);
146     commandAvailable = false;
147     commandReceived = false;
148     targetRunning = true;
149     targetRunningVar.notify_one();
150 }
