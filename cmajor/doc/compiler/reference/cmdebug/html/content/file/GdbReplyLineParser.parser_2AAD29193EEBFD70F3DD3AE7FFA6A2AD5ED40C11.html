 20     GdbReplyRecord : cmajor::debug::GdbReplyRecord*
 21         ::= GdbPrompt:prompt{ return prompt; }
 22         |   GdbResultRecord:resultRecord{ return resultRecord; }

 27     GdbResultRecord : cmajor::debug::GdbReplyRecord*
 28         ::= GdbDoneRecord:done{ return done; }
 29         |   GdbRunningRecord:running{ return running; }
 30         |   GdbConnectedRecord:connected{ return connected; }
 31         |   GdbErrorRecord:error{ return error; }
 32         |   GdbExitRecord:exit{ return exit; }
 33         ;
 34 
 35     GdbDoneRecord : cmajor::debug::GdbReplyRecord*
 36         ::=
 37         (

 41             return new cmajor::debug::GdbDoneRecord(results);
 42         }
 43         ;
 44

115     GdbNotifyAsyncRecord(var std::string notification) : cmajor::debug::GdbReplyRecord*
116         ::=
117         (
118             EQ IDENTIFIER{ notification = ToUtf8(lexer.GetToken(pos).match.ToString()); } GdbResults:results
119         )
120         {
121             return new cmajor::debug::GdbNotifyAsyncRecord(notification, results);
122         }

125     GdbStreamRecord(var std::string text) : cmajor::debug::GdbReplyRecord*
126         ::= TILDE STRING{ text = cmajor::debug::ParseString(lexer.GetToken(pos)); return new cmajor::debug::GdbConsoleOutputRecord(text); }
127         |   AT STRING{ text = cmajor::debug::ParseString(lexer.GetToken(pos)); return new cmajor::debug::GdbTargetOutputRecord(text); }
128         |   AMP STRING{ text = cmajor::debug::ParseString(lexer.GetToken(pos));  return new cmajor::debug::GdbLogOutputRecord(text); }
129         ;
130 
131     GdbPrompt : cmajor::debug::GdbReplyRecord*
132         ::= LPAREN GDB RPAREN{ return new cmajor::debug::GdbPrompt(); }
133         ;
134 
135     GdbResults(var std::unique_ptr results) : cmajor::debug::GdbResults*
136         ::= empty{ results.reset(new cmajor::debug::GdbResults()); }
137         (
138             (COMMA GdbResult:result{ results->Add(result); })*
139         )
140         {
141             return results.release();
142         }
143         ;
144 
145     GdbResult : cmajor::debug::GdbResult*
146         ::=
147         (
148             GdbVariable:variable EQ GdbValue:value
149         )
150         {
151             return new cmajor::debug::GdbResult(variable, value);
152         }
153         ;
154 
155     GdbVariable : std::string
156         ::= IDENTIFIER{ return ToUtf8(lexer.GetToken(pos).match.ToString()); }
157         ;
158 
159     GdbValue : cmajor::debug::GdbValue*
160         ::= GdbStringValue:string{ return string; }
161         |   GdbTupleValue:tuple{ return tuple; }
162         |   GdbListValue:list{ return list; }
163         ;
164 
165     GdbStringValue : cmajor::debug::GdbValue*
166         ::= STRING{ return new cmajor::debug::GdbStringValue(cmajor::debug::ParseString(lexer.GetToken(pos))); }
167         ;
168 
169     GdbTupleValue(var std::unique_ptr tuple) : cmajor::debug::GdbValue*
170         ::= LBRACE{ tuple.reset(new cmajor::debug::GdbTupleValue()); }
171             (GdbResult:result{ tuple->AddResult(result); } % COMMA)?
172             RBRACE{ return tuple.release(); }
173         ;
174 
175     GdbListValue(var std::unique_ptr list) : cmajor::debug::GdbValue*
176         ::= LBRACKET{ list.reset(new cmajor::debug::GdbListValue()); }
177             (   (GdbResult:result{ list->AddValue(result); } % COMMA)
178             |   (GdbValue:value{ list->AddValue(value); } % COMMA)
179             )?
180             RBRACKET{ return list.release(); }
181         ;
