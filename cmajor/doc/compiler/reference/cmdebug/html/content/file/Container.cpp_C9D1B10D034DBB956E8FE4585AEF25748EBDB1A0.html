  89 uint64_t Pointer::Address()
  90 {
  91     if (hasAddress)
  92     {
  93         return address;
  94     }
  95     std::unique_ptr<JsonValue> value = Value();
  96     if (value->Type() == JsonValueType::object)
  97     {
  98         JsonObject* jsonObject = static_cast<JsonObject*>(value.get());
  99         JsonValue* successValue = jsonObject->GetField(U"success");
 100         bool success = false;
 101         if (successValue && successValue->Type() == JsonValueType::boolean)
 102         {
 103             success = static_cast<JsonBool*>(successValue)->Value();
 104         }
 105         if (success)
 106         {
 107             JsonValue* addrValue = jsonObject->GetField(U"value");
 108             if (addrValue && addrValue->Type() == JsonValueType::string)
 109             {
 110                 JsonString* addrStr = static_cast<JsonString*>(addrValue);
 111                 std::string addressStr = ToUtf8(addrStr->Value());
 112                 address = ParseHex(addressStr);
 113                 hasAddress = true;
 114                 return address;
 115             }
 116         }
 117     }
 118     hasAddress = true;
 119     address = 0;
 120     return address;
 121 }

 159 uint64_t GetContainerAddress(Debugger& debugger, ContainerClassTemplateKind containerKind, const std::string& containerVarExpr)
 160 {
 161     uint64_t address = 0;
 162     std::string addressExpr;
 163     switch (containerKind)
 164     {
 165         case ContainerClassTemplateKind::forwardList:
 166         case ContainerClassTemplateKind::linkedList:
 167         {
 168             addressExpr = "&(" + containerVarExpr + ").head";
 169             break;
 170         }
 171         case ContainerClassTemplateKind::hashMap:
 172         case ContainerClassTemplateKind::hashSet:
 173         {
 174             addressExpr = "&(" + containerVarExpr + ").table";
 175             break;
 176         }
 177         case ContainerClassTemplateKind::list:
 178         {
 179             addressExpr = "&(" + containerVarExpr + ").items";
 180             break;
 181         }
 182         case ContainerClassTemplateKind::set:
 183         case ContainerClassTemplateKind::map:
 184         {
 185             addressExpr = "&(" + containerVarExpr + ").tree";
 186             break;
 187         }
 188         case ContainerClassTemplateKind::queue:
 189         case ContainerClassTemplateKind::stack:
 190         {
 191             addressExpr = "&(" + containerVarExpr + ").items";
 192             break;
 193         }
 194     }
 195     if (!addressExpr.empty())
 196     {
 197         EvaluationGuard evaluationGuard(debugger);
 198         debugger.Evaluate(addressExpr);
 199         JsonValue* resultValue = debugger.ReleaseResult();
 200         if (resultValue && resultValue->Type() == JsonValueType::object)
 201         {
 202             JsonObject* resultObject = static_cast<JsonObject*>(resultValue);
 203             JsonValue* successValue = resultObject->GetField(U"success");
 204             bool success = false;
 205             if (successValue && successValue->Type() == JsonValueType::boolean)
 206             {
 207                 JsonBool* successBool = static_cast<JsonBool*>(successValue);
 208                 success = successBool->Value();
 209             }
 210             if (success)
 211             {
 212                 JsonValue* addressValue = resultObject->GetField(U"value");
 213                 if (addressValue && addressValue->Type() == JsonValueType::string)
 214                 {
 215                     JsonString* addrStr = static_cast<JsonString*>(addressValue);
 216                     std::string addressStr = ToUtf8(addrStr->Value());
 217                     address = ParseHex(addressStr);
 218                 }
 219             }
 220         }
 221     }
 222     return address;
 223 }

 369 {
 370     if (IsEnd()) return this;
 371     Pointer* prev = bucketPtr;
 372     Pointer* nextPtr = bucketPtr->PtrMember("next");
 373     Container& container = GetContainer();

 375     if (nextPtr->IsNull())
 376     {

 390         {
 391             Iterator* endIterator = container.GetIterator(-1);
 392             if (endIterator)
 393             {
 394                 return endIterator;
 395             }
 396             HashtableIterator* endIter = new HashtableIterator(container, -1, container.EndPtr(), -1, -1, std::string());
 397             container.AddIterator(endIter);
 398             return endIter;
 399         }
 400         HashtableIterator* nextIterator = new HashtableIterator(container, index, bptr.get(), nextBucketIndex, bucketCount, bucketsExpr);
 401         container.AddPointer(bptr.release());

 414 {
 415     Pointer* left = nodePtr->PtrMember("left");
 416     while (!left->IsNull())
 417     {

 424 Pointer* TreeNext(Pointer* nodePtr)
 425 {
 426     Pointer* right = nodePtr->PtrMember("right");
 427     if (!right->IsNull())
 428     {
 429         return TreeMin(right);
 430     }
 431     else
 432     {
 433         Pointer* parent = nodePtr->PtrMember("parent");
 434         Pointer* parentRight = parent->PtrMember("right");
 435         while (nodePtr->Address() == parentRight->Address())
 436         {
 437             nodePtr = parent;
 438             parent = parent->PtrMember("parent");
 439             parentRight = parent->PtrMember("right");
 440         }
 441         right = nodePtr->PtrMember("right");
 442         if (right->Address() != parent->Address())
 443         {
 444             return parent;
 445         }
 446         return nodePtr;
 447     }
