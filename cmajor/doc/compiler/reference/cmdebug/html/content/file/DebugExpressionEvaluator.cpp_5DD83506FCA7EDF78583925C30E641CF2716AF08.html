 19 DebugExpressionEvaluator::DebugExpressionEvaluator(Debugger& debugger_, DebugInfo* debugInfo_) :
 20     debugger(debugger_), debugInfo(debugInfo_), integer(-1), rangeStart(-1), rangeEnd(-1), status(InitializationStatus::unknown)
 21 {
 22 }

 24 void DebugExpressionEvaluator::Visit(BoundDebugExpression& expr)
 25 {
 26     bool addTypes = false;
 27     BoundDebugNode* node = expr.Node();
 28     status = expr.Status();
 29     switch (node->GetKind())
 30     {
 31         case BoundDebugNode::Kind::typeNode:
 32         {
 33             DIType* type = node->Type();
 34             JsonObject* object = new JsonObject();
 35             object->AddField(U"type", type->ToJson());
 36             object->AddField(U"success", std::unique_ptr<JsonValue>(new JsonBool(true)));
 37             result.reset(object);
 38             break;
 39         }
 40         case BoundDebugNode::Kind::subscriptNode:
 41         {
 42             node->Accept(*this);
 43             break;
 44         }
 45         case BoundDebugNode::Kind::rangeNode:
 46         {
 47             node->Accept(*this);
 48             break;
 49         }
 50         default:
 51         {
 52             Evaluate(node);
 53             break;
 54         }
 55     }
 56 }

 63 void DebugExpressionEvaluator::Visit(BoundSubscriptNode& node)
 64 {
 65     integer = -1;
 66     node.Index()->Accept(*this);
 67     int64_t index = integer;
 68     if (index == -1)
 69     {
 70         throw std::runtime_error(node.ToString() + ": invalid subscript");
 71     }
 72     BoundDebugNode* subject = node.Subject();
 73     DIType* subjectType = subject->Type();
 74     switch (subjectType->GetKind())
 75     {
 76         case DIType::Kind::pointerType:
 77         case DIType::Kind::arrayType:
 78         {
 79             Evaluate(&node);
 80             break;
 81         }
 82         case DIType::Kind::specializationType:
 83         {
 84             ContainerClassTemplateKind containerKind = GetContainerKind(subjectType);
 85             if (containerKind != ContainerClassTemplateKind::notContainerClassTemplate)
 86             {
 87                 Container* container = debugger.GetContainer(containerKind, subject->SourceNode()->ToString());
 88                 result = container->Subscript(subject->SourceNode()->ToString(), index);
 89             }
 90             else
 91             {
 92                 throw std::runtime_error(node.ToString() + ": unknown container type");
 93             }
 94             break;
 95         }
 96     }
 97 }

 99 void DebugExpressionEvaluator::Visit(BoundRangeNode& node)
100 {
101     integer = -1;
102     node.RangeStart()->Accept(*this);
103     rangeStart = integer;
104     integer = -1;
105     node.RangeEnd()->Accept(*this);
106     rangeEnd = integer;
107     BoundDebugNode* subject = node.Subject();
108     DIType* subjectType = subject->Type();
109     switch (subjectType->GetKind())
110     {
111         case DIType::Kind::pointerType:
112         {
113             EvaluatePointerRange(&node, subject, rangeStart, rangeEnd);
114             break;
115         }
116         case DIType::Kind::arrayType:
117         {
118             EvaluateArrayRange(&node, subject, rangeStart, rangeEnd);
119             break;
120         }
121         case DIType::Kind::specializationType:
122         {
123             ContainerClassTemplateKind containerKind = GetContainerKind(subjectType);
124             if (containerKind != ContainerClassTemplateKind::notContainerClassTemplate)
125             {
126                 Container* container = debugger.GetContainer(containerKind, subject->SourceNode()->ToString());
127                 result = container->Range(subject->SourceNode()->ToString(), rangeStart, rangeEnd);
128             }
129             else
130             {
131                 throw std::runtime_error(node.ToString() + ": unknown container type");
132             }
133             break;
134         }
135     }
136 }

138 void DebugExpressionEvaluator::EvaluatePointerRange(BoundDebugNode* node, BoundDebugNode* subject, int64_t rangeStart, int64_t rangeEnd)
139 {
140     if (rangeEnd == -1)
141     {
142         throw std::runtime_error("must specify range end explicitly for pointer type range expression: " + node->ToString());
143     }
144     std::unique_ptr<JsonObject> rangeResult(new JsonObject());
145     std::unique_ptr<JsonArray> rangeArray(new JsonArray());
146     for (int64_t index = rangeStart; index < rangeEnd; ++index)
147     {
148         IntegerDebugExprNode integerLiteralNode(index);
149         BoundIntegerLiteralNode boundIntegerLiteralNode(debugInfo->GetMainProject()->GetLongType(), index, &integerLiteralNode);
150         SubscriptDebugExprNode subscriptNode(subject->SourceNode()->Clone(), integerLiteralNode.Clone());
151         BoundSubscriptNode boundSubscriptNode(node->Type(), subject->Clone(), boundIntegerLiteralNode.Clone(), &subscriptNode);
152         boundSubscriptNode.Accept(*this);
153         std::unique_ptr<JsonValue> subscriptResult(result.release());
154         rangeArray->AddItem(std::unique_ptr<JsonValue>(subscriptResult.release()));
155     }
156     rangeResult->AddField(U"range", std::unique_ptr<JsonValue>(rangeArray.release()));
157     result.reset(rangeResult.release());
158 }

160 void DebugExpressionEvaluator::EvaluateArrayRange(BoundDebugNode* node, BoundDebugNode* subject, int64_t rangeStart, int64_t rangeEnd)
161 {
162     DIType* subjectType = subject->Type();
163     if (rangeEnd == -1)
164     {
165         if (subjectType->GetKind() == DIType::Kind::arrayType)
166         {
167             DIArrayType* arrayType = static_cast<DIArrayType*>(subjectType);
168             if (arrayType->Size() != -1)
169             {
170                 rangeEnd = arrayType->Size();
171             }
172         }
173     }
174     if (rangeEnd == -1)
175     {
176         throw std::runtime_error("must specify range end explicitly for array type without size range expression: " + node->ToString());
177     }
178     std::unique_ptr<JsonObject> rangeResult(new JsonObject());
179     std::unique_ptr<JsonArray> rangeArray(new JsonArray());
180     for (int64_t index = rangeStart; index < rangeEnd; ++index)
181     {
182         IntegerDebugExprNode integerLiteralNode(index);
183         BoundIntegerLiteralNode boundIntegerLiteralNode(debugInfo->GetMainProject()->GetLongType(), index, &integerLiteralNode);
184         SubscriptDebugExprNode subscriptNode(subject->SourceNode()->Clone(), integerLiteralNode.Clone());
185         BoundSubscriptNode boundSubscriptNode(node->Type(), subject->Clone(), boundIntegerLiteralNode.Clone(), &subscriptNode);
186         boundSubscriptNode.Accept(*this);
187         std::unique_ptr<JsonValue> subscriptResult(result.release());
188         rangeArray->AddItem(std::unique_ptr<JsonValue>(subscriptResult.release()));
189     }
190     rangeResult->AddField(U"range", std::unique_ptr<JsonValue>(rangeArray.release()));
191     result.reset(rangeResult.release());
192 }

194 void DebugExpressionEvaluator::Evaluate(BoundDebugNode* node)
195 {
196     DebuggerVariable variable = debugger.GetNextDebuggerVariable();
197     std::string gdbExprString = node->GdbExprString();
198     GdbVarCreateCommand command(variable.GdbVarName(), "*", gdbExprString);
199     EvaluationGuard evaluationGuard(debugger);
200     bool succeeded = debugger.ExecuteGDBCommand(command);
201     result.reset(debugger.ReleaseResult());
202     bool uninitialized = false;
203     if (succeeded)
204     {
205         if (result->Type() == JsonValueType::object)
206         {
207             JsonObject* jsonObject = static_cast<JsonObject*>(result.get());
208             jsonObject->AddField(U"status", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(InitializationStatusStr(status)))));
209         }
210         AddTypes(node);
211         ContainerClassTemplateKind containerKind = GetContainerKind(node->Type());
212         if (containerKind != ContainerClassTemplateKind::notContainerClassTemplate)
213         {
214             Container* container = debugger.GetContainer(containerKind, node->SourceNode()->ToString());
215             if (result && result->Type() == JsonValueType::object)
216             {
217                 JsonObject* resultObject = static_cast<JsonObject*>(result.get());
218                 resultObject->AddField(U"container", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(ContainerName(container->GetKind())))));
219                 if (!uninitialized)
220                 {
221                     int64_t count = container->Count(node->SourceNode()->ToString());
222                     if (count != -1)
223                     {
224                         resultObject->AddField(U"count", std::unique_ptr<JsonValue>(new JsonString(ToUtf32(std::to_string(count)))));
225                     }
226                 }
227             }
228         }
229         std::unique_ptr<JsonValue> origResult(result.release());
230         GdbVarDeleteCommand varDeleteComand(variable.GdbVarName(), false);
231         bool succeeded = debugger.ExecuteGDBCommand(varDeleteComand);
232         result.reset(origResult.release());
233     }
234 }

236 void DebugExpressionEvaluator::AddTypes(BoundDebugNode* node)
237 {
238     if (result && result->Type() == JsonValueType::object)
239     {
240         JsonObject* resultObject = static_cast<JsonObject*>(result.get());
241         DITypeRef staticTypeRef(node->Type());
242         resultObject->AddField(U"static_type", staticTypeRef.ToJson());
243         DIType* dynamicType = debugger.GetDynamicType(node->Type(), node);
244         if (dynamicType)
245         {
246             DITypeRef dynamicTypeRef(dynamicType);
247             resultObject->AddField(U"dynamic_type", dynamicTypeRef.ToJson());
248         }
249     }
250 }

252 ContainerClassTemplateKind DebugExpressionEvaluator::GetContainerKind(DIType* type)
253 {
254     if (type->GetKind() == DIType::Kind::specializationType)
255     {
256         DIClassTemplateSpecializationType* specialization = static_cast<DIClassTemplateSpecializationType*>(type);
257         return specialization->GetContainerClassTemplateKind();
258     }
259     return ContainerClassTemplateKind::notContainerClassTemplate;
260 }
