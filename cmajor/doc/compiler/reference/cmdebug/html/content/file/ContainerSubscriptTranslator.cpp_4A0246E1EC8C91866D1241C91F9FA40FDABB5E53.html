 80 void ContainerSubscriptTranslator::Visit(BoundSubscriptNode& node)
 81 {
 82     BoundDebugNode* subject = node.Subject();
 83     DIType* subjectType = subject->Type();
 84     bool translated = false;
 85     if (subjectType->GetKind() == DIType::Kind::specializationType)
 86     {
 87         ContainerClassTemplateKind containerKind = GetContainerKind(subjectType);
 88         if (containerKind != ContainerClassTemplateKind::notContainerClassTemplate)
 89         {
 90             index = -1;
 91             bool prevEvaluateIndex = evaluateIndex;
 92             evaluateIndex = true;
 93             node.Index()->Accept(*this);
 94             evaluateIndex = prevEvaluateIndex;
 95             if (index == -1)
 96             {
 97                 throw std::runtime_error(node.ToString() + ": invalid subscript");
 98             }
 99             subject->Accept(*this);
100             std::string containerExpr = expression;
101             Container* container = debugger.GetContainer(containerKind, containerExpr);
102             expression = container->SubscriptExpressionString(containerExpr, index);
103             translated = true;
104         }
105     }
106     if (!translated)
107     {
108         node.Index()->Accept(*this);
109         std::string indexStr = expression;
110         node.Subject()->Accept(*this);
111         std::string subject = expression;
112         expression = subject + "[" + indexStr + "]";
113     }
114 }

117 {

120     node.RangeStart()->Accept(*this);
121     std::string rangeStart = expression;
122     node.RangeEnd()->Accept(*this);
123     std::string rangeEnd = expression;
124     expression = subject + "[" + rangeStart + ", " + rangeEnd + "]";
125 }

139 void ContainerSubscriptTranslator::Visit(BoundCastNode& node)

148 ContainerClassTemplateKind ContainerSubscriptTranslator::GetContainerKind(DIType* type)
149 {
150     if (type->GetKind() == DIType::Kind::specializationType)
151     {
152         DIClassTemplateSpecializationType* specialization = static_cast<DIClassTemplateSpecializationType*>(type);
153         return specialization->GetContainerClassTemplateKind();
154     }
155     return ContainerClassTemplateKind::notContainerClassTemplate;
156 }
