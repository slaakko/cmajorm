 6 [hpp]#include <sngcm/cmparser/ParserApi.hpp>
 7 [hpp]#include <sngcm/ast/Project.hpp>
 8 [cpp]#include <sngcm/cmlexer/ContainerFileLexer.hpp>
 9 [cpp]#include <sngcm/cmlexer/ContainerFileTokens.hpp>
10 
11 using namespace sngcm::ast;
12 using namespace ContainerFileTokens;
13 
14 parser api(SNGCM_PARSER_API) ProjectFileParser
15 {
16     uselexer ContainerFileLexer;
17     main;
18 
19     ProjectFile(std::string config, sngcm::ast::BackEnd backend, var std::unique_ptr project) : sngcm::ast::Project*
20         ::=
21         (
22             PROJECT QualifiedId:name! SEMICOLON!{ project.reset(new Project(name, lexer.FileName(), config, backend)); }
23             (Declaration:declaration{ project->AddDeclaration(declaration); })*
24         )
25         {
26             return project.release();
27         }
28         ;
29 
30     QualifiedId(var Span s) : std::u32string
31         ::=
32         (
33             ID{ s = span; } (DOT ID{ s.end = span.end; })*
34         )
35         {
36             return lexer.GetMatch(s);
37         }
38         ;
39 
40     Declaration : ProjectDeclaration*
41         ::= ReferenceDeclaration:referenceDeclaration{ return referenceDeclaration; }
42         |   SourceFileDeclaration:sourceFileDeclaration{ return sourceFileDeclaration; }
43         |   TextFileDeclaration:textFileDeclaration{ return textFileDeclaration; }
44         |   TargetDeclaration:targetDeclaration{ return targetDeclaration; }
45         ;
46 
47     ReferenceDeclaration(var std::string filePath) : ProjectDeclaration*
48         ::= REFERENCE FILEPATH!{ filePath = lexer.filePath; } SEMICOLON!{ return new sngcm::ast::ReferenceDeclaration(filePath); }
49         ;
50 
51     SourceFileDeclaration(var std::string filePath) : ProjectDeclaration*
52         ::= SOURCE FILEPATH!{ filePath = lexer.filePath; } SEMICOLON!{ return new sngcm::ast::SourceFileDeclaration(filePath); }
53         ;
54 
55     TextFileDeclaration(var std::string filePath) : ProjectDeclaration*
56         ::= TEXT FILEPATH!{ filePath = lexer.filePath; } SEMICOLON!{ return new sngcm::ast::TextFileDeclaration(filePath); }
57         ;
58 
59     TargetDeclaration : ProjectDeclaration*
60         ::= TARGET ASSIGN! Target:target! SEMICOLON!{ return new sngcm::ast::TargetDeclaration(target); }
61         ;
62 
63     Target : sngcm::ast::Target
64         ::= PROGRAM{ return sngcm::ast::Target::program; }
65         |   WINAPP{ return sngcm::ast::Target::winapp; }
66         |   LIBRARY{ return sngcm::ast::Target::library; }
67         |   WINLIB{ return sngcm::ast::Target::winlib; }
68         |   UNITTEST{ return sngcm::ast::Target::unitTest; }
69         ;
70 
71     ruleinfo
72     {
73         (ProjectFile, "project file"), (QualifiedId, "identifier"), (Declaration, "project file declaration"),
74         (ReferenceDeclaration, "reference declaration"), (SourceFileDeclaration, "source file declaration"),
75         (TextFileDeclaration, "text file declaration"), (TargetDeclaration, "target declaration"),
76         (Target, "target")
77     }
78 }
