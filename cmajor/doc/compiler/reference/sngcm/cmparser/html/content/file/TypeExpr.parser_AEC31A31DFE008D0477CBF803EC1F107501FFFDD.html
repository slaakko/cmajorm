29     TypeExpr(boost::uuids::uuid* moduleId, ParsingContext* ctx) : sngcm::ast::Node*
30         ::= PrefixTypeExpr(moduleId, ctx):prefixTypeExpr{ return prefixTypeExpr; }
31         ;
32 
33     PrefixTypeExpr(boost::uuids::uuid* moduleId, ParsingContext* ctx, var Span s) : sngcm::ast::Node*
34         ::= CONST{ s = span; } PostfixTypeExpr(moduleId, ctx):constTypeExpr{ s.end = span.end; return new ConstNode(s, *moduleId, constTypeExpr); }
35         |   PostfixTypeExpr(moduleId, ctx):postfixTypeExpr{ return postfixTypeExpr; }
36         ;
37 
38     PostfixTypeExpr(boost::uuids::uuid* moduleId, ParsingContext* ctx, var std::unique_ptr typeExpr, var Span s) : sngcm::ast::Node*
39         ::=
40         (   PrimaryTypeExpr(moduleId, ctx):primaryTypeExpr{ s = span; typeExpr.reset(primaryTypeExpr); }
41             (   DOT ID!{ s.end = span.end; typeExpr.reset(new DotNode(s, *moduleId, typeExpr.release(), new IdentifierNode(span, *moduleId, lexer.GetToken(pos)))); }
42             |   STAR{ s.end = span.end; typeExpr.reset(new PointerNode(s, *moduleId, typeExpr.release())); }
43             |   AMPAMP{ s.end = span.end; typeExpr.reset(new RValueRefNode(s, *moduleId, typeExpr.release())); }
44             |   AMP{ s.end = span.end; typeExpr.reset(new LValueRefNode(s, *moduleId, typeExpr.release())); }
45             |   LBRACKET Expression(moduleId, ctx):size? RBRACKET!{ s.end = span.end; typeExpr.reset(new ArrayNode(s, *moduleId, typeExpr.release(), size)); }
46             )*
47         )
48         {
49             return typeExpr.release();
50         }
51         ;
52 
53     PrimaryTypeExpr(boost::uuids::uuid* moduleId, ParsingContext* ctx) : sngcm::ast::Node*
54         ::= BasicType(moduleId):basicType{ return basicType; }
55         |   TemplateId(moduleId, ctx):templateId{ return templateId; }
56         |   ID{ return new IdentifierNode(span, *moduleId, lexer.GetToken(pos)); }
57         ;
