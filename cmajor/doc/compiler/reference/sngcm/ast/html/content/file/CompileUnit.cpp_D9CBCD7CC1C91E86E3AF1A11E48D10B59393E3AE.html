 42 void CompileUnitNode::ComputeLineStarts(const std::u32string& sourceFileContent)
 43 {
 44     int32_t index = 0;
 45     bool startOfLine = true;
 46     for (const char32_t& c : sourceFileContent)
 47     {
 48         if (startOfLine)
 49         {
 50             lineStarts.push_back(index);
 51             startOfLine = false;
 52         }
 53         if (c == U'\n')
 54         {
 55             startOfLine = true;
 56         }
 57         ++index;
 58     }
 59 }

 61 int CompileUnitNode::GetColumn(const Span& span) const
 62 {
 63     if (span.Valid())
 64     {
 65         int32_t pos = span.start;
 66         auto it = std::lower_bound(lineStarts.cbegin(), lineStarts.cend(), pos);
 67         if (it != lineStarts.cend())
 68         {
 69             int32_t start = *it;
 70             if (start != pos && it != lineStarts.cbegin())
 71             {
 72                 --it;
 73                 start = *it;
 74             }
 75             int column = 1 + pos - start;
 76             return column;
 77         }
 78     }
 79     return 1;
 80 }

 91 class NamespaceCombiner : public Visitor
 92 {
 93 public:
 94     void Visit(CompileUnitNode& compileUnitNode) override;

 96 };

 98 void NamespaceCombiner::Visit(CompileUnitNode& compileUnitNode)
 99 {
100     compileUnitNode.GlobalNs()->Accept(*this);
101 }
102 
103 void NamespaceCombiner::Visit(NamespaceNode& namespaceNode)
104 {
105     int n = namespaceNode.Members().Count();
106     for (int i = 0; i < n; ++i)
107     {
108         namespaceNode.Members()[i]->Accept(*this);
109     }
110     if (n == 1 && !namespaceNode.Id()->Str().empty())
111     {
112         Node* node = namespaceNode.Members()[0];
113         if (node->GetNodeType() == NodeType::namespaceNode)
114         {
115             std::unique_ptr<NamespaceNode> childNs(static_cast<NamespaceNode*>(namespaceNode.Members().Release(0)));
116             namespaceNode.Id()->SetStr(namespaceNode.Id()->Str() + U"." + childNs->Id()->Str());
117             namespaceNode.Members().Clear();
118             namespaceNode.Members().SetContent(childNs->Members().Content());
119             namespaceNode.Members().SetParent(&namespaceNode);
120         }
121     }
122 }

168 class UnnamedNamespaceProcessor : public Visitor
169 {
170 public:
171     UnnamedNamespaceProcessor();
172     void Visit(CompileUnitNode& compileUnitNode) override;
173     void Visit(NamespaceNode& namespaceNode) override;
174 private:
175     std::vector<NamespaceNode*> unnamedNamespaces;
176 };
177 
178 UnnamedNamespaceProcessor::UnnamedNamespaceProcessor()
179 {
180 }
181 
182 void UnnamedNamespaceProcessor::Visit(CompileUnitNode& compileUnitNode)
183 {
184     compileUnitNode.GlobalNs()->Accept(*this);
185     int index = 0;
186     for (NamespaceNode* unnamedNs : unnamedNamespaces)
187     {
188         CloneContext cloneContext;
189         IdentifierNode* unnamedNsId = static_cast<IdentifierNode*>(unnamedNs->Id()->Clone(cloneContext));
190         NamespaceImportNode* import = new NamespaceImportNode(compileUnitNode.GetSpan(), unnamedNsId);
191         compileUnitNode.GlobalNs()->Members().Insert(index, import);
192         ++index;
193     }
194 }
195 
196 void UnnamedNamespaceProcessor::Visit(NamespaceNode& namespaceNode)
197 {
198     if (namespaceNode.IsUnnamedNs())
199     {
200         unnamedNamespaces.push_back(&namespaceNode);
201     }
202     int n = namespaceNode.Members().Count();
203     for (int i = 0; i < n; ++i)
204     {
205         namespaceNode.Members()[i]->Accept(*this);
206     }
207 }
