 28 IdentifierNode::IdentifierNode(const Span& span_, const boost::uuids::uuid& moduleId_, NodeType nodeType_, const std::u32string& identifier_) : Node(nodeType_, span_, moduleId_), identifier(identifier_)
 29 {
 30     std::u32string result;
 31     for (char32_t c : identifier)
 32     {
 33         if (c != '`')
 34         {
 35             result.append(1, c);
 36         }
 37     }
 38     std::swap(result, identifier);
 39 }

 42 {
