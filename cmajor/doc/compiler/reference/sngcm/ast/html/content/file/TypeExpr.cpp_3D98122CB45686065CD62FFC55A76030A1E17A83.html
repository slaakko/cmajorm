172 ArrayNode::ArrayNode(const Span& span_, const boost::uuids::uuid& moduleId_, Node* subject_, Node* size_) :
173     Node(NodeType::arrayNode, span_, moduleId_), subject(subject_), size(size_)
174 {

177     {

179     }
180 }

182 Node* ArrayNode::Clone(CloneContext& cloneContext) const
183 {
184     Node* clonedSize = nullptr;
185     if (size)
186     {
187         clonedSize = size->Clone(cloneContext);
188     }
189     ArrayNode* clone = new ArrayNode(GetSpan(), ModuleId(), subject->Clone(cloneContext), clonedSize);
190     return clone;
191 }

196 }

198 void ArrayNode::Write(AstWriter& writer)
199 {
200     Node::Write(writer);
201     writer.Write(subject.get());
202     bool hasSize = size != nullptr;
203     writer.GetBinaryWriter().Write(hasSize);
204     if (hasSize)
205     {
206         writer.Write(size.get());
207     }
208 }
209 
210 void ArrayNode::Read(AstReader& reader)
211 {
212     Node::Read(reader);
213     subject.reset(reader.ReadNode());
214     subject->SetParent(this);
215     bool hasSize = reader.GetBinaryReader().ReadBool();
216     if (hasSize)
217     {
218         size.reset(reader.ReadNode());
219         size->SetParent(this);
220     }
221 }
222 
223 std::string ArrayNode::ToString() const
224 {
225     std::string s = subject->ToString();
226     s.append("[");
227     if (size)
228     {
229         s.append(size->ToString());
230     }
231     s.append("]");
232     return s;
233 }
