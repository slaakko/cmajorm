 37 Node* FunctionNode::Clone(CloneContext& cloneContext) const
 38 {
 39     Attributes* clonedAttributes = nullptr;
 40     if (attributes)
 41     {
 42         clonedAttributes = attributes->Clone();
 43     }
 44     Node* clonedReturnTypeExpr = nullptr;
 45     if (returnTypeExpr)
 46     {
 47         clonedReturnTypeExpr = returnTypeExpr->Clone(cloneContext);
 48     }
 49     FunctionNode* clone = new FunctionNode(cloneContext.MapSpan(GetSpan(), RootModuleId()), specifiers, clonedReturnTypeExpr, groupId, clonedAttributes);
 50     if (!cloneContext.InstantiateFunctionNode())
 51     {
 52         int nt = templateParameters.Count();
 53         for (int i = 0; i < nt; ++i)
 54         {
 55             clone->AddTemplateParameter(static_cast<TemplateParameterNode*>(templateParameters[i]->Clone(cloneContext)));
 56         }
 57     }
 58     int np = parameters.Count();
 59     for (int i = 0; i < np; ++i)
 60     {
 61         clone->AddParameter(static_cast<ParameterNode*>(parameters[i]->Clone(cloneContext)));
 62     }
 63     if (body)
 64     {
 65         if (cloneContext.InstantiateClassNode())
 66         {
 67             CloneContext bodyCloneContext;
 68             bodyCloneContext.SetSpanMapper(cloneContext.GetSpanMapper());
 69             clone->SetBodySource(static_cast<CompoundStatementNode*>(body->Clone(bodyCloneContext)));
 70         }
 71         else
 72         {
 73             clone->SetBody(static_cast<CompoundStatementNode*>(body->Clone(cloneContext)));
 74         }
 75     }
 76     if (whereConstraint)
 77     {
 78         clone->SetConstraint(static_cast<WhereConstraintNode*>(whereConstraint->Clone(cloneContext)));
 79     }
 80     clone->SetGroupIdSpan(cloneContext.MapSpan(groupIdSpan, RootModuleId()));
 81     clone->SetSpecifierSpan(cloneContext.MapSpan(specifierSpan, RootModuleId()));
 82     return clone;
 83 }

 85 void FunctionNode::CloneContent(FunctionNode* clone, CloneContext& cloneContext) const
 86 {
 87     if (attributes)
 88     {
 89         clone->attributes.reset(attributes->Clone());
 90     }
 91     clone->specifiers = specifiers;
 92     Node* clonedReturnTypeExpr = nullptr;
 93     if (returnTypeExpr)
 94     {
 95         clone->SetReturnTypeExpr(returnTypeExpr->Clone(cloneContext));
 96     }
 97     clone->groupId = groupId;
 98     int nt = templateParameters.Count();
 99     for (int i = 0; i < nt; ++i)
100     {
101         clone->AddTemplateParameter(static_cast<TemplateParameterNode*>(templateParameters[i]->Clone(cloneContext)));
102     }
103     int np = parameters.Count();
104     for (int i = 0; i < np; ++i)
105     {
106         clone->AddParameter(static_cast<ParameterNode*>(parameters[i]->Clone(cloneContext)));
107     }
108     if (body)
109     {
110         if (cloneContext.InstantiateClassNode())
111         {
112             CloneContext bodyCloneContext;
113             bodyCloneContext.SetSpanMapper(cloneContext.GetSpanMapper());
114             clone->SetBodySource(static_cast<CompoundStatementNode*>(body->Clone(bodyCloneContext)));
115         }
116         else
117         {
118             clone->SetBody(static_cast<CompoundStatementNode*>(body->Clone(cloneContext)));
119         }
120     }
121     if (whereConstraint)
122     {
123         clone->SetConstraint(static_cast<WhereConstraintNode*>(whereConstraint->Clone(cloneContext)));
124     }
125     clone->SetGroupIdSpan(cloneContext.MapSpan(groupIdSpan, RootModuleId()));
126     clone->SetSpecifierSpan(cloneContext.MapSpan(specifierSpan, RootModuleId()));
127 }

134 void FunctionNode::Write(AstWriter& writer)
135 {
136     Node::Write(writer);
137     bool hasAttributes = attributes != nullptr;
138     writer.GetBinaryWriter().Write(hasAttributes);
139     if (hasAttributes)
140     {
141         attributes->Write(writer);
142     }
143     writer.Write(specifiers);
144     bool hasReturnTypeExpr = returnTypeExpr != nullptr;
145     writer.GetBinaryWriter().Write(hasReturnTypeExpr);
146     if (hasReturnTypeExpr)
147     {
148         writer.Write(returnTypeExpr.get());
149     }
150     writer.GetBinaryWriter().Write(groupId);
151     templateParameters.Write(writer);
152     parameters.Write(writer);
153     bool hasConstraint = whereConstraint != nullptr;
154     writer.GetBinaryWriter().Write(hasConstraint);
155     if (hasConstraint)
156     {
157         writer.Write(whereConstraint.get());
158     }
159     bool hasBody = body != nullptr;
160     writer.GetBinaryWriter().Write(hasBody);
161     if (hasBody)
162     {
163         writer.Write(body.get());
164     }
165     bool hasBodySource = bodySource != nullptr;
166     writer.GetBinaryWriter().Write(hasBodySource);
167     if (hasBodySource)
168     {
169         writer.Write(bodySource.get());
170     }
171     writer.Write(groupIdSpan);
172     writer.Write(specifierSpan);
173 }

175 void FunctionNode::Read(AstReader& reader)
176 {
177     Node::Read(reader);
178     bool hasAttributes = reader.GetBinaryReader().ReadBool();
179     if (hasAttributes)
180     {
181         attributes.reset(new Attributes());
182         attributes->Read(reader);
183     }
184     specifiers = reader.ReadSpecifiers();
185     bool hasReturnTypeExpr = reader.GetBinaryReader().ReadBool();
186     if (hasReturnTypeExpr)
187     {
188         returnTypeExpr.reset(reader.ReadNode());
189         returnTypeExpr->SetParent(this);
190     }
191     groupId = reader.GetBinaryReader().ReadUtf32String();
192     templateParameters.Read(reader);
193     templateParameters.SetParent(this);
194     parameters.Read(reader);
195     parameters.SetParent(this);
196     bool hasConstraint = reader.GetBinaryReader().ReadBool();
197     if (hasConstraint)
198     {
199         whereConstraint.reset(reader.ReadWhereConstraintNode());
200     }
201     bool hasBody = reader.GetBinaryReader().ReadBool();
202     if (hasBody)
203     {
204         body.reset(reader.ReadCompoundStatementNode());
205         body->SetParent(this);
206     }
207     bool hasBodySource = reader.GetBinaryReader().ReadBool();
208     if (hasBodySource)
209     {
210         bodySource.reset(reader.ReadCompoundStatementNode());
211         bodySource->SetParent(this);
212     }
213     groupIdSpan = reader.ReadSpan();
214     specifierSpan = reader.ReadSpan();
215 }
