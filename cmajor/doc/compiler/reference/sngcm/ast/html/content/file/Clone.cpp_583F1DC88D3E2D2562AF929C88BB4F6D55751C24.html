28 Span CloneContext::MapSpan(const Span& span, const boost::uuids::uuid& rootModuleId)
29 {
30     if (!rootModuleId.is_nil())
31     {
32         if (!spanMapper)
33         {
34             throw std::runtime_error("CloneContext: root module not null and span mapper not set");
35         }
36         return spanMapper->MapSpan(span, rootModuleId);
37     }
38     else
39     {
40         return span;
41     }
42 }
