 9     public Section MakePoolSection(const List<string>& argv)
10     {
11         Section poolSection;
12         long n = argv.Count() + 1;
13         for (long i = 0; i < n; ++i;)
14         {
15             poolSection.EmitULong(0u);
16         }
17         List<ulong> argAddresses;
18         for (const string& arg : argv)
19         {
20             argAddresses.Add(poolSegmentBaseAddress + poolSection.Address());
21             poolSection.EmitString(arg);
22         }
23         poolSection.pos = 0u;
24         for (ulong argAddr : argAddresses)
25         {
26             poolSection.EmitULong(argAddr);
27         }
28         return poolSection;
29     }

31     public int Load(const string& programFileName, const List<string>& programArguments)
32     {
33         UniquePtr<BinaryFile> binaryFile = ReadBinaryFile(programFileName);
34         if (binaryFile.Get() is ExecutableFile*)
35         {
36             Machine& machine = GetMachine();
37             ProcessTable& processTable = GetProcessTable();
38             ExecutableFile* executable = cast<ExecutableFile*>(binaryFile.Get());
39             Section* codeSection = executable->GetCodeSection();
40             Section* dataSection = executable->GetDataSection();
41             string name = Path.GetFileName(programFileName);
42             List<string> argv;
43             argv.Add(name);
44             for (const string& arg : programArguments)
45             {
46                 argv.Add(arg);
47             }
48             Process* process = CreateSingleProcess(machine, processTable, name, codeSection->BaseAddress(), codeSection->Length(), dataSection->Length(), executable->PoolSize(), executable->StackSize());
49             if (codeSection->Length() > 0u)
50             {
51                 int result = WriteProcessMemory(machine, process, codeSection->BaseAddress(), codeSection->data.Begin().Ptr(), cast<ulong>(codeSection->data.Count()), 
52                     Protection.execute);
53                 if (result != 0)
54                 {
55                     Panic("loader error: error writing to code segment of program '" + programFileName + "'");
56                 }
57             }
58             if (dataSection->Length() > 0u)
59             {
60                 int result = WriteProcessMemory(machine, process, dataSection->BaseAddress(), dataSection->data.Begin().Ptr(), cast<ulong>(dataSection->data.Count()), 
61                     Protection.write);
62                 if (result != 0)
63                 {
64                     Panic("loader error: error writing to data segment of program '" + programFileName + "'");
65                 }
66             }
67             Section poolSection = MakePoolSection(argv);
68             if (!poolSection.data.IsEmpty())
69             {
70                 int result = WriteProcessMemory(machine, process, poolSegmentBaseAddress, poolSection.data.Begin().Ptr(), cast<ulong>(poolSection.data.Count()), 
71                     Protection.write);
72                 if (result != 0)
73                 {
74                     Panic("loader error: error writing to pool segment of program '" + programFileName + "'");
75                 }
76                 int ax = cast<int>(argv.Count());
77                 ulong bx = poolSegmentBaseAddress;
78                 result = WriteProcessMemory(machine, process, process->regAXAddress, cast<ulong>(ax), 8u);
79                 if (result != 0)
80                 {
81                     Panic("loader error: error writing to stack segment of program '" + programFileName + "'");
82                 }
83                 result = WriteProcessMemory(machine, process, process->regBXAddress, bx, 8u);
84                 if (result != 0)
85                 {
86                     Panic("loader error: error writing to stack segment of program '" + programFileName + "'");
87                 }
88             }
89             return process->pid;
90         }
91         else
92         {
93             Panic("file '" + programFileName + "' is not a cmsx executable");
94         }
95         return -1;
96     }
