551         if (process->state == Process.State.created || process->state == Process.State.readyToRun || process->state == Process.State.running || process->state == Process.State.asleep)
552         {
553             Memory& mem = machine.GetMemory();
554             Registers& regs = machine.GetRegisters();
555             ulong prevRV = regs.GetSpecial(Registers.rV);
556             MemoryTable& memoryTable = process->memoryTable;
557             regs.SetSpecial(Registers.rV, memoryTable.virtualTranslationRegisterValue);
558             ulong address = targetAddress;
559             for (ulong i = 0u; i < size; ++i;)
560             {
561                 mem.WriteByte(address, *source, access);
562                 ++address;
563                 ++source;
564             }
565             regs.SetSpecial(Registers.rV, prevRV);
566             return 0;
567         }
568         else
569         {
570             return -1;
571         }
572     }

728     public nothrow void Schedule(Machine& machine, ProcessTable& processTable)
729     {
730         Processor& processor = machine.GetProcessor();
731         Process* running =  processTable.GetRunning();
732         Process* prevRunning = running;
733         Process*  idle = processTable.GetIdle();
734         if (running == idle)
735         {
736             running->sp = processor.SaveContext(running->memoryTable.virtualTranslationRegisterValue, running->sp, &running->regAXAddress, &running->regBXAddress);
737             running = null;
738             processTable.SetRunning(running);
739         }
740         LinkReadyProcessesToRunningList(processTable);
741         running =  processTable.GetRunning();
742         if (running == null)
743         {
744             running = idle;
745         }
746         else
747         {
748             running = running->nextReadyToRun;
749             if (running->state != Process.State.readyToRun && running->state != Process.State.running)
750             {
751                 running = idle;
752             }
753         }
754         if (prevRunning != running)
755         {
756             if (prevRunning != null)
757             {
758                 if (prevRunning->state != Process.State.zombie)
759                 {
760                     prevRunning->sp = processor.SaveContext(prevRunning->memoryTable.virtualTranslationRegisterValue, prevRunning->sp, &prevRunning->regAXAddress, &prevRunning->regBXAddress);
761                 }
762                 if (prevRunning->state == Process.State.running)
763                 {
764                     prevRunning->state = Process.State.readyToRun;
765                 }
766                 else
767                 {
768                     RemoveProcessFromList(&prevRunning);
769                 }
770             }
771             processor.UnsaveContext(running->memoryTable.virtualTranslationRegisterValue, running->sp);
772         }
773         else
774         {
775             processor.GetRegisters().SetSpecial(Registers.rV, running->memoryTable.virtualTranslationRegisterValue);
776         }
777         processTable.SetRunning(running);
778         running->state = Process.State.running;
779     }
