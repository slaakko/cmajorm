28     public void SoftwareInterruptHandler(Machine& machine, byte irq)
29     {
30         Registers& regs = machine.GetRegisters();
31         ulong prevRV = regs.GetSpecial(Registers.rV);
32         ulong kernelRV = MakeVirtualTranslationRegisterValue(machine.GetKernelRootPageAddress(), kernelAddressSpaceNumber);
33         regs.SetSpecial(Registers.rV, kernelRV);
34         ulong trapValue = regs.GetSpecial(Registers.rX);
35         byte trapX = cast<byte>((trapValue & trapXMask) >> trapXShift);
36         byte trapY = cast<byte>((trapValue & trapYMask) >> trapYShift);
37         byte trapZ = cast<byte>((trapValue & trapZMask) >> trapZShift);
38         ulong trapAX = regs.Get(regAX);
39         ulong trapBX = regs.Get(regBX);
40         ulong trapCX = regs.Get(regCX);
41         ulong trapDX = regs.Get(regDX);
42         ulong trapEX = regs.Get(regEX);
43         ulong trapOffset = cast<ulong>(trapY) << 3u;
44         ulong trapBaseAddress = regs.GetSpecial(Registers.rT);
45         ulong trapHandlerPtrAddress = trapTableBaseAddress + trapOffset;
46         ulong trapHandlerAddress = machine.GetMemory().ReadULong(trapHandlerPtrAddress, Protection.read);
47         regs.SetSpecial(Registers.rV, prevRV);
48         if (trapHandlerAddress == 0u)
49         {
50             Panic("handler for trap " + ToString(trapY) + " is null");
51         }
52         TrapHandler trap = cast<TrapHandler>(cast<void*>(trapHandlerAddress));
53         long result = trap(machine, trapX, trapY, trapZ, trapAX, trapBX, trapCX, trapDX, trapEX);
54         ulong currentRV = regs.GetSpecial(Registers.rV);
55         if (currentRV == prevRV)
56         {
57             ulong returnValue = cast<ulong>(result);
58             regs.Set(regAX, returnValue);
59         }
60     }
