 98     {
 99         public nothrow DebuggerInputFile() : base("CONSOLE INPUT (DEBUGGER)")
100         {
101         }
102         public override long Read(Machine& machine, Process* reader, ulong bufferAddress, long count)
103         {
104             try
105             {
106                 if (bufferedInput.IsEmpty())
107                 {
108                     Kernel& kernel = Kernel.Instance();
109                     Debugger* debugger = kernel.GetDebugger();
110                     bufferedInput = debugger->ReadInputLine();
111                     if (bufferedInput.IsEmpty())
112                     {
113                         if (debugger->Eof())
114                         {
115                             debugger->ResetEof();
116                             return 0;
117                         }
118                     }
119                 }
120                 long n = Min(count, bufferedInput.Length());
121                 WriteProcessMemory(machine, reader, bufferAddress, cast<byte*>(cast<void*>(bufferedInput.Chars())), cast<ulong>(n), Protection.write);
122                 bufferedInput = bufferedInput.Substring(n);
123                 return n;
124             }
125             catch (const Exception& ex)
126             {
127                 throw SystemError(EFAIL, ex.Message());
128             }
129             return -1;
130         }

169         {

171         public override long Read(Machine& machine, Process* reader, ulong bufferAddress, long count)
172         {
173             try
174             {
175                 Kernel& kernel = Kernel.Instance();
176                 Session* session = kernel.CurrentSession();
177                 if (session->fgpid == reader->pid)
178                 {
179                     if (bufferedInput.IsEmpty())
180                     {
181                         ConsoleDriver* consoleDriver = kernel.GetConsoleDriver();
182                         if (consoleDriver->HasLine())
183                         {
184                             bufferedInput = consoleDriver->GetLine();
185                         }
186                         else if (consoleDriver->Eof())
187                         {
188                             consoleDriver->ResetEof();
189                             return 0;
190                         }
191                         else
192                         {
193                             EnqueueReadRequest(reader, bufferAddress, count);
194                             return RESULT_BLOCKED;
195                         }
196                     }
197                     long n = Min(count, bufferedInput.Length());
198                     WriteProcessMemory(machine, reader, bufferAddress, cast<byte*>(cast<void*>(bufferedInput.Chars())), cast<ulong>(n), Protection.write);
199                     bufferedInput = bufferedInput.Substring(n);
200                     return n;
201                 }
202                 else
203                 {
204                     EnqueueReadRequest(reader, bufferAddress, count);
205                     return RESULT_BLOCKED;
206                 }
207             }
208             catch (const Exception& ex)
209             {
210                 throw SystemError(EFAIL, ex.Message());
211             }
212             return -1;
213         }

263     public class ProcessFileTable
264     {
265         public void SetFile(int fd, File* file)
266         {
267             if (fd < 0)
268             {
269                 throw SystemError(EBADF, "value negative");
270             }
271             else if (fd >= openFileMax)
272             {
273                 throw SystemError(EBADF, "value exceeds max number of open files");
274             }
275             files[fd] = file;
276         }
277         public File* GetFile(int fd)
278         {
279             if (fd < 0)
280             {
281                 throw SystemError(EBADF, "value negative");
282             }
283             else if (fd >= openFileMax)
284             {
285                 throw SystemError(EBADF, "value exceeds max number of open files");
286             }
287             File* file = files[fd];
288             if (file == null)
289             {
290                 throw SystemError(EBADF, "file not open");
291             }
292             return file;
293         }
294         public void ShareFilesTo(ProcessFileTable& that)
295         {
296             for (long i = 0; i < openFileMax; ++i;)
297             {
298                 File* file = files[i];
299                 if (file != null)
300                 {
301                     that.files[i] = file->Share();
302                 }
303                 else
304                 {
305                     that.files[i] = null;
306                 }
307             }
308         }
309         public void ReleaseFiles()
310         {
311             for (long i = 0; i < openFileMax; ++i;)
312             {
313                 File* file = files[i];
314                 if (file != null)
315                 {
316                     file->Release();
317                     files[i] = null;
318                 }
319             }
320         }
321         private File*[openFileMax] files;
322     }
