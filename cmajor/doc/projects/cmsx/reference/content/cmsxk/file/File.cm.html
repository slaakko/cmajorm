505     public class FilePtr
506     {
507         public nothrow FilePtr(INode* inode_, OpenFlags flags_) : inode(inode_), pos(0), flags(flags_), useCount(1)
508         {
509             if ((flags & OpenFlags.append) != OpenFlags.none)
510             {
511                 pos = inode->GetFileSize();
512             }
513         }
514         public long Write(Machine& machine, Process* writer, ulong bufferAddress, long count)
515         {
516             while (inode->GetFlag(INode.Flags.locked))
517             {
518                 inode->AddWaitingProcess(writer);
519                 SleepProcess(writer, inodeUnlockedEvent, cast<ulong>(cast<void*>(inode)), 0u);
520             }
521             long bytesWritten = 0;
522             {
523                 INodeLock lock(inode);
524                 ulong offset = 0u;
525                 while (count > 0)
526                 {
527                     int blockNumber = 0;
528                     int blockOffset = 0;
529                     GetBlockManager().GetBlockNumber(inode, pos, blockNumber, blockOffset, true);
530                     long bytesToWrite = Min(count, blockSize - blockOffset);
531                     Block* block = null;
532                     if (blockOffset != 0 || bytesToWrite != blockSize)
533                     {
534                         block = GetBlockManager().ReadBlock(BlockKey(blockNumber, inode->Key().fsNumber, BlockKind.fileBlock), null);
535                     }
536                     else
537                     {
538                         block = GetBlockManager().GetBlock(BlockKey(blockNumber, inode->Key().fsNumber, BlockKind.fileBlock), null);
539                     }
540                     #assert(block is FileBlock*);
541                     BlockPutter blockPutter(block);
542                     FileBlock* fileBlock = cast<FileBlock*>(block);
543                     UniquePtr<byte> buffer(cast<byte*>(RtMemAlloc(bytesToWrite)));
544                     ReadProcessMemory(machine, writer, bufferAddress + offset, buffer.Get(), cast<ulong>(bytesToWrite));
545                     MemoryReader reader(buffer.Get(), bytesToWrite);
546                     for (long i = 0; i < bytesToWrite; ++i;)
547                     {
548                         byte x = reader.ReadByte();
549                         fileBlock->SetByte(blockOffset + i, x);
550                     }
551                     fileBlock->SetFlag(Block.Flags.dirty);
552                     GetBlockManager().WriteBlock(block, null);
553                     count = count - bytesToWrite;
554                     pos = pos + bytesToWrite;
555                     offset = offset + cast<ulong>(bytesToWrite);
556                     bytesWritten = bytesWritten + bytesToWrite;
557                 }
558                 if (pos > inode->GetFileSize())
559                 {
560                     inode->SetFileSize(pos);
561                 }
562             }
563             inode->SetMTime();
564             return bytesWritten;
565         }
566         public long Read(Machine& machine, Process* reader, ulong bufferAddress, long count)
567         {
568             while (inode->GetFlag(INode.Flags.locked))
569             {
570                 inode->AddWaitingProcess(reader);
571                 SleepProcess(reader, inodeUnlockedEvent, cast<ulong>(cast<void*>(inode)), 0u);
572             }
573             long bytesRead = 0;
574             {
575                 INodeLock lock(inode);
576                 ulong offset = 0u;
577                 while (count > 0)
578                 {
579                     int blockNumber = 0;
580                     int blockOffset = 0;
581                     GetBlockManager().GetBlockNumber(inode, pos, blockNumber, blockOffset, false);
582                     long bytesToRead = Min(count, blockSize - blockOffset);
583                     long bytesLeft = Max(inode->GetFileSize() - pos, 0);
584                     bytesToRead = Min(bytesToRead, bytesLeft);
585                     if (bytesToRead == 0)
586                     {
587                         break;
588                     }
589                     Block* block = null;
590                     UniquePtr<byte> buffer(cast<byte*>(RtMemAlloc(bytesToRead)));
591                     MemoryWriter writer(buffer.Get(), bytesToRead);
592                     if (blockNumber != 0)
593                     {
594                         block = GetBlockManager().ReadBlock(BlockKey(blockNumber, inode->Key().fsNumber, BlockKind.fileBlock), null);
595                         #assert(block is FileBlock*);
596                         BlockPutter blockPutter(block);
597                         FileBlock* fileBlock = cast<FileBlock*>(block);
598                         for (long i = 0; i < bytesToRead; ++i;)
599                         {
600                             byte x = fileBlock->GetByte(blockOffset + i);
601                             writer.Write(x);
602                         }
603                     }
604                     else
605                     {
606                         for (long i = 0; i < bytesToRead; ++i;)
607                         {
608                             byte x = 0u;
609                             writer.Write(x);
610                         }
611                     }
612                     WriteProcessMemory(machine, reader, bufferAddress + offset, buffer.Get(), cast<ulong>(bytesToRead), Protection.write);
613                     count = count - bytesToRead;
614                     pos = pos + bytesToRead;
615                     offset = offset + cast<ulong>(bytesToRead);
616                     bytesRead = bytesRead + bytesToRead;
617                 }
618             }
619             inode->SetATime();
620             return bytesRead;
621         }
622         public FilePtr* Share()
623         {
624             ++useCount;
625             return this;
626         }
627         public void Release()
628         {
629             --useCount;
630             if (useCount == 0)
631             {
632                 delete this;
633             }
634         }
635         public inline nothrow OpenFlags Flags() const
636         {
637             return flags;
638         }
639         private INode* inode;
640         private long pos;
641         private OpenFlags flags;
642         private int useCount;
643     }

645     public class ProcessFileTable
646     {
647         public void SetFile(int fd, File* file)
648         {
649             if (fd < 0)
650             {
651                 throw SystemError(EBADF, "value negative");
652             }
653             else if (fd >= openFileMax)
654             {
655                 throw SystemError(EBADF, "value exceeds max number of open files");
656             }
657             files[fd] = file;
658         }
659         public File* GetFile(int fd)
660         {
661             if (fd < 0)
662             {
663                 throw SystemError(EBADF, "value negative");
664             }
665             else if (fd >= openFileMax)
666             {
667                 throw SystemError(EBADF, "value exceeds maximum number of open files");
668             }
669             File* file = files[fd];
670             if (file == null)
671             {
672                 throw SystemError(EBADF, "file not open");
673             }
674             return file;
675         }
676         public int GetEmptyFileSlot()
677         {
678             for (long i = 0; i < openFileMax; ++i;)
679             {
680                 File* file = files[i];
681                 if (file == null)
682                 {
683                     return cast<int>(i);
684                 }
685             }
686             throw SystemError(ERLIMITEXCEEDED, "maximum number of open files (" + ToString(openFileMax) + ") exceeded");
687         }
688         public void ShareFilesTo(ProcessFileTable& that)
689         {
690             for (long i = 0; i < openFileMax; ++i;)
691             {
692                 File* file = files[i];
693                 if (file != null)
694                 {
695                     that.files[i] = file->Share();
696                 }
697                 else
698                 {
699                     that.files[i] = null;
700                 }
701             }
702         }
703         public void ReleaseFiles()
704         {
705             for (long i = 0; i < openFileMax; ++i;)
706             {
707                 File* file = files[i];
708                 if (file != null)
709                 {
710                     file->Release();
711                     files[i] = null;
712                 }
713             }
714         }
715         private File*[openFileMax] files;
716     }
