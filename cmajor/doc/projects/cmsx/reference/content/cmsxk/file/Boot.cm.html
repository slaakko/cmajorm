11     public nothrow void InitializeInterruptVector(Machine& machine)
12     {
13         Memory& mem = machine.GetMemory();
14         for (byte i = 0u; i < irqMax; ++i;)
15         {
16             mem.WriteULong(MakeInterruptHandlerPtrAddress(i), 0u, Protection.write);
17         }
18         InterruptHandler softwareInterruptHandler = SoftwareInterruptHandler;
19         ulong softwareInterruptHandlerAddress = cast<ulong>(cast<void*>(softwareInterruptHandler));
20         mem.WriteULong(MakeInterruptHandlerPtrAddress(irqSoftware), softwareInterruptHandlerAddress, Protection.write);
21         InterruptHandler intervalInterruptHandler = IntervalInterruptHandler;
22         ulong intervalInterruptHandlerAddress = cast<ulong>(cast<void*>(intervalInterruptHandler));
23         mem.WriteULong(MakeInterruptHandlerPtrAddress(irqInterval), intervalInterruptHandlerAddress, Protection.write);
24         InterruptHandler clockHandler = ClockHandler;
25         ulong clockHandlerAddress = cast<ulong>(cast<void*>(clockHandler));
26         mem.WriteULong(MakeInterruptHandlerPtrAddress(irqClock), clockHandlerAddress, Protection.write);
27         InterruptHandler pageFaultHandler = PageFaultHandler;
28         ulong pageFaultHandlerAddress = cast<ulong>(cast<void*>(pageFaultHandler));
29         mem.WriteULong(MakeInterruptHandlerPtrAddress(irqR), pageFaultHandlerAddress, Protection.write);
30         mem.WriteULong(MakeInterruptHandlerPtrAddress(irqW), pageFaultHandlerAddress, Protection.write);
31         mem.WriteULong(MakeInterruptHandlerPtrAddress(irqX), pageFaultHandlerAddress, Protection.write);
32         mem.WriteULong(MakeInterruptHandlerPtrAddress(irqCOW), pageFaultHandlerAddress, Protection.write);
33         InterruptHandler securityViolationHandler = SecurityViolationHandler;
34         ulong securityViolationHandlerAddress = cast<ulong>(cast<void*>(securityViolationHandler));
35         mem.WriteULong(MakeInterruptHandlerPtrAddress(irqS), securityViolationHandlerAddress, Protection.write);
36         machine.GetRegisters().SetSpecial(Registers.rTT, interruptVectorBaseAddress);
37         machine.GetRegisters().SetSpecial(Registers.rT, trapTableBaseAddress);
38     }

40     public nothrow void InitializeSystemCalls(Machine& machine)
41     {
42         Memory& mem = machine.GetMemory();
43         for (int i = 0; i < maxTraps; ++i;)
44         {
45             mem.WriteULong(MakeSystemCallPtrAddress(i), 0u, Protection.write);
46         }
47         TrapHandler exitSystemCallHandler = Exit;
48         ulong exitSystemCallAddress = cast<ulong>(cast<void*>(exitSystemCallHandler));
49         mem.WriteULong(MakeSystemCallPtrAddress(exitSystemCall), exitSystemCallAddress, Protection.write);
50         TrapHandler waitSystemCallHandler = Wait;
51         ulong waitSystemCallAddress = cast<ulong>(cast<void*>(waitSystemCallHandler));
52         mem.WriteULong(MakeSystemCallPtrAddress(waitSystemCall), waitSystemCallAddress, Protection.write);
53         TrapHandler forkSystemCallHandler = Fork;
54         ulong forkSystemCallAddress = cast<ulong>(cast<void*>(forkSystemCallHandler));
55         mem.WriteULong(MakeSystemCallPtrAddress(forkSystemCall), forkSystemCallAddress, Protection.write);
56         TrapHandler writeSystemCallHandler = Write;
57         ulong writeSystemCallAddress = cast<ulong>(cast<void*>(writeSystemCallHandler));
58         mem.WriteULong(MakeSystemCallPtrAddress(writeSystemCall), writeSystemCallAddress, Protection.write);
59     }

61     public void Boot(Machine& machine)
62     {
63         if (Machine.GetFlag(Machine.Flags.verbose))
64         {
65             Console.Out() << "Cmajor System X kernel version " << KernelVersion() << " booting...";
66         }
67         Registers& regs = machine.GetRegisters();
68         DisableAllInterrupts(regs);
69         InitializeInterruptVector(machine);
70         InitializeSystemCalls(machine);
71         Kernel& kernel = Kernel.Instance();
72         ProcessTable& processTable = kernel.GetProcessTable();
73         int pid = -1;
74         if (Machine.GetFlag(Machine.Flags.runProgram))
75         {
76             pid = Load(kernel.GetProgramFileName(), kernel.GetProgramArguments());
77             kernel.SetProgramPID(pid);
78         }
79         else
80         {
81             CreateInitAndIdleProcesses(machine, processTable);
82         }
83         Schedule(machine, processTable);
84         SetProcessorToUserMode(regs);
85         EnableAllInterrupts(regs);
86         if (Machine.GetFlag(Machine.Flags.verbose))
87         {
88             Console.Out() << "\b\b\b, done." << endl();
89         }
90         if (Machine.GetFlag(Machine.Flags.runProgram))
91         {
92             if (Machine.GetFlag(Machine.Flags.verbose))
93             {
94                 Console.Out() << "running program '" << kernel.GetProgramFileName() << "' as process " << pid << endl();
95             }
96         }
97     }
