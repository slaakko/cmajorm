159         {
160             int n = superBlock->GetNumINodeBitmapBlocks();

162             for (int i = 0; i < n; ++i;)
163             {
164                 bitmapBlocks.Add(UniquePtr<BitmapBlock>(new BitmapBlock(firstNumber + i)));
165             }
166         }
167         public int GetFreeINodeNumber()
168         {
169             int n = cast<int>(bitmapBlocks.Count());
170             for (int i = 0; i < n; ++i;)
171             {
172                 BitmapBlock* block = bitmapBlocks[i].Get();
173                 int index = block->GetFirstZeroBitIndex();
174                 if (index != -1)
175                 {
176                     block->SetBit(index, superBlock);
177                     int inodeNumber = numBitsInBitmapBlock * i + index;
178                     return inodeNumber;
179                 }
180             }
181             throw SystemError(RLIMITEXCEEDED, "no free inodes");
182         }
183         public void SetFreeINodeNumber(int inodeNumber)
184         {
185             int blockNumber = inodeNumber / numBitsInBitmapBlock;
186             int index = inodeNumber % numBitsInBitmapBlock;
187             BitmapBlock* block = bitmapBlocks[blockNumber].Get();
188             block->ResetBit(index, superBlock);
189         }
