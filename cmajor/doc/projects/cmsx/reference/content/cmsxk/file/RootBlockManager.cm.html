 18             }

 20             Block* sb = fs->GetBlockManager()->ReadBlock(BlockKey(0, fsNumber), null);

 22             SuperBlock* superBlock = cast<SuperBlock*>(sb);
 23             int firstNumber = superBlock->GetFirstBlockBitmapBlockNumber();

 31                     block->Clear();
 32                 }

 36                 }
