303         public override int ReadDirectory(Process* process, DirectoryFile& dirFile, ulong inodeNumberAddress, ulong entryNameAddress)
304         {
305             INode* dirINode = dirFile.DirINode();
306             if (Log())
307             {
308                 LogMessage("fs.root.readdirectory", dirINode->ToString());
309             }
310             while (dirINode->GetFlag(INode.Flags.locked))
311             {
312                 dirINode->AddWaitingProcess(process);
313                 SleepProcess(process, inodeUnlockedEvent, cast<ulong>(cast<void*>(dirINode)), 0u);
314             }
315             {
316                 INodeLock lock(dirINode);
317                 while (dirFile.Offset() < dirINode->GetFileSize())
318                 {
319                     int blockNumber = 0;
320                     int blockOffset = 0;
321                     GetBlockManager().GetBlockNumber(dirINode, dirFile.Offset(), blockNumber, blockOffset, false);
322                     if (blockNumber == 0)
323                     {
324                         return 0;
325                     }
326                     Block* block = GetBlockManager().ReadBlock(BlockKey(blockNumber, dirINode->Key().fsNumber, BlockKind.directoryBlock), null);
327                     BlockPutter putter(block);
328                     #assert(block is DirectoryBlock*);
329                     DirectoryBlock* directoryBlock = cast<DirectoryBlock*>(block);
330                     for (int index = blockOffset / directoryEntrySize; index < numDirectoryEntriesInBlock && dirFile.Offset() < dirINode->GetFileSize(); ++index;)
331                     {
332                         const DirectoryEntry& directoryEntry = directoryBlock->GetDirectoryEntry(index);
333                         if (directoryEntry.inodeNumber == 0)
334                         {
335                             dirFile.SetOffset(dirFile.Offset() + directoryEntrySize);
336                         }
337                         else
338                         {
339                             WriteProcessMemory(GetMachine(), process, inodeNumberAddress, cast<ulong>(directoryEntry.inodeNumber), 4u, Protection.write);
340                             WriteProcessMemory(GetMachine(), process, entryNameAddress, cast<byte*>(cast<void*>(&directoryEntry.name[0])), cast<ulong>(nameMax), Protection.write);
341                             dirFile.SetOffset(dirFile.Offset() + directoryEntrySize);
342                             return 1;
343                         }
344                     }
345                 }
346             }
347             return 0;

349         public override string GetCurrentWorkingDirectory(Process* process)
350         {
351             if (Log())
352             {
353                 LogMessage("fs.root.getcurrentworkingdirectory", "begin");
354             }
355             INode* rootINode = GetINodeManager().GetINode(process->rootDirINodeKey);
356             INodeLock rootINodeLock(rootINode);
357             INode* dirINode = rootINode;
358             INodeLock dirINodeLock;
359             if (process->workingDirINodeKey != process->rootDirINodeKey)
360             {
361                 dirINode = GetINodeManager().GetINode(process->workingDirINodeKey);
362                 dirINodeLock.Reset(dirINode);
363             }
364             INode* prevINode = null;
365             bool first = true;
366             string currentWorkindDirectoryPath;
367             while (prevINode != rootINode)
368             {
369                 INode* parentINode = null;
370                 bool found = false;
371                 for (long offset = 0; !found && offset < dirINode->GetFileSize();;)
372                 {
373                     int blockNumber = 0;
374                     int blockOffset = 0;
375                     GetBlockManager().GetBlockNumber(dirINode, offset, blockNumber, blockOffset, false);
376                     if (blockNumber == 0)
377                     {
378                         throw SystemError(EFAIL, "could not get block number for directory inode");
379                     }
380                     Block* block = GetBlockManager().ReadBlock(BlockKey(blockNumber, dirINode->Key().fsNumber, BlockKind.directoryBlock), null);
381                     BlockPutter putter(block);
382                     #assert(block is DirectoryBlock*);
383                     DirectoryBlock* directoryBlock = cast<DirectoryBlock*>(block);
384                     for (int index = blockOffset / directoryEntrySize; index < numDirectoryEntriesInBlock && offset < dirINode->GetFileSize(); ++index;)
385                     {
386                         const DirectoryEntry& directoryEntry = directoryBlock->GetDirectoryEntry(index);
387                         string entryName = &directoryEntry.name[0];
388                         if (!first && directoryEntry.inodeNumber == prevINode->Key().inodeNumber)
389                         {
390                             currentWorkindDirectoryPath = Path.Combine(entryName, currentWorkindDirectoryPath);
391                             found = true;
392                         }
393                         if (entryName == "..")
394                         {
395                             INodeKey key(directoryEntry.inodeNumber, dirINode->Key().fsNumber);
396                             if (key != dirINode->Key())
397                             {
398                                 parentINode = GetINodeManager().GetINode(key);
399                             }
400                             else
401                             {
402                                 parentINode = dirINode;
403                             }
404                         }
405                         offset = offset + directoryEntrySize;
406                     }
407                 }
408                 if (parentINode == null)
409                 {
410                     throw SystemError(EFAIL, "parent inode entry not found");
411                 }
412                 if (found)
413                 {
414                     prevINode = dirINode;
415                     dirINode = parentINode;
416                     if (dirINode != rootINode)
417                     {
418                         dirINodeLock.Reset(dirINode);
419                     }
420                 }
421                 else if (first)
422                 {
423                     prevINode = dirINode;
424                     dirINode = parentINode;
425                     if (dirINode != rootINode)
426                     {
427                         dirINodeLock.Reset(dirINode);
428                     }
429                     first = false;
430                 }
431                 else
432                 {
433                     throw SystemError(EFAIL, "could not find matching directory inode from parent inode");
434                 }
435             }
436             currentWorkindDirectoryPath = Path.Combine("/", currentWorkindDirectoryPath);
437             if (Log())
438             {
439                 LogMessage("fs.root.getcurrentworkingdirectory", "end.cwd=" + currentWorkindDirectoryPath);
440             }
441             return currentWorkindDirectoryPath;
442         }

508             SuperBlock* superBlock = cast<SuperBlock*>(block);
509             return superBlock->GetLastBlockNumber();
510         }
511         public override nothrow void SetLastBlockNumber(int blockNumber, SuperBlock* sb)

513             if (sb != null && sb->Key().fsNumber == 0 && sb->GetLastBlockNumber() < blockNumber)
514             {
515                 sb->SetLastBlockNumber(blockNumber);
516             }
517             else
518             {
519                 Block* block = GetBlockManager().ReadBlock(BlockKey(0, 0, BlockKind.superBlock), null);
520                 #assert(block is SuperBlock*);
521                 BlockPutter putter(block);
522                 SuperBlock* superBlock = cast<SuperBlock*>(block);
523                 if (superBlock->GetLastBlockNumber() < blockNumber)
524                 {
525                     superBlock->SetLastBlockNumber(blockNumber);
526                 }
527             }
528         }
529         private void CreateRootFs(int maxFiles, int maxBlocks)
530         {
531             try
532             {
533                 int numINodeBitmapBlocks = (maxFiles - 1) / (8 * blockSize) + 1;
534                 int numBlockBitmapBlocks = (maxBlocks - 1) / (8 * blockSize) + 1;
535                 int numINodeBlocks = (maxFiles - 1) / numINodesInBlock + 1;
536                 {
537                     Block* block = GetBlockManager().GetBlock(BlockKey(0, 0, BlockKind.superBlock), null);
538                     #assert(block is SuperBlock*);
539                     BlockPutter putter(block);
540                     SuperBlock* superBlock = cast<SuperBlock*>(block);
541                     superBlock->SetFirstINodeBitmapBlockNumber(1);
542                     superBlock->SetNumINodeBitmapBlocks(numINodeBitmapBlocks);
543                     superBlock->SetFirstBlockBitmapBlockNumber(superBlock->GetFirstINodeBitmapBlockNumber() + superBlock->GetNumINodeBitmapBlocks());
544                     superBlock->SetNumBlockBitmapBlocks(numBlockBitmapBlocks);
545                     superBlock->SetFirstINodeBlockNumber(superBlock->GetFirstBlockBitmapBlockNumber() + superBlock->GetNumBlockBitmapBlocks());
546                     superBlock->SetNumINodeBlocks(numINodeBlocks);
547                     superBlock->SetFirstDataBlockNumber(superBlock->GetFirstINodeBlockNumber() + superBlock->GetNumINodeBlocks());
548                     GetBlockManager().WriteBlock(block, superBlock);
549                 }
550                 blockBitmap.Reset(new BlockBitmap(0));
551                 inodeBitmap.Reset(new INodeBitmap(0));
552                 int rootDirINodeNumber = GetFreeINodeNumber();
553                 {
554                     Block* block = GetBlockManager().GetBlock(BlockKey(0, 0, BlockKind.superBlock), null);
555                     #assert(block is SuperBlock*);
556                     BlockPutter putter(block);
557                     SuperBlock* superBlock = cast<SuperBlock*>(block);
558                     superBlock->SetRootDirINodeNumber(rootDirINodeNumber);
559                     GetBlockManager().WriteBlock(block, superBlock);
560                 }
561                 GetBlockManager().Flush();
562                 FileSystem* fs = GetMountTable().GetFileSystem(0);
563                 fs->MakeDirectory(null, "/");
564                 GetBlockManager().Flush();
565             }
566             catch (const Exception& ex)
567             {
568                 GetDiskDriver().Stop();
569                 throw;
570             }
571         }
