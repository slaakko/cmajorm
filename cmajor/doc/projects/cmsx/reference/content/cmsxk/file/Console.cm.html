 69         public void WriteToConsole(byte* buffer, long count)
 70         {
 71             ustring u;
 72             for (long i = 0; i < count; ++i;)
 73             {
 74                 byte x = buffer[i];
 75                 unicodeEngine.Put(x);
 76                 if (unicodeEngine.ResultReady())
 77                 {
 78                     uchar c = unicodeEngine.Get();
 79                     if (c != '\r')
 80                     {
 81                         u.Append(c);
 82                     }
 83                 }
 84             }
 85             if (!u.IsEmpty())
 86             {
 87                 bool result = OsWriteConsole(consoleOutputHandle, u.Chars());
 88                 GetDimensions();
 89                 outputEndCursorPosX = cursorPosX;
 90                 outputEndCursorPosY = cursorPosY;
 91             }
 92         }

 98         public void HandleBufferedInput()
 99         {
100             GetDimensions();
101             ustring chars = GetCharsFromInputBuffer();
102             for (uchar c : chars)
103             {
104                 switch (c)
105                 {
106                     case cast<uchar>(keyEnter): HandleEnter(); break;
107                     case cast<uchar>(keyControlD): HandleEof(); break;
108                     case cast<uchar>(keyHome): HandleHome(); break;
109                     case cast<uchar>(keyEnd): HandleEnd(); break;
110                     case cast<uchar>(keyLeft): HandleLeft(); break;
111                     case cast<uchar>(keyRight): HandleRight(); break;
112                     case cast<uchar>(keyDel): HandleDelete(); break;
113                     case cast<uchar>(keyBackspace): HandleBackspace(); break;
114                     default: HandleChar(c); break;
115                 }
116             }
117         }
118         private void DeliverInputLine()
119         {
120             Machine& machine = GetMachine();
121             Kernel& kernel = GetKernel();
122             Session* session = kernel.CurrentSession();
123             if (session->fgpid != -1)
124             {
125                 ProcessTable& processTable = kernel.GetProcessTable();
126                 Process* reader = processTable.GetProcess(session->fgpid);
127                 if (reader != null)
128                 {
129                     if (reader->state == Process.State.asleep && reader->event == consoleInputEvent)
130                     {
131                         ulong bufferAddress = reader->eventData1;
132                         long count = cast<long>(reader->eventData2);
133                         GlobalFileTable& fileTable = kernel.FileTable();
134                         ConsoleInputFile* consoleInputFile = fileTable.GetConsoleInputFile();
135                         if (!consoleInputFile->HasBufferedInput())
136                         {
137                             if (HasLine())
138                             {
139                                 consoleInputFile->SetBufferedInput(GetLine());
140                             }
141                         }
142                         long result = consoleInputFile->Read(machine, reader, bufferAddress, count);
143                         #assert(result != RESULT_BLOCKED);
144                         WriteProcessMemory(machine, reader, reader->regAXAddress, cast<ulong>(result), 8u, Protection.write);
145                         WakeUpProcess(processTable, reader);
146                         Process* running = processTable.GetRunning();
147                         if (running == processTable.GetIdle())
148                         {
149                             Schedule(machine, processTable);
150                         }
151                     }
152                 }
153             }
154         }
155         private void HandleEnter()
156         {
157             line.Append('\n');
158             lines.Add(line);
159             line.Clear();
160             DeliverInputLine();
161             pos = 0;
162             ++cursorPosY;
163             SetCursorPos(0, cursorPosY);
164             outputEndCursorPosX = 0;
165             outputEndCursorPosX = cursorPosY;
166         }
167         private void HandleEof()
168         {
169             eof = true;
170             lines.Add(line);
171             line.Clear();
172             DeliverInputLine();
173             pos = 0;
174             ++cursorPosY;
175             SetCursorPos(0, cursorPosY);
176             outputEndCursorPosX = 0;
177             outputEndCursorPosX = cursorPosY;
178         }
179         private void HandleHome()

183         }

188         }
189         private void HandleLeft()

191             if (pos > 0)
192             {
193                 --pos;

195             }
196         }
197         private void HandleRight()
198         {
199             if (pos < line.Length())
200             {
201                 ++pos;
202                 SetCursorPos(outputEndCursorPosX + pos, outputEndCursorPosY);
203             }
204         }
205         private void HandleDelete()
206         {
207             if (pos < line.Length())
208             {
209                 ++pos;
210                 line = line.Substring(0, pos - 1) + line.Substring(pos);
211                 --pos;
212                 UpdateLine();
213             }
214         }
215         private void HandleBackspace()
216         {
217             if (pos > 0)
218             {
219                 line = line.Substring(0, pos - 1) + line.Substring(pos);
220                 --pos;
221                 UpdateLine();
222                 SetCursorPos(outputEndCursorPosX + pos, outputEndCursorPosY);
223             }
224         }
225         private void HandleChar(uchar c)
226         {
227             if (IsGraphic(c) && cursorPosX < screenSizeX - 1)
228             {
229                 if (pos < line.Length())
230                 {
231                     line = line.Substring(0, pos) + ustring(c, 1) + line.Substring(pos);
232                 }
233                 else
234                 {
235                     line.Append(c);
236                 }
237                 ++pos;
238                 UpdateLine();
239                 SetCursorPos(outputEndCursorPosX + pos, outputEndCursorPosY);
240             }
241         }
242         private void AddCharToInputBuffer(uchar c)
243         {
244             LockGuard<Mutex> lock(inputBufferMutex);
245             inputBuffer.Append(c);
246         }
247         private ustring GetCharsFromInputBuffer()
248         {
249             LockGuard<Mutex> lock(inputBufferMutex);
250             ustring chars;
251             Swap(chars, inputBuffer);
252             return chars;
253         }

260         private void UpdateLine()
261         {
262             int prevCursorPosX = cursorPosX;
263             int prevCursorPosY = cursorPosY;
264             SetCursorPos(outputEndCursorPosX, outputEndCursorPosY);
265             ustring updateLine;
266             cursorPosX = outputEndCursorPosX;
267             cursorPosY = outputEndCursorPosY;
268             for (int i = 0; i < line.Length(); ++i;)
269             {
270                 if (cursorPosX + i < screenSizeX)
271                 {
272                     updateLine.Append(line[i]);
273                 }
274                 else
275                 {
276                     OsWriteConsole(consoleOutputHandle, updateLine.Chars());
277                     updateLine.Clear();
278                     cursorPosX = 0;
279                     ++cursorPosY;
280                 }
281             }
282             if (!updateLine.IsEmpty())
283             {
284                 OsWriteConsole(consoleOutputHandle, updateLine.Chars());
285                 updateLine.Clear();
286             }
287             for (int i = cursorPosX; i < screenSizeX; ++i;)
288             {
289                 updateLine.Append(' ');
290             }
291             OsWriteConsole(consoleOutputHandle, updateLine.Chars());
292             SetCursorPos(prevCursorPosX, prevCursorPosY);
293         }
