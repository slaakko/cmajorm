 92         public HostFile* GetOrInsertHostFile(const string& hostFilePath)
 93         {

 95             if (it != hostFileMap.CEnd())
 96             {

 98             }
 99             else

126         public void Run()
127         {
128             try
129             {
130                 OsConvertThreadToFiber(null);
131                 while (!exit)
132                 {
133                     MakeEventHandleList();
134                     hostFilesChanged = false;
135                     OsSetEvent(osFilesUpdatedEventHandle);
136                     int eventIndex = OsWaitForMultipleObjects(cast<uint>(eventHandles.Count()), eventHandles.Begin().Ptr());
137                     if (eventIndex < 0)
138                     {
139                         Panic("error from disk driver run: wait for multiple objects failed");
140                     }
141                     else if (eventIndex == 0)
142                     {
143                         if (exit)
144                         {
145                             break;
146                         }
147                         else if (hostFilesChanged)
148                         {
149                             // cont
150                         }
151                     }
152                     else if (eventIndex < hostFiles.Count())
153                     {
154                         HostFile* hostFile = hostFiles[eventIndex].Get();
155                         if (hostFile != null)
156                         {
157                             ulong numberOfBytesTransferred = 0u;
158                             ulong completionKey = 0u;
159                             void* overlapped = null;
160                             bool retval = OsGetQueuedCompletionStatus(hostFile->GetCompletionPortHandle(), &numberOfBytesTransferred, &completionKey, &overlapped);
161                             if (Log())
162                             {
163                                 string msg;
164                                 msg.Append("ok=").Append(ToString(retval)).Append(".count=").Append(ToString(numberOfBytesTransferred)).Append(".overlapped=").Append(ToHexString(cast<ulong>(overlapped)));
165                                 LogMessage("fs.dd.ioready",  msg);
166                             }
167                             if (!retval)
168                             {
169                                 ulong lastError = OsGetLastError();
170                                 Panic("OsGetQueuedCompletionStatus failed: error=" + ToString(lastError));
171                             }
172                             bool handleCompletions = false;
173                             if (completionKey == hostFileCompletionKey)
174                             {
175                                 LockGuard<Mutex> lock(completionQueueMutex);
176                                 completionQueue.Put(IOCompletion(hostFile, numberOfBytesTransferred, overlapped));
177                                 if (GetKernel().Waiting())
178                                 {
179                                     handleCompletions = true;
180                                 }
181                                 else
182                                 {
183                                     GetMachine().GetRegisters().SetInterrupt(DISK_BIT);
184                                 }
185                             }
186                             else
187                             {
188                                 Panic("error from disk driver run: unknown completion key received");
189                             }
190                             if (handleCompletions)
191                             {
192                                 HandleIOCompletions();
193                             }
194                         }
195                         else
196                         {
197                             Panic("error from disk driver run: null host file");
198                         }
199                     }
200                     else
201                     {
202                         Panic("error from disk driver run: invalid event index from wait for multiple objects");
203                     }
204                 }
205             }
206             catch (const Exception& ex)
207             {
208                 Panic("error from disk driver run: " + ex.ToString());
209             }
210         }
211         public void HandleIOCompletions()
212         {
213             Kernel& kernel = GetKernel();
214             ProcessTable& processTable = kernel.GetProcessTable();
215             {
216                 LockGuard<Mutex> lock(completionQueueMutex);
217                 while (!completionQueue.IsEmpty())
218                 {
219                     IOCompletion ioCompletion = completionQueue.Get();
220                     IORequest& ioRequest = requestMap[ioCompletion.overlapped];
221                     ioRequest.numberOfBytesTransferred = ioCompletion.numberOfBytesTransferred;
222                     WakeUpProcess(processTable, ioRequest.process);
223                 }
224             }

226         public long Read(Block* block)
227         {
228             FileSystem* fs = GetMountTable().GetFileSystem(block->Key().fsNumber);
229             HostFile* hostFile = fs->GetHostFile();
230             void* evnt = hostFile->GetEventHandle();
231             ulong offset = cast<ulong>(block->Key().blockNumber * blockSize);
232             Overlapped overlapped(offset, evnt);
233             Buffer buffer;
234             void* fiberData = OsGetFiberData();
235             Process* process = cast<Process*>(fiberData);
236             requestMap[overlapped.Get()] = IORequest(process);
237             Kernel& kernel = GetKernel();
238             if (process == null)
239             {
240                 kernel.SetWaiting();
241             }
242             else
243             {
244                 kernel.ResetWaiting();
245             }
246             if (Log())
247             {
248                 LogMessage("fs.dd.read", block->ToString() + ".overlapped=" + ToHexString(cast<ulong>(overlapped.Get())));
249             }
250             bool result = OsReadFile(hostFile->GetFileHandle(), buffer.Get(), cast<uint>(blockSize), overlapped.Get());
251             ulong lastError = 0u;
252             if (!result)
253             {
254                 lastError = OsGetLastError();
255                 if (lastError == ERROR_IO_PENDING) lastError = 0u;
256             }
257             if (lastError == 0u)
258             {
259                 SleepProcess(process, blockIOEvent, cast<ulong>(cast<void*>(block)), 0u);
260                 MemoryReader reader(cast<byte*>(buffer.Get()), blockSize);
261                 block->Read(reader);
262                 IORequest& request = requestMap[overlapped.Get()];
263                 long result = cast<long>(request.numberOfBytesTransferred);
264                 requestMap.Remove(overlapped.Get());
265                 return result;
266             }
267             else
268             {
269                 requestMap.Remove(overlapped.Get());
270                 Buffer messageBuf;
271                 OsFormatMessage(lastError, cast<char*>(messageBuf.Get()));
272                 const char* msg = cast<const char*>(messageBuf.Get());
273                 string message = "Could not read from file '";
274                 message.Append(hostFile->GetHostFilePath()).Append("': ").Append(msg);
275                 throw SystemError(EIO, message);
276             }
277         }

330         private void MakeEventHandleList()
331         {
332             eventHandles.Clear();
333             bool first = true;
334             for (const UniquePtr<HostFile>& hostFile : hostFiles)
335             {
336                 if (first)
337                 {
338                     eventHandles.Add(interruptEventHandle);
339                     first = false;
340                 }
341                 else if (hostFile.IsNull())
342                 {
343                     throw Exception("error: host file is null");
344                 }
345                 else
346                 {
347                     eventHandles.Add(hostFile->GetEventHandle());
348                 }
349             }
