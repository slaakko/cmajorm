 20     {
 21         if (x != 0u || y != exitSystemCall || z != 0u)
 22         {
 23             Panic("invalid exit parameters: " + ToString(x) + ", " + ToString(y) + ", " + ToString(z));
 24         }
 25         Kernel& kernel = Kernel.Instance();
 26         ProcessTable& processTable = kernel.GetProcessTable();
 27         Process* process = processTable.GetRunning();
 28         if (process != null)
 29         {
 30             ushort exitCode = MakeProcessExitCode(0u, cast<byte>(ax));
 31             int retval = ExitProcess(machine, processTable, process, exitCode);
 32             if (retval != 0)
 33             {
 34                 Panic("exit failed!");
 35             }
 36             if (Machine.GetFlag(Machine.Flags.runProgram) && process->pid == kernel.GetProgramPID())
 37             {
 38                 machine.GetProcessor().Exit();
 39             }
 40             return 0;
 41         }
 42         else
 43         {
 44             Panic("exit called for a process that is not running!");
 45         }
 46         return 0;
 47     }
