 79         if (pageProtection == Protection.copyOnWrite)
 80         {
 81             rq = rq | COW_BIT;
 82         }
 83         else

 86             {
 87                 rq = rq | X_BIT;
 88             }
 89             else if ((access & Protection.write) != 0u)
 90             {
 91                 rq = rq | W_BIT;
 92             }
 93             else if ((access & Protection.read) != 0u)
 94             {
 95                 rq = rq | R_BIT;
 96             }
 97             else
 98             {
 99                 SetSecurityViolation(regs, virtualAddress);
100                 return;
101             }

104     }
