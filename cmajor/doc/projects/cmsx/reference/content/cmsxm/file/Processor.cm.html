 61             sp = sp + 8u;
 62             ulong rl = regs.GetSpecial(Registers.rL);
 63             #assert(rl < 256u);
 64             int n = cast<int>(rl);
 65             for (int i = 0u; i < n; ++i;)
 66             {
 67                 mem.WriteULong(sp, regs.Get(cast<byte>(i)), Protection.write);
 68                 sp = sp + 8u;
 69             }
 70             ulong rg = regs.GetSpecial(Registers.rG);
 71             #assert(rg < 256u);
 72             for (int i = cast<int>(rg); i < 256; ++i;)
 73             {
 74                 switch (i)
 75                 {
 76                     case regAX:
 77                     {
 78                         if (regAXAddress != null) *regAXAddress = sp;
 79                         break;
 80                     }
 81                     case regBX:
 82                     {
 83                         if (regBXAddress != null) *regBXAddress = sp;
 84                         break;
 85                     }
 86                 }
 87                 mem.WriteULong(sp, regs.Get(cast<byte>(i)), Protection.write);
 88                 sp = sp + 8u;
 89             }
 90             mem.WriteULong(sp, regs.GetSpecial(Registers.rL), Protection.write);
 91             sp = sp + 8u;
 92             mem.WriteULong(sp, regs.GetSpecial(Registers.rG), Protection.write);
 93             sp = sp + 8u;
 94             regs.SetSpecial(Registers.rV, prevRV);
 95             return sp;
 96         }

100             regs.SetSpecial(Registers.rV, virtualTranslateionRegisterValue);
101             sp = sp - 8u;

104             sp = sp - 8u;
105             ulong rl = mem.ReadULong(sp, Protection.read);
106             regs.SetSpecial(Registers.rL, rl);
107             for (int i = 255; i >= cast<int>(rg); --i;)
108             {
109                 sp = sp - 8u;
110                 regs.Set(cast<byte>(i), mem.ReadULong(sp, Protection.read));
111             }

114             {

150         private void DoRun()
151         {
152             ClockDriver clockDriver(Clock.Instance());
153             Memory& mem = machine.GetMemory();
154             ulong pc = regs.GetPC();
155             while (pc != invalidAddress && !exiting)
156             {
157                 ulong prevPC = pc;
158                 byte x = 0u;
159                 byte y = 0u;
160                 byte z = 0u;
161                 Instruction* inst = FetchInstruction(pc, mem, x, y, z);
162                 inst->Execute(regs, mem, x, y, z);
163                 SetPC(inst, pc, prevPC);
164                 IncrementCycleCounter();
165                 CheckInterrupts();
166                 DecrementIntervalCounter();
167                 pc = regs.GetPC();
168             }
169         }
170         private void Boot()
171         {
172             ulong kernelRootPageAddress = machine.GetKernelRootPageAddress();
173             Memory& mem = machine.GetMemory();
174             if (kernelRootPageAddress == 0u)
175             {
176                 kernelRootPageAddress = mem.AllocateRootPage();
177                 machine.SetKernelRootPageAddress(kernelRootPageAddress);
178             }
179             regs.SetPC(invalidAddress);
180             regs.SetSpecial(Registers.rV, MakeVirtualTranslationRegisterValue(kernelRootPageAddress, kernelAddressSpaceNumber));
181             ulong pte = mem.GetPageTableEntry(bootProcAddress);
182             if (pte != 0u)
183             {
184                 ulong bootProcValue = mem.ReadULong(bootProcAddress, Protection.read);
185                 if (bootProcValue != 0u)
186                 {
187                     BootProc boot = cast<BootProc>(cast<void*>(bootProcValue));
188                     boot(machine);
189                 }
190                 else
191                 {
192                     if (!Machine.GetFlag(Machine.Flags.quiet))
193                     {
194                         Console.Error() << "machine boot procedure not set" << endl();
195                     }
196                 }
197             }
198             else
199             {
200                 if (!Machine.GetFlag(Machine.Flags.quiet))
201                 {
202                     Console.Error() << "machine boot procedure not set" << endl();

232         private nothrow void DecrementIntervalCounter()
233         {
234             ulong intervalCounter = regs.GetSpecial(Registers.rI);
235             if (intervalCounter != 0u)
236             {
237                 --intervalCounter;
238                 regs.SetSpecial(Registers.rI, intervalCounter);
239                 if (intervalCounter == 0u)
240                 {
241                     ulong prevRQ = regs.GetSpecial(Registers.rQ);
242                     regs.SetSpecial(Registers.rQ, INTERVAL_INTERRUPT_BIT);
243                     CheckInterrupts();

245                 }
246             }

248         private nothrow void SetPC(Instruction* inst, ulong& newPC, ulong prevPC)
249         {
250             ulong pc = regs.GetPC();
251             if (!inst->IsJumpInst() && pc == prevPC)
252             {
253                 regs.SetPC(newPC);
254             }
255             else
256             {
257                 newPC = pc;
258             }
259             regs.SetSpecial(Registers.rW, prevPC);
260         }
261         private nothrow void CheckInterrupts()
262         {
263             ulong rq = regs.GetSpecial(Registers.rQ);
264             ulong rk = regs.GetSpecial(Registers.rK);
265             ulong bits = rq & rk;
266             if (bits != 0u)
267             {
268                 Memory& mem = machine.GetMemory();
269                 ulong handlerPtrAddress = regs.GetSpecial(Registers.rTT);
270                 ulong irqBit = 1u;
271                 for (byte irq = 0u; irq < irqMax; ++irq;)
272                 {
273                     byte bit = cast<byte>(bits & 1u);
274                     if (bit != 0u)
275                     {
276                         HandleInterrupt(irq, handlerPtrAddress);
277                         rq = rq & ~irqBit;
278                         regs.SetSpecial(Registers.rQ, rq);
279                     }
280                     bits = bits >> 1u;
281                     handlerPtrAddress = handlerPtrAddress + 8u;
282                     irqBit = irqBit << 1u;
283                 }
284             }
285         }
286         private void HandleInterrupt(byte irq, ulong handlerPtrAddress)
287         {
288             ulong prevRV = regs.GetSpecial(Registers.rV);
289             bool inKernelMode = IsProcessorInKernelMode(regs);
290             if (!inKernelMode)
291             {
292                 SetProcessorToKernelMode(regs);
293                 ulong kernelRV = MakeVirtualTranslationRegisterValue(machine.GetKernelRootPageAddress(), kernelAddressSpaceNumber);
294                 regs.SetSpecial(Registers.rV, kernelRV);
295             }
296             ulong handlerAddress = machine.GetMemory().ReadULong(handlerPtrAddress, Protection.read);
297             regs.SetSpecial(Registers.rV, prevRV);
298             if (handlerAddress != 0u)
299             {
300                 InterruptHandler handler = cast<InterruptHandler>(cast<void*>(handlerAddress));
301                 handler(machine, irq);
302             }
303             else
304             {
305                 Panic("cannot handle " + GetInterruptName(irq) + " interrupt " + ToString(irq) + ", handler address is null");
306             }
307             if (!inKernelMode)
308             {
309                 SetProcessorToUserMode(regs);
310             }
311         }
