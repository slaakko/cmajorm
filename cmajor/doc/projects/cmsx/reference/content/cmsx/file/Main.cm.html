  7 void Init(Machine& machine)
  8 {
  9     Kernel& kernel = Kernel.Instance();
 10     kernel.Init();
 11     MemoryTable& kernelMemoryTable = kernel.GetMemoryTable();
 12     machine.SetKernelRootPageAddress(kernelMemoryTable.rootPageAddress);
 13     Memory& mem = machine.GetMemory();
 14     AllocateMemory(mem, kernelMemoryTable);
 15     Registers& regs = machine.GetRegisters();
 16     SetProcessorToKernelMode(regs);
 17     regs.SetSpecial(Registers.rV, MakeVirtualTranslationRegisterValue(kernelMemoryTable.rootPageAddress, kernelMemoryTable.addressSpaceNumber));
 18     BootProc boot = Boot;
 19     ulong bootProcValue = cast<ulong>(cast<void*>(boot));
 20     mem.WriteULong(bootProcAddress, bootProcValue, Protection.write);

 28 void Terminate(Machine& machine)
 29 {
 30     Kernel& kernel = Kernel.Instance();
 31     if (Machine.GetFlag(Machine.Flags.runProgram))
 32     {
 33         int pid = kernel.GetProgramPID();
 34         Process* process = kernel.GetProcessTable().GetProcess(pid);
 35         if (process != null && Machine.GetFlag(Machine.Flags.verbose))
 36         {
 37             Console.Out() << "program exited with code " << process->exitCode << "." << endl();
 38         }
 39     }
 40     Registers& regs = machine.GetRegisters();
 41     SetProcessorToKernelMode(regs);
 42     regs.SetSpecial(Registers.rV, MakeVirtualTranslationRegisterValue(machine.GetKernelRootPageAddress(), kernelAddressSpaceNumber));
 43     MemoryTable& kernelMemoryTable = kernel.GetMemoryTable();
 44     FreeMemory(machine.GetMemory(), kernelMemoryTable);
 45 }

 47 void MachineThreadFunction(void* machineParam)
 48 {
 49     try
 50     {

 65 class MachineDriver
 66 {

 68     {
 69     }
 70     public void RunThread()
 71     {
 72         machineThread = Thread.StartFunction(MachineThreadFunction, &machine);

 74     public ~MachineDriver()
 75     {
 76         if (!Machine.GetFlag(Machine.Flags.runProgram))
 77         {
 78             machine.Exit();
 79         }
 80         machineThread.Join();
 81     }
 82     private Machine& machine;

 86 void Run()

 89     MachineDriver machineDriver(machine);
 90     machineDriver.RunThread();

 92     {
 93         string line = Console.ReadLine();
 94         while (!Console.In().EndOfStream())
 95         {
 96             if (line == "logout" || line == "exit")
 97             {
 98                 break;

105 int main(int argc, const char** argv)
106 {
107     Debug.Start();
108     try
109     {
110         string programFileName;
111         List<string> programArguments;
112         bool cmsxOptions = true;
113         for (int i = 1; i < argc; ++i;)
114         {
115             string arg = argv[i];
116             if (cmsxOptions && arg.StartsWith("-"))
117             {
118                 if (arg == "--verbose" || arg == "-v")
119                 {
120                     Machine.SetFlag(Machine.Flags.verbose);
121                 }
122                 else if (arg == "--quiet" || arg == "-q")
123                 {
124                     Machine.SetFlag(Machine.Flags.quiet);
125                 }
126                 else if (arg == "--debug" || arg == "-d")
127                 {
128                     Machine.SetFlag(Machine.Flags.debug);
129                 }
130                 else
131                 {
132                     throw Exception("unknown option '" + arg + "'");
133                 }
134             }
135             else if (programFileName.IsEmpty())
136             {
137                 programFileName = arg;
138                 cmsxOptions = false;
139                 Machine.SetFlag(Machine.Flags.runProgram);
140             }
141             else
142             {
143                 programArguments.Add(arg);
144             }
145         }
146         if (Machine.GetFlag(Machine.Flags.runProgram))
147         {
148             Kernel.Instance().SetProgramFileName(programFileName);
149             Kernel.Instance().SetProgramArguments(programArguments);
150         }
151         Run();
152     }
153     catch (const Exception& ex)
154     {
155         Console.Error() << ex.ToString() << endl();
156         return 1;
157     }
158     return 0;
