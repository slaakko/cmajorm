 20 void Init(Machine& machine)
 21 {
 22     Kernel& kernel = Kernel.Instance();
 23     kernel.Init();
 24     MemoryTable& kernelMemoryTable = kernel.GetMemoryTable();
 25     machine.SetKernelRootPageAddress(kernelMemoryTable.rootPageAddress);
 26     Memory& mem = machine.GetMemory();
 27     AllocateMemory(mem, kernelMemoryTable, false);
 28     Registers& regs = machine.GetRegisters();
 29     SetProcessorToKernelMode(regs);
 30     regs.SetSpecial(Registers.rV, MakeVirtualTranslationRegisterValue(kernelMemoryTable.rootPageAddress, kernelMemoryTable.addressSpaceNumber));
 31     BootProc boot = Boot;
 32     ulong bootProcValue = cast<ulong>(cast<void*>(boot));
 33     mem.WriteULong(bootProcAddress, bootProcValue, Protection.write);
 34     if (Machine.GetFlag(Machine.Flags.debug))
 35     {
 36         kernel.SetDebugger(DebuggerProvider.GetDebugger());
 37     }
 38 }

 45 void Terminate(Machine& machine)
 46 {
 47     GetDiskDriver().Stop();
 48     GetConsoleDriver().Stop();
 49     if (Machine.GetFlag(Machine.Flags.debug))
 50     {
 51         Debugger* debugger = DebuggerProvider.GetDebugger();
 52         debugger->Done();
 53     }
 54     Kernel& kernel = GetKernel();
 55     int pid = kernel.GetProgramPID();
 56     if (pid != -1)
 57     {
 58         Process* process = kernel.GetProcessTable().GetProcess(pid);
 59         if (process != null && Machine.GetFlag(Machine.Flags.verbose))
 60         {
 61             Console.Out() << "process exited with code " << process->exitCode << "." << endl();
 62         }
 63     }
 64     Registers& regs = machine.GetRegisters();
 65     SetProcessorToKernelMode(regs);
 66     regs.SetSpecial(Registers.rV, MakeVirtualTranslationRegisterValue(machine.GetKernelRootPageAddress(), kernelAddressSpaceNumber));
 67     MemoryTable& kernelMemoryTable = kernel.GetMemoryTable();
 68     FreeMemory(machine.GetMemory(), kernelMemoryTable);
 69 }

 71 void MachineThreadFunction(void* machineParam)
 72 {
 73     Machine* machine = cast<Machine*>(machineParam);
 74     try
 75     {
 76         if (machine != null)
 77         {
 78             Init(*machine);
 79             Run(*machine);
 80             Terminate(*machine);
 81         }
 82     }
 83     catch (const Exception& ex)
 84     {
 85         if (machine != null)
 86         {
 87             machine->SetException(CaptureCurrentException());
 88         }
 89     }
 90 }

149 bool CopyFs()
150 {
151     string rootFSPath;
152     string defaultFSPath;
153     try
154     {
155         string cmajorRootDir = RtGetEnvironmentVariable("CMAJOR_ROOT");
156         if (cmajorRootDir.IsEmpty())
157         {
158             throw Exception("CMAJOR_ROOT environment variable not set. Please set it to contain /path/to/cmajor directory");
159         }
160         string fsDirPath = GetFullPath(Path.Combine(Path.Combine(Path.Combine(cmajorRootDir, "projects"), "cmsx"), "fs"));
161         rootFSPath = GetFullPath(Path.Combine(fsDirPath, "rootfs"));
162         defaultFSPath = GetFullPath(Path.Combine(fsDirPath, "defaultfs"));
163         if (!System.IO.File.Exists(rootFSPath) && System.IO.File.Exists(defaultFSPath))
164         {
165             bool verbose = Machine.GetFlag(Machine.Flags.verbose);
166             if (verbose)
167             {
168                 Console.Out() << "rootfs does not exist: copying from defaultfs...";
169             }
170             System.IO.File.Copy(defaultFSPath, rootFSPath);
171             if (verbose)
172             {
173                 Console.Out() << "\b\b\b, done." << endl();
174             }
175             return true;
176         }
177     }
178     catch (const Exception& ex)
179     {
180         Console.Error() << endl();
181         throw Exception("could not copy '" + defaultFSPath + "' to '" + rootFSPath + "': " + ex.Message());
182     }
183     return false;
184 }

186 bool CheckValidUserAccount(const string& userAccount)
187 {
188     if (userAccount.IsEmpty()) return false;
189     for (char c : userAccount)
190     {
191         bool validChar = (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || (c >= '0' && c <= '9');
192         if (!validChar) return false;
193     }
194     return true;
195 }

197 void SetUser(const string& user, bool fsCopied)
198 {
199     if (!user.IsEmpty())
200     {
201         if (!CheckValidUserAccount(user))
202         {
203             throw Exception(user + " is not a valid CMSX user account name. It must not be empty and must consist of letters a-z, A-Z and digits 0-9.");
204         }
205     }
206     string defaultUserAccount = user;
207     bool defaultUserAccountCreated = false;
208     string cmajorRootDir = RtGetEnvironmentVariable("CMAJOR_ROOT");
209     if (cmajorRootDir.IsEmpty())
210     {
211         throw Exception("CMAJOR_ROOT environment variable not set. Please set it to contain /path/to/cmajor directory");
212     }
213     string configDirPath = GetFullPath(Path.Combine(Path.Combine(Path.Combine(cmajorRootDir, "projects"), "cmsx"), "config"));
214     string defaultUserFilePath = GetFullPath(Path.Combine(configDirPath, "default-user"));
215     if (fsCopied)
216     {
217         if (System.IO.File.Exists(defaultUserFilePath))
218         {
219             System.IO.File.Remove(defaultUserFilePath);
220         }
221     }
222     if (!defaultUserAccount.IsEmpty())
223     {
224         try
225         {
226             StreamWriter writer = System.IO.File.CreateText(defaultUserFilePath);
227             writer << defaultUserAccount << endl();
228         }
229         catch (const Exception& ex)
230         {
231             throw Exception("could not create '" + defaultUserFilePath + "': " + ex.Message());
232         }
233     }
234     else if (System.IO.File.Exists(defaultUserFilePath))
235     {
236         try
237         {
238             StreamReader reader = System.IO.File.OpenRead(defaultUserFilePath);
239             defaultUserAccount = reader.ReadLine();
240         }
241         catch (const Exception& ex)
242         {
243             throw Exception("could not read '" + defaultUserFilePath + "': " + ex.Message());
244         }
245     }
246     else
247     {
248         Console.Out() << "CMSX default user account not set.\nEnter name of a new CMSX user account to be created> ";
249         defaultUserAccount = Console.ReadLine();
250         while (true)
251         {
252             bool valid = CheckValidUserAccount(defaultUserAccount);
253             if (valid)
254             {
255                 break;
256             }
257             else
258             {
259                 Console.Out() << "String '" << defaultUserAccount << "' is not a valid CMSX user account name. It must not be empty and must consist of letters a-z, A-Z and digits 0-9." << endl();
260                 Console.Out() << "Enter name of a new CMSX user account to be created> ";
261                 defaultUserAccount = Console.ReadLine();
262             }
263         }
264         try
265         {
266             StreamWriter writer = System.IO.File.CreateText(defaultUserFilePath);
267             writer << defaultUserAccount << endl();
268             defaultUserAccountCreated = true;
269         }
270         catch (const Exception& ex)
271         {
272             throw Exception("could not create '" + defaultUserFilePath + "': " + ex.Message());
273         }
274     }
275     if (defaultUserAccountCreated || fsCopied)
276     {
277         Machine.SetFlag(Machine.Flags.install);
278     }
279     Kernel& kernel = Kernel.Instance();
280     kernel.SetUser(defaultUserAccount);
281 }

283 int main(int argc, const char** argv)
284 {
285     Debug.Start();
286     try
287     {
288         string programFileName;
289         List<string> programArguments;
290         bool cmsxOptions = true;
291         string user;
292         for (int i = 1; i < argc; ++i;)
293         {
294             string arg = argv[i];
295             if (cmsxOptions)
296             {
297                 if (arg.StartsWith("--"))
298                 {
299                     if (arg == "--help")
300                     {
301                         PrintHelp();
302                         return 1;
303                     }
304                     else if (arg == "--verbose")
305                     {
306                         Machine.SetFlag(Machine.Flags.verbose);
307                     }
308                     else if (arg == "--quiet")
309                     {
310                         Machine.SetFlag(Machine.Flags.quiet);
311                     }
312                     else if (arg == "--debug")
313                     {
314                         Machine.SetFlag(Machine.Flags.debug);
315                     }
316                     else if (arg == "--log")
317                     {
318                         SetLog();
319                     }
320                     else if (arg == "--install")
321                     {
322                         Machine.SetFlag(Machine.Flags.install);
323                     }
324                     else if (arg == "--followExec")
325                     {
326                         Machine.SetFlag(Machine.Flags.followExec);
327                     }
328                     else if (arg == "--sleeping-write")
329                     {
330                         Machine.SetFlag(Machine.Flags.sleepingWrite);
331                     }
332                     else if (arg == "--no-readahead")
333                     {
334                         Machine.SetFlag(Machine.Flags.noReadahead);
335                     }
336                     else if (arg == "--dont-use-translation-cache")
337                     {
338                         SetUseTranslationCacheValue(false);
339                     }
340                     else if (arg.StartsWith("--logfile="))
341                     {
342                         string logFile = arg.Substring(10);
343                         SetLog(logFile);
344                     }
345                     else if (arg.StartsWith("--user="))
346                     {
347                         user = arg.Substring(7);
348                     }
349                     else
350                     {
351                         throw Exception("unknown option '" + arg + "'");
352                     }
353                 }
354                 else if (arg.StartsWith("-"))
355                 {
356                     if (arg.StartsWith("-f="))
357                     {
358                         string logFile = arg.Substring(3);
359                         SetLog(logFile);
360                     }
361                     else if (arg.StartsWith("-u="))
362                     {
363                         user = arg.Substring(3);
364                     }
365                     else
366                     {
367                         string options = arg.Substring(1);
368                         if (options.IsEmpty())
369                         {
370                             throw Exception("unknown option '" + arg + "'");
371                         }
372                         else
373                         {
374                             for (char o : options)
375                             {
376                                 if (o == 'h')
377                                 {
378                                     PrintHelp();
379                                     return 1;
380                                 }
381                                 else if (o == 'v')
382                                 {
383                                     Machine.SetFlag(Machine.Flags.verbose);
384                                 }
385                                 else if (o == 'q')
386                                 {
387                                     Machine.SetFlag(Machine.Flags.quiet);
388                                 }
389                                 else if (o == 'd')
390                                 {
391                                     Machine.SetFlag(Machine.Flags.debug);
392                                 }
393                                 else if (o == 'x')
394                                 {
395                                     Machine.SetFlag(Machine.Flags.followExec);
396                                 }
397                                 else if (o == 'p')
398                                 {
399                                     Machine.SetFlag(Machine.Flags.sleepingWrite);
400                                 }
401                                 else if (o == 'n')
402                                 {
403                                     Machine.SetFlag(Machine.Flags.noReadahead);
404                                 }
405                                 else if (o == 'l')
406                                 {
407                                     SetLog();
408                                 }
409                                 else if (o == 'i')
410                                 {
411                                     Machine.SetFlag(Machine.Flags.install);
412                                 }
413                                 else
414                                 {
415                                     throw Exception("unknown option '" + arg + "'");
416                                 }
417                             }
418                         }
419                     }
420                 }
421                 else if (programFileName.IsEmpty())
422                 {
423                     programFileName = arg;
424                     cmsxOptions = false;
425                     Machine.SetFlag(Machine.Flags.runProgram);
426                 }
427                 else
428                 {
429                     throw Exception("invalid argument '" + arg + "'");
430                 }
431             }
432             else
