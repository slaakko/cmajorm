 55         {

 59         {
 60             s.Append('P');

 62         if ((flags & Value.Flag.address) != Value.Flag.undefined)
 63         {

 65         }
 66         if ((flags & Value.Flag.function) != Value.Flag.undefined)
 67         {
 68             s.Append('F');
 69         }
 70         if ((flags & Value.Flag.structure) != Value.Flag.undefined)
 71         {
