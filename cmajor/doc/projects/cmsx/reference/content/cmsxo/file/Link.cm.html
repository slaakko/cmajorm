174     public void LinkInternal(ObjectFile& objectFile)
175     {
176         LinkSection newLinkSection;
177         SymbolTable& symbolTable = objectFile.GetSymbolTable();
178         Section* linkSection = objectFile.linkSection;
179         #assert(linkSection != null);
180         Section* codeSection = objectFile.codeSection;
181         #assert(codeSection != null);
182         linkSection->pos = 0u;
183         int x = linkSection->GetByte();
184         while (x != -1)
185         {
186             LinkCode linkCode = cast<LinkCode>(cast<byte>(x));
187             switch (linkCode)
188             {
189                 case LinkCode.forwardLongJump:
190                 {
191                     uint symbolIndex = linkSection->GetUInt();
192                     ulong fromAddress = linkSection->GetULong();
193                     Symbol* symbol = symbolTable.GetSymbol(cast<int>(symbolIndex));
194                     #assert(symbol != null);
195                     #assert(symbol->segment == Segment.text);
196                     #assert(symbol->value.GetFlag(Value.Flag.address));
197                     ulong toAddress = symbol->value.value;
198                     #assert(toAddress >= fromAddress);
199                     ulong offset = (toAddress - fromAddress) >> 2u;
200                     if (offset >= cast<ulong>(MaxValue<ushort>()) << 8u)
201                     {
202                         throw Exception("error linking: forward long jump too far");
203                     }
204                     codeSection->EmitLongOffset(fromAddress, cast<uint>(offset));
205                     break;
206                 }
207                 case LinkCode.forwardShortJump:
208                 {
209                     uint symbolIndex = linkSection->GetUInt();
210                     ulong fromAddress = linkSection->GetULong();
211                     Symbol* symbol = symbolTable.GetSymbol(cast<int>(symbolIndex));
212                     #assert(symbol != null);
213                     #assert(symbol->segment == Segment.text);
214                     #assert(symbol->value.GetFlag(Value.Flag.address));
215                     ulong toAddress = symbol->value.value;
216                     #assert(toAddress >= fromAddress);
217                     ulong offset = (toAddress - fromAddress) >> 2u;
218                     if (offset >= cast<ulong>(MaxValue<ushort>()))
219                     {
220                         throw Exception("error linking: forward short jump too far");
221                     }
222                     codeSection->EmitShortOffset(fromAddress, cast<ushort>(offset));
223                     break;
224                 }
225                 case LinkCode.absoluteAddrValue:
226                 {
227                     uint index = linkSection->GetUInt();
228                     ulong address = linkSection->GetULong();
229                     newLinkSection.EmitByte(LinkCode.absoluteAddrValue);
230                     newLinkSection.EmitUInt(cast<uint>(index));
231                     newLinkSection.EmitULong(address);
232                     break;
233                 }
234                 case LinkCode.farOcta:
235                 {
236                     uint index = linkSection->GetUInt();
237                     ulong address = linkSection->GetULong();
238                     newLinkSection.EmitByte(LinkCode.farOcta);
239                     newLinkSection.EmitUInt(cast<uint>(index));
240                     newLinkSection.EmitULong(address);
241                     break;
242                 }
243                 case LinkCode.once:
244                 {
245                     uint index = linkSection->GetUInt();
246                     newLinkSection.EmitByte(LinkCode.once);
247                     newLinkSection.EmitUInt(cast<uint>(index));
248                     break;
249                 }
250             }
251             x = linkSection->GetByte();
252         }
253         Swap(linkSection->data, newLinkSection.data);
254     }

256     public void ProcessSymbols(LinkTable& linkTable, ExecutableFile& executable, ObjectFile* objectFile)
257     {
258         SymbolTable& objectFileSymbolTable = objectFile->GetSymbolTable();
259         objectFile->codeSection->removeOffset = 0u;
260         objectFile->dataSection->removeOffset = 0u;
261         SymbolTable& executableSymbolTable = executable.GetSymbolTable();
262         for (const UniquePtr<Symbol>& symbol : objectFileSymbolTable.Symbols())
263         {
264             if (symbol->linkage == Linkage.internal_)
265             {
266                 if (symbol->value.GetFlag(Value.Flag.address) && symbol->value.GetFlag(Value.Flag.address) && (symbol->segment == Segment.text || symbol->segment == Segment.data))
267                 {
268                     symbol->value.value = symbol->value.value - symbol->section->removeOffset;
269                     symbol->start = symbol->section->baseAddress + symbol->value.value;
270                     Symbol* executableSymbol = symbol->Clone();
271                     #assert(executableSymbol->section != null);
272                     if (executableSymbol->section is DataSection*)
273                     {
274                         executable.dataSection->AddSymbol(executableSymbol);
275                     }
276                     executableSymbolTable.AddInternalSymbol(executableSymbol, true);
277                 }
278             }
279             else if (symbol->linkage == Linkage.external || symbol->linkage == Linkage.undefined)
280             {
281                 if (symbol->value.GetFlag(Value.Flag.address) && symbol->value.GetFlag(Value.Flag.address) && (symbol->segment == Segment.text || symbol->segment == Segment.data))
282                 {
283                     Symbol* prevSymbol = executableSymbolTable.GetSymbol(symbol->name);
284                     if (prevSymbol != null)
285                     {
286                         #assert(prevSymbol->section != null);
287                         #assert(symbol->section != null);
288                         throw Exception("Duplicate external symbol '" + symbol->name + 
289                             "'. Original defined in " + prevSymbol->section->name + " section of file " + prevSymbol->section->file->FileName() + 
290                             ". Duplicate detected in " + symbol->section->name + " section of file " + symbol->section->file->FileName() + ".");
291                     }
292                     else
293                     {
294                         symbol->value.value = symbol->value.value - symbol->section->removeOffset;
295                         symbol->start = symbol->section->baseAddress + symbol->value.value;
296                         Symbol* executableSymbol = symbol->Clone();
297                         executableSymbol->linkage = Linkage.external;
298                         #assert(executableSymbol->section != null);
299                         if (executableSymbol->section is DataSection*)
300                         {
301                             executable.dataSection->AddSymbol(executableSymbol);
302                         }
303                         executableSymbolTable.AddSymbol(executableSymbol);
304                         linkTable.ExecuteLinkCommands(symbol->name, symbol->start);
305                     }
306                 }
307                 else if (symbol->value.flags == Value.Flag.undefined)
308                 {
309                     Symbol* prevSymbol = executableSymbolTable.GetSymbol(symbol->name);
310                     if (prevSymbol != null)
311                     {
312                         symbol->start = prevSymbol->start;
313                         symbol->linkage = Linkage.external;
314                     }
315                 }
316             }
317             else if (symbol->linkage == Linkage.once)
318             {
319                 if (symbol->value.GetFlag(Value.Flag.address) && symbol->value.GetFlag(Value.Flag.address) && (symbol->segment == Segment.text || symbol->segment == Segment.data))
320                 {
321                     Symbol* prevSymbol = executableSymbolTable.GetSymbol(symbol->name);
322                     if (prevSymbol == null)
323                     {
324                         symbol->value.value = symbol->value.value - symbol->section->removeOffset;
325                         symbol->start = symbol->section->baseAddress + symbol->value.value;
326                         Symbol* executableSymbol = symbol->Clone();
327                         #assert(executableSymbol->section != null);
328                         if (executableSymbol->section is DataSection*)
329                         {
330                             executable.dataSection->AddSymbol(executableSymbol);
331                         }
332                         executableSymbolTable.AddSymbol(executableSymbol);
333                         linkTable.ExecuteLinkCommands(symbol->name, symbol->start);
334                     }
335                     else
336                     {
337                         symbol->linkage = Linkage.remove;
338                         symbol->section->removeOffset = symbol->section->removeOffset + symbol->length;
339                     }
340                 }
341             }
342         }

345     public void ProcessLinkSection(LinkTable& linkTable, ExecutableFile& executable, ObjectFile* objectFile)
346     {
347         SymbolTable& executableSymbolTable = executable.GetSymbolTable();
348         Section* codeSection = objectFile->codeSection;
349         codeSection->copyStartPos = 0u;
350         codeSection->copyTargetSection = executable.codeSection;
351         Section* dataSection = objectFile->dataSection;
352         dataSection->copyStartPos = 0u;
353         dataSection->copyTargetSection = executable.dataSection;
354         Section* linkSection = objectFile->linkSection;
355         linkSection->pos = 0u;
356         int x = linkSection->GetByte();
357         ulong prevCodeSectionLength = codeSection->length;
358         ulong prevDataSectionLength = dataSection->length;
359         while (x != -1)
360         {
361             LinkCode linkCode = cast<LinkCode>(cast<byte>(x));
362             Symbol* symbol = null;
363             UniquePtr<LinkCommand> linkCommand;
364             switch (linkCode)
365             {
366                 case LinkCode.absoluteAddrValue:
367                 {
368                     uint index = linkSection->GetUInt();
369                     ulong address = linkSection->GetULong();
370                     symbol = objectFile->GetSymbolTable().GetSymbol(cast<int>(index));
371                     if (symbol->value.flags == Value.Flag.undefined)
372                     {
373                         Symbol* executableSymbol = executableSymbolTable.GetSymbol(symbol->name);
374                         if (executableSymbol != null)
375                         {
376                             symbol = executableSymbol;
377                         }
378                     }
379                     linkCommand.Reset(new LinkAbsoluteAddressCommand(objectFile, address));
380                     break;
381                 }
382                 case LinkCode.farOcta:
383                 {
384                     uint index = linkSection->GetUInt();
385                     ulong address = linkSection->GetULong();
386                     symbol = objectFile->GetSymbolTable().GetSymbol(cast<int>(index));
387                     if (symbol->value.flags == Value.Flag.undefined)
388                     {
389                         Symbol* executableSymbol = executableSymbolTable.GetSymbol(symbol->name);
390                         if (executableSymbol != null)
391                         {
392                             symbol = executableSymbol;
393                         }
394                     }
395                     linkCommand.Reset(new LinkFarOctaCommand(objectFile, address));
396                     break;
397                 }
398                 case LinkCode.once:
399                 {
400                     uint index = linkSection->GetUInt();
401                     symbol = objectFile->GetSymbolTable().GetSymbol(cast<int>(index));
402                     if (symbol->value.flags == Value.Flag.undefined)
403                     {
404                         Symbol* executableSymbol = executableSymbolTable.GetSymbol(symbol->name);
405                         if (executableSymbol != null)
406                         {
407                             symbol = executableSymbol;
408                         }
409                     }
410                     linkCommand.Reset(new LinkOnceCommand(objectFile));
411                     break;
412                 }
413                 default:
414                 {
415                     symbol = null;
416                     linkCommand.Reset();
417                     break;
418                 }
419             }
420             if (!linkCommand.IsNull() && symbol != null)
421             {
422                 if (symbol->linkage == Linkage.remove)
423                 {
424                     if (symbol->value.value >= symbol->section->copyStartPos)
425                     {
426                         linkTable.AddCopyRange(CopyRange(symbol->section, symbol->section->copyTargetSection, symbol->section->copyStartPos, symbol->value.value - symbol->section->copyStartPos));
427                     }
428                     symbol->section->copyStartPos = symbol->value.value + symbol->length;
429                     symbol->section->length = symbol->section->length - symbol->length;
430                 }
431                 if (symbol->value.GetFlag(Value.Flag.address))
432                 {
433                     if (symbol->value.value >= symbol->section->copyStartPos)
434                     {
435                         linkCommand->Execute(symbol->start);
436                     }
437                 }
438                 else if (symbol->value.flags == Value.Flag.undefined)
439                 {
440                     linkTable.AddLinkCommand(symbol->name, linkCommand.Release());
441                 }
442             }
443             x = linkSection->GetByte();
444         }
445         if (codeSection->copyStartPos < prevCodeSectionLength)
446         {
447             linkTable.AddCopyRange(CopyRange(codeSection, codeSection->copyTargetSection, codeSection->copyStartPos, prevCodeSectionLength - codeSection->copyStartPos));
448         }
449         if (dataSection->copyStartPos < prevDataSectionLength)
450         {
451             linkTable.AddCopyRange(CopyRange(dataSection, dataSection->copyTargetSection, dataSection->copyStartPos, prevDataSectionLength - dataSection->copyStartPos));
452         }
453         codeSection->copyStartPos = 0u;
454         codeSection->copyTargetSection = null;
455         dataSection->copyStartPos = 0u;
456         dataSection->copyTargetSection = null;
457     }

459     public void LinkObjectFile(LinkTable& linkTable, ExecutableFile& executable, ObjectFile* objectFile, Section*& prevCodeSection, Section*& prevDataSection)
460     {
461         SymbolTable& executableSymbolTable = executable.GetSymbolTable();
462         Section* codeSection = objectFile->codeSection;
463         if (prevCodeSection == null)
464         {
465             codeSection->baseAddress = cmsx.machine.textSegmentBaseAddress + 4096u;
466             executable.codeSection->baseAddress = codeSection->baseAddress;
467         }
468         else
469         {
470             codeSection->baseAddress = prevCodeSection->baseAddress + prevCodeSection->length;
471         }
472         prevCodeSection = codeSection;
473         Section* dataSection = objectFile->dataSection;
474         if (prevDataSection == null)
475         {
476             dataSection->baseAddress = cmsx.machine.dataSegmentBaseAddress;
477             executable.dataSection->baseAddress = dataSection->baseAddress;
478         }
479         else
480         {
481             dataSection->baseAddress = prevDataSection->baseAddress + prevDataSection->length;
482         }
483         prevDataSection = dataSection;
484         ProcessSymbols(linkTable, executable, objectFile);
485         ProcessLinkSection(linkTable, executable, objectFile);
486     }

488     public void Link(ExecutableFile& executable, List<UniquePtr<BinaryFile>>& binaryFiles)
489     {
490         LinkTable linkTable;
491         CodeSection* prevCodeSection = null;
492         DataSection* prevDataSection = null;
493         for (const UniquePtr<BinaryFile>& binaryFile : binaryFiles)
494         {
495             if (binaryFile.Get() is ObjectFile*)
496             {
497                 ObjectFile* objectFile = cast<ObjectFile*>(binaryFile.Get());
498                 LinkObjectFile(linkTable, executable, objectFile, prevCodeSection, prevDataSection);
499             }
500             else if (binaryFile.Get() is LibraryFile*)
501             {
502                 LibraryFile* libraryFile = cast<LibraryFile*>(binaryFile.Get());
503                 for (const UniquePtr<ObjectFile>& objectFile : libraryFile->objectFiles)
504                 {
505                     LinkObjectFile(linkTable, executable, objectFile.Get(), prevCodeSection, prevDataSection);
506                 }
507             }
508             else
509             {
510                 throw Exception("cannot link '" + binaryFile->FileName() + "': not an object or library file");
511             }
512         }
513         linkTable.ReportUnresolvedSymbols();
514         linkTable.CopyRanges();
515         executable.Finalize();
516         executable.Write();
517     }
