283     public void ProcessAssemblyFile(long index, const string& assemblyFilePath, bool verbose)
284     {
285         if (verbose)
286         {
287             Console.Out() << "> " << assemblyFilePath << endl();
288         }
289         string utf8Content = File.ReadAllText(assemblyFilePath);
290         ustring assemblyFileContent = ToUtf32(utf8Content);
291         List<UniquePtr<Instruction>> instructions;
292         AssemblyFileParser.Instance().Parse(index, assemblyFileContent, assemblyFilePath, &instructions);
293         string objectFilePath = Path.ChangeExtension(assemblyFilePath, ".o");
294         Assembler assembler(objectFilePath);
295         assembler.Assemble(instructions);
296         cmsx.object.LinkInternal(assembler.objectFile);
297         if (verbose)
298         {
299             Console.Out() << "==> " << objectFilePath << endl();
300         }
301         assembler.WriteObjectFile();
302     }
