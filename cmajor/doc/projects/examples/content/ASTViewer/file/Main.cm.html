101     private void OpenClick()
102     {
103         try
104         {
105             List<Pair<string, string>> descriptionFilterPairs;
106             descriptionFilterPairs.Add(Pair<string, string>("Cmajor source files (*.cm)", "*.cm"));
107             string initialDirectory;
108             const char* cmajorRoot = RtGetEnvironmentVariable("CMAJOR_ROOT");
109             string defaultFilePath;
110             if (cmajorRoot != null && *cmajorRoot != '\0')
111             {
112                 initialDirectory = cmajorRoot;
113             }
114             string filePath;
115             string currentDirectory;
116             List<string> fileNames;
117             bool selected = System.Windows.API.GetOpenFileName(Handle(), descriptionFilterPairs, initialDirectory, defaultFilePath, ".cm", 
118                 OFN_PATHMUSTEXIST | OFN_FILEMUSTEXIST, currentDirectory, fileNames);
119             if (selected)
120             {
121                 if (!fileNames.IsEmpty())
122                 {
123                     filePath = Path.Combine(currentDirectory, fileNames[0]);
124                 }
125                 if (!filePath.IsEmpty() && File.Exists(filePath))
126                 {
127                     treeView->SetRoot(null);
128                     content = ToUtf32(File.ReadAllText(filePath));
129                     lexer = CmajorLexer(content.Chars(), content.Chars() + content.Length(), filePath, 0);
130                     Cm.Parser.ParsingContext ctx;
131                     compileUnit = CompileUnitParser.Parse(lexer, &ctx);
132                     SyntaxTreeBuilder syntaxTreeBuilder(lexer);
133                     compileUnit->Accept(syntaxTreeBuilder);
134                     indexNodeMap = syntaxTreeBuilder.GetIndexNodeMap();
135                     for (const Pair<int, TreeViewNode*>& p : indexNodeMap)
136                     {
137                         indexNodeList.Add(p);
138                     }
139                     Sort(indexNodeList.Begin(), indexNodeList.End());
140                     nodeIndexMap = syntaxTreeBuilder.GetNodeIndexMap();
141                     treeView->SetRoot(syntaxTreeBuilder.GetRoot());
142                     sourceCodeView->SetTextContent(content);
143                     sourceCodeView->SetFocus();
144                 }
145                 else
146                 {
147                     throw Exception("file path is empty or does not exist");
148                 }
149             }
150         }
151         catch (const Exception& ex)
152         {
153             MessageBox.Show(ex.Message());
154         }
155     }

181     private void TreeViewNodeSelected(TreeViewNodeEventArgs& args)
182     {
183         TreeViewNode* node = args.node;
184         HashMap<TreeViewNode*, int>.ConstIterator it = nodeIndexMap.CFind(node);
185         if (it != nodeIndexMap.CEnd())
186         {
187             int index = it->second;
188             LogView* logView = Application.GetLogView();
189             if (logView != null)
190             {
191                 logView->WriteLine("index=" + ToString(index));
192             }

195         }

197     private void UpdateClick()
198     {
199         int caretLine = sourceCodeView->CaretLine();
200         int caretColumn = sourceCodeView->CaretColumn();
201         int index = sourceCodeView->GetCharIndex(caretLine, caretColumn);
202         if (index != -1)
203         {
204             Pair<int, TreeViewNode*> x(index, null);
205             List<Pair<int, TreeViewNode*>>.ConstIterator it = LowerBound(indexNodeList.CBegin(), indexNodeList.CEnd(), x);
206             if (it != indexNodeList.CBegin() && it == indexNodeList.CEnd())
207             {
208                 --it;
209             }
210             int i = -1;
211             if (it != indexNodeList.CEnd())
212             {
213                 int i = it->first;
214                 if (i > index)
215                 {
216                     if (it != indexNodeList.CBegin())
217                     {
218                         --it;
219                     }
220                 }
221             }
222             if (it != indexNodeList.CEnd())
223             {
224                 TreeViewNode* node = it->second;
225                 treeView->Root()->CollapseAll();
226                 TreeViewNode* p = node;
227                 while (p != null)
228                 {
229                     p->Expand();
230                     p = p->Parent();
231                 }
232                 treeView->Invalidate();
233                 treeView->Update();
234                 node->EnsureVisible();
235                 treeView->Invalidate();
236             }
237         }
238     }
