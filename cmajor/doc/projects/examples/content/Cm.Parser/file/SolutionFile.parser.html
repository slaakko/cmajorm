 6 using System;

11 parser SolutionFileParser

15 
16     SolutionFile(var UniquePtr<Solution> solution) : Solution*
17         ::=
18         (   SOLUTION QualifiedId:name! SEMICOLON!{ solution.Reset(new Solution(name, lexer.FileName())); }
19             (Declaration:declaration{ solution->AddDeclaration(declaration); })*
20         )
21         {
22             return solution.Release();
23         }
24         ;
25         
26     QualifiedId(var Span s) : ustring
27         ::= 
28         (
29             ID{ s = span; } (DOT ID{ s.end = span.end; })*
30         )
31         {
32             return lexer.GetMatch(s);
33         }
34         ;
35 
36     Declaration : SolutionDeclaration*
37         ::= SolutionProjectDeclaration:solutionProjectDeclaration{ return solutionProjectDeclaration; }
38         |   ActiveProjectDeclaration:activeProjectDeclaration{ return activeProjectDeclaration; }
39         ;
40         
41     SolutionProjectDeclaration(var string filePath) : SolutionDeclaration*
42         ::= PROJECT FILEPATH{ filePath = lexer.filePath; } SEMICOLON{ return new Cm.Ast.SolutionProjectDeclaration(filePath); }
43         ;
44         
45     ActiveProjectDeclaration : SolutionDeclaration*
46         ::= ACTIVEPROJECT QualifiedId:activeProjectName SEMICOLON{ return new Cm.Ast.SolutionActiveProjectDeclaration(activeProjectName); }
47         ;
48 }
