 15     public class CmajorSourceCodeView : SourceCodeView

 17         public CmajorSourceCodeView(const Point& location, const Size& size, Dock dock, Anchors anchors) : base(location, size, dock, anchors), lexer(null, null, string(), 0)
 18         {
 19             lexer.SetCommentTokenId(COMMENT);
 20             Set<int> blockCommentStates;
 21             blockCommentStates.Insert(124);
 22             blockCommentStates.Insert(125);
 23             blockCommentStates.Insert(126);
 24             blockCommentStates.Insert(127);
 25             lexer.SetBlockCommentStates(blockCommentStates);
 26         }
 27         protected override TokenLine TokenizeLine(const ustring& line, int lineNumber, int startState)
 28         {
 29             LogView* logView = Application.GetLogView();
 30             TokenLine tokenLine = lexer.TokenizeLine(line, lineNumber, startState);
 31             if (logView != null)
 32             {
 33                 logView->WriteLine("endState=" + ToString(tokenLine.endState));
 34             }
 35             int tokenIndex = 0;
 36             for (Token& token : tokenLine.tokens)
 37             {
 38                 ustring s = token.match.ToString();
 39                 string sl = MakeStringLiteral(ToUtf8(s));
 40                 switch (startState)
 41                 {
 42                     case 0:
 43                     {
 44                         if (token.id == CONTINUE_TOKEN)
 45                         {
 46                             if (s.Length() >= 2)
 47                             {
 48                                 if (s[0] == '/' && s[1] == '/')
 49                                 {
 50                                     token.id = COMMENT;
 51                                     tokenLine.endState = 0;
 52                                 }
 53                                 else if (s[0] == '/' && s[1] == '*')
 54                                 {
 55                                     token.id = COMMENT;
 56                                 }
 57                             }
 58                             else if (!s.IsEmpty())
 59                             {
 60                                 if (IsWhiteSpace(s[0]))
 61                                 {
 62                                     token.id = WS;
 63                                     if (tokenIndex == tokenLine.tokens.Count() - 1)
 64                                     {
 65                                         tokenLine.endState = 0;
 66                                     }
 67                                 }
 68                             }
 69                         }
 70                         break;
 71                     }
 72                     default:
 73                     {
 74                         if (lexer.BlockCommentStates().CFind(startState) != lexer.BlockCommentStates().CEnd())
 75                         {
 76                             token.id = COMMENT;
 77                         }
 78                         break;
 79                     }
 80                 }
 81                 if (logView != null)
 82                 {
 83                     logView->WriteLine("lexeme=" + sl + ", token=" + ToUtf8(GetTokenName(token.id)) + ", startState=" + ToString(startState));
 84                 }
 85                 startState = 0;
 86                 ++tokenIndex;
 87             }
 88             return tokenLine;
 89         }
 90         protected override nothrow SourceCodeTokenKind GetTokenKind(const Token& token) const
 91         {
 92             switch (token.id)
 93             {
 94                 case ASSERT: case ELIF: case ENDIF: case BOOL: case TRUE: case FALSE:
 95                 case SBYTE: case BYTE: case SHORT: case USHORT: case INT: case UINT:
 96                 case LONG: case ULONG: case FLOAT: case DOUBLE: case CHAR: case WCHAR:
 97                 case UCHAR: case VOID: case ENUM: case CAST: case INTERFACE: case NAMESPACE:
 98                 case USING: case STATIC: case EXTERN: case AS: case IS: case EXPLICIT:
 99                 case DELEGATE: case INLINE: case CDECL: case NOTHROW: case PUBLIC:
100                 case PROTECTED: case PRIVATE: case INTERNAL: case VIRTUAL: case ABSTRACT:
101                 case OVERRIDE: case SUPPRESS: case WINAPI: case OPERATOR: case CLASS:
102                 case RETURN: case IF: case ELSE: case SWITCH: case CASE: case DEFAULT:
103                 case WHILE: case DO: case FOR: case BREAK: case CONTINUE: case GOTO:
104                 case TYPEDEF: case TYPENAME: case TYPEID: case CONST: case CONSTEXPR:
105                 case NULLLIT: case THIS: case BASE: case CONSTRUCT: case DESTROY:
106                 case NEW: case DELETE: case SIZEOF: case TRY: case CATCH: case THROW:
107                 case CONCEPT: case WHERE: case AXIOM: case AND: case OR:
108                 {
109                     return SourceCodeTokenKind.keyword;
110                 }
111                 case ID:
112                 {
113                     return SourceCodeTokenKind.identifier;
114                 }
115                 case EXCLAMATION: case UNIT_TEST: case EQUIVALENCE: case IMPLICATION:
116                 case DISJUNCTION: case AMPAMP: case BITOR: case BITXOR: case AMP:
117                 case EQ: case NEQ: case LEQ: case GEQ: case SHIFTLEFT: case SHIFTRIGHT:
118                 case PLUS: case MINUS: case STAR: case DIV: case REM: case PLUSPLUS:
119                 case MINUSMINUS: case CPL: case DOT: case ARROW: case LBRACKET:
120                 case RBRACKET: case LPAREN: case RPAREN: case LANGLE: case RANGLE:
121                 case LBRACE: case RBRACE: case COMMA: case ASSIGN: case COLON:
122                 case SEMICOLON: case HASH: case WS:
123                 {
124                     return SourceCodeTokenKind.plain;
125                 }
126                 case FLOATINGLIT: case INTLIT:
127                 {
128                     return SourceCodeTokenKind.number;
129                 }
130                 case CHARLIT:
131                 {
132                     return SourceCodeTokenKind.character;
133                 }
134                 case STRINGLIT:
135                 {
136                     return SourceCodeTokenKind.string;
137                 }
138                 case COMMENT:
139                 {
140                     return SourceCodeTokenKind.comment;
141                 }
142             }
143             return SourceCodeTokenKind.plain;
144         }
