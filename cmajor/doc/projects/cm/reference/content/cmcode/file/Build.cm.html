 6 using System;
 7 using Cm.Service.Build;
 8 
 9 namespace cmcode
10 {
11     enum BuildRequestKind
12     {
13         none = 0, build = 1 << 0, clean = 1 << 1, rebuild = 1 << 2, buildDependencies = 1 << 3
14     }
15 
16     BuildRequest MakeBuildRequest(const string& backend, const string& config, const string& filePath, BuildRequestKind buildRequestKind)
17     {
18         BuildRequest buildRequest;
19         buildRequest.backend = backend;
20         buildRequest.config = config;
21         buildRequest.optimizationLevel = "default";
22         buildRequest.filePath = filePath;
23         buildRequest.numBuildThreads = "default";
24         buildRequest.toolChain = "default";
25         buildRequest.verbose = true;
26         if ((buildRequestKind & BuildRequestKind.clean) != BuildRequestKind.none)
27         {
28             buildRequest.clean = true;
29         }
30         if ((buildRequestKind & BuildRequestKind.rebuild) != BuildRequestKind.none)
31         {
32             buildRequest.rebuild = true;
33         }
34         if ((buildRequestKind & BuildRequestKind.buildDependencies) != BuildRequestKind.none)
35         {
36             buildRequest.buildAllDependencies = true;
37         }
38         return buildRequest;
39     }
40 }
