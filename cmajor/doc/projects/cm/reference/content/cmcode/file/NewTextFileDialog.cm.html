12     public nothrow bool IsValidTextFileName(const string& name)
13     {
14         if (name.IsEmpty()) return false;
15         ustring s(ToUtf32(name));
16         if (!IsIdStart(s[0]))
17         {
18             return false;
19         }
20         for (int i = 1; i < s.Length(); ++i;)
21         {
22             uchar c = s[i];
23             if (!IsIdCont(c) && c != '.')
24             {
25                 return false;
26             }
27         }
28         return true;
29     }

31     public class NewTextFileDialog : Window
32     {
33         public NewTextFileDialog() : base("cmcode.NewTextFileDialog", DefaultWindowClassStyle(), DialogWindowStyle(), DefaultExtendedWindowStyle(), DefaultControlBackgroundColor(), 
34             "New Text File", Point.Default(), Size(ScreenMetrics.Get().MMToHorizontalPixels(160), ScreenMetrics.Get().MMToVerticalPixels(60)), Dock.none, Anchors.none, WindowState.normal)
35         {
36             Size s = GetSize();
37             Size defaultControlSpacing = ScreenMetrics.Get().DefaultControlSpacing();
38             Size defaultButtonSize = ScreenMetrics.Get().DefaultButtonSize();
39             Size defaultTextBoxSize = ScreenMetrics.Get().DefaultTextBoxSize();
40             Size defaultLabelSize = ScreenMetrics.Get().DefaultLabelSize();
41             int borderWidth = DefaultSingleBorderWidth();
42 
43             Point textFileNameLabelLocation(16, 16);
44             UniquePtr<Label> textFileNameLabelPtr(new Label("Text file name:", textFileNameLabelLocation, defaultLabelSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left), true));
45             AddChild(textFileNameLabelPtr.Release());
46             UniquePtr<TextBox> textFileNameTextBoxPtr(new TextBox(""));
47             textFileNameTextBox = textFileNameTextBoxPtr.Get();
48             textFileNameTextBox->TextChangedEvent().AddHandler(TextFileNameTextBoxTextChanged);
49             Point textFileNameTextBoxLocation(20, 16 + 24);
50             Size textFileNameTextBoxSize(ScreenMetrics.Get().MMToHorizontalPixels(120), defaultTextBoxSize.h);
51             Rect textFileNameTextBoxRec(textFileNameTextBoxLocation, textFileNameTextBoxSize);
52             UniquePtr<Control> paddedTextFileNameTextBox = new PaddedControl(textFileNameTextBoxPtr.Release());
53             Rect paddedTextFileNameTextBoxRect = textFileNameTextBoxRec;
54             paddedTextFileNameTextBoxRect.Inflate(4, 4);
55             Rect borderedPaddedTextFileNameTextBoxRect = paddedTextFileNameTextBoxRect;
56             borderedPaddedTextFileNameTextBoxRect.Inflate(borderWidth, borderWidth);
57             UniquePtr<Control> borderedPaddedTextFileNameTextBox(new BorderedControl(paddedTextFileNameTextBox.Release(), borderedPaddedTextFileNameTextBoxRect.location, 
58                 borderedPaddedTextFileNameTextBoxRect.size, Dock.none, cast<Anchors>(Anchors.top | Anchors.left)));
59             AddChild(borderedPaddedTextFileNameTextBox.Release());
60 
61             int x = s.w - defaultButtonSize.w - defaultControlSpacing.w;
62             int y = s.h - defaultButtonSize.h - defaultControlSpacing.h;
63             UniquePtr<Button> cancelButtonPtr(new Button("Cancel", Point(x, y), defaultButtonSize, Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
64             cancelButton = cancelButtonPtr.Get();
65             cancelButton->SetDialogResult(DialogResult.cancel);
66             x = x - defaultButtonSize.w - defaultControlSpacing.w;
67             UniquePtr<Button> createButtonPtr(new Button(ButtonStyle.BS_DEFPUSHBUTTON, GetSystemColor(SystemColor.COLOR_BTNFACE), "Create", Point(x, y), defaultButtonSize, 
68                 Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
69             createButton = createButtonPtr.Get();
70             createButton->SetDialogResult(DialogResult.ok);
71             createButton->Disable();
72             AddChild(createButtonPtr.Release());
73             AddChild(cancelButtonPtr.Release());
74             SetDefaultButton(createButton);
75             SetCancelButton(cancelButton);
76         }
