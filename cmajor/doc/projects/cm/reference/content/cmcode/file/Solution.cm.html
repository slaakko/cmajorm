 19     UniquePtr<Cm.Ast.Project> ReadProject(const string& projectFilePath, List<UniquePtr<ustring>>& contents)
 20     {
 21         UniquePtr<ustring> content(new ustring(ToUtf32(File.ReadAllText(projectFilePath))));
 22         ContainerFileLexer lexer(content->Chars(), content->Chars() + content->Length(), projectFilePath, 0);
 23         UniquePtr<Cm.Ast.Project> project = ProjectFileParser.Parse(lexer, "debug", Cm.Ast.BackEnd.llvm);
 24         project->ResolveDeclarations();
 25         contents.Add(Rvalue(content));
 26         return project;
 27     }

 29     UniquePtr<Cm.Ast.Solution> ReadSolution(const string& solutionFilePath, List<UniquePtr<ustring>>& contents)
 30     {
 31         UniquePtr<ustring> content(new ustring(ToUtf32(File.ReadAllText(solutionFilePath))));
 32         ContainerFileLexer lexer(content->Chars(), content->Chars() + content->Length(), solutionFilePath, 0);
 33         UniquePtr<Cm.Ast.Solution> solution(SolutionFileParser.Parse(lexer));
 34         solution->ResolveDeclarations();
 35         for (const string& projectFilePath : solution->ProjectFilePaths())
 36         {
 37             UniquePtr<Cm.Ast.Project> project = ReadProject(projectFilePath, contents);
 38             solution->AddProject(Rvalue(project));
 39         }
 40         contents.Add(Rvalue(content));
 41         return solution;
 42     }

 44     class TreeViewNodeData
 45     {
 46         public enum Kind
 47         {
 48             solution, project, file
 49         }
 50         public nothrow TreeViewNodeData(Kind kind_, Cm.Ast.Solution* solution_, Cm.Ast.Project* project_, const string& filePath_, const string& fileName_) : 
 51             kind(kind_), solution(solution_), project(project_), filePath(filePath_), fileName(fileName_)
 52         {
 53             switch (kind)
 54             {
 55                 case Kind.solution:
 56                 {
 57                     key = System.Security.GetSha1MessageDigest("solution." + ToUtf8(solution->Name()));
 58                     break;
 59                 }
 60                 case Kind.project:
 61                 {
 62                     key = System.Security.GetSha1MessageDigest("solution." + ToUtf8(solution->Name()) + ".project." + ToUtf8(project->Name()));
 63                     break;
 64                 }
 65                 case Kind.file:
 66                 {
 67                     key = filePath;
 68                     break;
 69                 }
 70             }
 71         }
 72         public Kind kind;
 73         public Cm.Ast.Solution* solution;
 74         public Cm.Ast.Project* project;
 75         public string filePath;
 76         public string fileName;
 77         public string key;
 78     }

 80     class ByFileName : Rel<UniquePtr<TreeViewNodeData>>

 83         {

128     void SetActiveProject(Cm.Ast.Solution* solution)
129     {
130         if (solution->ActiveProjectName().IsEmpty())
131         {
132             if (!solution->Projects().IsEmpty())
133             {
134                 solution->SetActiveProject(solution->Projects()[0].Get());
135             }
136         }
137         else
138         {
139             for (const UniquePtr<Cm.Ast.Project>& project : solution->Projects())
140             {
141                 if (project->Name() == solution->ActiveProjectName())
142                 {
143                     solution->SetActiveProject(project.Get());
144                     break;
145                 }
146             }
147         }
148         if (solution->ActiveProject() == null)
149         {
150             if (!solution->Projects().IsEmpty())
151             {
152                 solution->SetActiveProject(solution->Projects()[0].Get());
153             }
154         }
155     }

157     UniquePtr<TreeViewNode> BuildSolutionTree(Cm.Ast.Solution* solution, UniquePtr<SolutionExplorerData>& solutionExplorerData, const List<UniquePtr<ustring>>& contents)
158     {
159         SetActiveProject(solution);
160         UniquePtr<SolutionExplorerData> newSolutionExplorerData(new SolutionExplorerData());
161         string solutionName = ToUtf8(solution->Name());
162         UniquePtr<TreeViewNode> solutionNode(new TreeViewNode(solutionName));
163         UniquePtr<TreeViewNodeData> solutionData(new TreeViewNodeData(TreeViewNodeData.Kind.solution, solution, null, string(), string()));
164         solutionNode->SetData(solutionData.Get());
165         newSolutionExplorerData->treeViewData.Add(Rvalue(solutionData));
166         for (const UniquePtr<Cm.Ast.Project>& project : solution->Projects())
167         {
168             string projectName = ToUtf8(project->Name());
169             UniquePtr<TreeViewNode> projectNode(new TreeViewNode(projectName));
170             UniquePtr<TreeViewNodeData> projectData(new TreeViewNodeData(TreeViewNodeData.Kind.project, solution, project.Get(), string(), string()));
171             projectNode->SetData(projectData.Get());
172             if (project.Get() == solution->ActiveProject())
173             {
174                 newSolutionExplorerData->SetActiveProjectNode(projectNode.Get());
175                 projectNode->SetActive(true);
176             }
177             newSolutionExplorerData->treeViewData.Add(Rvalue(projectData));
178             List<UniquePtr<TreeViewNodeData>> fileData;
179             for (const string& sourceFilePath : project->SourceFilePaths())
180             {
181                 UniquePtr<TreeViewNodeData> file(new TreeViewNodeData(TreeViewNodeData.Kind.file, solution, project.Get(), sourceFilePath, Path.GetFileName(sourceFilePath)));
182                 fileData.Add(Rvalue(file));
183             }
184             for (const string& resourceFilePath : project->ResourceFilePaths())
185             {
186                 UniquePtr<TreeViewNodeData> file(new TreeViewNodeData(TreeViewNodeData.Kind.file, solution, project.Get(), resourceFilePath, Path.GetFileName(resourceFilePath)));
187                 fileData.Add(Rvalue(file));
188             }
189             for (const string& textFilePath : project->TextFilePaths())
190             {
191                 UniquePtr<TreeViewNodeData> file(new TreeViewNodeData(TreeViewNodeData.Kind.file, solution, project.Get(), textFilePath, Path.GetFileName(textFilePath)));
192                 fileData.Add(Rvalue(file));
193             }
194             Sort(fileData.Begin(), fileData.End(), ByFileName());
195             for (const UniquePtr<TreeViewNodeData>& file : fileData)
196             {
197                 UniquePtr<TreeViewNode> fileNode(new TreeViewNode(file->fileName));
198                 fileNode->SetData(file.Get());
199                 projectNode->AddChild(fileNode.Release());
200             }
201             for (UniquePtr<TreeViewNodeData>& file : fileData)
202             {
203                 newSolutionExplorerData->treeViewData.Add(Rvalue(file));
204             }
205             solutionNode->AddChild(projectNode.Release());
206         }
207         solutionNode->Expand();
208         for (const UniquePtr<TreeViewNodeData>& nodeData : newSolutionExplorerData->treeViewData)
209         {
210             newSolutionExplorerData->treeViewDataMap[nodeData->key] = nodeData.Get();
211         }
212         solutionExplorerData.Reset(newSolutionExplorerData.Release());
213         return solutionNode;
214     }
