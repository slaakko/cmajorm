103         public nothrow void SetLocation(const string& location)
104         {
105             locationTextBox->SetText(location);
106         }
107         public nothrow string GetLocation() const
108         {
109             return locationTextBox->Text();
110         }
111         public nothrow string GetProjectName() const
112         {
113             return projectNameTextBox->Text();
114         }
115         public nothrow Cm.Ast.Target GetProjectType() const
116         {
117             return projectType;
118         }
119         private void ProjectNameTextBoxTextChanged()
120         {
121             if (IsValidProjectOrSolutionName(projectNameTextBox->Text()) && !locationTextBox->Text().IsEmpty())
122             {
123                 createButton->Enable();
124             }
125             else
126             {
127                 createButton->Disable();
128             }
129         }
130         private void LocationTextBoxTextChanged()
131         {
132             if (IsValidProjectOrSolutionName(projectNameTextBox->Text()) && !locationTextBox->Text().IsEmpty())
133             {
134                 createButton->Enable();
135             }
136             else
137             {
138                 createButton->Disable();
139             }
140         }
141         private void SelectProjectTypeButtonClick()
142         {
143             SelectProjectTypeDialog dialog;
144             if (dialog.ShowDialog(*this) == DialogResult.ok)
145             {
146                 projectType = dialog.GetSelectedProjectType();
147                 projectTypeText->SetText(GetProjectTypeName(projectType));
148                 projectTypeText->Invalidate();
149                 Invalidate();
150             }
151         }
152         private void SelectLocationButtonClick()
153         {
154             string folder = System.Windows.API.GetFolder(Handle(), locationTextBox->Text());
155             if (!folder.IsEmpty())
156             {
157                 SetLocation(folder);
158             }
159         }
160         private string GetProjectTypeName(Cm.Ast.Target projectType) const
161         {
162             HashMap<Cm.Ast.Target, string>.ConstIterator it = projectTypeNameMap.CFind(projectType);
163             if (it != projectTypeNameMap.CEnd())
164             {
165                 return it->second;
166             }
167             else
168             {
169                 return "Unknown project type";
170             }
171         }

174         private TextBox* projectNameTextBox;
175         private TextBox* locationTextBox;
176         private Label* projectTypeText;
177         private Cm.Ast.Target projectType;
178         private HashMap<Cm.Ast.Target, string> projectTypeNameMap;
179     }
