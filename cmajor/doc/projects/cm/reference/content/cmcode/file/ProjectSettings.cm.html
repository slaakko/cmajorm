 40     public ProjectSettings LoadProjectSettings(const string& settingsFilePath)
 41     {
 42         string settingsFileStr = File.ReadAllText(settingsFilePath);
 43         ustring settingsFileContent = ToUtf32(settingsFileStr);
 44         JsonLexer lexer(settingsFileContent, settingsFilePath, 0);
 45         UniquePtr<System.Json.JsonValue> jsonValue = JsonParser.Parse(lexer);
 46         ProjectSettings settings(jsonValue.Get());
 47         return settings;
 48     }

 50     public void SaveProjectSettings(const ProjectSettings& settings, const string& settingsFilePath)
 51     {
 52         UniquePtr<System.Json.JsonValue> jsonValue = settings.ToJson();
 53         StreamWriter writer = File.CreateText(settingsFilePath);
 54         CodeFormatter formatter(writer);
 55         jsonValue->Write(formatter);
 56     }

 58     public void ImportBreakpoints(Cm.Service.Debug.BreakpointCollection& debugBreakpoints, const ProjectSettings& projectSettings)
 59     {
 60         debugBreakpoints.Clear();
 61         for (const cmcode.settings.BreakpointList& breakpointList : projectSettings.breakpoints.breakpointLists)
 62         {
 63             Cm.Service.Debug.BreakpointList& debugBreakpointList = debugBreakpoints.GetBreakpointList(breakpointList.filePath);
 64             for (const cmcode.settings.Breakpoint& settingsBreakpoint : breakpointList.breakpoints)
 65             {
 66                 Cm.Service.Debug.Breakpoint* debugBreakpoint = new Cm.Service.Debug.Breakpoint(
 67                     ParseInt(settingsBreakpoint.line), settingsBreakpoint.condition, settingsBreakpoint.disabled);
 68                 debugBreakpointList.AddBreakpoint(debugBreakpoint);
 69              }
 70         }
 71     }
 72 
 73     public void ExportBreakpoints(ProjectSettings& projectSettings, const Cm.Service.Debug.BreakpointCollection& debugBreakpoints)
 74     {
 75         projectSettings.breakpoints.breakpointLists.Clear();
 76         for (const Pair<string, Cm.Service.Debug.BreakpointList>& filePathBreakpointListPair : debugBreakpoints.BreakpointListMap())
 77         {
 78             const Cm.Service.Debug.BreakpointList& breakpointList = filePathBreakpointListPair.second;
 79             BreakpointList settingsBreakpointList;
 80             settingsBreakpointList.filePath = breakpointList.FilePath();
 81             const List<Cm.Service.Debug.Breakpoint*>& breakpoints = breakpointList.Breakpoints();
 82             List<Cm.Service.Debug.Breakpoint*>.ConstIterator e = breakpoints.CEnd();
 83             for (List<Cm.Service.Debug.Breakpoint*>.ConstIterator i = breakpoints.CBegin(); i != e; ++i;)
 84             {
 85                 Cm.Service.Debug.Breakpoint* breakpoint = *i;
 86                 cmcode.settings.Breakpoint settingsBreakpoint;
 87                 settingsBreakpoint.line = ToString(breakpoint->line);
 88                 settingsBreakpoint.condition = breakpoint->condition;
 89                 settingsBreakpoint.disabled = breakpoint->disabled;
 90                 settingsBreakpointList.breakpoints.Add(Rvalue(settingsBreakpoint));
 91             }
 92             projectSettings.breakpoints.breakpointLists.Add(Rvalue(settingsBreakpointList));
 93         }
 94     }

 96     public void LoadProjectSettings(Cm.Ast.Project* project, SolutionExplorerData* solutionExplorerData)
 97     {
 98         string projectSettingsFilePath = Path.ChangeExtension(project->FilePath(), "cmcode.settings.json");
 99         if (File.Exists(projectSettingsFilePath))
100         {
101             cmcode.settings.ProjectSettings activeProjectSettings = cmcode.settings.LoadProjectSettings(projectSettingsFilePath);
102             solutionExplorerData->SetActiveProjectSettings(activeProjectSettings);
103             cmcode.settings.ImportBreakpoints(solutionExplorerData->breakpointCollection, activeProjectSettings);
104         }
105         else
106         {
107             cmcode.settings.ProjectSettings activeProjectSettings;
108             solutionExplorerData->SetActiveProjectSettings(activeProjectSettings);
109             solutionExplorerData->breakpointCollection.Clear();
110         }
111     }
112 
113     public void SaveProjectSettings(Cm.Ast.Project* project, SolutionExplorerData* solutionExplorerData)
114     {
115         string projectSettingsFilePath = Path.ChangeExtension(project->FilePath(), "cmcode.settings.json");
116         ExportBreakpoints(solutionExplorerData->ActiveProjectSettings(), solutionExplorerData->breakpointCollection);
117         SaveProjectSettings(solutionExplorerData->ActiveProjectSettings(), projectSettingsFilePath);
118     }
