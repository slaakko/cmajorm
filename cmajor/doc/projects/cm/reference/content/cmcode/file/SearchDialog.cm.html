  6 using System;
  7 using System.Collections;
  8 using System.Windows;
  9 
 10 namespace cmcode
 11 {
 12     public enum SearchExtent
 13     {
 14         currentFile = 0, activeProject = 1, entireSolution = 2
 15     }
 16 
 17     public class SearchDialog : Window
 18     {
 19         public SearchDialog(bool hasCurrentFile_) : base("cmcode.SearchDialog", DefaultWindowClassStyle(), DialogWindowStyle(), DefaultExtendedWindowStyle(), DefaultControlBackgroundColor(), 
 20             "Search", Point.Default(), Size(ScreenMetrics.Get().MMToHorizontalPixels(160), ScreenMetrics.Get().MMToVerticalPixels(90)), Dock.none, Anchors.none, WindowState.normal), 
 21             hasCurrentFile(hasCurrentFile_)
 22         {
 23             Size s = GetSize();
 24             Size defaultControlSpacing = ScreenMetrics.Get().DefaultControlSpacing();
 25             Size defaultButtonSize = ScreenMetrics.Get().DefaultButtonSize();
 26             Size defaultTextBoxSize = ScreenMetrics.Get().DefaultTextBoxSize();
 27             Size defaultLabelSize = ScreenMetrics.Get().DefaultLabelSize();
 28             int borderWidth = DefaultSingleBorderWidth();
 29 
 30             Point searchLabelLocation(16, 16);
 31             UniquePtr<Label> searchLabelPtr(new Label("Search:", searchLabelLocation, defaultLabelSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left), true));
 32             AddChild(searchLabelPtr.Release());
 33             UniquePtr<TextBox> searchTextBoxPtr(new TextBox(""));
 34             searchTextBox = searchTextBoxPtr.Get();
 35             searchTextBox->TextChangedEvent().AddHandler(SearchTextBoxTextChanged);
 36             Point searchTextBoxLocation(20, 16 + 24);
 37             Size searchTextBoxSize(ScreenMetrics.Get().MMToHorizontalPixels(120), defaultTextBoxSize.h);
 38             Rect searchTextBoxRec(searchTextBoxLocation, searchTextBoxSize);
 39             UniquePtr<Control> paddedSearchTextBox = new PaddedControl(searchTextBoxPtr.Release());
 40             Rect paddedSearchTextBoxRect = searchTextBoxRec;
 41             paddedSearchTextBoxRect.Inflate(4, 4);
 42             Rect borderedPaddedSearchTextBoxRect = paddedSearchTextBoxRect;
 43             borderedPaddedSearchTextBoxRect.Inflate(borderWidth, borderWidth);
 44             UniquePtr<Control> borderedPaddedSearchTextBox(new BorderedControl(paddedSearchTextBox.Release(), borderedPaddedSearchTextBoxRect.location, 
 45                 borderedPaddedSearchTextBoxRect.size, Dock.none, cast<Anchors>(Anchors.top | Anchors.left)));
 46             AddChild(borderedPaddedSearchTextBox.Release());
 47 
 48             Point wholeWordsCheckBoxLocation(16, 16 + 24 + 24);
 49             UniquePtr<CheckBox> wholeWordsCheckBoxPtr(new CheckBox("Whole words only", wholeWordsCheckBoxLocation, Size(), Dock.none, cast<Anchors>(Anchors.top | Anchors.left), true));
 50             wholeWordsCheckBox = wholeWordsCheckBoxPtr.Get();
 51             AddChild(wholeWordsCheckBoxPtr.Release());
 52 
 53             Point caseInsensitiveCheckBoxLocation(16, 16 + 16 + 24 + 24);
 54             UniquePtr<CheckBox> caseInsensitiveCheckBoxPtr(new CheckBox("Case-insensitive", caseInsensitiveCheckBoxLocation, Size(), Dock.none, cast<Anchors>(Anchors.top | Anchors.left), true));
 55             caseInsensitiveCheckBox = caseInsensitiveCheckBoxPtr.Get();
 56             AddChild(caseInsensitiveCheckBoxPtr.Release());
 57 
 58             Point regularExpressionCheckBoxLocation(16, 16 + 16 + 16 + 24 + 24);
 59             UniquePtr<CheckBox> regularExpressionCheckBoxPtr(new CheckBox("Regular expression", regularExpressionCheckBoxLocation, Size(), Dock.none, cast<Anchors>(Anchors.top | Anchors.left), true));
 60             regularExpressionCheckBox = regularExpressionCheckBoxPtr.Get();
 61             AddChild(regularExpressionCheckBoxPtr.Release());
 62 
 63             Point whereLabelLocation(16, cast<int>(16) + 16 + 16 + 16 + 24 + 24);
 64             UniquePtr<Label> whereLabelPtr(new Label("Where:", whereLabelLocation, defaultLabelSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left), true));
 65             AddChild(whereLabelPtr.Release());
 66             Point listBoxLocation(16, cast<int>(16) + 16 + 16 + 16 + 16 + 24 + 24);
 67             Size listBoxSize(ScreenMetrics.Get().MMToHorizontalPixels(70), ScreenMetrics.Get().MMToVerticalPixels(16));
 68             Rect listBoxRect(listBoxLocation, listBoxSize);
 69             listBoxRect.Inflate(-1, -1);
 70             UniquePtr<ListBox> listBoxPtr(new ListBox(listBoxRect.location, listBoxRect.size, Dock.none, Anchors.none));
 71             whereListBox = listBoxPtr.Get();
 72             UniquePtr<Control> borderedListBoxPtr(new BorderedControl(listBoxPtr.Release(), listBoxLocation, listBoxSize, Dock.none, cast< Anchors>(Anchors.top |  Anchors.left)));
 73             Control* borderedListBox = borderedListBoxPtr.Get();
 74             AddChild(borderedListBoxPtr.Release());
 75             Point realListBoxLocation = whereListBox->Location();
 76             Size realListBoxSize = whereListBox->GetSize();
 77             Rect realListBoxRect(realListBoxLocation, realListBoxSize);
 78             realListBoxRect.Inflate(1, 1);
 79             borderedListBox->SetSize(realListBoxRect.size);
 80             if (hasCurrentFile)
 81             {
 82                 whereListBox->AddItem("Current file");
 83             }
 84             whereListBox->AddItem("Active project");
 85             whereListBox->AddItem("Entire solution");
 86             whereListBox->SetSelectedIndex(0);
 87 
 88             int x = s.w - defaultButtonSize.w - defaultControlSpacing.w;
 89             int y = s.h - defaultButtonSize.h - defaultControlSpacing.h;
 90             UniquePtr<Button> cancelButtonPtr(new Button("Cancel", Point(x, y), defaultButtonSize, Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
 91             cancelButton = cancelButtonPtr.Get();
 92             cancelButton->SetDialogResult(DialogResult.cancel);
 93             x = x - defaultButtonSize.w - defaultControlSpacing.w;
 94             UniquePtr<Button> searchButtonPtr(new Button(ButtonStyle.BS_DEFPUSHBUTTON, GetSystemColor(SystemColor.COLOR_BTNFACE), "Search", Point(x, y), defaultButtonSize, 
 95                 Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
 96             searchButton = searchButtonPtr.Get();
 97             searchButton->SetDialogResult(DialogResult.ok);
 98             searchButton->Disable();
 99             AddChild(searchButtonPtr.Release());
100             AddChild(cancelButtonPtr.Release());
101             SetDefaultButton(searchButton);
102             SetCancelButton(cancelButton);
103             searchTextBox->SetFocus();
104         }
105         public nothrow const string& SeachText() const
106         {
107             return searchTextBox->Text();
108         }
109         public nothrow bool WholeWords() const
110         {
111             return wholeWordsCheckBox->Checked();
112         }
113         public nothrow bool CaseInsensitive() const
114         {
115             return caseInsensitiveCheckBox->Checked();
116         }
117         public nothrow bool RegularExpression() const
118         {
119             return regularExpressionCheckBox->Checked();
120         }
121         public nothrow SearchExtent GetSearchExtent() const
122         {
123             int selectedIndex = whereListBox->GetSelectedIndex();
124             if (!hasCurrentFile)
125             {
126                 return cast<SearchExtent>(selectedIndex + 1);
127             }
128             else
129             {
130                 return cast<SearchExtent>(selectedIndex);
131             }
132         }
133         private void SearchTextBoxTextChanged()
134         {
135             if (!searchTextBox->Text().IsEmpty())
136             {
137                 searchButton->Enable();
138             }
139             else
140             {
141                 searchButton->Disable();
142             }
143         }
144         private Button* searchButton;
145         private Button* cancelButton;
146         private TextBox* searchTextBox;
147         private CheckBox* wholeWordsCheckBox;
148         private CheckBox* caseInsensitiveCheckBox;
149         private CheckBox* regularExpressionCheckBox;
150         private ListBox* whereListBox;
151         private bool hasCurrentFile;
152     }
