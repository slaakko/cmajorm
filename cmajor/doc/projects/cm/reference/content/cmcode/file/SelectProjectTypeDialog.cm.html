  6 using System;
  7 using System.Collections;
  8 using System.Windows;
  9 
 10 namespace cmcode
 11 {
 12     public class SelectProjectTypeDialog : Window
 13     {
 14         public SelectProjectTypeDialog() : 
 15             base("cmcode.SelectProjectTypeDialog", DefaultWindowClassStyle(), DialogWindowStyle(), DefaultExtendedWindowStyle(), DefaultControlBackgroundColor(), 
 16                 "Select Project Type", Point.Default(), Size(ScreenMetrics.Get().MMToHorizontalPixels(180), ScreenMetrics.Get().MMToVerticalPixels(100)), Dock.none, Anchors.none, WindowState.normal)
 17         {
 18             Size s = GetSize();
 19             Size defaultButtonSize = ScreenMetrics.Get().DefaultButtonSize();
 20             Size defaultControlSpacing = ScreenMetrics.Get().DefaultControlSpacing();
 21             Point iconListViewLocation(defaultControlSpacing.w, defaultControlSpacing.h);
 22             UniquePtr<IconListView> iconListViewPtr(new IconListView(Point(0, 0), Size(80, 80), Dock.none, Anchors.none));
 23             iconListView = iconListViewPtr.Get();
 24             iconListView->AddItem("console.app.bitmap", "Console Application");
 25             iconListView->AddItem("win.gui.app.bitmap", "Windows GUI Application");
 26             iconListView->AddItem("lib.bitmap", "Library");
 27             iconListView->AddItem("win.lib.bitmap", "Windows Library");
 28             iconListView->SelectedIndexChangedEvent().AddHandler(IconListViewSelectedIndexChanged);
 29             iconListView->IconDoubleClickedEvent().AddHandler(IconListViewIconDoubleClicked);
 30             targetMap[0] = Cm.Ast.Target.program;
 31             targetMap[1] = Cm.Ast.Target.winguiapp;
 32             targetMap[2] = Cm.Ast.Target.library;
 33             targetMap[3] = Cm.Ast.Target.winlib;
 34             UniquePtr<BorderedControl> borderedIconListViewPtr(new BorderedControl(iconListViewPtr.Release(), BorderStyle.single, System.Windows.Color(204u, 206u, 219u), 
 35                 System.Windows.Color(204u, 206u, 219u), iconListViewLocation, Size(80, 80), Dock.none, cast<Anchors>(Anchors.top | Anchors.left)));
 36             borderedIconListView = borderedIconListViewPtr.Get();
 37             AddChild(borderedIconListViewPtr.Release());
 38             int x = s.w - defaultButtonSize.w - defaultControlSpacing.w;
 39             int y = s.h - defaultButtonSize.h - defaultControlSpacing.h;
 40             UniquePtr<Button> cancelButtonPtr(new Button("Cancel", Point(x, y), defaultButtonSize, Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
 41             cancelButton = cancelButtonPtr.Get();
 42             cancelButton->SetDialogResult(DialogResult.cancel);
 43             x = x - defaultButtonSize.w - defaultControlSpacing.w;
 44             UniquePtr<Button> selectButtonPtr(new Button(ButtonStyle.BS_DEFPUSHBUTTON, GetSystemColor(SystemColor.COLOR_BTNFACE), "Select", Point(x, y), defaultButtonSize, 
 45                 Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
 46             selectButton = selectButtonPtr.Get();
 47             selectButton->SetDialogResult(DialogResult.ok);
 48             selectButton->Disable();
 49             AddChild(selectButtonPtr.Release());
 50             AddChild(cancelButtonPtr.Release());
 51             SetDefaultButton(selectButton);
 52             SetCancelButton(cancelButton);
 53         }
 54         private void IconListViewSelectedIndexChanged()
 55         {
 56             int selectedIndex = iconListView->GetSelectedIndex();
 57             if (selectedIndex != -1)
 58             {
 59                 selectButton->Enable();
 60             }
 61             else
 62             {
 63                 selectButton->Disable();
 64             }
 65         }
 66         private void IconListViewIconDoubleClicked()
 67         {
 68             selectButton->DoClick();
 69         }
 70         public nothrow Cm.Ast.Target GetSelectedProjectType() const
 71         {
 72             int selectedIndex = iconListView->GetSelectedIndex();
 73             if (selectedIndex != -1)
 74             {
 75                 HashMap<int, Cm.Ast.Target>.ConstIterator it = targetMap.CFind(selectedIndex);
 76                 if (it != targetMap.CEnd())
 77                 {
 78                     return it->second;
 79                 }
 80             }
 81             return Cm.Ast.Target.program;
 82         }
 83         protected override void OnPaint(PaintEventArgs& args)
 84         {
 85             base->OnPaint(args);
 86             Size origSize = iconListView->GetSize();
 87             iconListView->Invalidate();
 88             iconListView->Update();
 89             Size size = iconListView->GetSize();
 90             if (size != origSize)
 91             {
 92                 size.w = size.w + 1;
 93                 size.h = size.h + 1;
 94                 borderedIconListView->SetSize(size);
 95             }
 96         }
 97         private Button* cancelButton;
 98         private Button* selectButton;
 99         private IconListView* iconListView;
100         private BorderedControl* borderedIconListView;
101         private HashMap<int, Cm.Ast.Target> targetMap;
102     }
103 }
