 17     public nothrow bool IsValidProjectOrSolutionName(const string& name)
 18     {
 19         if (name.IsEmpty()) return false;
 20         ustring s(ToUtf32(name));
 21         if (!IsIdStart(s[0]))
 22         {
 23             return false;
 24         }
 25         for (int i = 1; i < s.Length(); ++i;)
 26         {
 27             uchar c = s[i];
 28             if (!IsIdCont(c))
 29             {
 30                 return false;
 31             }
 32         }
 33         return true;
 34     }

242         {
