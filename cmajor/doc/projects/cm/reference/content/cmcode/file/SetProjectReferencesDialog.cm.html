  6 using System;
  7 using System.Collections;
  8 using System.Windows;
  9 
 10 namespace cmcode
 11 {
 12     public class SetProjectReferencesDialog : Window
 13     {
 14         public SetProjectReferencesDialog(Cm.Ast.Project* project_, Cm.Ast.Solution* solution_) : base("cmcode.ReferencesDialog", 
 15             DefaultWindowClassStyle(), DialogWindowStyle(), DefaultExtendedWindowStyle(), DefaultControlBackgroundColor(), 
 16             ToUtf8(project_->Name()) + " Project References", Point.Default(), Size(ScreenMetrics.Get().MMToHorizontalPixels(160), 
 17             ScreenMetrics.Get().MMToVerticalPixels(60)), Dock.none, Anchors.none, WindowState.normal), 
 18             project(project_), solution(solution_)
 19         {
 20             Size s = GetSize();
 21             Size defaultControlSpacing = ScreenMetrics.Get().DefaultControlSpacing();
 22             Size defaultButtonSize = ScreenMetrics.Get().DefaultButtonSize();
 23 
 24             UniquePtr<GroupBox> groupBoxPtr(new GroupBox("Project References", Point(16, 16), 
 25                 Size(ScreenMetrics.Get().MMToHorizontalPixels(120), ScreenMetrics.Get().MMToVerticalPixels(40)), 
 26                 Dock.none, cast<Anchors>(Anchors.top | Anchors.left)));
 27             List<Cm.Ast.Project*> referencedProjects = Cm.Ast.GetReferencedProjects(project, solution);
 28             Point loc(16, 24);
 29 
 30             for (const UniquePtr<Cm.Ast.Project>& proj : solution->Projects())
 31             {
 32                 if (proj->GetTarget() == Cm.Ast.Target.library || proj->GetTarget() == Cm.Ast.Target.winlib)
 33                 {
 34                     bool disable = false;
 35                     if (proj.Get() == project)
 36                     {
 37                         disable = true;
 38                     }
 39                     else
 40                     {
 41                         Set<Cm.Ast.Project*> projReferences = Cm.Ast.GetAllReferencedProjects(proj.Get(), solution);
 42                         if (projReferences.CFind(project) != projReferences.CEnd())
 43                         {
 44                             disable = true;
 45                         }
 46                     }
 47                     UniquePtr<CheckBox> checkBoxPtr(new CheckBox(ToUtf8(proj->Name()), loc, Size(), Dock.none, 
 48                         cast<Anchors>(Anchors.top | Anchors.left), true));
 49                     if (disable)
 50                     {
 51                         checkBoxPtr->Disable();
 52                     }
 53                     libraryProjects.Add(MakePair(checkBoxPtr.Get(), proj.Get()));
 54                     if (Find(referencedProjects.Begin(), referencedProjects.End(), proj.Get()) != referencedProjects.End())
 55                     {
 56                         checkBoxPtr->SetChecked(true);
 57                     }
 58                     groupBoxPtr->AddChild(checkBoxPtr.Release());
 59                     loc.Offset(0, 16);
 60                 }
 61             }
 62             loc.Offset(0, 16);
 63             Size gbs = groupBoxPtr->GetSize();
 64             GroupBox* groupBox = groupBoxPtr.Get();
 65             groupBox->SetSize(Size(gbs.w, loc.y));
 66             AddChild(groupBoxPtr.Release());
 67 
 68             Size sz = s;
 69             sz.h = groupBox->Location().y;
 70             sz.h = sz.h + groupBox->GetSize().h;
 71             sz.h = sz.h + 2 * (defaultButtonSize.h + defaultControlSpacing.h + 64);
 72             SetSize(sz);
 73             s = GetSize();
 74             
 75             int x = s.w - defaultButtonSize.w - defaultControlSpacing.w;
 76             int y = s.h - defaultButtonSize.h - defaultControlSpacing.h;
 77             UniquePtr<Button> cancelButtonPtr(new Button("Cancel", Point(x, y), defaultButtonSize, Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
 78             cancelButton = cancelButtonPtr.Get();
 79             cancelButton->SetDialogResult(DialogResult.cancel);
 80             x = x - defaultButtonSize.w - defaultControlSpacing.w;
 81             UniquePtr<Button> okButtonPtr(new Button(ButtonStyle.BS_DEFPUSHBUTTON, GetSystemColor(SystemColor.COLOR_BTNFACE), "OK", Point(x, y), defaultButtonSize, 
 82                 Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
 83             okButton = okButtonPtr.Get();
 84             okButton->SetDialogResult(DialogResult.ok);
 85             AddChild(okButtonPtr.Release());
 86             AddChild(cancelButtonPtr.Release());
 87             SetDefaultButton(okButton);
 88             SetCancelButton(cancelButton);
 89         }
 90         public nothrow List<Cm.Ast.Project*> GetReferencedProjects() const
 91         {
 92             List<Cm.Ast.Project*> referencedProjects;
 93             long n = libraryProjects.Count();
 94             for (long i = 0; i < n; ++i;)
 95             {
 96                 const Pair<CheckBox*, Cm.Ast.Project*>& p = libraryProjects[i];
 97                 CheckBox* checkBox = p.first;
 98                 if (checkBox->Checked())
 99                 {
100                     Cm.Ast.Project* project = p.second;
101                     referencedProjects.Add(project);
102                 }
103             }
104             return referencedProjects;
105         }
106         private Cm.Ast.Project* project;
107         private Cm.Ast.Solution* solution;
108         private List<Pair<CheckBox*, Cm.Ast.Project*>> libraryProjects;
109         private Button* okButton;
110         private Button* cancelButton;
111     }
112 }
