168     public class MainConfiguration
169     {

171         {

173         public static nothrow MainConfiguration& Instance()
174         {
175             return *instance;
176         }
177         private MainConfiguration()
178         {
179             string cmajorRoot;
180             const char* cmajorRootEnv = RtGetEnvironmentVariable("CMAJOR_ROOT");
181             if (cmajorRootEnv != null)
182             {
183                 cmajorRoot = GetFullPath(cmajorRootEnv);
184             }
185             else
186             {
187                 error = "CMAJOR_ROOT environment variable not set";
188             }
189             if (!cmajorRoot.IsEmpty())
190             {
191                 jsonFilePath = Path.Combine(Path.Combine(cmajorRoot, "config"), "cmcode.config.json");
192             }
193         }
194         public void Save(bool force)
195         {
196             try
197             {
198                 if (!force)
199                 {
200                     if (!dirty) return;
201                     if (Now() - configChangedTime < Duration.FromSeconds(3))
202                     {
203                         return;
204                     }
205                 }
206                 dirty = false;
207                 if (!jsonFilePath.IsEmpty())
208                 {
209                     UniquePtr<JsonValue> mainConfigJsonValue = mainConfig.ToJson();
210                     StreamWriter writer(File.CreateText(jsonFilePath));
211                     CodeFormatter formatter(writer);
212                     mainConfigJsonValue->Write(formatter);
213                 }
214             }
215             catch (const Exception& ex)
216             {
217             }
218         }
219         public void Load()
220         {
221             try
222             {
223                 if (!error.IsEmpty())
224                 {
225                     throw Exception(error);
226                 }
227                 if (!jsonFilePath.IsEmpty())
228                 {
229                     if (File.Exists(jsonFilePath))
230                     {
231                         string content = File.ReadAllText(jsonFilePath);
232                         UniquePtr<JsonValue> mainConfigJsonValue = ParseJson(content);
233                         FromJson(mainConfigJsonValue.Get(), mainConfig);
234                         loaded = true;
235                     }
236                 }
237             }
238             catch (const Exception& ex)
239             {
240                 MessageBox.Show("settings could not be loaded: " + ex.Message(), "Warning", null, cast<MessageBoxType>(MessageBoxType.MB_OK | MessageBoxType.MB_ICONSTOP));
241             }
242         }
