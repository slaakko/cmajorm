148     public class SetActiveProjectAction : ClickAction
149     {
150         public nothrow SetActiveProjectAction(MainWindow* mainWindow_, TreeView* treeView_, SolutionExplorerData* solutionExplorerData_, Cm.Ast.Solution* solution_, 
151             Cm.Ast.Project* project_, TreeViewNode* newActiveProjectNode_) : 
152             base(), mainWindow(mainWindow_), treeView(treeView_), solutionExplorerData(solutionExplorerData_), solution(solution_), project(project_), 
153             newActiveProjectNode(newActiveProjectNode_)
154         {
155         }
156         public override void Execute()
157         {
158             mainWindow->SaveBackendConfig();
159             TreeViewNode* activeProjectNode = solutionExplorerData->GetActiveProjectNode();
160             activeProjectNode->SetActive(false);
161             newActiveProjectNode->SetActive(true);
162             solutionExplorerData->SetActiveProjectNode(newActiveProjectNode);
163             solution->SetActiveProject(project);
164             cmcode.settings.LoadProjectSettings(project, solutionExplorerData);
165             mainWindow->HideContextMenu();
166             treeView->Invalidate();
167             mainWindow->SetControlStatus(UserInterfaceAction.none);
168             mainWindow->RestoreBackendConfig();
169         }
170         private MainWindow* mainWindow;
171         private TreeView* treeView;
172         private SolutionExplorerData* solutionExplorerData;
173         private Cm.Ast.Solution* solution;
174         private Cm.Ast.Project* project;
175         private TreeViewNode* newActiveProjectNode;
176     }
