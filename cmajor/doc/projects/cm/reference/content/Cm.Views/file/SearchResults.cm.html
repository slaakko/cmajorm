  6 using System;
  7 using System.Collections;
  8 using System.Windows;
  9 
 10 namespace Cm.Views
 11 {
 12     public class SearchResult
 13     {
 14         public nothrow SearchResult(const string& fileName_, long lineNumber_, const string& line_) : fileName(fileName_), lineNumber(lineNumber_), line(line_)
 15         {
 16         }
 17         public string fileName;
 18         public long lineNumber;
 19         public string line;
 20     }
 21 
 22     public class ViewSearchResultEventArgs
 23     {
 24         public nothrow ViewSearchResultEventArgs(SearchResult* searchResult_) : searchResult(searchResult_)
 25         {
 26         }
 27         public SearchResult* searchResult;
 28     }
 29 
 30     public class delegate void ViewSearchResultEventHandler(ViewSearchResultEventArgs& args);
 31 
 32     public class SearchResults : Control
 33     {
 34         public SearchResults() : base("Cm.Views.SearchResults", DefaultWindowClassStyle(), DefaultChildWindowStyle(), DefaultExtendedWindowStyle(), Color.White(), 
 35             "searchResults", Point(), Size(), Dock.fill, Anchors.none), container(this), treeView(null), child(null)
 36         {
 37             UniquePtr<TreeView> treeViewPtr(new TreeView(Point(), Size(), Dock.none, Anchors.none));
 38             treeView = treeViewPtr.Get();
 39             treeView->NodeDoubleClickEvent().AddHandler(TreeViewNodeDoubleClick);
 40             UniquePtr<Control> paddedTreeView(new PaddedControl(treeViewPtr.Release(), Padding(4, 4, 4, 4), Point(0, 0), Size(0, 0), Dock.none, Anchors.none));
 41             UniquePtr<Control> scrollableTreeView(new ScrollableControl(paddedTreeView.Release(), Point(0, 0), Size(0, 0), Dock.fill, Anchors.none));
 42             child = scrollableTreeView.Get();
 43             container.AddChild(scrollableTreeView.Release());
 44             Clear();
 45         }
 46         public void Clear()
 47         {
 48             treeView->SetRoot(new TreeViewNode("Search Results"));
 49             child->Invalidate();
 50         }
 51         public void Add(SearchResult* searchResult)
 52         {
 53             searchResults.Add(UniquePtr<SearchResult>(searchResult));
 54             string text = searchResult->fileName;
 55             text.Append(':').Append(ToString(searchResult->lineNumber));
 56             TreeViewNode* resultNode = new TreeViewNode(text);
 57             resultNode->SetData(searchResult);
 58             resultNode->AddChild(new TreeViewNode(searchResult->line));
 59             treeView->Root()->AddChild(resultNode);
 60         }
 61         private void TreeViewNodeDoubleClick(TreeViewNodeMouseClickEventArgs& args)
 62         {
 63             SearchResult* searchResult = cast<SearchResult*>(args.node->Data());
 64             OnViewSearchResult(searchResult);
 65         }
 66         protected override void OnLocationChanged()
 67         {
 68             base->OnLocationChanged();
 69             SetChildPos();
 70         }
 71         protected override void OnSizeChanged(uint windowState)
 72         {
 73             base->OnSizeChanged(windowState);
 74             SetChildPos();
 75         }
 76         private void SetChildPos()
 77         {
 78             Point loc;
 79             Size size = GetSize();
 80             Rect childRect(loc, size);
 81             child->SetLocation(childRect.location);
 82             child->SetSize(childRect.size);
 83         }
 84         protected override void OnPaint(PaintEventArgs& args)
 85         {
 86             treeView->Root()->ExpandAll();
 87             treeView->Invalidate();
 88             child->Invalidate();
 89         }
 90         protected virtual void OnViewSearchResult(SearchResult* searchResult)
 91         {
 92             ViewSearchResultEventArgs args(searchResult);
 93             viewSearchResultEvent.Fire(args);
 94         }
 95         public nothrow Event<ViewSearchResultEventHandler, ViewSearchResultEventArgs>& ViewSearchResultEvent()
 96         {
 97             return viewSearchResultEvent;
 98         }
 99         private Container container;
100         private TreeView* treeView;
101         private Control* child;
102         private List<UniquePtr<SearchResult>> searchResults;
103         private Event<ViewSearchResultEventHandler, ViewSearchResultEventArgs> viewSearchResultEvent;
104     }
105 }
