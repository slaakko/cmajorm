 6 using System;
 7 using System.Collections;
 8 using System.Windows;
 9 
10 namespace Cm.Views
11 {
12     public class Bitmaps
13     {
14         static Bitmaps() : instance(new Bitmaps())
15         {
16         }
17         public static nothrow Bitmaps& Instance()
18         {
19             return *instance;
20         }
21         public void Clear()
22         {
23             bitmaps.Clear();
24             bitmapMap.Clear();
25         }
26         public Bitmap* GetBitmap(const string& bitmapName)
27         {
28             HashMap<string, Bitmap*>.ConstIterator it = bitmapMap.CFind(bitmapName);
29             if (it != bitmapMap.CEnd())
30             {
31                 return it->second;
32             }
33             UniquePtr<Bitmap> bitmapPtr(new Bitmap(Bitmap.FromResource(bitmapName)));
34             Bitmap* bitmap = bitmapPtr.Get();
35             bitmapMap[bitmapName] = bitmap;
36             bitmaps.Add(Rvalue(bitmapPtr));
37             return bitmap;
38         }
39         private nothrow Bitmaps()
40         {
41         }
42         private static UniquePtr<Bitmaps> instance;
43         private HashMap<string, Bitmap*> bitmapMap;
44         private List<UniquePtr<Bitmap>> bitmaps;
45     }
46 }
