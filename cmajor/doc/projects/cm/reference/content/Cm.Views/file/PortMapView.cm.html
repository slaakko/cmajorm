  6 using System;
  7 using System.Collections;
  8 using System.Windows;
  9 
 10 namespace Cm.Views
 11 {
 12     public class PortMapView : Control
 13     {
 14         public PortMapView(MenuItem* startServerMenuItem_, MenuItem* stopServerMenuItem_) : 
 15             base("Cm.Views.PortMapView", DefaultWindowClassStyle(), DefaultChildWindowStyle(), DefaultExtendedWindowStyle(), Color.White(), 
 16             "portMapView", Point(), Size(), Dock.fill, Anchors.none), container(this), treeView(null), child(null), 
 17             startServerMenuItem(startServerMenuItem_), stopServerMenuItem(stopServerMenuItem_)
 18         {
 19             UniquePtr<TreeView> treeViewPtr(new TreeView(Point(), Size(), Dock.none, Anchors.none));
 20             treeView = treeViewPtr.Get();
 21             UniquePtr<Control> paddedTreeView(new PaddedControl(treeViewPtr.Release(), Padding(4, 4, 4, 4), Point(0, 0), Size(0, 0), Dock.none, Anchors.none));
 22             UniquePtr<Control> scrollableTreeView(new ScrollableControl(paddedTreeView.Release(), Point(0, 0), Size(0, 0), Dock.fill, Anchors.none));
 23             child = scrollableTreeView.Get();
 24             container.AddChild(scrollableTreeView.Release());
 25             Clear();
 26         }
 27         public void Clear()
 28         {
 29             treeView->SetRoot(new TreeViewNode("Port Map Entries"));
 30             child->Invalidate();
 31         }
 32         protected override void OnPaint(PaintEventArgs& args)
 33         {
 34             treeView->Root()->ExpandAll();
 35             treeView->Invalidate();
 36             child->Invalidate();
 37         }
 38         public void Update()
 39         {
 40             try
 41             {
 42                 Clear();
 43                 List<Cm.Service.PortMap.PortLease> portLeases = Cm.Service.PortMap.GetPortLeases();
 44                 for (const Cm.Service.PortMap.PortLease& portLease : portLeases)
 45                 {
 46                     TreeViewNode* node = new TreeViewNode("Port " + portLease.portNumber);
 47                     treeView->Root()->AddChild(node);
 48                     TreeViewNode* programNode = new TreeViewNode("Program: " + portLease.programName);
 49                     node->AddChild(programNode);
 50                     TreeViewNode* pidNode = new TreeViewNode("PID: " + portLease.pid);
 51                     node->AddChild(pidNode);
 52                     TreeViewNode* leaseStartTimeNode = new TreeViewNode("Lease start time: " + portLease.leaseStartTime);
 53                     node->AddChild(leaseStartTimeNode);
 54                     TreeViewNode* leaseRenewalTimeNode = new TreeViewNode("Lease renewal time: " + portLease.leaseRenewalTime);
 55                     node->AddChild(leaseRenewalTimeNode);
 56                     TreeViewNode* leaseStateNode = new TreeViewNode("Lease state: " + portLease.leaseState);
 57                     node->AddChild(leaseStateNode);
 58                 }
 59                 TreeViewNode* runningNode = new TreeViewNode("cmpms running");
 60                 TreeViewNode* countNode = new TreeViewNode(ToString(portLeases.Count()) + " ports in the port map");
 61                 runningNode->AddChild(countNode);
 62                 treeView->Root()->AddChild(runningNode);
 63                 Invalidate();
 64                 startServerMenuItem->Disable();
 65                 stopServerMenuItem->Enable();
 66             }
 67             catch (const Exception& ex)
 68             {
 69                 MessageBox.Show("Update failed: " + ex.Message());
 70                 SetStopped();
 71             }
 72         }
 73         public void SetStopped()
 74         {
 75             try
 76             {
 77                 Clear();
 78                 TreeViewNode* stoppedNode = new TreeViewNode("cmpms stopped");
 79                 treeView->Root()->AddChild(stoppedNode);
 80                 Invalidate();
 81                 stopServerMenuItem->Disable();
 82                 startServerMenuItem->Enable();
 83             }
 84             catch (const Exception& ex)
 85             {
 86                 MessageBox.Show(ex.Message());
 87             }
 88         }
 89         protected override void OnLocationChanged()
 90         {
 91             base->OnLocationChanged();
 92             SetChildPos();
 93         }
 94         protected override void OnSizeChanged(uint windowState)
 95         {
 96             base->OnSizeChanged(windowState);
 97             SetChildPos();
 98         }
 99         private void SetChildPos()
100         {
101             Point loc;
102             Size size = GetSize();
103             Rect childRect(loc, size);
104             child->SetLocation(childRect.location);
105             child->SetSize(childRect.size);
106         }
107         private Container container;
108         private TreeView* treeView;
109         private Control* child;
110         private MenuItem* startServerMenuItem;
111         private MenuItem* stopServerMenuItem;
112     }
113 }
