 52     public class DebugStrip : Control
 53     {
 54         public DebugStrip(int width, const Color& backgroundColor, BreakpointList& breakpointList_, TextView* view_) : 
 55             base("Cm.Views.DebugStrip", DefaultWindowClassStyle(), DefaultChildWindowStyle(), DefaultExtendedWindowStyle(), 
 56                 backgroundColor, "debugStrip", Point(), Size(width, 0), Dock.left, Anchors.none), breakpointList(breakpointList_), view(view_), 
 57                 breakpointColor(DefaultBreakpointColor()), redBrush(breakpointColor), whiteBrush(Color.White()), redPen(breakpointColor), 
 58                 topLineNumber(1), origin(), arrowBitmap(Bitmaps.Instance().GetBitmap("cm.views.arrow.bitmap")), 
 59                 arrowBitmapSize(Size(cast<int>(arrowBitmap->GetWidth()), cast<int>(arrowBitmap->GetHeight())))
 60         {
 61             view->ContentLocationChangedEvent().AddHandler(ViewContentLocationChanged);
 62             view->LineInsertedEvent().AddHandler(ViewLineInserted);
 63             view->LineDeletedEvent().AddHandler(ViewLineDeleted);
 64         }
 65         private void ViewContentLocationChanged()
 66         {
 67             topLineNumber = view->TopLine();
 68             origin = view->ContentLocation();
 69             Invalidate();
 70         }
 71         private void ViewLineInserted(LineEventArgs& args)
 72         {
 73             bool changed = false;
 74             int line = args.lineIndex + 1 - 1;
 75             for (Breakpoint* breakpoint : breakpointList.Breakpoints())
 76             {
 77                 if (breakpoint->line >= line)
 78                 {
 79                     ++breakpoint->line;
 80                     changed = true;
 81                 }
 82             }
 83             if (changed)
 84             {
 85                 Invalidate();
 86             }
 87         }
 88         private void ViewLineDeleted(LineEventArgs& args)
 89         {
 90             bool changed = false;
 91             List<Breakpoint*> breakpointsToDelete;
 92             int line = args.lineIndex + 1;
 93             for (Breakpoint* breakpoint : breakpointList.Breakpoints())
 94             {
 95                 if (breakpoint->line == line)
 96                 {
 97                     breakpointsToDelete.Add(breakpoint);
 98                 }
 99                 else if (breakpoint->line > line)
100                 {
101                     --breakpoint->line;
102                     changed = true;
103                 }
104             }
105             for (Breakpoint* breakpointToDelete : breakpointsToDelete)
106             {
107                 breakpointList.RemoveBreakpoint(breakpointToDelete);
108                 changed = true;
109             }
110             if (changed)
111             {
112                 Invalidate();
113             }
114         }
115         protected override void OnMouseDown(MouseEventArgs& args)
116         {
117             try
118             {
119                 Point location = args.location;
120                 view->TranslateMousePos(location);
121                 int line = cast<int>(location.y / view->CharHeight() + 1);
122                 if (args.buttons == MouseButtons.lbutton && args.clicks == 1)
123                 {
124                     Breakpoint* bp = GetBreakpoint(line);
125                     if (bp != null)
126                     {
127                         RemoveBreakpoint(bp);
128                         Invalidate();
129                     }
130                     else
131                     {
132                         Breakpoint* bp = new Breakpoint(line);
133                         AddBreakpoint(bp);
134                         Invalidate();
135                     }
136                 }
137                 else if (args.buttons == MouseButtons.rbutton && args.clicks == 1)
138                 {
139                     // todo    
140                 }
141             }
142             catch (const Exception& ex)
143             {
144                 MessageBox.Show(ex.Message(), "Error", null, cast<MessageBoxType>(MessageBoxType.MB_OK | MessageBoxType.MB_ICONSTOP));
145             }
146         }
147         protected override void OnPaint(PaintEventArgs& args)
148         {
149             try
150             {
151                 args.graphics.Clear(BackgroundColor());
152                 if (breakpointList.IsEmpty() && debugLocation == SourceSpan()) return;
153                 SmoothingMode prevSmoothingMode = args.graphics.GetSmoothingModeChecked();
154                 args.graphics.SetSmoothingModeChecked(SmoothingMode.highQuality);
155                 Point myOrigin = origin;
156                 view->TranslateContentLocation(myOrigin);
157                 int visibleLineCount = view->GetVisibleLineCount();
158                 float topLineDiff = view->TopLineDiff();
159                 PointF o(myOrigin.x, myOrigin.y);
160                 float charHeight = view->CharHeight();
161                 Size s = GetSize();
162                 SizeF size(s.w, s.w);
163                 for (int line = topLineNumber; line < topLineNumber + visibleLineCount; ++line;)
164                 {
165                     Breakpoint* bp = GetBreakpoint(line);
166                     RectF r(o, size);
167                     Size inflate = DefaultBreakpointInflate();
168                     r.Inflate(inflate.w, inflate.h);
169                     Point offset = DefaultBreakpointOffset();
170                     r.Offset(offset.x, offset.y);
171                     r.Offset(0.0f, topLineDiff);
172                     if (bp != null)
173                     {
174                         if (bp->disabled)
175                         {
176                             args.graphics.FillEllipseChecked(whiteBrush, r);
177                             args.graphics.DrawEllipseChecked(redPen, r);
178                         }
179                         else
180                         {
181                             args.graphics.FillEllipseChecked(redBrush, r);
182                         }
183                     }
184                     if (debugLocation.line == line)
185                     {
186                         DrawArrow(args.graphics, r.location);
187                     }
188                     o.y = o.y + charHeight;
189                 }
190                 args.graphics.SetSmoothingModeChecked(prevSmoothingMode);
191                 base->OnPaint(args);
192             }
193             catch (const Exception& ex)
194             {
195                 MessageBox.Show(ex.Message(), "Error", null, cast<MessageBoxType>(MessageBoxType.MB_OK | MessageBoxType.MB_ICONSTOP));
196             }
197         }

202         public void AddBreakpoint(Breakpoint* breakpoint)
203         {
204             breakpointList.AddBreakpoint(breakpoint);

206         }
207         protected virtual void OnBreakpointAdded(const string& filePath, Breakpoint* breakpoint)
208         {
209             AddBreakpointEventArgs args;
210             args.filePath = filePath;
211             args.breakpoint = breakpoint;
212             breakpointAddedEvent.Fire(args);
213             breakpoint->disabled = !breakpoint->info.success;
214             Invalidate();
215         }
216         public void RemoveBreakpoint(Breakpoint* breakpoint)
217         {
218             string breakpointId = breakpoint->info.breakpointId;
219             breakpointList.RemoveBreakpoint(breakpoint);
220             OnBreakpointRemoved(breakpointId);
221         }
222         protected virtual void OnBreakpointRemoved(const string& breakpointId)
223         {
224             RemoveBreakpointEventArgs args;
225             args.breakpointId = breakpointId;
226             breakpointRemovedEvent.Fire(args);
227             Invalidate();
228         }

241         private void DrawArrow(Graphics& graphics, const PointF& location)
242         {
243             PointF bitmapLocation(location);
244             Point offset = DefaultArrowOffset();
245             bitmapLocation.x = bitmapLocation.x + offset.x;
246             bitmapLocation.y = bitmapLocation.y + offset.y;
247             RectF r(bitmapLocation, SizeF(arrowBitmapSize.w, arrowBitmapSize.h));
248             ImageAttributes attributes;
249             attributes.SetColorKey(TransparentColor(), TransparentColor(), ColorAdjustType.default_);
250             RectF s(PointF(), SizeF(arrowBitmapSize.w, arrowBitmapSize.h));
251             graphics.DrawImageChecked(*arrowBitmap, r, s, Unit.pixel, attributes);
252         }
