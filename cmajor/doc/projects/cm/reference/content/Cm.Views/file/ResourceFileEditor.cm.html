 6 using System;
 7 using System.Windows;
 8 
 9 namespace Cm.Views
10 {
11     public class ResourceFileEditor : Editor
12     {
13         public ResourceFileEditor(const string& filePath) : base(filePath)
14         {
15             UniquePtr<TextView> textViewPtr(new TextView(Point(), Size(), Dock.none, Anchors.none));
16             textView = textViewPtr.Get();
17             UniquePtr<ScrollableControl> scrollableTextView(new ScrollableControl(textViewPtr.Release(), Point(), Size(), Dock.fill, Anchors.none));
18             AddChild(scrollableTextView.Release());
19             ustring content = ToUtf32(File.ReadAllText(filePath));
20             textView->SetTextContent(content);
21             textView->SetFixed();
22         }
23         public override nothrow TextView* GetTextView() const
24         {
25             return textView;
26         }
27         public override void SaveTo(const string& filePath)
28         {
29             textView->SaveText(filePath);
30         }
31         public override nothrow bool IsDirty() const
32         {
33             return textView->IsDirty();
34         }
35         public override nothrow void ResetDirty()
36         {
37             textView->ResetDirty();
38         }
39         public override nothrow void Select()
40         {
41             textView->Select();
42         }
43         public override nothrow void Undo()
44         {
45             textView->Undo();
46         }
47         public override nothrow void Redo()
48         {
49             textView->Redo();
50         }
51         private TextView* textView;
52     }
53 }
