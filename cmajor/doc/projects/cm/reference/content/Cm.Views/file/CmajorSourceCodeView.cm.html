 85             if (bp != null)

 97         protected override void OnMouseHover(MouseEventArgs& args)
 98         {
 99             try
100             {
101                 base->OnMouseHover(args);
102                 if (GenerateExpressionHoverEvents())
103                 {
104                     int lineNumber = 0;
105                     short columnNumber = 0;
106                     GetLineColumn(args.location, lineNumber, columnNumber);
107                     if (lineNumber >= 1 && lineNumber <= Lines().Count() && columnNumber >= 1 && columnNumber <= GetLineLength(lineNumber))
108                     {
109                         ustring tokenText = GetTokenText(lineNumber, columnNumber);
110                         if (!tokenText.IsEmpty())
111                         {
112                             if (tokenText != prevTokenText)
113                             {
114                                 SetExpressionHovered();
115                                 prevTokenText = tokenText;
116                                 expressionHoverTime = Now();
117                                 OnExpressionHovered(tokenText, args.location);
118                                 SetTimer(expressionHoverTimerId, 3000u);
119                                 return;
120                             }
121                         }
122                         else
123                         {
124                             prevTokenText.Clear();
125                         }
126                     }
127                     else
128                     {
129                         prevTokenText.Clear();
130                     }
131                 }
132             }
133             catch (const Exception& ex)
134             {
135                 MessageBox.Show(ex.Message(), "Error", null, cast<MessageBoxType>(MessageBoxType.MB_OK | MessageBoxType.MB_ICONSTOP));
136             }
137         }
138         protected override void OnTimer(TimerEventArgs& args)
139         {
140             base->OnTimer(args);
141             if (args.timerId == expressionHoverTimerId && ExpressionHovered())
142             {
143                 KillTimer(expressionHoverTimerId);
144                 ResetExpressionHovered();
145                 OnExpressionExitHover();
146                 prevTokenText.Clear();
147             }
148         }

170         private void DrawDebugLocationHilite(Graphics& graphics, const SourceSpan& debugLocation, const PointF& origin)
171         {
172             string hiliteText = ToUtf8(GetText(debugLocation));
173             int n = Max(cast<int>(0), cast<int>(debugLocation.ecol - debugLocation.scol));
174             RectF r(PointF(origin.x + CharWidth() * (debugLocation.scol - 1), origin.y), SizeF(CharWidth() * n, CharHeight()));
175             Brush* backgroundBrush = GetOrInsertBrush(debugLocationBackgroundColor);
176             Padding padding = DebugLocationPadding();
177             RectF f(PointF(r.location.x - padding.left, r.location.y - padding.top), SizeF(r.size.w + padding.Horizontal(), r.size.h + padding.Vertical()));
178             graphics.FillRectangleChecked(*backgroundBrush, f);
179             if (!hiliteText.IsEmpty())
180             {
181                 Brush* textBrush = GetOrInsertBrush(debugLocationTextColor);
182                 Font* font = GetOrInsertFont(FontStyle.regular);
183                 graphics.DrawStringChecked(hiliteText, *font, r.location, *textBrush);
184             }

186         protected override TokenLine TokenizeLine(const ustring& line, int lineNumber, int startState)
187         {
188             TokenLine tokenLine = lexer.TokenizeLine(line, lineNumber, startState);
189             int tokenIndex = 0;
190             for (Token& token : tokenLine.tokens)
191             {
192                 ustring s = token.match.ToString();
193                 string sl = MakeStringLiteral(ToUtf8(s));
194                 switch (startState)
195                 {
196                     case 0:
197                     {
198                         if (token.id == CONTINUE_TOKEN)
199                         {
200                             if (s.Length() >= 2)
201                             {
202                                 if (s[0] == '/' && s[1] == '/')
203                                 {
204                                     token.id = COMMENT;
205                                     tokenLine.endState = 0;
206                                 }
207                                 else if (s[0] == '/' && s[1] == '*')
208                                 {
209                                     token.id = COMMENT;
210                                 }
211                             }
212                             else if (!s.IsEmpty())
213                             {
214                                 if (IsWhiteSpace(s[0]))
215                                 {
216                                     token.id = WS;
217                                     if (tokenIndex == tokenLine.tokens.Count() - 1)
218                                     {
219                                         tokenLine.endState = 0;
220                                     }
221                                 }
222                             }
223                         }
224                         break;
225                     }
226                     default:
227                     {
228                         if (lexer.BlockCommentStates().CFind(startState) != lexer.BlockCommentStates().CEnd())
229                         {
230                             token.id = COMMENT;
231                         }
232                         break;
233                     }
234                 }
235                 startState = 0;
236                 ++tokenIndex;
237             }
238             return tokenLine;
239         }
240         protected override nothrow SourceCodeTokenKind GetTokenKind(const Token& token) const
241         {
242             switch (token.id)
243             {
244                 case ASSERT: case ELIF: case ENDIF: case BOOL: case TRUE: case FALSE:
245                 case SBYTE: case BYTE: case SHORT: case USHORT: case INT: case UINT:
246                 case LONG: case ULONG: case FLOAT: case DOUBLE: case CHAR: case WCHAR:
247                 case UCHAR: case VOID: case ENUM: case CAST: case INTERFACE: case NAMESPACE:
248                 case USING: case STATIC: case EXTERN: case AS: case IS: case EXPLICIT:
249                 case DELEGATE: case INLINE: case CDECL: case NOTHROW: case PUBLIC:
250                 case PROTECTED: case PRIVATE: case INTERNAL: case VIRTUAL: case ABSTRACT:
251                 case OVERRIDE: case SUPPRESS: case WINAPI: case OPERATOR: case CLASS:
252                 case RETURN: case IF: case ELSE: case SWITCH: case CASE: case DEFAULT:
253                 case WHILE: case DO: case FOR: case BREAK: case CONTINUE: case GOTO:
254                 case TYPEDEF: case TYPENAME: case TYPEID: case CONST: case CONSTEXPR:
255                 case NULLLIT: case THIS: case BASE: case CONSTRUCT: case DESTROY:
256                 case NEW: case DELETE: case SIZEOF: case TRY: case CATCH: case THROW:
257                 case CONCEPT: case WHERE: case AXIOM: case AND: case OR:
258                 {
259                     return SourceCodeTokenKind.keyword;
260                 }
261                 case ID:
262                 {
263                     return SourceCodeTokenKind.identifier;
264                 }
265                 case EXCLAMATION: case UNIT_TEST: case EQUIVALENCE: case IMPLICATION:
266                 case DISJUNCTION: case AMPAMP: case BITOR: case BITXOR: case AMP:
267                 case EQ: case NEQ: case LEQ: case GEQ: case SHIFTLEFT: case SHIFTRIGHT:
268                 case PLUS: case MINUS: case STAR: case DIV: case REM: case PLUSPLUS:
269                 case MINUSMINUS: case CPL: case DOT: case ARROW: case LBRACKET:
270                 case RBRACKET: case LPAREN: case RPAREN: case LANGLE: case RANGLE:
271                 case COMMA: case ASSIGN: case COLON:
272                 case SEMICOLON: case HASH:
273                 {
274                     return SourceCodeTokenKind.plain;
275                 }
276                 case LBRACE:
277                 {
278                     return SourceCodeTokenKind.beginBlock;
279                 }
280                 case RBRACE:
281                 {
282                     return SourceCodeTokenKind.endBlock;
283                 }
284                 case WS:
285                 {
286                     return SourceCodeTokenKind.space;
287                 }
288                 case FLOATINGLIT: case INTLIT:
289                 {
290                     return SourceCodeTokenKind.number;
291                 }
292                 case CHARLIT:
293                 {
294                     return SourceCodeTokenKind.character;
295                 }
296                 case STRINGLIT:
297                 {
298                     return SourceCodeTokenKind.string;
299                 }
300                 case COMMENT:
301                 {
302                     return SourceCodeTokenKind.comment;
303                 }
304             }
305             return SourceCodeTokenKind.plain;
306         }
