 27     public class delegate void ViewErrorEventHandler(ViewErrorEventArgs& args);

 29     public class ErrorView : Control
 30     {
 31         public ErrorView() : base("Cm.Views.ErrorView", DefaultWindowClassStyle(), DefaultChildWindowStyle(), DefaultExtendedWindowStyle(), Color.White(), 
 32             "errorView", Point(), Size(), Dock.fill, Anchors.none), container(this), treeView(null), child(null)
 33         {
 34             UniquePtr<TreeView> treeViewPtr(new TreeView(Point(), Size(), Dock.none, Anchors.none));
 35             treeView = treeViewPtr.Get();
 36             treeView->SetNormalNodeFont(Font(FontFamily("Consolas"), 9.0f));
 37             treeView->NodeDoubleClickEvent().AddHandler(TreeViewNodeDoubleClick);
 38             UniquePtr<Control> paddedTreeView(new PaddedControl(treeViewPtr.Release(), Padding(4, 4, 4, 4), Point(0, 0), Size(0, 0), Dock.none, Anchors.none));
 39             UniquePtr<Control> scrollableTreeView(new ScrollableControl(paddedTreeView.Release(), Point(0, 0), Size(0, 0), Dock.fill, Anchors.none));
 40             child = scrollableTreeView.Get();
 41             container.AddChild(scrollableTreeView.Release());
 42             Clear();
 43         }
 44         public void Clear()
 45         {
 46             treeView->SetRoot(new TreeViewNode("Errors"));
 47             child->Invalidate();
 48         }
 49         public void SetErrors(List<CompileError>&& errors_)
 50         {
 51             errors = errors_;
 52             Clear();
 53             for (const CompileError& error : errors)
 54             {
 55                 List<ustring> lines = GetErrorLines(ToUtf32(error.message));
 56                 string firstLine;
 57                 if (!lines.IsEmpty())
 58                 {
 59                     firstLine = ToUtf8(lines.Front());
 60                 }
 61                 TreeViewNode* node = new TreeViewNode(firstLine);
 62                 if (!error.project.IsEmpty())
 63                 {
 64                     TreeViewNode* project = new TreeViewNode("Module: " + error.project);
 65                     node->AddChild(project);
 66                 }
 67                 TreeViewNode* file = new TreeViewNode("File: " + error.file);
 68                 node->AddChild(file);
 69                 TreeViewNode* line = new TreeViewNode("Line: " + error.line);
 70                 node->AddChild(line);
 71                 TreeViewNode* code = new TreeViewNode("Code:");
 72                 for (long i = 1; i < lines.Count(); ++i;)
 73                 {
 74                     TreeViewNode* codeNode = new TreeViewNode(ToUtf8(lines[i]));
 75                     code->AddChild(codeNode);
 76                 }
 77                 node->AddChild(code);
 78                 node->SetData(&error);
 79                 treeView->Root()->AddChild(node);
 80             }
 81             Invalidate();
 82         }
 83         private void TreeViewNodeDoubleClick(TreeViewNodeMouseClickEventArgs& args)
 84         {
 85             CompileError* error = cast<CompileError*>(args.node->Data());
 86             OnViewError(error);
 87         }
 88         protected virtual void OnViewError(CompileError* error)
 89         {
 90             ViewErrorEventArgs args(error);
 91             viewErrorEvent.Fire(args);
 92         }
 93         protected override void OnLocationChanged()
 94         {
 95             base->OnLocationChanged();
 96             SetChildPos();
 97         }
 98         protected override void OnSizeChanged(uint windowState)
 99         {
100             base->OnSizeChanged(windowState);
101             SetChildPos();
102         }
103         private void SetChildPos()
104         {
105             Point loc;
106             Size size = GetSize();
107             Rect childRect(loc, size);
108             child->SetLocation(childRect.location);
109             child->SetSize(childRect.size);
110         }
111         protected override void OnPaint(PaintEventArgs& args)
112         {
113             treeView->Root()->ExpandAll();
114             treeView->Invalidate();
115             child->Invalidate();
116         }
117         public nothrow Event<ViewErrorEventHandler, ViewErrorEventArgs>& ViewErrorEvent()
118         {
119             return viewErrorEvent;
120         }

125         private Event<ViewErrorEventHandler, ViewErrorEventArgs> viewErrorEvent;
126     }
