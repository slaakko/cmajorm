 56     public class LocalQuery : Query
 57     {
 58         public nothrow LocalQuery(TreeView* treeView_, TreeViewNode* root_) : treeView(treeView_), root(root_), localCount(-1)
 59         {
 60         }
 61         public override void Init(Cm.Service.Debug.DebugService* debugService)
 62         {
 63             if (localCount == -1)
 64             {
 65                 localCount = debugService->LocalVariableCount();
 66                 root->RemoveChildren();
 67                 for (int i = 0; i < localCount; ++i;)
 68                 {
 69                     string nodeText = System.ToString(i);
 70                     TreeViewNode* node = new TreeViewNode(nodeText);
 71                     node->SetIndex(i);
 72                     TreeViewNode* dummyNode = new TreeViewNode();
 73                     dummyNode->SetIndex(-1);
 74                     node->AddChild(dummyNode);
 75                     root->AddChild(node);
 76                     childResultData.Clear();
 77                 }
 78                 root->Expand();
 79                 treeView->Invalidate();
 80             }
 81         }
 82         public override void Execute(Cm.Service.Debug.DebugService* debugService)
 83         {
 84             List<TreeViewNode*> visibleNodes = treeView->GetVisibleNodes(1);
 85             if (visibleNodes.IsEmpty()) return;
 86             TreeViewNode* first = visibleNodes.Front();
 87             TreeViewNode* last = visibleNodes.Back();
 88             while (first->Data() != null && first != last)
 89             {
 90                 first = cast<TreeViewNode*>(first->NextSibling());
 91             }
 92             while (last->Data() != null && last != first)
 93             {
 94                 last = cast<TreeViewNode*>(last->PrevSibling());
 95             }
 96             int start = first->Index();
 97             int count = last->Index() - start + 1;
 98             if (count == 1 && first->Data() != null)
 99             {
100                 return;
101             }
102             Pair<List<Cm.Service.Debug.ChildResult>, bool> localVariablesSuccess = debugService->LocalVariables(start, count);
103             if (!localVariablesSuccess.second)
104             {
105                 return;
106             }
107             List<Cm.Service.Debug.ChildResult>& localVariables = localVariablesSuccess.first;
108             childResultData.Resize(start + count);
109             for (int i = start; i < start + count; ++i;)
110             {
111                 childResultData[i].Reset(new ChildResultNodeData(localVariables[i - start]));
112             }
113             int n = cast<int>(visibleNodes.Count());
114             for (int i = 0; i < n; ++i;)
115             {
116                 TreeViewNode* node = visibleNodes[i];
117                 ChildResultNodeData* nodeData = childResultData[node->Index()].Get();
118                 node->SetData(nodeData);
119             }
120             for (int i = 0; i < n; ++i;)
121             {
122                 TreeViewNode* node = visibleNodes[i];
123                 ChildResultNodeData* data = cast<ChildResultNodeData*>(node->Data());
124                 string text = data->Result().name;
125                 if (!data->Result().value.IsEmpty())
126                 {
127                     text.Append(" = ").Append(data->Result().value);
128                 }
129                 node->SetText(text);
130                 int count = ParseInt(data->Result().count);
131                 if (count == 0)
132                 {
133                     node->RemoveChildren();
134                 }
135             }
136             treeView->Invalidate();
137         }
138         private int localCount;
139         private TreeView* treeView;
140         private TreeViewNode* root;
141         private List<UniquePtr<ChildResultNodeData>> childResultData;
142     }

214         {
215             TreeViewNode* treeViewNode = args.node;
216             if (treeViewNode->Data() != null)
217             {
218                 NodeData* nodeData = cast<NodeData*>(treeViewNode->Data());
219                 if (nodeData is ChildResultNodeData*)
220                 {
221                     ChildResultNodeData* childNodeData = cast<ChildResultNodeData*>(nodeData);
222                     treeViewNode->RemoveChildren();
223                     int count = ParseInt(childNodeData->Result().count);
224                     for (int i = 0; i < count; ++i;)
225                     {
226                         string nodeText = ToString(i);
227                         TreeViewNode* childNode = new TreeViewNode(nodeText);
228                         childNode->SetIndex(i);
229                         treeViewNode->AddChild(childNode);
230                     }
231                 }
232             }

235         protected override void OnPaint(PaintEventArgs& args)
236         {
237             try
238             {
239                 UpdateLocalsEventArgs updateArgs;
240                 updateArgs.debugging = false;
241                 updateArgs.targetStopped = false;
242                 OnUpdateLocals(updateArgs);
243                 if (updateArgs.debugging && updateArgs.targetStopped)
244                 {
245                     debugService = updateArgs.debugService;
246                     TreeViewNode* root = treeView->Root();
247                     Size size = root->GetSize();
248                     if (size.h == 0)
249                     {
250                         root->MeasureSize(args.graphics, treeView);
251                         size = root->GetSize();
252                     }
253                     if (debugService != null)
254                     {
255                         localQuery->Init(debugService);
256                         localQuery->Execute(debugService);
257                     }
258                     List<TreeViewNode*> visibleNodes = treeView->GetVisibleNodes();
259                     for (TreeViewNode* node : visibleNodes)
260                     {
261                         if (node->Data() == null)
262                         {
263                             TreeViewNode* parent = node->Parent();
264                             if (parent != null)
265                             {
266                                 if (parent->Data() != null)
267                                 {
268                                     NodeData* nodeData = cast<NodeData*>(parent->Data());
269                                     if (nodeData is ChildResultNodeData*)
270                                     {
271                                         ChildResultNodeData* resultNodeData = cast<ChildResultNodeData*>(nodeData);
272                                         List<TreeViewNode*> visibleChildren;
273                                         parent->GetVisibleNodes(visibleChildren, treeView, parent->Level() + 1);
274                                         FetchChildData(visibleChildren, resultNodeData->Result().expr);
275                                         treeView->Invalidate();
276                                     }
277                                 }
278                             }
279                         }
280                     }
281                 }
282             }
283             catch (const Exception& ex)
284             {
285                 MessageBox.Show(ex.Message());
286             }

288         private void FetchChildData(const List<TreeViewNode*>& childNodes, const string& expression)
289         {
290             if (childNodes.IsEmpty()) return;
291             TreeViewNode* first = childNodes.Front();
292             TreeViewNode* last = childNodes.Back();
293             int firstIndex = first->Index();
294             while (first != last && first->Data() != null)
295             {
296                 first = cast<TreeViewNode*>(first->NextSibling());
297                 firstIndex = first->Index();
298             }
299             int lastIndex = last->Index();
300             while (last != first && last->Data() != null)
301             {
302                 last = cast<TreeViewNode*>(last->PrevSibling());
303                 lastIndex = last->Index();
304             }
305             if (firstIndex == lastIndex && first->Data() != null)
306             {
307                 return;
308             }
309             int count = lastIndex - firstIndex + 1;
310             Pair<List<Cm.Service.Debug.ChildResult>, bool> resultsSuccess = debugService->Children(expression, firstIndex, count);
311             if (!resultsSuccess.second)
312             {
313                 return;
314             }
315             const List<Cm.Service.Debug.ChildResult>& childResults = resultsSuccess.first;
316             long start = childResultData.Count();
317             long n = childResults.Count();
318             for (long i = 0; i < n; ++i;)
319             {
320                 childResultData.Add(UniquePtr<ChildResultNodeData>(new ChildResultNodeData(childResults[i])));
321             }
322             for (TreeViewNode* node : childNodes)
323             {
324                 if (node->Data() == null)
325                 {
326                     int index = node->Index();
327                     int i = index - firstIndex;
328                     if (i >= 0)
329                     {
330                         long dataIndex = start + i;
331                         ChildResultNodeData* nodeData = childResultData[dataIndex].Get();
332                         node->SetData(nodeData);
333                         string nodeText = nodeData->Result().name;
334                         if (!nodeData->Result().value.IsEmpty())
335                         {
336                             nodeText.Append(" = ").Append(nodeData->Result().value);
337                         }
338                         node->SetText(nodeText);
339                         int cnt = ParseInt(nodeData->Result().count);
340                         if (cnt > 0)
341                         {
342                             TreeViewNode* dummyNode = new TreeViewNode();
343                             dummyNode->SetIndex(-1);
344                             node->AddChild(dummyNode);
345                         }
346                     }
347                 }
348             }
