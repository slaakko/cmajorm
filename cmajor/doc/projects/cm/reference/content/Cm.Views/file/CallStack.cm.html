 49         }

 94         protected override void OnPaint(PaintEventArgs& args)
 95         {
 96             UpdateCallStackEventArgs updateArgs;
 97             updateArgs.debugging = false;
 98             updateArgs.targetStopped = false;
 99             OnUpdateCallStack(updateArgs);
100             if (updateArgs.debugging && updateArgs.targetStopped)
101             {
102                 TreeViewNode* root = treeView->Root();
103                 Size size = root->GetSize();
104                 if (size.h == 0)
105                 {
106                     root->MeasureSize(args.graphics, treeView);
107                     size = root->GetSize();
108                 }
109                 PopulateTreeView(root);
110                 List<TreeViewNode*> visibleNodes = treeView->GetVisibleNodes(1);
111                 UpdateNodeData(visibleNodes, root);
112                 for (TreeViewNode* node : visibleNodes)
113                 {
114                     void* nodeData = node->Data();
115                     if (nodeData != null)
116                     {
117                         Frame* frame = cast<Frame*>(node->Data());
118                         string func = frame->location.func;
119                         if (func.IsEmpty())
120                         {
121                             func = "<external>";
122                         }
123                         string nodeText = ToString(node->Index()) + " - " + func;
124                         node->SetText(nodeText);
125                         node->RemoveChildren();
126                         if (!frame->location.func.IsEmpty())
127                         {
128                             TreeViewNode* fileNode = new TreeViewNode(frame->location.file);
129                             node->AddChild(fileNode);
130                             string lineText = "Line " + frame->location.line;
131                             TreeViewNode* lineNode = new TreeViewNode(lineText);
132                             node->AddChild(lineNode);
133                         }
134                     }
135                 }
136                 treeView->Invalidate();
137             }
138         }
139         private void PopulateTreeView(TreeViewNode* root)
140         {
141             if (depth == -1)
142             {
143                 depth = GetCallStackDepth();
144                 for (int i = 0; i < depth; ++i;)
145                 {
146                     string nodeText = ToString(i);
147                     TreeViewNode* node = new TreeViewNode(nodeText);
148                     node->SetIndex(i);
149                     root->AddChild(node);
150                 }
151                 root->Expand();
152                 treeView->Invalidate();
153             }
154         }
155         private void UpdateNodeData(List<TreeViewNode*>& nodes, TreeViewNode* root)
156         {
157             if (nodes.IsEmpty()) return;
158             TreeViewNode* first = nodes.Front();
159             TreeViewNode* last = nodes.Back();
160             int firstIndex = 0;
161             while (first->Data() != null && first != last)
162             {
163                 first = cast<TreeViewNode*>(first->NextSibling());
164                 ++firstIndex;
165             }
166             int lastIndex = cast<int>(nodes.Count() - 1);
167             while (last->Data() != null && last != first)
168             {
169                 last = cast<TreeViewNode*>(last->PrevSibling());
170                 --lastIndex;
171             }
172             int lowFrame = first->Index();
173             int highFrame = last->Index();
174             if (lowFrame == highFrame && first->Data() != null)
175             {
176                 return;
177             }
178             Pair<List<Cm.Service.Debug.Location>, bool> locationsSuccess = debugService->Frames(lowFrame, highFrame);
179             if (!locationsSuccess.second)
180             {
181                 return;
182             }
183             frames.Resize(highFrame + 1);
184             List<Cm.Service.Debug.Location>& locations = locationsSuccess.first;
185             for (int i = lowFrame; i <= highFrame; ++i;)
186             {
187                 frames[i].Reset(new Frame(Rvalue(locations[i - lowFrame])));
188             }
189             for (int i = firstIndex; i <= lastIndex; ++i;)
190             {
191                 TreeViewNode* node = nodes[i];
192                 node->SetData(frames[node->Index()].Get());
193             }
194         }
