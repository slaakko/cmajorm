 12 void PrintHelp()
 13 {
 14     Console.Out() << "Usage: cmcmparser [options] { file.cm | file.cmp | file.cms }" << endl();
 15     Console.Out() << "Parse Cmajor source file file.cm, project file.cmp or solution file.cms" << endl();
 16     Console.Out() << "Options:" << endl();
 17     Console.Out() << "--help | -h" << endl();
 18     Console.Out() << "  Print help and exit." << endl();
 19     Console.Out() << "--verbose | -v" << endl();
 20     Console.Out() << "  Be verbose." << endl();
 21 }

 23 void ParseSourceFile(const string& fileName, int index, bool verbose)
 24 {
 25     if (verbose)
 26     {
 27         Console.Out() << ">>> " << fileName << endl();
 28     }
 29     CmajorLexer lexer(ToUtf32(File.ReadAllText(fileName)), fileName, index);
 30     ParsingContext ctx;
 31     UniquePtr<CompileUnitNode> compileUnit = CompileUnitParser.Parse(lexer, &ctx);
 32 }
 33 
 34 void ParseProject(const string& projectFileName, int index, bool verbose)
 35 {
 36     if (verbose)
 37     {
 38         Console.Out() << ">> " << projectFileName << endl();
 39     }
 40     ContainerFileLexer lexer(ToUtf32(File.ReadAllText(projectFileName)), projectFileName, index);
 41     UniquePtr<Project> project = ProjectFileParser.Parse(lexer, "debug", BackEnd.llvm);
 42     project->ResolveDeclarations();
 43     long n = project->SourceFilePaths().Count();
 44     for (long i = 0; i < n; ++i;)
 45     {
 46         ParseSourceFile(project->SourceFilePaths()[i], cast<int>(i), verbose);
 47     }
 48 }
 49 
 50 void ParseSolution(const string& solutionFileName, int index, bool verbose)
 51 {
 52     if (verbose)
 53     {
 54         Console.Out() << "> " << solutionFileName << endl();
 55     }
 56     ContainerFileLexer lexer(ToUtf32(File.ReadAllText(solutionFileName)), solutionFileName, index);
 57     UniquePtr<Solution> solution = SolutionFileParser.Parse(lexer);
 58     solution->ResolveDeclarations();
 59     long n = solution->ProjectFilePaths().Count();
 60     for (long i = 0; i < n; ++i;)
 61     {
 62         ParseProject(solution->ProjectFilePaths()[i], cast<int>(i), verbose);
 63     }
 64 }

 66 int main(int argc, const char** argv)
 67 {
 68     try
 69     {
 70         List<string> files;
 71         bool verbose = false;
 72         for (int i = 1; i < argc; ++i;)
 73         {
 74             string arg = argv[i];
 75             if (arg.StartsWith("--"))
 76             {
 77                 if (arg == "--help")
 78                 {
 79                     PrintHelp();
 80                     return 1;
 81                 }
 82                 else if (arg == "--verbose")
 83                 {
 84                     verbose = true;
 85                 }
 86                 else
 87                 {
 88                     throw Exception("unknown option '" + arg + "'");
 89                 }
 90             }
 91             else if (arg.StartsWith("-"))
 92             {
 93                 string options = arg.Substring(1);
 94                 string unknownOption;
 95                 for (char o : options)
 96                 {
 97                     switch (o)
 98                     {
 99                         case 'h': PrintHelp(); return 1;
100                         case 'v': verbose = true; break;
101                         default:
102                         {
103                             unknownOption = string(o);
104                             break;
105                         }
106                     }
107                     if (!unknownOption.IsEmpty())
108                     {
109                         throw Exception("unknown option '-" + unknownOption + "'");
110                     }
111                 }
112             }
113             else
114             {
115                 files.Add(arg);
116             }
117         }
118         long n = files.Count();
119         if (n == 0)
120         {
121             throw Exception("no files given");
122         }
123         for (long i = 0; i < n; ++i;)
124         {
125             string fileName = GetFullPath(files[i]);
126             string extension = Path.GetExtension(fileName);
127             if (extension == ".cm")
128             {
129                 ParseSourceFile(fileName, cast<int>(i), verbose);
130             }
131             else if (extension == ".cmp")
132             {
133                 ParseProject(fileName, cast<int>(i), verbose);
134             }
135             else if (extension == ".cms")
136             {
137                 ParseSolution(fileName, cast<int>(i), verbose);
138             }
139             else
140             {
141                 throw Exception("unknown extension '" + extension + "'");
142             }
143         }
144     }
145     catch (const Exception& ex)
146     {
147         Console.Error() << ex.Message() << endl();
148         return 1;
149     }
150     return 0;
151 }
