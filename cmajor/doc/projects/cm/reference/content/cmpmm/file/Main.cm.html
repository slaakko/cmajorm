  6 using System;
  7 using System.Windows;
  8 using Cm.Service;
  9 using cmpmm;
 10 
 11 namespace cmpmm
 12 {
 13     class MainWindow : Window
 14     {
 15         public MainWindow() : 
 16             base("cmpmm.MainWindow", WindowStyle.WS_OVERLAPPEDWINDOW, DefaultExtendedWindowStyle(), "Cmajor Port Map Monitor", 
 17             Point.Default(), Size.Default(), Dock.none, Anchors.none, WindowState.normal)
 18         {
 19             UniquePtr<MenuBar> menuBar(new MenuBar());
 20             UniquePtr<MenuItem> fileMenuItemPtr(new MenuItem("&File"));
 21             UniquePtr<MenuItem> exitMenuItemPtr(new MenuItem("E&xit"));
 22             exitMenuItemPtr->SetShortcut(cast<Keys>(Keys.altModifier | Keys.f4));
 23             exitMenuItemPtr->ClickEvent().AddHandler(ExitClick);
 24             fileMenuItemPtr->AddMenuItem(exitMenuItemPtr.Release());
 25             menuBar->AddMenuItem(fileMenuItemPtr.Release());
 26             UniquePtr<MenuItem> serverMenuItemPtr(new MenuItem("&Server"));
 27             UniquePtr<MenuItem> startMenuItemPtr(new MenuItem("&Start"));
 28             startServerMenuItem = startMenuItemPtr.Get();
 29             startServerMenuItem->ClickEvent().AddHandler(StartServerClick);
 30             startServerMenuItem->Disable();
 31             serverMenuItemPtr->AddMenuItem(startMenuItemPtr.Release());
 32             UniquePtr<MenuItem> stopMenuItemPtr(new MenuItem("&Stop"));
 33             stopServerMenuItem = stopMenuItemPtr.Get();
 34             stopServerMenuItem->ClickEvent().AddHandler(StopServerClick);
 35             stopServerMenuItem->Disable();
 36             serverMenuItemPtr->AddMenuItem(stopMenuItemPtr.Release());
 37             menuBar->AddMenuItem(serverMenuItemPtr.Release());
 38             UniquePtr<MenuItem> windowMenuItemPtr(new MenuItem("&Window"));
 39             UniquePtr<MenuItem> updateMenuItemPtr(new MenuItem("&Update"));
 40             updateMenuItemPtr->SetShortcut(Keys.f5);
 41             updateMenuItemPtr->ClickEvent().AddHandler(UpdateClick);
 42             windowMenuItemPtr->AddMenuItem(updateMenuItemPtr.Release());
 43             menuBar->AddMenuItem(windowMenuItemPtr.Release());
 44             AddChild(menuBar.Release());
 45 
 46             UniquePtr<Cm.Views.PortMapView> portMapViewPtr(new Cm.Views.PortMapView(startServerMenuItem, stopServerMenuItem));
 47             portMapView = portMapViewPtr.Get();
 48             AddChild(portMapViewPtr.Release());
 49 
 50             UpdateClick();
 51         }
 52         private void ExitClick()
 53         {
 54             Close();
 55         }
 56         private void UpdateClick()
 57         {
 58             portMapView->Update();
 59             Invalidate();
 60         }
 61         private void StartServerClick()
 62         {
 63             try
 64             {
 65                 Cm.Service.PortMap.StartPortMapServer();
 66                 UpdateClick();
 67             }
 68             catch (const Exception& ex)
 69             {
 70                 MessageBox.Show(ex.Message());
 71             }
 72         }
 73         private void StopServerClick()
 74         {
 75             try
 76             {
 77                 Cm.Service.PortMap.StopPortMapServer();
 78                 portMapView->SetStopped();
 79             }
 80             catch (const Exception& ex)
 81             {
 82                 MessageBox.Show(ex.Message());
 83             }
 84         }
 85         private Cm.Views.PortMapView* portMapView;
 86         private MenuItem* startServerMenuItem;
 87         private MenuItem* stopServerMenuItem;
 88     }
 89 }
 90 
 91 int main(int argc, const char** argv)
 92 {
 93     try
 94     {
 95         MainWindow mainWindow;
 96         mainWindow.SetSmallIcon(Application.GetResourceManager().GetIcon("cmpmm.icon"));
 97         int exitCode = Application.Run(mainWindow);
 98         return exitCode;
 99     }
100     catch (const Exception& ex)
101     {
102         Application.SetLogView(null);
103         MessageBox.Show(ex.Message(), "Error", null, cast<MessageBoxType>(MessageBoxType.MB_OK | MessageBoxType.MB_ICONSTOP));
104         LogException(ex);
105         return 1;
106     }
107     return 0;
108 }
