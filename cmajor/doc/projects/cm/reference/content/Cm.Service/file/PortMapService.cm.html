 93     int GetPortMapServicePortNumber()
 94     {
 95         try
 96         {
 97             string portMapConfigFileName = PortMapConfigFileName();
 98             try
 99             {
100                 UniquePtr<Document> portMapConfigDoc = ReadDocument(portMapConfigFileName);
101                 UniquePtr<XPathObject> result = Evaluate(u"/cmpm", portMapConfigDoc.Get());
102                 if (result.Get() is XPathNodeSet*)
103                 {
104                     XPathNodeSet* nodeSet = cast<XPathNodeSet*>(result.Get());
105                     if (nodeSet->Length() == 1)
106                     {
107                         Node* node = (*nodeSet)[0];
108                         if (node is Element*)
109                         {
110                             Element* element = cast<Element*>(node);
111                             ustring portMapServicePortAttribute = element->GetAttribute(u"portMapServicePort");
112                             if (!portMapServicePortAttribute.IsEmpty())
113                             {
114                                 return ParseInt(ToUtf8(portMapServicePortAttribute));
115                             }
116                             else
117                             {
118                                 throw Exception("'portMapServicePort' attribute not found from 'cmpm' element");
119                             }
120                         }
121                         else
122                         {
123                             throw Exception("cmpm: element node expected");
124                         }
125                     }
126                     else
127                     {
128                         throw Exception("single 'cmpm' node expected");
129                     }
130                 }
131                 else
132                 {
133                     throw Exception("cmpm: node set expected");
134                 }
135             }
136             catch (const Exception& ex)
137             {
138                 throw Exception("Cm.Service.PortMap.PortMapService: GetPortMapServicePortNumber: file=" + portMapConfigFileName + ": " + ex.Message());
139             }
140         }
141         catch (const Exception& ex)
142         {
143             LogWriter logWriter(PortMapLogFilePath());
144             logWriter << ex.ToString() << endl();
145         }
146         return -1;
147     }

149     string GetMessage(JsonValue* jsonValue)
150     {
151         if (jsonValue is JsonObject*)
152         {
153             JsonObject* jsonObject = cast<JsonObject*>(jsonValue);
154             JsonValue* message = jsonObject->GetField("message");
155             if (message != null && message is JsonString*)
156             {
157                 JsonString* messageStr = cast<JsonString*>(message);
158                 return ToUtf8(messageStr->Value());
159             }
160         }
161         return string();
162     }

164     public void StartPortMapServer()
165     {
166         try
167         {
168             Process* portMapServer = new Process("cmpms", cast<Process.Redirections>(Process.Redirections.processStdIn | Process.Redirections.processStdOut | Process.Redirections.processStdErr));
169             string line = portMapServer->ReadLine(Process.StdHandle.stdOut);
170             if (line == "port-map-server-ready")
171             {
172                 Sleep(Duration.FromSeconds(1));
173             }
174             else if (line == "port-map-server-error")
175             {
176                 line = portMapServer->ReadLine(Process.StdHandle.stdOut);
177                 throw Exception(line);
178             }
179         }
180         catch (const Exception& ex)
181         {
182             throw Exception("Cm.Service.PortMap.PortMapService: StartPortMapServer: " + ex.Message());
183         }
184     }

186     public void StopPortMapServer()
187     {
188         int port = GetPortMapServicePortNumber();
189         if (port == -1)
190         {
191             throw Exception("could not resolve port map service port number");
192         }
193         TcpSocket portMapServerConnection("localhost", ToString(port));
194         StopPortMapServerRequest request;
195         UniquePtr<JsonValue> requestValue = request.ToJson();
196         string requestStr = requestValue->ToString();
197         Write(portMapServerConnection, requestStr);
198         string replyStr = ReadStr(portMapServerConnection);
199         UniquePtr<JsonValue> replyValue = ParseJson(replyStr);
200         string message = GetMessage(replyValue.Get());
201     }

203     public int GetFreePortNumber(const string& programName, int pid, int& portMapServicePortNumber)
204     {
205         portMapServicePortNumber = -1;
206         try
207         {
208             bool portMapServerStarted = false;
209             int port = GetPortMapServicePortNumber();
210             if (port == -1)
211             {
212                 StartPortMapServer();
213                 portMapServerStarted = true;
214             }
215             port = GetPortMapServicePortNumber();
216             if (port == -1)
217             {
218                 throw Exception("could not resolve port map service port number");
219             }
220             for (int connectionAttempt = 1; connectionAttempt <= 2; ++connectionAttempt;)
221             {
222                 try
223                 {
224                     TcpSocket portMapServerConnection("localhost", ToString(port));
225                     GetFreePortNumberRequest request(programName, pid, GetCurrentDateTime().ToString());
226                     UniquePtr<JsonValue> requestValue = request.ToJson();
227                     string requestStr = requestValue->ToString();
228                     Write(portMapServerConnection, requestStr);
229                     string replyStr = ReadStr(portMapServerConnection);
230                     UniquePtr<JsonValue> replyValue = ParseJson(replyStr);
231                     string message = GetMessage(replyValue.Get());
232                     if (message == "GetFreePortNumberReply")
233                     {
234                         GetFreePortNumberReply reply(replyValue.Get());
235                         string portNumberStr = reply.portNumber;
236                         if (portNumberStr.IsEmpty())
237                         {
238                             throw Exception("portNumber field in 'GetFreePortNumberReply' is empty");
239                         }
240                         int freePortNumber = ParseInt(portNumberStr);
241                         portMapServicePortNumber = port;
242                         return freePortNumber;
243                     }
244                     else
245                     {
246                         throw Exception("'GetFreePortNumberReply' expected, message=" + message);
247                     }
248                 }
249                 catch (const Exception& ex)
250                 {
251                     if (connectionAttempt > 1)
252                     {
253                         throw ex;
254                     }
255                 }
256                 if (connectionAttempt == 1 && !portMapServerStarted)
257                 {
258                     StartPortMapServer();
259                 }
260                 else
261                 {
262                     break;
263                 }
264             }
265         }
266         catch (const Exception& ex)
267         {
268             LogWriter logWriter(PortMapLogFilePath());
269             logWriter << ex.ToString() << endl();
270         }
271         return -1;
272     }
