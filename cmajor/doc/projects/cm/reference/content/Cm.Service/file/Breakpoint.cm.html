  6 using System;

  9 namespace Cm.Service.Debug

 11     public class BreakpointCollection
 12     {
 13         public BreakpointList& GetBreakpointList(const string& filePath)
 14         {
 15             BreakpointList& breakpointList = breakpointListMap[filePath];
 16             if (breakpointList.FilePath().IsEmpty())
 17             {
 18                 breakpointList.SetFilePath(filePath);
 19             }
 20             return breakpointList;
 21         }
 22         public nothrow const Map<string, BreakpointList>& BreakpointListMap() const
 23         {
 24             return breakpointListMap;
 25         }
 26         public void Clear()
 27         {
 28             breakpointListMap.Clear();
 29         }
 30         private Map<string, BreakpointList> breakpointListMap;
 31     }

 33     public class ByLine : Rel<Breakpoint*>
 34     {
 35         public inline nothrow bool operator()(Breakpoint* left, Breakpoint* right) const
 36         {
 37             return left->line < right->line;
 38         }
 39     }
 40 
 41     public class BreakpointList
 42     {
 43         public nothrow BreakpointList()
 44         {
 45         }
 46         public inline nothrow bool IsEmpty() const
 47         {
 48             return breakpoints.IsEmpty();
 49         }
 50         public nothrow Breakpoint* GetBreakpoint(int line) const
 51         {
 52             if (IsEmpty()) return null;
 53             Breakpoint bp(line);
 54             List<Breakpoint*>.ConstIterator it = LowerBound(breakpoints.CBegin(), breakpoints.CEnd(), &bp, ByLine());
 55             if (it != breakpoints.CEnd())
 56             {
 57                 Breakpoint* foundBp = *it;
 58                 if (foundBp->line == line)
 59                 {
 60                     return foundBp;
 61                 }
 62             }
 63             return null;
 64         }
 65         public void AddBreakpoint(Breakpoint* breakpoint)
 66         {
 67             if (GetBreakpoint(breakpoint->line) != null)
 68             {
 69                 throw Exception("already has breakpoint on line " + ToString(breakpoint->line));
 70             }
 71             breakpoints.Add(breakpoint);
 72             Sort(breakpoints.Begin(), breakpoints.End(), ByLine());
 73         }
 74         public void RemoveBreakpoint(Breakpoint* breakpoint)
 75         {
 76             breakpoints.Remove(breakpoint);
 77             delete breakpoint;
 78         }
 79         public void Clear()
 80         {
 81             for (Breakpoint* bp : breakpoints)
 82             {
 83                 RemoveBreakpoint(bp);
 84             }
 85         }
 86         public inline nothrow const string& FilePath() const
 87         {
 88             return filePath;
 89         }
 90         public nothrow void SetFilePath(const string& filePath_)
 91         {
 92             filePath = filePath_;
 93         }
 94         public inline nothrow const List<Breakpoint*>& Breakpoints() const
 95         {
 96             return breakpoints;
 97         }
 98         public inline nothrow List<Breakpoint*>& Breakpoints()
 99         {
100             return breakpoints;
101         }
102         private string filePath;
103         private List<Breakpoint*> breakpoints;
104     }
