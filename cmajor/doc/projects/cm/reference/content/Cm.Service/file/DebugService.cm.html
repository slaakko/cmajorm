210         private string MakeDebugServerStartCommand()
211         {
212             string serverStartCommand = "cmdb";
213             if (args.debugServer)
214             {
215                 serverStartCommand = "cmdbd";
216             }
217             serverStartCommand.Append(" --server");
218             port = Cm.Service.PortMap.GetFreePortNumber(args.programName, args.pid, portMapServicePortNumber);
219             bool gotNoPort = false;
220             if (port == -1)
221             {
222                 port = args.port;
223                 LogMessage("DebugService: got no port from port map server, resorting to default port " + ToString(port));
224                 gotNoPort = true;
225             }
226             serverStartCommand.Append(" --port=" + ToString(port));
227             if (!gotNoPort)
228             {
229                 sessionPort = Cm.Service.PortMap.GetFreePortNumber(args.programName, args.pid, portMapServicePortNumber);
230             }
231             if (sessionPort == -1)
232             {
233                 sessionPort = args.sessionPort;
234                 LogMessage("DebugService: got no port from port map server, resorting to default session port " + ToString(sessionPort));
235             }
236             if (portMapServicePortNumber != -1)
237             {
238                 serverStartCommand.Append(" --portMapServicePort=" + ToString(portMapServicePortNumber));
239             }
240             serverStartCommand.Append(" --sessionPort=" + ToString(sessionPort));
241             if (args.log)
242             {
243                 serverStartCommand.Append(" --log");
244             }
245             if (args.wait)
246             {
247                 serverStartCommand.Append(" --wait");
248             }
249             if (args.verbose)
250             {
251                 serverStartCommand.Append(" --verbose");
252             }
253             serverStartCommand.Append(" \"").Append(args.GetExecutablePath()).Append("\"");
254             if (!args.arguments.IsEmpty())
255             {
256                 serverStartCommand.Append(" ").Append(args.arguments);
257             }
258             return serverStartCommand;
259         }

265         public void Start(const BreakpointCollection& breakpointCollection)
266         {
267             targetState.Reset();
268             if (state != State.initialized)
269             {
270                 return;
271             }
272             {
273                 RequestGuard requestGuard(this);
274                 string serverStartCommand = MakeDebugServerStartCommand();
275                 LogMessage("DebugService: Starting Debug Server (port=" + ToString(port) + ", sessionPort=" + ToString(sessionPort) + ")");
276                 debugServer.Reset(new Process(serverStartCommand, 
277                     cast<Process.Redirections>(Process.Redirections.processStdIn | Process.Redirections.processStdOut | Process.Redirections.processStdErr)));
278                 string line = debugServer->ReadLine(Process.StdHandle.stdOut);
279                 if (line == "debug-server-error")
280                 {
281                     line = debugServer->ReadLine(Process.StdHandle.stdOut);
282                     throw Exception("DebugServer failed: " + line);
283                 }
284                 LogMessage(line);
285                 line = debugServer->ReadLine(Process.StdHandle.stdOut);
286                 LogMessage(line);
287                 Sleep(Duration.FromSeconds(serverStartWaitSeconds));
288                 socket.Connect("localhost", ToString(port));
289                 List<Breakpoint*> breakpoints = SendStartRequest(breakpointCollection);
290                 UniquePtr<JsonValue> reply = ReadReply(MessageKind.startReply);
291                 ProcessStartReply(reply.Get(), breakpoints);
292                 LogMessage("DebugService: Debug Server started.");
293             }
294             state = State.debuggingStarted;
295         }
296         private List<Breakpoint*> SendStartRequest(const BreakpointCollection& breakpointCollection)
297         {
298             List<Breakpoint*> breakpoints;
299             StartRequest startRequest;
300             for (const Pair<string, BreakpointList>& p : breakpointCollection.BreakpointListMap())
301             {
302                 const BreakpointList& bpList = p.second;
303                 for (Breakpoint* breakpoint : bpList.Breakpoints())
304                 {
305                     breakpoints.Add(breakpoint);
306                     SourceLoc breakpointLocation;
307                     breakpointLocation.path = bpList.FilePath();
308                     breakpointLocation.line = ToString(breakpoint->line);
309                     startRequest.breakpointLocations.Add(breakpointLocation);
310                 }
311             }
312             UniquePtr<JsonValue> request = startRequest.ToJson();
313             WriteRequest(request.Get());
314             return breakpoints;
315         }
316         private void ProcessStartReply(JsonValue* reply, const List<Breakpoint*>& breakpoints)
317         {
318             StartReply startReply(reply);
319             if (startReply.success)
320             {
321                 int n = cast<int>(breakpoints.Count());
322                 for (int i = 0; i < n; ++i;)
323                 {
324                     Breakpoint* breakpoint = breakpoints[i];
325                     breakpoint->info = startReply.breakpointInfos[i];
326                     if (breakpoint->info.success)
327                     {
328                         breakpoint->disabled = false;
329                     }
330                     else
331                     {
332                         breakpoint->disabled = true;
333                     }
334                 }
335                 if (setDebugLocation != SetDebugLocationMethod())
336                 {
337                     setDebugLocation(startReply.location, true);
338                 }
339             }
340             else
341             {
342                 LogMessage("start request failed: " + startReply.error);
343             }
344         }

780         private void ProcessMessage(JsonValue* message, MessageKind messageKind, const string& messageKindStr)
781         {
782             switch (messageKind)
783             {
784                 case MessageKind.targetRunningRequest:
785                 {
786                     TargetRunningRequest targetRunningRequest(message);
787                     ProcessTargetRunningRequest(targetRunningRequest);
788                     break;
789                 }
790                 case MessageKind.targetInputRequest:
791                 {
792                     TargetInputRequest targetInputRequest(message);
793                     ProcessTargetInputRequest(targetInputRequest);
794                     break;
795                 }
796                 case MessageKind.targetOutputRequest:
797                 {
798                     TargetOutputRequest targetOutputRequest(message);
799                     ProcessTargetOutputRequest(targetOutputRequest);
800                     break;
801                 }
802                 case MessageKind.logMessageRequest:
803                 {
804                     LogMessageRequest logMessageRequest(message);
805                     ProcessLogMessageRequest(logMessageRequest);
806                     break;
807                 }
808                 case MessageKind.genericErrorReply:
809                 {
810                     GenericErrorReply errorReply(message);
811                     ProcessErrorReply(errorReply);
812                     break;
813                 }
814                 default:
815                 {
816                     LogMessage("unexpected " + messageKindStr + " received");
817                     break;
818                 }
819             }
820         }

866         private MessageKind GetMessageKind(JsonValue* message, string& messageKindStr)
867         {
868             messageKindStr = "unknown message kind";
869             if (message is JsonObject*)
870             {
871                 JsonObject* jsonObject = cast<JsonObject*>(message);
872                 JsonValue* messageKindValue = jsonObject->GetField("messageKind");
873                 if (messageKindValue is JsonString*)
874                 {
875                     JsonString* messageKind = cast<JsonString*>(messageKindValue);
876                     messageKindStr = ToUtf8(messageKind->Value());
877                     Map<string, MessageKind>.ConstIterator it = messageKindMap.CFind(messageKindStr);
878                     if (it != messageKindMap.CEnd())
879                     {
880                         return it->second;
881                     }
882                 }
883             }
884             return MessageKind.none;
885         }

896         private UniquePtr<JsonValue> ReadReply(MessageKind replyMessageKind)
897         {
898             string replyStr = ReadStr(socket);
899             JsonLexer lexer(ToUtf32(replyStr), "", 0);
900             UniquePtr<JsonValue> replyValue = JsonParser.Parse(lexer);
901             string messageKindStr;
902             MessageKind messageKind = GetMessageKind(replyValue.Get(), messageKindStr);
903             while (messageKind != replyMessageKind)
904             {
905                 ProcessMessage(replyValue.Get(), messageKind, messageKindStr);
906                 replyStr = ReadStr(socket);
907                 JsonLexer lexer(ToUtf32(replyStr), "", 0);
908                 replyValue = JsonParser.Parse(lexer);
909                 messageKind = GetMessageKind(replyValue.Get(), messageKindStr);
910             }
911             return replyValue;
912         }
