113         private string MakeServerStartCommand()
114         {
115             string serverStartCommand = "cmbs";
116             if (args.debugServer)
117             {
118                 serverStartCommand = "cmbsd";
119             }
120             port = Cm.Service.PortMap.GetFreePortNumber(args.programName, args.pid, portMapServicePortNumber);
121             bool gotNoPort = false;
122             if (port == -1)
123             {
124                 port = args.port;
125                 LogMessage("BuildService: got no port from port map server, resorting to default port " + ToString(port));
126                 gotNoPort = true;
127             }
128             serverStartCommand.Append(" --port=" + ToString(port));
129             if (!gotNoPort)
130             {
131                 keepAliveServerPort = Cm.Service.PortMap.GetFreePortNumber(args.programName, args.pid, portMapServicePortNumber);
132             }
133             serverStartCommand.Append(" --keepAliveServerPort=" + ToString(keepAliveServerPort));
134             if (keepAliveServerPort == -1)
135             {
136                 keepAliveServerPort = args.keepAliveServerPort;
137                 LogMessage("BuildService: got no port from port map server, resorting to default keep alive server port " + ToString(keepAliveServerPort));
138             }
139             if (portMapServicePortNumber != -1)
140             {
141                 serverStartCommand.Append(" --portMapServicePort=" + ToString(portMapServicePortNumber));
142             }
143             if (args.log)
144             {
145                 serverStartCommand.Append(" --log");
146             }
147             if (args.wait)
148             {
149                 serverStartCommand.Append(" --wait");
150             }
151             return serverStartCommand;
152         }

170         public bool StartServer()
171         {
172             try
173             {
174                 if (ServerRunning())
175                 {
176                     StopServer(true);
177                 }
178                 string serverStartCommand = MakeServerStartCommand();
179                 LogMessage("BuildService: Starting Build Server (port=" + ToString(port) + ", keep alive server port=" + ToString(keepAliveServerPort) + ")...");
180                 buildServer.Reset(
181                     new Process(serverStartCommand, 
182                     cast<Process.Redirections>(Process.Redirections.processStdIn | Process.Redirections.processStdOut | Process.Redirections.processStdErr)));
183                 string line = buildServer->ReadLine(Process.StdHandle.stdOut);
184                 if (line == "build-server-error")
185                 {
186                     line = buildServer->ReadLine(Process.StdHandle.stdOut);
187                     throw Exception("BuildServer failed: " + line);
188                 }
189                 LogMessage(line);
190                 Sleep(Duration.FromSeconds(args.serverStartWaitSeconds));
191                 keepAliveThread = Thread.StartMethod(KeepAliveServer);
192                 LogMessage("BuildService: Build Server started.");
193                 notifyBuildServerStart();
194             }
195             catch (const Exception& ex)
196             {
197                 LogException(ex);
198                 LogMessage("BuildService: " + ex.Message());
199                 return false;
200             }
201             return true;
202         }
203         public void StopServer(bool log)
204         {
205             try
206             {
207                 if (!ServerRunning()) return;
208                 if (keepAliveServerRunning)
209                 {
210                     stopKeepAlives = true;
211                     stopKeepAlivesVar.NotifyOne();
212                     keepAliveThread.Join();
213                 }
214                 if (log)
215                 {
216                     LogMessage("BuildService: Stopping Build Server...");
217                 }
218                 SendReceiveStopRequest();
219                 bool serverStopped = false;
220                 for (int i = 0; i < args.serverStopWaitSeconds; ++i;)
221                 {
222                     if (buildServer->Running())
223                     {
224                         Sleep(Duration.FromSeconds(1));
225                     }
226                     else
227                     {
228                         serverStopped = true;
229                         break;
230                     }
231                 }
232                 if (!serverStopped)
233                 {
234                     if (log)
235                     {
236                         LogMessage("BuildService: Build Server not stopped after " + ToString(args.serverStopWaitSeconds) + " seconds, terminating server process...");
237                     }
238                     buildServer->Terminate();
239                 }
240                 int exitCode = buildServer->ExitCode();
241                 string exitCodeStr;
242                 if (exitCode != 0)
243                 {
244                     exitCodeStr.Append(" Exit code=" + ToString(exitCode));
245                 }
246                 if (log)
247                 {
248                     LogMessage("BuildService: Build Server stopped." + exitCodeStr);
249                 }
250             }
251             catch (const Exception& ex)
252             {
253                 LogException(ex);
254                 if (log)
255                 {
256                     LogMessage("BuildService: StopServer: " + ex.Message());
257                 }
258             }
259         }

303         public void KeepAliveServer()
304         {
305             KeepAliveServerGuard keepAliveServerGuard(keepAliveServerRunning);
306             try
307             {
308                 stopKeepAlives = false;
309                 while (!stopKeepAlives)
310                 {
311                     LockGuard<RecursiveMutex> lock(keepAliveMutex);
312                     if (stopKeepAlivesVar.WaitFor(keepAliveMutex, StopKeepAlives, null, Duration.FromSeconds(defaultKeepAliveIntervalSeconds)))
313                     {
314                         return;
315                     }
316                     else
317                     {
318                         SendReceiveKeepAliveRequest();
319                     }
320                 }
321             }
322             catch (const Exception& ex)
323             {
324                 LogException(ex);
325             }
326         }
327         private void SendReceiveStopRequest()
328         {
329             TcpSocket socket("localhost", ToString(port));
330             StopRequest stopRequest;
331             UniquePtr<JsonValue> requestJsonValue = stopRequest.ToJson();
332             string requestStr = requestJsonValue->ToString();
333             Write(socket, requestStr);
334             string replyStr = ReadStr(socket);
335             UniquePtr<JsonValue> replyJsonValue = ParseJson(replyStr);
336             string messageKind = GetMessageKind(replyJsonValue.Get());
337             if (messageKind == "stopReply")
338             {
339                 StopReply reply(replyJsonValue.Get());
340             }
341             else
342             {
343                 throw Exception("stopReply expected, unknown reply message kind '" + messageKind + "' received");
344             }
345         }
346         private void SendReceiveKeepAliveRequest()
347         {
348             TcpSocket socket("localhost", ToString(keepAliveServerPort));
349             KeepAliveRequest keepAliveRequest;
350             UniquePtr<JsonValue> requestJsonValue = keepAliveRequest.ToJson();
351             string requestStr = requestJsonValue->ToString();
352             Write(socket, requestStr);
353             string replyStr = ReadStr(socket);
354             UniquePtr<JsonValue> replyJsonValue = ParseJson(replyStr);
355             string messageKind = GetMessageKind(replyJsonValue.Get());
356             if (messageKind == "keepAliveReply")
357             {
358                 KeepAliveReply reply(replyJsonValue.Get());
359             }
360             else
361             {
362                 throw Exception("keepAliveReply expected, unknown reply message kind '" + messageKind + "' received");
363             }

365         public List<CompileError> ProcessBuildRequest(const BuildRequest& request)
366         {
367             try
368             {
369                 terminating = false;
370                 if (!ServerRunning())
371                 {
372                     if (!StartServer())
373                     {
374                         return List<CompileError>();
375                     }
376                 }
377                 UniquePtr<JsonValue> requestJsonValue = request.ToJson();
378                 StreamWriter requestWriter(File.CreateText(Path.Combine(CmCodeLogDir(), "build-request.json")));
379                 CodeFormatter requestFormatter(requestWriter);
380                 requestJsonValue->Write(requestFormatter);
381                 if (saveRequestOnly) return List<CompileError>();
382                 TcpSocket socket("localhost", ToString(port));
383                 string requestStr = requestJsonValue->ToString();
384                 Write(socket, requestStr);
385                 string replyStr = ReadStr(socket);
386                 UniquePtr<JsonValue> replyJsonValue = ParseJson(replyStr);
387                 string messageKind = GetMessageKind(replyJsonValue.Get());
388                 while (messageKind == "logMessageRequest" || messageKind == "progressMessage")
389                 {
390                     if (messageKind == "logMessageRequest")
391                     {
392                         LogMessageRequest logMessageRequest(replyJsonValue.Get());
393                         string message = logMessageRequest.message;
394                         LogMessage(message);
395                         LogMessageReply logMessageReply;
396                         logMessageReply.ok = true;
397                         UniquePtr<JsonValue> logMessageReplyJsonValue = logMessageReply.ToJson();
398                         string logMessageReplyStr = logMessageReplyJsonValue->ToString();
399                         Write(socket, logMessageReplyStr);
400                     }
401                     else if (messageKind == "progressMessage")
402                     {
403                         ProgressMessage progressMessage(replyJsonValue.Get());
404                         notifyProgress();
405                     }
406                     replyStr = ReadStr(socket);
407                     replyJsonValue = ParseJson(replyStr);
408                     messageKind = GetMessageKind(replyJsonValue.Get());
409                 }
410                 if (messageKind == "buildReply")
411                 {
412                     StreamWriter replyWriter(File.CreateText(Path.Combine(CmCodeLogDir(), "build-reply.json")));
413                     CodeFormatter replyFormatter(replyWriter);
414                     replyJsonValue->Write(replyFormatter);
415                     BuildReply buildReply(replyJsonValue.Get());
416                     return ProcessBuildReply(buildReply);
417                 }
418                 else if (messageKind == "genericErrorReply")
419                 {
420                     GenericErrorReply genericErrorReply(replyJsonValue.Get());
421                     LogMessage("BuildService: ProcessBuildRequest: generic error reply received: " + genericErrorReply.error);
422                 }
423                 else
424                 {
425                     throw Exception("unknown reply message kind received");
426                 }
427             }
428             catch (const Exception& ex)
429             {
430                 if (!terminating)
431                 {
432                     LogException(ex);
433                     LogMessage("BuildService: ProcessBuildRequest: " + ex.Message());
434                 }
435             }
436             return List<CompileError>();
437         }
438         private List<CompileError> ProcessBuildReply(const BuildReply& buildReply)
439         {
440             if (buildReply.success)
441             {
442                 LogMessage("BuildService: build successful, time=" + buildReply.time);
443             }
444             else
445             {
446                 LogMessage("BuildService: build unsuccessful");
447             }
448             if (!buildReply.requestValid)
449             {
450                 throw Exception("server claims build request not valid: " + buildReply.requestErrorMessage);
451             }
452             if (!buildReply.logException.IsEmpty())
453             {
454                 throw Exception("server log exception: " + buildReply.logException);
455             }
456             return Rvalue(buildReply.errors);
457         }
458         private string GetMessageKind(JsonValue* jsonValue)
459         {
460             if (jsonValue is JsonObject*)
461             {
462                 JsonObject* jsonObject = cast<JsonObject*>(jsonValue);
463                 JsonValue* messageKindValue = jsonObject->GetField(u"messageKind");
464                 if (messageKindValue is JsonString*)
465                 {
466                     JsonString* messageKindStr = cast<JsonString*>(messageKindValue);
467                     string messageKind = ToUtf8(messageKindStr->Value());
468                     return messageKind;
469                 }
470             }
471             return string();
472         }
