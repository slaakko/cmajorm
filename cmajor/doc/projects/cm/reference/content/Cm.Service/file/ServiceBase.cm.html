 6 using System;
 7 using System.Collections;
 8 using System.Threading;
 9 using System.IO;
10 using Cm.Ast;
11 
12 namespace Cm.Service
13 {
14     public class delegate void LogMessageNotificationFunction();
15 
16     public string CmCodeLogDir()
17     {
18         string cmajorRootDir = CmajorRootDir();
19         string logDir = Path.Combine(Path.Combine(Path.Combine(Path.Combine(cmajorRootDir, "projects"), "cm"), "cmcode"), "log");
20         Directory.CreateDirectories(logDir);
21         return logDir;
22     }
23 
24     public void LogException(const Exception& ex)
25     {
26         try
27         {
28             StreamWriter errorWriter = File.CreateText(Path.Combine(CmCodeLogDir(), "error.txt"));
29             errorWriter << ex.ToString() << endl();
30         }
31         catch (const Exception&)
32         {
33         }
34     }
35 
36     public class ServiceBase
37     {
38         public ServiceBase() : notifyLogMessageAvailable()
39         {
40         }
41         public nothrow void SetLogMessageNotificationFunction(const LogMessageNotificationFunction& notifyLogMessageAvailable_)
42         {
43             notifyLogMessageAvailable = notifyLogMessageAvailable_;
44         }
45         public void LogMessage(const string& message)
46         {
47             {
48                 LockGuard<RecursiveMutex> lock(messageLogMutex);
49                 messageLog.Add(message);
50             }
51             if (notifyLogMessageAvailable != LogMessageNotificationFunction())
52             {
53                 notifyLogMessageAvailable();
54             }
55         }
56         public nothrow bool MessageLogEmpty() const
57         {
58             LockGuard<RecursiveMutex> lock(messageLogMutex);
59             return messageLog.IsEmpty();
60         }
61         public string FetchLogMessage()
62         {
63             LockGuard<RecursiveMutex> lock(messageLogMutex);
64             if (!messageLog.IsEmpty())
65             {
66                 string message = messageLog.Front();
67                 messageLog.RemoveFirst();
68                 return message;
69             }
70             return string();
71         }
72         private LinkedList<string> messageLog;
73         private RecursiveMutex messageLogMutex;
74         private LogMessageNotificationFunction notifyLogMessageAvailable;
75     }
76 }
