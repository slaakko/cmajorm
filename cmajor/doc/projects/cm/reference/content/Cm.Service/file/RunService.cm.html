 68     public class RunService : ServiceBase
 69     {
 70         public nothrow RunService(RunServiceArgs& args_) : base(), args(args_), processStarted(false), cmdbSessionBridge(this)
 71         {
 72         }

 85         public void Execute()
 86         {
 87             bool sessionServerStarted = false;
 88             string cmdbSessionFilePath = Path.ChangeExtension(args.executableName, ".cmdbs");
 89             try
 90             {
 91                 int portMapServicePortNumber = -1;
 92                 int port = Cm.Service.PortMap.GetFreePortNumber("cmcode", args.pid, portMapServicePortNumber);
 93                 if (port == -1)
 94                 {
 95                     port = args.sessionPort;
 96                     LogMessage("RunService: got no port from port map server, resorting to default port " + ToString(port));
 97                 }
 98                 sessionServer.Reset(new CmdbSessionServer(port, cmdbSessionFilePath, &cmdbSessionBridge));
 99                 sessionServer->Start();
100                 sessionServerStarted = true;
101                 processStarted = false;
102                 bool sessionFileRemoved = false;
103                 LogMessage("Starting '" + args.executableName + "' with arguments [" + args.arguments + "]...");
104                 runProcessThread = Thread.StartMethod(RunProcess);
105                 while (!processStarted)
106                 {
107                     Sleep(Duration.FromSeconds(1));
108                 }
109                 while (!process.IsNull() && process->Running())
110                 {
111                     if (!sessionFileRemoved && sessionServer->SessionOpen())
112                     {
113                         sessionFileRemoved = true;
114                         sessionServer->RemoveSessionFilePath();
115                     }
116                     CmdbSessionRequest* request = sessionServer->GetRequest();
117                     if (request != null)
118                     {
119                         request->Execute(&cmdbSessionBridge);
120                     }
121                     if (targetRunning != TargetRunningMethod())
122                     {
123                         targetRunning();
124                     }
125                 }
126                 runProcessThread.Join();
127                 int exitCode = process->ExitCode();
128                 LogMessage("Process exited with code " + ToString(exitCode));
129                 if (!sessionFileRemoved)
130                 {
131                     try
132                     {
133                         File.Remove(cmdbSessionFilePath);
134                     }
135                     catch (const Exception& ex)
136                     {
137                     }
138                 }
139                 if (sessionServerStarted)
140                 {
141                     sessionServer->Stop();
142                 }
143             }
144             catch (const Exception& ex)
145             {
146                 LogMessage("RunService: exception from Run(): " + ex.Message());
147                 LogException(ex);
148                 if (File.Exists(cmdbSessionFilePath))
149                 {
150                     try
151                     {
152                         File.Remove(cmdbSessionFilePath);
153                     }
154                     catch (const Exception& ex)
155                     {
156                     }
157                 }
158                 if (sessionServerStarted)
159                 {
160                     sessionServer->Stop();
161                 }
162             }
163         }
164         public void TerminateProcess()
165         {
166             try
167             {
168                 if (!process.IsNull() && process->Running())
169                 {
170                     process->Terminate();
171                 }
172             }
173             catch (const Exception& ex)
174             {
175                 LogMessage("RunService: exception from TerminateProcess: " + ex.Message());
176                 LogException(ex);
177             }
178         }
179         public string GetTargetInputBytes()
180         {
181             if (readTargetInput != ReadTargetInputMethod())
182             {
183                 string targetInputLine;
184                 bool targetInputEof = false;
185                 readTargetInput(targetInputLine, targetInputEof);
186                 if (targetInputEof)
187                 {
188                     return string();
189                 }
190                 else
191                 {
192                     string targetInputBytes;
193                     for (char c : targetInputLine)
194                     {
195                         byte x = cast<byte>(c);
196                         targetInputBytes.Append(ToHexString(x));
197                     }
198                     targetInputBytes.Append(ToHexString(cast<byte>('\n')));
199                     return targetInputBytes;
200                 }
201             }
202             return string();
203         }
204         public void WriteTargetOutput(int handle, const string& output)
205         {
206             if (writeTargetOutput != WriteTargetOutputMethod())
207             {
208                 writeTargetOutput(handle, output);
209             }
210         }
211         private void RunProcess()
212         {
213             try
214             {
215                 string command = args.executableName;
216                 command.Append(" ").Append(args.arguments);
217                 process.Reset(new Process(command, Process.Redirections.none));
218                 processStarted = true;
219                 process->WaitForExit();
220             }
221             catch (const Exception& ex)
222             {
223                 LogException(ex);
224                 processStarted = true;
225             }
226         }

229         private ReadTargetInputMethod readTargetInput;
230         private TargetRunningMethod targetRunning;
231         private bool processStarted;
232         private CmdbSessionBridge cmdbSessionBridge;
233         private UniquePtr<Process> process;
234         private Thread runProcessThread;
235         private UniquePtr<CmdbSessionServer> sessionServer;
236     }
