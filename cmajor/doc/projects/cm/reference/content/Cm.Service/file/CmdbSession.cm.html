 17     public abstract class CmdbSessionClient
 18     {
 19         public virtual default ~CmdbSessionClient();
 20         public abstract void WriteOutputMessage(const string& message);

 23     }

 25     public abstract class CmdbSessionRequest
 26     {
 27         public virtual default ~CmdbSessionRequest();

 29     }

 31     public abstract class CmdbSessionReply
 32     {
 33         public virtual default ~CmdbSessionReply();

 36     public class CmdbInputRequest : CmdbSessionRequest
 37     {
 38         public nothrow CmdbInputRequest(CmdbSessionServer* server_) : server(server_)
 39         {
 40         }
 41         public override void Execute(CmdbSessionClient* client)
 42         {
 43             server->SetReply(new CmdbInputReply(client->GetTargetInputBytes()));
 44         }

 46     }

101         public CmdbSessionRequest* GetRequest()
102         {
103             LockGuard<RecursiveMutex> lock(mtx);
104             if (requestVar.WaitFor(mtx, RequestWaiting, null, Duration.FromMilliseconds(250)))
105             {
106                 requestWaiting = false;
107                 replyReady = false;
108                 return request.Get();
109             }
110             return null;
111         }
112         public void SetReply(CmdbSessionReply* reply_)
113         {
114             LockGuard<RecursiveMutex> lock(mtx);
115             reply.Reset(reply_);
116             replyReady = true;
117             replyVar.NotifyOne();
118         }
119         public void Start()
120         {
121             try
122             {
123                 StartCmdbSession();
124                 thread = Thread.StartMethod(Run);
125                 while (!started)
126                 {
127                     Sleep(Duration.FromSeconds(1));
128                 }
129             }
130             catch (const Exception& ex)
131             {
132                 client->WriteOutputMessage("CmdbSession: could not start CMDB session: " + ex.Message());
133             }
134         }
135         public void Stop()
136         {
137             try
138             {
139                 if (started)
140                 {
141                     exiting = true;
142                     if (running)
143                     {
144                         try
145                         {
146                             TcpSocket socket("localhost", ToString(port));
147                         }
148                         catch (const Exception& ex)
149                         {
150                         }
151                     }
152                     thread.Join();
153                 }
154             }
155             catch (const Exception& ex)
156             {
157                 client->WriteOutputMessage("CmdbSession: could not stop CMDB session: " + ex.Message());
158             }
159         }
160         private void StartCmdbSession()
161         {
162             StreamWriter cmdbSessionFile = File.CreateText(cmdbSessionFilePath);
163             CodeFormatter formatter(cmdbSessionFile);
164             formatter.SetIndentSize(1);
165             Document cmdbSessionDoc;
166             Element* sessionElement = new Element(u"cmdbSession");
167             Element* timestampElement = new Element(u"timestamp");
168             long timestamp = RtGetCurrentTime();
169             timestampElement->SetAttribute(u"value", ToUtf32(ToString(timestamp)));
170             sessionElement->AppendChild(UniquePtr<Node>(timestampElement));
171             Element* skeyElement = new Element(u"skey");
172             Uuid skey = Uuid.Random();
173             skeyStr = ToString(skey);
174             skeyElement->SetAttribute(u"value", ToUtf32(skeyStr));
175             sessionElement->AppendChild(UniquePtr<Node>(skeyElement));
176             Uuid rkey = Uuid.Random();
177             rkeyStr = ToString(rkey);
178             Element* rkeyElement = new Element(u"rkey");
179             rkeyElement->SetAttribute(u"value", ToUtf32(rkeyStr));
180             sessionElement->AppendChild(UniquePtr<Node>(rkeyElement));
181             Element* portElement = new Element(u"port");
182             portElement->SetAttribute(u"value", ToUtf32(ToString(port)));
183             sessionElement->AppendChild(UniquePtr<Node>(portElement));
184             cmdbSessionDoc.AppendChild(UniquePtr<Node>(sessionElement));
185             cmdbSessionDoc.Write(formatter);
186         }
187         private void Run()
188         {
189             try
190             {
191                 listenSocket.Bind(port);
192                 listenSocket.Listen(10);
193                 started = true;
194                 running = true;
195                 while (!exiting)
196                 {
197                     TcpSocket socket = listenSocket.Accept();
198                     if (exiting) return;
199                     sessionClosed = false;
200                     while (!sessionClosed)
201                     {
202                         Transact(socket);
203                     }
204                 }
205             }
206             catch (const Exception& ex)
207             {
208                 client->WriteOutputMessage("CmdbSession.Run failed: " + ex.Message());
209                 started = true;
210             }
211             running = false;
212         }
213         private void Transact(TcpSocket& socket)
214         {
215             ustring requestStr = ToUtf32(ReadStr(socket));
216             UniquePtr<Document> requestDoc = ParseDocument(requestStr, "socket");
217             UniquePtr<XPathObject> messageObject = Evaluate(u"/cmdbMessage", requestDoc.Get());
218             if (messageObject.Get() is XPathNodeSet*)
219             {
220                 XPathNodeSet* nodeSet = cast<XPathNodeSet*>(messageObject.Get());
221                 if (nodeSet->Length() == 1)
222                 {
223                     Node* node = (*nodeSet)[0];
224                     if (node is Element*)
225                     {
226                         Element* element = cast<Element*>(node);
227                         ustring kind = element->GetAttribute(u"kind");
228                         if (kind == u"openSessionRequest")
229                         {
230                             OpenSession(socket, element);
231                         }
232                         else if (kind == u"closeSessionRequest")
233                         {
234                             CloseSession();
235                         }
236                         else if (kind == u"inputRequest")
237                         {
238                             GetInput(socket);
239                         }
240                         else if (kind == u"outputRequest")
241                         {
242                             PutOutput(socket, element);
243                         }
244                         else
245                         {
246                             throw Exception("invalid CMDB session request received: kind='" + ToUtf8(kind) + "'");
247                         }
248                     }
249                     else
250                     {
251                         throw Exception("invalid CMDB session request received: element node expected");
252                     }
253                 }
254                 else
255                 {
256                     throw Exception("invalid CMDB session request received: one node expected");
257                 }
258             }
259             else
260             {
261                 throw Exception("invalid CMDB session request received: node set expected");
262             }
263         }
264         private void OpenSession(TcpSocket& socket, Element* request)
265         {
266             ustring skeyValue = request->GetAttribute(u"key");
267             if (!skeyValue.IsEmpty())
268             {
269                 if (GetSha1MessageDigest(skeyStr) == ToUtf8(skeyValue))
270                 {
271                     Document openSessionResponse;
272                     Element* responseMessage = new Element(u"cmdbMessage");
273                     responseMessage->SetAttribute(u"kind", u"openSessionResponse");
274                     responseMessage->SetAttribute(u"key", ToUtf32(GetSha1MessageDigest(rkeyStr)));
275                     openSessionResponse.AppendChild(UniquePtr<Node>(responseMessage));
276                     WriteDocumentToSocket(openSessionResponse, socket);
277                     sessionOpen = true;
278                 }
279             }
280         }
281         private void CloseSession()
282         {
283             sessionClosed = true;
284         }
285         private bool ReplyReady(void* arg)
286         {
287             return replyReady;
288         }
289         private void GetInput(TcpSocket& socket)
290         {
291             string bytes;
292             requestWaiting = true;
293             replyReady = false;
294             request.Reset(new CmdbInputRequest(this));
295             requestVar.NotifyOne();
296             {
297                 LockGuard<RecursiveMutex> lock(mtx);
298                 replyVar.Wait(mtx, ReplyReady, null);
299                 if (!reply.IsNull() && (reply.Get() is CmdbInputReply*))
300                 {
301                     CmdbInputReply* inputReply = cast<CmdbInputReply*>(reply.Get());
302                     bytes = inputReply->bytes;
303                 }
304             }
305             requestWaiting = false;
306             Document inputResponse;
307             Element* inputResponseElement = new Element(u"cmdbMessage");
308             inputResponseElement->SetAttribute(u"kind", u"inputResponse");
309             inputResponseElement->SetAttribute(u"bytes", ToUtf32(bytes));
310             inputResponse.AppendChild(UniquePtr<Node>(inputResponseElement));
311             WriteDocumentToSocket(inputResponse, socket);
312         }
313         private void PutOutput(TcpSocket& socket, Element* requestElement)
314         {
315             ustring handleValue = requestElement->GetAttribute(u"handle");
316             if (!handleValue.IsEmpty())
317             {
318                 string handleStr = ToUtf8(handleValue);
319                 int handle = ParseInt(handleStr);
320                 if (handle == 1 || handle == 2)
321                 {
322                     ustring bytesValue = requestElement->GetAttribute(u"bytes");
323                     if (!bytesValue.IsEmpty())
324                     {
325                         string output;
326                         string bytesStr = ToUtf8(bytesValue);
327                         long n = bytesStr.Length() / 2;
328                         for (long i = 0; i < n; ++i;)
329                         {
330                             string hex;
331                             hex.Append(bytesStr[i * 2]);
332                             hex.Append(bytesStr[i * 2 + 1]);
333                             byte x = ParseHexByte(hex);
334                             output.Append(cast<char>(x));
335                         }
336                         requestWaiting = true;
337                         replyReady = false;
338                         request.Reset(new CmdbOutputRequest(this, handle, output));
339                         requestVar.NotifyOne();
340                         {
341                             LockGuard<RecursiveMutex> lock(mtx);
342                             replyVar.Wait(mtx, ReplyReady, null);
343                         }
344                         requestWaiting = false;
345                         Document outputResponse;
346                         Element* outputResponseElement = new Element(u"cmdbMessage");
347                         outputResponseElement->SetAttribute(u"kind", u"outputResponse");
348                         outputResponse.AppendChild(UniquePtr<Node>(outputResponseElement));
349                         WriteDocumentToSocket(outputResponse, socket);
350                     }
351                 }
352             }
353         }
354         private void WriteDocumentToSocket(const Document& doc, TcpSocket& socket)
355         {
356             StringWriter writer;
357             CodeFormatter formatter(writer);
358             formatter.SetIndentSize(1);
359             doc.Write(formatter);
360             string str = writer.GetString();
361             Write(socket, str);
362         }
