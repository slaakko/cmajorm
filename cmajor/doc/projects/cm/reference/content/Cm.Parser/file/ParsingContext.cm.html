 6 using System;
 7 using System.Collections;
 8 
 9 namespace Cm.Parser
10 {
11     public class ParsingContext
12     {
13         public nothrow ParsingContext() : 
14             parsingExpressionStatement(false), parsingLvalue(false), parsingArguments(false), parsingIsOrAs(false), parsingTypeExpr(false), parsingConcept(false), parsingTemplateId(false)
15         {
16         }
17         public nothrow void BeginParsingTemplateId()
18         {
19             parsingTemplateIdStack.Push(parsingTemplateId);
20             parsingTemplateId = true;
21         }
22         public nothrow void EndParsingTemplateId()
23         {
24             parsingTemplateId = parsingTemplateIdStack.Pop();
25         }
26         public nothrow void PushParsingIsOrAs(bool parsingIsOrAs_)
27         {
28             parsingIsOrAsStack.Push(parsingIsOrAs);
29             parsingIsOrAs = parsingIsOrAs_;
30         }
31         public nothrow void PopParsingIsOrAs()
32         {
33             parsingIsOrAs = parsingIsOrAsStack.Pop();
34         }
35         public nothrow void BeginParsingTypeExpr()
36         {
37             parsingTypeExprStack.Push(parsingTypeExpr);
38             parsingTypeExpr = true;
39         }
40         public nothrow void EndParsingTypeExpr()
41         {
42             parsingTypeExpr = parsingTypeExprStack.Pop();
43         }
44         public nothrow void PushParsingLvalue(bool parsingLvalue_)
45         {
46             parsingLvalueStack.Push(parsingLvalue);
47             parsingLvalue = parsingLvalue_;
48         }
49         public nothrow void PopParsingLvalue()
50         {
51             parsingLvalue = parsingLvalueStack.Pop();
52         }
53         public nothrow void PushParsingExpressionStatement(bool parsingExpressionStatement_)
54         {
55             parsingExpressionStatementStack.Push(parsingExpressionStatement);
56             parsingExpressionStatement = parsingExpressionStatement_;
57         }
58         public nothrow void PopParsingExpressionStatement()
59         {
60             parsingExpressionStatement = parsingExpressionStatementStack.Pop();
61         }
62         public nothrow void BeginParsingArguments()
63         {
64             parsingArgumentsStack.Push(parsingArguments);
65             parsingArguments = true;
66         }
67         public nothrow void EndParsingArguments()
68         {
69             parsingArguments = parsingArgumentsStack.Pop();
70         }
71         public nothrow void BeginParsingConcept()
72         {
73             parsingConceptStack.Push(parsingConcept);
74             parsingConcept = true;
75         }
76         public nothrow void EndParsingConcept()
77         {
78             parsingConcept = parsingConceptStack.Pop();
79         }
80         public bool parsingExpressionStatement;
81         public bool parsingLvalue;
82         public bool parsingArguments;
83         public bool parsingIsOrAs;
84         public bool parsingTypeExpr;
85         public bool parsingConcept;
86         public bool parsingTemplateId;
87         private Stack<bool> parsingExpressionStatementStack;
88         private Stack<bool> parsingLvalueStack;
89         private Stack<bool> parsingArgumentsStack;
90         private Stack<bool> parsingIsOrAsStack;
91         private Stack<bool> parsingTypeExprStack;
92         private Stack<bool> parsingConceptStack;
93         private Stack<bool> parsingTemplateIdStack;
94     }
95 }
