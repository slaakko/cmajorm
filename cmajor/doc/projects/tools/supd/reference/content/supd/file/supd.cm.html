 14 void Make(const List<string>& files, bool force, bool verbose, bool colors)
 15 {
 16     for (const string& file : files)
 17     {
 18         if (File.Exists(file))
 19         {
 20             if (!force)
 21             {
 22                 throw Exception("cannot make: file '" + file + "' already exists and no --force option specified");
 23             }
 24         }
 25         if (verbose)
 26         {
 27             if (colors)
 28             {
 29                 Console.Out() << TextColor.darkCyan << "make" << Color.reset << ": " << TextColor.white << file << Color.reset << endl();
 30             }
 31             else
 32             {
 33                 Console.Out() << "make" << ": " << file << endl();
 34             }
 35         }
 36         string dir = Path.GetDirectoryName(file);
 37         Document doc;
 38         Element* supdElement = new Element(u"supd");
 39         doc.AppendChild(UniquePtr<Node>(supdElement));
 40         List<string> directories = Directory.GetDirectories(dir);
 41         for (const string& directory : directories)
 42         {
 43             string dirName = Path.GetFileName(directory);
 44             if (verbose)
 45             {
 46                 Console.Out() << TextColor.green << "add" << Color.reset << ": " << TextColor.white << dirName << Color.reset << endl();
 47             }
 48             Element* dirElement = new Element(u"dir");
 49             dirElement->SetAttribute(u"name", ToUtf32(dirName));
 50             supdElement->AppendChild(UniquePtr<Node>(dirElement));
 51         }
 52         StreamWriter writer(File.CreateText(file));
 53         CodeFormatter formatter(writer);
 54         formatter.SetIndentSize(1);
 55         doc.Write(formatter);
 56         if (verbose)
 57         {
 58             Console.Out() << TextColor.yellow << "==> " << Color.reset << TextColor.white << file << Color.reset << endl();
 59         }
 60     }
 61 }

 63 class FileMask
 64 {
 65     public enum Kind
 66     {
 67         include, exclude
 68     }
 69     public enum Actions
 70     {
 71         none = 0, add = 1 << 0, update = 1 << 1
 72     }
 73     public nothrow FileMask(Context& context, Kind kind_, const ustring& mask_, Actions actions_) : kind(kind_), mask(mask_), actions(actions_)
 74     {
 75         nfa = CompileFilePattern(context, mask);
 76     }
 77     public nothrow FileMask(Context& context, Kind kind, const ustring& mask) : this(context, kind, mask, cast<Actions>(Actions.update | Actions.add))
 78     {
 79     }
 80     public nothrow bool Enabled(Actions action) const
 81     {
 82         return (actions & action) != Actions.none;
 83     }
 84     public void Match(const ustring& filename, bool& include)
 85     {
 86         bool match = PatternMatch(filename, nfa);
 87         if (kind == Kind.include)
 88         {
 89             if (match)
 90             {
 91                 include = true;
 92             }
 93         }
 94         else if (kind == Kind.exclude)
 95         {
 96             if (match)
 97             {
 98                 include = false;
 99             }
100         }
101     }
102     public nothrow Kind GetKind() const
103     {
104         return kind;
105     }
106     private Kind kind;
107     private ustring mask;
108     private Nfa nfa;
109     private Actions actions;
110 }

179         }
180         return include;

183     {

187             fileMask->Match(fileName, include);

193                 }
194             }
195         }
196         return true;
197     }

291 PatternSet ReadPatterns(Context& context, Document* doc, const string& file, bool verbose, bool colors)
292 {
293     if (verbose)
294     {
295         if (colors)
296         {
297             Console.Out() << TextColor.darkCyan << "reading" << Color.reset << " patterns from " << TextColor.white << file << Color.reset << endl();
298         }
299         else
300         {
301             Console.Out() << "reading" << " patterns from " << file << endl();
302         }
303     }
304     PatternSet patterns;
305     UniquePtr<XPathObject> patternObject = Evaluate(u"/supd/pattern", doc);
306     if (!patternObject.IsNull())
307     {
308         if (patternObject.Get() is XPathNodeSet*)
309         {
310             XPathNodeSet* nodeSet = cast<XPathNodeSet*>(patternObject.Get());
311             for (int i = 0; i < nodeSet->Length(); ++i;)
312             {
313                 Node* node = (*nodeSet)[i];
314                 if (node is Element*)
315                 {
316                     Element* patternElement = cast<Element*>(node);
317                     string name = ToUtf8(patternElement->GetAttribute(u"name"));
318                     if (name.IsEmpty())
319                     {
320                         throw Exception("'supd/pattern' element index " + ToString(i) + " has no 'name' attribute");
321                     }
322                     UniquePtr<Pattern> pattern(new Pattern(name));
323                     UniquePtr<XPathObject> fileObject = Evaluate(u"file", patternElement);
324                     if (!fileObject.IsNull())
325                     {
326                         if (fileObject.Get() is XPathNodeSet*)
327                         {
328                             XPathNodeSet* nodeSet = cast<XPathNodeSet*>(fileObject.Get());
329                             for (int i = 0; i < nodeSet->Length(); ++i;)
330                             {
331                                 Node* node = (*nodeSet)[i];
332                                 if (node is Element*)
333                                 {
334                                     Element* fileElement = cast<Element*>(node);
335                                     ustring action = fileElement->GetAttribute(u"action");
336                                     ustring mask = fileElement->GetAttribute(u"include");
337                                     if (mask.IsEmpty())
338                                     {
339                                         mask = fileElement->GetAttribute(u"exclude");
340                                         if (mask.IsEmpty())
341                                         {
342                                             throw Exception("file element index " + ToString(i) + " in pattern '" + pattern->Name() + "' has no include/exclude attribute");
343                                         }
344                                         else
345                                         {
346                                             pattern->AddFileMask(new FileMask(context, FileMask.Kind.exclude, mask));
347                                         }
348                                     }
349                                     else
350                                     {
351                                         if (action.IsEmpty())
352                                         {
353                                             pattern->AddFileMask(new FileMask(context, FileMask.Kind.include, mask));
354                                         }
355                                         else
356                                         {
357                                             FileMask.Actions actions = FileMask.Actions.none;
358                                             if (action == u"add")
359                                             {
360                                                 actions = cast<FileMask.Actions>(actions | FileMask.Actions.add);
361                                             }
362                                             else if (action == u"update")
363                                             {
364                                                 actions = cast<FileMask.Actions>(actions | FileMask.Actions.update);
365                                             }
366                                             pattern->AddFileMask(new FileMask(context, FileMask.Kind.include, mask, actions));
367                                         }
368                                     }
369                                 }
370                             }
371                         }
372                     }
373                     UniquePtr<XPathObject> dirObject = Evaluate(u"dir", patternElement);
374                     if (!dirObject.IsNull())
375                     {
376                         if (dirObject.Get() is XPathNodeSet*)
377                         {
378                             XPathNodeSet* nodeSet = cast<XPathNodeSet*>(dirObject.Get());
379                             for (int i = 0; i < nodeSet->Length(); ++i;)
380                             {
381                                 Node* node = (*nodeSet)[i];
382                                 if (node is Element*)
383                                 {
384                                     Element* dirElement = cast<Element*>(node);
385                                     ustring mask = dirElement->GetAttribute(u"include");
386                                     if (mask.IsEmpty())
387                                     {
388                                         mask = dirElement->GetAttribute(u"exclude");
389                                         if (mask.IsEmpty())
390                                         {
391                                             throw Exception("dir element index " + ToString(i) + " in pattern '" + pattern->Name() + "' has no include/exclude attribute");
392                                         }
393                                         else
394                                         {
395                                             pattern->AddDirMask(new DirMask(context, DirMask.Kind.exclude, mask));
396                                         }
397                                     }
398                                     else
399                                     {
400                                         pattern->AddDirMask(new DirMask(context, DirMask.Kind.include, mask));
401                                     }
402                                 }
403                             }
404                         }
405                     }
406                     patterns.AddPattern(pattern.Release());
407                 }
408             }
409         }
410     }
411     return patterns;
412 }

471 bool FilesDiffer(const string& sourceFilePath, const string& targetFilePath)
472 {
473     long size = File.Size(sourceFilePath);
474     if (size != File.Size(targetFilePath))
475     {
476         return true;
477     }
478     BinaryReader sourceReader = File.OpenBinary(sourceFilePath);
479     BinaryReader targetReader = File.OpenBinary(targetFilePath);
480     for (ulong i = 0u; i < size; ++i;)
481     {
482         byte sourceByte = sourceReader.ReadByte();
483         byte targetByte = targetReader.ReadByte();
484         if (sourceByte != targetByte)
485         {
486             return true;
487         }
488     }
489     return false;
490 }

549 void ProcessDir(const string& sourceDir, const string& targetDir, Pattern* pattern, bool includeSubdirs, bool verbose, bool add, bool update, int& updateCount, int& addCount, bool colors)
550 {
551     if (verbose)
552     {
553         if (colors)
554         {
555             Console.Out() << TextColor.darkCyan << "from " << Color.reset << TextColor.white << sourceDir << Color.reset << endl();
556         }
557         else
558         {
559             Console.Out() << "from " << sourceDir << endl();
560         }
561     }
562     bool targetDirExists = false;
563     List<string> sourceFiles = Directory.GetFiles(sourceDir);
564     for (const string& sourceFile : sourceFiles)
565     {
566         string fileName = Path.GetFileName(sourceFile);
567         ustring fname;
568         try
569         {
570             fname = ToUtf32(fileName);
571         }
572         catch (const Exception& ex)
573         {
574             if (colors)
575             {
576                 Console.Out() << TextColor.magenta << "warning" << Color.reset << ": invalid file name" << ": could not convert file name to UTF-32" << endl();
577             }
578             else
579             {
580                 Console.Out() << "warning" << ": invalid file name" << ": could not convert file name to UTF-32" << endl();
581             }
582         }
583         if (!fname.IsEmpty())
584         {
585             if (pattern->MatchFile(fname))
586             {
587                 string targetFile = Path.Combine(targetDir, fileName);
588                 if (!targetDirExists && (add || update))
589                 {
590                     targetDirExists = Directory.Exists(targetDir);
591                     if (!targetDirExists)
592                     {
593                         targetDirExists = true;
594                         Directory.CreateDirectories(targetDir);
595                         if (colors)
596                         {
597                             Console.Out() << TextColor.green << "added" << Color.reset << ": " << TextColor.white << targetDir << Color.reset << endl();
598                         }
599                         else
600                         {
601                             Console.Out() << "added" << ": " << targetDir << endl();
602                         }
603                     }
604                 }
605                 if (File.Exists(targetFile))
606                 {
607                     if (pattern->Enabled(fname, FileMask.Actions.update))
608                     {
609                         if (FilesDiffer(sourceFile, targetFile))
610                         {
611                             UpdateFile(sourceFile, targetFile, update, updateCount, colors);
612                         }
613                     }
614                 }
615                 else
616                 {
617                     if (pattern->Enabled(fname, FileMask.Actions.add))
618                     {
619                         AddFile(sourceFile, targetFile, add, addCount, colors);
620                     }
621                 }
622             }
623         }
624     }
625     if (includeSubdirs)
626     {
627         List<string> sourceSubdirs = Directory.GetDirectories(sourceDir);
628         for (const string& sourceSubdir : sourceSubdirs)
629         {
630             string dirName = Path.GetFileName(sourceSubdir);
631             ustring dname;
632             try
633             {
634                 dname = ToUtf32(dirName);
635             }
636             catch (const Exception& ex)
637             {
638                 if (colors)
639                 {
640                     Console.Out() << TextColor.magenta << "warning" << Color.reset << ": invalid directory name" << ": could not convert file name to UTF-32" << endl();
641                 }
642                 else
643                 {
644                     Console.Out() << "warning" << ": invalid file name" << ": could not convert directory name to UTF-32" << endl();
645                 }
646             }
647             string targetSubdir = Path.Combine(targetDir, dirName);
648             if (pattern->MatchDir(dname))
649             {
650                 ProcessDir(sourceSubdir, targetSubdir, pattern, includeSubdirs, verbose, add, update, updateCount, addCount, colors);
651             }
652         }
653     }

656 void Process(const string& file, const string& from, bool verbose, bool add, bool update, bool colors)
657 {
658     int updateCount = 0;
659     int addCount = 0;
660     if (!File.Exists(file))
661     {
662         throw Exception("file '" + file + "' does not exist");
663     }
664     string targetRoot = Path.GetDirectoryName(file);
665     UniquePtr<Document> doc = ReadDocument(file);
666     string sourceRoot = from;
667     if (sourceRoot.IsEmpty())
668     {
669         UniquePtr<XPathObject> supdObject = Evaluate(u"/supd", doc.Get());
670         if (!supdObject.IsNull())
671         {
672             if (supdObject.Get() is XPathNodeSet*)
673             {
674                 XPathNodeSet* nodeSet = cast<XPathNodeSet*>(supdObject.Get());
675                 if (nodeSet->Length() == 1)
676                 {
677                     Node* node = (*nodeSet)[0];
678                     if (node is Element*)
679                     {
680                         Element* element = cast<Element*>(node);
681                         sourceRoot = ToUtf8(element->GetAttribute(u"from"));
682                     }
683                 }
684             }
685         }
686     }
687     if (sourceRoot.IsEmpty())
688     {
689         throw Exception("'" + file + "' does not contain 'supd' element with from=\"FROM_DIR_PATH\" attribute set and no --from=FROM_DIR_PATH option specified. stop.");
690     }
691     if (!Directory.Exists(sourceRoot))
692     {
693         throw Exception("source root directory '" + sourceRoot + "' not found. stop.");
694     }
695     Context context;
696     PatternSet patterns = ReadPatterns(context, doc.Get(), file, verbose, colors);
697     DirSet dirs = ReadDirs(doc.Get(), patterns, file, verbose, colors);
698     List<string> sourceDirs = Directory.GetDirectories(sourceRoot);
699     for (const string& sourceDir : sourceDirs)
700     {
701         string dirName = Path.GetFileName(sourceDir);
702         Dir* dir = dirs.GetDir(dirName);
703         if (dir != null)
704         {
705             string targetDir = Path.Combine(targetRoot, dirName);
706             ProcessDir(sourceDir, targetDir, dir->GetPattern(), dir->IncludeSubdirs(), verbose, add, update, updateCount, addCount, colors);
707         }
708     }
709     if (add)
710     {
711         if (colors)
712         {
713             Console.Out() << addCount << " files " << TextColor.green << "added" << Color.reset << endl();
714         }
715         else
716         {
717             Console.Out() << addCount << " files " << "added" << endl();
718         }
719     }
720     else
721     {
722         if (colors)
723         {
724             Console.Out() << addCount << " files to be " << TextColor.green << "added" << Color.reset << endl();
725         }
726         else
727         {
728             Console.Out() << addCount << " files to be " << "added" << endl();
729         }
730     }
731     if (update)
732     {
733         if (colors)
734         {
735             Console.Out() << updateCount << " files " << TextColor.yellow << "updated" << Color.reset << endl();
736         }
737         else
738         {
739             Console.Out() << updateCount << " files " << "updated" << endl();
740         }
741     }
742     else
743     {
744         if (colors)
745         {
746             Console.Out() << updateCount << " files to be " << TextColor.yellow << "updated" << Color.reset << endl();
747         }
748         else
749         {
750             Console.Out() << updateCount << " files to be " << "updated" << endl();
751         }
752     }
753 }

879                 }
880             }

884                 {
885                     List<string> components = arg.Split('=');
886                     if (components.Count() == 2)
887                     {
888                         if (components[0] == "-c")
