 6 using System;
 7 using System.Collections;
 8 using System.Windows;
 9 
10 namespace spring
11 {
12     public class NotePropertiesDialog : Window
13     {
14         public NotePropertiesDialog(NoteElement* noteElement_) : 
15             base("spring.notePropertiesDialog", DefaultWindowClassStyle(), DialogWindowStyle(), DefaultExtendedWindowStyle(), 
16             DefaultControlBackgroundColor(), "Note Properties", Point.Default(), Size(ScreenMetrics.Get().MMToHorizontalPixels(100), ScreenMetrics.Get().MMToVerticalPixels(70)), 
17             Dock.none, Anchors.none, WindowState.normal), noteElement(noteElement_)
18         {
19             Size s = GetSize();
20             Size defaultButtonSize = ScreenMetrics.Get().DefaultButtonSize();
21             Size defaultControlSpacing = ScreenMetrics.Get().DefaultControlSpacing();
22             Size defaultLabelSize = ScreenMetrics.Get().DefaultLabelSize();
23             Size defaultTextBoxSize = ScreenMetrics.Get().DefaultTextBoxSize();
24             Point textBoxLoc(defaultControlSpacing.w, defaultControlSpacing.h);
25             Size textBoxSize(ScreenMetrics.Get().MMToHorizontalPixels(80), ScreenMetrics.Get().MMToVerticalPixels(30));
26             string noteText;
27             bool first = true;
28             for (const string& line : noteElement->Lines())
29             {
30                 if (first)
31                 {
32                     first = false;
33                 }
34                 else
35                 {
36                     noteText.Append("\r\n");
37                 }
38                 noteText.Append(line);
39             }
40             UniquePtr<TextBox> textBoxPtr(new TextBox(cast<TextBoxStyle>(TextBoxStyle.ES_LEFT | TextBoxStyle.ES_AUTOHSCROLL | TextBoxStyle.ES_AUTOVSCROLL | TextBoxStyle.ES_MULTILINE), 
41                 System.Windows.Color.White(), noteText, textBoxLoc, textBoxSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left)));
42             textBox = textBoxPtr.Get();
43             textBox->TextChangedEvent().AddHandler(NoteTextChanged);
44             textBox->SelectAll();
45             UniquePtr<PaddedControl> paddedTextBox(new PaddedControl(textBoxPtr.Release()));
46             UniquePtr<BorderedControl> borderedPaddedTextBox(new BorderedControl(paddedTextBox.Release(), BorderStyle.style3D, textBoxLoc, textBoxSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left)));
47             AddChild(borderedPaddedTextBox.Release());
48             int x = s.w - defaultButtonSize.w - defaultControlSpacing.w;
49             int y = s.h - defaultButtonSize.h - defaultControlSpacing.h;
50             UniquePtr<Button> cancelButtonPtr(new Button("Cancel", Point(x, y), defaultButtonSize, Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
51             cancelButton = cancelButtonPtr.Get();
52             cancelButton->SetDialogResult(DialogResult.cancel);
53             x = x - defaultButtonSize.w - defaultControlSpacing.w;
54             UniquePtr<Button> okButtonPtr(new Button(ButtonStyle.BS_DEFPUSHBUTTON, GetSystemColor(SystemColor.COLOR_BTNFACE), "OK", Point(x, y), defaultButtonSize, 
55                 Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
56             okButton = okButtonPtr.Get();
57             okButton->SetDialogResult(DialogResult.ok);
58             AddChild(okButtonPtr.Release());
59             AddChild(cancelButtonPtr.Release());
60             SetDefaultButton(okButton);
61             SetCancelButton(cancelButton);
62             textBox->SetFocus();
63         }
64         private void NoteTextChanged()
65         {
66             ustring text = ToUtf32(textBox->Text());
67             List<ustring> ulines = SplitTextIntoLines(text);
68             List<string> lines;
69             for (const ustring& line : ulines)
70             {
71                 lines.Add(ToUtf8(line));
72             }
73             noteElement->SetLines(lines);
74         }
75         private NoteElement* noteElement;
76         private Button* okButton;
77         private Button* cancelButton;
78         private TextBox* textBox;
79     }
80 }
