  6 using System;
  7 using System.Windows;
  8 
  9 namespace spring
 10 {
 11     public class ClassPropertiesDialog : Window
 12     {
 13         public ClassPropertiesDialog(ClassElement* classElement_) : 
 14             base("spring.classPropertiesDialog", DefaultWindowClassStyle(), DialogWindowStyle(), DefaultExtendedWindowStyle(), 
 15             DefaultControlBackgroundColor(), "Class Properties", Point.Default(), Size(ScreenMetrics.Get().MMToHorizontalPixels(100), ScreenMetrics.Get().MMToVerticalPixels(70)), 
 16             Dock.none, Anchors.none), classElement(classElement_)
 17         {
 18             int column1Width = ScreenMetrics.Get().MMToHorizontalPixels(20);
 19             Size s = GetSize();
 20             Size defaultButtonSize = ScreenMetrics.Get().DefaultButtonSize();
 21             Size defaultControlSpacing = ScreenMetrics.Get().DefaultControlSpacing();
 22             Size defaultLabelSize = ScreenMetrics.Get().DefaultLabelSize();
 23             Size defaultTextBoxSize = ScreenMetrics.Get().DefaultTextBoxSize();
 24             int column2Width = ScreenMetrics.Get().MMToHorizontalPixels(30);
 25             int textBoxPadding = DefaultControlPadding();
 26             int borderWidth = DefaultSingleBorderWidth();
 27             int line1Top = defaultControlSpacing.h + textBoxPadding + borderWidth;
 28             Point classNameLabelLoc(defaultControlSpacing.w, line1Top);
 29             UniquePtr<Label> classNameLabel(new Label("Class name:", classNameLabelLoc, defaultLabelSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left), true));
 30             int line2Top = classNameLabel->Location().y + classNameLabel->GetSize().h + defaultControlSpacing.h;
 31             AddChild(classNameLabel.Release());
 32             Point classNameTextBoxLoc(defaultControlSpacing.w * 2 + column1Width, line1Top);
 33             Size classNameTextBoxSize(column2Width - 2 * (textBoxPadding + borderWidth), defaultTextBoxSize.h);
 34             Rect classNameRect(classNameTextBoxLoc, classNameTextBoxSize);
 35             Rect paddedClassNameRect = classNameRect;
 36             paddedClassNameRect.Inflate(textBoxPadding, textBoxPadding);
 37             UniquePtr<TextBox> classNameTextBoxPtr(new TextBox(classElement->Name()));
 38             classNameTextBox = classNameTextBoxPtr.Get();
 39             classNameTextBox->SelectAll();
 40             classNameTextBox->TextChangedEvent().AddHandler(ClassNameChanged);
 41             UniquePtr<Control> paddedClassNameTextBox(new PaddedControl(classNameTextBoxPtr.Release()));
 42             Rect borderedPaddedClassNameTextBoxRect = paddedClassNameRect;
 43             borderedPaddedClassNameTextBoxRect.Inflate(borderWidth, borderWidth);
 44             UniquePtr<Control> borderedPaddedClassNameTextBox(new BorderedControl(paddedClassNameTextBox.Release(), borderedPaddedClassNameTextBoxRect.location, borderedPaddedClassNameTextBoxRect.size, 
 45                 Dock.none, cast<Anchors>(Anchors.top | Anchors.left)));
 46             AddChild(borderedPaddedClassNameTextBox.Release());
 47             Point abstractCheckBoxLoc(borderedPaddedClassNameTextBoxRect.location.x, line2Top);
 48             UniquePtr<CheckBox> abstractCheckBoxPtr(new CheckBox("Abstract", abstractCheckBoxLoc, Size(), Dock.none, cast<Anchors>(Anchors.top | Anchors.left), true));
 49             abstractCheckBox = abstractCheckBoxPtr.Get();
 50             abstractCheckBox->SetChecked(classElement->IsAbstract());
 51             abstractCheckBox->CheckedChangedEvent().AddHandler(AbstractCheckBoxChanged);
 52             AddChild(abstractCheckBoxPtr.Release());
 53             Size abstractCheckBoxSize = abstractCheckBox->GetSize();
 54             int line3Top = abstractCheckBoxLoc.y + abstractCheckBoxSize.h + defaultControlSpacing.h;
 55             Point operationssLabelLoc(defaultControlSpacing.w, line3Top);
 56             UniquePtr<Label> operationsLabel(new Label("Operations:", operationssLabelLoc, defaultLabelSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left), true));
 57             Size operationsLabelSize = operationsLabel->GetSize();
 58             AddChild(operationsLabel.Release());
 59             Point editOperationsButtonLoc(defaultControlSpacing.w + column1Width + defaultControlSpacing.w - (textBoxPadding + borderWidth), line3Top - (textBoxPadding + borderWidth));
 60             Size editOperationsButtonSize(ScreenMetrics.Get().MMToHorizontalPixels(8), defaultButtonSize.h);
 61             UniquePtr<Button> editOperationsButtonPtr(new Button("...", editOperationsButtonLoc, editOperationsButtonSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left)));
 62             editOperationsButton = editOperationsButtonPtr.Get();
 63             editOperationsButton->ClickEvent().AddHandler(EditOperations);
 64             AddChild(editOperationsButtonPtr.Release());
 65             int line4Top = operationssLabelLoc.y + operationsLabelSize.h + defaultControlSpacing.h + textBoxPadding + borderWidth;
 66             Point fieldsLabelLoc(defaultControlSpacing.w, line3Top);
 67             UniquePtr<Label> fieldsLabel(new Label("Fields:", fieldsLabelLoc, defaultLabelSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left), true));
 68             Size fieldsLabelSize = fieldsLabel->GetSize();
 69             AddChild(fieldsLabel.Release());
 70             Point editFieldsButtonLoc(defaultControlSpacing.w + column1Width + defaultControlSpacing.w - (textBoxPadding + borderWidth), line4Top - (textBoxPadding + borderWidth));
 71             Size editFieldsButtonSize(ScreenMetrics.Get().MMToHorizontalPixels(8), defaultButtonSize.h);
 72             UniquePtr<Button> editFieldsButtonPtr(new Button("...", editFieldsButtonLoc, editFieldsButtonSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left)));
 73             editFieldsButton = editFieldsButtonPtr.Get();
 74             editFieldsButton->ClickEvent().AddHandler(EditFields);
 75             AddChild(editFieldsButtonPtr.Release());
 76             int x = s.w - defaultButtonSize.w - defaultControlSpacing.w;
 77             int y = s.h - defaultButtonSize.h - defaultControlSpacing.h;
 78             UniquePtr<Button> cancelButtonPtr(new Button("Cancel", Point(x, y), defaultButtonSize, Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
 79             cancelButton = cancelButtonPtr.Get();
 80             cancelButton->SetDialogResult(DialogResult.cancel);
 81             x = x - defaultButtonSize.w - defaultControlSpacing.w;
 82             UniquePtr<Button> okButtonPtr(new Button(ButtonStyle.BS_DEFPUSHBUTTON, GetSystemColor(SystemColor.COLOR_BTNFACE), "OK", Point(x, y), defaultButtonSize, 
 83                 Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
 84             okButton = okButtonPtr.Get();
 85             okButton->SetDialogResult(DialogResult.ok);
 86             if (classElement->Name().IsEmpty())
 87             {
 88                 okButton->Disable();
 89                 editOperationsButton->Disable();
 90                 editFieldsButton->Disable();
 91             }
 92             AddChild(okButtonPtr.Release());
 93             AddChild(cancelButtonPtr.Release());
 94             SetDefaultButton(okButton);
 95             SetCancelButton(cancelButton);
 96         }
 97         private void ClassNameChanged()
 98         {
 99             if (classNameTextBox->Text().IsEmpty())
100             {
101                 okButton->Disable();
102                 editOperationsButton->Disable();
103                 editFieldsButton->Disable();
104             }
105             else
106             {
107                 okButton->Enable();
108                 editOperationsButton->Enable();
109                 editFieldsButton->Enable();
110                 classElement->SetName(classNameTextBox->Text());
111             }
112         }
113         private void AbstractCheckBoxChanged()
114         {
115             bool isAbstract = abstractCheckBox->Checked();
116             LogView* logView = Application.GetLogView();
117             if (logView != null)
118             {
119                 logView->WriteLine("abstract=" + ToString(isAbstract));
120             }
121             classElement->SetAbstract(isAbstract);
122         }
123         private void EditOperations()
124         {
125             bool mayHaveAbstractOperations = classElement->IsAbstract();
126             IndexList<OperationElement> clonedOperationElements = classElement->Operations().Clone();
127             if (ExecuteEditOperationsDialog(clonedOperationElements, classElement, mayHaveAbstractOperations, *this))
128             {
129                 classElement->SetOperations(clonedOperationElements);
130             }
131         }
132         private void EditFields()
133         {
134             IndexList<FieldElement> clonedFieldElements = classElement->Fields().Clone();
135             if (ExecuteEditFieldsDialog(clonedFieldElements, classElement, *this))
136             {
137                 classElement->SetFields(clonedFieldElements);
138             }
139         }
140         private ClassElement* classElement;
141         private Button* okButton;
142         private Button* cancelButton;
143         private Button* editOperationsButton;
144         private Button* editFieldsButton;
145         private TextBox* classNameTextBox;
146         private CheckBox* abstractCheckBox;
147     }
148 }
