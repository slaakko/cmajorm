 84         void SetCursor(const PointF& location)
 85         {
 86             if (Tools.Current() is SelectTool*)
 87             {
 88                 if (diagram->GetOperation() is ResizeRectangleSelectionOperation*)
 89                 {
 90                     ResizeRectangleSelectionOperation* operation = cast<ResizeRectangleSelectionOperation*>(diagram->GetOperation());
 91                     SetCursor(*operation->GetCursor());
 92                     return;
 93                 }
 94                 else
 95                 {
 96                     if (diagram->GetSelection() is RectangleSelection*)
 97                     {
 98                         RectangleSelection* selection = cast<RectangleSelection*>(diagram->GetSelection());
 99                         SetCursor(*selection->GetCursor(location));
100                         return;
101                     }
102                 }
103             }
104             if (buttonPressed)
105             {
106                 SetCursor(*Tools.Current()->GetPressedCursor());
107             }
108             else
109             {
110                 SetCursor(*Tools.Current()->GetCursor());
111             }
112         }
113         protected override void OnPaint(PaintEventArgs& args)
114         {
115             SetCursor();
116             Unit prevUnit = args.graphics.GetPageUnitChecked();
117             TextRenderingHint prevTextRenderingHint = args.graphics.GetTextRenderingHint();
118             args.graphics.SetTextRenderingHintChecked(TextRenderingHint.clearTypeGridFit);
119             SmoothingMode prevSmoothingMode = args.graphics.GetSmoothingModeChecked();
120             args.graphics.SetSmoothingModeChecked(SmoothingMode.highQuality);
121             args.graphics.SetPageUnitChecked(Unit.millimeter);
122             args.graphics.ClearChecked(BackgroundColor());
123             args.graphics.TranslateTransformChecked(-viewOrigin.x, -viewOrigin.y);
124             diagram->Draw(args.graphics);
125             args.graphics.SetTextRenderingHintChecked(prevTextRenderingHint);
126             args.graphics.SetPageUnitChecked(prevUnit);
127             args.graphics.SetSmoothingModeChecked(prevSmoothingMode);
128         }

145         protected override void OnMouseDown(MouseEventArgs& args)
146         {
147             base->OnMouseDown(args);
148             PointF location = ToDiagramLocation(args.location);
149             DiagramElement* element = null;
150             int index = diagram->GetIndexOfElementAt(location);
151             if (index != -1)
152             {
153                 element = diagram->GetElementByIndex(index);
154             }
155             MouseButtons buttons = args.buttons;
156             bool shift = (buttons & MouseButtons.shift) != MouseButtons.none;
157             bool control = (buttons & MouseButtons.control) != MouseButtons.none;
158             bool left = (buttons & MouseButtons.lbutton) != MouseButtons.none;
159             bool right = (buttons & MouseButtons.rbutton) != MouseButtons.none;
160             mouseDownArgs = MouseArgs(diagram.Get(), location, index, element, shift, control, left, right);
161             operationCreated = false;
162             if (left)
163             {
164                 buttonPressed = true;
165             }
166             SetCursor(location);
167         }
168         protected override void OnMouseMove(MouseEventArgs& args)
169         {
170             base->OnMouseMove(args);
171             PointF location = ToDiagramLocation(args.location);
172             MouseButtons buttons = args.buttons;
173             bool shift = (buttons & MouseButtons.shift) != MouseButtons.none;
174             bool control = (buttons & MouseButtons.control) != MouseButtons.none;
175             bool left = (buttons & MouseButtons.lbutton) != MouseButtons.none;
176             bool right = (buttons & MouseButtons.rbutton) != MouseButtons.none;
177             DiagramElement* element = null;
178             int index = -1;
179             if (left || right)
180             {
181                 index = diagram->GetIndexOfElementAt(location);
182                 if (index != -1)
183                 {
184                     element = diagram->GetElementByIndex(index);
185                 }
186             }
187             mouseMoveArgs = MouseArgs(diagram.Get(), location, index, element, shift, control, left, right);
188             if (Tools.Current()->CanMove())
189             {
190                 if (left && !right && !shift && !control)
191                 {
192                     if (!operationCreated)
193                     {
194                         operationCreated = true;
195                         Tools.Current()->CreateOperation(mouseDownArgs);
196                     }
197                     Tools.Current()->ContinueOperation(mouseMoveArgs);
198                 }
199             }
200             else if (Tools.Current() is RelationshipTool*)
201             {
202                 diagram->AdvanceOperation(mouseMoveArgs);
203             }
204             SetCursor(location);
205         }
206         protected override void OnMouseUp(MouseEventArgs& args)
207         {
208             base->OnMouseUp(args);
209             PointF location = ToDiagramLocation(args.location);
210             DiagramElement* element = null;
211             int index = diagram->GetIndexOfElementAt(location);
212             if (index != -1)
213             {
214                 element = diagram->GetElementByIndex(index);
215             }
216             MouseButtons buttons = args.buttons;
217             bool shift = (buttons & MouseButtons.shift) != MouseButtons.none;
218             bool control = (buttons & MouseButtons.control) != MouseButtons.none;
219             bool left = (buttons & MouseButtons.lbutton) != MouseButtons.none;
220             bool right = (buttons & MouseButtons.rbutton) != MouseButtons.none;
221             mouseUpArgs = MouseArgs(diagram.Get(), location, index, element, shift, control, left, right);
222             if (Tools.Current()->CanMove())
223             {
224                 Duration duration = mouseUpArgs.time - mouseDownArgs.time;
225                 if (duration <= Duration.FromMilliseconds(clickThresholdMs))
226                 {
227                     if (Tools.Current() is SelectTool*)
228                     {
229                         if (operationCreated)
230                         {
231                             operationCreated = false;
232                             Tools.Current()->CancelOperation(mouseUpArgs);
233                         }
234                         Tools.Current()->MouseClick(mouseUpArgs);
235                     }
236                 }
237                 else
238                 {
239                     if (operationCreated)
240                     {
241                         operationCreated = false;
242                         Tools.Current()->CommitOperation(mouseUpArgs);
243                     }
244                 }
245             }
246             else if (Tools.Current() is SimpleClickTool*)
247             {
248                 Tools.Current()->MouseClick(mouseUpArgs);
249             }
250             buttonPressed = false;
251             SetCursor(location);
252         }

271         public nothrow PointF ToDiagramLocation(const Point& controlLocation)
272         {
273             float x = ScreenMetrics.Get().HorizontalPixelsToMM(controlLocation.x);
274             float y = ScreenMetrics.Get().VerticalPixelsToMM(controlLocation.y);
275             PointF loc(x, y);
276             loc.Offset(viewOrigin.x, viewOrigin.y);
277             return loc;
278         }
279         public nothrow Point ToControlLocation(const PointF& diagramLocation)
280         {
281             PointF loc(diagramLocation);
282             loc.Offset(-viewOrigin.x, -viewOrigin.y);
283             int x = ScreenMetrics.Get().MMToHorizontalPixels(loc.x);
284             int y = ScreenMetrics.Get().MMToVerticalPixels(loc.y);
285             return Point(x, y);
286         }
