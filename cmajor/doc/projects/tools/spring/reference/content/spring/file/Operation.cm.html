110     public class MoveContainerElementsOperation : Operation
111     {
112         public nothrow MoveContainerElementsOperation(Diagram* diagram, const PointF& startLocation) : base(diagram, startLocation), resetSelection(false), moveCommand(new MoveCommand(*diagram))
113         {
114             diagram->GetSelection()->SetStartLocation(startLocation);
115         }
116         public void ResetSelection()
117         {
118             resetSelection = true;
119         }
120         public override nothrow void AddElementIndex(int index)
121         {
122             elementIndeces.Add(index);
123         }
124         public override nothrow void SetElementIndeces(List<int>&& elementIndeces_)
125         {
126             elementIndeces = elementIndeces_;
127         }
128         public override void AdvanceTo(MouseArgs& args)
129         {
130             int n = cast<int>(elementIndeces.Count());
131             #assert(n == cast<int>(originalLocations.Count()));
132             PointF startLocation = StartLocation();
133             PointF nextLocation = args.location;
134             if (args.control)
135             {
136                 Line line(startLocation, nextLocation);
137                 Vector v(line.ToVector());
138                 switch (MainDirection(v))
139                 {
140                     case 0:
141                     case 180:
142                     {
143                         nextLocation.y = startLocation.y;
144                         break;
145                     }
146                     case 90:
147                     case 270:
148                     {
149                         nextLocation.x = startLocation.x;
150                         break;
151                     }
152                 }
153             }
154             float dx = nextLocation.x - startLocation.x;
155             float dy = nextLocation.y - startLocation.y;
156             for (int i = 0; i < n; ++i;)
157             {
158                 int elementIndex = elementIndeces[i];
159                 CompoundLocation newLocation = originalLocations[i];
160                 newLocation.Offset(dx, dy);
161                 DiagramElement* element = GetDiagram()->GetElementByIndex(elementIndex);
162                 element->SetCompoundLocation(newLocation);
163             }
164             GetDiagram()->GetSelection()->Move(args);
165         }
166         public override void Draw(Graphics& graphics, Layout& layout)
167         {
168         }
169         public override void Cancel()
170         {
171             int n = cast<int>(elementIndeces.Count());
172             #assert(n == cast<int>(originalLocations.Count()));
173             for (int i = 0; i < n; ++i;)
174             {
175                 int elementIndex = elementIndeces[i];
176                 const CompoundLocation& originalLocation = originalLocations[i];
177                 DiagramElement* element = GetDiagram()->GetElementByIndex(elementIndex);
178                 element->SetCompoundLocation(originalLocation);
179             }
180         }
181         public override void Commit(MouseArgs& args)
182         {
183             AdvanceTo(args);
184             if (resetSelection)
185             {
186                 GetDiagram()->ResetSelection();
187             }
188             for (int index : elementIndeces)
189             {
190                 DiagramElement* element = GetDiagram()->GetElementByIndex(index);
191                 moveCommand->AddNewLocation(element->GetCompoundLocation());
192             }
193             GetDiagram()->GetCommandList().AddCommand(moveCommand.Release());
194         }
195         public override nothrow void FetchOriginalLocations()
196         {
197             for (int elementIndex : elementIndeces)
198             {
199                 DiagramElement* element = GetDiagram()->GetElementByIndex(elementIndex);
200                 CompoundLocation compoundLocation = element->GetCompoundLocation();
201                 originalLocations.Add(compoundLocation);
202                 moveCommand->AddOldLocation(elementIndex, compoundLocation);
203             }
204         }
205         public override nothrow bool IsEmpty() const
206         {
207             return elementIndeces.IsEmpty();
208         }
209         private List<int> elementIndeces;
210         private List<CompoundLocation> originalLocations;
211         private bool resetSelection;
212         private UniquePtr<MoveCommand> moveCommand;
213     }

215     public class AddRelationshipOperation : Operation
216     {
217         public nothrow AddRelationshipOperation(Diagram* diagram, RelationshipElement* relationshipElement_) : base(diagram), relationshipElement(relationshipElement_)
218         {
219         }
220         public override nothrow bool IsEmpty() const
221         {
222             return false;
223         }
224         public override void AdvanceTo(MouseArgs& args)
225         {
226             PointF lastPoint = relationshipElement->LastPoint();
227             PointF targetPoint = args.location;
228             if (args.control)
229             {
230                 Line line(lastPoint, targetPoint);
231                 Vector v(line.ToVector());
232                 switch (MainDirection(v))
233                 {
234                     case 0:
235                     case 180:
236                     {
237                         targetPoint.y = lastPoint.y;
238                         break;
239                     }
240                     case 90:
241                     case 270:
242                     {
243                         targetPoint.x = lastPoint.x;
244                         break;
245                     }
246                 }
247             }
248             relationshipElement->target.point = targetPoint;
249         }
250         public override void Draw(Graphics& graphics, Layout& layout)
251         {
252             relationshipElement->Draw(graphics, layout);
253         }
254         public override void Cancel()
255         {
256             relationshipElement->RemoveFromElements();
257         }
258         public override void Commit(MouseArgs& args)
259         {
260             relationshipElement->AddToElements();
261             Diagram* diagram = GetDiagram();
262             UniquePtr<AddElementsCommand> addElementsCommand(new AddElementsCommand(*diagram));
263             addElementsCommand->AddIndex(diagram->NextIndex());
264             diagram->AddElement(relationshipElement.Release());
265             Tools.SetCurrent(Tools.GetSelectTool());
266             diagram->GetCommandList().AddCommand(addElementsCommand.Release());
267         }
268         public nothrow RelationshipElement& GetRelationshipElement()
269         {
270             return *relationshipElement;
271         }
272         private UniquePtr<RelationshipElement> relationshipElement;
273     }

311     public class ResizeRectangleSelectionOperation : Operation
312     {
313         public ResizeRectangleSelectionOperation(Diagram* diagram, const PointF& startLocation, ResizeHandle* resizeHandle_) : base(diagram, startLocation), resizeHandle(resizeHandle_)
314         {
315             RectangleSelection* selection = resizeHandle->GetSelection();
316             selection->SetStartLocation(startLocation);
317         }
318         public override nothrow bool IsEmpty() const
319         {
320             return resizeHandle == null;
321         }
322         public override void AdvanceTo(MouseArgs& args)
323         {
324             if (resizeHandle == null) return;
325             PointF currentLocation = args.location;
326             PointF startLocation = StartLocation();
327             float dx = currentLocation.x - startLocation.x;
328             float dy = currentLocation.y - startLocation.y;
329             resizeHandle->Move(dx, dy);
330         }
331         public override void Cancel()
332         {
333             if (resizeHandle != null)
334             {
335                 resizeHandle->GetSelection()->SetRect(resizeHandle->GetSelection()->StartRect());
336                 resizeHandle = null;
337             }
338         }
339         public override void Commit(MouseArgs& args)
340         {
341             if (resizeHandle != null)
342             {
343                 AdvanceTo(args);
344                 if (!resizeHandle->GetSelection()->IsEmpty())
345                 {
346                     GetDiagram()->EnableCopy();
347                     GetDiagram()->EnableCut();
348                 }
349                 else
350                 {
351                     GetDiagram()->DisableCopy();
352                     GetDiagram()->DisableCut();
353                 }
354                 resizeHandle = null;
355             }
356         }
357         public override void Draw(Graphics& graphics, Layout& layout)
358         {
359         }
360         public nothrow Cursor* GetCursor() const
361         {
362             return resizeHandle->GetCursor();
363         }
364         private ResizeHandle* resizeHandle;
365     }
