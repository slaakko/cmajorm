 207         public override nothrow void Delete()
 208         {
 209             UniquePtr<DeleteElementsCommand> deleteElementsCommand(new DeleteElementsCommand(GetDiagram()));
 210             List<RelationshipElement*> relationships;
 211             int n = cast<int>(indeces.Count());
 212             for (int i = 0; i < n; ++i;)
 213             {
 214                 int index = indeces[i];
 215                 DiagramElement* element = GetDiagram().GetElementByIndex(index);
 216                 if (element is ContainerElement*)
 217                 {
 218                     ContainerElement* containerElement = cast<ContainerElement*>(element);
 219                     List<RelationshipElement*> containerElementRelationships = containerElement->GetAllRelationships();
 220                     for (RelationshipElement* relationship : containerElementRelationships)
 221                     {
 222                         if (Find(relationships.CBegin(), relationships.CEnd(), relationship) == relationships.CEnd())
 223                         {
 224                             relationships.Add(relationship);
 225                         }
 226                     }
 227                 }
 228             }
 229             for (RelationshipElement* relationship : relationships)
 230             {
 231                 int index = GetDiagram().GetIndexOfElement(relationship);
 232                 if (index != -1)
 233                 {
 234                     if (Find(indeces.CBegin(), indeces.CEnd(), index) == indeces.CEnd())
 235                     {
 236                         indeces.Add(index);
 237                     }
 238                 }
 239             }
 240             SortByIndex();
 241             int m = cast<int>(indeces.Count());
 242             for (int i = m - 1; i >= 0; --i;)
 243             {
 244                 int index = indeces[i];
 245                 UniquePtr<DiagramElement> element = GetDiagram().RemoveElementByIndex(index);
 246                 if (element.Get() is RelationshipElement*)
 247                 {
 248                     RelationshipElement* relationship = cast<RelationshipElement*>(element.Get());
 249                     relationship->RemoveFromElements();
 250                 }
 251                 deleteElementsCommand->AddDeletedElement(element.Release(), index);
 252             }
 253             GetDiagram().GetCommandList().AddCommand(deleteElementsCommand.Release());
 254             indeces.Clear();
 255         }

 262         public override nothrow List<UniquePtr<DiagramElement>> Clone() const
 263         {
 264             List<UniquePtr<DiagramElement>> clonedElements;
 265             List<ContainerElement*> containerElements;
 266             List<RelationshipElement*> relationshipElements;
 267             System.Windows.RectF boundingRect;
 268             int n = cast<int>(indeces.Count());
 269             for (int i = 0; i < n; ++i;)
 270             {
 271                 int index = indeces[i];
 272                 DiagramElement* element = GetDiagram().GetElementByIndex(index);
 273                 if (element is ContainerElement*)
 274                 {
 275                     if (boundingRect.IsEmpty())
 276                     {
 277                         boundingRect = element->BoundingRect();
 278                     }
 279                     else
 280                     {
 281                         boundingRect = System.Windows.RectF.Union(boundingRect, element->BoundingRect());
 282                     }
 283                     containerElements.Add(cast<ContainerElement*>(element));
 284                 }
 285             }
 286             for (ContainerElement* containerElement : containerElements)
 287             {
 288                 List<RelationshipElement*> containerRelationships = containerElement->GetAllRelationships();
 289                 for (RelationshipElement* relationship : containerRelationships)
 290                 {
 291                     bool sourceInClosure = false;
 292                     if (relationship->source.element != null)
 293                     {
 294                         ContainerElement* sourceContainerElement = relationship->source.element->GetContainerElement();
 295                         if (Find(containerElements.CBegin(), containerElements.CEnd(), sourceContainerElement) != containerElements.CEnd())
 296                         {
 297                             sourceInClosure = true;
 298                         }
 299                     }
 300                     bool sourcesInClosure = false;
 301                     if (!relationship->sourceEndpoints.IsEmpty())
 302                     {
 303                         sourcesInClosure = true;
 304                         for (const Endpoint& endpoint : relationship->sourceEndpoints)
 305                         {
 306                             ContainerElement* sourceContainerElement = endpoint.element->GetContainerElement();
 307                             if (Find(containerElements.CBegin(), containerElements.CEnd(), sourceContainerElement) == containerElements.CEnd())
 308                             {
 309                                 sourcesInClosure = false;
 310                                 break;
 311                             }
 312                         }
 313                     }
 314                     bool targetInClosure = false;
 315                     if (relationship->target.element != null)
 316                     {
 317                         ContainerElement* targetContainerElement = relationship->target.element->GetContainerElement();
 318                         if (Find(containerElements.CBegin(), containerElements.CEnd(), targetContainerElement) != containerElements.CEnd())
 319                         {
 320                             targetInClosure = true;
 321                         }
 322                     }
 323                     else if (relationship->kind == RelationshipElement.kind.reference && relationship->cardinality == Cardinality.zero)
 324                     {
 325                         targetInClosure = true;
 326                     }
 327                     if ((sourceInClosure || sourcesInClosure) && targetInClosure)
 328                     {
 329                         if (boundingRect.IsEmpty())
 330                         {
 331                             boundingRect = relationship->BoundingRect();
 332                         }
 333                         else
 334                         {
 335                             boundingRect = System.Windows.RectF.Union(boundingRect, relationship->BoundingRect());
 336                         }
 337                         if (Find(relationshipElements.CBegin(), relationshipElements.CEnd(), relationship) == relationshipElements.CEnd())
 338                         {
 339                             relationshipElements.Add(relationship);
 340                         }
 341                     }
 342                 }
 343             }
 344             HashMap<DiagramElement*, DiagramElement*> cloneMap;
 345             HashMap<ContainerElement*, int> containerElementIndexMap;
 346             int nc = cast<int>(containerElements.Count());
 347             for (int i = 0; i < nc; ++i;)
 348             {
 349                 ContainerElement* containerElement = containerElements[i];
 350                 containerElementIndexMap[containerElement] = i;
 351                 ContainerElement* clone = cast<ContainerElement*>(containerElement->Clone());
 352                 cloneMap[containerElement] = clone;
 353                 clonedElements.Add(UniquePtr<DiagramElement>(clone));
 354             }
 355             for (RelationshipElement* relationshipElement : relationshipElements)
 356             {
 357                 RelationshipElement* clone = cast<RelationshipElement*>(relationshipElement->Clone());
 358                 clone->SetContainerElementIndeces(containerElementIndexMap);
 359                 clone->MapContainerElements(cloneMap);
 360                 clone->AddToElements();
 361                 clonedElements.Add(UniquePtr<DiagramElement>(clone));
 362             }
 363             for (UniquePtr<DiagramElement>& clonedElement : clonedElements)
 364             {
 365                 clonedElement->Offset(-boundingRect.location.x, -boundingRect.location.y);
 366             }
 367             return clonedElements;
 368         }

 711         public void CombineInheritanceRelationships()
 712         {
 713             List<RelationshipElement*> inheritanceRelationships;
 714             for (int index : indeces)
 715             {
 716                 DiagramElement* element = GetDiagram().GetElementByIndex(index);
 717                 if (element is RelationshipElement*)
 718                 {

 720                     if (relationshipElement->kind == RelationshipElement.Kind.inheritance)
 721                     {

 723                     }
 724                 }
 725             }
 726             List<RelationshipElement*> sourceRelationships;
 727             UniquePtr<RelationshipElement> combinedInheritanceRelationship = CombineInheritanceRelationships(inheritanceRelationships, sourceRelationships);
 728             if (!combinedInheritanceRelationship.IsNull())
 729             {
 730                 List<int> inheritanceRelationshipIndeces;
 731                 for (RelationshipElement* sourceRelationship : sourceRelationships)
 732                 {

 734                     if (index != -1)
 735                     {
 736                         inheritanceRelationshipIndeces.Add(index);
 737                     }
 738                 }
 739                 UniquePtr<DeleteElementsCommand> deleteElementsCommand(new DeleteElementsCommand(GetDiagram()));
 740                 Sort(inheritanceRelationshipIndeces);

 742                 for (int i = n - 1; i >= 0; --i;)

 747                     {

 749                         relationship->RemoveFromElements();

 755                 int addIndex = GetDiagram().NextIndex();
 756                 UniquePtr<AddElementsCommand> addElementsCommand(new AddElementsCommand(GetDiagram()));
 757                 addElementsCommand->AddIndex(addIndex);

 760                 GetDiagram().GetCommandList().AddCommand(addElementsCommand.Release());
