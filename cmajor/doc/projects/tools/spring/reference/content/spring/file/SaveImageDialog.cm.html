177         private void FormatRadioButtonStateChanged()
178         {
179             if (!ready) return;
180             RadioButton* checkedRadioButton = formatGroupBox->CheckedRadioButton();
181             if (checkedRadioButton == pngRadioButton)
182             {
183                 format = ImageFormat.png;
184             }
185             else if (checkedRadioButton == jpgRadioButton)
186             {
187                 format = ImageFormat.jpeg;
188             }
189             else if (checkedRadioButton == gifRadioButton)
190             {
191                 format = ImageFormat.gif;
192             }
193             else if (checkedRadioButton == tiffRadioButton)
194             {
195                 format = ImageFormat.tiff;
196             }
197             else if (checkedRadioButton == bmpRadioButton)
198             {
199                 format = ImageFormat.bmp;
200             }
201             if (!fileName.IsEmpty())
202             {
203                 switch (format)
204                 {
205                     case ImageFormat.png:
206                     {
207                         fileName = Path.ChangeExtension(fileName, ".png");
208                         break;
209                     }
210                     case ImageFormat.jpeg:
211                     {
212                         fileName = Path.ChangeExtension(fileName, ".jpg");
213                         break;
214                     }
215                     case ImageFormat.gif:
216                     {
217                         fileName = Path.ChangeExtension(fileName, ".gif");
218                         break;
219                     }
220                     case ImageFormat.tiff:
221                     {
222                         fileName = Path.ChangeExtension(fileName, ".tiff");
223                         break;
224                     }
225                     case ImageFormat.bmp:
226                     {
227                         fileName = Path.ChangeExtension(fileName, ".bmp");
228                         break;
229                     }
230                 }
231                 if (fileNameTextBox != null)
232                 {
233                     fileNameTextBox->SetText(fileName);
234                 }
235             }
236         }

374         private void BrowseClick()
375         {
376             try
377             {
378                 List<Pair<string, string>> descriptionFilterPairs;
379                 descriptionFilterPairs.Add(Pair<string, string>("PNG files (*.png)", "*.png"));
380                 descriptionFilterPairs.Add(Pair<string, string>("JPG files (*.jpg)", "*.jpg"));
381                 descriptionFilterPairs.Add(Pair<string, string>("GIF files (*.gif)", "*.gif"));
382                 descriptionFilterPairs.Add(Pair<string, string>("TIFF files (*.tiff)", "*.tiff"));
383                 descriptionFilterPairs.Add(Pair<string, string>("BMP files (*.bmp)", "*.bmp"));
384 
385                 string initialDirectory = DefaultDiagramDirectory();
386                 string currentDirectory;
387                 List<string> fileNames;
388                 string defaultFilePath = fileName;
389                 string filePath;
390                 bool selected = System.Windows.API.GetSaveFileName(Handle(), descriptionFilterPairs, initialDirectory, defaultFilePath, ".png", 0u, currentDirectory, filePath);
391                 if (selected)
392                 {
393                     if (File.Exists(filePath))
394                     {
395                         if (MessageBox.Show("File '" + filePath + "' already exists. Do you want to replace it?", "File Exists", this, MessageBoxType.MB_YESNO) != MessageBoxResult.yes)
396                         {
397                             return;
398                         }
399                     }
400                     fileName = filePath;
401                     fileNameTextBox->SetText(fileName);
402                 }
403             }
404             catch (const Exception& ex)
405             {
406                 MessageBox.Show(ex.Message());
407             }
