213     public class DeleteDiagramElementAction : DiagramElementAction
214     {
215         public nothrow DeleteDiagramElementAction(Diagram& diagram, int elementIndex) : base(diagram, elementIndex)
216         {
217         }
218         protected override void Execute(Diagram& diagram, int elementIndex)
219         {
220             UniquePtr<DeleteElementsCommand> deleteElementsCommand(new DeleteElementsCommand(diagram));
221             UniquePtr<DiagramElement> element = diagram.RemoveElementByIndex(elementIndex);
222             if (element.Get() is ContainerElement*)
223             {
224                 ContainerElement* containerElement = cast<ContainerElement*>(element.Get());
225                 List<RelationshipElement*> containerElementRelationships = containerElement->GetAllRelationships();
226                 List<RelationshipElement*> relationships;
227                 for (RelationshipElement* relationship : containerElementRelationships)
228                 {
229                     if (Find(relationships.CBegin(), relationships.CEnd(), relationship) == relationships.CEnd())
230                     {
231                         relationships.Add(relationship);
232                     }
233                 }
234                 List<int> relationshipIndeces;
235                 for (RelationshipElement* relationship : relationships)
236                 {
237                     int index = diagram.GetIndexOfElement(relationship);
238                     if (index != -1)
239                     {
240                         if (Find(relationshipIndeces.CBegin(), relationshipIndeces.CEnd(), index) == relationshipIndeces.CEnd())
241                         {
242                             relationshipIndeces.Add(index);
243                         }
244                     }
245                 }
246                 Sort(relationshipIndeces.Begin(), relationshipIndeces.End());
247                 int m = cast<int>(relationshipIndeces.Count());
248                 for (int i = m - 1; i >= 0; --i;)
249                 {
250                     int index = relationshipIndeces[i];
251                     UniquePtr<DiagramElement> element = diagram.RemoveElementByIndex(index);
252                     deleteElementsCommand->AddDeletedElement(element.Release(), index);
253                 }
254             }
255             deleteElementsCommand->AddDeletedElement(element.Release(), elementIndex);
256             diagram.GetCommandList().AddCommand(deleteElementsCommand.Release());
257         }
258     }

282         {
283             DiagramElement* diagramElement = diagram.GetElementByIndex(elementIndex);
284             if (diagramElement is RelationshipElement*)
285             {
286                 RelationshipElement* relationship = cast<RelationshipElement*>(diagramElement);
287                 if (relationship->kind == RelationshipElement.Kind.inheritance && !relationship->sourceEndpoints.IsEmpty())
288                 {
289                     List<UniquePtr<RelationshipElement>> relationships = SplitCombinedInheritanceRelationship(relationship);
290                     UniquePtr<DeleteElementsCommand> deleteCommand(new DeleteElementsCommand(diagram));
291                     UniquePtr<DiagramElement> element(diagram.RemoveElementByIndex(elementIndex));
292                     deleteCommand->AddDeletedElement(element.Release(), elementIndex);
293                     diagram.GetCommandList().AddCommand(deleteCommand.Release());
294                     UniquePtr<AddElementsCommand> addElementsCommand(new AddElementsCommand(diagram));
295                     for (UniquePtr<RelationshipElement>& relationship : relationships)
296                     {
297                         int addIndex = diagram.NextIndex();
298                         addElementsCommand->AddIndex(addIndex);
299                         diagram.AddElement(relationship.Release());
300                     }
301                     diagram.GetCommandList().AddCommand(addElementsCommand.Release());
302                 }
303             }

311         }
312         protected override void Execute(Diagram& diagram, int elementIndex)
313         {
314             DiagramElement* element = diagram.GetElementByIndex(elementIndex);
315             if (element is ClassElement*)
316             {
317                 ClassElement* classElement = cast<ClassElement*>(element);
318                 UniquePtr<ClassElement> clone = cast<ClassElement*>(classElement->Clone());
319                 UniquePtr<ClassPropertiesDialog> classPropertiesDialog = new ClassPropertiesDialog(clone.Get());
320                 Canvas* canvas = diagram.GetCanvas();
321                 Window* window = canvas->GetWindow();
322                 if (window != null)
323                 {
324                     diagram.HideContextMenu();
325                     DialogResult result = classPropertiesDialog->ShowDialog(*window);
326                     if (result == DialogResult.ok)
327                     {
328                         if (!clone->IsAbstract())
329                         {
330                             clone->ResetAbstractOperations();
331                         }
332                         clone->SetDiagram(&diagram);
333                         UniquePtr<ReplaceElementCommand> replaceElementCommand(new ReplaceElementCommand(diagram, elementIndex));
334                         HashMap<DiagramElement*, DiagramElement*> cloneMap;
335                         cloneMap[classElement] = clone.Get();
336                         clone->MapChildObjects(classElement, cloneMap);
337                         List<RelationshipElement*> relationships = classElement->GetAllRelationships();
338                         for (RelationshipElement* relationship : relationships)
339                         {
340                             relationship->MapContainerElements(cloneMap);
341                             relationship->AddToElements();
342                         }
343                         diagram.SetElementByIndex(clone.Release(), elementIndex);
344                         diagram.SetChanged();
345                         diagram.Invalidate();
346                         diagram.GetCommandList().AddCommand(replaceElementCommand.Release());
347                     }
348                 }
