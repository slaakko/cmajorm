 80             UniquePtr<RadioButton> manyRadioButtonPtr(new RadioButton("Many",

136         private void SetKindRadioButton()
137         {
138             switch (relationshipElement->kind)
139             {
140                 case RelationshipElement.Kind.inheritance:
141                 {
142                     kindGroupBox->SetCheckedRadioButton(inheritanceRadioButton);
143                     zeroRadioButton->Disable();
144                     break;
145                 }
146                 case RelationshipElement.Kind.aggregation:
147                 {
148                     kindGroupBox->SetCheckedRadioButton(aggregationRadioButton);
149                     zeroRadioButton->Disable();
150                     break;
151                 }
152                 case RelationshipElement.Kind.reference:
153                 {
154                     kindGroupBox->SetCheckedRadioButton(referenceRadioButton);
155                     break;
156                 }
157                 case RelationshipElement.Kind.createInstance:
158                 {
159                     kindGroupBox->SetCheckedRadioButton(createInstanceRadioButton);
160                     zeroRadioButton->Disable();
161                     break;
162                 }
163             }
164         }
165         private void SetCardinalityRadioButton()
166         {
167             switch (relationshipElement->cardinality)
168             {
169                 case Cardinality.zero:
170                 {
171                     cardinalityGroupBox->SetCheckedRadioButton(zeroRadioButton);
172                     break;
173                 }
174                 case Cardinality.one:
175                 {
176                     cardinalityGroupBox->SetCheckedRadioButton(oneRadioButton);
177                     break;
178                 }
179                 case Cardinality.many:
180                 {
181                     cardinalityGroupBox->SetCheckedRadioButton(manyRadioButton);
182                     break;
183                 }
184             }
185         }
186         private void KindRadioButtonStateChanged()
187         {
188             if (ready)
189             {
190                 RadioButton* checkedRadioButton = kindGroupBox->CheckedRadioButton();
191                 if (checkedRadioButton == inheritanceRadioButton)
192                 {
193                     relationshipElement->kind = RelationshipElement.Kind.inheritance;
194                 }
195                 else if (checkedRadioButton == aggregationRadioButton)
196                 {
197                     relationshipElement->kind = RelationshipElement.Kind.aggregation;
198                 }
199                 else if (checkedRadioButton == referenceRadioButton)
200                 {
201                     relationshipElement->kind = RelationshipElement.Kind.reference;
202                 }
203                 else if (checkedRadioButton == createInstanceRadioButton)
204                 {
205                     relationshipElement->kind = RelationshipElement.Kind.createInstance;
206                 }
207                 if (checkedRadioButton == inheritanceRadioButton)
208                 {
209                     if (manyRadioButton->Checked())
210                     {
211                         oneRadioButton->SetChecked(true);
212                     }
213                     else if (zeroRadioButton->Checked())
214                     {
215                         oneRadioButton->SetChecked(true);
216                     }
217                     manyRadioButton->Disable();
218                     zeroRadioButton->Disable();
219                 }
220                 else if (checkedRadioButton == aggregationRadioButton)
221                 {
222                     if (zeroRadioButton->Checked())
223                     {
224                         oneRadioButton->SetChecked(true);
225                     }
226                     zeroRadioButton->Disable();
227                     manyRadioButton->Enable();
228                 }
229                 else if (checkedRadioButton == referenceRadioButton)
230                 {
231                     zeroRadioButton->Enable();
232                     manyRadioButton->Enable();
233                 }
234                 else if (checkedRadioButton == createInstanceRadioButton)
235                 {
236                     if (zeroRadioButton->Checked())
237                     {
238                         oneRadioButton->SetChecked(true);
239                     }
240                     zeroRadioButton->Disable();
241                     manyRadioButton->Enable();
242                 }
243             }
244         }
245         private void CardinalityRadioButtonStateChanged()
246         {
247             if (ready)
248             {
249                 RadioButton* checkedRadioButton = cardinalityGroupBox->CheckedRadioButton();
250                 if (checkedRadioButton == zeroRadioButton)
251                 {
252                     relationshipElement->cardinality = Cardinality.zero;
253                 }
254                 else if (checkedRadioButton == oneRadioButton)
255                 {
256                     relationshipElement->cardinality = Cardinality.one;
257                 }
258                 else if (checkedRadioButton == manyRadioButton)
259                 {
260                     relationshipElement->cardinality = Cardinality.many;
261                 }
262             }
263         }

267         }
268         private void TargetTextBoxTextChanged()

270             relationshipElement->target.text = targetTextBox->Text();
