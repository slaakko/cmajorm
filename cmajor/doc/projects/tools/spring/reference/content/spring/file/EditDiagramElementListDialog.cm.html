245                 if (selectedIndex > 0)
246                 {

248                 }

252                 }
253                 if (selectedIndex < operationList->Count() - 1)
254                 {
255                     GetMoveDownButton()->Enable();
256                 }
257                 else
258                 {
259                     GetMoveDownButton()->Disable();
260                 }
261             }
262             else

265                 GetDeleteButton()->Disable();
266             }
267         }
268         private void AddOperation()
269         {
270             string operationName = operationNameTextBox->Text();
271             bool isAbstract = abstractCheckBox->Checked();
272             OperationElement* operationElement = new OperationElement(operationName, isAbstract);
273             operationElement->SetContainerElement(containerElement);
274             operationList->Add(operationElement);
275             string listBoxItemString = operationName;
276             if (isAbstract)
277             {
278                 listBoxItemString.Append(", abstract=true");
279             }
280             GetListBox()->AddItem(listBoxItemString);
281             operationNameTextBox->Clear();
282             abstractCheckBox->SetChecked(false);
283             selectedIndex = -1;
284             GetDeleteButton()->Disable();
285             GetMoveUpButton()->Disable();
286             GetMoveDownButton()->Disable();
287         }
288         private void ChangeOperation()
289         {
290             if (selectedIndex != -1 && selectedOperation != null)
291             {
292                 string operationName = operationNameTextBox->Text();
293                 bool isAbstract = abstractCheckBox->Checked();
294                 selectedOperation->SetName(operationName);
295                 selectedOperation->SetAbstract(isAbstract);
296                 string listBoxItemString = operationName;
297                 if (isAbstract)
298                 {
299                     listBoxItemString.Append(", abstract=true");
300                 }
301                 GetListBox()->SetItem(selectedIndex, listBoxItemString);
302                 operationNameTextBox->Clear();
303                 operationNameTextBox->SetFocus();
304                 abstractCheckBox->SetChecked(false);
305                 selectedIndex = -1;
306                 SetDefaultButton(GetAddButton());
307                 GetDeleteButton()->Disable();
308                 GetMoveUpButton()->Disable();
309                 GetMoveDownButton()->Disable();
310             }
311         }
312         private void DeleteOperation()
313         {
314             if (selectedIndex != -1 && selectedOperation != null)
315             {
316                 GetListBox()->DeleteItem(selectedIndex);
317                 operationList->Remove(selectedIndex);
318                 operationNameTextBox->Clear();
319                 abstractCheckBox->SetChecked(false);
320                 operationNameTextBox->SetFocus();
321                 selectedIndex = -1;
322                 SetDefaultButton(GetAddButton());
323                 GetDeleteButton()->Disable();
324                 GetMoveUpButton()->Disable();
325                 GetMoveDownButton()->Disable();
326             }
327         }
328         private void MoveUpOperation()
329         {
330             if (selectedIndex > 0)
331             {
332                 GetListBox()->DeleteItem(selectedIndex);
333                 operationList->MoveUp(selectedIndex);
334                 --selectedIndex;
335                 selectedOperation = operationList->Get(selectedIndex);
336                 string operationName = selectedOperation->Name();
337                 bool isAbstract = selectedOperation->IsAbstract();
338                 string listBoxItemString = operationName;
339                 if (isAbstract)
340                 {
341                     listBoxItemString.Append(", abstract=true");
342                 }
343                 GetListBox()->InsertItem(selectedIndex, listBoxItemString);
344                 GetListBox()->SetSelectedIndex(selectedIndex);
345                 SelectedIndexChanged();
346             }
347         }
348         private void MoveDownOperation()
349         {
350             if (selectedIndex < operationList->Count() - 1)
351             {
352                 GetListBox()->DeleteItem(selectedIndex);
353                 operationList->MoveDown(selectedIndex);
354                 ++selectedIndex;
355                 selectedOperation = operationList->Get(selectedIndex);
356                 string operationName = selectedOperation->Name();
357                 bool isAbstract = selectedOperation->IsAbstract();
358                 string listBoxItemString = operationName;
359                 if (isAbstract)
360                 {
361                     listBoxItemString.Append(", abstract=true");
362                 }
363                 if (selectedIndex < operationList->Count())
364                 {
365                     GetListBox()->InsertItem(selectedIndex, listBoxItemString);
366                 }
367                 else
368                 {
369                     GetListBox()->AddItem(listBoxItemString);
370                 }
371                 GetListBox()->SetSelectedIndex(selectedIndex);
372                 SelectedIndexChanged();
373             }
374         }
