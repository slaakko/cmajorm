 12     public class CommandList
 13     {
 14         public nothrow CommandList() : currentIndex(0), commands()
 15         {
 16         }
 17         public nothrow void SetMenuItems(MenuItem* undoMenuItem_, MenuItem* redoMenuItem_)
 18         {
 19             undoMenuItem = undoMenuItem_;
 20             redoMenuItem = redoMenuItem_;
 21             UpdateMenuItems();
 22         }
 23         public nothrow void AddCommand(Command* command)
 24         {
 25             command->Finalize();
 26             if (currentIndex >= commands.Count())
 27             {
 28                 commands.Add(UniquePtr<Command>(command));
 29                 currentIndex = cast<int>(commands.Count());
 30             }
 31             else
 32             {
 33                 commands[currentIndex].Reset(command);
 34                 ++currentIndex;
 35             }
 36             commands.Resize(currentIndex);
 37             UpdateMenuItems();
 38         }
 39         public nothrow bool CanUndo() const
 40         {
 41             return currentIndex > 0;
 42         }
 43         public nothrow void Undo()
 44         {
 45             if (CanUndo())
 46             {
 47                 --currentIndex;
 48                 Command* command = commands[currentIndex].Get();
 49                 command->Undo();
 50             }
 51             UpdateMenuItems();
 52         }
 53         public nothrow bool CanRedo() const
 54         {
 55             return currentIndex < commands.Count();
 56         }
 57         public nothrow void Redo()
 58         {
 59             if (CanRedo())
 60             {
 61                 Command* command = commands[currentIndex].Get();
 62                 command->Redo();
 63                 ++currentIndex;
 64             }
 65             UpdateMenuItems();
 66         }
 67         private void UpdateMenuItems()
 68         {
 69             if (CanUndo())
 70             {
 71                 undoMenuItem->Enable();
 72             }
 73             else
 74             {
 75                 undoMenuItem->Disable();
 76             }
 77             if (CanRedo())
 78             {
 79                 redoMenuItem->Enable();
 80             }
 81             else
 82             {
 83                 redoMenuItem->Disable();
 84             }
 85         }
 86         private int currentIndex;
 87         private List<UniquePtr<Command>> commands;
 88         private MenuItem* undoMenuItem;
 89         private MenuItem* redoMenuItem;
 90     }

 92     public abstract class Command
 93     {
 94         public nothrow Command(Diagram& diagram_) : diagram(diagram_)
 95         {
 96         }
 97         public virtual default ~Command();
 98         public abstract nothrow void Finalize();
 99         public abstract nothrow void Undo();
100         public abstract nothrow void Redo();
101         public inline nothrow Diagram& GetDiagram()
102         {
103             return diagram;
104         }
105         private Diagram& diagram;
106     }

134     {
135         public inline nothrow bool operator()(const ElementIndexPair& left, const ElementIndexPair& right) const

149     public class DeleteElementsCommand : Command
150     {
151         public nothrow DeleteElementsCommand(Diagram& diagram) : base(diagram)
152         {
153         }
154         public nothrow void AddDeletedElement(DiagramElement* element, int index)
155         {
156             deletedElements.Add(ElementIndexPair(element, index));
157         }
158         public override nothrow void Finalize()
159         {
160             Sort(deletedElements.Begin(), deletedElements.End(), DecreasingElementIndexPairOrder());
161         }
162         public override nothrow void Undo()
163         {
164             for (ElementIndexPair& deleted : deletedElements)
165             {
166                 deleted.Get()->ResetSelected();
167                 GetDiagram().InsertElement(deleted.Release(), deleted.Index());
168             }
169         }
170         public override nothrow void Redo()
171         {
172             for (ElementIndexPair& deleted : deletedElements)
173             {
174                 deleted.Set(GetDiagram().RemoveElementByIndex(deleted.Index()));
175             }
176         }
177         private List<ElementIndexPair> deletedElements;
178     }

180     public class AddElementsCommand : Command
181     {
182         public nothrow AddElementsCommand(Diagram& diagram) : base(diagram)
183         {
184         }
185         public nothrow void AddIndex(int index)
186         {
187             indeces.Add(index);
188         }
189         public override nothrow void Finalize()
190         {
191             Sort(indeces.Begin(), indeces.End(), DecreasingIndexOrder());
192         }
193         public override nothrow void Undo()
194         {
195             for (int index : indeces)
196             {
197                 UniquePtr<DiagramElement> element = GetDiagram().RemoveElementByIndex(index);
198                 addedElements.Add(ElementIndexPair(element.Release(), index));
199             }
200         }
201         public override nothrow void Redo()
202         {
203             for (ElementIndexPair& addedElement : addedElements)
204             {
205                 addedElement.Get()->ResetSelected();
206                 GetDiagram().InsertElement(addedElement.Release(), addedElement.Index());
207             }
208             addedElements.Clear();
209         }
210         private List<int> indeces;
211         private List<ElementIndexPair> addedElements;
212     }

214     public class ReplaceElementCommand : Command
215     {
216         public nothrow ReplaceElementCommand(Diagram& diagram, int elementIndex_) : base(diagram), elementIndex(elementIndex_)
217         {
218             element.Reset(diagram.ReleaseElementByIndex(elementIndex));
219         }
220         public override nothrow void Finalize()
221         {
222         }
223         public override nothrow void Undo()
224         {
225             UniquePtr<DiagramElement> newElement(GetDiagram().ReleaseElementByIndex(elementIndex));
226             if (element.Get() is ContainerElement*)
227             {
228                 ContainerElement* containerElement = cast<ContainerElement*>(element.Get());
229                 HashMap<DiagramElement*, DiagramElement*> cloneMap;
230                 cloneMap[newElement.Get()] = containerElement;
231                 containerElement->MapChildObjects(cast<ContainerElement*>(newElement.Get()), cloneMap);
232             }
233             element.Get()->ResetSelected();
234             GetDiagram().SetElementByIndex(element.Release(), elementIndex);
235             element = Rvalue(newElement);
236         }
237         public override nothrow void Redo()
238         {
239             UniquePtr<DiagramElement> oldElement(GetDiagram().ReleaseElementByIndex(elementIndex));
240             if (element.Get() is ContainerElement*)
241             {
242                 ContainerElement* containerElement = cast<ContainerElement*>(element.Get());
243                 HashMap<DiagramElement*, DiagramElement*> cloneMap;
244                 cloneMap[oldElement.Get()] = containerElement;
245                 containerElement->MapChildObjects(cast<ContainerElement*>(oldElement.Get()), cloneMap);
246             }
247             element.Get()->ResetSelected();
248             GetDiagram().SetElementByIndex(element.Release(), elementIndex);
249             element = Rvalue(oldElement);
250         }
251         private int elementIndex;
252         private UniquePtr<DiagramElement> element;
253     }

255     public class MoveCommand : Command
256     {
257         public nothrow MoveCommand(Diagram& diagram) : base(diagram)
258         {
259         }
260         public override nothrow void Finalize()
261         {
262         }
263         public nothrow void AddOldLocation(int index, const CompoundLocation& oldLocation)
264         {
265             indeces.Add(index);
266             oldLocations.Add(oldLocation);
267         }
268         public nothrow void AddNewLocation(const CompoundLocation& newLocation)
269         {
270             newLocations.Add(newLocation);
271         }
272         public override nothrow void Undo()
273         {
274             int n = cast<int>(indeces.Count());
275             for (int i = 0; i < n; ++i;)
276             {
277                 int index = indeces[i];
278                 DiagramElement* element = GetDiagram().GetElementByIndex(index);
279                 element->SetCompoundLocation(oldLocations[i]);
280             }
281         }
282         public override nothrow void Redo()
283         {
284             int n = cast<int>(indeces.Count());
285             for (int i = 0; i < n; ++i;)
286             {
287                 int index = indeces[i];
288                 DiagramElement* element = GetDiagram().GetElementByIndex(index);
289                 element->SetCompoundLocation(newLocations[i]);
290             }
291             newLocations.Clear();
292         }
293         private List<int> indeces;
294         private List<CompoundLocation> oldLocations;
295         private List<CompoundLocation> newLocations;
296     }
