 494         public override void Read(System.Dom.Element* parentElement, Layout& layout)
 495         {
 496             ustring en = ToUtf32(elementName);
 497             UniquePtr<XPathObject> result = Evaluate(en, parentElement);
 498             if (!result.IsNull())
 499             {
 500                 if (result->Type() == XPathObjectType.nodeSet)
 501                 {
 502                     XPathNodeSet* nodeSet = cast<XPathNodeSet*>(result.Get());
 503                     int n = nodeSet->Length();
 504                     if (n == 1)
 505                     {
 506                         System.Dom.Node* node = (*nodeSet)[0];
 507                         if (node->GetNodeType() == System.Dom.NodeType.elementNode)
 508                         {
 509                             System.Dom.Element* paddingElement = cast<System.Dom.Element*>(node);
 510                             left = ParseFloat(ToUtf8(paddingElement->GetAttribute(u"left")));
 511                             top = ParseFloat(ToUtf8(paddingElement->GetAttribute(u"top")));
 512                             right = ParseFloat(ToUtf8(paddingElement->GetAttribute(u"right")));
 513                             bottom = ParseFloat(ToUtf8(paddingElement->GetAttribute(u"bottom")));
 514                         }
 515                     }
 516                 }
 517             }
 518         }

 520         {
