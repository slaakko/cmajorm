323                     }

326                         args.diagram->DisposeContextMenu();

384         public override void CreateOperation(MouseArgs& args)
385         {
386             if (args.diagram == null) return;
387             ResizeHandle* resizeHandle = null;
388             bool leftButtonPressed = args.left && !args.right && !args.control && !args.shift;
389             if (leftButtonPressed && args.diagram->GetSelection() is RectangleSelection*)
390             {
391                 RectangleSelection* selection = cast<RectangleSelection*>(args.diagram->GetSelection());
392                 PointF location = args.location;
393                 resizeHandle = selection->GetResizeHandle(location);
394             }
395             if (resizeHandle != null)
396             {
397                 UniquePtr<Operation> operation(new ResizeRectangleSelectionOperation(args.diagram, args.location, resizeHandle));
398                 args.diagram->SetOperation(operation.Release());
399             }
400             else
401             {
402                 UniquePtr<Operation> operation(new MoveContainerElementsOperation(args.diagram, args.location));
403                 if (args.diagram->GetSelection()->Count() <= 1)
404                 {
405                     if (args.diagram->GetSelection()->Count() == 1)
406                     {
407                         args.elementIndex = args.diagram->GetSelection()->GetFirstIndex();
408                         if (args.elementIndex != -1)
409                         {
410                             args.element = args.diagram->GetElementByIndex(args.elementIndex);
411                         }
412                         else
413                         {
414                             args.element = null;
415                         }
416                     }
417                     if (args.element != null)
418                     {
419                         if (args.element is ContainerElement*)
420                         {
421                             if (args.element->Contains(args.location))
422                             {
423                                 operation->AddElementIndex(args.elementIndex);
424                                 ContainerElement* containerElement = cast<ContainerElement*>(args.element);
425                                 if (containerElement != null)
426                                 {
427                                     List<RelationshipElement*> relationships = containerElement->GetAllRelationships();
428                                     for (RelationshipElement* relationship : relationships)
429                                     {
430                                         DiagramElement* sourceElement = relationship->source.element;
431                                         ContainerElement* sourceContainerElement = null;
432                                         if (sourceElement != null)
433                                         {
434                                             sourceContainerElement = sourceElement->GetContainerElement();
435                                         }
436                                         DiagramElement* targetElement = relationship->target.element;
437                                         ContainerElement* targetContainerElement = null;
438                                         if (targetElement != null)
439                                         {
440                                             targetContainerElement = targetElement->GetContainerElement();
441                                         }
442                                         if (sourceContainerElement == containerElement && targetContainerElement == containerElement)
443                                         {
444                                             int relationshipIndex = args.diagram->GetIndexOfElement(relationship);
445                                             if (relationshipIndex != -1)
446                                             {
447                                                 operation->AddElementIndex(relationshipIndex);
448                                             }
449                                         }
450                                         else if (targetElement == null && relationship->kind == RelationshipElement.kind.reference && relationship->cardinality == Cardinality.zero)
451                                         {
452                                             int relationshipIndex = args.diagram->GetIndexOfElement(relationship);
453                                             if (relationshipIndex != -1)
454                                             {
455                                                 operation->AddElementIndex(relationshipIndex);
456                                             }
457                                         }
458                                     }
459                                 }
460                             }
461                         }
462                     }
463                     else
464                     {
465                         operation.Reset(new CreateRectangleSelectionOperation(args.diagram, args.location));
466                         args.diagram->SetOperation(operation.Release());
467                         return;
468                     }
469                 }
470                 else
471                 {
472                     if (args.diagram->GetSelection()->GetContainerElementCount() > 0)
473                     {
474                         if (args.diagram->GetSelection()->Contains(args.location))
475                         {
476                             List<int> elementIndeces = args.diagram->GetSelection()->GetContainerElementIndeces();
477                             List<int> relationshipIndeces;
478                             for (int index : elementIndeces)
479                             {
480                                 DiagramElement* element = args.diagram->GetElementByIndex(index);
481                                 if (element is ContainerElement*)
482                                 {
483                                     ContainerElement* containerElement = cast<ContainerElement*>(element);
484                                     List<RelationshipElement*> containerElementRelationships = containerElement->GetAllRelationships();
485                                     for (RelationshipElement* relationship : containerElementRelationships)
486                                     {
487                                         bool sourceInSelection = false;
488                                         DiagramElement* sourceElement = relationship->source.element;
489                                         if (sourceElement != null)
490                                         {
491                                             ContainerElement* sourceContainerElement = sourceElement->GetContainerElement();
492                                             int sourceContainerElementIndex = args.diagram->GetIndexOfElement(sourceContainerElement);
493                                             if (sourceContainerElementIndex != -1)
494                                             {
495                                                 if (Find(elementIndeces.CBegin(), elementIndeces.CEnd(), sourceContainerElementIndex) != elementIndeces.CEnd())
496                                                 {
497                                                     sourceInSelection = true;
498                                                 }
499                                             }
500                                         }
501                                         bool targetInSelection = false;
502                                         DiagramElement* targetElement = relationship->target.element;
503                                         if (targetElement != null)
504                                         {
505                                             ContainerElement* targetContainerElement = targetElement->GetContainerElement();
506                                             int targetContainerElementIndex = args.diagram->GetIndexOfElement(targetContainerElement);
507                                             if (targetContainerElementIndex != -1)
508                                             {
509                                                 if (Find(elementIndeces.CBegin(), elementIndeces.CEnd(), targetContainerElementIndex) != elementIndeces.CEnd())
510                                                 {
511                                                     targetInSelection = true;
512                                                 }
513                                             }
514                                         }
515                                         else if (relationship->kind == RelationshipElement.Kind.reference && relationship->cardinality == Cardinality.zero)
516                                         {
517                                             targetInSelection = true;
518                                         }
519                                         if (sourceInSelection && targetInSelection)
520                                         {
521                                             int relationshipIndex = args.diagram->GetIndexOfElement(relationship);
522                                             if (relationshipIndex != -1)
523                                             {
524                                                 if (Find(relationshipIndeces.CBegin(), relationshipIndeces.CEnd(), relationshipIndex) == relationshipIndeces.CEnd())
525                                                 {
526                                                     relationshipIndeces.Add(relationshipIndex);
527                                                 }
528                                             }
529                                         }
530                                     }
531                                 }
532                             }
533                             for (int relationshipIndex : relationshipIndeces)
534                             {
535                                 elementIndeces.Add(relationshipIndex);
536                             }
537                             operation->SetElementIndeces(Rvalue(elementIndeces));
538                         }
539                     }
540                 }
541                 operation->FetchOriginalLocations();
542                 if (!operation->IsEmpty())
543                 {
544                     args.diagram->SetOperation(operation.Release());
545                 }
546             }
547         }

670     public abstract class RelationshipTool : SimpleClickTool
671     {
672         public override void Initialize(Diagram& diagram)
673         {
674             diagram.ResetSelection();
675             diagram.ResetOperation();
676         }
677         public virtual nothrow bool Accept(DiagramElement* element, DiagramElement* sourceElement, EndpointType endpointType)
678         {
679             return false;
680         }
681         public abstract nothrow RelationshipElement.Kind GetRelationshipKind() const;
682         public override void MouseNormalLeftClick(MouseArgs& args)
683         {
684             args.diagram->ResetSelection();
685             DiagramElement* element = args.element;
686             if (element == null)
687             {
688                 Operation* currentOperation = args.diagram->GetOperation();
689                 if (currentOperation is AddRelationshipOperation*)
690                 {
691                     AddRelationshipOperation* operation = cast<AddRelationshipOperation*>(currentOperation);
692                     RelationshipElement& relationshipElement = operation->GetRelationshipElement();
693                     relationshipElement.intermediatePoints.Add(args.location);
694                 }
695                 return;
696             }
697             DiagramElement* sourceElement = null;
698             EndpointType endpointType = EndpointType.source;
699             Operation* currentOperation = args.diagram->GetOperation();
700             if (currentOperation is AddRelationshipOperation*)
701             {
702                 AddRelationshipOperation* operation = cast<AddRelationshipOperation*>(currentOperation);
703                 RelationshipElement& relationshipElement = operation->GetRelationshipElement();
704                 sourceElement = relationshipElement.source.element;
705                 if (relationshipElement.target.element == null)
706                 {
707                     endpointType = EndpointType.target;
708                 }
709             }
710             if (!Accept(element, sourceElement, endpointType)) return;
711             List<Endpoint> endpoints = args.element->GetEndpoints(endpointType, this);
712             if (!endpoints.IsEmpty())
713             {
714                 List<Endpoint*> endpointPtrs;
715                 for (const Endpoint& endpoint : endpoints)
716                 {
717                     endpointPtrs.Add(&endpoint);
718                 }
719                 Sort(endpointPtrs.Begin(), endpointPtrs.End(), EndpointNearer(args.location));
720                 Endpoint* nearestEndpoint = endpointPtrs.Front();
721                 LogView* logView = Application.GetLogView();
722                 if (logView != null)
723                 {
724                     logView->WriteLine(nearestEndpoint->ToString());
725                 }
726                 if (currentOperation is EmptyOperation*)
727                 {
728                     RelationshipElement.Kind relationshipKind = GetRelationshipKind();
729                     UniquePtr<RelationshipElement> relationshipElement(new RelationshipElement(relationshipKind));
730                     if (endpointType == EndpointType.source)
731                     {
732                         if (nearestEndpoint->element != null)
733                         {
734                             nearestEndpoint->element->AddRelationship(relationshipElement.Get());
735                         }
736                     }
737                     relationshipElement->source = *nearestEndpoint;
738                     UniquePtr<AddRelationshipOperation> operation(new AddRelationshipOperation(args.diagram, relationshipElement.Release()));
739                     args.diagram->SetOperation(operation.Release());
740                 }
741                 else if (currentOperation is AddRelationshipOperation*)
742                 {
743                     AddRelationshipOperation* operation = cast<AddRelationshipOperation*>(currentOperation);
744                     RelationshipElement& relationshipElement = operation->GetRelationshipElement();
745                     if (relationshipElement.source.element == element && relationshipElement.intermediatePoints.IsEmpty())
746                     {
747                         relationshipElement.source = *nearestEndpoint;
748                     }
749                     else if (relationshipElement.target.element == null)
750                     {
751                         if (args.control)
752                         {
753                             PointF lastPoint = relationshipElement.LastPoint();
754                             PointF nextPoint = nearestEndpoint->point;
755                             Line line(lastPoint, nextPoint);
756                             Vector v(line.ToVector());
757                             switch (MainDirection(v))
758                             {
759                                 case 0:
760                                 case 180:
761                                 {
762                                     lastPoint.y = nextPoint.y;
763                                     break;
764                                 }
765                                 case 90:
766                                 case 270:
767                                 {
768                                     lastPoint.x = nextPoint.x;
769                                     break;
770                                 }
771                             }
772                             relationshipElement.SetLastPoint(lastPoint);
773                         }
774                         relationshipElement.target = *nearestEndpoint;
775                         nearestEndpoint->element->AddRelationship(&relationshipElement);
776                         args.diagram->CommitOperation(args);
777                         Tools.SetCurrent(Tools.GetSelectTool());
778                     }
779                     else
780                     {
781                         #assert(false);
782                     }
783                 }
784             }
785         }
786         public override void MouseControlLeftClick(MouseArgs& args)
787         {
788             args.diagram->ResetSelection();

795                 PointF lastPoint = relationshipElement.LastPoint();
796                 PointF nextPoint = args.location;

799                 switch (MainDirection(v))
800                 {

806                     }
