242     private void OpenClick()
243     {
244         try
245         {
246             if (canvas->GetDiagram().Modified())
247             {
248                 MessageBoxResult result = MessageBox.Show("Current diagram is modified. Do you want to save changes?", "Diagram modified", this, MessageBoxType.MB_YESNOCANCEL);
249                 if (result == MessageBoxResult.yes)
250                 {
251                     SaveClick();
252                 }
253                 else if (result == MessageBoxResult.cancel)
254                 {
255                     return;
256                 }
257             }
258             List<Pair<string, string>> descriptionFilterPairs;
259             descriptionFilterPairs.Add(Pair<string, string>("Spring diagram files (*.spring.xml)", "*.spring.xml"));
260             string initialDirectory = DefaultDiagramDirectory();
261             string filePath;
262             string currentDirectory;
263             List<string> fileNames;
264             string defaultFilePath;
265             bool selected = System.Windows.API.GetOpenFileName(Handle(), descriptionFilterPairs, initialDirectory, defaultFilePath, ".spring.xml", 
266                 OFN_PATHMUSTEXIST | OFN_FILEMUSTEXIST, currentDirectory, fileNames);
267             if (selected)
268             {
269                 if (!fileNames.IsEmpty())
270                 {
271                     filePath = Path.Combine(currentDirectory, fileNames[0]);
272                 }
273                 if (!filePath.IsEmpty() && File.Exists(filePath))
274                 {
275                     UniquePtr<Diagram> diagram(new Diagram());
276                     diagram->Load(filePath);
277                     diagramFilePath = filePath;
278                     filePathStatusBarItem->SetText(diagramFilePath);
279                     modifiedStatusBarItem->SetText(" ");
280                     canvas->SetDiagram(diagram.Release());
281                     canvas->GetDiagram().ResetModified();
282                 }
283                 else
284                 {
285                     throw Exception("file path is empty or does not exist");
286                 }
287             }
288         }
289         catch (const Exception& ex)
290         {
291             MessageBox.Show(ex.Message());
292         }
293     }

333     private void SaveAsClick()
334     {
335         try
336         {
337             List<Pair<string, string>> descriptionFilterPairs;
338             descriptionFilterPairs.Add(Pair<string, string>("Spring diagram files (*.spring.xml)", "*.spring.xml"));
339             string initialDirectory = DefaultDiagramDirectory();
340             string currentDirectory;
341             List<string> fileNames;
342             string defaultFilePath;
343             string filePath;
344             bool selected = System.Windows.API.GetSaveFileName(Handle(), descriptionFilterPairs, initialDirectory, defaultFilePath, ".spring.xml", 0u, currentDirectory, filePath);
345             if (selected)
346             {
347                 if (File.Exists(filePath))
348                 {
349                     if (MessageBox.Show("File '" + filePath + "' already exists. Do you want to replace it?", "File Exists", this, MessageBoxType.MB_YESNO) != MessageBoxResult.yes)
350                     {
351                         return;
352                     }
353                 }
354                 canvas->GetDiagram().Save(filePath);
355                 diagramFilePath = filePath;
356                 filePathStatusBarItem->SetText(diagramFilePath);
357                 canvas->GetDiagram().ResetModified();
358             }
359         }
360         catch (const Exception& ex)
361         {
362             MessageBox.Show(ex.Message());
363         }

419     protected override void OnKeyDown(KeyEventArgs& args)
420     {
421         base->OnKeyDown(args);
422         switch (args.key)
423         {
424             case cast<Keys>(Keys.controlModifier | Keys.insert):
425             {
426                 if (copyMenuItem->IsEnabled())
427                 {
428                     EditCopy();
429                 }
430                 args.handled = true;
431                 break;
432             }
433             case cast<Keys>(Keys.shiftModifier | Keys.insert):
434             {
435                 if (pasteMenuItem->IsEnabled())
436                 {
437                     EditPaste();
438                 }
439                 args.handled = true;
440                 break;
441             }
442             case cast<Keys>(Keys.shiftModifier | Keys.delete_):
443             {
444                 if (cutMenuItem->IsEnabled())
445                 {
446                     EditCut();
447                 }
448                 args.handled = true;
449                 break;
450             }
451         }
452         if (!args.handled)
453         {
454             canvas->KeyDown(args);
455         }
