  6 using System;
  7 using System.Windows;
  8 
  9 namespace spring
 10 {
 11     public class ObjectPropertiesDialog : Window
 12     {
 13         public ObjectPropertiesDialog(ObjectElement* objectElement_) : 
 14             base("spring.objectPropertiesDialog", DefaultWindowClassStyle(), DialogWindowStyle(), DefaultExtendedWindowStyle(), 
 15             DefaultControlBackgroundColor(), "Object Properties", Point.Default(), Size(ScreenMetrics.Get().MMToHorizontalPixels(100), ScreenMetrics.Get().MMToVerticalPixels(70)), 
 16             Dock.none, Anchors.none, WindowState.normal), objectElement(objectElement_)
 17         {
 18             int column1Width = ScreenMetrics.Get().MMToHorizontalPixels(20);
 19             Size s = GetSize();
 20             Size defaultButtonSize = ScreenMetrics.Get().DefaultButtonSize();
 21             Size defaultControlSpacing = ScreenMetrics.Get().DefaultControlSpacing();
 22             Size defaultLabelSize = ScreenMetrics.Get().DefaultLabelSize();
 23             Size defaultTextBoxSize = ScreenMetrics.Get().DefaultTextBoxSize();
 24             int column2Width = ScreenMetrics.Get().MMToHorizontalPixels(30);
 25             int textBoxPadding = DefaultControlPadding();
 26             int borderWidth = DefaultSingleBorderWidth();
 27             int line1Top = defaultControlSpacing.h + textBoxPadding + borderWidth;
 28             Point objectNameLabelLoc(defaultControlSpacing.w, line1Top);
 29             UniquePtr<Label> objectNameLabel(new Label("Object name:", objectNameLabelLoc, defaultLabelSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left), true));
 30             int line2Top = objectNameLabel->Location().y + objectNameLabel->GetSize().h + defaultControlSpacing.h;
 31             AddChild(objectNameLabel.Release());
 32             Point objectNameTextBoxLoc(defaultControlSpacing.w * 2 + column1Width, line1Top);
 33             Size objectNameTextBoxSize(column2Width - 2 * (textBoxPadding + borderWidth), defaultTextBoxSize.h);
 34             Rect objectNameRect(objectNameTextBoxLoc, objectNameTextBoxSize);
 35             Rect paddedObjectNameRect = objectNameRect;
 36             paddedObjectNameRect.Inflate(textBoxPadding, textBoxPadding);
 37             UniquePtr<TextBox> objectNameTextBoxPtr(new TextBox(objectElement->Name()));
 38             objectNameTextBox = objectNameTextBoxPtr.Get();
 39             objectNameTextBox->SelectAll();
 40             objectNameTextBox->TextChangedEvent().AddHandler(ObjectNameChanged);
 41             UniquePtr<Control> paddedObjectNameTextBox(new PaddedControl(objectNameTextBoxPtr.Release()));
 42             Rect borderedPaddedObjectNameTextBoxRect = paddedObjectNameRect;
 43             borderedPaddedObjectNameTextBoxRect.Inflate(borderWidth, borderWidth);
 44             UniquePtr<Control> borderedPaddedObjectNameTextBox(new BorderedControl(paddedObjectNameTextBox.Release(), borderedPaddedObjectNameTextBoxRect.location, borderedPaddedObjectNameTextBoxRect.size, 
 45                 Dock.none, cast<Anchors>(Anchors.top | Anchors.left)));
 46             AddChild(borderedPaddedObjectNameTextBox.Release());
 47             Point abstractCheckBoxLoc(borderedPaddedObjectNameTextBoxRect.location.x, line2Top);
 48             Point fieldsLabelLoc(defaultControlSpacing.w, line2Top);
 49             UniquePtr<Label> fieldsLabel(new Label("Fields:", fieldsLabelLoc, defaultLabelSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left), true));
 50             Size fieldsLabelSize = fieldsLabel->GetSize();
 51             AddChild(fieldsLabel.Release());
 52             Point editFieldsButtonLoc(defaultControlSpacing.w + column1Width + defaultControlSpacing.w - (textBoxPadding + borderWidth), line2Top - (textBoxPadding + borderWidth));
 53             Size editFieldsButtonSize(ScreenMetrics.Get().MMToHorizontalPixels(8), defaultButtonSize.h);
 54             UniquePtr<Button> editFieldsButtonPtr(new Button("...", editFieldsButtonLoc, editFieldsButtonSize, Dock.none, cast<Anchors>(Anchors.top | Anchors.left)));
 55             editFieldsButton = editFieldsButtonPtr.Get();
 56             editFieldsButton->ClickEvent().AddHandler(EditFields);
 57             AddChild(editFieldsButtonPtr.Release());
 58             int x = s.w - defaultButtonSize.w - defaultControlSpacing.w;
 59             int y = s.h - defaultButtonSize.h - defaultControlSpacing.h;
 60             UniquePtr<Button> cancelButtonPtr(new Button("Cancel", Point(x, y), defaultButtonSize, Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
 61             cancelButton = cancelButtonPtr.Get();
 62             cancelButton->SetDialogResult(DialogResult.cancel);
 63             x = x - defaultButtonSize.w - defaultControlSpacing.w;
 64             UniquePtr<Button> okButtonPtr(new Button(ButtonStyle.BS_DEFPUSHBUTTON, GetSystemColor(SystemColor.COLOR_BTNFACE), "OK", Point(x, y), defaultButtonSize, 
 65                 Dock.none, cast<Anchors>(Anchors.bottom | Anchors.right)));
 66             okButton = okButtonPtr.Get();
 67             okButton->SetDialogResult(DialogResult.ok);
 68             if (objectElement->Name().IsEmpty())
 69             {
 70                 okButton->Disable();
 71                 editFieldsButton->Disable();
 72             }
 73             AddChild(okButtonPtr.Release());
 74             AddChild(cancelButtonPtr.Release());
 75             SetDefaultButton(okButton);
 76             SetCancelButton(cancelButton);
 77         }
 78         private void ObjectNameChanged()
 79         {
 80             if (objectNameTextBox->Text().IsEmpty())
 81             {
 82                 okButton->Disable();
 83                 editFieldsButton->Disable();
 84             }
 85             else
 86             {
 87                 okButton->Enable();
 88                 editFieldsButton->Enable();
 89                 objectElement->SetName(objectNameTextBox->Text());
 90             }
 91         }
 92         private void EditFields()
 93         {
 94             IndexList<FieldElement> clonedFieldElements = objectElement->Fields().Clone();
 95             if (ExecuteEditFieldsDialog(clonedFieldElements, objectElement, *this))
 96             {
 97                 objectElement->SetFields(clonedFieldElements);
 98             }
 99         }
100         private ObjectElement* objectElement;
101         private Button* okButton;
102         private Button* cancelButton;
103         private Button* editFieldsButton;
104         private TextBox* objectNameTextBox;
105     }
106 }
