 6 using Cm.Ast;
 7 using ParserFileTokens;
 8 
 9 parser TypeExprParser
10 {
11     uselexer ParserFileLexer;
12     
13     using BasicTypeParser.BasicType;
14     using ExpressionParser.Expression;
15     using TemplateParser.TemplateId;
16 
17     TypeExpr(ParsingContext* ctx) : Node*
18         ::= PrefixTypeExpr(ctx):prefixTypeExpr{ return prefixTypeExpr; }
19         ;
20 
21     PrefixTypeExpr(ParsingContext* ctx, var Span s) : Node*
22         ::= CONST{ s = span; } PostfixTypeExpr(ctx):constTypeExpr{ s.end = span.end; return new ConstNode(s, constTypeExpr); }
23         |   PostfixTypeExpr(ctx):postfixTypeExpr{ return postfixTypeExpr; }
24         ;
25 
26     PostfixTypeExpr(ParsingContext* ctx, var UniquePtr<Node> typeExpr, var Span s) : Node*
27         ::=
28         (   PrimaryTypeExpr(ctx):primaryTypeExpr{ s = span; typeExpr.Reset(primaryTypeExpr); }
29             (   DOT ID!{ s.end = span.end; typeExpr.Reset(new DotNode(s, typeExpr.Release(), new IdentifierNode(span, lexer.GetToken(pos)))); }
30             |   STAR{ s.end = span.end; typeExpr.Reset(new PointerNode(s, typeExpr.Release())); }
31             |   AMPAMP{ s.end = span.end; typeExpr.Reset(new RValueRefNode(s, typeExpr.Release())); }
32             |   AMP{ s.end = span.end; typeExpr.Reset(new LValueRefNode(s, typeExpr.Release())); }
33             |   LBRACKET Expression(ctx):size? RBRACKET!{ s.end = span.end; typeExpr.Reset(new ArrayNode(s, typeExpr.Release(), size)); }
34             )*
35         )
36         {
37             return typeExpr.Release();
38         }
39         ;
40 
41     PrimaryTypeExpr(ParsingContext* ctx) : Node*
42         ::= BasicType:basicType{ return basicType; }
43         |   TemplateId(ctx):templateId{ return templateId; }
44         |   ID{ return new IdentifierNode(span, lexer.GetToken(pos)); }
45         ;
46 
47     ruleinfo
48     {
49         (TypeExpr, "type expression"), (PrefixTypeExpr, "prefix type expression"), (PostfixTypeExpr, "postfix type expression"), (PrimaryTypeExpr, "primary type expression")
50     }
51 }
