633             if (!found)
634             {
635                 if (diffs.IsEmpty())
636                 {
637                     difference.Add(left);
638                 }
639                 else

641                     for (const Range& diff : diffs)

643                         difference.Add(diff);
644                     }
645                 }
646             }
647         }
648         Class* d = lexerContext.MakeClass();
649         for (const Range& r : difference)
