121     internal class NodeStringComputer : Visitor
122     {
123         public nothrow NodeStringComputer(Node* node)
124         {
125             if (node == null)
126             {
127                 nodeString = "null";
128             }
129             else
130             {
131                 node->Accept(*this);
132                 if (nodeString.IsEmpty())
133                 {
134                     nodeString = typename(*node);
135                 }
136             }
137         }
138         public override void Visit(SimpleTypeNode& simpleTypeNode)
139         {
140             nodeString.Append(ToUtf8(SimpleTypeSpecifierStr(simpleTypeNode.simpleTypeSpecifiers)));
141         }
142         public override void Visit(IdentifierNode& identifierNode)
143         {
144             nodeString.Append(ToUtf8(identifierNode.str));
145         }
146         public override void Visit(ColonColonNode& colonColonNode)
147         {
148             if (!colonColonNode.left.IsNull())
149             {
150                 colonColonNode.left->Accept(*this);
151             }
152             nodeString.Append("::");
153             colonColonNode.right->Accept(*this);
154         }
155         public override void Visit(DotNode& dotNode)
156         {
157             dotNode.left->Accept(*this);
158             nodeString.Append('.');
159             dotNode.right->Accept(*this);
160         }
161         public override void Visit(ArrowNode& arrowNode)
162         {
163             arrowNode.left->Accept(*this);
164             nodeString.Append("->");
165             arrowNode.right->Accept(*this);
166         }
167         public override void Visit(TemplateIdNode& templateIdNode)
168         {
169             templateIdNode.id->Accept(*this);
170             nodeString.Append('<');
171             bool first = true;
172             for (const UniquePtr<Node>& node : templateIdNode.nodes)
173             {
174                 if (first)
175                 {
176                     first = false;
177                 }
178                 else
179                 {
180                     nodeString.Append(", ");
181                 }
182                 node->Accept(*this);
183             }
184             nodeString.Append('>');
185         }
186         public override void Visit(ConstNode& constNode)
187         {
188             nodeString.Append("const ");
189             constNode.child->Accept(*this);
190         }
191         public override void Visit(VolatileNode& volatileNode)
192         {
193             nodeString.Append("volatile ");
194             volatileNode.child->Accept(*this);
195         }
196         public override void Visit(PointerNode& pointerNode)
197         {
198             pointerNode.child->Accept(*this);
199             nodeString.Append('*');
200         }
201         public override void Visit(RvalueRefNode& rvalueRefNode)
202         {
203             rvalueRefNode.child->Accept(*this);
204             nodeString.Append("&&");
205         }
206         public override void Visit(LvalueRefNode& lvalueRefNode)
207         {
208             lvalueRefNode.child->Accept(*this);
209             nodeString.Append('&');
210         }
211         public override void Visit(ArrayTypeNode& arrayTypeNode)
212         {
213             arrayTypeNode.left->Accept(*this);
214             nodeString.Append('[');
215             if (!arrayTypeNode.right.IsNull())
216             {
217                 arrayTypeNode.right->Accept(*this);
218             }
219             nodeString.Append(']');
220         }
221         public override void Visit(FunctionPtrIdNode& functionPtrIdNode)
222         {
223             nodeString.Append('*');
224             functionPtrIdNode.child->Accept(*this);
225         }
226         public override void Visit(MemberFunctionPtrIdNode& memberFunctionPtrIdNode)
227         {
228             memberFunctionPtrIdNode.left->Accept(*this);
229             nodeString.Append('*');
230             memberFunctionPtrIdNode.right->Accept(*this);
231         }
232         public override void Visit(DestructorIdNode& destructorIdNode)
233         {
234             nodeString.Append('~');
235             destructorIdNode.child->Accept(*this);
236         }
237         public string nodeString;
238     }
