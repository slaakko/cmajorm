 13     public class TypeResolver : Visitor
 14     {
 15         public nothrow TypeResolver(SymbolTable& symbolTable_, Scope* scope_, const Span& span_, List<UniquePtr<Exception>>& errors_) : 
 16             symbolTable(symbolTable_), scope(scope_), span(span_), errors(errors_)
 17         {
 18         }
 19         public override void Visit(SimpleTypeNode& simpleTypeNode)
 20         {
 21             type = symbolTable.MakeSimpleTypeSymbol(simpleTypeNode, simpleTypeNode.span);
 22         }
 23         public override void Visit(ConstNode& constNode)
 24         {
 25             TypeSymbol* childType = ResolveType(constNode.child.Get(), symbolTable, scope, span, errors);
 26             type = symbolTable.MakeConstTypeSymbol(childType, span);
 27         }
 28         public override void Visit(VolatileNode& volatileNode)
 29         {
 30             TypeSymbol* childType = ResolveType(volatileNode.child.Get(), symbolTable, scope, span, errors);
 31             type = symbolTable.MakeVolatileTypeSymbol(childType, span);
 32         }
 33         public override void Visit(LvalueRefNode& lvalueRefNode)
 34         {
 35             TypeSymbol* childType = ResolveType(lvalueRefNode.child.Get(), symbolTable, scope, span, errors);
 36             type = symbolTable.MakeLvalueRefTypeSymbol(childType, span);
 37         }
 38         public override void Visit(RvalueRefNode& rvalueRefNode)
 39         {
 40             TypeSymbol* childType = ResolveType(rvalueRefNode.child.Get(), symbolTable, scope, span, errors);
 41             type = symbolTable.MakeRvalueRefTypeSymbol(childType, span);
 42         }
 43         public override void Visit(PointerNode& pointerNode)
 44         {
 45             TypeSymbol* childType = ResolveType(pointerNode.child.Get(), symbolTable, scope, span, errors);
 46             type = symbolTable.MakePointerTypeSymbol(childType, span);
 47         }
 48         public override void Visit(IdentifierNode& identifierNode)
 49         {
 50             if (scope == null)
 51             {
 52                 type = null;
 53             }
 54             else
 55             {
 56                 Symbol* s = scope->Lookup(identifierNode.str, ScopeLookup.this_and_bases_and_parent, span, errors);
 57                 if (s != null)
 58                 {
 59                     ResolveSymbol(s);
 60                 }
 61                 else
 62                 {
 63                     type = null;
 64                 }
 65             }
 66         }
 67         public override void Visit(ColonColonNode& colonColonNode)
 68         {
 69             if (colonColonNode.left.IsNull())
 70             {
 71                 scope = &symbolTable.globalNs.scope;
 72             }
 73             else
 74             {
 75                 colonColonNode.left->Accept(*this);
 76             }
 77             colonColonNode.right->Accept(*this);
 78         }
 79         private void ResolveSymbol(Symbol* s)
 80         {
 81             if (s is TypeSymbol*)
 82             {
 83                 type = cast<TypeSymbol*>(s);
 84                 scope = &type->scope;
 85             }
 86             else if (s is NamespaceSymbol*)
 87             {
 88                 NamespaceSymbol* ns = cast<NamespaceSymbol*>(s);
 89                 type = null;
 90                 scope = &ns->scope;
 91             }
 92             else if (s is TypeAliasSymbol*)
 93             {
 94                 TypeAliasSymbol* alias = cast<TypeAliasSymbol*>(s);
 95                 type = alias->type;
 96                 scope = &type->scope;
 97             }
 98             else if (s is TypedefSymbol*)
 99             {
100                 TypedefSymbol* typedefSymbol = cast<TypedefSymbol*>(s);
101                 type = typedefSymbol->type;
102                 scope = &type->scope;
103             }
104             else
105             {
106                 type = null;
107                 scope = null;
108             }
109         }

122     public TypeSymbol* ResolveType(Node* typeExpr, SymbolTable& symbolTable, Scope* scope, const Span& span, TypeResolvingFlags flags, List<UniquePtr<Exception>>& errors)
123     {
124         TypeResolver resolver(symbolTable, scope, span, errors);
125         typeExpr->Accept(resolver);
126         if (resolver.type != null)
127         {
128             if ((flags & TypeResolvingFlags.leaveClassGroup) != TypeResolvingFlags.none)
129             {
130                 return resolver.type;
131             }
132             else
133             {
134                 if (resolver.type is ClassGroupTypeSymbol*)
135                 {
136                     ClassGroupTypeSymbol* cg = cast<ClassGroupTypeSymbol*>(resolver.type);
137                     for (const UniquePtr<Symbol>& symbol : cg->symbols)
138                     {
139                         if (symbol.Get() is ClassTypeSymbol*)
140                         {
141                             ClassTypeSymbol* cls = cast<ClassTypeSymbol*>(symbol.Get());
142                             if (cls->Arity() == 0)
143                             {
144                                 return cls;
145                             }
146                         }
147                     }
148                 }
149                 else
150                 {
151                     return resolver.type;
152                 }
153             }
154         }
155         TypeSymbol* externalType = symbolTable.MakeExternalTypeSymbol(ToUtf32(MakeNodeString(typeExpr)), span);
156         return externalType;
157     }
