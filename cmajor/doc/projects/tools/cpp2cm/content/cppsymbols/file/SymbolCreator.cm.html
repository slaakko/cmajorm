 59         public override void Visit(IdentifierNode& identifierNode)
 60         {
 61             if (HasSimpleDeclarationNodeParent(&identifierNode) || HasMemberDeclarationNodeParent(&identifierNode))
 62             {
 63                 if (currentType == null)
 64                 {
 65                     errors.Add(UniquePtr<Exception>(new SymbolException("internal error: current type is null", identifierNode.span)));
 66                     return;
 67                 }
 68                 if (symbolTable.container is NamespaceSymbol*)
 69                 {
 70                     Symbol* s = symbolTable.container->scope.Lookup(identifierNode.str, identifierNode.span, errors);
 71                     if (s != null)
 72                     {
 73                         if (!(s is GlobalVariableSymbol*))
 74                         {
 75                             errors.Add(UniquePtr<Exception>(new SymbolException("global variable symbol expected", identifierNode.span)));;
 76                             return;
 77                         }
 78                     }
 79                     else
 80                     {
 81                         GlobalVariableSymbol* globalVariable = new GlobalVariableSymbol(identifierNode.span, identifierNode.str, currentType);
 82                         symbolTable.container->AddSymbol(globalVariable, errors);
 83                     }
 84                 }
 85                 else if (symbolTable.container is ClassTypeSymbol*)
 86                 {
 87                     Symbol* s = symbolTable.container->scope.Lookup(identifierNode.str, identifierNode.span, errors);
 88                     if (s != null)
 89                     {
 90                         if (!(s is MemberVariableSymbol*))
 91                         {
 92                             errors.Add(UniquePtr<Exception>(new SymbolException("member variable symbol expected", identifierNode.span)));
 93                             return;
 94                         }
 95                     }
 96                     else
 97                     {
 98                         MemberVariableSymbol* memberVariable = new MemberVariableSymbol(identifierNode.span, identifierNode.str, currentType);
 99                         symbolTable.container->AddSymbol(memberVariable, errors);
100                     }
101                 }
102                 else if (symbolTable.container is BlockSymbol*)
103                 {
104                     Symbol* s = symbolTable.container->scope.Lookup(identifierNode.str, identifierNode.span, errors);
105                     if (s != null)
106                     {
107                         if (s is LocalVariableSymbol*)
108                         {
109                             errors.Add(UniquePtr<Exception>(new SymbolException("local variable symbol of the same name already declared in the current scope", identifierNode.span)));
110                             return;
111                         }
112                         else
113                         {
114                             errors.Add(UniquePtr<Exception>(new SymbolException("conflicting symbol '" + ToUtf8(s->FullName()) + "'", identifierNode.span)));
115                             return;
116                         }
117                     }
118                     else
119                     {
120                         LocalVariableSymbol* localVariable = new LocalVariableSymbol(identifierNode.span, identifierNode.str, currentType);
121                         symbolTable.container->AddSymbol(localVariable, errors);
122                     }
123                 }
124                 else
125                 {
126                     errors.Add(UniquePtr<Exception>(new SymbolException("variable can be declared only in namespace, class or block scope", identifierNode.span)));
127                     return;
128                 }
129             }

301         public override void Visit(FunctionNode& functionNode)
302         {
303             functionNode.declarator->Accept(*this);
304             if (symbolTable.container is FunctionSymbol*)
305             {
306                 FunctionSymbol* prevFunctionSymbol = currentFunctionSymbol;
307                 FunctionSymbol* functionSymbol = cast<FunctionSymbol*>(symbolTable.container);
308                 currentFunctionSymbol = functionSymbol;
309                 if (functionSymbol->body != null)
310                 {
311                     errors.Add(UniquePtr<Exception>(new SymbolException("function '" + ToUtf8(functionSymbol->FullName()) + "' already has a body", functionNode.span)));
312                     currentFunctionSymbol = prevFunctionSymbol;
313                     symbolTable.EndFunction();
314                     if (symbolTable.container is FunctionGroupSymbol*)
315                     {
316                         symbolTable.EndFunctionGroup();
317                     }
318                     return;
319                 }
320                 TypeSymbol* returnType = ResolveType(functionNode.returnTypeExpr.Get(), symbolTable, &functionSymbol->scope, functionNode.span, errors);
321                 if (functionSymbol->returnType == null)
322                 {
323                     functionSymbol->returnType = returnType;
324                 }
325                 else if (functionSymbol->returnType != returnType)
326                 {
327                     errors.Add(UniquePtr<Exception>(
328                         new SymbolException("conflicting return type for function '" + ToUtf8(functionSymbol->FullName()) + "'. In declaration '" + ToUtf8(functionSymbol->returnType->FullName()) + 
329                         "', in definition '" + ToUtf8(returnType->FullName()) + "'.", functionNode.span)));
330                     currentFunctionSymbol = prevFunctionSymbol;
331                     symbolTable.EndFunction();
332                     if (symbolTable.container is FunctionGroupSymbol*)
333                     {
334                         symbolTable.EndFunctionGroup();
335                     }
336                     return;
337                 }
338                 if (!functionNode.body.IsNull())
339                 {
340                     functionNode.body->Accept(*this);
341                 }
342                 else
343                 {
344                     // todo default or delete
345                 }
346                 currentFunctionSymbol = prevFunctionSymbol;
347                 symbolTable.EndFunction();
348                 if (symbolTable.container is FunctionGroupSymbol*)
349                 {
350                     symbolTable.EndFunctionGroup();
351                 }
352             }
353             else
354             {
355                 errors.Add(UniquePtr<Exception>(new SymbolException("function declarator not supported", functionNode.span)));
356                 return;
357             }

559                 symbolTable.BeginClassGroup(classNameNode, errors);
560                 symbolTable.BeginClass(forwardClassDeclarationNode.classKey, forwardClassDeclarationNode.span, 0, errors);
561                 symbolTable.EndClass();
562                 symbolTable.EndClassGroup();
563             }
564             else
565             {
566                 errors.Add(UniquePtr<Exception>(new SymbolException("class declarator '" + MakeNodeString(classNameNode) + "' not supported", classNameNode->span)));
567                 return;
568             }
569         }
570         public override void Visit(ClassNode& classNode)
571         {
572             Node* classNameNode = classNode.className.Get();
573             if (classNameNode is ColonColonNode*)
574             {
575                 ColonColonNode* qualifiedId = cast<ColonColonNode*>(classNameNode);
576                 Symbol* s = symbolTable.container->scope.Lookup(qualifiedId->left.Get(), ScopeLookup.this_and_parent, classNode.span, errors);
577                 if (s != null)
578                 {
579                     if (s is ContainerSymbol*)
580                     {
581                         ContainerSymbol* c = cast<ContainerSymbol*>(s);
582                         symbolTable.BeginContainer(c);
583                         symbolTable.BeginClassGroup(qualifiedId->right.Get(), errors);
584                         symbolTable.BeginClass(classNode.classKey, classNode.span, 0, errors);
585                         if (!classNode.baseClasses.IsNull())
586                         {
587                             classNode.baseClasses->Accept(*this);
588                         }
589                         for (const UniquePtr<Node>& memberDeclaration : classNode.memberDeclarations)
590                         {
591                             memberDeclaration->Accept(*this);
592                         }
593                         symbolTable.EndClass();
594                         symbolTable.EndClassGroup();
595                         symbolTable.EndContainer();
596                     }
597                     else
598                     {
599                         errors.Add(UniquePtr<Exception>(new SymbolException("symbol '" + ToUtf8(s->FullName()) + "' does not denote a container symbol", qualifiedId->left->span)));
600                         return;
601                     }
602                 }
603                 else
604                 {
605                     errors.Add(UniquePtr<Exception>(new SymbolException("container symbol '" + MakeNodeString(qualifiedId->left.Get()) + "' not found", qualifiedId->left->span)));
606                     return;
607                 }
608             }
609             else if (classNameNode is IdentifierNode*)
610             {
611                 symbolTable.BeginClassGroup(classNameNode, errors);
612                 symbolTable.BeginClass(classNode.classKey, classNode.span, 0, errors);
613                 if (!classNode.baseClasses.IsNull())
614                 {
615                     classNode.baseClasses->Accept(*this);
616                 }
617                 for (const UniquePtr<Node>& memberDeclaration : classNode.memberDeclarations)
618                 {
619                     memberDeclaration->Accept(*this);
620                 }
621                 symbolTable.EndClass();
622                 symbolTable.EndClassGroup();
623             }
624             else
625             {
626                 errors.Add(UniquePtr<Exception>(new SymbolException("class declarator '" + MakeNodeString(classNameNode) + "' not supported", classNameNode->span)));
627                 return;
628             }
629         }

637         public override void Visit(BaseClassNode& baseClassNode)
638         {
639             Node* baseClassName = baseClassNode.baseClassName.Get();
640             TypeSymbol* baseType = ResolveType(baseClassName, symbolTable, &symbolTable.container->scope, baseClassNode.span, errors);
641             if (baseType is ClassTypeSymbol*)
642             {
643                 ClassTypeSymbol* baseClassType = cast<ClassTypeSymbol*>(baseType);
644                 if (symbolTable.container is ClassTypeSymbol*)
645                 {
646                     ClassTypeSymbol* classType = cast<ClassTypeSymbol*>(symbolTable.container);
647                     classType->AddBaseClass(baseClassType);
648                 }
649                 else
650                 {
651                     errors.Add(UniquePtr<Exception>(new SymbolException("internal error: class type expected", baseClassNode.span)));
652                     return;
653                 }
654             }
655             else if (baseType is ExternalTypeSymbol*)
656             {
657                 errors.Add(UniquePtr<Exception>(new SymbolException("type symbol '" + ToUtf8(baseType->FullName()) + "' is external", baseClassNode.span)));
658                 return;
659             }
660             else
661             {
662                 errors.Add(UniquePtr<Exception>(new SymbolException("type symbol '" + ToUtf8(baseType->FullName()) + "' does not denote a class type", baseClassNode.span)));
663                 return;
664             }
665         }
666         public override void Visit(MemberDeclarationNode& memberDeclarationNode)

670             {
671                 type = ResolveType(memberDeclarationNode.type.Get(), symbolTable, &symbolTable.container->scope, memberDeclarationNode.span, errors);

676             currentType = prevType;
677             if (symbolTable.container is FunctionSymbol*)
678             {
679                 FunctionSymbol* functionSymbol = cast<FunctionSymbol*>(symbolTable.container);
680                 functionSymbol->returnType = type;
681                 symbolTable.EndFunction();
682                 if (symbolTable.container is FunctionGroupSymbol*)
683                 {
684                     symbolTable.EndFunctionGroup();
685                 }
686             }
687         }
