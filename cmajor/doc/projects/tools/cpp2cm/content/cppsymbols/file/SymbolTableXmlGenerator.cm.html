 10         public void ToXml(SymbolTable& symbolTable)
 11         {
 12             symbolTableElement.Reset(new DomElement( u"symbolTable"));
 13             currentElement = symbolTableElement.Get();
 14             symbolTable.globalNs.Accept(*this);
 15             DomElement* prevElement = currentElement;
 16             UniquePtr<DomElement> typesElement(new DomElement( u"types"));
 17             currentElement = typesElement.Get();
 18             for (const UniquePtr<TypeSymbol>& type : symbolTable.types)
 19             {
 20                 type->Accept(*this);
 21             }
 22             currentElement = prevElement;
 23             currentElement->AppendChild(UniquePtr<DomNode>(typesElement.Release()));
 24         }

 55                 }

 58             }

 60             {

 63                 currentElement = membersElement.Get();

 66                     symbol->Accept(*this);

 69                 currentElement->AppendChild(UniquePtr<DomNode>(membersElement.Release()));

 73         }
