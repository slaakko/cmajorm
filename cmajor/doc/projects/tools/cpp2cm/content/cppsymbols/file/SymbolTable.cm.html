 32         public void BeginNamespace(const Span& span, const ustring& namespaceName, List<UniquePtr<Exception>>& errors)
 33         {
 34             Symbol* s = container->scope.Lookup(namespaceName, span, errors);
 35             if (s != null)
 36             {
 37                 if (s is ContainerSymbol*)
 38                 {
 39                     ContainerSymbol* c = cast<ContainerSymbol*>(s);
 40                     BeginContainer(c);
 41                 }
 42                 else
 43                 {
 44                     throw SymbolException(string(typename(*s)) + " symbol '" + ToUtf8(s->name) + "' does not denote a namespace", span);
 45                 }
 46             }
 47             else
 48             {
 49                 ustring nsName = namespaceName;
 50                 bool isUnnamedNs = false;
 51                 if (nsName.IsEmpty())
 52                 {
 53                     nsName =  u"ns_" + ToUtf32(ToString(Uuid.Random()));
 54                     isUnnamedNs = true;
 55                 }
 56                 NamespaceSymbol* ns = new NamespaceSymbol(span, nsName, isUnnamedNs);
 57                 container->AddSymbol(ns, errors);
 58                 BeginContainer(ns);
 59             }
 60         }

 62         {
 63             EndContainer();

 65         public void BeginFunctionGroup(Node* node, List<UniquePtr<Exception>>& errors)
 66         {
 67             ustring parentClassName;
 68             if (container is ClassTypeSymbol*)
 69             {
 70                 parentClassName = container->name;
 71             }
 72             ustring functionGroupName = MakeFunctionGroupName(node, parentClassName);
 73             Symbol* f = container->scope.Lookup(functionGroupName, node->span, errors);
 74             if (f != null)
 75             {
 76                 if (f is FunctionGroupSymbol*)
 77                 {
 78                     FunctionGroupSymbol* fg = cast<FunctionGroupSymbol*>(f);
 79                     BeginContainer(fg);
 80                 }
 81                 else
 82                 {
 83                     errors.Add(UniquePtr<Exception>(new SymbolException("symbol '" + ToUtf8(f->FullName()) + "' does not denote a function group", node->span)));
 84                     return;
 85                 }
 86             }
 87             else
 88             {
 89                 FunctionGroupSymbol* fg = new FunctionGroupSymbol(node->span, functionGroupName);
 90                 container->AddSymbol(fg, errors);
 91                 BeginContainer(fg);
 92             }
 93         }

 98         public void BeginFunction(List<UniquePtr<ParameterSymbol>>& parameterList, CVSpecifiers cvSpecifiers, const Span& span, List<UniquePtr<Exception>>& errors)
 99         {
100             FunctionGroupSymbol* fg = container as FunctionGroupSymbol*;
101             if (fg != null)
102             {
103                 for (const UniquePtr<Symbol>& symbol : fg->symbols)
104                 {
105                     if (symbol.Get() is FunctionSymbol*)
106                     {
107                         FunctionSymbol* functionSymbol = cast<FunctionSymbol*>(symbol.Get());
108                         long n = functionSymbol->Arity();
109                         if (n == parameterList.Count() && functionSymbol->cvSpecifiers == cvSpecifiers)
110                         {
111                             bool found = true;
112                             for (long i = 0; i < n; ++i;)
113                             {
114                                 TypeSymbol* paramType = parameterList[i]->type;
115                                 TypeSymbol* functionParamType = functionSymbol->parameters[i]->type;
116                                 if (paramType != functionParamType)
117                                 {
118                                     found = false;
119                                     break;
120                                 }
121                             }
122                             if (found)
123                             {
124                                 BeginContainer(functionSymbol);
125                                 return;
126                             }
127                         }
128                     }
129                 }
130                 const ustring& functionGroupName = fg->name;
131                 FunctionSymbol* functionSymbol = new FunctionSymbol(span, functionGroupName, cvSpecifiers);
132                 for (UniquePtr<ParameterSymbol>& parameterSymbol : parameterList)
133                 {
134                     functionSymbol->AddSymbol(parameterSymbol.Release(), errors);
135                 }
136                 fg->AddSymbol(functionSymbol, errors);
137                 functionSymbol->parent = fg->parent;
138                 BeginContainer(functionSymbol);
139             }
140             else
141             {
142                 errors.Add(UniquePtr<Exception>(new SymbolException("internal error : function group expected", container->span)));
143                 return;
144             }
145         }
146         public void EndFunction()
147         {
148             EndContainer();
149         }
150         public void BeginClassGroup(Node* node, List<UniquePtr<Exception>>& errors)
151         {
152             if (node is IdentifierNode*)
153             {
154                 IdentifierNode* idNode = cast<IdentifierNode*>(node);
155                 Symbol* s = container->scope.Lookup(idNode->str, node->span, errors);
156                 if (s != null)
157                 {
158                     if (s is ClassGroupTypeSymbol*)
159                     {
160                         ClassGroupTypeSymbol* cg = cast<ClassGroupTypeSymbol*>(s);
161                         BeginContainer(cg);
162                     }
163                     else
164                     {
165                         errors.Add(UniquePtr<Exception>(new SymbolException("symbol '" + ToUtf8(s->FullName()) + " does not denote a class group", node->span)));
166                         return;
167                     }
168                 }
169                 else
170                 {
171                     ClassGroupTypeSymbol* cg = new ClassGroupTypeSymbol(node->span, idNode->str);
172                     container->AddSymbol(cg, errors);
173                     BeginContainer(cg);
174                 }
175             }
176             else if (node is TemplateIdNode*)
177             {

179             }
180             else

183                 return;
184             }

190         public void BeginClass(ClassKey classKey, const Span& span, int arity, List<UniquePtr<Exception>>& errors)
191         {

209                 classTypeSymbol->parent = cg->parent;
210                 BeginContainer(classTypeSymbol);
211             }
212             else
213             {
214                 errors.Add(UniquePtr<Exception>(new SymbolException("internal error: class group expected", span)));
215             }
