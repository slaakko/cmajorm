 49         public Symbol* Lookup(Node* node, ScopeLookup lookup, const Span& span, List<UniquePtr<Exception>>& errors, bool& errorOccurred, HashSet<Scope*>& scopesLookedUp)
 50         {
 51             if (node == null)
 52             {
 53                 return containerSymbol->GlobalNs();
 54             }
 55             else if (node is IdentifierNode*)
 56             {
 57                 IdentifierNode* identifierNode = cast<IdentifierNode*>(node);
 58                 return Lookup(identifierNode->str, lookup, span, errors, errorOccurred, scopesLookedUp);
 59             }
 60             else if (node is ColonColonNode*)
 61             {
 62                 ColonColonNode* qualifiedId = cast<ColonColonNode*>(node);
 63                 if ((lookup & ScopeLookup.parent) != ScopeLookup.none)
 64                 {
 65                     Symbol* s = Lookup(qualifiedId->left.Get(), lookup, span, errors, errorOccurred, scopesLookedUp);
 66                     if (errorOccurred) return null;
 67                     if (s != null)
 68                     {
 69                         if (s is ContainerSymbol*)
 70                         {
 71                             ContainerSymbol* c = cast<ContainerSymbol*>(s);
 72                             Symbol* n = c->scope.Lookup(qualifiedId->right.Get(), cast<ScopeLookup>(lookup & ~ScopeLookup.parent), span, errors, errorOccurred, scopesLookedUp);
 73                             if (errorOccurred) return null;
 74                             if (n != null)
 75                             {
 76                                 return n;
 77                             }
 78                             else if (node->parent != null)
 79                             {
 80                                 return Lookup(node->parent, lookup, span, errors, errorOccurred, scopesLookedUp);
 81                             }
 82                             else
 83                             {
 84                                 return null;
 85                             }
 86                         }
 87                         else if (node->parent != null)
 88                         {
 89                             return Lookup(node->parent, lookup, span, errors, errorOccurred, scopesLookedUp);
 90                         }
 91                         else
 92                         {
 93                             return null;
 94                         }
 95                     }
 96                     else if (node->parent != null)
 97                     {
 98                         return Lookup(node->parent, lookup, span, errors, errorOccurred, scopesLookedUp);
 99                     }
100                     else
101                     {
102                         return null;
103                     }
104                 }
105                 else if ((lookup & ScopeLookup.this_) != ScopeLookup.none)
106                 {
107                     Symbol* s = Lookup(qualifiedId->left.Get(), lookup, span, errors, errorOccurred, scopesLookedUp);
108                     if (errorOccurred) return null;

115                         }
116                         else

119                         }

152         public Symbol* Lookup(const ustring& name, ScopeLookup lookup, const Span& span, List<UniquePtr<Exception>>& errors, bool& errorOccurred, HashSet<Scope*>& scopesLookedUp)
153         {
154             if ((lookup & ScopeLookup.this_) != ScopeLookup.none)
155             {
156                 HashMap<ustring, Symbol*>.ConstIterator it = symbolMap.CFind(name);
157                 if (it != symbolMap.CEnd())
158                 {
159                     return it->second;
160                 }
161             }
162             if ((lookup & ScopeLookup.bases) != ScopeLookup.none)
163             {
164                 long n = containerSymbol->NumBases();
165                 for (long i = 0; i < n; ++i;)
166                 {
167                     ContainerSymbol* baseClass = containerSymbol->GetBase(i);
168                     Symbol* s = baseClass->scope.Lookup(name, lookup, span, errors, errorOccurred, scopesLookedUp);
169                     if (errorOccurred) return null;
170                     if (s != null)
171                     {
172                         return s;
173                     }
174                 }
175             }
176             if ((lookup & ScopeLookup.parent) != ScopeLookup.none)
177             {
178                 if (containerSymbol->parent != null)
179                 {
180                     Symbol* s = containerSymbol->parent->scope.Lookup(name, lookup, span, errors, errorOccurred, scopesLookedUp);
181                     if (errorOccurred) return null;
182                     if (s != null)
183                     {
184                         return s;
185                     }
186                 }
187             }
188             if (scopesLookedUp.CFind(this) != scopesLookedUp.CEnd()) return null;
189             scopesLookedUp.Insert(this);
190             HashSet<Symbol*> matched;
191             for (NamespaceImportSymbol* namespaceImport : namespaceImports)
192             {
193                 Symbol* s = namespaceImport->ns->scope.Lookup(name, ScopeLookup.this_, span, errors, errorOccurred, scopesLookedUp);
194                 if (errorOccurred) return null;
195                 if (s != null)
196                 {
197                     matched.Insert(s);
198                 }
199             }
200             if (matched.IsEmpty())
201             {
202                 return null;
203             }
204             else if (matched.Count() > 1)
205             {
206                 string matchedString;
207                 bool first = true;
208                 for (Symbol* s : matched)
209                 {
210                     if (first)
211                     {
212                         first = false;
213                     }
214                     else
215                     {
216                         matchedString.Append(" or ");
217                     }
218                     matchedString.Append(ToUtf8(s->FullName()));
219                 }
220                 errors.Add(UniquePtr<Exception>(new SymbolException("reference to symbol '" + ToUtf8(name) + "' is ambiguous: " + matchedString, span)));
221                 errorOccurred = true;
222                 return null;
223             }
224             else
225             {
226                 return *matched.CBegin();
227             }
228         }
