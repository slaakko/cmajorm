 46         for (int i = 1; i < argc; ++i;)
 47         {
 48             string arg = argv[i];

 50             {

 71                     if (components.Count() == 2)
 72                     {
 73                         if (components[0] == "--cpp" || components[0] == "-c")
 74                         {
 75                             xmlFilePath = components[1];
 76                         }

 78                         {
 79                             outputFilePath = components[1];
 80                         }
 81                         else if (components[0] == "--include" || components[0] == "-I")
 82                         {
 83                             List<string> directories = components[1].Split(';');
 84                             for (const string& includeDir : directories)
 85                             {
 86                                 includeDirectories.Add(includeDir);
 87                             }
 88                         }
 89                         else
 90                         {
 91                             throw Exception("unknown option '" + arg + "'");
 92                         }
 93                     }
 94                     else
 95                     {
 96                         throw Exception("unknown option '" + arg + "'");
 97                     }
 98                 }
 99                 else
100                 {
101                     throw Exception("unknown option '" + arg + "'");
102                 }
103             }
104             else
105             {
106                 sourceFilePaths.Add(arg);
107             }
108         }
109         if (sourceFilePaths.IsEmpty())

113         }

125         }

128         {

130         }
131         for (const string& sourceFilePath : sourceFilePaths)
132         {
133             if (Flags.Verbose())

136             }

138             for (const PPFile& ppFile : ppFiles)
139             {
140                 if (Flags.Verbose())
141                 {

143                 }

145                 out << ppFile.content << endl();
146             }
147         }
148     }
149     catch (const Exception& ex)

151         Console.Error() << ex.Message() << endl();
152         return 1;
153     }
