ExpressionGrammar Grammar

ExpressionList → (empty InitializerClause (',' InitializerClause)*)
PossiblyEmptyArgumentList → ExpressionList | empty
Expression → (AssignmentExpression (',' AssignmentExpression)*)
ConstantExpression → ConditionalExpression
AssignmentExpression → ConcreteAssignmentExpression | ConditionalExpression | ThrowExpression
ConcreteAssignmentExpression → empty (LogicalOrExpression AssignmentOperator InitializerClause)
AssignmentOperator → ("=" − "==") | "*=" | "/=" | "%=" | "+=" | "-=" | ">>=" | "<<=" | "&=" | "^=" | "|="
ConditionalExpression → (LogicalOrExpression ('?' Expression ':' AssignmentExpression)?)
ThrowExpression → (keyword("throw") AssignmentExpression?)
LogicalOrExpression → (LogicalAndExpression ("||" LogicalAndExpression)*)
LogicalAndExpression → (InclusiveOrExpression ("&&" InclusiveOrExpression)*)
InclusiveOrExpression → (ExclusiveOrExpression (('|' − ("|=" | "||")) ExclusiveOrExpression)*)
ExclusiveOrExpression → (AndExpression (('^' − "^=") AndExpression)*)
AndExpression → (EqualityExpression (('&' − ("&=" | "&&")) EqualityExpression)*)
EqualityExpression → (RelationalExpression (("==" | "!=") RelationalExpression)*)
RelationalExpression → (ShiftExpression (("<=" | ">=" | ('<' − ("<<" | "<<=")) | ('>' − (">>" | ">>="))) ShiftExpression)*)
ShiftExpression → (AdditiveExpression ((("<<" − "<<=") | (">>" − ">>=")) (AdditiveExpression))*)
AdditiveExpression → (MultiplicativeExpression ((('+' − "+=") | ('-' − "-=")) MultiplicativeExpression)*)
MultiplicativeExpression → (PMExpression ((('*' − "*=") | ('/' − "/=") | ('%' − "%=")) PMExpression)*)
PMExpression → (CastExpression ((".*" | "->*") CastExpression)*)
CastExpression → (('(' TypeExpr ')' CastExpression) | UnaryExpression)
UnaryExpression → ("++" CastExpression) | "--" CastExpression | UnaryOperator CastExpression | (keyword("sizeof") UnaryExpression) | (keyword("alignof") '(' TypeExpr ')') | NewExpression | DeleteExpression | PostfixExpression
UnaryOperator → ('*' − "*=") | ('&' − "&=") | ('+' − "+=") | ('-' − "-=") | '!' | '~'
NewExpression → (keyword("new") NewPlacement? NewTypeId NewInitializer?)
NewPlacement → ('(' ExpressionList ')')
NewTypeId → TypeExpr (NewDeclarator)?
NewDeclarator → '[' Expression ']'
NewInitializer → ('(' PossiblyEmptyArgumentList ')')
DeleteExpression → (keyword("delete") ('[' ']')? CastExpression)
PostfixExpression → (PrimaryExpression ('[' Expression ']' | '(' PossiblyEmptyArgumentList ')' | (('.' − ".*") IdExpression) | (("->" − "->*") IdExpression) | "++" | "--")*)
PrimaryExpression → Literal | keyword("this") | empty DefinedExpr | ('(' Expression ')') | CppCastExpression | TypeIdExpression | (SimpleTypeSpecifier '(' PossiblyEmptyArgumentList ')') | (SimpleTypeSpecifier BracedInitializerList) | (TypeExpr '(' PossiblyEmptyArgumentList ')') | (TypeExpr BracedInitializerList) | IdExpression | LambdaExpression
CppCastExpression → ((keyword("dynamic_cast") | keyword("static_cast") | keyword("reinterpret_cast") | keyword("const_cast")) '<' TypeExpr '>' '(' Expression ')')
DefinedExpr → (keyword("defined") '(' identifier ')') | (keyword("defined") identifier) | (keyword("defined") '(' Literal ')') | (keyword("defined") Literal)
TypeIdExpression → (keyword("typeid") '(' TypeExpr ')') | (keyword("typeid") '(' Expression ')')
IdExpression → QualifiedIdExpr | UnqualifiedIdExpr
UnqualifiedIdExpr → TemplateId | Identifier | OperatorFunctionId | ConversionFunctionId | ('~' Identifier)
QualifiedIdExpr → ("::" TemplateId) | ("::" NestedNameSpecifier UnqualifiedIdExpr) | ("::" Identifier) | ("::" OperatorFunctionId) | (NestedNameSpecifier UnqualifiedIdExpr)
OperatorFunctionId → (keyword("operator") Operator)
Operator → (keyword("new") "[]") | (keyword("delete") "[]") | keyword("new") | keyword("delete") | AssignmentOperator | "++" | "--" | "<<" | ">>" | "&&" | "||" | "==" | "!=" | "<=" | ">=" | ".*" | "->*" | "->" | "()" | "[]" | '<' | '>' | '+' | '-' | '*' | '/' | '%' | '^' | '&' | '|' | '~' | '!' | ','
ConversionFunctionId → (keyword("operator") TypeExpr)
LambdaExpression → '[' LambdaCaptures ']'! (ParameterList)? (CompoundStatement)!
LambdaCaptures → (CaptureDefault)? Captures?
CaptureDefault → '=' | '&'
Captures → Capture (',' Capture)*
Capture → Identifier | keyword("this")
