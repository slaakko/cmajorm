 56         {

 86                 {

 89                     {
 90                         if (components[0] == "--cpp" || components[0] == "-c")
 91                         {
 92                             xmlFilePath = components[1];
 93                         }
 94                         else if (components[0] == "--out" || components[0] == "-o")
 95                         {
 96                             outputFilePath = components[1];
 97                         }
 98                         else if (components[0] == "--include" || components[0] == "-I")
 99                         {

102                             {
103                                 includeDirectories.Add(includeDir);
104                             }
105                         }
106                         else
107                         {
108                             throw Exception("unknown option '" + arg + "'");
109                         }
110                     }
111                     else
112                     {
113                         throw Exception("unknown option '" + arg + "'");
114                     }

118                     throw Exception("unknown option '" + arg + "'");
119                 }

134             cppDoc = ParseXmlFileToDomDocument(xmlFilePath);
135         }
136         DomElement* cppElement = null;

138         {
139             cppElement = cppDoc->DocumentElement();
140         }

142         if (Flags.Debug())
