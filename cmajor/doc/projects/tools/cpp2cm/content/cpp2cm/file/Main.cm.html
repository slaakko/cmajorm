26 int main(int argc, const char** argv)
27 {
28     try
29     {
30         List<string> inputXmlFilePaths;
31         for (int i = 1; i < argc; ++i;)
32         {

34             if (arg.StartsWith("-"))
35             {
36                 if (arg == "--help" || arg == "-h")
37                 {
38                     PrintHelp();

40                 }

55                     throw Exception("unknown option '" + arg + "'");
56                 }
57             }
58             else
59             {
60                 inputXmlFilePaths.Add(GetFullPath(arg));
61             }
62         }

64         {
65             string defaultInputXml = GetFullPath("cpp2cm.xml");
66             if (File.Exists(defaultInputXml))
67             {

69             }

75         }
76         for (const string& inputXmlFilePath : inputXmlFilePaths)
77         {

80                 string workXmlFilePath = MakeWorkXml(inputXmlFilePath);
81                 Process(workXmlFilePath);
82             }
83             else
84             {
85                 throw Exception("error: input file '" + inputXmlFilePath + "' does not have .xml extension");
86             }
87         }
88     }
89     catch (const Exception& ex)
90     {
91         Console.Error() << ex.ToString() << endl();
92         return 1;
93     }
94     return 0;
