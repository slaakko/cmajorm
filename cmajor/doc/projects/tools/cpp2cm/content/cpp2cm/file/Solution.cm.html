33     internal void Visit(List<Project*>& buildOrder, Project* project, HashSet<Project*>& visited, HashSet<Project*>& tempVisit, Solution* solution)
34     {
35         if (tempVisit.CFind(project) == tempVisit.CEnd())
36         {
37             if (visited.CFind(project) == visited.CEnd())
38             {
39                 tempVisit.Insert(project);
40                 for (const ustring& dependendsOnGuid : project->dependencies)
41                 {
42                     Project* dependsOnProject = solution->GetProject(dependendsOnGuid);
43                     Visit(buildOrder, dependsOnProject, visited, tempVisit, solution);
44                 }
45                 tempVisit.Remove(project);
46                 visited.Insert(project);
47                 buildOrder.Add(project);
48             }
49         }

62         public List<Project*> BuildOrder()
63         {
64             for (const UniquePtr<Project>& project : projects)
65             {
66                 guidProjectMap[project->guid] = project.Get();
67             }
68             List<Project*> buildOrder;
69             HashSet<Project*> visited;
70             HashSet<Project*> tempVisit;
71             for (const UniquePtr<Project>& project : projects)
72             {
73                 if (visited.CFind(project.Get()) == visited.CEnd())
74                 {
75                     Visit(buildOrder, project.Get(), visited, tempVisit, this);
76                 }
77             }
78             return buildOrder;
79         }
80         public Project* GetProject(const ustring& projectGuid) const
81         {

83             if (it != guidProjectMap.CEnd())
84             {
85                 return it->second;
86             }
87             else
88             {
89                 throw Exception("project id " + ToUtf8(projectGuid) + " not found");
90             }
91         }
