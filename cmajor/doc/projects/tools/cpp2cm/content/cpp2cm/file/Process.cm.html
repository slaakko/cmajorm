  9     internal string MakeDirectoryPath(DomElement* element)
 10     {
 11         DomParentNode* parentNode = element->Parent();
 12         DomElement* parentElement = parentNode as DomElement*;
 13         if (parentElement != null && parentElement->Name() !=  u"work")
 14         {
 15             string directoryName = MakeDirectoryPath(parentElement);
 16             string name = ToUtf8(parentElement->GetAttribute( u"name"));
 17             if (!name.IsEmpty())
 18             {
 19                 if (directoryName.IsEmpty())
 20                 {
 21                     return name;
 22                 }

 24                 {

 28             }

 36             return string();
 37         }

 40     internal void ParseSourceFile(const string& filePath, const string& fileName, const string& localDirectory, const string& localFilePath)
 41     {
 42         bool parsingFailed = false;
 43         string errorMessage;
 44         try
 45         {
 46             SourceFileGrammar* grammar = Parser.GetSourceFileGrammar();
 47             ustring fileContent = ToUtf32(File.ReadAllText(filePath));
 48             if (Flags.Verbose())
 49             {
 50                 Console.Out() << "> " << filePath << endl();
 51             }
 52             int fileIndex = FileRegistry.Instance().RegisterFilePath(filePath);
 53             grammar->Parse(fileContent.Chars(), fileContent.Chars() + fileContent.Length(), fileIndex, filePath);
 54         }
 55         catch (const Exception& ex)
 56         {
 57             parsingFailed = true;
 58             errorMessage = ex.Message();
 59         }
 60         if (parsingFailed)
 61         {
 62             if (Flags.Verbose())
 63             {
 64                 Console.Out() << errorMessage << endl();
 65             }
 66             Directory.CreateDirectories(localDirectory);
 67             if (Flags.Verbose())
 68             {
 69                 Console.Out() << filePath << " -> " << localFilePath << endl();
 70             }
 71             if (File.Exists(localFilePath))
 72             {
 73                 File.Remove(localFilePath);
 74             }
 75             File.Copy(filePath, localFilePath);
 76             string errorFilePath = localFilePath + ".error.txt";
 77             if (Flags.Verbose())
 78             {
 79                 Console.Out() << "=> " << errorFilePath << endl();
 80             }
 81             StreamWriter writer = File.CreateText(errorFilePath);
 82             writer << "parsing " << filePath << " failed:" << endl() << errorMessage << endl();
 83         }
