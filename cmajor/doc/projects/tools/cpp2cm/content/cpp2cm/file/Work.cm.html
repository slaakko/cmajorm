 61     internal string ToRegExPattern(const string& filePattern)
 62     {
 63         string regExPattern;
 64         for (char c : filePattern)
 65         {
 66             if (c == '.')
 67             {
 68                 regExPattern.Append("\\.");
 69             }
 70             else if (c == '*')
 71             {
 72                 regExPattern.Append(".*");
 73             }
 74             else if (c == '?')
 75             {
 76                 regExPattern.Append(".");
 77             }
 78             else
 79             {
 80                 regExPattern.Append(c);
 81             }
 82         }
 83         return regExPattern;
 84     }

 86     internal class Filter
 87     {
 88         public enum Kind { include, exclude }
 89         
 90         public Filter(Item.Kind itemKind_, Kind kind_, const string& filePattern_) : itemKind(itemKind_), kind(kind_), filePattern(filePattern_), regEx(ToRegExPattern(filePattern))
 91         {
 92         }
 93         public void ApplyTo(Item& item)
 94         {
 95             if (item.kind == itemKind)
 96             {
 97                 if (regEx.Matches(item.name))
 98                 {
 99                     if (kind == Kind.include)
100                     {
101                         item.included = true;
102                         if (Flags.Diagnostics())
103                         {
104                             Console.Out() << "include " << ItemKindStr(itemKind) << " '" << filePattern << "' applied to " << item.name << endl();
105                         }
106                         for (Item& subitem : item.items)
107                         {
108                             for (Filter& filter : filters)
109                             {
110                                 filter.ApplyTo(subitem);
111                             }
112                         }
113                     }
114                     else if (kind == Kind.exclude)
115                     {
116                         item.included = false;
117                         if (Flags.Diagnostics())
118                         {
119                             Console.Out() << "exclude " << ItemKindStr(itemKind) << " '" << filePattern << "' applied to " << item.name << endl();
120                         }
121                     }
122                 }
123             }
124         }
125         public Item.Kind itemKind;
126         public Kind kind;
127         public string filePattern;
128         public RegEx regEx;
129         public List<Filter> filters;
130     }

137     internal List<string> GetFiles(const string& filePath)
138     {
139         List<string> files;
140         if (ContainsWildCards(filePath))
141         {
142             List<string> allFiles = Directory.GetFiles(Path.GetDirectoryName(filePath));
143             string filePattern = Path.GetFileName(filePath);
144             RegEx regEx(ToRegExPattern(filePattern), RegExFlags.compiled);
145             for (const string& file : allFiles)
146             {
147                 if (regEx.Matches(Path.GetFileName(file)))
148                 {
149                     files.Add(file);
150                 }
151             }
152         }
153         else
154         {
155             files.Add(filePath);
156         }
157         return files;

183     internal List<Item> ReadItems(Item.Kind itemKind, const string& filePath, const string& name)
184     {
185         if (Flags.Diagnostics())
186         {
187             Console.Out() << "reading " << ItemKindStr(itemKind) << " items from " << filePath << " : " << name << endl();
188         }
189         List<Item> items;
190         if (itemKind == Item.Kind.solution)
191         {
192             Item solutionItem(Item.Kind.solution, Path.GetFileNameWithoutExtension(filePath), string());
193             ustring solutionFileContent = ToUtf32(File.ReadAllText(filePath));
194             UniquePtr<Solution> solution(SolutionParser.GetGrammar()->Parse(solutionFileContent.Chars(), solutionFileContent.Chars() + solutionFileContent.Length(), 0, filePath));
195             List<Project*> buildOrder = solution->BuildOrder();
196             for (Project* project : buildOrder)
197             {
198                 string fullProjectFilePath = GetFullPath(Path.Combine(Path.GetDirectoryName(filePath), project->filePath));
199                 List<Item> projectItems = ReadItems(Item.Kind.project, fullProjectFilePath, ToUtf8(project->name));
200                 for (Item& projectItem : projectItems)
201                 {
202                     solutionItem.items.Add(Rvalue(projectItem));
203                 }
204             }
205             items.Add(Rvalue(solutionItem));
206         }
207         else if (itemKind == Item.Kind.project)
208         {
209             UniquePtr<DomDocument> projectDoc = ParseXmlFileToDomDocument(filePath);
210             Item projectItem(Item.Kind.project, name, string());
211             UniquePtr<XPathObject> headerFilesObject = Evaluate( u"/Project/ItemGroup/ClInclude", projectDoc.Get());
212             XPathNodeSet* headerFilesNodeSet = headerFilesObject.Get() as XPathNodeSet*;
213             if (headerFilesNodeSet != null)
214             {
215                 long n = headerFilesNodeSet->Length();
216                 for (long i = 0; i < n; ++i;)
217                 {
218                     DomNode* headerFileNode = (*headerFilesNodeSet)[i];
219                     DomElement* headerFileElement = headerFileNode as DomElement*;
220                     if (headerFileElement != null)
221                     {
222                         string headerFileName = ToUtf8(headerFileElement->GetAttribute( u"Include"));
223                         string fullSourceFilePath = GetFullPath(Path.Combine(Path.GetDirectoryName(filePath), headerFileName));
224                         Item sourceItem(Item.Kind.source, Path.GetFileName(headerFileName), fullSourceFilePath);
225                         projectItem.items.Add(Rvalue(sourceItem));
226                     }
227                 }
228             }
229             UniquePtr<XPathObject> sourceFilesObject = Evaluate( u"/Project/ItemGroup/ClCompile", projectDoc.Get());
230             XPathNodeSet* sourceFilesNodeSet = sourceFilesObject.Get() as XPathNodeSet*;
231             if (sourceFilesNodeSet != null)
232             {
233                 long n = sourceFilesNodeSet->Length();
234                 for (long i = 0; i < n; ++i;)
235                 {
236                     DomNode* sourceFileNode = (*sourceFilesNodeSet)[i];
237                     DomElement* sourceFileElement = sourceFileNode as DomElement*;
238                     if (sourceFileElement != null)
239                     {
240                         string sourceFileName = ToUtf8(sourceFileElement->GetAttribute( u"Include"));
241                         string fullSourceFilePath = GetFullPath(Path.Combine(Path.GetDirectoryName(filePath), sourceFileName));
242                         Item sourceItem(Item.Kind.source, Path.GetFileName(sourceFileName), fullSourceFilePath);
243                         projectItem.items.Add(Rvalue(sourceItem));
244                     }
245                 }
246             }
247             items.Add(Rvalue(projectItem));
248         }
249         else if (itemKind == Item.Kind.source)
250         {
251             List<string> files = GetFiles(filePath);
252             for (const string& file : files)
253             {
254                 Item sourceItem(Item.Kind.source, Path.GetFileName(file), file);
255                 items.Add(Rvalue(sourceItem));
256             }
257         }
258         if (Flags.Diagnostics())
259         {
260             Console.Out() << ItemKindStr(itemKind) << " items read" << endl();
261         }
262         return items;
263     }

265     internal void CreateSubFilters(Filter& parentFilter, DomNode* parentNode)
266     {
267         Item.Kind itemKind = Item.Kind.work;
268         ustring itemType;
269         if (parentFilter.itemKind == Item.Kind.solution)
270         {
271            itemKind = Item.Kind.project;
272            itemType =  u"project";
273         }
274         else if (parentFilter.itemKind == Item.Kind.project)
275         {
276             itemKind = Item.Kind.source;
277             itemType =  u"source";
278         }
279         else
280         {
281             return;
282         }
283         UniquePtr<XPathObject> xpathObject = Evaluate(itemType, parentNode);
284         XPathNodeSet* nodeSet = xpathObject.Get() as XPathNodeSet*;
285         if (nodeSet != null)
286         {
287             long n = nodeSet->Length();
288             for (long i = 0; i < n; ++i;)
289             {
290                 DomNode* node = (*nodeSet)[i];
291                 DomElement* element = node as DomElement*;
292                 if (element != null)
293                 {
294                     string name = ToUtf8(element->GetAttribute( u"name"));
295                     string file = ToUtf8(element->GetAttribute( u"file"));
296                     string include = ToUtf8(element->GetAttribute( u"include"));
297                     string exclude = ToUtf8(element->GetAttribute( u"exclude"));
298                     if (!name.IsEmpty())
299                     {
300                         throw Exception("a filter element cannot contain other than filter elements (" + 
301                             ToUtf8(ItemKindStr(parentFilter.itemKind)) + " filter contains element '" + ToUtf8(element->Name()) + "' with 'name' attribute)");
302                     }
303                     if (!file.IsEmpty())
304                     {
305                         throw Exception("a filter element cannot contain other than filter elements (" + 
306                             ToUtf8(ItemKindStr(parentFilter.itemKind)) + " filter contains element '" + ToUtf8(element->Name()) + "' with 'file' attribute");
307                     }
308                     if (!include.IsEmpty())
309                     {
310                         if (Flags.Diagnostics())
311                         {
312                             Console.Out() << "creating include " << ItemKindStr(itemKind) << " subfilter " << include << endl();
313                         }
314                         Filter includeFilter(itemKind, Filter.Kind.include, include);
315                         parentFilter.filters.Add(includeFilter);
316                     }
317                     else if (!exclude.IsEmpty())
318                     {
319                         if (Flags.Diagnostics())
320                         {
321                             Console.Out() << "creating exclude " << ItemKindStr(itemKind) << " subfilter " << exclude << endl();
322                         }
323                         Filter excludeFilter(itemKind, Filter.Kind.exclude, exclude);
324                         parentFilter.filters.Add(excludeFilter);
325                     }
326                 }
327             }
328         }

331     internal void PopulateItems(Item& parentItem, DomNode* parentNode, Item.Kind itemKind, const string& baseDir)
332     {
333         if (Flags.Diagnostics())
334         {
335             Console.Out() << "populating " << parentItem.KindStr() << " : " << parentItem.name << " : " << parentItem.filePath << " : " << ItemKindStr(itemKind) << endl();
336         }
337         List<Filter> filters;
338         ustring itemType;
339         if (itemKind == Item.Kind.solution)
340         {
341             itemType =  u"solution";
342         }
343         else if (itemKind == Item.Kind.project)
344         {
345             itemType =  u"project";
346         }
347         else if (itemKind == Item.Kind.source)
348         {
349             itemType =  u"source";
350         }
351         UniquePtr<XPathObject> xpathObject = Evaluate(itemType, parentNode);
352         XPathNodeSet* nodeSet = xpathObject.Get() as XPathNodeSet*;
353         if (nodeSet != null)
354         {
355             long n = nodeSet->Length();
356             for (long i = 0; i < n; ++i;)
357             {
358                 DomNode* node = (*nodeSet)[i];
359                 DomElement* element = node as DomElement*;
360                 if (element != null)
361                 {
362                     int numAttributesSet = 0;
363                     string name = ToUtf8(element->GetAttribute( u"name"));
364                     string file = ToUtf8(element->GetAttribute( u"file"));
365                     string include = ToUtf8(element->GetAttribute( u"include"));
366                     string exclude = ToUtf8(element->GetAttribute( u"exclude"));
367                     if (!name.IsEmpty())
368                     {
369                         ++numAttributesSet;
370                     }
371                     if (!file.IsEmpty())
372                     {
373                         ++numAttributesSet;
374                     }
375                     if (!include.IsEmpty())
376                     {
377                         ++numAttributesSet;
378                     }
379                     if (!exclude.IsEmpty())
380                     {
381                         ++numAttributesSet;
382                     }
383                     if (numAttributesSet == 0)
384                     {
385                         throw Exception(ToUtf8(element->Name()) + " element must have one of the following attributes set: 'name', 'file', 'include', or 'exclude'");
386                     }
387                     else if (numAttributesSet > 1)
388                     {
389                         throw Exception(ToUtf8(element->Name()) + " element cannot have more than one of these attributes set: 'name', 'file', 'include', and 'exclude'");
390                     }
391                     if (!name.IsEmpty())
392                     {
393                         if (itemKind == Item.Kind.source)
394                         {
395                             throw Exception(ToUtf8(element->Name()) + " source element cannot have 'name' attribute");
396                         }
397                         Item item(itemKind, name, string());
398                         if (itemKind == Item.Kind.solution)
399                         {
400                             PopulateItems(item, node, Item.Kind.project, baseDir);
401                         }
402                         else if (itemKind == Item.Kind.project)
403                         {
404                             PopulateItems(item, node, Item.Kind.source, baseDir);
405                         }
406                         parentItem.items.Add(Rvalue(item));
407                     }
408                     else if (!include.IsEmpty())
409                     {
410                         if (Flags.Diagnostics())
411                         {
412                             Console.Out() << "creating include filter for " << ItemKindStr(itemKind) << " : " << include << endl();
413                         }
414                         Filter includeFilter(itemKind, Filter.Kind.include, include);
415                         CreateSubFilters(includeFilter, node);
416                         filters.Add(includeFilter);
417                     }
418                     else if (!exclude.IsEmpty())
419                     {
420                         if (Flags.Diagnostics())
421                         {
422                             Console.Out() << "creating exclude filter for " << ItemKindStr(itemKind) << " : " << exclude << endl();
423                         }
424                         Filter excludeFilter(itemKind, Filter.Kind.exclude, exclude);
425                         CreateSubFilters(excludeFilter, node);
426                         filters.Add(excludeFilter);
427                     }
428                     else if (!file.IsEmpty())
429                     {
430                         string name = Path.GetFileName(file);
431                         if (itemKind == Item.Kind.solution || itemKind == Item.Kind.project)
432                         {
433                             name = Path.GetFileNameWithoutExtension(file);
434                         }
435                         string filePath = GetFullPath(Path.Combine(baseDir, file));
436                         List<Item> items = ReadItems(itemKind, filePath, name);
437                         for (Item& item : items)
438                         {
439                             if (itemKind == Item.Kind.solution)
440                             {
441                                 PopulateItems(item, node, Item.Kind.project, Path.GetDirectoryName(filePath));
442                             }
443                             else if (itemKind == Item.Kind.project)
444                             {
445                                 PopulateItems(item, node, Item.Kind.source, Path.GetDirectoryName(filePath));
446                             }
447                             parentItem.items.Add(Rvalue(item));
448                         }
449                     }
450                 }
451             }
452             if (Flags.Diagnostics())
453             {
454                 Console.Out() << "applying filters" << endl();
455             }
456             ApplyFilters(parentItem.items, filters);
457         }

460     internal string MakeWorkXml(const string& inputXmlFilePath)
461     {
462         if (Flags.Verbose())
463         {
464             Console.Out() << "> " << inputXmlFilePath << endl();
465         }
466         string workXmlFilePath = Path.ChangeExtension(inputXmlFilePath, ".work.xml");
467         bool make = Flags.Force();
468         if (!make)
469         {
470             if (!File.Exists(workXmlFilePath))
471             {
472                 make = true;
473             }
474             else
475             {
476                 if (File.LastWriteTimeLess(workXmlFilePath, inputXmlFilePath))
477                 {
478                     make = true;
479                 }
480             }
481         }
482         if (!make)
483         {
484             if (Flags.Verbose())
485             {
486                 Console.Out() << workXmlFilePath << " is up-to-date" << endl();
487             }
488             return workXmlFilePath;
489         }
490         string inputBaseDir = Path.GetDirectoryName(inputXmlFilePath);
491         UniquePtr<DomDocument> inputXmlDoc = ParseXmlFileToDomDocument(inputXmlFilePath);
492         string targetDir = ToUtf8(inputXmlDoc->DocumentElement()->GetAttribute( u"targetDir"));
493         if (targetDir.IsEmpty())
494         {
495             targetDir = inputBaseDir;
496         }
497         Item workItem(Item.Kind.work, string(), targetDir);
498         PopulateItems(workItem, inputXmlDoc->DocumentElement(), Item.Kind.solution, inputBaseDir);
499         if (workItem.items.IsEmpty())
500         {
501             PopulateItems(workItem, inputXmlDoc->DocumentElement(), Item.Kind.project, inputBaseDir);
502             if (workItem.items.IsEmpty())
503             {
504                 PopulateItems(workItem, inputXmlDoc->DocumentElement(), Item.Kind.source, inputBaseDir);
505             }
506         }
507         DomDocument workXmlDoc;
508         workXmlDoc.AppendChild(UniquePtr<DomNode>(workItem.ToDomElement().Release()));
509         StreamWriter writer = File.CreateText(workXmlFilePath);
510         CodeFormatter formatter(writer);
511         formatter.SetIndentSize(1);
512         workXmlDoc.Write(formatter);
513         if (Flags.Verbose())
514         {
515             Console.Out() << "==> " << workXmlFilePath << endl();
516         }
517         return workXmlFilePath;
518     }
