 61     public class Evaluator : Visitor
 62     {
 63         public nothrow Evaluator(PP& pp_, int line_) : pp(pp_), line(line_)
 64         {
 65         }
 66         public override void Visit(IdentifierNode& identifierNode)
 67         {
 68             throw PreprocessingException("preprocessing symbol '" + ToUtf8(identifierNode.str) + "' at line " + ToString(line) + " not defined");
 69         }

 73         }

 81         }

 90         public override void Visit(BinaryExpressionNode& binaryExpressionNode)
 91         {
 92             int result = 0;
 93             UniquePtr<Value> left = Evaluate(pp, binaryExpressionNode.left.Get(), line);
 94             int leftInt = left->ToInteger();
 95             if (binaryExpressionNode.op == Operator.logicalAnd && leftInt == 0)
 96             {
 97                 result = 0;
 98             }
 99             else if (binaryExpressionNode.op == Operator.logicalOr && leftInt != 0)
100             {
101                 result = 1;
102             }
103             else
104             {
105                 UniquePtr<Value> right = Evaluate(pp, binaryExpressionNode.right.Get(), line);
106                 int rightInt = right->ToInteger();
107                 switch (binaryExpressionNode.op)
108                 {
109                     case Operator.shiftLeft: result = leftInt << rightInt; break;
110                     case Operator.shiftRight: result = leftInt >> rightInt; break;
111                     case Operator.logicalAnd: result = cast<int>(leftInt != 0 && rightInt != 0); break;
112                     case Operator.logicalOr: result =  cast<int>(leftInt != 0 || rightInt != 0); break;
113                     case Operator.equal: result = cast<int>(leftInt == rightInt); break;
114                     case Operator.notEqual: result = cast<int>(leftInt != rightInt); break;
115                     case Operator.lessOrEqual: result = cast<int>(leftInt <= rightInt); break;
116                     case Operator.greaterOrEqual: result = cast<int>(leftInt >= rightInt); break;
117                     case Operator.less: result = cast<int>(leftInt < rightInt); break;
118                     case Operator.greater: result = cast<int>(leftInt > rightInt); break;
119                     case Operator.add: result = leftInt + rightInt; break;
120                     case Operator.sub: result = leftInt - rightInt; break;
121                     case Operator.mul: result = leftInt * rightInt; break;
122                     case Operator.div: result = leftInt / rightInt; break;
123                     case Operator.rem: result = leftInt % rightInt; break;
124                     case Operator.xor: result = leftInt ^ rightInt; break;
125                     case Operator.and_: result = leftInt & rightInt; break;
126                     case Operator.or_: result = leftInt | rightInt; break;
127                 }
128             }
129             value.Reset(new IntegerValue(result));
130         }
131         public override void Visit(UnaryExpressionNode& unaryExpressionNode)
132         {
133             UniquePtr<Value> operandValue = Evaluate(pp, unaryExpressionNode.child.Get(), line);
134             int intValue = operandValue->ToInteger();
135             int result = 0;
136             switch (unaryExpressionNode.op)
137             {
138                 case Operator.neg: result = ~intValue; break;
139                 case Operator.not_: result = cast<int>(!(intValue != 0)); break;
140                 case Operator.unaryMinus: result = -intValue; break;
141             }
142             value.Reset(new IntegerValue(result));
143         }
144         public override void Visit(ConditionalExpressionNode& conditionalExpressionNode)
145         {
146             UniquePtr<Value> exprValue = Evaluate(pp, conditionalExpressionNode.expr.Get(), line);
147             int intValue = exprValue->ToInteger();
148             if (intValue != 0)
149             {
150                 value = Evaluate(pp, conditionalExpressionNode.thenExpr.Get(), line);
151             }
152             else
153             {
154                 value = Evaluate(pp, conditionalExpressionNode.elseExpr.Get(), line);
155             }
156         }
157         public override void Visit(DefinedNode& definedNode)
158         {
159             bool defined = pp.Defined(definedNode.identifier);
160             if (defined)
161             {
162                 value.Reset(new IntegerValue(1));
163             }
164             else
165             {
166                 value.Reset(new IntegerValue(0));
167             }
168         }

174     public UniquePtr<Value> Evaluate(PP& pp, Node* expression, int line)
175     {
176         Evaluator evaluator(pp, line);
177         if (expression != null)
178         {
179             if (Flags.Diagnostics())
180             {
181                 Console.Out() << "> evaluating " << typename(*expression) << " node" << endl();
182             }
183             expression->Accept(evaluator);
184         }
185         if (evaluator.value.Get() != null)
186         {
187             return Rvalue(evaluator.value);
188         }
189         else
190         {
191             throw PreprocessingException("cannot evaluate constant expression at line " + ToString(line));
192         }
193     }
