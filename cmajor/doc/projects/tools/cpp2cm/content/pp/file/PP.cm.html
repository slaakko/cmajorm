343         }

345         int state = 0;
346         int numLinesCombined = 0;
347         for (uchar c : sourceFileContent)
348         {
349             switch (state)
350             {
351                 case 0:
352                 {

354                     {
355                         state = 1;
356                     }
357                     else if (c == '\n')
358                     {
359                         for (int i = 0; i < numLinesCombined; ++i;)

364                         numLinesCombined = 0;
365                     }

367                     {
368                         result.Append(c);
369                     }
370                     break;
371                 }
372                 case 1:
373                 {
374                     if (c == '\n')
375                     {
376                         ++numLinesCombined;
377                         state = 0;
378                     }
379                     else if (c != '\r')
380                     {
381                         result.Append('\\').Append(c);
382                         state = 0;
383                     }
384                     break;
385                 }
386             }
387         }
388         if (Flags.Diagnostics())

603     public List<PPFile> PreprocessUsingCl(const string& sourceFilePath, DomElement* ppDefs, const List<string>& includeDirectories)
604     {
605         List<PPFile> ppFiles;
606         string preprocessFilePath = Path.ChangeExtension(sourceFilePath, ".i");
607         string includeDirOptions;
608         bool first = true;
609         for (const string& includeDir : includeDirectories)
610         {
611             if (first)
612             {
613                 first = false;
614             }
615             else
616             {
617                includeDirOptions.Append(' ');
618             }
619             includeDirOptions.Append("/I\"").Append(includeDir).Append("\"");
620         }
621         string clCommand = "cl /EP /P /Fi\"" + preprocessFilePath + "\" " + includeDirOptions + " \"" + sourceFilePath + "\"";
622         int retVal = Execute(clCommand);
623         if (retVal != 0)
624         {
625             throw PreprocessingException("cl preprocessing command '" + clCommand + "' failed.");
626         }
627         PPFile ppFile(sourceFilePath);
628         string content = File.ReadAllText(preprocessFilePath);
629         if (!content.IsEmpty())
630         {
631             ppFile.content = ToUtf32(content);
632         }
633         ppFiles.Add(Rvalue(ppFile));
634         if (!Flags.LeavePPOutput())
635         {
636             File.Remove(preprocessFilePath);
637         }
638         else if (Flags.Verbose())
639         {
640             Console.Out() << "==> " << preprocessFilePath << endl();
641         }
642         return ppFiles;
643     }
