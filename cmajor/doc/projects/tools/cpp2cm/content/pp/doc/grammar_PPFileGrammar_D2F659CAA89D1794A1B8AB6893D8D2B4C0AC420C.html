PPFileGrammar Grammar

PPFile → Group?
Group → (IfSection | ControlLine | TextLine)+
IfSection → IfGroup (ElifGroup)* ElseGroup? EndIfLine
IfGroup → (S? '#' S? keyword("if") S? ConstantExpression S? newline Group?) | (S? '#' S? keyword("ifdef") S? identifier S? newline Group?) | (S? '#' S? keyword("ifndef") S? identifier S? newline Group?)
ElifGroup → (S? '#' S? keyword("elif") S? ConstantExpression S? newline Group?)
ElseGroup → S? '#' S? keyword("else") S? newline Group?
EndIfLine → S? '#' S? keyword("endif") S? newline
ControlLine → IncludeDirective | DefineFunctionMacro | DefineObjectMacro | UndefineMacro | PragmaDirective | ErrorDirective | LineDirective | NullDirective
IncludeDirective → (S? '#' S? keyword("include") S? NonemptyTokenLine newline)
DefineFunctionMacro → (S? '#' S? keyword("define") S? token(identifier '(') MacroParams ')' S? PossiblyEmptyTokenLine newline)
MacroParams → (S? identifier (S? ',' S? identifier)* S?)?
DefineObjectMacro → (S? '#' S? keyword("define") S? identifier S? PossiblyEmptyTokenLine newline)
UndefineMacro → (S? '#' S? keyword("undef") S? identifier S? newline)
PragmaDirective → S? '#' S? keyword("pragma") S? token((anychar − newline)* newline)
ErrorDirective → S? '#' S? keyword("error") S? token((anychar − newline)* newline)
LineDirective → S? '#' S? keyword("line") S? token((anychar − newline)* newline)
NullDirective → (S? '#' S? newline)
ConstantExpression → NonemptyTokenLine
TextLine → (PossiblyEmptyTokenLine newline)
PossiblyEmptyTokenLine → empty (token(PPToken*))
NonemptyTokenLine → empty (token(PPToken+))
PPToken → empty ("##" | '#') | empty HeaderName | empty FunctionMacroInvocation | empty ('@' S?)? identifier | empty identifier | PPNumber | CharacterLiteral | StringLiteral | empty ('(' | ')' | ',' | (anychar − ('#' | '(' | ')' | ',' | newline))) | empty (anychar − ('#' | newline))
FunctionMacroInvocation → (token(identifier '(') MacroArgumentList ')')
MacroArgumentList → (MacroArgument (',' MacroArgument)*)?
MacroArgument → token((PPToken)+)
PPTokenCatenation → (PPToken (S? '$' S? PPToken)*)
PPTokenSequence → PPTokenCatenation (PPTokenCatenation)*
MacroReplacement → empty ((PPTokenSequence)?)
PPNumber → token((digit | ('.' digit)) (digit | '.' | [eE] sign | [a-zA-Z_])*)
HeaderName → token('<' [^>]+ '>') | token('\"' [^\"]+ '\"')
S → [ \t]+
