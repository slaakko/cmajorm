 51     internal void GenerateCode(ParserFileContent* parserFile)
 52     {
 53         const string& filePath = parserFile->FilePath();
 54         string cmFilePath = Path.ChangeExtension(filePath, ".cm");
 55         if (!Flags.Instance().ForceCodeGen() && !Changed(filePath, cmFilePath))
 56         {
 57             return;
 58         }
 59         StreamWriter cmFile = File.CreateText(cmFilePath);
 60         CodeFormatter formatter(cmFile);
 61         formatter.WriteLine("using System;");
 62         formatter.WriteLine("using System.Collections;");
 63         formatter.WriteLine("using System.Text.Parsing;");
 64         for (const ustring& usingImport : parserFile->UsingImports())
 65         {
 66             if (usingImport ==  u"System" || usingImport == u"System.Collections" || usingImport == u"System.Text.Parsing")continue;
 67             formatter.WriteLine("using " + ToUtf8(usingImport) + ";");
 68         }
 69         for (const ustring& usingAlias : parserFile->UsingAliases())
 70         {
 71             formatter.WriteLine("using " + ToUtf8(usingAlias) + ";");
 72         }
 73         formatter.WriteLine();
 74         formatter.WriteLine("// this source file has been automatically generated from '" + Path.GetFileName(parserFile->FilePath()) + "' using cmparsergen version 2.0.0.");
 75         formatter.WriteLine();
 76         ObjectSelectorVisitor selectorVisitor(parserFile->Id());
 77         parserFile->GetParsingDomain()->GlobalNamespace()->Accept(selectorVisitor);
 78         const Map<Namespace*, NamespaceContent>& nsContentMap = selectorVisitor.NsContentMap();
 79         for (const Pair<Namespace*, NamespaceContent>& p : nsContentMap)
 80         {
 81             Namespace* ns = p.first;
 82             const NamespaceContent& nsContent = p.second;
 83             if (!ns->FullName().IsEmpty())
 84             {
 85                 formatter.WriteLine("namespace " + ToUtf8(ns->FullName()));
 86                 formatter.WriteLine("{");
 87                 formatter.IncIndent();
 88             }
 89             for (Grammar* grammar : nsContent.Grammars())
 90             {
 91                 CodeGeneratorVisitor codeGeneratorVisitor(formatter);
 92                 grammar->Accept(codeGeneratorVisitor);
 93             }
 94             if (!ns->FullName().IsEmpty())
 95             {
 96                 formatter.DecIndent();
 97                 formatter.Write("} ");
 98                 formatter.WriteLine("// " + ToUtf8(ns->FullName()));
 99             }
100         }
101         if (Flags.Instance().Verbose())
102         {
103             Console.WriteLine("=> " + cmFilePath);
104         }
105     }
