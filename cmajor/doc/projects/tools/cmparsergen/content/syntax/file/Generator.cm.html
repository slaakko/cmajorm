 87     public void ParseSourceFiles(void* data)
 88     {
 89         ParserData* parserData = cast<ParserData*>(data);
 90         ParserFileGrammar* parserFileGrammar = parserData->GetParserFileGrammar();
 91         long sourceFileIndex = -1;
 92         try
 93         {
 94             while (!parserData->Stop())
 95             {
 96                 string sourceFilePath;
 97                 {
 98                     LockGuard<Mutex> lock(parserData->GetMutex());
 99                     if (parserData->FilePos() >= parserData->NumFiles()) return;
100                     sourceFileIndex = parserData->NextFilePos();
101                     sourceFilePath = parserData->GetSourceFilePath(sourceFileIndex);
102                     if (Flags.Instance().Verbose())
103                     {
104                         Console.WriteLine("> " + sourceFilePath);
105                     }
106                 }
107                 ParsingDomain* parsingDomain = parserData->GetParsingDomain(sourceFileIndex);
108                 ustring sourceFileContent = ToUtf32(File.ReadAllText(sourceFilePath));
109                 ParsingContext parsingContext;
110                 UniquePtr<ParserFileContent> parserFileContent = parserFileGrammar->Parse(sourceFileContent.Chars(), sourceFileContent.Chars() + sourceFileContent.Length(), 
111                     cast<int>(sourceFileIndex), sourceFilePath, &parsingContext, cast<int>(sourceFileIndex), parsingDomain);
112                 parserData->SetParserFileContent(sourceFileIndex, Rvalue(parserFileContent));
113             }
114         }
115         catch (const Exception& ex)
116         {
117             if (sourceFileIndex != -1)
118             {
119                 parserData->SetExceptionPtr(sourceFileIndex, CaptureCurrentException());
120                 parserData->SetStop();
121             }
122         }
123     }

125     public void ParseSourcesConcurrently(ParserFileGrammar* parserFileGrammar, const List<string>& sourceFiles, ParsingDomain* parsingDomain, int numThreads, 
126         List<UniquePtr<ParserFileContent>>& parserFiles, List<UniquePtr<ParsingDomain>>& parsingDomains)
127     {
128         long n = sourceFiles.Count();
129         parserFiles.Resize(n);
130         List<ExceptionPtr> exceptionPtrs;
131         exceptionPtrs.Resize(n);
132         for (long i = 0; i < n; ++i;)
133         {
134             UniquePtr<ParsingDomain> pd(new ParsingDomain());
135             pd->SetOwned();
136             pd->Import(parsingDomain, true);
137             parsingDomains.Add(Rvalue(pd));
138         }
139         ParserData parserData(parserFileGrammar, sourceFiles, parserFiles, exceptionPtrs, parsingDomains);
140         List<Thread> threads;
141         for (int i = 0; i < numThreads; ++i;)
142         {
143             threads.Add(Thread.StartFunction(ParseSourceFiles, &parserData));
144             if (parserData.Stop()) break;
145         }
146         long numThreadsStarted = threads.Count();
147         for (long i = 0; i < numThreadsStarted; ++i;)
148         {
149             threads[i].Join();
150         }
151         for (long i = 0; i < n; ++i;)
152         {
153             if (exceptionPtrs[i].Exception() != null)
154             {
155                 ThrowCapturedException(exceptionPtrs[i]);
156             }
157         }
158         for (long i = 0; i < n; ++i;)
159         {
160             ParsingDomain* pd = parsingDomains[i].Get();
161             if (pd != null)
162             {
163                 parsingDomain->Import(pd, false);
164             }
165         }

168     public string ResolveReferenceFilePath(const string& relativeReferenceFilePath, const string& projectBasePath, const List<string>& libraryDirectories)
169     {
170         string referenceFilePath = Path.Combine(projectBasePath, relativeReferenceFilePath);
171         if (File.Exists(referenceFilePath))
172         {
173             return referenceFilePath;
174         }
175         long n = libraryDirectories.Count();
176         for (long i = 0; i < n; ++i;)
177         {
178             referenceFilePath = Path.Combine(libraryDirectories[i], relativeReferenceFilePath);
179             if (File.Exists(referenceFilePath))
180             {
181                 return referenceFilePath;
182             }
183         }
184         throw Exception("library file path '" + relativeReferenceFilePath + "' not found");
185     }
