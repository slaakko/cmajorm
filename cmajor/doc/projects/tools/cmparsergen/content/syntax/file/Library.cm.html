  7 namespace Syntax
  8 {
  9     public class LibraryFileGeneratorVisitor : Visitor
 10     {
 11         public LibraryFileGeneratorVisitor(CodeFormatter& formatter_) : formatter(formatter_)
 12         {
 13         }
 14         public override void BeginVisit(Grammar& grammar)
 15         {
 16             if (!grammar.IsExternal())
 17             {
 18                 List<Grammar*>& grammars = nsGrammarMap[grammar.Ns()];
 19                 grammars.Add(&grammar);
 20             }
 21         }
 22         public void GenerateLibraryFile()
 23         {
 24             HashMap<Namespace*, List<Grammar*>>.ConstIterator e = nsGrammarMap.CEnd();
 25             for (HashMap<Namespace*, List<Grammar*>>.ConstIterator i = nsGrammarMap.CBegin(); i != e; ++i;)
 26             {
 27                 Namespace* ns = i->first;
 28                 const List<Grammar*>& grammars = i->second;
 29                 if (!ns->FullName().IsEmpty())
 30                 {
 31                     formatter.WriteLine("namespace " + ToUtf8(ns->FullName()));
 32                     formatter.WriteLine("{");
 33                     formatter.IncIndent();
 34                 }
 35                 long n = grammars.Count();
 36                 for (long i = 0; i < n; ++i;)
 37                 {
 38                     Grammar* grammar = grammars[i];
 39                     formatter.WriteLine("grammar " + ToUtf8(grammar->Name()));
 40                     formatter.WriteLine("{");
 41                     formatter.IncIndent();
 42                     const List<RuleParser*>& rules = grammar->Rules();
 43                     long nr = rules.Count();
 44                     for (long i = 0; i < nr; ++i;)
 45                     {
 46                         RuleParser* rule = rules[i];
 47                         string ruleLine = ToUtf8(rule->Name());
 48                         if (!rule->InheritedAttributes().IsEmpty() || !rule->LocalVariables().IsEmpty())
 49                         {
 50                             ruleLine.Append("(");
 51                             long ni = rule->InheritedAttributes().Count();
 52                             bool first = true;
 53                             for (long i = 0; i < ni; ++i;)
 54                             {
 55                                 const AttrOrVar& attr = rule->InheritedAttributes()[i];
 56                                 if (first)
 57                                 {
 58                                     first = false;
 59                                 }
 60                                 else
 61                                 {
 62                                     ruleLine.Append(", ");
 63                                 }
 64                                 ruleLine.Append(ToUtf8(attr.Type()) + " " + ToUtf8(attr.Name()));
 65                             }
 66                             long nl = rule->LocalVariables().Count();
 67                             for (long i = 0; i < nl; ++i;)
 68                             {
 69                                 const AttrOrVar& var = rule->LocalVariables()[i];
 70                                 if (first)
 71                                 {
 72                                     first = false;
 73                                 }
 74                                 else
 75                                 {
 76                                     ruleLine.Append(", ");
 77                                 }
 78                                 ruleLine.Append("var " + ToUtf8(var.Type()) + " " + ToUtf8(var.Name()));
 79                             }
 80                             ruleLine.Append(")");
 81                         }
 82                         if (!rule->ValueTypeName().IsEmpty())
 83                         {
 84                             ruleLine.Append(" : ").Append(ToUtf8(rule->ValueTypeName()));
 85                         }
 86                         ruleLine.Append(";");
 87                         formatter.WriteLine(ruleLine);
 88                     }
 89                     formatter.DecIndent();
 90                     formatter.WriteLine("}");
 91                 }
 92                 if (!ns->FullName().IsEmpty())
 93                 {
 94                     formatter.DecIndent();
 95                     formatter.WriteLine("}");
 96                 }
 97             }
 98         }
 99         private CodeFormatter& formatter;
100         private HashMap<Namespace*, List<Grammar*>> nsGrammarMap;
101     }
102     
103     public void GenerateLibraryFile(Project* project, ParsingDomain* parsingDomain)
104     {
105         string libraryFilePath = Path.ChangeExtension(project->FilePath(), ".pl");
106         StreamWriter plFile = File.CreateText(libraryFilePath);
107         CodeFormatter formatter(plFile);
108         LibraryFileGeneratorVisitor visitor(formatter);
109         parsingDomain->GlobalNamespace()->Accept(visitor);
110         visitor.GenerateLibraryFile();
111         if (Flags.Instance().Verbose())
112         {
113             Console.WriteLine("=> " + libraryFilePath);
114         }
115     }
