  7 internal void AddToLibraryDirectories(const string& libraryPath, List<string>& libraryDirectories)
  8 {
  9     List<string> dirs = libraryPath.Split(';');
 10     for (const string& dir : dirs)
 11     {
 12         libraryDirectories.Add(dir);
 13     }
 14 }

 56 internal int main(int argc, const char** argv)
 57 {
 58     try
 59     {
 60         if (argc < 2)
 61         {
 62             PrintHelp();
 63             return 0;
 64         }
 65         List<string> projectFiles;
 66         List<string> libraryDirectories;
 67         bool prevWasLib = false;
 68         string cmajorRoot = RtGetEnvironmentVariable("CMAJOR_ROOT");
 69         if (cmajorRoot.IsEmpty())
 70         {
 71             throw Exception("CMAJOR_ROOT environment variable not defined. Please set it to contain /path/to/cmajor directory.");
 72         }
 73         string libraryPath = Path.Combine(Path.Combine(cmajorRoot, "system"), "System.Text.Parsing");
 74         AddToLibraryDirectories(libraryPath, libraryDirectories);
 75         for (int i = 1; i < argc; ++i;)
 76         {
 77             string arg = argv[i];
 78             if (arg.StartsWith("-"))
 79             {
 80                 if (arg == "--force" || arg == "-F")
 81                 {
 82                     Syntax.Flags.Instance().SetForceCodeGen();
 83                 }
 84                 else if (arg == "--verbose" || arg == "-v")
 85                 {
 86                     Syntax.Flags.Instance().SetVerbose();
 87                 }
 88                 else if (arg == "--lib" || arg == "-L")
 89                 {
 90                     prevWasLib = true;
 91                 }
 92                 else if (arg == "-dp")
 93                 {
 94                     Syntax.Flags.Instance().SetDebugProject();
 95                 }
 96                 else if (arg == "-dl")
 97                 {
 98                     Syntax.Flags.Instance().SetDebugLibrary();
 99                 }
100                 else if (arg == "-da")
101                 {
102                     Syntax.Flags.Instance().SetDebugParse();
103                 }
104                 else if (arg.Find('=') != -1)
105                 {
106                     List<string> components = arg.Split('=');
107                     if (components.Count() != 2)
108                     {
109                         throw Exception("unknown option '" + arg + "'");
110                     }
111                     if (components[0] == "--debug")
112                     {
113                         if (components[1] == "project")
114                         {
115                             Syntax.Flags.Instance().SetDebugProject();
116                         }
117                         else if (components[1] == "library")
118                         {
119                             Syntax.Flags.Instance().SetDebugLibrary();
120                         }
121                         else if (components[1] == "parse")
122                         {
123                             Syntax.Flags.Instance().SetDebugParse();
124                         }
125                         else
126                         {
127                             throw Exception("unknown option '" + arg + "'");
128                         }
129                     }
130                     else if (components[0] == "--threads" || components[0] == "-t")
131                     {
132                         Syntax.Flags.Instance().SetNumThreads(ParseInt(components[1]));
133                     }
134                     else
135                     {
136                         throw Exception("unknown option '" + arg + "'");
137                     }
138                 }
139                 else
140                 {
141                     throw Exception("unknown option '" + arg + "'");
142                 }
143             }
144             else if (prevWasLib)
145             {
146                 AddToLibraryDirectories(arg, libraryDirectories);
147                 prevWasLib = false;
148             }
149             else
150             {
151                 if (arg.EndsWith(".pp"))
152                 {
153                     projectFiles.Add(GetFullPath(arg));
154                 }
155                 else
156                 {
157                     throw Exception("invalid parser project file extension (not .pp) '" + arg + "'");
158                 }
159             }
160         }
161         if (Syntax.Flags.Instance().DebugParse() && Syntax.Flags.Instance().NumThreads() != 1)
162         {
163             throw Exception("--debug=parse option requires --threads=1");
164         }
165         if (Syntax.Flags.Instance().NumThreads() < 0)
166         {
167             throw Exception("invalid thread count '" + ToString(Syntax.Flags.Instance().NumThreads()));
168         }
169         for (const string& projectFile : projectFiles)
170         {
171             Generate(projectFile, libraryDirectories);
172         }
173     }
174     catch (const Exception& ex)
175     {
176         Console.Error() << ex.Message() << endl();
177         return 1;
178     }
179     return 0;
180 }
