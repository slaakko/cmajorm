 10 internal void AddToLibraryDirectories(const string& libraryPath, List<string>& libraryDirectories)
 11 {
 12     List<string> dirs = libraryPath.Split(';');

 35 internal int main(int argc, const char** argv)
 36 {
 37     try
 38     {
 39         bool verbose = false;
 40         List<string> libraryDirectories;
 41         string cmajorRoot = RtGetEnvironmentVariable("CMAJOR_ROOT");
 42         if (cmajorRoot.IsEmpty())
 43         {
 44             throw Exception("CMAJOR_ROOT environment variable not defined. Please set it to contain /path/to/cmajor directory.");
 45         }
 46         string libraryPath = Path.Combine(Path.Combine(cmajorRoot, "system"), "System.Text.Parsing");
 47         AddToLibraryDirectories(libraryPath, libraryDirectories);
 48         List<string> moduleDirs;
 49         for (int i = 1; i < argc; ++i;)
 50         {
 51             string arg = argv[i];
 52             if (arg.StartsWith("-"))
 53             {
 54                 if (arg == "--verbose" || arg == "-v")
 55                 {
 56                     verbose = true;
 57                     Syntax.Flags.Instance().SetVerbose();
 58                 }
 59                 else
 60                 {
 61                     throw Exception("unknown argument '" + arg + "'");
 62                 }
 63             }
 64             else
 65             {
 66                 moduleDirs.Add(GetFullPath(Path.MakeCanonical(arg)));
 67             }
 68         }
 69         if (moduleDirs.IsEmpty())
 70         {
 71             throw Exception("no module directories given");
 72         }
 73         for (const string& moduleDir : moduleDirs)
 74         {
 75             HashMap<ustring, string> grammarMap;
 76             if (verbose)
 77             {
 78                Console.Out() << "> " << moduleDir << endl();
 79             }
 80             string docDir = Path.Combine(moduleDir, "doc");
 81             string ppXmlFile = Path.Combine(moduleDir, "pp.xml");
 82             string gmXmlFile = Path.Combine(moduleDir, "gm.xml");
 83             UniquePtr<DomDocument> ppDoc = ParseXmlFileToDomDocument(ppXmlFile);
 84             UniquePtr<DomDocument> gmDoc = ParseXmlFileToDomDocument(gmXmlFile);
 85             UniquePtr<XPathObject> grammarsObject = Evaluate( u"/grammars/grammar", gmDoc.Get());
 86             if (grammarsObject.Get() is XPathNodeSet*)
 87             {
 88                 XPathNodeSet* grammarsNodeSet = cast<XPathNodeSet*>(grammarsObject.Get());
 89                 long n = grammarsNodeSet->Length();
 90                 for (long i = 0; i < n; ++i;)
 91                 {
 92                     DomNode* grammarNode = (*grammarsNodeSet)[i];
 93                     if (grammarNode is DomElement*)
 94                     {
 95                         DomElement* grammarElement = cast<DomElement*>(grammarNode);
 96                         ustring grammarId = grammarElement->GetAttribute( u"id");
 97                         ustring docPath = grammarElement->GetAttribute( u"docPath");
 98                         grammarMap[grammarId] = ToUtf8(docPath);
 99                     }
100                 }
101             }
102             UniquePtr<XPathObject> parserProjectsObject = Evaluate( u"/parserProjects/parserProject", ppDoc.Get());
103             if (parserProjectsObject.Get() is XPathNodeSet*)
104             {
105                 XPathNodeSet* parserProjectsNodeSet = cast<XPathNodeSet*>(parserProjectsObject.Get());
106                 long n = parserProjectsNodeSet->Length();
107                 for (long i = 0; i < n; ++i;)
108                 {
109                     DomNode* parserProjectNode = (*parserProjectsNodeSet)[i];
110                     if (parserProjectNode is DomElement*)
111                     {
112                         DomElement* parserProjectElement = cast<DomElement*>(parserProjectNode);
113                         string relativePPFilePath = ToUtf8(parserProjectElement->GetAttribute( u"filePath"));
114                         string styleFilePath = ToUtf8(parserProjectElement->GetAttribute( u"styleFilePath"));
115                         if (!relativePPFilePath.IsEmpty())
116                         {
117                             string ppFilePath = GetFullPath(Path.Combine(moduleDir, relativePPFilePath));
118                             if (verbose)
119                             {
120                                Console.Out() << ">> " << ppFilePath << endl();
121                             }
122                             List<GrammarInfo> outputGrammars;
123                             ProcessParserProject(grammarMap, outputGrammars, ppFilePath, docDir, styleFilePath, libraryDirectories, verbose);
124                             DomDocument outputGrammarDoc;
125                             UniquePtr<DomElement> grammarsElement(new DomElement( u"grammars"));
126                             for (const GrammarInfo& info : outputGrammars)
127                             {
128                                 UniquePtr<DomElement> grammarElement(new DomElement( u"grammar"));
129                                 grammarElement->SetAttribute( u"name", info.GrammarName());
130                                 grammarElement->SetAttribute( u"id", info.GrammarId());
131                                 grammarElement->SetAttribute( u"docPath", ToUtf32(Path.Combine("doc", info.GrammarFileName())));
132                                 grammarsElement->AppendChild(UniquePtr<DomNode>(grammarElement.Release()));
133                                 grammarMap[info.GrammarId()] = info.GrammarFileName();
134                             }
135                             outputGrammarDoc.AppendChild(UniquePtr<DomNode>(grammarsElement.Release()));
136                             StreamWriter writer = File.CreateText(Path.Combine(moduleDir, "grammars.xml"));
137                             CodeFormatter formatter(writer);
138                             formatter.SetIndentSize(1);
139                             outputGrammarDoc.Write(formatter);
140                         }
141                     }
142                 }
143             }
144         }
145     }
146     catch (const Exception& ex)
147     {
148         Console.Error() << ex.ToString() << endl();
149         return 1;
150     }
151     return 0;
