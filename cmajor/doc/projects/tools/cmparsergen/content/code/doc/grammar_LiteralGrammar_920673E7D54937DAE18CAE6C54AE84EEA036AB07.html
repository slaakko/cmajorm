LiteralGrammar Grammar

Literal → BooleanLiteral | FloatingLiteral | IntegerLiteral | CharLiteral | StringLiteral | NullLiteral
BooleanLiteral → keyword("true") | keyword("false")
FloatingLiteral → token(FloatingLiteralValue [fF]?)
FloatingLiteralValue → token(FractionalFloatingLiteral | ExponentFloatingLiteral)
FractionalFloatingLiteral → token(digit_sequence? '.' digit_sequence ExponentPart?) | token(digit_sequence '.')
ExponentFloatingLiteral → token(digit_sequence ExponentPart)
ExponentPart → token([eE] sign digit_sequence)
IntegerLiteral → token(IntegerLiteralValue [uU]?)
IntegerLiteralValue → token(HexIntegerLiteral | DecIntegerLiteral)
HexIntegerLiteral → token(("0x" | "0X") hexdigit+)
DecIntegerLiteral → token(digit_sequence)
CharLiteral → token('\'' (([^'\\\r\n]+) | CharEscape) '\''!) | ('w' token('\'' (([^'\\\r\n]+) | CharEscape) '\''!)) | ('u' token('\'' (([^'\\\r\n]+) | CharEscape) '\''!))
StringLiteral → ('@' token('\"' ([^\"]*) '\"')) | ('w' '@' token('\"' ([^\"]*) '\"')) | ('u' '@' token('\"' ([^\"]*) '\"')) | token('\"' (([^\"\\\r\n]+) | CharEscape)* '\"') | ('w' token('\"' (([^\"\\\r\n]+) | CharEscape)* '\"')) | ('u' token('\"' (([^\"\\\r\n]+) | CharEscape)* '\"'))
NullLiteral → keyword("null")
CharEscape → token('\\' ([xX] hexdigit+ | [dD] digit_sequence | [0-7]+ | 'u' hexdigit hexdigit hexdigit hexdigit | 'U' hexdigit hexdigit hexdigit hexdigit hexdigit hexdigit hexdigit hexdigit | anychar))
