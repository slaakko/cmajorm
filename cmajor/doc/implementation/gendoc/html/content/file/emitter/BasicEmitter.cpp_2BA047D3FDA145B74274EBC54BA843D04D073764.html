 100 void BasicEmitter::Visit(BoundCompileUnit& boundCompileUnit)
 101 {
 102     if (GetGlobalFlag(GlobalFlags::generateDebugInfo) && boundCompileUnit.GetCompileUnitNode() && !boundCompileUnit.GetCompileUnitNode()->IsSynthesizedUnit())
 103     {
 104         compileUnitModule->addModuleFlag(llvm::Module::Warning, "Debug Info Version", llvm::DEBUG_METADATA_VERSION);
 105 #ifdef _WIN32
 106         compileUnitModule->addModuleFlag(llvm::Module::Warning, "CodeView", 1);
 107 #endif
 108         diBuilder.reset(new llvm::DIBuilder(*compileUnitModule));
 109         SetDIBuilder(diBuilder.get());
 110         std::string sourceFilePath = boundCompileUnit.GetCompileUnitNode()->FilePath();
 111         llvm::DIFile* sourceFile = diBuilder->createFile(Path::GetFileName(sourceFilePath), Path::GetDirectoryName(sourceFilePath));
 112         SetDIFile(sourceFile);
 113         llvm::DICompileUnit* diCompileUnit = diBuilder->createCompileUnit(cmajorLanguageTag, sourceFile, "Cmajor compiler version " + GetCompilerVersion(), GetGlobalFlag(GlobalFlags::release), "", 0);
 114         SetDICompileUnit(diCompileUnit);
 115         SetCurrentCompileUnitNode(boundCompileUnit.GetCompileUnitNode());
 116         SetCompileUnitIndex(boundCompileUnit.CompileUnitIndex());
 117         PushScope(sourceFile);
 118     }
 119     compileUnit = &boundCompileUnit;
 120     symbolTable = &boundCompileUnit.GetSymbolTable();
 121     ConstantArrayRepository& constantArrayRepository = boundCompileUnit.GetConstantArrayRepository();
 122     for (ConstantSymbol* constantSymbol : constantArrayRepository.ConstantArrays())
 123     {
 124         constantSymbol->ArrayIrObject(*this, true);
 125     }
 126     ConstantStructureRepository& constantStructureRepository = boundCompileUnit.GetConstantStructureRepository();
 127     for (ConstantSymbol* constantSymbol : constantStructureRepository.ConstantStructures())
 128     {
 129         constantSymbol->StructureIrObject(*this, true);
 130     }
 131     int n = boundCompileUnit.BoundNodes().size();
 132     for (int i = 0; i < n; ++i)
 133     {
 134         BoundNode* boundNode = boundCompileUnit.BoundNodes()[i].get();
 135         boundNode->Accept(*this);
 136     }
 137     if (diBuilder)
 138     {
 139         ReplaceForwardDeclarations();
 140         diBuilder->finalize();
 141     }
 142     if (GetGlobalFlag(GlobalFlags::emitLlvm))
 143     {
 144         std::ofstream llFile(boundCompileUnit.LLFilePath());
 145         llvm::raw_os_ostream llOs(llFile);
 146         compileUnitModule->print(llOs, nullptr);
 147     }
 148     std::string errorMessageStore;
 149     llvm::raw_string_ostream errorMessage(errorMessageStore);
 150     if (verifyModule(*compileUnitModule, &errorMessage))
 151     {
 152         throw std::runtime_error("Emitter: verification of module '" + compileUnitModule->getSourceFileName() + "' failed. " + errorMessage.str());
 153     }
 154     llvm::legacy::PassManager passManager;
 155     std::error_code errorCode;
 156     llvm::raw_fd_ostream objectFile(boundCompileUnit.ObjectFilePath(), errorCode, llvm::sys::fs::F_None);
 157     if (emittingContext.GetEmittingContextImpl()->TargetMachine().addPassesToEmitFile(passManager, objectFile, llvm::TargetMachine::CGFT_ObjectFile))
 158     {
 159         throw std::runtime_error("Emitter: cannot emit object code file '" + boundCompileUnit.ObjectFilePath() + "': addPassesToEmitFile failed");
 160     }
 161     passManager.run(*compileUnitModule);
 162     objectFile.flush();
 163     if (objectFile.has_error())
 164     {
 165         throw std::runtime_error("Emitter: could not emit object code file '" + boundCompileUnit.ObjectFilePath() + "': " + errorCode.message());
 166     }
 167     if (GetGlobalFlag(GlobalFlags::emitOptLlvm))
 168     {
 169         std::string optCommandLine;
 170         optCommandLine.append("opt -O").append(std::to_string(GetOptimizationLevel())).append(" ").append(QuotedPath(boundCompileUnit.LLFilePath())).append(" -S -o ");
 171         optCommandLine.append(QuotedPath(boundCompileUnit.OptLLFilePath()));
 172         System(optCommandLine);
 173     }
 174     if (diBuilder)
 175     {
 176         PopScope();
 177         diBuilder.reset();
 178     }
 179 }

 181 void BasicEmitter::Visit(BoundNamespace& boundNamespace)
 182 {
 183     int numComponents = 0;
 184     if (diBuilder)
 185     {
 186         if (!boundNamespace.GetNamespaceNode().Id()->Str().empty())
 187         {
 188             std::vector<std::u32string> components = Split(boundNamespace.GetNamespaceNode().Id()->Str(), '.');

 191             {

 194             }
 195         }
 196         else

 198             PushScope(GetFile(boundNamespace.GetSpan().FileIndex()));
 199         }
 200     }

 203     {
 204         BoundNode* member = boundNamespace.Members()[i].get();
 205         member->Accept(*this);
 206     }
 207     if (diBuilder)
 208     {
 209         if (!boundNamespace.GetNamespaceNode().Id()->Str().empty())
 210         {
 211             for (int i = 0; i < numComponents; ++i)
 212             {
 213                 PopScope();
 214             }
 215         }
 216         else
 217         {
 218             PopScope();
 219         }

 223 void BasicEmitter::Visit(BoundClass& boundClass)
 224 {
 225     classStack.push(currentClass);
 226     currentClass = &boundClass;
 227     if (diBuilder)
 228     {
 229         llvm::DIType* diType = GetDIType(currentClass->GetClassTypeSymbol());
 230         if (diType)
 231         {
 232             PushScope(diType);
 233         }
 234         else
 235         {
 236             llvm::DIType* baseClass = nullptr;
 237             if (currentClass->GetClassTypeSymbol()->BaseClass())
 238             {
 239                 baseClass = currentClass->GetClassTypeSymbol()->BaseClass()->GetDIType(*this);
 240             }
 241             llvm::DIType* vtableHolderClass = nullptr;
 242             if (currentClass->GetClassTypeSymbol()->IsPolymorphic() && currentClass->GetClassTypeSymbol()->VmtPtrHolderClass())
 243             {
 244                 vtableHolderClass = currentClass->GetClassTypeSymbol()->VmtPtrHolderClass()->CreateDIForwardDeclaration(*this);
 245                 MapFwdDeclaration(vtableHolderClass, currentClass->GetClassTypeSymbol()->VmtPtrHolderClass());
 246             }
 247             llvm::MDNode* templateParams = nullptr;
 248             uint64_t sizeInBits = DataLayout()->getStructLayout(      llvm::cast<llvm::StructType>(currentClass->GetClassTypeSymbol()->IrType(*this)))->getSizeInBits();
 249             uint32_t alignInBits = 8 * DataLayout()->getStructLayout(      llvm::cast<llvm::StructType>(currentClass->GetClassTypeSymbol()->IrType(*this)))->getAlignment();
 250             uint64_t offsetInBits = 0;
 251             llvm::DINode::DIFlags flags = llvm::DINode::DIFlags::FlagZero;
 252             Span classSpan = currentClass->GetClassTypeSymbol()->GetSpan();
 253             if (currentClass->GetClassTypeSymbol()->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
 254             {
 255                 ClassTemplateSpecializationSymbol* specialization = static_cast<ClassTemplateSpecializationSymbol*>(currentClass->GetClassTypeSymbol());
 256                 classSpan = specialization->GetClassTemplate()->GetSpan();
 257             }
 258             llvm::DIType* forwardDeclaration = diBuilder->createReplaceableCompositeType(llvm::dwarf::DW_TAG_class_type, ToUtf8(currentClass->GetClassTypeSymbol()->Name()), 
 259                 CurrentScope(), GetFile(classSpan.FileIndex()), classSpan.LineNumber(), 
 260                 0, sizeInBits, alignInBits, llvm::DINode::DIFlags::FlagZero, ToUtf8(currentClass->GetClassTypeSymbol()->MangledName()));
 261             SetDIType(currentClass->GetClassTypeSymbol(), forwardDeclaration);
 262             std::vector<llvm::Metadata*> elements;
 263             for (MemberVariableSymbol* memberVariable : currentClass->GetClassTypeSymbol()->MemberVariables())
 264             {
 265                 int memberVariableLayoutIndex = memberVariable->LayoutIndex();
 266                 uint64_t offsetInBits = DataLayout()->getStructLayout(      llvm::cast<llvm::StructType>(currentClass->GetClassTypeSymbol()->IrType(*this)))->getElementOffsetInBits(memberVariableLayoutIndex);
 267                 elements.push_back(memberVariable->GetDIMemberType(*this, offsetInBits));
 268             }
 269             llvm::DICompositeType* cls = diBuilder->createClassType(CurrentScope(), ToUtf8(currentClass->GetClassTypeSymbol()->Name()), 
 270                 GetFile(classSpan.FileIndex()), 
 271                 classSpan.LineNumber(), sizeInBits, alignInBits, offsetInBits, flags, baseClass, diBuilder->getOrCreateArray(elements), vtableHolderClass, templateParams, 
 272                 ToUtf8(currentClass->GetClassTypeSymbol()->MangledName()));
 273             MapFwdDeclaration(forwardDeclaration, currentClass->GetClassTypeSymbol());
 274             SetDIType(currentClass->GetClassTypeSymbol(), cls);
 275             PushScope(cls);
 276         }
 277     }
 278     int n = boundClass.Members().size();
 279     for (int i = 0; i < n; ++i)
 280     {
 281         BoundNode* boundNode = boundClass.Members()[i].get();
 282         boundNode->Accept(*this);
 283     }
 284     if (diBuilder)
 285     {
 286         PopScope();
 287     }
 288     currentClass = classStack.top();
 289     classStack.pop();
 290 }

 292 void BasicEmitter::Visit(BoundEnumTypeDefinition& boundEnumTypeDefinition)

 330 void BasicEmitter::Visit(BoundFunction& boundFunction)
 331 {
 332     if (!boundFunction.Body()) return;
 333     currentFunction = &boundFunction;
 334     handlerBlock = nullptr;
 335     cleanupBlock = nullptr;
 336     newCleanupNeeded = false;
 337     currentPad = nullptr;
 338     prevLineNumber = 0;
 339     destructorCallGenerated = false;
 340     lastInstructionWasRet = false;
 341     basicBlockOpen = false;
 342     lastAlloca = nullptr;
 343     compoundLevel = 0;
 344     cleanups.clear();
 345     pads.clear();
 346     labeledStatementMap.clear();
 347     FunctionSymbol* functionSymbol = boundFunction.GetFunctionSymbol();
 348     llvm::FunctionType* functionType = functionSymbol->IrType(*this);
 349     function =       llvm::cast<llvm::Function>(compileUnitModule->getOrInsertFunction(ToUtf8(functionSymbol->MangledName()), functionType));
 350     if (GetGlobalFlag(GlobalFlags::release) && functionSymbol->IsInline())
 351     {
 352         function->addFnAttr(llvm::Attribute::InlineHint);
 353         functionSymbol->SetLinkOnceOdrLinkage();
 354     }
 355     if (functionSymbol->HasLinkOnceOdrLinkage())
 356     {
 357         llvm::Comdat* comdat = compileUnitModule->getOrInsertComdat(ToUtf8(functionSymbol->MangledName()));
 358         comdat->setSelectionKind(llvm::Comdat::SelectionKind::Any);
 359         function->setLinkage(llvm::GlobalValue::LinkageTypes::LinkOnceODRLinkage);
 360         function->setComdat(comdat);
 361     }
 362     SetFunction(function);
 363     if (diBuilder)
 364     {
 365         SetInPrologue(true);
 366         SetCurrentDebugLocation(Span());
 367         std::vector<llvm::Metadata*> elementTypes;
 368         if (functionSymbol->ReturnType())
 369         {
 370             elementTypes.push_back(functionSymbol->ReturnType()->GetDIType(*this));
 371         }
 372         else
 373         {
 374             elementTypes.push_back(symbolTable->GetTypeByName(U"void")->GetDIType(*this));
 375         }
 376         for (ParameterSymbol* parameter : functionSymbol->Parameters())
 377         {
 378             TypeSymbol* parameterType = parameter->GetType();
 379             elementTypes.push_back(parameterType->GetDIType(*this));
 380         }
 381         llvm::DISubroutineType* subroutineType = diBuilder->createSubroutineType(diBuilder->getOrCreateTypeArray(elementTypes));
 382         llvm::DISubprogram* subprogram = nullptr;
 383         if (functionSymbol->IsConstructorDestructorOrNonstaticMemberFunction())
 384         {
 385             unsigned int virtuality = 0;
 386             if (functionSymbol->IsVirtualAbstractOrOverride())
 387             {
 388                 if (functionSymbol->IsAbstract())
 389                 {
 390                     virtuality = llvm::dwarf::DW_VIRTUALITY_pure_virtual;
 391                 }
 392                 else
 393                 {
 394                     virtuality = llvm::dwarf::DW_VIRTUALITY_virtual;
 395                 }
 396             }
 397             llvm::DIType* vtableHolder = nullptr;
 398             Symbol* parent = functionSymbol->Parent();
 399             if (parent && parent->IsClassTypeSymbol())
 400             {
 401                 ClassTypeSymbol* cls = static_cast<ClassTypeSymbol*>(parent);
 402                 if (cls->IsPolymorphic() && cls->VmtPtrHolderClass())
 403                 {
 404                     vtableHolder = cls->VmtPtrHolderClass()->GetDIType(*this);
 405                 }
 406             }
 407             llvm::DINode::DIFlags flags = llvm::DINode::DIFlags::FlagZero;
 408             if (functionSymbol->IsStatic())
 409             {
 410                 flags |= llvm::DINode::FlagStaticMember;
 411             }
 412             flags |= AccessFlag(functionSymbol->Access());
 413             if (functionSymbol->IsExplicit())
 414             {
 415                 flags |= llvm::DINode::FlagExplicit;
 416             }
 417             unsigned vtableIndex = 0;
 418             if (functionSymbol->VmtIndex() != -1)
 419             {
 420                 vtableIndex = functionSymbol->VmtIndex();
 421             }
 422             subprogram = diBuilder->createMethod(CurrentScope(), ToUtf8(functionSymbol->Name()), ToUtf8(functionSymbol->MangledName()), GetFile(functionSymbol->GetSpan().FileIndex()), functionSymbol->GetSpan().LineNumber(), 
 423                 subroutineType, false, true, virtuality, vtableIndex, 0, vtableHolder, flags);
 424         }
 425         else
 426         {
 427             llvm::DINode::DIFlags flags = AccessFlag(functionSymbol->Access());
 428             subprogram = diBuilder->createFunction(CurrentScope(), ToUtf8(functionSymbol->Name()), ToUtf8(functionSymbol->MangledName()), GetFile(functionSymbol->GetSpan().FileIndex()), 
 429                 functionSymbol->GetSpan().LineNumber(), 
 430                 subroutineType, false, true, functionSymbol->GetSpan().LineNumber(), flags);
 431         }
 432         function->setSubprogram(subprogram);
 433         PushScope(subprogram);
 434     }
 435     llvm::BasicBlock* entryBlock = llvm::BasicBlock::Create(context, "entry", function);
 436     entryBasicBlock = entryBlock;
 437     SetCurrentBasicBlock(entryBlock);
 438     if (currentClass)
 439     {
 440         ClassTypeSymbol* classTypeSymbol = currentClass->GetClassTypeSymbol();
 441         classTypeSymbol->SetModule(&symbolsModule);
 442         if (!classTypeSymbol->IsVmtObjectCreated())
 443         {
 444             classTypeSymbol->VmtObject(*this, true);
 445         }
 446         if (!classTypeSymbol->IsStaticObjectCreated())
 447         {
 448             classTypeSymbol->StaticObject(*this, true);
 449         }
 450     }
 451     int np = functionSymbol->Parameters().size();
 452     for (int i = 0; i < np; ++i)
 453     {
 454         ParameterSymbol* parameter = functionSymbol->Parameters()[i];
 455         llvm::AllocaInst* allocaInst = builder.CreateAlloca(parameter->GetType()->IrType(*this));
 456         parameter->SetIrObject(allocaInst);
 457         lastAlloca = allocaInst;
 458         if (diBuilder)
 459         {
 460             llvm::SmallVector<int64_t, 13> expr;
 461             llvm::DILocalVariable* paramVar = diBuilder->createParameterVariable(CurrentScope(), ToUtf8(parameter->Name()), i + 1, GetFile(parameter->GetSpan().FileIndex()),

 598     {
 599         lastStatement = body->Statements().back().get();

 602     {
 603         CreateExitFunctionCall();

 605         {

 623         function->addFnAttr(llvm::Attribute::NoUnwind);
 624     }
 625     else
 626     {
 627         function->addFnAttr(llvm::Attribute::UWTable);

 630     if (diBuilder)
 631     {
 632         PopScope();

 652 void BasicEmitter::ExitBlocks(BoundCompoundStatement* targetBlock)
 653 {
 654     if (diBuilder)
 655     {
 656         if (currentBlock->EndSpan().Valid())
 657         {
 658             SetCurrentDebugLocation(currentBlock->EndSpan());
 659         }
 660     }
 661     bool createBasicBlock = false;
 662     BoundStatement* lastStatement = nullptr;
 663     if (!currentFunction->Body()->Statements().empty())
 664     {
 665         lastStatement = currentFunction->Body()->Statements().back().get();
 666     }
 667     BoundStatement* currentBlockLastStatement = nullptr;
 668     if (currentBlock && !currentBlock->Statements().empty())
 669     {
 670         currentBlockLastStatement = currentBlock->Statements().back().get();
 671     }
 672     if (lastStatement && currentBlockLastStatement && lastStatement == currentBlockLastStatement && currentBlockLastStatement->GetBoundNodeType() == BoundNodeType::boundReturnStatement)
 673     {
 674         createBasicBlock = true;
 675     }
 676     int n = blocks.size();
 677     for (int i = n - 1; i >= 0; --i)
 678     {

 681         {
 682             break;

 685         if (it != blockDestructionMap.cend())

 688             int nd = destructorCallVec.size();

 690             {
 691                 std::unique_ptr<BoundFunctionCall>& destructorCall = destructorCallVec[i];

 693                 {
 694                     if (createBasicBlock)
 695                     {

 697                         if (!lastInstructionWasRet)
 698                         {
 699                             builder.CreateBr(nextBlock);
 700                         }
 701                         SetCurrentBasicBlock(nextBlock);
 702                         createBasicBlock = false;
 703                     }
 704                     destructorCall->Accept(*this);
 705                     destructorCallGenerated = true;
 706                     newCleanupNeeded = true;
 707                 }
 708             }
 709         }
 710     }
