 22 void WindowsEmitter::Visit(BoundReturnStatement& boundReturnStatement)
 23 {
 24     SetCurrentDebugLocation(boundReturnStatement.GetSpan());
 25     destructorCallGenerated = false;
 26     lastInstructionWasRet = false;
 27     basicBlockOpen = false;
 28     SetTarget(&boundReturnStatement);
 29     Pad* prevCurrentPad = currentPad;
 30     while (currentPad != nullptr)
 31     {
 32         llvm::BasicBlock* returnTarget = llvm::BasicBlock::Create(context, "return", function);
 33         builder.CreateCatchRet(      llvm::cast<llvm::CatchPadInst>(currentPad->value), returnTarget);
 34         SetCurrentBasicBlock(returnTarget);
 35         currentPad = currentPad->parent;
 36     }
 37     BoundFunctionCall* returnFunctionCall = boundReturnStatement.ReturnFunctionCall();
 38     if (returnFunctionCall)
 39     {
 40         returnFunctionCall->Accept(*this);
 41         llvm::Value* returnValue = stack.Pop();
 42         if (sequenceSecond)
 43         {
 44             sequenceSecond->SetGenerated();
 45             sequenceSecond->Accept(*this);
 46         }
 47         ExitBlocks(nullptr);
 48         CreateExitFunctionCall();
 49         builder.CreateRet(returnValue);
 50         lastInstructionWasRet = true;
 51     }
 52     else
 53     {
 54         ExitBlocks(nullptr);
 55         CreateExitFunctionCall();
 56         builder.CreateRetVoid();
 57         lastInstructionWasRet = true;
 58     }
 59     BoundCompoundStatement* body = currentFunction->Body();
 60     BoundStatement* lastStatement = nullptr;
 61     if (!body->Statements().empty())
 62     {
 63         lastStatement = body->Statements().back().get();
 64     }
 65     if (lastStatement && lastStatement != &boundReturnStatement)
 66     {
 67         llvm::BasicBlock* nextBlock = llvm::BasicBlock::Create(context, "next", function);
 68         SetCurrentBasicBlock(nextBlock);
 69         basicBlockOpen = true;
 70         lastInstructionWasRet = false;
 71     }
 72     currentPad = prevCurrentPad;
 73 }

 75 void WindowsEmitter::Visit(BoundGotoCaseStatement& boundGotoCaseStatement)
 76 {
 77     SetCurrentDebugLocation(boundGotoCaseStatement.GetSpan());
 78     destructorCallGenerated = false;
 79     lastInstructionWasRet = false;
 80     basicBlockOpen = false;
 81     SetTarget(&boundGotoCaseStatement);
 82     Assert(breakTargetBlock, "break target not set");
 83     Pad* prevCurrentPad = currentPad;
 84     BoundStatement* parent = currentBlock;
 85     while (currentPad != nullptr && parent && parent != breakTargetBlock)
 86     {
 87         if (parent->GetBoundNodeType() == BoundNodeType::boundTryStatement)
 88         {
 89             llvm::BasicBlock* fromCatchTarget = llvm::BasicBlock::Create(context, "fromCatch", function);
 90             builder.CreateCatchRet(      llvm::cast<llvm::CatchPadInst>(currentPad->value), fromCatchTarget);
 91             SetCurrentBasicBlock(fromCatchTarget);
 92             currentPad = currentPad->parent;
 93         }
 94         parent = parent->Parent();
 95     }
 96     ExitBlocks(breakTargetBlock);
 97     IntegralValue integralCaseValue(boundGotoCaseStatement.CaseValue());
 98     auto it = currentCaseMap->find(integralCaseValue);
 99     if (it != currentCaseMap->cend())
100     {
101         llvm::BasicBlock* caseDest = it->second;
102         builder.CreateBr(caseDest);
103     }
104     else
105     {
106         throw Exception(&symbolsModule, "case not found", boundGotoCaseStatement.GetSpan());
107     }
108     currentPad = prevCurrentPad;
109 }

242 void WindowsEmitter::Visit(BoundTryStatement& boundTryStatement)
243 {
244     ClearFlags();
245     SetTarget(&boundTryStatement);
246     llvm::BasicBlock* prevHandlerBlock = handlerBlock;
247     llvm::BasicBlock* prevCleanupBlock = cleanupBlock;
248     handlerBlock = llvm::BasicBlock::Create(context, "handlers", function);
249     cleanupBlock = nullptr;
250     boundTryStatement.TryBlock()->Accept(*this);
251     llvm::BasicBlock* nextTarget = llvm::BasicBlock::Create(context, "next", function);
252     builder.CreateBr(nextTarget);
253     SetCurrentBasicBlock(handlerBlock);
254     handlerBlock = prevHandlerBlock;
255     Pad* parentPad = currentPad;
256     llvm::CatchSwitchInst* catchSwitch = nullptr;
257     if (parentPad == nullptr)
258     {
259         catchSwitch = builder.CreateCatchSwitch(llvm::ConstantTokenNone::get(context), prevHandlerBlock, 1);
260     }
261     else
262     {
263         catchSwitch = builder.CreateCatchSwitch(parentPad->value, prevHandlerBlock, 1);
264     }
265     Pad* pad = new Pad();
266     pads.push_back(     std::unique_ptr<Pad>(pad));
267     pad->parent = parentPad;
268     pad->value = catchSwitch;
269     currentPad = pad;
270     llvm::BasicBlock* catchPadTarget = llvm::BasicBlock::Create(context, "catchpad", function);
271     catchSwitch->addHandler(catchPadTarget);
272     SetCurrentBasicBlock(catchPadTarget);
273     ArgVector catchPadArgs;
274     catchPadArgs.push_back(llvm::Constant::getNullValue(builder.getInt8PtrTy()));
275     catchPadArgs.push_back(builder.getInt32(64));
276     catchPadArgs.push_back(llvm::Constant::getNullValue(builder.getInt8PtrTy()));
277     llvm::CatchPadInst* catchPad = builder.CreateCatchPad(currentPad->value, catchPadArgs);
278     currentPad->value = catchPad;
279     llvm::BasicBlock* catchTarget = llvm::BasicBlock::Create(context, "catch", function);
280     llvm::BasicBlock* resumeTarget = llvm::BasicBlock::Create(context, "resume", function);
281     builder.CreateBr(catchTarget);
282     int n = boundTryStatement.Catches().size();
283     for (int i = 0; i < n; ++i)
284     {
285         const std::unique_ptr<BoundCatchStatement>& boundCatchStatement = boundTryStatement.Catches()[i];
286         SetCurrentBasicBlock(catchTarget);
287         std::vector<llvm::Type*> handleExceptionParamTypes;
288         handleExceptionParamTypes.push_back(builder.getInt8PtrTy());
289         llvm::FunctionType* handleExceptionFunctionType = llvm::FunctionType::get(builder.getInt1Ty(), handleExceptionParamTypes, false);
290         ArgVector handleExceptionArgs;
291         UuidValue uuidValue(boundCatchStatement->GetSpan(), boundCatchStatement->CatchedTypeUuidId());
292         llvm::Value* catchTypeIdValue = uuidValue.IrValue(*this);
293         handleExceptionArgs.push_back(catchTypeIdValue);
294         llvm::Function* handleException =       llvm::cast<llvm::Function>(compileUnitModule->getOrInsertFunction("RtHandleException", handleExceptionFunctionType));
295         llvm::Value* handleThisEx = nullptr;
296         if (currentPad == nullptr)
297         {
298             handleThisEx = builder.CreateCall(handleException, handleExceptionArgs);
299         }
300         else
301         {
302             std::vector<llvm::OperandBundleDef> bundles;
303             std::vector<llvm::Value*> inputs;
304             inputs.push_back(currentPad->value);
305             bundles.push_back(llvm::OperandBundleDef("funclet", inputs));
306             handleThisEx = llvm::CallInst::Create(handleException, handleExceptionArgs, bundles, "", CurrentBasicBlock());
307             if (diBuilder)
308             {
309                       llvm::cast<llvm::CallInst>(handleThisEx)->setDebugLoc(GetDebugLocation(boundCatchStatement->GetSpan()));
310             }
311         }
312         llvm::BasicBlock* nextHandlerTarget = nullptr;
313         if (i < n - 1)
314         {
315             catchTarget = llvm::BasicBlock::Create(context, "catch", function);
316             nextHandlerTarget = catchTarget;
317         }
318         else
319         {
320             nextHandlerTarget = resumeTarget;
321         }
322         llvm::BasicBlock* thisHandlerTarget = llvm::BasicBlock::Create(context, "handler", function);
323         builder.CreateCondBr(handleThisEx, thisHandlerTarget, nextHandlerTarget);
324         SetCurrentBasicBlock(thisHandlerTarget);
325         boundCatchStatement->CatchBlock()->Accept(*this);
326         builder.CreateCatchRet(      llvm::cast<llvm::CatchPadInst>(currentPad->value), nextTarget);
327     }
328     SetCurrentBasicBlock(resumeTarget);
329 
330     std::vector<llvm::Type*> cxxThrowFunctionParamTypes;
331     cxxThrowFunctionParamTypes.push_back(builder.getInt8PtrTy());
332     cxxThrowFunctionParamTypes.push_back(builder.getInt8PtrTy());
333     llvm::FunctionType* cxxThrowFunctionType = llvm::FunctionType::get(builder.getVoidTy(), cxxThrowFunctionParamTypes, false);
334     llvm::Function* cxxThrowFunction =       llvm::cast<llvm::Function>(compileUnitModule->getOrInsertFunction("_CxxThrowException", cxxThrowFunctionType));
335     ArgVector rethrowArgs;
336     rethrowArgs.push_back(llvm::Constant::getNullValue(builder.getInt8PtrTy()));
337     rethrowArgs.push_back(llvm::Constant::getNullValue(builder.getInt8PtrTy()));
338     std::vector<llvm::OperandBundleDef> bundles;
339     std::vector<llvm::Value*> inputs;
340     inputs.push_back(currentPad->value);
341     bundles.push_back(llvm::OperandBundleDef("funclet", inputs));
342     llvm::CallInst* callInst = llvm::CallInst::Create(cxxThrowFunction, rethrowArgs, bundles, "", resumeTarget);
343     if (diBuilder)
344     {
345         callInst->setDebugLoc(GetCurrentDebugLocation());
346     }
347     builder.CreateBr(nextTarget);
348     currentPad = parentPad;
349     SetCurrentBasicBlock(nextTarget);
350     basicBlockOpen = true;
351     cleanupBlock = prevCleanupBlock;
352 }

385 void WindowsEmitter::CreateCleanup()
386 {
387     cleanupBlock = llvm::BasicBlock::Create(context, "cleanup", function);
388     BoundCompoundStatement* targetBlock = nullptr;
389     BoundStatement* parent = currentBlock->Parent();
390     while (parent && parent->GetBoundNodeType() != BoundNodeType::boundTryStatement)
391     {
392         parent = parent->Parent();
393     }
394     if (parent)
395     {
396         targetBlock = parent->Block();
397     }
398     Cleanup* cleanup = new Cleanup(cleanupBlock, handlerBlock, currentPad);
399     int n = blocks.size();
400     for (int i = n - 1; i >= 0; --i)
401     {
402         BoundCompoundStatement* block = blocks[i];
403         if (block == targetBlock)
404         {
405             break;
406         }
407         auto it = blockDestructionMap.find(block);
408         if (it != blockDestructionMap.cend())
409         {
410             std::vector<std::unique_ptr<BoundFunctionCall>>& destructorCallVec = it->second;
411             int nd = destructorCallVec.size();
412             for (int i = nd - 1; i >= 0; --i)
413             {
414                 std::unique_ptr<BoundFunctionCall>& destructorCall = destructorCallVec[i];
415                 if (destructorCall)
416                 {
417                     cleanup->destructors.push_back(     std::unique_ptr<BoundFunctionCall>(static_cast<BoundFunctionCall*>(destructorCall->Clone())));
418                 }
419             }
420         }
421     }
422     cleanups.push_back(     std::unique_ptr<Cleanup>(cleanup));
423     newCleanupNeeded = false;
424 }

426 void WindowsEmitter::GenerateCodeForCleanups()
427 {
428     for (const std::unique_ptr<Cleanup>& cleanup : cleanups)
429     {
430         SetCurrentBasicBlock(cleanup->cleanupBlock);
431         Pad* parentPad = cleanup->currentPad;
432         llvm::CleanupPadInst* cleanupPad = nullptr;
433         if (parentPad)
434         {
435             ArgVector args;
436             cleanupPad = builder.CreateCleanupPad(parentPad->value, args);
437         }
438         else
439         {
440             ArgVector args;
441             cleanupPad = builder.CreateCleanupPad(llvm::ConstantTokenNone::get(context), args);
442         }
443         Pad pad;
444         pad.parent = parentPad;
445         pad.value = cleanupPad;
446         currentPad = &pad;
447         for (const std::unique_ptr<BoundFunctionCall>& destructorCall : cleanup->destructors)
448         {
449             destructorCall->Accept(*this);
450         }
451         llvm::BasicBlock* unwindTarget = cleanup->handlerBlock;
452         builder.CreateCleanupRet(      llvm::cast<llvm::CleanupPadInst>(cleanupPad), unwindTarget);
453     }
454 }
