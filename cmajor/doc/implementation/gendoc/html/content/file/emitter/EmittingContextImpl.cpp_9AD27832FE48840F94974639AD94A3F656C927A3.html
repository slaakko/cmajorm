11 using namespace cmajor::symbols;
12 
13 EmittingContextImpl::EmittingContextImpl() : targetTriple(), optimizationLevel(0), targetMachine(), dataLayout()
14 {
15     llvm::InitializeAllTargetInfos();
16     llvm::InitializeAllTargets();
17     llvm::InitializeAllTargetMCs();
18     llvm::InitializeAllAsmParsers();
19     llvm::InitializeAllAsmPrinters();
20     llvm::PassRegistry* passRegistry = llvm::PassRegistry::getPassRegistry();
21     llvm::initializeCore(*passRegistry);
22     llvm::initializeScalarOpts(*passRegistry);
23     llvm::initializeVectorization(*passRegistry);
24     llvm::initializeIPO(*passRegistry);
25     llvm::initializeAnalysis(*passRegistry);
26     llvm::initializeTransformUtils(*passRegistry);
27     llvm::initializeInstCombine(*passRegistry);
28     llvm::initializeInstrumentation(*passRegistry);
29     llvm::initializeTarget(*passRegistry);
30     llvm::initializeCodeGen(*passRegistry);
31     targetTriple = llvm::sys::getDefaultTargetTriple();
32     std::string error;
33     const llvm::Target* target = llvm::TargetRegistry::lookupTarget(targetTriple, error);
34     if (!target)
35     {
36         throw std::runtime_error("EmittingContext: TargetRegistry::lookupTarget failed: " + error);
37     }
38     llvm::TargetOptions targetOptions = {};
39 #ifdef _WIN32
40     targetOptions.ExceptionModel = llvm::ExceptionHandling::WinEH;
41 #endif
42     llvm::Optional<llvm::Reloc::Model> relocModel;
43     llvm::CodeModel::Model codeModel = llvm::CodeModel::Large;
44     llvm::CodeGenOpt::Level codeGenLevel = llvm::CodeGenOpt::None;
45     optimizationLevel = GetOptimizationLevel();
46     switch (optimizationLevel)
47     {
48         case 0: codeGenLevel = llvm::CodeGenOpt::None; break;
49         case 1: codeGenLevel = llvm::CodeGenOpt::Less; break;
50         case 2: codeGenLevel = llvm::CodeGenOpt::Default; break;
51         case 3: codeGenLevel = llvm::CodeGenOpt::Aggressive; break;
52     }
53     targetMachine.reset(target->createTargetMachine(targetTriple, "generic", "", targetOptions, relocModel, codeModel, codeGenLevel));
54     dataLayout.reset(new llvm::DataLayout(targetMachine->createDataLayout()));
55 }
56 
57 } } // namespace cmajor::emitter
