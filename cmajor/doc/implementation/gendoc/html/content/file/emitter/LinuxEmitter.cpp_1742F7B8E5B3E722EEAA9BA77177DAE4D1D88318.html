 22 void LinuxEmitter::Visit(BoundReturnStatement& boundReturnStatement)
 23 {
 24     SetCurrentDebugLocation(boundReturnStatement.GetSpan());
 25     destructorCallGenerated = false;
 26     lastInstructionWasRet = false;
 27     basicBlockOpen = false;
 28     SetTarget(&boundReturnStatement);
 29     BoundFunctionCall* returnFunctionCall = boundReturnStatement.ReturnFunctionCall();
 30     if (returnFunctionCall)
 31     {
 32         returnFunctionCall->Accept(*this);
 33         llvm::Value* returnValue = stack.Pop();
 34         if (sequenceSecond)
 35         {
 36             sequenceSecond->SetGenerated();
 37             sequenceSecond->Accept(*this);
 38         }
 39         ExitBlocks(nullptr);
 40         CreateExitFunctionCall();
 41         builder.CreateRet(returnValue);
 42         lastInstructionWasRet = true;
 43     }
 44     else
 45     {
 46         ExitBlocks(nullptr);
 47         CreateExitFunctionCall();
 48         builder.CreateRetVoid();
 49         lastInstructionWasRet = true;
 50     }
 51     BoundCompoundStatement* body = currentFunction->Body();
 52     BoundStatement* lastStatement = nullptr;
 53     if (!body->Statements().empty())
 54     {
 55         lastStatement = body->Statements().back().get();
 56     }
 57     if (lastStatement && lastStatement != &boundReturnStatement)
 58     {
 59         llvm::BasicBlock* nextBlock = llvm::BasicBlock::Create(context, "next", function);
 60         SetCurrentBasicBlock(nextBlock);
 61         basicBlockOpen = true;
 62         lastInstructionWasRet = false;
 63     }
 64 }

163 void LinuxEmitter::Visit(BoundTryStatement& boundTryStatement)
164 {
165     ClearFlags();
166     SetTarget(&boundTryStatement);
167     llvm::BasicBlock* prevHandlerBlock = handlerBlock;
168     llvm::BasicBlock* prevCleanupBlock = cleanupBlock;
169     handlerBlock = llvm::BasicBlock::Create(context, "lpad", function);
170     cleanupBlock = nullptr;
171     boundTryStatement.TryBlock()->Accept(*this);
172     llvm::BasicBlock* nextTarget = llvm::BasicBlock::Create(context, "next", function);
173     builder.CreateBr(nextTarget);
174     SetCurrentBasicBlock(handlerBlock);
175     handlerBlock = prevHandlerBlock;
176     std::vector<llvm::Type*> lpElemTypes;
177     lpElemTypes.push_back(builder.getInt8PtrTy());
178     lpElemTypes.push_back(builder.getInt32Ty());
179     llvm::StructType* lpType = llvm::StructType::get(context, lpElemTypes);
180     llvm::LandingPadInst* lp = builder.CreateLandingPad(lpType, 1);
181     std::vector<llvm::Type*> exTypeInfoElemTypes;
182     exTypeInfoElemTypes.push_back(builder.getInt8PtrTy());
183     exTypeInfoElemTypes.push_back(builder.getInt8PtrTy());
184     llvm::StructType* exTypeInfoType = llvm::StructType::get(context, exTypeInfoElemTypes);
185     llvm::Constant* exceptionTypeId = compileUnitModule->getOrInsertGlobal("_ZTIN6cmajor2eh9ExceptionE", exTypeInfoType);
186     lp->addClause(      llvm::cast<llvm::Constant>(builder.CreateBitCast(exceptionTypeId, builder.getInt8PtrTy())));
187     std::vector<unsigned int> exPtrIndex;
188     exPtrIndex.push_back(0);
189     llvm::Value* exPtr = builder.CreateExtractValue(lp, exPtrIndex);
190     llvm::AllocaInst* exPtrAlloca = new llvm::AllocaInst(builder.getInt8PtrTy(), 0);
191     InsertAllocaIntoEntryBlock(exPtrAlloca);
192     builder.CreateStore(exPtr, exPtrAlloca);
193     std::vector<unsigned int> exIdIndex;
194     exIdIndex.push_back(1);
195     llvm::Value* exId = builder.CreateExtractValue(lp, exIdIndex);
196     llvm::AllocaInst* exIdAlloca = new llvm::AllocaInst(builder.getInt32Ty(), 0);
197     builder.CreateStore(exId, exIdAlloca);
198     InsertAllocaIntoEntryBlock(exIdAlloca);
199     llvm::BasicBlock* testBlock = llvm::BasicBlock::Create(context, "test", function);
200     builder.CreateBr(testBlock);
201     SetCurrentBasicBlock(testBlock);
202     llvm::Value* loadedExId = builder.CreateLoad(exIdAlloca);
203     std::vector<llvm::Type*> llvmEhParamTypes;
204     llvmEhParamTypes.push_back(builder.getInt8PtrTy());
205     llvm::FunctionType* llvmEHTypeIdForType = llvm::FunctionType::get(builder.getInt32Ty(), llvmEhParamTypes, false);
206     llvm::Function* llvmEHTypeIdFor =       llvm::cast<llvm::Function>(compileUnitModule->getOrInsertFunction("llvm.eh.typeid.for", llvmEHTypeIdForType));
207     ArgVector llvmEHTypeIdForArgs;
208     llvmEHTypeIdForArgs.push_back(      llvm::cast<llvm::Constant>(builder.CreateBitCast(exceptionTypeId, builder.getInt8PtrTy())));
209     llvm::Value* ehSelector = builder.CreateCall(llvmEHTypeIdFor, llvmEHTypeIdForArgs);
210     llvm::Value* match = builder.CreateICmpEQ(loadedExId, ehSelector);
211     llvm::BasicBlock* beginCatchBlock = llvm::BasicBlock::Create(context, "handlers", function);
212     llvm::BasicBlock* resumeBlock = llvm::BasicBlock::Create(context, "resume", function);
213     builder.CreateCondBr(match, beginCatchBlock, resumeBlock);
214     SetCurrentBasicBlock(beginCatchBlock);
215     std::vector<llvm::Type*> cxaBeginCatchParamTypes;
216     cxaBeginCatchParamTypes.push_back(builder.getInt8PtrTy());
217     llvm::FunctionType* cxaBeginCatchType = llvm::FunctionType::get(builder.getInt8PtrTy(), cxaBeginCatchParamTypes, false);
218     llvm::Function* cxaBeginCatch =       llvm::cast<llvm::Function>(compileUnitModule->getOrInsertFunction("__cxa_begin_catch", cxaBeginCatchType));
219     ArgVector cxaBeginCatchArgs;
220     llvm::Value* loadedExPtr = builder.CreateLoad(exPtrAlloca);
221     cxaBeginCatchArgs.push_back(loadedExPtr);
222     llvm::Value* cxaBeginCatchValue = builder.CreateCall(cxaBeginCatch, cxaBeginCatchArgs);
223     llvm::BasicBlock* catchTarget = llvm::BasicBlock::Create(context, "catch", function);
224     builder.CreateBr(catchTarget);
225     int n = boundTryStatement.Catches().size();
226     for (int i = 0; i < n; ++i)
227     {
228         const std::unique_ptr<BoundCatchStatement>& boundCatchStatement = boundTryStatement.Catches()[i];
229         SetCurrentBasicBlock(catchTarget);
230         std::vector<llvm::Type*> handleExceptionParamTypes;
231         handleExceptionParamTypes.push_back(builder.getInt8PtrTy());
232         llvm::FunctionType* handleExceptionFunctionType = llvm::FunctionType::get(builder.getInt1Ty(), handleExceptionParamTypes, false);
233         ArgVector handleExceptionArgs;
234         UuidValue uuidValue(boundCatchStatement->GetSpan(), boundCatchStatement->CatchedTypeUuidId());
235         llvm::Value* catchTypeIdValue = uuidValue.IrValue(*this);
236         handleExceptionArgs.push_back(catchTypeIdValue);
237         llvm::Function* handleException =       llvm::cast<llvm::Function>(compileUnitModule->getOrInsertFunction("RtHandleException", handleExceptionFunctionType));
238         llvm::Value* handleThisEx = builder.CreateCall(handleException, handleExceptionArgs);
239         llvm::BasicBlock* nextHandlerTarget = nullptr;
240         if (i < n - 1)
241         {
242             catchTarget = llvm::BasicBlock::Create(context, "catch", function);
243             nextHandlerTarget = catchTarget;
244         }
245         else
246         {
247             nextHandlerTarget = resumeBlock;
248         }
249         llvm::BasicBlock* thisHandlerTarget = llvm::BasicBlock::Create(context, "handler", function);
250         builder.CreateCondBr(handleThisEx, thisHandlerTarget, nextHandlerTarget);
251         SetCurrentBasicBlock(thisHandlerTarget);
252         boundCatchStatement->CatchBlock()->Accept(*this);
253         std::vector<llvm::Type*> cxaEndCatchParamTypes;
254         llvm::FunctionType* cxaEndCatchType = llvm::FunctionType::get(builder.getVoidTy(), cxaEndCatchParamTypes, false);
255         llvm::Function* cxaEndCatch =       llvm::cast<llvm::Function>(compileUnitModule->getOrInsertFunction("__cxa_end_catch", cxaEndCatchType));
256         ArgVector cxaEndCatchArgs;
257         llvm::Value* cxaEndCatchValue = builder.CreateCall(cxaEndCatch, cxaEndCatchArgs);
258         builder.CreateBr(nextTarget);
259     }
260     SetCurrentBasicBlock(resumeBlock);
261     llvm::Value* loadedExPtrForResume = builder.CreateLoad(exPtrAlloca);
262     llvm::Value* loadedExIdForResume = builder.CreateLoad(exIdAlloca);
263     llvm::Value* resume1 = builder.CreateInsertValue(llvm::UndefValue::get(lpType), loadedExPtrForResume, exPtrIndex);
264     llvm::Value* resume2 = builder.CreateInsertValue(resume1, loadedExIdForResume, exIdIndex);
265     builder.CreateResume(resume2);
266     SetCurrentBasicBlock(nextTarget);
267     basicBlockOpen = true;
268 }

284 void LinuxEmitter::CreateCleanup()
285 {
286     cleanupBlock = llvm::BasicBlock::Create(context, "cleanup", function);
287     BoundCompoundStatement* targetBlock = nullptr;
288     BoundStatement* parent = currentBlock->Parent();
289     while (parent && parent->GetBoundNodeType() != BoundNodeType::boundTryStatement)
290     {
291         parent = parent->Parent();
292     }
293     if (parent)
294     {
295         targetBlock = parent->Block();
296     }
297     Cleanup* cleanup = new Cleanup(cleanupBlock, handlerBlock, currentPad);
298     int n = blocks.size();
299     for (int i = n - 1; i >= 0; --i)
300     {
301         BoundCompoundStatement* block = blocks[i];
302         if (block == targetBlock)
303         {
304             break;
305         }
306         auto it = blockDestructionMap.find(block);
307         if (it != blockDestructionMap.cend())
308         {
309             std::vector<std::unique_ptr<BoundFunctionCall>>& destructorCallVec = it->second;
310             int nd = destructorCallVec.size();
311             for (int i = nd - 1; i >= 0; --i)
312             {
313                 std::unique_ptr<BoundFunctionCall>& destructorCall = destructorCallVec[i];
314                 if (destructorCall)
315                 {
316                     cleanup->destructors.push_back(     std::unique_ptr<BoundFunctionCall>(static_cast<BoundFunctionCall*>(destructorCall->Clone())));
317                 }
318             }
319         }
320     }
321     cleanups.push_back(     std::unique_ptr<Cleanup>(cleanup));
322     newCleanupNeeded = false;
323 }

325 void LinuxEmitter::GenerateCodeForCleanups()
326 {
327     for (const std::unique_ptr<Cleanup>& cleanup : cleanups)
328     {
329         SetCurrentBasicBlock(cleanup->cleanupBlock);
330         std::vector<llvm::Type*> lpElemTypes;
331         lpElemTypes.push_back(builder.getInt8PtrTy());
332         lpElemTypes.push_back(builder.getInt32Ty());
333         llvm::StructType* lpType = llvm::StructType::get(context, lpElemTypes);
334         llvm::LandingPadInst* lp = builder.CreateLandingPad(lpType, 0);
335         lp->setCleanup(true);
336         for (const std::unique_ptr<BoundFunctionCall>& destructorCall : cleanup->destructors)
337         {
338             destructorCall->Accept(*this);
339         }
340         builder.CreateResume(lp);
341     }
342 }
