 35 bool BoundAtomicConstraint::Subsume(BoundConstraint* that) const
 36 {
 37     if (that->IsBinaryConstraint())
 38     {
 39         BoundBinaryConstraint* thatBinaryConstraint = static_cast<BoundBinaryConstraint*>(that);
 40         BoundConstraint* thatLeft = thatBinaryConstraint->Left();
 41         BoundConstraint* thatRight = thatBinaryConstraint->Right();
 42         bool subsumeLeft = Subsume(thatLeft);
 43         bool subsumeRight = Subsume(thatRight);
 44         if (that->GetBoundNodeType() == BoundNodeType::boundConjunctiveConstraint)
 45         {
 46             return subsumeLeft && subsumeRight;
 47         }
 48         else if (that->GetBoundNodeType() == BoundNodeType::boundDisjunctiveConstraint)
 49         {
 50             return subsumeLeft || subsumeRight;
 51         }
 52         else
 53         {
 54             Assert(false, "unknown binary constraint type");
 55             return false;
 56         }
 57     }
 58     else if (that->GetBoundNodeType() == BoundNodeType::boundAtomicConstraint)
 59     {
 60         BoundAtomicConstraint* thatAtomic = static_cast<BoundAtomicConstraint*>(that);
 61         if (satisfied && !thatAtomic->Satisfied())
 62         {
 63             return true;
 64         }
 65         else if (!satisfied && thatAtomic->Satisfied())
 66         {
 67             return false;
 68         }
 69         else
 70         {
 71             if (concept && !thatAtomic->concept)
 72             {
 73                 return true;
 74             }
 75             else if (!concept && thatAtomic->concept)
 76             {
 77                 return false;
 78             }
 79             else if (!concept && !thatAtomic->concept)
 80             {
 81                 return true;
 82             }
 83             else
 84             {
 85                 if (concept == thatAtomic->concept)
 86                 {
 87                     return true;
 88                 }
 89                 ConceptSymbol* refinedConcept = concept->RefinedConcept();
 90                 while (refinedConcept)
 91                 {
 92                     if (refinedConcept == thatAtomic->concept)
 93                     {
 94                         return true;
 95                     }
 96                     else
 97                     {
 98                         refinedConcept = refinedConcept->RefinedConcept();
 99                     }
100                 }
101                 return false;
102             }
103         }
104     }
105     else
106     {
107         return false;
108     }
109 }

134 bool BoundDisjunctiveConstraint::Subsume(BoundConstraint* that) const
135 {
136     BoundConstraint* left = Left();
137     BoundConstraint* right = Right();
138     if (that->IsBinaryConstraint())
139     {
140         BoundBinaryConstraint* thatBinaryConstraint = static_cast<BoundBinaryConstraint*>(that);
141         BoundConstraint* thatLeft = thatBinaryConstraint->Left();
142         BoundConstraint* thatRight = thatBinaryConstraint->Right();
143         bool leftSubsumeThatLeft = left->Subsume(thatLeft);
144         bool leftSubsumeThatRight = left->Subsume(thatRight);
145         bool rightSubsumeThatLeft = right->Subsume(thatLeft);
146         bool rightSubsumeThatRight = right->Subsume(thatRight);
147         bool leftSubsumeThatLeftOrThatRight = leftSubsumeThatLeft || leftSubsumeThatRight;
148         bool rightSubsumeThatLeftOrThatRight = rightSubsumeThatLeft || rightSubsumeThatRight;
149         if (that->GetBoundNodeType() == BoundNodeType::boundConjunctiveConstraint)
150         {
151             return leftSubsumeThatLeftOrThatRight && rightSubsumeThatLeftOrThatRight;
152         }
153         else if (that->GetBoundNodeType() == BoundNodeType::boundDisjunctiveConstraint)
154         {
155             return leftSubsumeThatLeftOrThatRight || rightSubsumeThatLeftOrThatRight;
156         }
157         else
158         {
159             Assert(false, "unknown binary constraint type");
160             return false;
161         }
162     }
163     else
164     {
165         bool leftSubsumeThat = left->Subsume(that);
166         bool rightSubsumeThat = right->Subsume(that);
167         return leftSubsumeThat && rightSubsumeThat;
168     }
169 }

202         bool rightSubsumeThatRight = right->Subsume(thatRight);
203         bool leftOrRightSubsumeThatLeft = leftSubsumeThatLeft || rightSubsumeThatLeft;

208         }

212         }
213         else
214         {
215             Assert(false, "unknown binary constraint type");
216             return false;
217         }
218     }

237 std::u32string MakeBoundConceptName(ConceptSymbol* conceptSymbol, const std::vector<TypeSymbol*>& typeArguments)
238 {
239     std::u32string s = conceptSymbol->GroupName();
240     s.append(1, '<');
241     int n = typeArguments.size();
242     for (int i = 0; i < n; ++i)
243     {
244         if (i > 0)
245         {
246             s.append(U", ");
247         }
248         TypeSymbol* typeSymbol = typeArguments[i];
249         s.append(typeSymbol->FullName());
250     }
251     s.append(1, '>');
252     return s;
253 }

285 bool operator==(const BoundConceptKey& left, const BoundConceptKey& right)
286 {
287     if (left.conceptSymbol != right.conceptSymbol) return false;
288     if (left.Arity() != right.Arity()) return false;
289     int n = left.Arity();
290     for (int i = 0; i < n; ++i)
291     {
292         if (!TypesEqual(left.typeArguments[i], right.typeArguments[i])) return false;
293     }
294     return true;
295 }
