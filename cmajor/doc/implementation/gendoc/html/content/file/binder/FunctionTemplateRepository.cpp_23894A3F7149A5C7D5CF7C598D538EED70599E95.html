 20 bool operator==(const FunctionTemplateKey& left, const FunctionTemplateKey& right)
 21 {
 22     if (left.functionTemplate != right.functionTemplate) return false;
 23     if (left.templateArgumentTypes.size() != right.templateArgumentTypes.size()) return false;
 24     int n = left.templateArgumentTypes.size();
 25     for (int i = 0; i < n; ++i)
 26     {
 27         if (!TypesEqual(left.templateArgumentTypes[i], right.templateArgumentTypes[i])) return false;
 28     }
 29     return true;
 30 }
 31 
 32 bool operator!=(const FunctionTemplateKey& left, const FunctionTemplateKey& right)
 33 {
 34     return !(left == right);
 35 }

 41 FunctionSymbol* FunctionTemplateRepository::Instantiate(FunctionSymbol* functionTemplate, const std::unordered_map<TemplateParameterSymbol*, TypeSymbol*>& templateParameterMapping, 
 42     const Span& span)
 43 {
 44     std::vector<TypeSymbol*> templateArgumentTypes;
 45     for (TemplateParameterSymbol* templateParameter : functionTemplate->TemplateParameters())
 46     {
 47         auto it = templateParameterMapping.find(templateParameter);
 48         if (it != templateParameterMapping.cend())
 49         {
 50             TypeSymbol* templateArgumentType = it->second;
 51             templateArgumentTypes.push_back(templateArgumentType);
 52         }
 53         else
 54         {
 55             throw Exception(&boundCompileUnit.GetModule(), "template parameter type not found", span);
 56         }
 57     }
 58     FunctionTemplateKey key(functionTemplate, templateArgumentTypes);
 59     auto it = functionTemplateMap.find(key);
 60     if (it != functionTemplateMap.cend())
 61     {
 62         return it->second;
 63     }
 64     SymbolTable& symbolTable = boundCompileUnit.GetSymbolTable();
 65     Node* node = symbolTable.GetNodeNoThrow(functionTemplate);
 66     if (!node)
 67     {
 68         functionTemplate->ReadAstNodes();
 69         node = functionTemplate->GetFunctionNode();
 70         Assert(node, "function node not read");
 71     }
 72     Assert(node->GetNodeType() == NodeType::functionNode, "function node expected");
 73     instantiated.insert(functionTemplate);
 74     FunctionNode* functionNode = static_cast<FunctionNode*>(node);
 75     std::unique_ptr<NamespaceNode> globalNs(new NamespaceNode(functionTemplate->GetSpan(), new IdentifierNode(functionTemplate->GetSpan(), U"")));
 76     NamespaceNode* currentNs = globalNs.get();
 77     CloneContext cloneContext;
 78     cloneContext.SetInstantiateFunctionNode();
 79     int n = functionTemplate->UsingNodes().Count();
 80     for (int i = 0; i < n; ++i)
 81     {
 82         Node* usingNode = functionTemplate->UsingNodes()[i];
 83         globalNs->AddMember(usingNode->Clone(cloneContext));
 84     }
 85     bool fileScopeAdded = false;
 86     if (!functionTemplate->Ns()->IsGlobalNamespace())
 87     {
 88         FileScope* primaryFileScope = new FileScope(&boundCompileUnit.GetModule());
 89         primaryFileScope->AddContainerScope(functionTemplate->Ns()->GetContainerScope());
 90         boundCompileUnit.AddFileScope(primaryFileScope);
 91         fileScopeAdded = true;
 92         std::u32string fullNsName = functionTemplate->Ns()->FullName();
 93         std::vector<std::u32string> nsComponents = Split(fullNsName, '.');
 94         for (const std::u32string& nsComponent : nsComponents)
 95         {
 96             NamespaceNode* nsNode = new NamespaceNode(functionTemplate->GetSpan(), new IdentifierNode(functionTemplate->GetSpan(), nsComponent));
 97             currentNs->AddMember(nsNode);
 98             currentNs = nsNode;
 99         }
100     }
101     FunctionNode* functionInstanceNode = static_cast<FunctionNode*>(functionNode->Clone(cloneContext));
102     currentNs->AddMember(functionInstanceNode);
103     symbolTable.SetCurrentCompileUnit(boundCompileUnit.GetCompileUnitNode());
104     SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
105     globalNs->Accept(symbolCreatorVisitor);
106     Symbol* symbol = symbolTable.GetSymbol(functionInstanceNode);
107     Assert(symbol->GetSymbolType() == SymbolType::functionSymbol, "function symbol expected");
108     FunctionSymbol* functionSymbol = static_cast<FunctionSymbol*>(symbol);
109     functionSymbol->SetLinkOnceOdrLinkage();
110     functionSymbol->SetTemplateSpecialization();
111     functionSymbol->SetFunctionTemplate(functionTemplate);
112     functionSymbol->SetTemplateArgumentTypes(templateArgumentTypes);
113     functionTemplateMap[key] = functionSymbol;
114     for (TemplateParameterSymbol* templateParameter : functionTemplate->TemplateParameters())
115     {
116         auto it = templateParameterMapping.find(templateParameter);
117         if (it != templateParameterMapping.cend())
118         {
119             TypeSymbol* boundType = it->second;
120             BoundTemplateParameterSymbol* boundTemplateParameter = new BoundTemplateParameterSymbol(span, templateParameter->Name());
121             boundTemplateParameter->SetType(boundType);
122             functionSymbol->AddMember(boundTemplateParameter);
123         }
124         else
125         {
126             throw Exception(&boundCompileUnit.GetModule(), "template parameter type not found", span);
127         }
128     }
129     TypeBinder typeBinder(boundCompileUnit);
130     globalNs->Accept(typeBinder);
131     StatementBinder statementBinder(boundCompileUnit);
132     globalNs->Accept(statementBinder);
133     if (fileScopeAdded)
134     {
135         boundCompileUnit.RemoveLastFileScope();
136     }
137     functionSymbol->SetGlobalNs(std::move(globalNs));
138     return functionSymbol;
139 }
