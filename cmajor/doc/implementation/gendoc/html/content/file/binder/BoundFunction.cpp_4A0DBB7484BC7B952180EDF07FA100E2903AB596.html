13 BoundFunction::BoundFunction(Module* module_, FunctionSymbol* functionSymbol_) :
14     BoundNode(module_, functionSymbol_->GetSpan(), BoundNodeType::boundFunction), functionSymbol(functionSymbol_), hasGotos(false)
15 {
16 }
17 
18 void BoundFunction::Load(Emitter& emitter, OperationFlags flags)
19 {
20     throw Exception(GetModule(), "cannot load from function", GetSpan());
21 }
22 
23 void BoundFunction::Store(Emitter& emitter, OperationFlags flags)
24 {
25     throw Exception(GetModule(), "cannot store to function", GetSpan());
26 }
27 
28 void BoundFunction::Accept(BoundNodeVisitor& visitor)
29 {
30     visitor.Visit(*this);
31 }
32 
33 void BoundFunction::SetBody(std::unique_ptr<BoundCompoundStatement>&& body_)
34 {
35     body = std::move(body_);
36 }
37 
38 void BoundFunction::AddTemporaryDestructorCall(std::unique_ptr<BoundFunctionCall>&& destructorCall)
39 {
40     temporaryDestructorCalls.push_back(std::move(destructorCall));
41 }
42 
43 void BoundFunction::MoveTemporaryDestructorCallsTo(BoundExpression& expression)
44 {
45     for (std::unique_ptr<BoundFunctionCall>& destructorCall : temporaryDestructorCalls)
46     {
47         expression.AddTemporaryDestructorCall(std::move(destructorCall));
48     }
49     temporaryDestructorCalls.clear();
50 }
51 
52 void BoundFunction::AddLabeledStatement(BoundStatement* labeledStatement)
53 {
54     labeledStatements.push_back(labeledStatement);
55 }
