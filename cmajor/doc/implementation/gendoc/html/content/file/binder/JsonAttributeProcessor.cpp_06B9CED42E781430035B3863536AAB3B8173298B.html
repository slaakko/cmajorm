 33 void JsonAttributeProcessor::TypeCheck(Attribute* attribute, Symbol* symbol)
 34 {
 35     switch (symbol->GetSymbolType())
 36     {
 37         case SymbolType::classTypeSymbol: case SymbolType::classTemplateSpecializationSymbol: case SymbolType::memberVariableSymbol:
 38         {
 39             if (attribute->Value() == U"true" || attribute->Value() == U"false")
 40             {
 41                 if (symbol->IsClassTypeSymbol())
 42                 {
 43                     ClassTypeSymbol* classTypeSymbol = static_cast<ClassTypeSymbol*>(symbol);
 44                     ClassTypeSymbol* baseClass = classTypeSymbol->BaseClass();
 45                     if (baseClass)
 46                     {
 47                         bool baseClassHasJsonAttribute = false;
 48                         Attributes* baseClassAttributes = baseClass->GetAttributes();
 49                         if (baseClassAttributes)
 50                         {
 51                             Attribute* jsonAttribute = baseClassAttributes->GetAttribute(U"json");
 52                             if (jsonAttribute)
 53                             {
 54                                 baseClassHasJsonAttribute = true;
 55                             }
 56                         }
 57                         if (!baseClassHasJsonAttribute)
 58                         {
 59                             Warning warning(module->GetCurrentProjectName(), "base class '" + ToUtf8(baseClass->FullName()) + "' of json-attributed class '" + 
 60                                 ToUtf8(classTypeSymbol->FullName()) + "' does not explicitly declare 'json' attribute value to \"true\" or \"false\"");
 61                             warning.SetDefined(classTypeSymbol->GetSpan());
 62                             std::vector<Span> references;
 63                             references.push_back(baseClass->GetSpan());
 64                             warning.SetReferences(references);
 65                             module->WarningCollection().AddWarning(warning);
 66                         }
 67                     }
 68                 }
 69                 return;
 70             }
 71             else
 72             {
 73                 throw Exception(module, "unknown attribute value '" + ToUtf8(attribute->Value()) + "' for attribute '" + ToUtf8(attribute->Name()) + "'", attribute->GetSpan());
 74             }
 75         }
 76     }
 77     AttributeProcessor::TypeCheck(attribute, symbol);
 78 }

 80 void JsonAttributeProcessor::GenerateSymbols(Attribute* attribute, Symbol* symbol, BoundCompileUnit& boundCompileUnit, ContainerScope* containerScope)
 81 {

 91             GenerateJsonConstructorSymbol(attribute, classTypeSymbol);

 94         }
 95     }

 98 void JsonAttributeProcessor::CheckMemberVariableJsonFieldNames(ClassTypeSymbol* classTypeSymbol, std::map<std::u32string, Attribute*>& memberVariableFieldNames)
 99 {
100     if (classTypeSymbol->BaseClass())
101     {
102         CheckMemberVariableJsonFieldNames(classTypeSymbol->BaseClass(), memberVariableFieldNames);
103     }
104     for (MemberVariableSymbol* memberVariableSymbol : classTypeSymbol->MemberVariables())
105     {
106         Attributes* attributes = memberVariableSymbol->GetAttributes();
107         if (attributes)
108         {
109             Attribute* jsonAttribute = attributes->GetAttribute(U"json");
110             if (jsonAttribute)
111             {
112                 if (jsonAttribute->Value() == U"false")
113                 {
114                     continue;
115                 }
116             }
117             Attribute* jsonFieldNameAttribute = attributes->GetAttribute(U"jsonFieldName");
118             if (jsonFieldNameAttribute)
119             {
120                 auto it = memberVariableFieldNames.find(jsonFieldNameAttribute->Value());
121                 if (it != memberVariableFieldNames.cend())
122                 {
123                     Attribute* prev = it->second;
124                     throw Exception(module, "error in JSON field name generation: 'jsonFieldName' attribute not unique among member variable names of the current class and its base classes", 
125                         jsonFieldNameAttribute->GetSpan(), prev->GetSpan());
126                 }
127                 memberVariableFieldNames[jsonFieldNameAttribute->Value()] = jsonFieldNameAttribute;
128             }
129             else
130             {
131                 throw Exception(module, "internal error in JSON field name generation: 'jsonFieldName' attribute not found", memberVariableSymbol->GetSpan());
132             }
133         }
134         else
135         {
136             throw Exception(module, "internal error in JSON field name generation: attributes not found", memberVariableSymbol->GetSpan());
137         }
138     }
139 }

150             if (jsonAttribute)

153                 {

158             if (jsonFieldNameAttribute)

160                 hasJsonFieldAttribute = true;
161             }

175 void JsonAttributeProcessor::GenerateJsonCreatorFunctionSymbol(Attribute* attribute, ClassTypeSymbol* classTypeSymbol)
176 {
177     MemberFunctionSymbol* jsonCreatorFunctionSymbol = new MemberFunctionSymbol(attribute->GetSpan(), U"Create");
178     jsonCreatorFunctionSymbol->SetGroupName(U"Create");
179     jsonCreatorFunctionSymbol->SetSymbolTable(&module->GetSymbolTable());
180     jsonCreatorFunctionSymbol->SetModule(module);
181     jsonCreatorFunctionSymbol->SetOriginalModule(module);
182     module->GetSymbolTable().SetFunctionIdFor(jsonCreatorFunctionSymbol);
183     jsonCreatorFunctionSymbol->SetAccess(SymbolAccess::public_);
184     jsonCreatorFunctionSymbol->SetStatic();
185     Symbol* jsonValue = classTypeSymbol->GetSymbolTable()->GlobalNs().GetContainerScope()->Lookup(U"System.Json.JsonValue");
186     if (!jsonValue || jsonValue->GetSymbolType() != SymbolType::classGroupTypeSymbol)
187     {
188         throw Exception(module, "System.Json.JsonValue class not found from the symbol table", attribute->GetSpan(), classTypeSymbol->GetSpan());
189     }
190     ClassGroupTypeSymbol* jsonValueGroup = static_cast<ClassGroupTypeSymbol*>(jsonValue);
191     ClassTypeSymbol* jsonValueClass = jsonValueGroup->GetClass(0);
192     if (!jsonValueClass)
193     {
194         throw Exception(module, "System.Json.JsonValue class not found from the symbol table", attribute->GetSpan(), classTypeSymbol->GetSpan());
195     }
196     ParameterSymbol* jsonValueParam = new ParameterSymbol(attribute->GetSpan(), U"@value");
197     jsonValueParam->SetType(jsonValueClass->AddPointer(attribute->GetSpan()));
198     jsonCreatorFunctionSymbol->AddMember(jsonValueParam);
199     jsonCreatorFunctionSymbol->SetReturnType(classTypeSymbol->GetSymbolTable()->GetTypeByName(U"void")->AddPointer(attribute->GetSpan()));
200     classTypeSymbol->AddMember(jsonCreatorFunctionSymbol);
201     jsonCreatorFunctionSymbol->ComputeName();
202     jsonCreatorMap[classTypeSymbol] = jsonCreatorFunctionSymbol;
203 }

343 void JsonAttributeProcessor::GenerateImplementation(Attribute* attribute, Symbol* symbol, StatementBinder* statementBinder)
344 {
345     if (symbol->IsClassTypeSymbol())
346     {
347         ClassTypeSymbol* classTypeSymbol = static_cast<ClassTypeSymbol*>(symbol);
348         auto it = jsonConstructorMap.find(symbol);
349         if (it != jsonConstructorMap.cend())
350         {
351             ConstructorSymbol* jsonConstructorSymbol = it->second;
352             GenerateJsonConstructorImplementation(attribute, classTypeSymbol, jsonConstructorSymbol, statementBinder);
353         }
354         else
355         {
356             throw Exception(module, "internal error in JSON attribute implementation: constructor symbol for symbol '" + ToUtf8(symbol->FullName()) + "' not found", attribute->GetSpan());
357         }
358         auto it1 = jsonCreatorMap.find(symbol);
359         if (it1 != jsonCreatorMap.cend())
360         {
361             MemberFunctionSymbol* jsonCreatorSymbol = it1->second;
362             GenerateJsonCreatorImplementation(attribute, classTypeSymbol, jsonCreatorSymbol, statementBinder);
363         }
364         else
365         {
366             throw Exception(module, "internal error in JSON attribute implementation: Creator function symbol for symbol '" + ToUtf8(symbol->FullName()) + "' not found", attribute->GetSpan());
367         }
368         auto it2 = toJsonJsonObjectMemberFunctionSymbolMap.find(symbol);
369         if (it2 != toJsonJsonObjectMemberFunctionSymbolMap.cend())
370         {
371             MemberFunctionSymbol* toJsonJsonObjectMemberFunctionSymbol = it2->second;
372             GenerateToJsonJsonObjectImplementation(attribute, classTypeSymbol, toJsonJsonObjectMemberFunctionSymbol, statementBinder);
373         }
374         else
375         {
376             throw Exception(module, "internal error in JSON attribute implementation: member function 'ToJson' symbol for symbol '" + ToUtf8(symbol->FullName()) + "' not found", attribute->GetSpan());
377         }
378         auto it3 = toJsonObjectMemberFunctionSymbolMap.find(symbol);
379         if (it3 != toJsonObjectMemberFunctionSymbolMap.cend())
380         {
381             MemberFunctionSymbol* toJsonMemberFunctionSymbol = it3->second;
382             GenerateToJsonImplementation(attribute, classTypeSymbol, toJsonMemberFunctionSymbol, statementBinder);
383         }
384         else
385         {
386             throw Exception(module, "internal error in JSON attribute implementation: member function 'ToJson' symbol for symbol '" + ToUtf8(symbol->FullName()) + "' not found", attribute->GetSpan());
387         }
388         SymbolTable* symbolTable = classTypeSymbol->GetSymbolTable();
389         symbolTable->AddJsonClass(classTypeSymbol->FullName());
390     }
391 }

393 void JsonAttributeProcessor::GenerateJsonCreatorImplementation(Attribute* attribute, ClassTypeSymbol* classTypeSymbol, MemberFunctionSymbol* jsonCreatorFunctionSymbol, StatementBinder* statementBinder)
394 {
395     try
396     {
397         FileScope* fileScope = new FileScope(&statementBinder->GetBoundCompileUnit().GetModule());
398         Symbol* jsonValue = classTypeSymbol->GetSymbolTable()->GlobalNs().GetContainerScope()->Lookup(U"System.Json.JsonValue");
399         if (jsonValue)
400         {
401             fileScope->AddContainerScope(jsonValue->Ns()->GetContainerScope());
402         }
403         statementBinder->GetBoundCompileUnit().AddFileScope(fileScope);
404         std::unique_ptr<BoundFunction> boundFunction(new BoundFunction(module, jsonCreatorFunctionSymbol));
405         const Span& span = attribute->GetSpan();
406         CompoundStatementNode compoundStatementNode(span);
407         compoundStatementNode.SetEndBraceSpan(span);
408         NewNode* newNode = new NewNode(span, new IdentifierNode(span, classTypeSymbol->FullName()));
409         newNode->AddArgument(new IdentifierNode(span, U"@value"));
410         ReturnStatementNode* returnStatementNode = new ReturnStatementNode(span, newNode);
411         compoundStatementNode.AddStatement(returnStatementNode);
412         SymbolTable& symbolTable = statementBinder->GetBoundCompileUnit().GetSymbolTable();
413         symbolTable.BeginContainer(jsonCreatorFunctionSymbol);
414         SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
415         compoundStatementNode.Accept(symbolCreatorVisitor);
416         symbolTable.EndContainer();
417         TypeBinder typeBinder(statementBinder->GetBoundCompileUnit());
418         ContainerScope* containerScope = statementBinder->GetContainerScope();
419         typeBinder.SetContainerScope(containerScope);
420         statementBinder->SetContainerScope(jsonCreatorFunctionSymbol->GetContainerScope());
421         typeBinder.SetCurrentFunctionSymbol(jsonCreatorFunctionSymbol);
422         compoundStatementNode.Accept(typeBinder);
423         BoundFunction* prevFunction = statementBinder->CurrentFunction();
424         statementBinder->SetCurrentFunction(boundFunction.get());
425         compoundStatementNode.Accept(*statementBinder);
426         statementBinder->SetContainerScope(containerScope);
427         BoundStatement* boundStatement = statementBinder->ReleaseStatement();
428         Assert(boundStatement->GetBoundNodeType() == BoundNodeType::boundCompoundStatement, "bound compound statement expected");
429         BoundCompoundStatement* compoundStatement = static_cast<BoundCompoundStatement*>(boundStatement);
430         boundFunction->SetBody(     std::unique_ptr<BoundCompoundStatement>(compoundStatement));
431         statementBinder->CurrentClass()->AddMember(std::move(boundFunction));
432         statementBinder->SetCurrentFunction(prevFunction);
433         statementBinder->GetBoundCompileUnit().RemoveLastFileScope();
434     }
435     catch (const Exception& ex;)
436     {
437         std::vector<Span> references;
438         references.push_back(ex.Defined());
439         references.insert(references.end(), ex.References().begin(), ex.References().end());
440         throw Exception(module, "error in JSON attribute generation: could not create JSON Create() function for class '" + ToUtf8(classTypeSymbol->FullName()) + "': " + ex.Message(), 
441             classTypeSymbol->GetSpan(), references);
442     }
443 }

445 void JsonAttributeProcessor::GenerateJsonConstructorImplementation(Attribute* attribute, ClassTypeSymbol* classTypeSymbol, ConstructorSymbol* jsonConstructorSymbol, StatementBinder* statementBinder)
446 {
447     try
448     {
449         FileScope* fileScope = new FileScope(&statementBinder->GetBoundCompileUnit().GetModule());
450         Symbol* jsonValue = classTypeSymbol->GetSymbolTable()->GlobalNs().GetContainerScope()->Lookup(U"System.Json.JsonValue");
451         if (jsonValue)
452         {
453             fileScope->AddContainerScope(jsonValue->Ns()->GetContainerScope());
454         }
455         statementBinder->GetBoundCompileUnit().AddFileScope(fileScope);
456         std::unique_ptr<BoundFunction> boundFunction(new BoundFunction(module, jsonConstructorSymbol));
457         const Span& span = attribute->GetSpan();
458         ConstructorNode constructorNode(span);
459         CompoundStatementNode compoundStatementNode(span);
460         compoundStatementNode.SetEndBraceSpan(span);
461         ClassTypeSymbol* baseClass = classTypeSymbol->BaseClass();
462         if (baseClass)
463         {
464             Attributes* attributes = baseClass->GetAttributes();
465             if (attributes)
466             {
467                 Attribute* jsonAttribute = attributes->GetAttribute(U"json");
468                 if (jsonAttribute)
469                 {
470                     if (jsonAttribute->Value() == U"true")
471                     {
472                         BaseInitializerNode* baseInitializer = new BaseInitializerNode(span);
473                         baseInitializer->AddArgument(new IdentifierNode(span, U"@value"));
474                         constructorNode.AddInitializer(baseInitializer);
475                     }
476                 }
477             }
478         }
479         for (MemberVariableSymbol* memberVariableSymbol : classTypeSymbol->MemberVariables())
480         {
481             std::u32string jsonFieldName = memberVariableSymbol->Name();
482             Attributes* attributes = memberVariableSymbol->GetAttributes();
483             if (attributes)
484             {
485                 Attribute* jsonAttribute = attributes->GetAttribute(U"json");
486                 if (jsonAttribute)
487                 {
488                     if (jsonAttribute->Value() == U"false")
489                     {
490                         continue;
491                     }
492                 }
493                 Attribute* jsonFieldNameAttribute = attributes->GetAttribute(U"jsonFieldName");
494                 if (jsonFieldNameAttribute)
495                 {
496                     jsonFieldName = jsonFieldNameAttribute->Value();
497                 }
498             }

500             invokeNode->AddArgument(new IdentifierNode(memberVariableSymbol->GetSpan(), U"@value"));
501             invokeNode->AddArgument(new UStringLiteralNode(memberVariableSymbol->GetSpan(), jsonFieldName));
502             invokeNode->AddArgument(new IdentifierNode(memberVariableSymbol->GetSpan(), memberVariableSymbol->Name()));
503             ExpressionStatementNode* fromJsonStatement = new ExpressionStatementNode(memberVariableSymbol->GetSpan(), invokeNode);
504             compoundStatementNode.AddStatement(fromJsonStatement);
505         }
506         SymbolTable& symbolTable = statementBinder->GetBoundCompileUnit().GetSymbolTable();
507         symbolTable.BeginContainer(jsonConstructorSymbol);
508         SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
509         compoundStatementNode.Accept(symbolCreatorVisitor);
510         symbolTable.EndContainer();
511         TypeBinder typeBinder(statementBinder->GetBoundCompileUnit());
512         ContainerScope* containerScope = statementBinder->GetContainerScope();
513         typeBinder.SetContainerScope(containerScope);
514         statementBinder->SetContainerScope(jsonConstructorSymbol->GetContainerScope());
515         typeBinder.SetCurrentFunctionSymbol(jsonConstructorSymbol);
516         compoundStatementNode.Accept(typeBinder);
517         BoundFunction* prevFunction = statementBinder->CurrentFunction();
518         statementBinder->SetCurrentFunction(boundFunction.get());
519         ConstructorSymbol* prevConstructorSymbol = statementBinder->CurrentConstructorSymbol();
520         ConstructorNode* prevConstructorNode = statementBinder->CurrentConstructorNode();
521         statementBinder->SetCurrentConstructor(jsonConstructorSymbol, &constructorNode);
522         compoundStatementNode.Accept(*statementBinder);
523         statementBinder->SetContainerScope(containerScope);
524         BoundStatement* boundStatement = statementBinder->ReleaseStatement();
525         Assert(boundStatement->GetBoundNodeType() == BoundNodeType::boundCompoundStatement, "bound compound statement expected");
526         BoundCompoundStatement* compoundStatement = static_cast<BoundCompoundStatement*>(boundStatement);
527         boundFunction->SetBody(     std::unique_ptr<BoundCompoundStatement>(compoundStatement));
528         statementBinder->CurrentClass()->AddMember(std::move(boundFunction));
529         statementBinder->SetCurrentConstructor(prevConstructorSymbol, prevConstructorNode);
530         statementBinder->SetCurrentFunction(prevFunction);
531         statementBinder->GetBoundCompileUnit().RemoveLastFileScope();
532     }
533     catch (const Exception& ex;)
534     {
535         std::vector<Span> references;
536         references.push_back(ex.Defined());
537         references.insert(references.end(), ex.References().begin(), ex.References().end());
538         throw Exception(module, "error in JSON attribute generation: could not create JSON constructor for class '" + ToUtf8(classTypeSymbol->FullName()) + "': " + ex.Message(), 
539             classTypeSymbol->GetSpan(), references);
540     }
