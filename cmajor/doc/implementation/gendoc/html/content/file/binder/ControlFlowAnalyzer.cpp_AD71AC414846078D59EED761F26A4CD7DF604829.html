 64 void ControlFlowAnalyzer::CollectLabel(BoundStatement& statement)
 65 {
 66     if (!statement.Label().empty())
 67     {
 68         currentFunction->AddLabeledStatement(&statement);
 69         auto it = labelStatementMap.find(statement.Label());
 70         if (it == labelStatementMap.cend())
 71         {
 72             labelStatementMap[statement.Label()] = &statement;
 73         }
 74         else
 75         {
 76             throw Exception(module, "duplicate label '" + ToUtf8(statement.Label()) + "'", statement.GetSpan(), it->second->GetSpan());
 77         }
 78     }
 79 }

 81 void ControlFlowAnalyzer::ResolveGoto(BoundGotoStatement& boundGotoStatement)
 82 {
 83     const std::u32string& target = boundGotoStatement.Target();
 84     auto it = labelStatementMap.find(target);
 85     if (it != labelStatementMap.cend())
 86     {
 87         BoundStatement* targetStatement = it->second;
 88         BoundCompoundStatement* targetBlock = targetStatement->Block();
 89         Assert(targetBlock, "target block not found");
 90         boundGotoStatement.SetTargetStatement(targetStatement);
 91         boundGotoStatement.SetTargetBlock(targetBlock);
 92         BoundCompoundStatement* gotoBlock = boundGotoStatement.Block();
 93         Assert(gotoBlock, "goto block not found");
 94         while (gotoBlock && gotoBlock != targetBlock)
 95         {
 96             if (gotoBlock->Parent())
 97             {
 98                 gotoBlock = gotoBlock->Parent()->Block();
 99             }
100             else
101             {
102                 gotoBlock = nullptr;
103             }
104         }
105         if (!gotoBlock)
106         {
107             throw Exception(module, "goto target '" + ToUtf8(target) + "' not in enclosing block", boundGotoStatement.GetSpan(), targetStatement->GetSpan());
108         }
109     }
110     else
111     {
112         throw Exception(module, "goto target '" + ToUtf8(target) + "' not found", boundGotoStatement.GetSpan());
113     }
114 }

116 void ControlFlowAnalyzer::Visit(BoundCompileUnit& boundCompileUnit)
117 {
118     module = &boundCompileUnit.GetModule();
119     int n = boundCompileUnit.BoundNodes().size();
120     for (int i = 0; i < n; ++i)
121     {
122         BoundNode* boundNode = boundCompileUnit.BoundNodes()[i].get();
123         boundNode->Accept(*this);
124     }
125 }

129     int n = boundClass.Members().size();
130     for (int i = 0; i < n; ++i)
131     {
132         BoundNode* boundNode = boundClass.Members()[i].get();
133         boundNode->Accept(*this);
134     }
135 }

137 void ControlFlowAnalyzer::Visit(BoundFunction& boundFunction)
138 {
139     if (!boundFunction.HasGotos()) return;
140     BoundFunction* prevFunction = currentFunction;
141     currentFunction = &boundFunction;
142     bool prevCollectLabels = collectLabels;
143     collectLabels = true;
144     boundFunction.Body()->Accept(*this);
145     collectLabels = prevCollectLabels;
146     bool prevResolveGotos = resolveGotos;
147     resolveGotos = true;
148     boundFunction.Body()->Accept(*this);
149     resolveGotos = prevResolveGotos;
150     currentFunction = prevFunction;
151 }

171     {
172         BoundStatement* statement = boundCompoundStatement.Statements()[i].get();
173         statement->Accept(*this);
174     }
175 }

228 void ControlFlowAnalyzer::Visit(BoundSwitchStatement& boundSwitchStatement)
229 {
230     if (collectLabels)
231     {
232         CollectLabel(boundSwitchStatement);
233     }
234     int n = boundSwitchStatement.CaseStatements().size();
235     for (int i = 0; i < n; ++i)
236     {
237         BoundCaseStatement* caseStatement = boundSwitchStatement.CaseStatements()[i].get();
238         caseStatement->Accept(*this);
239     }
240     if (boundSwitchStatement.DefaultStatement())
241     {
242         boundSwitchStatement.DefaultStatement()->Accept(*this);
243     }
244 }
