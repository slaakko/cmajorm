268 FunctionSymbol* BoundCompileUnit::GetConversion(TypeSymbol* sourceType, TypeSymbol* targetType, ContainerScope* containerScope, BoundFunction* currentFunction, const Span& span, ArgumentMatch& argumentMatch)
269 {
270     FunctionSymbol* conversion = symbolTable.GetConversion(sourceType, targetType, span);
271     if (conversion && conversion->GetSymbolType() == SymbolType::conversionFunctionSymbol)
272     {
273         argumentMatch.preReferenceConversionFlags = OperationFlags::addr;
274     }
275     if (!conversion)
276     {
277         conversion = conversionTable.GetConversion(sourceType, targetType, span);
278         if (!conversion)
279         {
280             if (sourceType->IsNullPtrType() && targetType->IsPointerType() && !targetType->IsReferenceType())
281             {
282                 std::unique_ptr<FunctionSymbol> nullPtrToPtrConversion(new NullPtrToPtrConversion(symbolTable.GetTypeByName(U"@nullptr_type"), targetType));
283                 conversion = nullPtrToPtrConversion.get();
284                 conversionTable.AddConversion(conversion);
285                 conversionTable.AddGeneratedConversion(std::move(nullPtrToPtrConversion));
286                 return conversion;
287             }
288             else if (sourceType->IsVoidPtrType() && targetType->IsPointerType() && !targetType->IsReferenceType())
289             {
290                 std::unique_ptr<FunctionSymbol> voidPtrToPtrConversion(new VoidPtrToPtrConversion(symbolTable.GetTypeByName(U"void")->AddPointer(span), targetType));
291                 conversion = voidPtrToPtrConversion.get();
292                 conversionTable.AddConversion(conversion);
293                 conversionTable.AddGeneratedConversion(std::move(voidPtrToPtrConversion));
294                 return conversion;
295             }
296             else if (sourceType->PlainType(span)->IsPointerType() && targetType == symbolTable.GetTypeByName(U"ulong"))
297             {
298                 std::unique_ptr<FunctionSymbol> ptrToULongConversion(new PtrToULongConversion(sourceType->PlainType(span), symbolTable.GetTypeByName(U"ulong")));
299                 conversion = ptrToULongConversion.get();
300                 conversionTable.AddConversion(conversion);
301                 conversionTable.AddGeneratedConversion(std::move(ptrToULongConversion));
302                 return conversion;
303             }
304             else if (sourceType->PlainType(span)->IsPointerType() && targetType->RemoveConst(span)->IsVoidPtrType())
305             {
306                 std::unique_ptr<FunctionSymbol> ptrToVoidPtrConversion(new PtrToVoidPtrConversion(sourceType->PlainType(span), symbolTable.GetTypeByName(U"void")->AddPointer(span)));
307                 conversion = ptrToVoidPtrConversion.get();
308                 conversionTable.AddConversion(conversion);
309                 conversionTable.AddGeneratedConversion(std::move(ptrToVoidPtrConversion));
310                 return conversion;
311             }
312             else if (sourceType->BaseType()->IsClassTypeSymbol() && targetType->BaseType()->IsClassTypeSymbol())
313             {
314                 if (sourceType->PointerCount() == targetType->PointerCount() && 
315                     (!sourceType->IsArrayType() && !targetType->IsArrayType()))
316                 {
317                     ClassTypeSymbol* sourceClassType = static_cast<ClassTypeSymbol*>(sourceType->BaseType());
318                     ClassTypeSymbol* targetClassType = static_cast<ClassTypeSymbol*>(targetType->BaseType());
319                     uint8_t conversionDistance = 0;
320                     if (sourceClassType->HasBaseClass(targetClassType, conversionDistance))
321                     {
322                         if (targetType->IsLvalueReferenceType() && !sourceType->IsReferenceType())
323                         {
324                             argumentMatch.preReferenceConversionFlags = OperationFlags::addr;
325                             sourceType = sourceType->AddLvalueReference(span);
326                             if (targetType->IsConstType())
327                             {
328                                 sourceType = sourceType->AddConst(span);
329                             }
330                         }
331                         std::u32string conversionName = sourceType->FullName() + U"2" + targetType->FullName();
332                         std::unique_ptr<FunctionSymbol> implicitClassTypeConversion(new ClassTypeConversion(conversionName, ConversionType::implicit_, conversionDistance, sourceType, targetType));
333                         conversion = implicitClassTypeConversion.get();
334                         // do not add entry to the conversion table
335                         conversionTable.AddGeneratedConversion(std::move(implicitClassTypeConversion));
336                         return conversion;
337                     }
338                     else
339                     {
340                         uint8_t conversionDistance = 0;
341                         if (targetClassType->HasBaseClass(sourceClassType, conversionDistance))
342                         {
343                             if (targetType->IsLvalueReferenceType() && !sourceType->IsReferenceType())
344                             {
345                                 argumentMatch.preReferenceConversionFlags = OperationFlags::addr;
346                                 sourceType = sourceType->AddLvalueReference(span);
347                                 if (targetType->IsConstType())
348                                 {
349                                     sourceType = sourceType->AddConst(span);
350                                 }
351                             }
352                             std::u32string conversionName = sourceType->FullName() + U"2" + targetType->FullName();
353                             std::unique_ptr<FunctionSymbol> explicitClassTypeConversion(new ClassTypeConversion(conversionName, ConversionType::explicit_, conversionDistance, sourceType, targetType));
354                             conversion = explicitClassTypeConversion.get();
355                             // do not add entry to the conversion table
356                             conversionTable.AddGeneratedConversion(std::move(explicitClassTypeConversion));
357                             return conversion;
358                         }
359                     }
360                 }
361             }
362             else if ((sourceType->GetSymbolType() == SymbolType::functionGroupTypeSymbol || sourceType->GetSymbolType() == SymbolType::memberExpressionTypeSymbol) && 
363                 targetType->GetSymbolType() == SymbolType::delegateTypeSymbol)
364             {
365                 FunctionGroupSymbol* functionGroupSymbol = nullptr;
366                 BoundMemberExpression* boundMemberExpression = nullptr;
367                 if (sourceType->GetSymbolType() == SymbolType::functionGroupTypeSymbol)
368                 {
369                     FunctionGroupTypeSymbol* functionGroupTypeSymbol = static_cast<FunctionGroupTypeSymbol*>(sourceType);
370                     functionGroupSymbol = functionGroupTypeSymbol->FunctionGroup();
371                 }
372                 else if (sourceType->GetSymbolType() == SymbolType::memberExpressionTypeSymbol)
373                 {
374                     MemberExpressionTypeSymbol* memberExpressionTypeSymbol = static_cast<MemberExpressionTypeSymbol*>(sourceType);
375                     boundMemberExpression = static_cast<BoundMemberExpression*>(memberExpressionTypeSymbol->BoundMemberExpression());
376                     if (boundMemberExpression->Member()->GetBoundNodeType() == BoundNodeType::boundFunctionGroupExpression)
377                     {
378                         BoundFunctionGroupExpression* boundFunctionGroupExpression = static_cast<BoundFunctionGroupExpression*>(boundMemberExpression->Member());
379                         functionGroupSymbol = boundFunctionGroupExpression->FunctionGroup();
380                     }
381                 }
382                 if (functionGroupSymbol)
383                 {
384                     DelegateTypeSymbol* delegateTypeSymbol = static_cast<DelegateTypeSymbol*>(targetType);
385                     int arity = delegateTypeSymbol->Arity();
386                     std::unordered_set<FunctionSymbol*> viableFunctions;
387                     functionGroupSymbol->CollectViableFunctions(arity, viableFunctions);
388                     for (FunctionSymbol* viableFunction : viableFunctions)
389                     {
390                         if (viableFunction->GetSymbolType() == SymbolType::memberFunctionSymbol && !viableFunction->IsStatic()) continue;
391                         bool found = true;
392                         for (int i = 0; i < arity; ++i)
393                         {
394                             ParameterSymbol* sourceParam = viableFunction->Parameters()[i];
395                             ParameterSymbol* targetParam = delegateTypeSymbol->Parameters()[i];
396                             if (!TypesEqual(sourceParam->GetType(), targetParam->GetType()))
397                             {
398                                 found = false;
399                                 break;
400                             }
401                         }
402                         if (found)
403                         {
404                             found = TypesEqual(viableFunction->ReturnType(), delegateTypeSymbol->ReturnType());
405                         }
406                         if (found)
407                         {
408                             if (boundMemberExpression)
409                             {
410                                 boundMemberExpression->ResetClassPtr();
411                             }
412                             if (viableFunction->IsFunctionTemplate())
413                             {
414                                 if (sourceType->GetSymbolType() == SymbolType::functionGroupTypeSymbol)
415                                 {
416                                     FunctionGroupTypeSymbol* functionGroupTypeSymbol = static_cast<FunctionGroupTypeSymbol*>(sourceType);
417                                     BoundFunctionGroupExpression* boundFunctionGroupExpression = static_cast<BoundFunctionGroupExpression*>(functionGroupTypeSymbol->BoundFunctionGroup());
418                                     std::unordered_map<TemplateParameterSymbol*, TypeSymbol*> templateParameterMap;
419                                     int n = viableFunction->TemplateParameters().size();
420                                     if (boundFunctionGroupExpression->TemplateArgumentTypes().size() == n)
421                                     {
422                                         for (int i = 0; i < n; ++i)
423                                         {
424                                             TemplateParameterSymbol* templateParameterSymbol = viableFunction->TemplateParameters()[i];
425                                             TypeSymbol* templateArgumentType = boundFunctionGroupExpression->TemplateArgumentTypes()[i];
426                                             templateParameterMap[templateParameterSymbol] = templateArgumentType;
427                                         }
428                                         viableFunction = InstantiateFunctionTemplate(viableFunction, templateParameterMap, span);
429                                     }
430                                     else
431                                     {
432                                         return nullptr;
433                                     }
434                                 }
435                                 else
436                                 {
437                                     return nullptr;
438                                 }
439                             }
440                             std::unique_ptr<FunctionSymbol> functionToDelegateConversion(new FunctionToDelegateConversion(sourceType, delegateTypeSymbol, viableFunction));
441                             conversion = functionToDelegateConversion.get();
442                             conversionTable.AddConversion(conversion);
443                             conversionTable.AddGeneratedConversion(std::move(functionToDelegateConversion));
444                             return conversion;
445                         }
446                     }
447                 }
448             }
449             else if ((sourceType->GetSymbolType() == SymbolType::functionGroupTypeSymbol || sourceType->GetSymbolType() == SymbolType::memberExpressionTypeSymbol) && 
450                 targetType->PlainType(span)->GetSymbolType() == SymbolType::classDelegateTypeSymbol && currentFunction)
451             {
452                 ClassDelegateTypeSymbol* classDelegateType = static_cast<ClassDelegateTypeSymbol*>(targetType->PlainType(span));
453                 FunctionGroupSymbol* functionGroup = nullptr;
454                 if (sourceType->GetSymbolType() == SymbolType::functionGroupTypeSymbol)
455                 {
456                     FunctionGroupTypeSymbol* functionGroupTypeSymbol = static_cast<FunctionGroupTypeSymbol*>(sourceType);
457                     BoundFunctionGroupExpression* boundFunctionGroup = static_cast<BoundFunctionGroupExpression*>(functionGroupTypeSymbol->BoundFunctionGroup());
458                     functionGroup = functionGroupTypeSymbol->FunctionGroup();
459                 }
460                 else if (sourceType->GetSymbolType() == SymbolType::memberExpressionTypeSymbol)
461                 {
462                     MemberExpressionTypeSymbol* memberExpressionType = static_cast<MemberExpressionTypeSymbol*>(sourceType);
463                     BoundMemberExpression* boundMemberExpr = static_cast<BoundMemberExpression*>(memberExpressionType->BoundMemberExpression());
464                     if (boundMemberExpr->Member()->GetBoundNodeType() == BoundNodeType::boundFunctionGroupExpression)
465                     {
466                         BoundFunctionGroupExpression* boundFunctionGroup = static_cast<BoundFunctionGroupExpression*>(boundMemberExpr->Member());
467                         functionGroup = boundFunctionGroup->FunctionGroup();
468                     }
469                 }
470                 if (functionGroup)
471                 {
472                     int arity = classDelegateType->Arity();
473                     std::unordered_set<FunctionSymbol*> viableFunctions;
474                     functionGroup->CollectViableFunctions(arity + 1, viableFunctions);
475                     for (FunctionSymbol* viableFunction : viableFunctions)
476                     {
477                         bool found = true;
478                         for (int i = 1; i < arity + 1; ++i)
479                         {
480                             ParameterSymbol* sourceParam = viableFunction->Parameters()[i];
481                             ParameterSymbol* targetParam = classDelegateType->Parameters()[i - 1];
482                             if (!TypesEqual(sourceParam->GetType(), targetParam->GetType()))
483                             {
484                                 found = false;
485                                 break;
486                             }
487                         }
488                         if (found)
489                         {
490                             found = TypesEqual(viableFunction->ReturnType(), classDelegateType->ReturnType());
491                         }
492                         if (found)
493                         {
494                             LocalVariableSymbol* objectDelegatePairVariable = currentFunction->GetFunctionSymbol()->CreateTemporary(classDelegateType->ObjectDelegatePairType(), span);
495                             std::unique_ptr<FunctionSymbol> memberFunctionToClassDelegateConversion(new MemberFunctionToClassDelegateConversion(span, sourceType, classDelegateType, viableFunction, 
496                                 objectDelegatePairVariable));
497                             conversion = memberFunctionToClassDelegateConversion.get();
498                             conversionTable.AddConversion(conversion);
499                             conversionTable.AddGeneratedConversion(std::move(memberFunctionToClassDelegateConversion));
500                             return conversion;
501                         }
502                     }
503                 }
504             }
505             else if (targetType->PlainType(span)->GetSymbolType() == SymbolType::interfaceTypeSymbol && currentFunction)
506             {
507                 InterfaceTypeSymbol* targetInterfaceType = static_cast<InterfaceTypeSymbol*>(targetType->PlainType(span));
508                 if (sourceType->IsClassTypeSymbol())
509                 {
510                     ClassTypeSymbol* sourceClassType = static_cast<ClassTypeSymbol*>(sourceType);
511                     int32_t n = sourceClassType->ImplementedInterfaces().size();
512                     for (int32_t i = 0; i < n; ++i)
513                     {
514                         InterfaceTypeSymbol* sourceInterfaceType = sourceClassType->ImplementedInterfaces()[i];
515                         if (TypesEqual(targetInterfaceType, sourceInterfaceType))
516                         {
517                             LocalVariableSymbol* temporaryInterfaceObjectVar = currentFunction->GetFunctionSymbol()->CreateTemporary(targetInterfaceType, span);
518                             std::unique_ptr<FunctionSymbol> classToInterfaceConversion(new ClassToInterfaceConversion(sourceClassType, targetInterfaceType, temporaryInterfaceObjectVar, i, span));
519                             classToInterfaceConversion->SetModule(&GetModule());
520                             classToInterfaceConversion->SetOriginalModule(&GetModule());
521                             classToInterfaceConversion->SetSymbolTable(&symbolTable);
522                             conversion = classToInterfaceConversion.get();
523                             conversionTable.AddConversion(conversion);
524                             conversionTable.AddGeneratedConversion(std::move(classToInterfaceConversion));
525                             return conversion;
526                         }
527                     }
528                 }
529             }
530         }
531     }
532     if (conversion)
533     {
534         if (conversion->Parent() && !conversion->IsGeneratedFunction() && conversion->Parent()->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
535         {
536             InstantiateClassTemplateMemberFunction(conversion, containerScope, currentFunction, span);
537         }
538         else if (GetGlobalFlag(GlobalFlags::release) && conversion->IsInline())
539         {
540             InstantiateInlineFunction(conversion, containerScope, span);
541         }
542     }
543     return conversion;
544 }

660 void BoundCompileUnit::FinalizeBinding(ClassTypeSymbol* classType)
661 {
662     if (classType->GetSymbolType() != SymbolType::classTemplateSpecializationSymbol) return;
663     ClassTemplateSpecializationSymbol* classTemplateSpecialization = static_cast<ClassTemplateSpecializationSymbol*>(classType);
664     if (classTemplateSpecialization->StatementsNotBound())
665     {
666         classTemplateSpecialization->ResetStatementsNotBound();
667         FileScope* fileScope = classTemplateSpecialization->ReleaseFileScope();
668         bool fileScopeAdded = false;
669         if (fileScope)
670         {
671             AddFileScope(fileScope);
672             fileScopeAdded = true;
673         }
674         StatementBinder statementBinder(*this);
675         classTemplateSpecialization->GlobalNs()->Accept(statementBinder);
676         if (fileScopeAdded)
677         {
678             RemoveLastFileScope();
679         }
680     }
681 }
