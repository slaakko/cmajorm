  61 void BoundParameter::Load(Emitter& emitter, OperationFlags flags)
  62 {
  63     emitter.SetCurrentDebugLocation(GetSpan());
  64     if ((flags & OperationFlags::addr) != OperationFlags::none)
  65     {
  66         emitter.Stack().Push(parameterSymbol->IrObject());
  67     }
  68     else if ((flags & OperationFlags::deref) != OperationFlags::none)
  69     {
  70         llvm::Value* value = emitter.Builder().CreateLoad(parameterSymbol->IrObject());
  71         uint8_t n = GetDerefCount(flags);
  72         for (uint8_t i = 0; i < n; ++i)
  73         {
  74             value = emitter.Builder().CreateLoad(value);
  75         }
  76         emitter.Stack().Push(value);
  77     }
  78     else
  79     {
  80         emitter.Stack().Push(emitter.Builder().CreateLoad(parameterSymbol->IrObject()));
  81     }
  82     DestroyTemporaries(emitter);
  83 }

 198 void BoundMemberVariable::Load(Emitter& emitter, OperationFlags flags)
 199 {
 200     emitter.SetCurrentDebugLocation(GetSpan());
 201     Assert(memberVariableSymbol->LayoutIndex() != -1, "layout index of the member variable not set");
 202     if (memberVariableSymbol->IsStatic())
 203     {
 204         ClassTypeSymbol* classType = static_cast<ClassTypeSymbol*>(memberVariableSymbol->Parent());
 205         if (staticInitNeeded)
 206         {
 207             if (classType->StaticConstructor())
 208             {
 209                 BoundFunctionCall staticConstructorCall(GetModule(), GetSpan(), classType->StaticConstructor());
 210                 staticConstructorCall.Load(emitter, OperationFlags::none);
 211             }
 212         }
 213         emitter.Stack().Push(classType->StaticObject(emitter, false));
 214     }
 215     else
 216     {
 217         classPtr->Load(emitter, OperationFlags::none);
 218     }
 219     llvm::Value* ptr = emitter.Stack().Pop();
 220     ArgVector indeces;
 221     indeces.push_back(emitter.Builder().getInt32(0));
 222     indeces.push_back(emitter.Builder().getInt32(memberVariableSymbol->LayoutIndex()));
 223     llvm::Value* memberVariablePtr = emitter.Builder().CreateGEP(ptr, indeces);
 224     if ((flags & OperationFlags::addr) != OperationFlags::none)
 225     {
 226         emitter.Stack().Push(memberVariablePtr);
 227     }
 228     else if ((flags & OperationFlags::deref) != OperationFlags::none)
 229     {
 230         llvm::Value* value = emitter.Builder().CreateLoad(memberVariablePtr);
 231         uint8_t n = GetDerefCount(flags);
 232         for (uint8_t i = 0; i < n; ++i)
 233         {
 234             value = emitter.Builder().CreateLoad(value);
 235         }
 236         emitter.Stack().Push(value);
 237     }
 238     else
 239     {
 240         emitter.Stack().Push(emitter.Builder().CreateLoad(memberVariablePtr));
 241     }
 242     DestroyTemporaries(emitter);
 243 }

 429 std::unique_ptr<Value> BoundLiteral::ToValue(BoundCompileUnit& boundCompileUnit) const
 430 {
 431     switch (value->GetValueType())
 432     {
 433         case ValueType::stringValue:
 434         {
 435             StringValue* stringValue = static_cast<StringValue*>(value.get());
 436             return      std::unique_ptr<Value>(new PointerValue(GetSpan(), boundCompileUnit.GetSymbolTable().GetTypeByName(U"char")->AddPointer(Span()), boundCompileUnit.GetUtf8CharPtr(stringValue->StringId())));
 437         }
 438         case ValueType::wstringValue:
 439         {
 440             WStringValue* wstringValue = static_cast<WStringValue*>(value.get());
 441             return      std::unique_ptr<Value>(new PointerValue(GetSpan(), boundCompileUnit.GetSymbolTable().GetTypeByName(U"wchar")->AddPointer(Span()), boundCompileUnit.GetUtf16CharPtr(wstringValue->StringId())));
 442         }
 443         case ValueType::ustringValue:
 444         {
 445             UStringValue* ustringValue = static_cast<UStringValue*>(value.get());
 446             return      std::unique_ptr<Value>(new PointerValue(GetSpan(), boundCompileUnit.GetSymbolTable().GetTypeByName(U"uchar")->AddPointer(Span()), boundCompileUnit.GetUtf32CharPtr(ustringValue->StringId())));
 447         }
 448         default:
 449         {
 450             return      std::unique_ptr<Value>(value->Clone());
 451         }
 452     }
 453     return      std::unique_ptr<Value>();
