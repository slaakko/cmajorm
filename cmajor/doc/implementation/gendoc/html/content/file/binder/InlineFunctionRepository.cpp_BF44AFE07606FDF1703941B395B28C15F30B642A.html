 21 InlineFunctionRepository::InlineFunctionRepository(BoundCompileUnit& boundCompileUnit_) : boundCompileUnit(boundCompileUnit_)
 22 {
 23 }

 25 InlineFunctionRepository::~InlineFunctionRepository()
 26 {
 27     for (FunctionSymbol* inlineFunctionSymbol : instantiatedInlineFunctions)
 28     {
 29         if (inlineFunctionSymbol->IsTemplateSpecialization())
 30         {
 31             FunctionGroupSymbol* functionGroup = inlineFunctionSymbol->FunctionGroup();
 32             if (functionGroup)
 33             {
 34                 functionGroup->RemoveFunction(inlineFunctionSymbol);
 35             }
 36         }
 37     }
 38 }

 40 void InlineFunctionRepository::Instantiate(FunctionSymbol* inlineFunction, ContainerScope* containerScope, const Span& span)
 41 {
 42     if (inlineFunction->GetCompileUnit() == boundCompileUnit.GetCompileUnitNode()) return;
 43     if (instantiatedInlineFunctions.find(inlineFunction) != instantiatedInlineFunctions.cend()) return;
 44     instantiatedInlineFunctions.insert(inlineFunction);
 45     SymbolTable& symbolTable = boundCompileUnit.GetSymbolTable();
 46     Node* node = symbolTable.GetNodeNoThrow(inlineFunction);
 47     if (!node)
 48     {
 49         inlineFunction->ReadAstNodes();
 50         node = inlineFunction->GetFunctionNode();
 51         Assert(node, "function node not read");
 52     }
 53     FunctionNode* functionNode = static_cast<FunctionNode*>(node);
 54     std::unique_ptr<NamespaceNode> globalNs(new NamespaceNode(inlineFunction->GetSpan(), new IdentifierNode(inlineFunction->GetSpan(), U"")));
 55     NamespaceNode* currentNs = globalNs.get();
 56     CloneContext cloneContext;
 57     cloneContext.SetInstantiateFunctionNode();
 58     bool fileScopeAdded = false;
 59     int n = inlineFunction->UsingNodes().Count();
 60     if (!inlineFunction->Ns()->IsGlobalNamespace() || n > 0)
 61     {
 62         FileScope* primaryFileScope = new FileScope(&boundCompileUnit.GetModule());
 63         if (!inlineFunction->Ns()->IsGlobalNamespace())
 64         {
 65             primaryFileScope->AddContainerScope(inlineFunction->Ns()->GetContainerScope());
 66         }
 67         for (int i = 0; i < n; ++i)
 68         {
 69             Node* usingNode = inlineFunction->UsingNodes()[i];
 70             if (usingNode->GetNodeType() == NodeType::namespaceImportNode)
 71             {
 72                 primaryFileScope->InstallNamespaceImport(containerScope, static_cast<NamespaceImportNode*>(usingNode));
 73             }
 74             else if (usingNode->GetNodeType() == NodeType::aliasNode)
 75             {
 76                 primaryFileScope->InstallAlias(containerScope, static_cast<AliasNode*>(usingNode));
 77             }
 78         }
 79         boundCompileUnit.AddFileScope(primaryFileScope);
 80         fileScopeAdded = true;
 81         std::u32string fullNsName = inlineFunction->Ns()->FullName();
 82         std::vector<std::u32string> nsComponents = Split(fullNsName, '.');
 83         for (const std::u32string& nsComponent : nsComponents)
 84         {
 85             NamespaceNode* nsNode = new NamespaceNode(inlineFunction->GetSpan(), new IdentifierNode(inlineFunction->GetSpan(), nsComponent));
 86             currentNs->AddMember(nsNode);
 87             currentNs = nsNode;
 88         }
 89     }
 90     FunctionNode* functionInstanceNode = static_cast<FunctionNode*>(functionNode->Clone(cloneContext));
 91     currentNs->AddMember(functionInstanceNode);
 92     symbolTable.SetCurrentCompileUnit(boundCompileUnit.GetCompileUnitNode());
 93     SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
 94     symbolTable.BeginContainer(inlineFunction);
 95     functionInstanceNode->Body()->Accept(symbolCreatorVisitor);
 96     symbolTable.EndContainer();
 97     TypeBinder typeBinder(boundCompileUnit);
 98     typeBinder.SetContainerScope(inlineFunction->GetContainerScope());
 99     functionInstanceNode->Body()->Accept(typeBinder);
100     StatementBinder statementBinder(boundCompileUnit);
101     std::unique_ptr<BoundClass> boundClass;
102     std::unique_ptr<BoundFunction> boundFunction(new BoundFunction(&boundCompileUnit.GetModule(), inlineFunction));
103     statementBinder.SetCurrentFunction(boundFunction.get());
104     if (inlineFunction->GetSymbolType() == SymbolType::constructorSymbol || 
105         inlineFunction->GetSymbolType() == SymbolType::memberFunctionSymbol)
106     {
107         boundClass.reset(new BoundClass(&boundCompileUnit.GetModule(), static_cast<ClassTypeSymbol*>(inlineFunction->Parent())));
108         statementBinder.SetCurrentClass(boundClass.get());
109         if (inlineFunction->GetSymbolType() == SymbolType::constructorSymbol)
110         {
111             ConstructorSymbol* constructorSymbol = static_cast<ConstructorSymbol*>(inlineFunction);
112             Assert(node->GetNodeType() == NodeType::constructorNode, "constructor node expected");
113             ConstructorNode* constructorNode = static_cast<ConstructorNode*>(node);
114             statementBinder.SetCurrentConstructor(constructorSymbol, constructorNode);
115         }
116         else if (inlineFunction->GetSymbolType() == SymbolType::memberFunctionSymbol)
117         {
118             MemberFunctionSymbol* memberFunctionSymbol = static_cast<MemberFunctionSymbol*>(inlineFunction);
119             Assert(node->GetNodeType() == NodeType::memberFunctionNode, "member function node expected");
120             MemberFunctionNode* memberFunctionNode = static_cast<MemberFunctionNode*>(node);
121             statementBinder.SetCurrentMemberFunction(memberFunctionSymbol, memberFunctionNode);
122         }
123     }
124     statementBinder.SetContainerScope(inlineFunction->GetContainerScope());
125     functionInstanceNode->Body()->Accept(statementBinder);
126     BoundStatement* boundStatement = statementBinder.ReleaseStatement();
127     Assert(boundStatement->GetBoundNodeType() == BoundNodeType::boundCompoundStatement, "bound compound statement expected");
128     BoundCompoundStatement* compoundStatement = static_cast<BoundCompoundStatement*>(boundStatement);
129     boundFunction->SetBody(     std::unique_ptr<BoundCompoundStatement>(compoundStatement));
130     if (boundClass)
131     {
132         boundClass->AddMember(std::move(boundFunction));
133         boundCompileUnit.AddBoundNode(std::move(boundClass));
134     }
135     else
136     {
137         boundCompileUnit.AddBoundNode(std::move(boundFunction));
138     }
139     if (fileScopeAdded)
140     {
141         boundCompileUnit.RemoveLastFileScope();
142     }
143     inlineFunction->SetGlobalNs(std::move(globalNs));
144 }
