 20 enum class BoundExpressionFlags : uint8_t 
 21 {
 22     none = 0, 
 23     argIsExplicitThisOrBasePtr = 1 << 0, 
 24     bindToRvalueReference = 1 << 1, 
 25     virtualCall = 1 << 2, 
 26     deref = 1 << 3, 
 27     exceptionCapture = 1 << 4
 28 };
 29 
 30 inline BoundExpressionFlags operator|(BoundExpressionFlags left, BoundExpressionFlags right)
 31 {
 32     return BoundExpressionFlags(uint8_t(left) | uint8_t(right));
 33 }
 34 
 35 inline BoundExpressionFlags operator&(BoundExpressionFlags left, BoundExpressionFlags right)
 36 {
 37     return BoundExpressionFlags(uint8_t(left) & uint8_t(right));
 38 }

 43 class BoundExpression : public BoundNode
 44 {
 45 public:
 46     BoundExpression(Module* module_, const Span& span_, BoundNodeType boundNodeType_, TypeSymbol* type_);
 47     virtual BoundExpression* Clone() = 0;
 48     virtual bool IsComplete() const { return true; }
 49     virtual bool IsLvalueExpression() const { return false; }
 50     virtual bool HasValue() const { return false; }
 51     virtual std::string TypeString() const { return "expression"; }
 52     virtual std::unique_ptr<Value> ToValue(BoundCompileUnit& boundCompileUnit) const { return      std::unique_ptr<Value>();}
 53     virtual bool ContainsExceptionCapture() const { return GetFlag(BoundExpressionFlags::exceptionCapture); }
 54     const TypeSymbol* GetType() const { return type; }
 55     TypeSymbol* GetType() { return type; }
 56     bool GetFlag(BoundExpressionFlags flag) const { return (flags & flag) != BoundExpressionFlags::none;  }
 57     void SetFlag(BoundExpressionFlags flag) { flags = flags | flag; }
 58     void AddTemporaryDestructorCall(std::unique_ptr<BoundFunctionCall>&& destructorCall);
 59     void MoveTemporaryDestructorCallsTo(BoundExpression& expression);
 60     void DestroyTemporaries(Emitter& emitter);
 61 private:
 62     TypeSymbol* type;
 63     BoundExpressionFlags flags;
 64     std::vector<std::unique_ptr<BoundFunctionCall>> temporaryDestructorCalls;
 65 };
