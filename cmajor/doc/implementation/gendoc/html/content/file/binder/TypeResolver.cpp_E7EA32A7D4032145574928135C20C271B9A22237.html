188 void TypeResolver::Visit(ArrayNode& arrayNode)
189 {
190     type = ResolveType(arrayNode.Subject(), boundCompileUnit, containerScope);
191     if (type->IsReferenceType())
192     {
193         throw Exception(module, "cannot have array of reference type", arrayNode.GetSpan());
194     }
195     int64_t size = -1;
196     if (arrayNode.Size())
197     {
198         std::unique_ptr<Value> sizeValue = Evaluate(arrayNode.Size(), boundCompileUnit.GetSymbolTable().GetTypeByName(U"long"), containerScope, boundCompileUnit, false, nullptr, arrayNode.GetSpan());
199         if (sizeValue->GetValueType() == ValueType::longValue)
200         {
201             LongValue* longSizeValue = static_cast<LongValue*>(sizeValue.get());
202             size = longSizeValue->GetValue();
203         }
204         else
205         {
206             throw Exception(module, "long type value expected ", arrayNode.Size()->GetSpan());
207         }
208     }
209     type = symbolTable.MakeArrayType(type, size, arrayNode.GetSpan());
210 }

212 void TypeResolver::ResolveSymbol(Node& node, IdentifierNode* idNode, Symbol* symbol)
213 {
214     if (symbol->IsTypeSymbol())
215     {
216         type = static_cast<TypeSymbol*>(symbol);
217         if (GetGlobalFlag(GlobalFlags::cmdoc))
218         {
219             TypeSymbol* cmdocType = type;
220             if (type->GetSymbolType() == SymbolType::classGroupTypeSymbol)
221             {
222                 ClassGroupTypeSymbol* group = static_cast<ClassGroupTypeSymbol*>(type);
223                 cmdocType = group->GetClass(0);
224             }
225             symbolTable.MapSymbol(idNode, cmdocType);
226         }
227     }
228     else
229     {
230         switch (symbol->GetSymbolType())
231         {
232             case SymbolType::typedefSymbol:
233             {
234                 TypedefSymbol* typedefSymbol = static_cast<TypedefSymbol*>(symbol);
235                 if (typedefSymbol->IsProject() && !typedefSymbol->IsBound())
236                 {
237                     TypeBinder typeBinder(boundCompileUnit);
238                     typeBinder.SetContainerScope(typedefSymbol->Parent()->GetContainerScope());
239                     Node* node = symbolTable.GetNode(typedefSymbol);
240                     Assert(node->GetNodeType() == NodeType::typedefNode, "typedef node expected");
241                     TypedefNode* typedefNode = static_cast<TypedefNode*>(node);
242                     typeBinder.BindTypedef(typedefSymbol, typedefNode, false);
243                 }
244                 type = typedefSymbol->GetType();
245                 if (GetGlobalFlag(GlobalFlags::cmdoc))
246                 {
247                     symbolTable.MapSymbol(idNode, typedefSymbol);
248                 }
249                 break;
250             }
251             case SymbolType::boundTemplateParameterSymbol:
252             {
253                 BoundTemplateParameterSymbol* boundTemplateParameterSymbol = static_cast<BoundTemplateParameterSymbol*>(symbol);
254                 type = boundTemplateParameterSymbol->GetType();
255                 break;
256             }
257             case SymbolType::namespaceSymbol:
258             {
259                 NamespaceSymbol* ns = static_cast<NamespaceSymbol*>(symbol);
260                 nsTypeSymbol.reset(new NamespaceTypeSymbol(ns));
261                 type = nsTypeSymbol.get();
262                 break;
263             }
264             default:
265             {
266                 throw Exception(module, "symbol '" + ToUtf8(symbol->FullName()) + "' does not denote a type", node.GetSpan(), symbol->GetSpan());
267             }
268         }
269     }
270 }

272 void TypeResolver::Visit(IdentifierNode& identifierNode)
273 {
274     symbolTable.SetLatestIdentifier(&identifierNode);
275     std::u32string name = identifierNode.Str();
276     Symbol* symbol = containerScope->Lookup(name, ScopeLookup::this_and_base_and_parent);
277     if (!symbol)
278     {
279         for (const std::unique_ptr<FileScope>& fileScope : boundCompileUnit.FileScopes())
280         {
281             symbol = fileScope->Lookup(name);
282             if (symbol)
283             {
284                 break;
285             }
286         }
287     }
288     if (symbol)
289     {
290         ResolveSymbol(identifierNode, &identifierNode, symbol);
291     }
292     else
293     {
294         throw Exception(module, "type symbol '" + ToUtf8(name) + "' not found", identifierNode.GetSpan());
295     }
296 }

298 void TypeResolver::Visit(TemplateIdNode& templateIdNode)
299 {
300     IdentifierNode* prevId = symbolTable.GetLatestIdentifier();
301     int arity = templateIdNode.TemplateArguments().Count();
302     TypeSymbol* primaryTemplateType = ResolveType(templateIdNode.Primary(), boundCompileUnit, containerScope, TypeResolverFlags::resolveClassGroup);
303     if (primaryTemplateType->GetSymbolType() == SymbolType::classGroupTypeSymbol)
304     {
305         ClassGroupTypeSymbol* classGroup = static_cast<ClassGroupTypeSymbol*>(primaryTemplateType);
306         ClassTypeSymbol* classType = classGroup->GetClass(arity);
307         if (classType)
308         {
309             primaryTemplateType = classType;
310         }
311         else
312         {
313             throw Exception(classGroup->GetModule(), "primary class template with arity '" + std::to_string(arity) + "' not found", classGroup->GetSpan());
314         }
315     }
316     if (!primaryTemplateType->IsClassTypeSymbol())
317     {
318         throw Exception(module, "class type symbol expected", templateIdNode.Primary()->GetSpan());
319     }
320     ClassTypeSymbol* classTemplate = static_cast<ClassTypeSymbol*>(primaryTemplateType);
321     if (!classTemplate->IsClassTemplate())
322     {
323         throw Exception(module, "class template expected", templateIdNode.Primary()->GetSpan());
324     }
325     IdentifierNode* idNode = symbolTable.GetLatestIdentifier();
326     symbolTable.SetLatestIdentifier(prevId);
327     if (GetGlobalFlag(GlobalFlags::cmdoc))
328     {
329         symbolTable.MapSymbol(idNode, classTemplate);
330     }
331     std::vector<TypeSymbol*> templateArgumentTypes;
332     int n = arity;
333     for (int i = 0; i < n; ++i)
334     {
335         IdentifierNode* prevId = symbolTable.GetLatestIdentifier();
336         TypeSymbol* templateArgumentType = ResolveType(templateIdNode.TemplateArguments()[i], boundCompileUnit, containerScope);
337         templateArgumentTypes.push_back(templateArgumentType);
338         IdentifierNode* idNode = symbolTable.GetLatestIdentifier();
339         if (idNode && GetGlobalFlag(GlobalFlags::cmdoc))
340         {
341             symbolTable.MapSymbol(idNode, templateArgumentType);
342         }
343         symbolTable.SetLatestIdentifier(prevId);
344     }
345     int m = classTemplate->TemplateParameters().size();
346     if (n < m)
347     {
348         classTemplateRepository.ResolveDefaultTemplateArguments(templateArgumentTypes, classTemplate, containerScope, templateIdNode.GetSpan());
349     }
350     ClassTemplateSpecializationSymbol* classTemplateSpecialization = symbolTable.MakeClassTemplateSpecialization(classTemplate, templateArgumentTypes, templateIdNode.GetSpan());
351     if (!classTemplateSpecialization->IsBound())
352     {
353         classTemplateRepository.BindClassTemplateSpecialization(classTemplateSpecialization, containerScope, templateIdNode.GetSpan());
354     }
355     type = classTemplateSpecialization;
356 }

358 void TypeResolver::Visit(DotNode& dotNode)
359 {
360     dotNode.Subject()->Accept(*this);
361     Scope* scope = nullptr;
362     if (type->GetSymbolType() == SymbolType::classGroupTypeSymbol)
363     {
364         ClassGroupTypeSymbol* classGroup = static_cast<ClassGroupTypeSymbol*>(type);
365         type = classGroup->GetClass(0);
366         if (!type)
367         {
368             throw Exception(module, "symbol '" + ToUtf8(type->FullName()) + "' does not denote a class type, an array type or a namespace", dotNode.GetSpan(), type->GetSpan());
369         }
370     }
371     if (type->GetSymbolType() == SymbolType::namespaceTypeSymbol)
372     {
373         NamespaceTypeSymbol* nsType = static_cast<NamespaceTypeSymbol*>(type);
374         scope = nsType->Ns()->GetContainerScope();
375     }
376     else if (type->IsClassTypeSymbol() || type->IsArrayType() || (flags & TypeResolverFlags::createMemberSymbols) != TypeResolverFlags::none && type->GetSymbolType() == SymbolType::templateParameterSymbol)
377     {
378         scope = type->GetContainerScope();
379     }
380     else
381     {
382         throw Exception(module, "symbol '" + ToUtf8(type->FullName()) + "' does not denote a class type, an array type or a namespace", dotNode.GetSpan(), type->GetSpan());
383     }
384     std::u32string name = dotNode.MemberId()->Str();
385     Symbol* symbol = scope->Lookup(name, ScopeLookup::this_and_base);
386     if (symbol)
387     {
388         ResolveSymbol(dotNode, dotNode.MemberId(), symbol);
389     }
390     else
391     {
392         if ((flags & TypeResolverFlags::createMemberSymbols) != TypeResolverFlags::none && type->GetSymbolType() == SymbolType::templateParameterSymbol)
393         {
394             TemplateParameterSymbol* templateParameterSymbol = new TemplateParameterSymbol(dotNode.GetSpan(), name);
395             templateParameterSymbol->SetModule(module);
396             templateParameterSymbol->SetOriginalModule(module);
397             templateParameterSymbol->SetSymbolTable(&symbolTable);
398             symbolTable.SetTypeIdFor(templateParameterSymbol);
399             type->AddMember(templateParameterSymbol);
400             ResolveSymbol(dotNode, dotNode.MemberId(), templateParameterSymbol);
401         }
402         else
403         {
404             throw Exception(module, "type symbol '" + ToUtf8(name) + "' not found", dotNode.GetSpan());
405         }
406     }
407 }

410 {
411     return ResolveType(typeExprNode, boundCompileUnit, containerScope, TypeResolverFlags::none);

414 TypeSymbol* ResolveType(Node* typeExprNode, BoundCompileUnit& boundCompileUnit, ContainerScope* containerScope, TypeResolverFlags flags)
415 {
416     Module* module = &boundCompileUnit.GetModule();
417     bool resolveClassGroup = (flags & TypeResolverFlags::resolveClassGroup) != TypeResolverFlags::none;
418     TypeResolver typeResolver(boundCompileUnit, containerScope, flags);
419     typeExprNode->Accept(typeResolver);
420     TypeSymbol* type = typeResolver.GetType();
421     if (resolveClassGroup && type && type->GetSymbolType() == SymbolType::classGroupTypeSymbol)
422     {
423         return type;
424     }
425     if (type && type->GetSymbolType() == SymbolType::classGroupTypeSymbol)
426     {
427         ClassGroupTypeSymbol* classGroup = static_cast<ClassGroupTypeSymbol*>(type);
428         type = classGroup->GetClass(0);
429     }
430     if (!type || type->IsInComplete())
431     {
432         throw Exception(module, "incomplete type expression", typeExprNode->GetSpan());
433     }
434     TypeDerivationRec derivationRec = UnifyDerivations(typeResolver.DerivationRec(), type->DerivationRec());
435     if (!derivationRec.derivations.empty())
436     {
437         return boundCompileUnit.GetSymbolTable().MakeDerivedType(type->BaseType(), derivationRec, typeExprNode->GetSpan());
438     }
439     return type;
440 }
