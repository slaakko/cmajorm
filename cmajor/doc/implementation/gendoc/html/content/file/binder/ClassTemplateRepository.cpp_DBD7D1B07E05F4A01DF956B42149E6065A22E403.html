 30 void ClassTemplateRepository::ResolveDefaultTemplateArguments(std::vector<TypeSymbol*>& templateArgumentTypes, ClassTypeSymbol* classTemplate, ContainerScope* containerScope, const Span& span)
 31 {
 32     int n = classTemplate->TemplateParameters().size();
 33     int m = templateArgumentTypes.size();
 34     if (m == n) return;
 35     SymbolTable& symbolTable = boundCompileUnit.GetSymbolTable();
 36     Node* node = symbolTable.GetNodeNoThrow(classTemplate);
 37     if (!node)
 38     {
 39         classTemplate->ReadAstNodes();
 40         node = classTemplate->GetClassNode();
 41         Assert(node, "class node not read");
 42     }
 43     Assert(node->GetNodeType() == NodeType::classNode, "class node expected");
 44     ClassNode* classNode = static_cast<ClassNode*>(node);
 45     int numFileScopeAdded = 0;
 46     int nu = classTemplate->UsingNodes().Count();
 47     if (nu > 0)
 48     {
 49         FileScope* fileScope = new FileScope(&boundCompileUnit.GetModule());
 50         for (int i = 0; i < nu; ++i)
 51         {
 52             Node* usingNode = classTemplate->UsingNodes()[i];
 53             if (usingNode->GetNodeType() == NodeType::namespaceImportNode)
 54             {
 55                 NamespaceImportNode* namespaceImportNode = static_cast<NamespaceImportNode*>(usingNode);
 56                 fileScope->InstallNamespaceImport(containerScope, namespaceImportNode);
 57             }
 58             else if (usingNode->GetNodeType() == NodeType::aliasNode)
 59             {
 60                 AliasNode* aliasNode = static_cast<AliasNode*>(usingNode);
 61                 fileScope->InstallAlias(containerScope, aliasNode);
 62             }
 63         }
 64         boundCompileUnit.AddFileScope(fileScope);
 65         ++numFileScopeAdded;
 66     }
 67     if (!classTemplate->Ns()->IsGlobalNamespace())
 68     {
 69         FileScope* primaryFileScope = new FileScope(&boundCompileUnit.GetModule());
 70         primaryFileScope->AddContainerScope(classTemplate->Ns()->GetContainerScope());
 71         boundCompileUnit.AddFileScope(primaryFileScope);
 72         ++numFileScopeAdded;
 73     }
 74     ContainerScope resolveScope;
 75     resolveScope.SetParent(containerScope);
 76     std::vector<std::unique_ptr<BoundTemplateParameterSymbol>> boundTemplateParameters;
 77     for (int i = 0; i < n; ++i)
 78     {
 79         TemplateParameterSymbol* templateParameterSymbol = classTemplate->TemplateParameters()[i];
 80         BoundTemplateParameterSymbol* boundTemplateParameter = new BoundTemplateParameterSymbol(span, templateParameterSymbol->Name());
 81         boundTemplateParameters.push_back(     std::unique_ptr<BoundTemplateParameterSymbol>(boundTemplateParameter));
 82         if (i < m)
 83         {
 84             boundTemplateParameter->SetType(templateArgumentTypes[i]);
 85             resolveScope.Install(boundTemplateParameter);
 86         }
 87         else
 88         {
 89             if (i >= classNode->TemplateParameters().Count())
 90             {
 91                 throw Exception(&boundCompileUnit.GetModule(), "too few template arguments", span);
 92             }
 93             Node* defaultTemplateArgumentNode = classNode->TemplateParameters()[i]->DefaultTemplateArgument();
 94             if (!defaultTemplateArgumentNode)
 95             {
 96                 throw Exception(&boundCompileUnit.GetModule(), "too few template arguments", span);
 97             }
 98             TypeSymbol* templateArgumentType = ResolveType(defaultTemplateArgumentNode, boundCompileUnit, &resolveScope);
 99             templateArgumentTypes.push_back(templateArgumentType);
100         }
101     }
102     for (int i = 0; i < numFileScopeAdded; ++i)
103     {
104         boundCompileUnit.RemoveLastFileScope();
105     }
106 }

108 void ClassTemplateRepository::BindClassTemplateSpecialization(ClassTemplateSpecializationSymbol* classTemplateSpecialization, ContainerScope* containerScope, const Span& span)
109 {
110     if (classTemplateSpecialization->IsBound()) return;
111     SymbolTable& symbolTable = boundCompileUnit.GetSymbolTable();
112     ClassTypeSymbol* classTemplate = classTemplateSpecialization->GetClassTemplate();
113     Node* node = symbolTable.GetNodeNoThrow(classTemplate);
114     if (!node)
115     {
116         classTemplate->ReadAstNodes();
117         node = classTemplate->GetClassNode();
118         Assert(node, "class node not read");
119     }
120     Assert(node->GetNodeType() == NodeType::classNode, "class node expected");
121     ClassNode* classNode = static_cast<ClassNode*>(node);
122     std::unique_ptr<NamespaceNode> globalNs(new NamespaceNode(classTemplate->GetSpan(), new IdentifierNode(classTemplate->GetSpan(), U"")));
123     NamespaceNode* currentNs = globalNs.get();
124     CloneContext cloneContext;
125     cloneContext.SetInstantiateClassNode();
126     int nu = classTemplate->UsingNodes().Count();
127     for (int i = 0; i < nu; ++i)
128     {
129         Node* usingNode = classTemplate->UsingNodes()[i];
130         globalNs->AddMember(usingNode->Clone(cloneContext));
131     }
132     bool fileScopeAdded = false;
133     if (!classTemplate->Ns()->IsGlobalNamespace())
134     {
135         FileScope* primaryFileScope = new FileScope(&boundCompileUnit.GetModule());
136         primaryFileScope->AddContainerScope(classTemplate->Ns()->GetContainerScope());
137         boundCompileUnit.AddFileScope(primaryFileScope);
138         fileScopeAdded = true;
139         std::u32string fullNsName = classTemplate->Ns()->FullName();
140         std::vector<std::u32string> nsComponents = Split(fullNsName, '.');
141         for (const std::u32string& nsComponent : nsComponents)
142         {
143             NamespaceNode* nsNode = new NamespaceNode(classTemplate->GetSpan(), new IdentifierNode(classTemplate->GetSpan(), nsComponent));
144             currentNs->AddMember(nsNode);
145             currentNs = nsNode;
146         }
147     }
148     ClassNode* classInstanceNode = static_cast<ClassNode*>(classNode->Clone(cloneContext));
149     currentNs->AddMember(classInstanceNode);
150     int n = classTemplate->TemplateParameters().size();
151     int m = classTemplateSpecialization->TemplateArgumentTypes().size();
152     if (n != m)
153     {
154         throw Exception(&boundCompileUnit.GetModule(), "wrong number of template arguments", span);
155     }
156     bool markedExport = classTemplateSpecialization->MarkedExport();
157     bool templateParameterBinding = false;
158     ContainerScope resolveScope;
159     resolveScope.SetParent(containerScope);
160     for (int i = 0; i < n; ++i)
161     {
162         TemplateParameterSymbol* templateParameter = classTemplate->TemplateParameters()[i];
163         BoundTemplateParameterSymbol* boundTemplateParameter = new BoundTemplateParameterSymbol(span, templateParameter->Name());
164         TypeSymbol* templateArgumentType = classTemplateSpecialization->TemplateArgumentTypes()[i];
165         boundTemplateParameter->SetType(templateArgumentType);
166         if (templateArgumentType->GetSymbolType() == SymbolType::templateParameterSymbol)
167         {
168             templateParameterBinding = true;
169             if (classTemplateSpecialization->IsPrototype())
170             {
171                 if (classTemplateSpecialization->IsProject())
172                 {
173                     resolveScope.Install(boundTemplateParameter);
174                     TemplateParameterNode* templateParameterNode = classNode->TemplateParameters()[i];
175                     Node* defaultTemplateArgumentNode = templateParameterNode->DefaultTemplateArgument();
176                     if (defaultTemplateArgumentNode)
177                     {
178                         TypeSymbol* templateArgumentType = ResolveType(defaultTemplateArgumentNode, boundCompileUnit, &resolveScope);
179                         templateParameter->SetDefaultType(templateArgumentType);
180                     }
181                 }
182             }
183         }
184         classTemplateSpecialization->AddMember(boundTemplateParameter);
185     }
186     symbolTable.SetCurrentCompileUnit(boundCompileUnit.GetCompileUnitNode());
187     SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
188     symbolCreatorVisitor.SetClassInstanceNode(classInstanceNode);
189     symbolCreatorVisitor.SetClassTemplateSpecialization(classTemplateSpecialization);
190     globalNs->Accept(symbolCreatorVisitor);
191     TypeBinder typeBinder(boundCompileUnit);
192     if (templateParameterBinding)
193     {
194         typeBinder.CreateMemberSymbols();
195     }
196     globalNs->Accept(typeBinder);
197     if (templateParameterBinding)
198     {
199         classTemplateSpecialization->SetGlobalNs(std::move(globalNs));
200         if (fileScopeAdded)
201         {
202             boundCompileUnit.RemoveLastFileScope();
203         }
204     }
205     else if (boundCompileUnit.BindingTypes())
206     {
207         classTemplateSpecialization->SetGlobalNs(std::move(globalNs));
208         classTemplateSpecialization->SetStatementsNotBound();
209         if (fileScopeAdded)
210         {
211             FileScope* fileScope = boundCompileUnit.ReleaseLastFileScope();
212             classTemplateSpecialization->SetFileScope(fileScope);
213         }
214     }
215     else
216     {
217         StatementBinder statementBinder(boundCompileUnit);
218         globalNs->Accept(statementBinder);
219         classTemplateSpecialization->SetGlobalNs(std::move(globalNs));
220         if (fileScopeAdded)
221         {
222             boundCompileUnit.RemoveLastFileScope();
223         }
224     }

227 void ClassTemplateRepository::Instantiate(FunctionSymbol* memberFunction, ContainerScope* containerScope, BoundFunction* currentFunction, const Span& span)
228 {
229     if (instantiatedMemberFunctions.find(memberFunction) != instantiatedMemberFunctions.cend()) return;
230     instantiatedMemberFunctions.insert(memberFunction);
231     try
232     {
233         SymbolTable& symbolTable = boundCompileUnit.GetSymbolTable();
234         Symbol* parent = memberFunction->Parent();
235         Assert(parent->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol, "class template specialization expected");
236         ClassTemplateSpecializationSymbol* classTemplateSpecialization = static_cast<ClassTemplateSpecializationSymbol*>(parent);
237         boundCompileUnit.FinalizeBinding(classTemplateSpecialization);
238         ClassTypeSymbol* classTemplate = classTemplateSpecialization->GetClassTemplate();
239         std::unordered_map<TemplateParameterSymbol*, TypeSymbol*> templateParameterMap;
240         int n = classTemplateSpecialization->TemplateArgumentTypes().size();
241         for (int i = 0; i < n; ++i)
242         {
243             TemplateParameterSymbol* templateParameter = classTemplate->TemplateParameters()[i];
244             TypeSymbol* templateArgument = classTemplateSpecialization->TemplateArgumentTypes()[i];
245             templateParameterMap[templateParameter] = templateArgument;
246         }
247         if (!classTemplateSpecialization->IsConstraintChecked())
248         {
249             classTemplateSpecialization->SetConstraintChecked();
250             if (classTemplate->Constraint())
251             {
252                 std::unique_ptr<BoundConstraint> boundConstraint;
253                 std::unique_ptr<Exception> conceptCheckException;
254                 if (!CheckConstraint(classTemplate->Constraint(), classTemplate->UsingNodes(), boundCompileUnit, containerScope, currentFunction, classTemplate->TemplateParameters(), 
255                     templateParameterMap, boundConstraint, span, memberFunction, conceptCheckException))
256                 {
257                     if (conceptCheckException)
258                     {
259                         throw Exception(&boundCompileUnit.GetModule(), "concept check of class template specialization '" + ToUtf8(classTemplateSpecialization->FullName()) + "' failed: " + conceptCheckException->Message(), span, 
260                             conceptCheckException->References());
261                     }
262                     else
263                     {
264                         throw Exception(&boundCompileUnit.GetModule(), "concept check of class template specialization '" + ToUtf8(classTemplateSpecialization->FullName()) + "' failed.", span);
265                     }
266                 }
267             }
268         }
269         FileScope* fileScope = new FileScope(&boundCompileUnit.GetModule());
270         int nu = classTemplate->UsingNodes().Count();
271         for (int i = 0; i < nu; ++i)
272         {
273             Node* usingNode = classTemplate->UsingNodes()[i];
274             if (usingNode->GetNodeType() == NodeType::namespaceImportNode)
275             {
276                 NamespaceImportNode* namespaceImportNode = static_cast<NamespaceImportNode*>(usingNode);
277                 fileScope->InstallNamespaceImport(containerScope, namespaceImportNode);
278             }
279             else if (usingNode->GetNodeType() == NodeType::aliasNode)
280             {
281                 AliasNode* aliasNode = static_cast<AliasNode*>(usingNode);
282                 fileScope->InstallAlias(containerScope, aliasNode);
283             }
284         }
285         if (!classTemplate->Ns()->IsGlobalNamespace())
286         {
287             fileScope->AddContainerScope(classTemplate->Ns()->GetContainerScope());
288         }
289         boundCompileUnit.AddFileScope(fileScope);
290         Node* node = symbolTable.GetNode(memberFunction);
291         Assert(node->IsFunctionNode(), "function node expected");
292         FunctionNode* functionInstanceNode = static_cast<FunctionNode*>(node);
293         Assert(functionInstanceNode->BodySource(), "body source expected");
294         CloneContext cloneContext;
295         functionInstanceNode->SetBody(static_cast<CompoundStatementNode*>(functionInstanceNode->BodySource()->Clone(cloneContext)));
296         if (functionInstanceNode->WhereConstraint())
297         {
298             std::unique_ptr<BoundConstraint> boundConstraint;
299             std::unique_ptr<Exception> conceptCheckException;
300             FileScope* classTemplateScope = new FileScope(&boundCompileUnit.GetModule());
301             classTemplateScope->AddContainerScope(classTemplateSpecialization->GetContainerScope());
302             boundCompileUnit.AddFileScope(classTemplateScope);
303             if (!CheckConstraint(functionInstanceNode->WhereConstraint(), classTemplate->UsingNodes(), boundCompileUnit, containerScope, currentFunction, classTemplate->TemplateParameters(), 
304                 templateParameterMap, boundConstraint, span, memberFunction, conceptCheckException))
305             {
306                 boundCompileUnit.RemoveLastFileScope();
307                 if (conceptCheckException)
308                 {
309                     std::vector<Span> references;
310                     references.push_back(conceptCheckException->Defined());
311                     references.insert(references.end(), conceptCheckException->References().begin(), conceptCheckException->References().end());
312                     throw Exception(&boundCompileUnit.GetModule(), "concept check of class template member function '" + ToUtf8(memberFunction->FullName()) + "' failed: " + conceptCheckException->Message(), span, references);
313                 }
314                 else
315                 {
316                     throw Exception(&boundCompileUnit.GetModule(), "concept check of class template template member function '" + ToUtf8(memberFunction->FullName()) + "' failed.", span);
317                 }
318             }
319             else
320             {
321                 boundCompileUnit.RemoveLastFileScope();
322             }
323         }
324         symbolTable.SetCurrentCompileUnit(boundCompileUnit.GetCompileUnitNode());
325         SymbolCreatorVisitor symbolCreatorVisitor(symbolTable);
326         symbolTable.BeginContainer(memberFunction);
327         functionInstanceNode->Body()->Accept(symbolCreatorVisitor);
328         symbolTable.EndContainer();
329         TypeBinder typeBinder(boundCompileUnit);
330         typeBinder.SetContainerScope(memberFunction->GetContainerScope());
331         functionInstanceNode->Body()->Accept(typeBinder);
332         StatementBinder statementBinder(boundCompileUnit);
333         std::unique_ptr<BoundClass> boundClass(new BoundClass(&boundCompileUnit.GetModule(), classTemplateSpecialization));
334         statementBinder.SetCurrentClass(boundClass.get());
335         std::unique_ptr<BoundFunction> boundFunction(new BoundFunction(&boundCompileUnit.GetModule(), memberFunction));
336         statementBinder.SetCurrentFunction(boundFunction.get());
337         statementBinder.SetContainerScope(memberFunction->GetContainerScope());
338         if (memberFunction->GetSymbolType() == SymbolType::constructorSymbol)
339         {
340             ConstructorSymbol* constructorSymbol = static_cast<ConstructorSymbol*>(memberFunction);
341             Node* node = symbolTable.GetNode(memberFunction);
342             Assert(node->GetNodeType() == NodeType::constructorNode, "constructor node expected");
343             ConstructorNode* constructorNode = static_cast<ConstructorNode*>(node);
344             statementBinder.SetCurrentConstructor(constructorSymbol, constructorNode);
345         }
346         else if (memberFunction->GetSymbolType() == SymbolType::destructorSymbol)
347         {
348             DestructorSymbol* destructorSymbol = static_cast<DestructorSymbol*>(memberFunction);
349             Node* node = symbolTable.GetNode(memberFunction);
350             Assert(node->GetNodeType() == NodeType::destructorNode, "destructor node expected");
351             DestructorNode* destructorNode = static_cast<DestructorNode*>(node);
352             statementBinder.SetCurrentDestructor(destructorSymbol, destructorNode);
353         }
354         else if (memberFunction->GetSymbolType() == SymbolType::memberFunctionSymbol)
355         {
356             MemberFunctionSymbol* memberFunctionSymbol = static_cast<MemberFunctionSymbol*>(memberFunction);
357             Node* node = symbolTable.GetNode(memberFunction);
358             Assert(node->GetNodeType() == NodeType::memberFunctionNode, "member function node expected");
359             MemberFunctionNode* memberFunctionNode = static_cast<MemberFunctionNode*>(node);
360             statementBinder.SetCurrentMemberFunction(memberFunctionSymbol, memberFunctionNode);
361         }
362         functionInstanceNode->Body()->Accept(statementBinder);
363         BoundStatement* boundStatement = statementBinder.ReleaseStatement();
364         Assert(boundStatement->GetBoundNodeType() == BoundNodeType::boundCompoundStatement, "bound compound statement expected");
365         BoundCompoundStatement* compoundStatement = static_cast<BoundCompoundStatement*>(boundStatement);
366         boundFunction->SetBody(     std::unique_ptr<BoundCompoundStatement>(compoundStatement));
367         boundClass->AddMember(std::move(boundFunction));
368         boundCompileUnit.AddBoundNode(std::move(boundClass));
369         boundCompileUnit.RemoveLastFileScope();
370         InstantiateDestructorAndVirtualFunctions(classTemplateSpecialization, containerScope, currentFunction, span);
371     }
372     catch (const Exception& ex;)
373     {
374         std::vector<Span> references;
375         references.push_back(memberFunction->GetSpan());
376         references.push_back(ex.Defined());
377         references.insert(references.end(), ex.References().begin(), ex.References().end());
378         throw Exception(&boundCompileUnit.GetModule(), "could not instantiate member function '" + ToUtf8(memberFunction->FullName()) + "'. Reason: " + ex.Message(), span, references);
379     }
380 }
381 
382 void ClassTemplateRepository::InstantiateDestructorAndVirtualFunctions(ClassTemplateSpecializationSymbol* classTemplateSpecialization, ContainerScope* containerScope, BoundFunction* currentFunction, const Span& span)
383 {
384     for (FunctionSymbol* virtualMemberFunction : classTemplateSpecialization->Vmt())
385     {
386         if (virtualMemberFunction->Parent() == classTemplateSpecialization && !virtualMemberFunction->IsGeneratedFunction())
387         {
388             Instantiate(virtualMemberFunction, containerScope, currentFunction, span);
389         }
390     }
391     if (classTemplateSpecialization->Destructor())
392     {
393         if (!classTemplateSpecialization->Destructor()->IsGeneratedFunction())
394         {
395             Instantiate(classTemplateSpecialization->Destructor(), containerScope, currentFunction, span);
396         }
397     }
398 }
