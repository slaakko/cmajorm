 18 BoundCompoundStatement* BoundStatement::Block()
 19 {
 20     if (GetBoundNodeType() == BoundNodeType::boundCompoundStatement)
 21     {
 22         return static_cast<BoundCompoundStatement*>(this);
 23     }
 24     return parent->Block();
 25 }
