 22 enum class BoundStatementFlags : uint8_t 
 23 {
 24     none = 0, 
 25     postfix = 1 << 0, 
 26     generated = 1 << 1
 27 };
 28 
 29 inline BoundStatementFlags operator|(BoundStatementFlags left, BoundStatementFlags right)
 30 {
 31     return BoundStatementFlags(uint8_t(left) | uint8_t(right));
 32 }
 33 
 34 inline BoundStatementFlags operator&(BoundStatementFlags left, BoundStatementFlags right)
 35 {
 36     return BoundStatementFlags(uint8_t(left) & uint8_t(right));
 37 }

 39 class BoundStatement : public BoundNode
 40 {
 41 public:
 42     BoundStatement(Module* module_, const Span& span_, BoundNodeType boundNodeType_);
 43     void Load(Emitter& emitter, OperationFlags flags) override;
 44     void Store(Emitter& emitter, OperationFlags flags) override;
 45     BoundStatement* Parent() { return parent; }
 46     void SetParent(BoundStatement* parent_) { parent = parent_; }
 47     BoundCompoundStatement* Block();
 48     void SetLabel(const std::u32string& label_);
 49     const std::u32string& Label() const { return label; }
 50     void SetPostfix() { SetFlag(BoundStatementFlags::postfix); }
 51     bool Postfix() const { return GetFlag(BoundStatementFlags::postfix); }
 52     void SetGenerated() { SetFlag(BoundStatementFlags::generated); }
 53     bool Generated() const { return GetFlag(BoundStatementFlags::generated); }
 54 private:
 55     std::u32string label;
 56     BoundStatement* parent;
 57     BoundStatementFlags flags;
 58     bool GetFlag(BoundStatementFlags flag) const { return (flags & flag) != BoundStatementFlags::none;  }
 59     void SetFlag(BoundStatementFlags flag) { flags = flags | flag; }
 60 };
