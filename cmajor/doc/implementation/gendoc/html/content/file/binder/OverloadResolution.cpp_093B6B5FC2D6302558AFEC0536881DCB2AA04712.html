  22 bool BetterFunctionMatch::operator()(const FunctionMatch& left, const FunctionMatch& right) const
  23 {
  24     int leftBetterArgumentMatches = 0;
  25     int rightBetterArgumentMatches = 0;
  26     int n = std::max(int(left.argumentMatches.size()), int(right.argumentMatches.size()));
  27     for (int i = 0; i < n; ++i)
  28     {
  29         ArgumentMatch leftMatch;
  30         if (i < int(left.argumentMatches.size()))
  31         {
  32             leftMatch = left.argumentMatches[i];
  33         }
  34         ArgumentMatch rightMatch;
  35         if (i < int(right.argumentMatches.size()))
  36         {
  37             rightMatch = right.argumentMatches[i];
  38         }
  39         if (BetterArgumentMatch(leftMatch, rightMatch))
  40         {
  41             ++leftBetterArgumentMatches;
  42         }
  43         else if (BetterArgumentMatch(rightMatch, leftMatch))
  44         {
  45             ++rightBetterArgumentMatches;
  46         }
  47     }
  48     if (leftBetterArgumentMatches > rightBetterArgumentMatches)
  49     {
  50         return true;
  51     }
  52     if (rightBetterArgumentMatches > leftBetterArgumentMatches)
  53     {
  54         return false;
  55     }
  56     if (left.numConversions < right.numConversions)
  57     {
  58         return true;
  59     }
  60     if (left.numConversions > right.numConversions)
  61     {
  62         return false;
  63     }
  64     if (left.numQualifyingConversions < right.numQualifyingConversions)
  65     {
  66         return true;
  67     }
  68     if (left.numQualifyingConversions > right.numQualifyingConversions)
  69     {
  70         return false;
  71     }
  72     if (!left.fun->IsFunctionTemplate() && right.fun->IsFunctionTemplate())
  73     {
  74         return true;
  75     }

  77     {
  78         return false;
  79     }
  80     if (!left.fun->IsTemplateSpecialization() && right.fun->IsTemplateSpecialization())
  81     {

  83     }

  85     {

  87     }
  88     if (left.boundConstraint && !right.boundConstraint)
  89     {
  90         return true;
  91     }
  92     if (!left.boundConstraint && right.boundConstraint)
  93     {
  94         return false;

 101         {
 102             return true;
 103         }
 104         if (rightSubsumeLeft && !leftSubsumeRight)
 105         {
 106             return false;
 107         }
 108     }

 246 bool FindTemplateParameterMatch(TypeSymbol* sourceType, TypeSymbol* targetType, ConversionType conversionType, BoundExpression* argument, 
 247     BoundCompileUnit& boundCompileUnit, FunctionMatch& functionMatch, ContainerScope* containerScope, BoundFunction* currentFunction, const Span& span)

 251     TypeSymbol* templateArgumentType = nullptr;

 253     if (it == functionMatch.templateParameterMap.cend())
 254     {
 255         templateArgumentType = sourceType->RemoveDerivations(targetType->DerivationRec(), span);
 256         if (templateArgumentType)
 257         {
 258             functionMatch.templateParameterMap[templateParameter] = templateArgumentType;
 259         }
 260         else
 261         {
 262             return false;
 263         }
 264     }
 265     else
 266     {
 267         templateArgumentType = it->second;
 268     }
 269     targetType = targetType->Unify(templateArgumentType, span);
 270     if (!targetType)
 271     {
 272         return false;
 273     }
 274     if (TypesEqual(sourceType, targetType))
 275     {
 276         functionMatch.argumentMatches.push_back(ArgumentMatch());
 277         return true;
 278     }
 279     else
 280     {
 281         bool qualificationConversionMatch = false;

 284         {
 285             qualificationConversionMatch = FindQualificationConversion(sourceType, targetType, argument, conversionType, span, functionMatch, argumentMatch);
 286         }

 311                         }

 316                         {
 317                             functionMatch.argumentMatches.push_back(argumentMatch);
 318                             return true;
 319                         }

 323                         }
 324                     }

 329                 }

 455 bool FindConversions(BoundCompileUnit& boundCompileUnit, FunctionSymbol* function, std::vector<std::unique_ptr<BoundExpression>>& arguments, FunctionMatch& functionMatch, 
 456     ConversionType conversionType, ContainerScope* containerScope, BoundFunction* currentFunction, const Span& span)
 457 {
 458     if (!currentFunction)
 459     {
 460         if (function->IsProject() && !function->IsBound())
 461         {
 462             Node* node = boundCompileUnit.GetSymbolTable().GetNodeNoThrow(function);
 463             if (node)
 464             {
 465                 TypeBinder typeBinder(boundCompileUnit);
 466                 typeBinder.SetContainerScope(containerScope);
 467                 node->Accept(typeBinder);
 468             }
 469         }
 470     }
 471     int arity = arguments.size();
 472     if (arity == 1 && function->GroupName() == U"@constructor" && arguments[0]->GetType()->IsReferenceType())
 473     {
 474         functionMatch.referenceMustBeInitialized = true;
 475         return false;
 476     }
 477     Assert(arity == function->Arity(), "wrong arity");
 478     for (int i = 0; i < arity; ++i)
 479     {
 480         BoundExpression* argument = arguments[i].get();
 481         TypeSymbol* sourceType = argument->GetType();
 482         if (sourceType->RemoveConst(span)->IsBasicTypeSymbol())
 483         {
 484             sourceType = sourceType->RemoveConst(span);
 485         }
 486         ParameterSymbol* parameter = function->Parameters()[i];
 487         TypeSymbol* targetType = parameter->GetType();
 488         if (arity == 2 && function->GroupName() == U"operator=")
 489         {
 490             if (targetType->IsRvalueReferenceType() && !sourceType->IsRvalueReferenceType() && !argument->GetFlag(BoundExpressionFlags::bindToRvalueReference))
 491             {
 492                 ++functionMatch.numQualifyingConversions;
 493             }
 494             if (i == 0)
 495             {
 496                 if (targetType->IsConstType() && targetType->PointerCount() <= 1)
 497                 {
 498                     functionMatch.cannotAssignToConstObject = true;
 499                     return false;
 500                 }
 501             }
 502             if (i == 0 && TypesEqual(sourceType, targetType))
 503             {
 504                 if (sourceType->IsReferenceType() && !function->IsConstructorDestructorOrNonstaticMemberFunction())
 505                 {
 506                     functionMatch.argumentMatches.push_back(ArgumentMatch(OperationFlags::none, nullptr, OperationFlags::deref, 1));
 507                     ++functionMatch.numConversions;
 508                     continue;
 509                 }
 510             }
 511             else if (i == 1 && sourceType->IsReferenceType() && targetType->IsLvalueReferenceType() && TypesEqual(sourceType->RemoveReference(span)->RemoveConst(span), targetType->RemoveReference(span)))
 512             {
 513                 if (!function->IsConstructorDestructorOrNonstaticMemberFunction())
 514                 {
 515                     functionMatch.argumentMatches.push_back(ArgumentMatch(OperationFlags::none, nullptr, OperationFlags::deref, 1));
 516                     ++functionMatch.numConversions;
 517                     continue;
 518                 }
 519             }
 520             else if (i == 1 && sourceType->IsReferenceType() && targetType->IsRvalueReferenceType() && TypesEqual(sourceType->RemoveReference(span), targetType->RemoveReference(span)))
 521             {
 522                 if (!function->IsConstructorDestructorOrNonstaticMemberFunction())
 523                 {
 524                     functionMatch.argumentMatches.push_back(ArgumentMatch(OperationFlags::none, nullptr, OperationFlags::none, 0));
 525                     continue;
 526                 }
 527             }
 528             else if (i == 1 && !sourceType->IsReferenceType() && argument->IsLvalueExpression() && targetType->IsReferenceType() && TypesEqual(sourceType, targetType->RemoveReference(span)))
 529             {
 530                 if (!function->IsConstructorDestructorOrNonstaticMemberFunction())
 531                 {
 532                     functionMatch.argumentMatches.push_back(ArgumentMatch(OperationFlags::none, nullptr, OperationFlags::none,  0));
 533                     continue;
 534                 }
 535             }
 536             else if (i == 1 && function->IsLvalueReferenceCopyAssignment() && TypesEqual(sourceType, targetType->RemoveReference(span)))
 537             {
 538                 functionMatch.argumentMatches.push_back(ArgumentMatch(OperationFlags::none, nullptr, OperationFlags::none, 0));
 539                 continue;
 540             }

 542             {
