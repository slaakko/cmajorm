 131 void TypeBinder::Visit(FunctionNode& functionNode)
 132 {
 133     ContainerScope* prevContainerScope = containerScope;
 134     Symbol* symbol = symbolTable.GetSymbol(&functionNode);
 135     Assert(symbol->GetSymbolType() == SymbolType::functionSymbol, "function symbol expected");
 136     FunctionSymbol* functionSymbol = static_cast<FunctionSymbol*>(symbol);
 137     if (functionSymbol->IsBound()) return;
 138     functionSymbol->SetBound();
 139     if (GetGlobalFlag(GlobalFlags::cmdoc))
 140     {
 141         symbolTable.MapSymbol(&functionNode, functionSymbol);
 142     }
 143     FunctionSymbol* prevFunctionSymbol = currentFunctionSymbol;
 144     currentFunctionSymbol = functionSymbol;
 145     if (GetGlobalFlag(GlobalFlags::cmdoc) && functionNode.WhereConstraint())
 146     {
 147         cmdoclib::BindConstraintSymbols(functionNode.WhereConstraint(), containerScope, boundCompileUnit);
 148     }
 149     if (functionSymbol->IsFunctionTemplate())
 150     {
 151         functionSymbol->CloneUsingNodes(usingNodes);
 152         if (functionNode.WhereConstraint())
 153         {
 154             CloneContext cloneContext;
 155             functionSymbol->SetConstraint(static_cast<WhereConstraintNode*>(functionNode.WhereConstraint()->Clone(cloneContext)));
 156         }
 157     }
 158     if (GetGlobalFlag(GlobalFlags::cmdoc) && functionSymbol->Constraint())
 159     {
 160         cmdoclib::BindConstraintSymbols(functionSymbol->Constraint(), functionSymbol->GetContainerScope(), boundCompileUnit);
 161     }
 162     containerScope = functionSymbol->GetContainerScope();
 163     Specifiers specifiers = functionNode.GetSpecifiers();
 164     functionSymbol->SetSpecifiers(specifiers);
 165     if (GetGlobalFlag(GlobalFlags::release) && functionSymbol->IsInline())
 166     {
 167         functionSymbol->CloneUsingNodes(usingNodes);
 168     }
 169     int n = functionNode.Parameters().Count();
 170     for (int i = 0; i < n; ++i)
 171     {
 172         ParameterNode* parameterNode = functionNode.Parameters()[i];
 173         TypeSymbol* parameterType = ResolveType(parameterNode->TypeExpr(), boundCompileUnit, containerScope, typeResolverFlags);
 174         Symbol* symbol = symbolTable.GetSymbol(parameterNode);
 175         Assert(symbol->GetSymbolType() == SymbolType::parameterSymbol, "parameter symbol expected");
 176         ParameterSymbol* parameterSymbol = static_cast<ParameterSymbol*>(symbol);
 177         parameterSymbol->SetType(parameterType);
 178     }
 179     TypeSymbol* returnType = ResolveType(functionNode.ReturnTypeExpr(), boundCompileUnit, containerScope, typeResolverFlags);
 180     functionSymbol->SetReturnType(returnType);
 181     if (!functionSymbol->Constraint() && functionNode.WhereConstraint())
 182     {
 183         CloneContext cloneContext;
 184         functionSymbol->SetConstraint(static_cast<WhereConstraintNode*>(functionNode.WhereConstraint()->Clone(cloneContext)));
 185     }
 186     functionSymbol->ComputeName();
 187     for (ParameterSymbol* parameterSymbol : functionSymbol->Parameters())
 188     {
 189         parameterSymbol->ComputeMangledName();
 190     }
 191     if (functionSymbol->ReturnsClassInterfaceOrClassDelegateByValue())
 192     {
 193         ParameterSymbol* returnParam = new ParameterSymbol(functionNode.ReturnTypeExpr()->GetSpan(), U"@return");
 194         returnParam->SetParent(functionSymbol);
 195         returnParam->SetType(returnType->AddPointer(functionNode.GetSpan()));
 196         functionSymbol->SetReturnParam(returnParam);
 197     }
 198     if (functionNode.Body() && !functionSymbol->IsFunctionTemplate())
 199     {
 200         functionNode.Body()->Accept(*this);
 201     }
 202     else
 203     {
 204         if (!functionSymbol->IsExternal() && !functionSymbol->IsFunctionTemplate())
 205         {
 206             throw Exception(module, "function has no body", functionSymbol->GetSpan());
 207         }
 208     }
 209     boundCompileUnit.GetAttributeBinder()->BindAttributes(functionNode.GetAttributes(), symbol, boundCompileUnit, containerScope);
 210     containerScope = prevContainerScope;
 211     currentFunctionSymbol = prevFunctionSymbol;
 212 }

 222 void TypeBinder::BindClassTemplate(ClassTypeSymbol* classTemplate, ClassNode* classNode)
 223 {
 224     classTemplate->CloneUsingNodes(usingNodes);
 225     if (classNode->WhereConstraint())
 226     {
 227         CloneContext cloneContext;
 228         classTemplate->SetConstraint(static_cast<ConstraintNode*>(classNode->WhereConstraint()->Clone(cloneContext)));
 229         if (GetGlobalFlag(GlobalFlags::cmdoc))
 230         {
 231             cmdoclib::BindConstraintSymbols(classNode->WhereConstraint(), classTemplate->GetContainerScope(), boundCompileUnit);
 232         }
 233     }
 234     classTemplate->SetAccess(classNode->GetSpecifiers() & Specifiers::access_);
 235     classTemplate->ComputeName();
 236     std::vector<TypeSymbol*> templateArgumentTypes;
 237     for (TemplateParameterSymbol* templateParam : classTemplate->TemplateParameters())
 238     {
 239         templateArgumentTypes.push_back(templateParam);
 240     }
 241     ClassTemplateSpecializationSymbol* prototype = symbolTable.MakeClassTemplateSpecialization(classTemplate, templateArgumentTypes, classTemplate->GetSpan());
 242     prototype->SetPrototype();
 243     prototype->MarkExport();
 244     prototype->SetAccess(SymbolAccess::public_);
 245     boundCompileUnit.GetClassTemplateRepository().BindClassTemplateSpecialization(prototype, containerScope, classTemplate->GetSpan());
 246     classTemplate->SetPrototype(prototype);
 247     if (GetGlobalFlag(GlobalFlags::cmdoc) && prototype->Constraint())
 248     {
 249         cmdoclib::BindConstraintSymbols(prototype->Constraint(), prototype->GetContainerScope(), boundCompileUnit);
 250     }
 251 }

 253 void TypeBinder::BindClass(ClassTypeSymbol* classTypeSymbol, ClassNode* classNode, bool fromOwnCompileUnit)
 254 {
 255     if (classTypeSymbol->IsBound()) return;
 256     classTypeSymbol->SetBound();
 257     if (GetGlobalFlag(GlobalFlags::cmdoc))
 258     {
 259         symbolTable.MapSymbol(classNode->Id(), classTypeSymbol);
 260     }
 261     if (!fromOwnCompileUnit)
 262     {
 263         AddUsingNodesToCurrentCompileUnit(classNode);
 264     }
 265     if (classTypeSymbol->IsClassTemplate())
 266     {
 267         BindClassTemplate(classTypeSymbol, classNode);
 268         return;
 269     }
 270     ContainerScope* prevContainerScope = containerScope;
 271     containerScope = classTypeSymbol->GetContainerScope();
 272     classTypeSymbol->SetSpecifiers(classNode->GetSpecifiers());
 273     if (classNode->WhereConstraint())
 274     {
 275         CloneContext cloneContext;
 276         classTypeSymbol->SetConstraint(static_cast<ConstraintNode*>(classNode->WhereConstraint()->Clone(cloneContext)));
 277         if (GetGlobalFlag(GlobalFlags::cmdoc))
 278         {
 279             cmdoclib::BindConstraintSymbols(classNode->WhereConstraint(), containerScope, boundCompileUnit);
 280         }
 281     }
 282     classTypeSymbol->ComputeName();
 283     int nb = classNode->BaseClassOrInterfaces().Count();
 284     for (int i = 0; i < nb; ++i)
 285     {
 286         Node* baseOrInterfaceNode = classNode->BaseClassOrInterfaces()[i];
 287         TypeSymbol* baseOrInterfaceSymbol = ResolveType(baseOrInterfaceNode, boundCompileUnit, containerScope, typeResolverFlags);
 288         if (baseOrInterfaceSymbol->IsClassTypeSymbol())
 289         {
 290             ClassTypeSymbol* baseClassSymbol = static_cast<ClassTypeSymbol*>(baseOrInterfaceSymbol);
 291             if (baseClassSymbol->IsProject() && !GetGlobalFlag(GlobalFlags::info))
 292             {
 293                 Node* node = symbolTable.GetNode(baseClassSymbol);
 294                 Assert(node->GetNodeType() == NodeType::classNode, "class node expected");
 295                 ClassNode* baseClassNode = static_cast<ClassNode*>(node);
 296                 BindClass(baseClassSymbol, baseClassNode, false);
 297             }
 298             if (classTypeSymbol->BaseClass())
 299             {
 300                 throw Exception(module, "class type can have at most one base class", classTypeSymbol->GetSpan(), baseClassSymbol->GetSpan());
 301             }
 302             else if (baseClassSymbol == classTypeSymbol)
 303             {
 304                 throw Exception(module, "class cannot derive from itself", classTypeSymbol->GetSpan());
 305             }
 306             else
 307             {
 308                 classTypeSymbol->SetBaseClass(baseClassSymbol);
 309             }
 310         }
 311         else if (baseOrInterfaceSymbol->GetSymbolType() == SymbolType::interfaceTypeSymbol)
 312         {
 313             InterfaceTypeSymbol* interfaceTypeSymbol = static_cast<InterfaceTypeSymbol*>(baseOrInterfaceSymbol);
 314             if (interfaceTypeSymbol->IsProject())
 315             {
 316                 Node* node = symbolTable.GetNode(interfaceTypeSymbol);
 317                 Assert(node->GetNodeType() == NodeType::interfaceNode, "interface node expected");
 318                 InterfaceNode* interfaceNode = static_cast<InterfaceNode*>(node);
 319                 BindInterface(interfaceTypeSymbol, interfaceNode, false);
 320             }
 321             classTypeSymbol->AddImplementedInterface(interfaceTypeSymbol);
 322         }
 323         else
 324         {
 325             throw Exception(module, "symbol '" + ToUtf8(baseOrInterfaceSymbol->FullName()) + "' is not a class or interface type symbol", baseOrInterfaceNode->GetSpan(), baseOrInterfaceSymbol->GetSpan());
 326         }
 327     }
 328     int nm = classNode->Members().Count();
 329     for (int i = 0; i < nm; ++i)
 330     {
 331         Node* member = classNode->Members()[i];
 332         member->Accept(*this);
 333     }
 334     if (classTypeSymbol->BaseClass())
 335     {
 336         classTypeSymbol->GetContainerScope()->SetBase(classTypeSymbol->BaseClass()->GetContainerScope());
 337     }
 338     boundCompileUnit.GetAttributeBinder()->BindAttributes(classNode->GetAttributes(), classTypeSymbol, boundCompileUnit, containerScope);
 339     classTypeSymbol->InitVmt();
 340     classTypeSymbol->InitImts();
 341     classTypeSymbol->CreateLayouts();
 342     if (classTypeSymbol->IsPolymorphic() && !classTypeSymbol->IsPrototypeTemplateSpecialization())
 343     {
 344         symbolTable.AddPolymorphicClass(classTypeSymbol);
 345     }
 346     if (classTypeSymbol->StaticConstructor())
 347     {
 348         symbolTable.AddClassHavingStaticConstructor(classTypeSymbol);
 349     }
 350     if (classTypeSymbol->HasNontrivialDestructor())
 351     {
 352         classTypeSymbol->CreateDestructorSymbol();
 353     }
 354     containerScope = prevContainerScope;
 355 }

 371     if (staticConstructorSymbol->Parent()->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
 372     {
 373         staticConstructorSymbol->SetTemplateSpecialization();
 374         staticConstructorSymbol->SetLinkOnceOdrLinkage();
 375     }
 376     if (!staticConstructorSymbol->Constraint() && staticConstructorNode.WhereConstraint())
 377     {
 378         CloneContext cloneContext;
 379         staticConstructorSymbol->SetConstraint(static_cast<WhereConstraintNode*>(staticConstructorNode.WhereConstraint()->Clone(cloneContext)));
 380     }
 381     staticConstructorSymbol->ComputeName();
 382     if (GetGlobalFlag(GlobalFlags::cmdoc) && staticConstructorNode.WhereConstraint())
 383     {
 384         cmdoclib::BindConstraintSymbols(staticConstructorNode.WhereConstraint(), containerScope, boundCompileUnit);
 385     }
 386     if (GetGlobalFlag(GlobalFlags::cmdoc) && staticConstructorSymbol->Constraint())
 387     {
 388         cmdoclib::BindConstraintSymbols(staticConstructorSymbol->Constraint(), containerScope, boundCompileUnit);
 389     }
 390     if (staticConstructorNode.Body())
 391     {
 392         staticConstructorNode.Body()->Accept(*this);
 393     }
 394     else
 395     {
 396         if (!staticConstructorSymbol->IsTemplateSpecialization())
 397         {
 398             throw Exception(module, "static constructor has no body", staticConstructorSymbol->GetSpan());
 399         }
 400     }
 401     boundCompileUnit.GetAttributeBinder()->BindAttributes(staticConstructorNode.GetAttributes(), symbol, boundCompileUnit, containerScope);
 402     containerScope = prevContainerScope;
 403     currentFunctionSymbol = prevFunctionSymbol;

 406 void TypeBinder::Visit(ConstructorNode& constructorNode)
 407 {
 408     Symbol* symbol = symbolTable.GetSymbol(&constructorNode);
 409     Assert(symbol->GetSymbolType() == SymbolType::constructorSymbol, "constructor symbol expected");
 410     ConstructorSymbol* constructorSymbol = static_cast<ConstructorSymbol*>(symbol);
 411     if (constructorSymbol->IsBound()) return;
 412     constructorSymbol->SetBound();
 413     if (GetGlobalFlag(GlobalFlags::cmdoc))
 414     {
 415         symbolTable.MapSymbol(constructorNode.ClassId(), constructorSymbol);
 416     }
 417     FunctionSymbol* prevFunctionSymbol = currentFunctionSymbol;
 418     currentFunctionSymbol = constructorSymbol;
 419     ContainerScope* prevContainerScope = containerScope;
 420     containerScope = constructorSymbol->GetContainerScope();
 421     constructorSymbol->SetSpecifiers(constructorNode.GetSpecifiers());
 422     if (GetGlobalFlag(GlobalFlags::release) && constructorSymbol->IsInline())
 423     {
 424         constructorSymbol->CloneUsingNodes(usingNodes);
 425     }
 426     Symbol* parent = constructorSymbol->Parent();
 427     if (parent->IsStatic())
 428     {
 429         throw Exception(module, "static class cannot contain instance constructors", constructorSymbol->GetSpan(), parent->GetSpan());
 430     }
 431     if (parent->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
 432     {
 433         constructorSymbol->SetTemplateSpecialization();
 434         constructorSymbol->SetLinkOnceOdrLinkage();
 435     }
 436     Assert(parent->IsClassTypeSymbol(), "class type symbol expected");
 437     ClassTypeSymbol* classType = static_cast<ClassTypeSymbol*>(parent);
 438     int n = constructorNode.Parameters().Count();
 439     for (int i = 0; i < n; ++i)
 440     {
 441         ParameterNode* parameterNode = constructorNode.Parameters()[i];
 442         TypeSymbol* parameterType = ResolveType(parameterNode->TypeExpr(), boundCompileUnit, containerScope, typeResolverFlags);
 443         Symbol* symbol = symbolTable.GetSymbol(parameterNode);
 444         Assert(symbol->GetSymbolType() == SymbolType::parameterSymbol, "parameter symbol expected");
 445         ParameterSymbol* parameterSymbol = static_cast<ParameterSymbol*>(symbol);
 446         parameterSymbol->SetType(parameterType);
 447     }
 448     if (constructorNode.WhereConstraint())
 449     {
 450     }
 451     if (!constructorSymbol->Constraint() && constructorNode.WhereConstraint())
 452     {
 453         CloneContext cloneContext;
 454         constructorSymbol->SetConstraint(static_cast<WhereConstraintNode*>(constructorNode.WhereConstraint()->Clone(cloneContext)));
 455     }
 456     constructorSymbol->ComputeName();
 457     if (GetGlobalFlag(GlobalFlags::cmdoc) && constructorNode.WhereConstraint())
 458     {
 459         cmdoclib::BindConstraintSymbols(constructorNode.WhereConstraint(), containerScope, boundCompileUnit);
 460     }
 461     if (GetGlobalFlag(GlobalFlags::cmdoc) && constructorSymbol->Constraint())
 462     {
 463         cmdoclib::BindConstraintSymbols(constructorSymbol->Constraint(), containerScope, boundCompileUnit);
 464     }
 465     for (ParameterSymbol* parameterSymbol : constructorSymbol->Parameters())
 466     {
 467         parameterSymbol->ComputeMangledName();
 468     }
 469     if (constructorSymbol->IsDefaultConstructor())
 470     {
 471         classType->SetDefaultConstructor(constructorSymbol);
 472     }
 473     else if (constructorSymbol->IsCopyConstructor())
 474     {
 475         classType->SetCopyConstructor(constructorSymbol);
 476     }
 477     else if (constructorSymbol->IsMoveConstructor())
 478     {
 479         classType->SetMoveConstructor(constructorSymbol);
 480     }
 481     else if (constructorSymbol->Arity() == 2 && !constructorSymbol->IsExplicit() && !constructorSymbol->IsGeneratedFunction())
 482     {
 483         constructorSymbol->SetConversion();
 484         symbolTable.AddConversion(constructorSymbol);
 485     }
 486     if (constructorNode.Body())
 487     {
 488         if (constructorSymbol->IsDefault() || constructorSymbol->IsSuppressed())
 489         {
 490             throw Exception(module, "default or suppressed constructor cannot have a body", constructorSymbol->GetSpan());
 491         }
 492         constructorNode.Body()->Accept(*this);
 493     }
 494     else
 495     {
 496         if (!constructorSymbol->IsDefault() && !constructorSymbol->IsSuppressed() && !constructorSymbol->IsTemplateSpecialization())
 497         {
 498             throw Exception(module, "constructor has no body", constructorSymbol->GetSpan());
 499         }
 500     }
 501     boundCompileUnit.GetAttributeBinder()->BindAttributes(constructorNode.GetAttributes(), symbol, boundCompileUnit, containerScope);
 502     containerScope = prevContainerScope;
