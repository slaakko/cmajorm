  59 void ConceptIdResolver::Visit(IdentifierNode& identifierNode)
  60 {
  61     const std::u32string& name = identifierNode.Str();
  62     Symbol* symbol = containerScope->Lookup(name, ScopeLookup::this_and_base_and_parent);
  63     if (!symbol)
  64     {
  65         for (const std::unique_ptr<FileScope>& fileScope : boundCompileUnit.FileScopes())
  66         {
  67             symbol = fileScope->Lookup(name);
  68             if (symbol)
  69             {
  70                 break;
  71             }
  72         }
  73     }
  74     if (symbol)
  75     {
  76         if (symbol->GetSymbolType() == SymbolType::conceptGroupSymbol)
  77         {
  78             conceptGroup = static_cast<ConceptGroupSymbol*>(symbol);
  79         }
  80         else if (symbol->GetSymbolType() == SymbolType::namespaceSymbol)
  81         {
  82             ns = static_cast<NamespaceSymbol*>(symbol);
  83         }
  84         else
  85         {
  86             throw Exception(&boundCompileUnit.GetModule(), "symbol '" + ToUtf8(name) + "' does not denote a concept", identifierNode.GetSpan());
  87         }
  88     }
  89     else
  90     {
  91         throw Exception(&boundCompileUnit.GetModule(), "concept symbol '" + ToUtf8(name) + "' not found", identifierNode.GetSpan());
  92     }
  93 }

 380 void ConstraintChecker::Visit(IdentifierNode& identifierNode)
 381 {
 382     Reset();
 383     const std::u32string& name = identifierNode.Str();
 384     Symbol* symbol = containerScope->Lookup(name, ScopeLookup::this_and_base_and_parent);
 385     if (!symbol)
 386     {
 387         for (const std::unique_ptr<FileScope>& fileScope : boundCompileUnit.FileScopes())
 388         {
 389             symbol = fileScope->Lookup(name);
 390             if (symbol)
 391             {
 392                 break;
 393             }
 394         }
 395     }
 396     if (symbol)
 397     {
 398         if (symbol->IsTypeSymbol())
 399         {
 400             type = static_cast<TypeSymbol*>(symbol);
 401         }
 402         else
 403         {
 404             switch (symbol->GetSymbolType())
 405             {
 406                 case SymbolType::typedefSymbol:
 407                 {
 408                     TypedefSymbol* typedefSymbol = static_cast<TypedefSymbol*>(symbol);
 409                     type = typedefSymbol->GetType();
 410                     break;
 411                 }
 412                 case SymbolType::boundTemplateParameterSymbol:
 413                 {
 414                     BoundTemplateParameterSymbol* boundTemplateParameterSymbol = static_cast<BoundTemplateParameterSymbol*>(symbol);
 415                     type = boundTemplateParameterSymbol->GetType();
 416                     break;
 417                 }
 418                 case SymbolType::conceptGroupSymbol:
 419                 {
 420                     conceptGroup = static_cast<ConceptGroupSymbol*>(symbol);
 421                     break;
 422                 }
 423                 case SymbolType::namespaceSymbol:
 424                 {
 425                     NamespaceSymbol* ns = static_cast<NamespaceSymbol*>(symbol);
 426                     NamespaceTypeSymbol* namespaceTypeSymbol = new NamespaceTypeSymbol(ns);
 427                     namespaceTypeSymbols.push_back(     std::unique_ptr<NamespaceTypeSymbol>(namespaceTypeSymbol));
 428                     type = namespaceTypeSymbol;
 429                     FileScope* fileScope = new FileScope(&boundCompileUnit.GetModule());
 430                     NamespaceImportNode importNode(span, new IdentifierNode(span, ns->FullName()));
 431                     fileScope->InstallNamespaceImport(containerScope, &importNode);
 432                     boundCompileUnit.AddFileScope(fileScope);
 433                     ++fileScopesAdded;
 434                     break;
 435                 }
 436                 default:
 437                 {
 438                     throw Exception(&boundCompileUnit.GetModule(), "symbol '" + ToUtf8(symbol->FullName()) + "' does not denote a type or a concept", symbol->GetSpan());
 439                 }
 440             }
 441         }
 442     }
 443     else
 444     {
 445         throw Exception(&boundCompileUnit.GetModule(), "type or concept symbol '" + ToUtf8(name) + "' not found", identifierNode.GetSpan());
 446     }
 447 }

 454 void ConstraintChecker::Visit(DotNode& dotNode)
 455 {
 456     Reset();
 457     dotNode.Subject()->Accept(*this);
 458     TypeSymbol* subjectType = GetType();
 459     if (!subjectType)
 460     {
 461         throw Exception(&boundCompileUnit.GetModule(), "symbol '" + dotNode.Subject()->ToString() + "' does not denote a type", dotNode.Subject()->GetSpan());
 462     }
 463     Scope* typeContainerScope = nullptr;
 464     if (subjectType->IsPointerType())
 465     {
 466         typeContainerScope = subjectType->GetContainerScope();
 467     }
 468     else
 469     {
 470         typeContainerScope = subjectType->BaseType()->GetContainerScope();
 471     }
 472     if (subjectType->GetSymbolType() == SymbolType::namespaceTypeSymbol)
 473     {
 474         NamespaceTypeSymbol* nsTypeSymbol = static_cast<NamespaceTypeSymbol*>(subjectType);
 475         typeContainerScope = nsTypeSymbol->Ns()->GetContainerScope();
 476     }
 477     Reset();
 478     const std::u32string& name = dotNode.MemberId()->Str();
 479     Symbol* symbol = typeContainerScope->Lookup(name, ScopeLookup::this_and_base);
 480     if (symbol)
 481     {
 482         switch (symbol->GetSymbolType())
 483         {
 484             case SymbolType::typedefSymbol:
 485             {
 486                 TypedefSymbol* typedefSymbol = static_cast<TypedefSymbol*>(symbol);
 487                 type = typedefSymbol->GetType();
 488                 break;
 489             }
 490             case SymbolType::boundTemplateParameterSymbol:
 491             {
 492                 BoundTemplateParameterSymbol* boundTemplateParameterSymbol = static_cast<BoundTemplateParameterSymbol*>(symbol);
 493                 type = boundTemplateParameterSymbol->GetType();
 494                 break;
 495             }
 496             case SymbolType::conceptGroupSymbol:
 497             {
 498                 conceptGroup = static_cast<ConceptGroupSymbol*>(symbol);
 499                 break;
 500             }
 501             case SymbolType::namespaceSymbol:
 502             {
 503                 NamespaceSymbol* ns = static_cast<NamespaceSymbol*>(symbol);
 504                 NamespaceTypeSymbol* namespaceTypeSymbol = new NamespaceTypeSymbol(ns);
 505                 namespaceTypeSymbols.push_back(     std::unique_ptr<NamespaceTypeSymbol>(namespaceTypeSymbol));
 506                 type = namespaceTypeSymbol;
 507                 FileScope* fileScope = new FileScope(&boundCompileUnit.GetModule());
 508                 NamespaceImportNode importNode(span, new IdentifierNode(span, ns->FullName()));
 509                 fileScope->InstallNamespaceImport(containerScope, &importNode);
 510                 boundCompileUnit.AddFileScope(fileScope);
 511                 ++fileScopesAdded;
 512                 break;
 513             }
 514             default:
 515             {
 516                 throw Exception(&boundCompileUnit.GetModule(), "symbol '" + ToUtf8(symbol->FullName()) + "' does not denote a type or a concept", symbol->GetSpan());
 517             }
 518         }
 519     }
 520     else
 521     {
 522         throw Exception(&boundCompileUnit.GetModule(), "type or concept symbol '" + ToUtf8(name) + "' not found", dotNode.GetSpan());
 523     }
 524 }
