11 namespace cmajor.parser
12 {
13     [hpp]using namespace cmajor::ast;
14 
15     grammar SourceToken
16     {
17         using Keyword.Keyword;
18         using Keyword.PPKeyword;
19         using Identifier.QualifiedId;
20         using Literal.FloatingLiteral;
21         using Literal.IntegerLiteral;
22         using Literal.CharLiteral;
23         using Literal.StringLiteral;
24         using stdlib.block_comment;
25         using stdlib.newline;
26 
27         SourceTokens(SourceTokenFormatter* formatter)
28             ::= empty{ formatter->BeginFormat(); }
29                 SourceToken(formatter)*
30                 empty{ formatter->EndFormat(); }
31             ;
32 
33         SourceToken(SourceTokenFormatter* formatter)
34             ::= PPKeyword{ formatter->Keyword(std::u32string(matchBegin, matchEnd)); }
35             |   Keyword{ formatter->Keyword(std::u32string(matchBegin, matchEnd)); }
36             |   QualifiedId{ formatter->Identifier(std::u32string(matchBegin, matchEnd)); }
37             |   (FloatingLiteral | IntegerLiteral){ formatter->Number(std::u32string(matchBegin, matchEnd)); }
38             |   CharLiteral{ formatter->Char(std::u32string(matchBegin, matchEnd)); }
39             |   StringLiteral{ formatter->String(std::u32string(matchBegin, matchEnd)); }
40             |   Spaces{ formatter->Spaces(Spaces); }
41             |   LineComment{ formatter->Comment(std::u32string(matchBegin, matchEnd)); }
42             |   block_comment{ formatter->Comment(std::u32string(matchBegin, matchEnd)); }
43             |   newline{ formatter->NewLine(std::u32string(matchBegin, matchEnd)); }
44             |   Other{ formatter->Other(Other); }
45             ;
46 
47         Spaces : std::u32string
48             ::= ([\t ]+){ value = std::u32string(matchBegin, matchEnd); }
49             ;
50 
51         LineComment 
52             ::= "//" [^\r\n]* newline?
53             ;
54 
55         Other : std::u32string
56             ::= [^\r\n]{ value = std::u32string(matchBegin, matchEnd); }
57             ;
58     }
