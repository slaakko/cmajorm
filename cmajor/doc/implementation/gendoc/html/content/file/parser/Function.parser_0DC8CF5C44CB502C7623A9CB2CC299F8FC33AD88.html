33             ::= (((Attributes{ attributes.reset(Attributes); })? Specifiers{ specifierSpan = span; } TypeExpr(ctx) FunctionGroupId(ctx)
34                 { 
35                     groupIdSpan = span;

39                 }
40                 TemplateParameterList(ctx, fun.get())?
41                 ParameterList(ctx, fun.get())){ fun->GetSpan() = span; }
42                 (WhereConstraint(ctx){ fun->SetConstraint(WhereConstraint); })?
43                 (CompoundStatement(ctx){ fun->SetBody(CompoundStatement); } | ';')){ value = fun.release(); }
