17     grammar Enumeration
18     {
19         using Specifier.Specifiers;
20         using Identifier.Identifier;
21         using TypeExpr.TypeExpr;
22         using Expression.Expression;
23 
24         EnumType(ParsingContext* ctx, var Span beginBraceSpan, var Span endBraceSpan) : EnumTypeNode*
25             ::= (Specifiers keyword("enum") Identifier:enumTypeId!){ value = new EnumTypeNode(span, Specifiers, enumTypeId); }
26                 (UnderlyingType(ctx){ value->SetUnderlyingType(UnderlyingType); })?
27                 '{'!{ beginBraceSpan = span; } EnumConstants(ctx, value) '}'!
28                 { 
29                     endBraceSpan = span; 
30                     value->GetSpan().SetEnd(span.End()); 
31                     value->SetBeginBraceSpan(beginBraceSpan); 
32                     value->SetEndBraceSpan(endBraceSpan);
33                 }
34             ;
35 
36         UnderlyingType(ParsingContext* ctx) : Node*
37             ::= ':' TypeExpr(ctx){ value = TypeExpr; }
38             ;
39 
40         EnumConstants(ParsingContext* ctx, EnumTypeNode* enumType)
41             ::= EnumConstant(ctx, enumType)!{ enumType->AddConstant(EnumConstant); } % ','
42             ;
43 
44         EnumConstant(ParsingContext* ctx, EnumTypeNode* enumType, var Span s) : EnumConstantNode*
45             ::= Identifier:constantId!{ s = span; } 
46             (   '=' Expression(ctx):constantValue
47                     { 
48                         s.SetEnd(span.End()); 
49                         value = new EnumConstantNode(s, constantId, constantValue); 
50                         value->SetHasValue(); 
51                         value->SetStrValue(std::u32string(matchBegin, matchEnd));
52                     }
53             |   empty{ s.SetEnd(span.End()); value = new EnumConstantNode(s, constantId, MakeNextEnumConstantValue(s, enumType)); }
54             )
55             ;
56     }
