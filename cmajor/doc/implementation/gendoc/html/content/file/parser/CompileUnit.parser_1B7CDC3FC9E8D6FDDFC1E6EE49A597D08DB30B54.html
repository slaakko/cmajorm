 38         CompileUnit(ParsingContext* ctx) : CompileUnitNode*
 39             ::= empty{ value = new CompileUnitNode(span, fileName); }
 40                 NamespaceContent(ctx, value->GlobalNs())
 41             ;
 42 
 43         NamespaceContent(ParsingContext* ctx, NamespaceNode* ns)
 44             ::= UsingDirectives(ctx, ns)
 45                 Definitions(ctx, ns)
 46             ;
 47 
 48         UsingDirectives(ParsingContext* ctx, NamespaceNode* ns)
 49             ::= UsingDirective(ctx, ns)*
 50             ;
 51 
 52         UsingDirective(ParsingContext* ctx, NamespaceNode* ns)
 53             ::= UsingAliasDirective{ ns->AddMember(UsingAliasDirective); }
 54             |   UsingNamespaceDirective{ ns->AddMember(UsingNamespaceDirective); }
 55             ;
 56 
 57         UsingAliasDirective(var std::unique_ptr<IdentifierNode> id) : Node*
 58             ::= (keyword("using") Identifier{ id.reset(Identifier); } '=' QualifiedId! ';'!){ value = new AliasNode(span, id.release(), QualifiedId); }
 59             ;
 60 
 61         UsingNamespaceDirective : Node*
 62             ::= (keyword("using") QualifiedId! ';'!){ value = new NamespaceImportNode(span, QualifiedId); }
 63             ;
 64 
 65         Definitions(ParsingContext* ctx, NamespaceNode* ns)
 66             ::= (Definition(ctx, ns){ ns->AddMember(Definition); })*
 67             ;
 68 
 69         Definition(ParsingContext* ctx, NamespaceNode* ns) : Node*
 70             ::= NamespaceDefinition(ctx, ns){ value = NamespaceDefinition; }
 71             |   TypedefDeclaration(ctx){ value = TypedefDeclaration; }
 72             |   ConceptDefinition(ctx){ value = ConceptDefinition; }
 73             |   FunctionDefinition(ctx){ value = FunctionDefinition; }
 74             |   ClassDefinition(ctx){ value = ClassDefinition; }
 75             |   InterfaceDefinition(ctx){ value = InterfaceDefinition; }
 76             |   EnumTypeDefinition(ctx){ value = EnumTypeDefinition; }
 77             |   ConstantDefinition(ctx){ value = ConstantDefinition; }
 78             |   DelegateDefinition(ctx){ value = DelegateDefinition; }
 79             |   ClassDelegateDefinition(ctx){ value = ClassDelegateDefinition; }
 80             ;
 81 
 82         NamespaceDefinition(ParsingContext* ctx, NamespaceNode* ns) : NamespaceNode*
 83             ::= (keyword("namespace") QualifiedId!){ value = new NamespaceNode(span, QualifiedId); } '{'! NamespaceContent(ctx, value)! '}'!
 84             ;
