 37 void FileRegistry::PushObtainSystemFileIndeces()
 38 {
 39     obtainsSystemFileIndecesStack.push(obtainSystemFileIndeces);
 40     obtainSystemFileIndeces = true;
 41 }

 44 {

 46     obtainsSystemFileIndecesStack.pop();

 49 uint32_t FileRegistry::RegisterNewFile(const std::string& filePath)
 50 {
 51     std::lock_guard<std::mutex> lock(mtx);
 52     if (obtainSystemFileIndeces)
 53     {
 54         uint32_t fileIndex = SystemFileIndex::Instance().RegisterSystemSourceFile(filePath);
 55         fileMap[fileIndex] = std::make_pair(true, filePath);
 56         return fileIndex;
 57     }
 58     else
 59     {
 60         uint32_t fileIndex = nextUserFileIndex++;
 61         fileMap[fileIndex] = std::make_pair(true, filePath);
 62         return fileIndex;
 63     }
 64 }

 66 void FileRegistry::RegisterExistingFile(uint32_t fileIndex, const std::string& filePath)
 67 {
 68     fileMap[fileIndex] = std::make_pair(false, filePath);
 69 }

 71 std::vector<std::pair<uint32_t, std::string>> FileRegistry::GetFileMap()
 72 {
 73     std::vector<std::pair<uint32_t, std::string>> result;
 74     for (const std::pair<uint32_t, std::pair<bool, std::string>>& p : fileMap)
 75     {
 76         if (p.second.first)
 77         {
 78             result.push_back(std::make_pair(p.first, p.second.second));
 79         }
 80     }
 81     return result;
 82 }

 84 std::string FileRegistry::GetFilePath(uint32_t filePathIndex)
 85 {
 86     std::lock_guard<std::mutex> lock(mtx);
 87     if (filePathIndex >= firstSystemFileIndex)
 88     {
 89         return SystemFileIndex::Instance().GetSystemSourceFilePath(filePathIndex);
 90     }
 91     else
 92     {
 93         auto it = fileMap.find(filePathIndex);
 94         if (it != fileMap.cend())
 95         {
 96             return it->second.second;
 97         }
 98         return std::string();
 99     }
