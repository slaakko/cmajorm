 6 #ifndef CMAJOR_AST_SPECIFIER_INCLUDED

11 namespace cmajor { namespace ast {
12 
13 enum class Specifiers : uint32_t 
14 {
15     none = 0, 
16     public_ = 1 << 0, 
17     protected_ = 1 << 1, 
18     private_ = 1 << 2, 
19     internal_ = 1 << 3, 
20     static_ = 1 << 4, 
21     virtual_ = 1 << 5, 
22     override_ = 1 << 6, 
23     abstract_ = 1 << 7, 
24     inline_ = 1 << 8, 
25     explicit_ = 1 << 9, 
26     external_ = 1 << 10, 
27     suppress_ = 1 << 11, 
28     default_ = 1 << 12, 
29     constexpr_ = 1 << 13, 
30     cdecl_ = 1 << 14, 
31     nothrow_ = 1 << 15, 
32     throw_ = 1 << 16, 
33     new_ = 1 << 17, 
34     const_ = 1 << 18, 
35     unit_test_ = 1 << 19, 
36     access_ = public_ | protected_ | private_ | internal_
37 };
38 
39 inline Specifiers operator|(Specifiers left, Specifiers right)
40 {
41     return Specifiers(uint32_t(left) | uint32_t(right));
42 }
43 
44 inline Specifiers operator&(Specifiers left, Specifiers right)
45 {
46     return Specifiers(uint32_t(left) & uint32_t(right));
47 }
48 
49 inline Specifiers operator~(Specifiers operand)
50 {
51     return Specifiers(~uint32_t(operand));
52 }
53 
54 bool StaticConstructorSpecifiers(Specifiers specifiers);
55 
56 std::string SpecifierStr(Specifiers specifiers);
57 
58 } } // namespace cmajor::ast
