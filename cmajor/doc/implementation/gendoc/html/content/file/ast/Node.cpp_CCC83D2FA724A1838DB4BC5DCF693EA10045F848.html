338 void NodeFactory::Register(NodeType nodeType, NodeCreator* creator)
339 {
340     creators[static_cast<size_t>(nodeType)] =      std::unique_ptr<NodeCreator>(creator);
341 }
342 
343 Node* NodeFactory::CreateNode(NodeType nodeType, const Span& span)
344 {
345     const std::unique_ptr<NodeCreator>& creator = creators[static_cast<size_t>(nodeType)];
346     if (creator)
347     {
348         Node* value = creator->CreateNode(span);
349         if (value)
350         {
351             return value;
352         }
353         else
354         {
355             throw std::runtime_error("could not create node");
356         }
357     }
358     else
359     {
360         throw std::runtime_error("no creator for node type '" + NodeTypeStr(nodeType) + "'");
361     }
362 }
