 37 Attribute* Attributes::GetAttribute(const std::u32string& name) const
 38 {
 39     auto it = attributeMap.find(name);
 40     if (it != attributeMap.cend())
 41     {
 42         return it->second;
 43     }
 44     return nullptr;
 45 }

 47 void Attributes::AddAttribute(const Span& span, const std::u32string& name)
 48 {
 49     AddAttribute(span, name, U"true");
 50 }
 51 
 52 void Attributes::AddAttribute(const Span& span, const std::u32string& name, const std::u32string& value)
 53 {
 54     Attribute* prev = GetAttribute(name);
 55     if (prev != nullptr)
 56     {
 57         throw AttributeNotUniqueException("attribute '" + ToUtf8(name) + "' not unique", span, prev->GetSpan());
 58     }
 59     Attribute* attribute = new Attribute(span, name, value);
 60     AddAttribute(attribute);
 61 }
 62 
 63 void Attributes::AddAttribute(Attribute* attribute)
 64 {
 65     attributes.push_back(     std::unique_ptr<Attribute>(attribute));
 66     attributeMap[attribute->Name()] = attribute;
 67 }
 68 
 69 Attributes* Attributes::Clone() const
 70 {
 71     std::unique_ptr<Attributes> clone(new Attributes());
 72     for (const std::unique_ptr<Attribute>& attribute : attributes)
 73     {
 74         clone->AddAttribute(attribute->GetSpan(), attribute->Name(), attribute->Value());
 75     }
 76     return clone.release();
 77 }

 79 void Attributes::Write(AstWriter& writer)
 80 {
 81     writer.GetBinaryWriter().WriteULEB128UInt(attributes.size());
 82     for (const std::unique_ptr<Attribute>& attribute : attributes)
 83     {
 84         attribute->Write(writer);
 85     }
 86 }
 87 
 88 void Attributes::Read(AstReader& reader)
 89 {
 90     uint32_t n = reader.GetBinaryReader().ReadULEB128UInt();
 91     for (uint32_t i = 0; i < n; ++i)
 92     {
 93         Attribute* attribute = new Attribute();
 94         attribute->Read(reader);
 95         AddAttribute(attribute);
 96     }
 97 }
