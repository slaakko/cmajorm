 22 Node* NamespaceNode::Clone(CloneContext& cloneContext) const
 23 {
 24     NamespaceNode* clone = new NamespaceNode(GetSpan(), static_cast<IdentifierNode*>(id->Clone(cloneContext)));
 25     int n = members.Count();
 26     for (int i = 0; i < n; ++i)
 27     {
 28         Node* member = members[i];
 29         if (cloneContext.MakeTestUnits() && member->GetNodeType() == NodeType::functionNode && (static_cast<FunctionNode*>(member)->GetSpecifiers() & Specifiers::unit_test_) != Specifiers::none)
 30         {
 31             FunctionNode* unitTestFunction = static_cast<FunctionNode*>(member->Clone(cloneContext));
 32             unitTestFunction->SetParent(const_cast<NamespaceNode*>(this));
 33             cloneContext.AddUnitTestFunction(     std::unique_ptr<FunctionNode>(unitTestFunction));
 34         }
 35         else
 36         {
 37             clone->AddMember(member->Clone(cloneContext));
 38         }
 39     }
 40     return clone;
 41 }
