 6 #ifndef CMAJOR_AST_NODE_LIST_INCLUDED
 7 #define CMAJOR_AST_NODE_LIST_INCLUDED
 8 #include <cmajor/ast/AstWriter.hpp>
 9 #include <cmajor/ast/AstReader.hpp>
10 
11 namespace cmajor { namespace ast {
12 
13 template<typename T>
14 class NodeList 
15 {
16 public:
17     int Count() const
18     {
19         return static_cast<int>(nodes.size());
20     }
21     T* operator[](int index) const
22     {
23         return nodes[index].get();
24     }
25     void Add(T* node)
26     {
27         nodes.push_back(     std::unique_ptr<T>(node));
28     }
29     void SetParent(Node* parent)
30     {
31         for (const std::unique_ptr<T>& node : nodes)
32         {
33             node->SetParent(parent);
34         }
35     }
36     void Write(AstWriter& writer)
37     {
38         uint32_t n = static_cast<uint32_t>(nodes.size());
39         writer.GetBinaryWriter().WriteULEB128UInt(n);
40         for (uint32_t i = 0; i < n; ++i)
41         {
42             writer.Write(nodes[i].get());
43         }
44     }
45     void Read(AstReader& reader)
46     {
47         uint32_t n = reader.GetBinaryReader().ReadULEB128UInt();
48         for (uint32_t i = 0; i < n; ++i)
49         {
50             Node* node = reader.ReadNode();
51             T* asTNode = dynamic_cast<T*>(node);
52             Assert(asTNode, "wrong node type");
53             nodes.push_back(     std::unique_ptr<T>(asTNode));
54         }
55     }
56     std::vector<std::unique_ptr<T>>& Nodes() { return nodes; }
57 private:
58     std::vector<std::unique_ptr<T>> nodes;
59 };
60 
61 } } // namespace cmajor::ast
62 
63 #endif // CMAJOR_AST_NODE_LIST_INCLUDED
