 54 class Node 
 55 {
 56 public:
 57     Node(NodeType nodeType_, const Span& span_);
 58     virtual ~Node();
 59     NodeType GetNodeType() const { return nodeType; }
 60     virtual Node* Clone(CloneContext& cloneContext) const = 0;
 61     virtual void Accept(Visitor& visitor) = 0;
 62     virtual void Write(AstWriter& writer);
 63     virtual void Read(AstReader& reader);
 64     virtual std::string ToString() const { return std::string(); }
 65     virtual void AddArgument(Node* argument) { Assert(false, "AddArgument not overridden");  }
 66     virtual void AddParameter(ParameterNode* parameter) { Assert(false, "AddParameter not overridden"); }
 67     virtual void AddTemplateParameter(TemplateParameterNode* templateParameter) { Assert(false, "AddTemplateParameter not overridden"); }
 68     virtual bool IsUnsignedTypeNode() const { return false; }
 69     virtual bool IsStatementNode() const { return false; }
 70     virtual bool IsConstraintNode() const { return false; }
 71     virtual bool IsConceptNode() const { return false; }
 72     virtual bool IsFunctionNode() const { return false; }
 73     virtual bool IsIntrinsicConceptNode() const { return false; }
 74     virtual bool IsConditionalCompilationExpressionNode() const { return false; }
 75     const Span& GetSpan() const { return span; }
 76     Span& GetSpan() { return span; }
 77     const Node* Parent() const { return parent; }
 78     Node* Parent() { return parent; }
 79     void SetParent(Node* parent_) { parent = parent_; }
 80 private:
 81     NodeType nodeType;
 82     Span span;
 83     Node* parent;
 84 };
