13 
14 ConstantNode::ConstantNode(const Span& span_) : Node(NodeType::constantNode, span_), specifiers(Specifiers::none)
15 {
16 }
17 
18 ConstantNode::ConstantNode(const Span& span_, Specifiers specifiers_, Node* typeExpr_, IdentifierNode* id_, Node* value_) :
19     Node(NodeType::constantNode, span_), specifiers(specifiers_), typeExpr(typeExpr_), id(id_), value(value_)
20 {
21     typeExpr->SetParent(this);
22     id->SetParent(this);
23     value->SetParent(this);
24 }
25 
26 Node* ConstantNode::Clone(CloneContext& cloneContext) const
27 {
28     return new ConstantNode(GetSpan(), specifiers, typeExpr->Clone(cloneContext), static_cast<IdentifierNode*>(id->Clone(cloneContext)), value->Clone(cloneContext));
29 }
30 
31 void ConstantNode::Accept(Visitor& visitor)
32 {
33     visitor.Visit(*this);
34 }
35 
36 void ConstantNode::Write(AstWriter& writer)
37 {
38     Node::Write(writer);
39     writer.Write(specifiers);
40     writer.Write(typeExpr.get());
41     writer.Write(id.get());
42     writer.Write(value.get());
43     writer.GetBinaryWriter().Write(strValue);
44 }
45 
46 void ConstantNode::Read(AstReader& reader)
47 {
48     Node::Read(reader);
49     specifiers = reader.ReadSpecifiers();
50     typeExpr.reset(reader.ReadNode());
51     typeExpr->SetParent(this);
52     id.reset(reader.ReadIdentifierNode());
53     id->SetParent(this);
54     value.reset(reader.ReadNode());
55     value->SetParent(this);
56     strValue = reader.GetBinaryReader().ReadUtf32String();
57 }
