 18 Node* AstReader::ReadNode()
 19 {
 20     NodeType nodeType = static_cast<NodeType>(binaryReader.ReadByte());
 21     Span span = ReadSpan();
 22     Node* node = NodeFactory::Instance().CreateNode(nodeType, span);
 23     node->Read(*this);
 24     return node;
 25 }

175 Span AstReader::ReadSpan()
176 {
177     bool valid = binaryReader.ReadBool();
178     if (!valid)
179     {
180         return Span();
181     }
182     else
183     {
184         uint32_t fileIndex = binaryReader.ReadULEB128UInt();
185         if (moduleId != -1)
186         {
187             int16_t fileId = GetFileId(fileIndex);
188             fileIndex = static_cast<uint32_t>(MakeFileIndex(moduleId, fileId));
189         }
190         uint32_t lineNumber = binaryReader.ReadULEB128UInt();
191         uint32_t start = binaryReader.ReadULEB128UInt();
192         uint32_t end = binaryReader.ReadULEB128UInt();
193         return Span(fileIndex, lineNumber, start, end);
194     }
