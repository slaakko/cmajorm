34 void CompileUnitNode::ComputeLineStarts(const std::u32string& sourceFileContent)
35 {
36     int32_t index = 0;
37     bool startOfLine = true;
38     for (const char32_t& c : sourceFileContent)
39     {
40         if (startOfLine)
41         {
42             lineStarts.push_back(index);
43             startOfLine = false;
44         }
45         if (c == U'\n')
46         {
47             startOfLine = true;
48         }
49         ++index;
50     }
51 }

53 int CompileUnitNode::GetColumn(const Span& span) const
54 {
55     if (span.Valid())
56     {
57         int32_t pos = span.Start();
58         auto it = std::lower_bound(lineStarts.cbegin(), lineStarts.cend(), pos);
59         if (it != lineStarts.cend())
60         {
61             int32_t start = *it;
62             if (start != pos && it != lineStarts.cbegin())
63             {
64                 --it;
65                 start = *it;
66             }
67             int column = 1 + pos - start;
68             return column;
69         }
70     }
71     return 1;
72 }
