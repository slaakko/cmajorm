174 Node* ArrayNode::Clone(CloneContext& cloneContext) const
175 {
176     Node* clonedSize = nullptr;
177     if (size)
178     {
179         clonedSize = size->Clone(cloneContext);
180     }
181     return new ArrayNode(GetSpan(), subject->Clone(cloneContext), clonedSize);
182 }

187 }
188 
189 void ArrayNode::Write(AstWriter& writer)
190 {
191     Node::Write(writer);
192     writer.Write(subject.get());
193     bool hasSize = size != nullptr;
194     writer.GetBinaryWriter().Write(hasSize);
195     if (hasSize)
196     {
197         writer.Write(size.get());
198     }
199 }
200 
201 void ArrayNode::Read(AstReader& reader)
202 {
203     Node::Read(reader);
204     subject.reset(reader.ReadNode());
205     subject->SetParent(this);
206     bool hasSize = reader.GetBinaryReader().ReadBool();
207     if (hasSize)
208     {
209         size.reset(reader.ReadNode());
210         size->SetParent(this);
211     }
212 }
213 
214 std::string ArrayNode::ToString() const
215 {
216     std::string s = subject->ToString();
217     s.append("[");
218     if (size)
219     {
220         s.append(size->ToString());
221     }
222     s.append("]");
223     return s;
224 }
