 14 std::string CmajorRootDir()
 15 {
 16     char* e = getenv("CMAJOR_ROOT");
 17     if (e == nullptr || !*e)
 18     {
 19         throw std::runtime_error("please set 'CMAJOR_ROOT' environment variable to contain /path/to/cmajor directory.");
 20     }
 21     return std::string(e);
 22 }

 64 Project::Project(const std::u32string& name_, const std::string& filePath_, const std::string& config_) :
 65     name(name_), filePath(filePath_), config(config_), target(Target::program), basePath(filePath), isSystemProject(false), logStreamId(0), built(false)
 66 {
 67     basePath.remove_filename();
 68     systemLibDir = CmajorSystemLibDir(config);
 69     boost::filesystem::path mfp(filePath);
 70     boost::filesystem::path fn = mfp.filename();
 71     mfp.remove_filename();
 72     mfp /= "lib";
 73     mfp /= config;
 74     mfp /= fn;
 75     mfp.replace_extension(".cmm");
 76     moduleFilePath = GetFullPath(mfp.generic_string());
 77     boost::filesystem::path lfp(mfp);
 78 #ifdef _WIN32
 79     lfp.replace_extension(".lib");
 80 #else
 81     lfp.replace_extension(".a");
 82 #endif
 83     libraryFilePath = GetFullPath(lfp.generic_string());
 84     boost::filesystem::path efp(filePath);
 85     efp.remove_filename();
 86     efp /= "bin";
 87     efp /= config;
 88     efp /= fn;
 89 #ifdef _WIN32
 90     efp.replace_extension(".exe");
 91 #else
 92     efp.replace_extension();
 93 #endif
 94     executableFilePath = GetFullPath(efp.generic_string());
 95 }

102 void Project::ResolveDeclarations()
103 {
104     for (const std::unique_ptr<ProjectDeclaration>& declaration : declarations)
105     {
106         switch (declaration->GetDeclarationType())
107         {
108             case ProjectDeclarationType::referenceDeclaration:
109             {
110                 ReferenceDeclaration* reference = static_cast<ReferenceDeclaration*>(declaration.get());
111                 boost::filesystem::path rp(reference->FilePath());
112                 relativeReferencedProjectFilePaths.push_back(rp.generic_string());
113                 if (rp.is_absolute())
114                 {
115                     referencedProjectFilePaths.push_back(GetFullPath(rp.generic_string()));
116                 }
117                 else
118                 {
119                     boost::filesystem::path ar = basePath / rp;
120                     referencedProjectFilePaths.push_back(GetFullPath(ar.generic_string()));
121                 }
122                 boost::filesystem::path fn = rp.filename();
123                 rp.remove_filename();
124                 if (rp.is_relative())
125                 {
126                     rp = systemLibDir / rp;
127                 }
128                 rp /= fn;
129                 if (rp.extension() == ".cmp" || rp.extension() == ".cmproj")
130                 {
131                     rp.replace_extension(".cmm");
132                 }
133                 if (rp.extension() != ".cmm")
134                 {
135                     throw std::runtime_error("invalid reference path extension '" + rp.generic_string() + "' (not .cmp, .cmproj or .cmm)");
136                 }
137                 if (!boost::filesystem::exists(rp))
138                 {
139                     rp = reference->FilePath();
140                     rp.remove_filename();
141                     if (rp.is_relative())
142                     {
143                         rp = basePath / rp;
144                     }
145                     rp /= "lib";
146                     rp /= config;
147                     rp /= fn;
148                     if (rp.extension() == ".cmp" || rp.extension() == ".cmproj")
149                     {
150                         rp.replace_extension(".cmm");
151                     }
152                     if (rp.extension() != ".cmm")
153                     {
154                         throw std::runtime_error("invalid reference path extension '" + rp.generic_string() + "' (not .cmp, .cmproj or .cmm)");
155                     }
156                 }
157                 std::string referencePath = GetFullPath(rp.generic_string());
158                 if (std::find(references.cbegin(), references.cend(), referencePath) == references.cend())
159                 {
160                     references.push_back(referencePath);
161                 }
162                 break;
163             }
164             case ProjectDeclarationType::sourceFileDeclaration:
165             {
166                 SourceFileDeclaration* sourceFileDeclaration = static_cast<SourceFileDeclaration*>(declaration.get());
167                 boost::filesystem::path sfp(sourceFileDeclaration->FilePath());
168                 relativeSourceFilePaths.push_back(sfp.generic_string());
169                 if (sfp.is_relative())
170                 {
171                     sfp = basePath / sfp;
172                 }
173                 if (sfp.extension() != ".cm")
174                 {
175                     throw std::runtime_error("invalid source file extension '" + sfp.generic_string() + "' (not .cm)");
176                 }
177                 if (!boost::filesystem::exists(sfp))
178                 {
179                     throw std::runtime_error("source file path '" + GetFullPath(sfp.generic_string()) + "' not found");
180                 }
181                 std::string sourceFilePath = GetFullPath(sfp.generic_string());
182                 if (std::find(sourceFilePaths.cbegin(), sourceFilePaths.cend(), sourceFilePath) == sourceFilePaths.cend() && sourceFilePath != excludeSourceFilePath)
183                 {
184                     sourceFilePaths.push_back(sourceFilePath);
185                 }
186                 break;
187             }
188             case ProjectDeclarationType::targetDeclaration:
189             {
190                 TargetDeclaration* targetDeclaration = static_cast<TargetDeclaration*>(declaration.get());
191                 target = targetDeclaration->GetTarget();
192                 break;
193             }
194             case ProjectDeclarationType::textFileDeclaration:
195             {
196                 TextFileDeclaration* textFileDeclaration = static_cast<TextFileDeclaration*>(declaration.get());
197                 boost::filesystem::path tfp(textFileDeclaration->FilePath());
198                 relativeTextFilePaths.push_back(tfp.generic_string());
199                 break;
200             }
201             default:
202             {
203                 throw std::runtime_error("unknown project declaration");
204             }
205         }
206     }
207 }

229 bool Project::IsUpToDate(const std::string& systemModuleFilePath) const
230 {
231     if (!boost::filesystem::exists(moduleFilePath))
232     {
233         return false;
234     }
235     for (const std::string& sourceFilePath : sourceFilePaths)
236     {
237         if (boost::filesystem::last_write_time(sourceFilePath) > boost::filesystem::last_write_time(moduleFilePath))
238         {
239             return false;
240         }
241     }
242     if (!systemModuleFilePath.empty() && !IsSystemProject() && boost::filesystem::exists(systemModuleFilePath))
243     {
244         if (boost::filesystem::last_write_time(systemModuleFilePath) > boost::filesystem::last_write_time(moduleFilePath))
245         {
246             return false;
247         }
248     }
249     return true;
250 }
