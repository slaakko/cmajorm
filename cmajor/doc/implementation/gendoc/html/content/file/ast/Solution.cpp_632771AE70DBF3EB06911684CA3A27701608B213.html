 47 void Solution::ResolveDeclarations()
 48 {
 49     for (const std::unique_ptr<SolutionDeclaration>& declaration : declarations)
 50     {
 51         if (SolutionProjectDeclaration* solutionProjectDeclaration=  dynamic_cast<SolutionProjectDeclaration*>(declaration.get());)
 52         {
 53             boost::filesystem::path pp(solutionProjectDeclaration->FilePath());
 54             relativeProjectFilePaths.push_back(pp.generic_string());
 55             if (pp.is_relative())
 56             {
 57                 pp = basePath / pp;
 58             }
 59             if (pp.extension() != ".cmp")
 60             {
 61                 throw std::runtime_error("invalid project file extension '" + pp.generic_string() + "' (not .cmp)");
 62             }
 63             if (!boost::filesystem::exists(pp))
 64             {
 65                 throw std::runtime_error("project file '" + GetFullPath(pp.generic_string()) + "' not found");
 66             }
 67             std::string projectFilePath = GetFullPath(pp.generic_string());
 68             if (std::find(projectFilePaths.cbegin(), projectFilePaths.cend(), projectFilePath) == projectFilePaths.cend())
 69             {
 70                 projectFilePaths.push_back(projectFilePath);
 71             }
 72         }
 73         else if (SolutionActiveProjectDeclaration* activeProjectDeclaration=  dynamic_cast<SolutionActiveProjectDeclaration*>(declaration.get());)
 74         {
 75             activeProjectName = activeProjectDeclaration->ActiveProjectName();
 76         }
 77         else if (ProjectDependencyDeclaration* projectDependencyDeclaration=  dynamic_cast<ProjectDependencyDeclaration*>(declaration.get());)
 78         {
 79             dependencyMap[projectDependencyDeclaration->ProjectName()] = projectDependencyDeclaration;
 80         }
 81         else
 82         {
 83             throw std::runtime_error("unknown solution declaration");
 84         }
 85     }
 86 }

 93 void Visit(std::vector<std::u32string>& order, const std::u32string& projectName, std::unordered_set<std::u32string>& visited, std::unordered_set<std::u32string>& tempVisit, 
 94     const std::unordered_map<std::u32string, ProjectDependencyDeclaration*>& dependencyMap, Solution* solution)
 95 {
 96     if (tempVisit.find(projectName) == tempVisit.end())
 97     {
 98         if (visited.find(projectName) == visited.end())
 99         {
100             tempVisit.insert(projectName);
101             auto i = dependencyMap.find(projectName);
102             if (i != dependencyMap.end())
103             {
104                 ProjectDependencyDeclaration* dependencyDeclaration = i->second;
105                 for (const std::u32string& dependentProject : dependencyDeclaration->DependsOnProjects())
106                 {
107                     Visit(order, dependentProject, visited, tempVisit, dependencyMap, solution);
108                 }
109                 tempVisit.erase(projectName);
110                 visited.insert(projectName);
111                 order.push_back(projectName);
112             }
113             else
114             {
115                 throw std::runtime_error("project '" + ToUtf8(projectName) + "' not found in dependencies of solution '" + ToUtf8(solution->Name()) + "' (" + 
116                     GetFullPath(solution->FilePath()) + ")");
117             }
118         }
119     }
120     else
121     {
122         throw std::runtime_error("circular project dependency '" + ToUtf8(projectName) + "' detected in dependencies of solution '" + ToUtf8(solution->Name()) + "' (" + 
123             GetFullPath(solution->FilePath()) + ")");
124     }
125 }

127 void Solution::AddDependencies()
128 {
129     for (const std::unique_ptr<Project>& project : projects)
130     {
131         ProjectDependencyDeclaration* dependencyDeclaration = nullptr;
132         auto it = dependencyMap.find(project->Name());
133         if (it != dependencyMap.cend())
134         {
135             dependencyDeclaration = it->second;
136         }
137         else
138         {
139             ProjectDependencyDeclaration* additionalDeclaration = new ProjectDependencyDeclaration(project->Name());
140             additionalDependencyDeclarations.push_back(     std::unique_ptr<ProjectDependencyDeclaration>(additionalDeclaration));
141             dependencyDeclaration = additionalDeclaration;
142             dependencyMap[project->Name()] = dependencyDeclaration;
143         }
144         for (const std::unique_ptr<Project>& projectToCheck : projects)
145         {
146             if (projectToCheck != project)
147             {
148                 if (project->DependsOn(projectToCheck.get()))
149                 {
150                     project->AddDependsOnProjects(projectToCheck.get());
151                     if (std::find(dependencyDeclaration->DependsOnProjects().cbegin(), dependencyDeclaration->DependsOnProjects().cend(), projectToCheck->Name()) == dependencyDeclaration->DependsOnProjects().cend())
152                     {
153                         dependencyDeclaration->AddDependency(projectToCheck->Name());
154                     }
155                 }
156             }
157         }
158     }
159 }

161 std::vector<Project*> Solution::CreateBuildOrder()
162 {
163     AddDependencies();
164     std::vector<Project*> buildOrder;
165     std::unordered_map<std::u32string, Project*> projectMap;
166     for (const std::unique_ptr<Project>& project : projects)
167     {
168         projectMap[project->Name()] = project.get();
169     }
170     std::vector<std::u32string> order;
171     std::unordered_set<std::u32string> visited;
172     std::unordered_set<std::u32string> tempVisit;
173     for (const std::unique_ptr<Project>& project : projects)
174     {
175         if (visited.find(project->Name()) == visited.end())
176         {
177             Visit(order, project->Name(), visited, tempVisit, dependencyMap, this);
178         }
179     }
180     for (const std::u32string& projectName : order)
181     {
182         auto i = projectMap.find(projectName);
183         if (i != projectMap.end())
184         {
185             buildOrder.push_back(i->second);
186         }
187         else
188         {
189             throw std::runtime_error("project name '" + ToUtf8(projectName) + "' not found in solution '" + ToUtf8(Name()) + "' (" + GetFullPath(FilePath()) + ")");
190         }
191     }
192     return buildOrder;
193 }
