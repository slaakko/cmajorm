13 
14 TypedefNode::TypedefNode(const Span& span_) : Node(NodeType::typedefNode, span_), specifiers(Specifiers::none), typeExpr(), id()
15 {
16 }
17 
18 TypedefNode::TypedefNode(const Span& span_, Specifiers specifiers_, Node* typeExpr_, IdentifierNode* id_) : Node(NodeType::typedefNode, span_), specifiers(specifiers_), typeExpr(typeExpr_), id(id_)
19 {
20     typeExpr->SetParent(this);
21     id->SetParent(this);
22 }
23 
24 Node* TypedefNode::Clone(CloneContext& cloneContext) const
25 {
26     return new TypedefNode(GetSpan(), specifiers, typeExpr->Clone(cloneContext), static_cast<IdentifierNode*>(id->Clone(cloneContext)));
27 }
28 
29 void TypedefNode::Accept(Visitor& visitor)
30 {
31     visitor.Visit(*this);
32 }
33 
34 void TypedefNode::Write(AstWriter& writer)
35 {
36     Node::Write(writer);
37     writer.Write(specifiers);
38     writer.Write(typeExpr.get());
39     writer.Write(id.get());
40 }
41 
42 void TypedefNode::Read(AstReader& reader)
43 {
44     Node::Read(reader);
45     specifiers = reader.ReadSpecifiers();
46     typeExpr.reset(reader.ReadNode());
47     typeExpr->SetParent(this);
48     id.reset(reader.ReadIdentifierNode());
49     id->SetParent(this);
50 }
