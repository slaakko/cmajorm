  72 std::vector<std::unique_ptr<CompileUnitNode>> ParseSourcesInMainThread(Module* module, const std::vector<std::string>& sourceFilePaths, bool& stop)
  73 {
  74     if (!compileUnitGrammar)
  75     {
  76         compileUnitGrammar = CompileUnit::Create();
  77     }
  78     if (GetGlobalFlag(GlobalFlags::verbose))
  79     {
  80         std::string s;
  81         if (sourceFilePaths.size() != 1)
  82         {
  83             s = "s";
  84         }
  85         LogMessage(module->LogStreamId(), "Parsing " + std::to_string(sourceFilePaths.size()) + " source file" + s + " in main thread...");
  86     }
  87     std::vector<std::unique_ptr<CompileUnitNode>> compileUnits;
  88     for (const std::string& sourceFilePath : sourceFilePaths)
  89     {
  90         if (stop)
  91         {
  92             return      std::vector<std::unique_ptr<CompileUnitNode>>();
  93         }
  94         if (boost::filesystem::file_size(sourceFilePath) == 0)
  95         {
  96             std::unique_ptr<CompileUnitNode> compileUnit(new CompileUnitNode(Span(), sourceFilePath));
  97             compileUnits.push_back(std::move(compileUnit));
  98         }
  99         else
 100         {
 101             MappedInputFile sourceFile(sourceFilePath);
 102             int32_t fileIndex = module->GetFileTable().RegisterFilePath(sourceFilePath);
 103             ParsingContext parsingContext;
 104             if (GetGlobalFlag(GlobalFlags::debugParsing))
 105             {
 106                 compileUnitGrammar->SetLog(&std::cout);
 107             }
 108             std::u32string s(ToUtf32(std::string(sourceFile.Begin(), sourceFile.End())));
 109             std::unique_ptr<CompileUnitNode> compileUnit(compileUnitGrammar->Parse(&s[0], &s[0] + s.length(), fileIndex, sourceFilePath, &parsingContext));
 110             if (GetGlobalFlag(GlobalFlags::ast2xml))
 111             {
 112                 std::unique_ptr<dom::Document> ast2xmlDoc = cmajor::ast2dom::GenerateAstDocument(compileUnit.get());
 113                 std::string ast2xmlFilePath = Path::ChangeExtension(sourceFilePath, ".ast.xml");
 114                 std::ofstream ast2xmlFile(ast2xmlFilePath);
 115                 CodeFormatter formatter(ast2xmlFile);
 116                 formatter.SetIndentSize(1);
 117                 ast2xmlDoc->Write(formatter);
 118             }
 119             if (GetGlobalFlag(GlobalFlags::generateDebugInfo) || GetGlobalFlag(GlobalFlags::cmdoc))
 120             {
 121                 compileUnit->ComputeLineStarts(s);
 122             }
 123             compileUnits.push_back(std::move(compileUnit));
 124         }
 125     }
 126     if (GetGlobalFlag(GlobalFlags::verbose))
 127     {
 128         std::string s;
 129         if (sourceFilePaths.size() != 1)
 130         {
 131             s = "s";
 132         }
 133         LogMessage(module->LogStreamId(), "Source file" + s + " parsed.");
 134     }
 135     return compileUnits;
 136 }

 139 {
 140     ParserData(const std::vector<std::string>& sourceFilePaths_, std::vector<std::unique_ptr<CompileUnitNode>>& compileUnits_, const std::vector<uint32_t>& fileIndeces_, 
 141         std::vector<std::exception_ptr>& exceptions_, bool& stop_) : sourceFilePaths(sourceFilePaths_), compileUnits(compileUnits_), fileIndeces(fileIndeces_), stop(stop_), 
 142         exceptions(exceptions_)
 143     {
 144     }
 145     const std::vector<std::string>& sourceFilePaths;
 146     std::vector<std::unique_ptr<CompileUnitNode>>& compileUnits;
 147     const std::vector<uint32_t>& fileIndeces;
 148     std::list<int> indexQueue;

 154 void ParseSourceFile(ParserData* parserData)
 155 {
 156     int index = -1;
 157     try
 158     {
 159         while (!parserData->stop)
 160         {
 161             {
 162                 std::lock_guard<std::mutex> lock(parserData->indexQueueMutex);
 163                 if (parserData->indexQueue.empty()) return;
 164                 index = parserData->indexQueue.front();
 165                 parserData->indexQueue.pop_front();
 166             }
 167             const std::string& sourceFilePath = parserData->sourceFilePaths[index];
 168             if (boost::filesystem::file_size(sourceFilePath) == 0)
 169             {
 170                 std::unique_ptr<CompileUnitNode> compileUnit(new CompileUnitNode(Span(), sourceFilePath));
 171                 parserData->compileUnits[index].reset(compileUnit.release());
 172             }
 173             else
 174             {
 175                 MappedInputFile sourceFile(sourceFilePath);
 176                 ParsingContext parsingContext;
 177                 int fileIndex = parserData->fileIndeces[index];
 178                 std::u32string s(ToUtf32(std::string(sourceFile.Begin(), sourceFile.End())));
 179                 std::unique_ptr<CompileUnitNode> compileUnit(compileUnitGrammar->Parse(&s[0], &s[0] + s.length(), fileIndex, sourceFilePath, &parsingContext));
 180                 if (GetGlobalFlag(GlobalFlags::ast2xml))
 181                 {
 182                     std::unique_ptr<dom::Document> ast2xmlDoc = cmajor::ast2dom::GenerateAstDocument(compileUnit.get());
 183                     std::string ast2xmlFilePath = Path::ChangeExtension(sourceFilePath, ".ast.xml");
 184                     std::ofstream ast2xmlFile(ast2xmlFilePath);
 185                     CodeFormatter formatter(ast2xmlFile);
 186                     formatter.SetIndentSize(1);
 187                     ast2xmlDoc->Write(formatter);
 188                 }
 189                 if (GetGlobalFlag(GlobalFlags::generateDebugInfo) || GetGlobalFlag(GlobalFlags::cmdoc))
 190                 {
 191                     compileUnit->ComputeLineStarts(s);
 192                 }
 193                 parserData->compileUnits[index].reset(compileUnit.release());
 194             }
 195         }
 196     }
 197     catch (...)
 198     {
 199         if (index != -1)
 200         {
 201             parserData->exceptions[index] = std::current_exception();
 202             parserData->stop = true;
 203         }
 204     }
 205 }

 207 std::vector<std::unique_ptr<CompileUnitNode>> ParseSourcesConcurrently(Module* module, const std::vector<std::string>& sourceFilePaths, int numThreads, bool& stop)
 208 {
 209     if (!compileUnitGrammar)
 210     {
 211         compileUnitGrammar = CompileUnit::Create();
 212     }
 213     if (GetGlobalFlag(GlobalFlags::verbose))
 214     {
 215         std::string s;
 216         if (sourceFilePaths.size() != 1)
 217         {
 218             s = "s";
 219         }
 220         LogMessage(module->LogStreamId(), "Parsing " + std::to_string(sourceFilePaths.size()) + " source file" + s + " using " + std::to_string(numThreads) + " threads...");
 221     }
 222     std::vector<std::unique_ptr<CompileUnitNode>> compileUnits;
 223     int n = int(sourceFilePaths.size());
 224     compileUnits.resize(n);
 225     std::vector<uint32_t> fileIndeces;
 226     for (int i = 0; i < n; ++i)
 227     {
 228         const std::string& sourceFilePath = sourceFilePaths[i];
 229         int32_t fileIndex = module->GetFileTable().RegisterFilePath(sourceFilePath);
 230         fileIndeces.push_back(fileIndex);
 231     }
 232     std::vector<std::exception_ptr> exceptions;
 233     exceptions.resize(n);
 234     ParserData parserData(sourceFilePaths, compileUnits, fileIndeces, exceptions, stop);
 235     for (int i = 0; i < n; ++i)
 236     {
 237         parserData.indexQueue.push_back(i);
 238     }
 239     std::vector<std::thread> threads;
 240     for (int i = 0; i < numThreads; ++i)
 241     {
 242         threads.push_back(std::thread(ParseSourceFile, &parserData));
 243         if (parserData.stop) break;
 244     }
 245     int numStartedThreads = int(threads.size());
 246     for (int i = 0; i < numStartedThreads; ++i)
 247     {
 248         if (threads[i].joinable())
 249         {
 250             threads[i].join();
 251         }
 252     }
 253     for (int i = 0; i < n; ++i)
 254     {
 255         if (exceptions[i])
 256         {
 257             std::rethrow_exception(exceptions[i]);
 258         }
 259     }
 260     if (GetGlobalFlag(GlobalFlags::verbose))
 261     {
 262         std::string s;
 263         if (sourceFilePaths.size() != 1)
 264         {
 265             s = "s";
 266         }
 267         LogMessage(module->LogStreamId(), "Source file" + s + " parsed.");
 268     }
 269     return compileUnits;
 270 }

 272 std::vector<std::unique_ptr<CompileUnitNode>> ParseSources(Module* module, const std::vector<std::string>& sourceFilePaths, bool& stop)
 273 {
 274     try
 275     {
 276         int numCores = std::thread::hardware_concurrency();
 277         if (numCores == 0 || sourceFilePaths.size() < numCores || GetGlobalFlag(GlobalFlags::debugParsing))
 278         {
 279             return ParseSourcesInMainThread(module, sourceFilePaths, stop);
 280         }
 281         else
 282         {
 283             return ParseSourcesConcurrently(module, sourceFilePaths, numCores, stop);
 284         }
 285     }
 286     catch (ParsingException& ex;)
 287     {
 288         ex.SetProject(ToUtf8(module->Name()));
 289         throw ex;
 290     }
 291     catch (const AttributeNotUniqueException& ex;)
 292     {
 293         throw Exception(module, ex.what(), ex.GetSpan(), ex.PrevSpan());
 294     }
 295 }

 311 std::vector<std::unique_ptr<BoundCompileUnit>> BindTypes(Module& module, const std::vector<std::unique_ptr<CompileUnitNode>>& compileUnits, AttributeBinder* attributeBinder, 
 312     bool& stop)
 313 {
 314     std::vector<std::unique_ptr<BoundCompileUnit>> boundCompileUnits;
 315     for (const std::unique_ptr<CompileUnitNode>& compileUnit : compileUnits)
 316     {
 317         if (stop)
 318         {
 319             return      std::vector<std::unique_ptr<BoundCompileUnit>>();
 320         }
 321         std::unique_ptr<BoundCompileUnit> boundCompileUnit(new BoundCompileUnit(module, compileUnit.get(), attributeBinder));
 322         boundCompileUnit->PushBindingTypes();
 323         TypeBinder typeBinder(*boundCompileUnit);
 324         compileUnit->Accept(typeBinder);
 325         boundCompileUnit->PopBindingTypes();
 326         boundCompileUnits.push_back(std::move(boundCompileUnit));
 327     }
 328     return boundCompileUnits;
 329 }

 442 void Link(const std::string& executableFilePath, const std::string& libraryFilePath, const std::vector<std::string>& libraryFilePaths, Module& module)
 443 {
 444     if (GetGlobalFlag(GlobalFlags::verbose) && !GetGlobalFlag(GlobalFlags::unitTest))
 445     {
 446         LogMessage(module.LogStreamId(), "Linking...");
 447     }
 448     if (GetGlobalFlag(GlobalFlags::linkUsingMsLink))
 449     {
 450         module.SetCurrentToolName(U"link");
 451     }
 452     else
 453     {
 454         module.SetCurrentToolName(U"lld-link");
 455     }
 456     boost::filesystem::path bdp = executableFilePath;
 457     bdp.remove_filename();
 458     boost::filesystem::create_directories(bdp);
 459     std::vector<std::string> args;
 460     args.push_back("/machine:x64");
 461     args.push_back("/entry:main");
 462     args.push_back("/debug");
 463     args.push_back("/out:" + QuotedPath(executableFilePath));
 464     args.push_back("/stack:16777216");
 465     std::string defFilePath = GetFullPath(boost::filesystem::path(libraryFilePath).replace_extension(".def").generic_string());
 466     CreateDefFile(defFilePath, module);
 467     args.push_back("/def:" + QuotedPath(defFilePath));
 468     std::string cmrtLibName = "cmrt230.lib";
 469     if (GetGlobalFlag(GlobalFlags::linkWithDebugRuntime))
 470     {
 471         cmrtLibName = "cmrt230d.lib";
 472     }
 473     args.push_back(QuotedPath(Path::Combine(Path::Combine(CmajorRootDir(), "lib"), cmrtLibName)));
 474     int n = libraryFilePaths.size();
 475     for (int i = 0; i < n; ++i)
 476     {
 477         args.push_back(QuotedPath(libraryFilePaths[i]));
 478     }
 479     std::string linkCommandLine;
 480     std::string linkErrorFilePath;
 481     if (GetGlobalFlag(GlobalFlags::linkUsingMsLink))
 482     {
 483         linkErrorFilePath = Path::Combine(Path::GetDirectoryName(executableFilePath), "link.error");
 484         linkCommandLine = "cmfileredirector -2 " + linkErrorFilePath + " link";
 485     }
 486     else
 487     {
 488         linkErrorFilePath = Path::Combine(Path::GetDirectoryName(executableFilePath), "lld-link.error");
 489         linkCommandLine = "cmfileredirector -2 " + linkErrorFilePath + " lld-link";

 539 void Link(const std::string& executableFilePath, const std::string& libraryFilePath, const std::vector<std::string>& libraryFilePaths, Module& module)
 540 {
 541     if (GetGlobalFlag(GlobalFlags::verbose) && !GetGlobalFlag(GlobalFlags::unitTest))
 542     {
 543         LogMessage(module.LogStreamId(), "Linking...");
 544     }
 545     module.SetCurrentToolName(U"clang++");
 546     boost::filesystem::path bdp = executableFilePath;
 547     bdp.remove_filename();
 548     boost::filesystem::create_directories(bdp);
 549     std::vector<std::string> args;
 550     args.push_back("-L" + Path::Combine(CmajorRootDir(), "lib"));
 551     char* cmajorLibDir = getenv("CMAJOR_LIBDIR");

 557     CreateDynamicListFile(dynamicListFilePath, module);
 558     args.push_back("-Wl,--dynamic-list=" + dynamicListFilePath);
 559     args.push_back("-Xlinker --start-group");
 560     int n = libraryFilePaths.size();
 561     args.push_back(QuotedPath(libraryFilePaths.back()));
 562     for (int i = 0; i < n - 1; ++i)
 563     {

 566     if (GetGlobalFlag(GlobalFlags::linkWithDebugRuntime))
 567     {

 578     args.push_back("-lboost_filesystem -lboost_iostreams -lboost_system -lgmp -lbz2 -lz");
 579     args.push_back("-Xlinker --end-group");
 580     args.push_back("-o " + QuotedPath(executableFilePath));
 581     std::string linkCommandLine;
 582     std::string linkErrorFilePath;
 583     linkErrorFilePath = Path::Combine(Path::GetDirectoryName(executableFilePath), "clang++.error");
 584     linkCommandLine = "cmfileredirector -2 " + linkErrorFilePath + " clang++";
 585     for (const std::string& arg : args)
 586     {
 587         linkCommandLine.append(1, ' ').append(arg);
 588     }
 589     try
 590     {
 591         System(linkCommandLine);
 592         boost::filesystem::remove(boost::filesystem::path(linkErrorFilePath));
 593     }
 594     catch (const std::exception& ex;)
 595     {
 596         std::string errors = ReadFile(linkErrorFilePath);
 597         throw std::runtime_error("linking executable '" + executableFilePath + "' failed: " + ex.what() + ":\nerrors:\n" + errors);
 598     }
 599     if (GetGlobalFlag(GlobalFlags::verbose) && !GetGlobalFlag(GlobalFlags::unitTest))
 600     {
 601         LogMessage(module.LogStreamId(), "==> " + executableFilePath);

 629 void CheckMainFunctionSymbol(Module& module)
 630 {
 631     FunctionSymbol* userMain = module.GetSymbolTable().MainFunctionSymbol();
 632     if (!userMain)
 633     {
 634         throw Exception(&module, "program has no main function", Span());
 635     }
 636     if (!userMain->Parameters().empty())
 637     {
 638         if (userMain->Parameters().size() != 2)
 639         {
 640             throw Exception(&module, "main function must either take no parameters or take two parameters", userMain->GetSpan());
 641         }
 642         if (!TypesEqual(userMain->Parameters()[0]->GetType(), module.GetSymbolTable().GetTypeByName(U"int")))
 643         {
 644             throw Exception(&module, "first parameter of main function must be of int type", userMain->GetSpan());
 645         }
 646         if (!TypesEqual(userMain->Parameters()[1]->GetType(), module.GetSymbolTable().GetTypeByName(U"char")->AddConst(userMain->GetSpan())->AddPointer(userMain->GetSpan())->AddPointer(userMain->GetSpan())))
 647         {
 648             throw Exception(&module, "second parameter of main function must be of 'const char**' type", userMain->GetSpan());
 649         }
 650     }
 651     if (userMain->ReturnType() && !userMain->ReturnType()->IsVoidType())
 652     {
 653         if (!TypesEqual(userMain->ReturnType(), module.GetSymbolTable().GetTypeByName(U"int")))
 654         {
 655             throw Exception(&module, "main function must either be void function or return an int", userMain->GetSpan());
 656         }
 657     }
 658 }

 660 void CreateJsonRegistrationUnit(std::vector<std::string>& objectFilePaths, Module& module, EmittingContext& emittingContext, AttributeBinder* attributeBinder)

 690     {

 697 void CreateMainUnit(std::vector<std::string>& objectFilePaths, Module& module, EmittingContext& emittingContext, AttributeBinder* attributeBinder)
 698 {
 699     CompileUnitNode mainCompileUnit(Span(), boost::filesystem::path(module.OriginalFilePath()).parent_path().append("__main__.cm").generic_string());
 700     mainCompileUnit.SetSynthesizedUnit();
 701     mainCompileUnit.GlobalNs()->AddMember(new NamespaceImportNode(Span(), new IdentifierNode(Span(), U"System")));
 702     FunctionNode * mainFunction(new FunctionNode(Span(), Specifiers::public_, new IntNode(Span()), U"main", nullptr));
 703 #ifndef _WIN32
 704     mainFunction->AddParameter(new ParameterNode(Span(), new IntNode(Span()), new IdentifierNode(Span(), U"argc")));
 705     mainFunction->AddParameter(new ParameterNode(Span(), new PointerNode(Span(), new PointerNode(Span(), new CharNode(Span()))), new IdentifierNode(Span(), U"argv")));
 706 #endif
 707     mainFunction->SetProgramMain();
 708     CompoundStatementNode* mainFunctionBody = new CompoundStatementNode(Span());
 709     ConstructionStatementNode* constructExitCode = new ConstructionStatementNode(Span(), new IntNode(Span()), new IdentifierNode(Span(), U"exitCode"));
 710     mainFunctionBody->AddStatement(constructExitCode);
 711     ExpressionStatementNode* rtInitCall = nullptr;
 712     if (GetGlobalFlag(GlobalFlags::profile))
 713     {
 714         rtInitCall = new ExpressionStatementNode(Span(), new InvokeNode(Span(), new IdentifierNode(Span(), U"RtStartProfiling")));
 715     }
 716     else
 717     {
 718         rtInitCall = new ExpressionStatementNode(Span(), new InvokeNode(Span(), new IdentifierNode(Span(), U"RtInit")));
 719     }
 720     mainFunctionBody->AddStatement(rtInitCall);
 721 #ifdef _WIN32
 722     ConstructionStatementNode* argc = new ConstructionStatementNode(Span(), new IntNode(Span()), new IdentifierNode(Span(), U"argc"));
 723     argc->AddArgument(new InvokeNode(Span(), new IdentifierNode(Span(), U"RtArgc")));
 724     mainFunctionBody->AddStatement(argc);
 725     ConstructionStatementNode* argv = new ConstructionStatementNode(Span(), new ConstNode(Span(), new PointerNode(Span(), new PointerNode(Span(), new CharNode(Span())))), new IdentifierNode(Span(), U"argv"));
 726     argv->AddArgument(new InvokeNode(Span(), new IdentifierNode(Span(), U"RtArgv")));
 727     mainFunctionBody->AddStatement(argv);
 728 #endif
 729     CompoundStatementNode* tryBlock = new CompoundStatementNode(Span());
 730     if (!module.GetSymbolTable().JsonClasses().empty())
 731     {
 732         ExpressionStatementNode* registerJsonClassesCall = new ExpressionStatementNode(Span(), new InvokeNode(Span(), new IdentifierNode(Span(), U"RegisterJsonClasses")));
 733         tryBlock->AddStatement(registerJsonClassesCall);
 734     }
 735     FunctionSymbol* userMain = module.GetSymbolTable().MainFunctionSymbol();
 736     InvokeNode* invokeMain = new InvokeNode(Span(), new IdentifierNode(Span(), userMain->GroupName()));
 737     if (!userMain->Parameters().empty())
 738     {
 739         invokeMain->AddArgument(new IdentifierNode(Span(), U"argc"));
 740         invokeMain->AddArgument(new IdentifierNode(Span(), U"argv"));
 741     }
 742     StatementNode* callMainStatement = nullptr;
 743     if (!userMain->ReturnType() || userMain->ReturnType()->IsVoidType())
 744     {
 745         callMainStatement = new ExpressionStatementNode(Span(), invokeMain);
 746     }
 747     else
 748     {
 749         callMainStatement = new AssignmentStatementNode(Span(), new IdentifierNode(Span(), U"exitCode"), invokeMain);
 750     }
 751     tryBlock->AddStatement(callMainStatement);
 752     TryStatementNode* tryStatement = new TryStatementNode(Span(), tryBlock);
 753     CompoundStatementNode* catchBlock = new CompoundStatementNode(Span());
 754     InvokeNode* consoleError = new InvokeNode(Span(), new DotNode(Span(), new IdentifierNode(Span(), U"System.Console"), new IdentifierNode(Span(), U"Error")));
 755     DotNode* writeLine = new DotNode(Span(), consoleError, new IdentifierNode(Span(), U"WriteLine"));
 756     InvokeNode* printEx = new InvokeNode(Span(), writeLine);
 757     InvokeNode* exToString = new InvokeNode(Span(), new DotNode(Span(), new IdentifierNode(Span(), U"ex"), new IdentifierNode(Span(), U"ToString")));
 758     printEx->AddArgument(exToString);
 759     ExpressionStatementNode* printExStatement = new ExpressionStatementNode(Span(), printEx);
 760     catchBlock->AddStatement(printExStatement);
 761     AssignmentStatementNode* assignExitCodeStatement = new AssignmentStatementNode(Span(), new IdentifierNode(Span(), U"exitCode"), new IntLiteralNode(Span(), 1));
 762     catchBlock->AddStatement(assignExitCodeStatement);
 763     CatchNode* catchAll = new CatchNode(Span(), new ConstNode(Span(), new LValueRefNode(Span(), new IdentifierNode(Span(), U"System.Exception"))), new IdentifierNode(Span(), U"ex"), catchBlock);
 764     tryStatement->AddCatch(catchAll);
 765     mainFunctionBody->AddStatement(tryStatement);
 766     ExpressionStatementNode* rtDoneCall = nullptr;
 767     if (GetGlobalFlag(GlobalFlags::profile))
 768     {
 769         rtDoneCall = new ExpressionStatementNode(Span(), new InvokeNode(Span(), new IdentifierNode(Span(), U"RtEndProfiling")));
 770     }
 771     else
 772     {
 773         rtDoneCall = new ExpressionStatementNode(Span(), new InvokeNode(Span(), new IdentifierNode(Span(), U"RtDone")));
 774     }
 775     mainFunctionBody->AddStatement(rtDoneCall);
 776     InvokeNode* exitCall = new InvokeNode(Span(), new IdentifierNode(Span(), U"RtExit"));
 777     exitCall->AddArgument(new IdentifierNode(Span(), U"exitCode"));
 778     ExpressionStatementNode* rtExitCall = new ExpressionStatementNode(Span(), exitCall);
 779     mainFunctionBody->AddStatement(rtExitCall);
 780     ReturnStatementNode* returnStatement = new ReturnStatementNode(Span(), new IdentifierNode(Span(), U"exitCode"));
 781     mainFunctionBody->AddStatement(returnStatement);
 782     mainFunction->SetBody(mainFunctionBody);
 783     mainCompileUnit.GlobalNs()->AddMember(mainFunction);
 784     SymbolCreatorVisitor symbolCreator(module.GetSymbolTable());
 785     mainCompileUnit.Accept(symbolCreator);
 786     BoundCompileUnit boundMainCompileUnit(module, &mainCompileUnit, attributeBinder);
 787     boundMainCompileUnit.PushBindingTypes();
 788     TypeBinder typeBinder(boundMainCompileUnit);
 789     mainCompileUnit.Accept(typeBinder);
 790     boundMainCompileUnit.PopBindingTypes();
 791     StatementBinder statementBinder(boundMainCompileUnit);
 792     mainCompileUnit.Accept(statementBinder);
 793     if (boundMainCompileUnit.HasGotos())
 794     {
 795         AnalyzeControlFlow(boundMainCompileUnit);
 796     }
 797     GenerateCode(emittingContext, boundMainCompileUnit);
 798     objectFilePaths.push_back(boundMainCompileUnit.ObjectFilePath());
 799 }

 916         }
