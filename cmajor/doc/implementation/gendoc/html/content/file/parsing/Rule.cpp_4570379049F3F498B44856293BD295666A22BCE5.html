132 class CodeExpandingVisitor : public cmajor::codedom::Visitor
133 {
134 public:
135     CodeExpandingVisitor(std::unordered_map<std::u32string, std::u32string>& expandMap_): expandMap(expandMap_) {}
136     virtual void Visit(IdExpr& expr)
137     {
138         std::unordered_map<std::u32string, std::u32string>                 i = expandMap.find(expr.Name());
139         if (i != expandMap.end())
140         {
141             expr.SetName(i->second);
142         }
143     }
144 private:
145     std::unordered_map<std::u32string, std::u32string>& expandMap;
146 };

148 void Rule::ExpandCode()
149 {
150     std::unordered_map<std::u32string, std::u32string> expandMap;
151     expandMap[U"value"] = U"context->value";
152     int n = int(nonterminals.size());
153     for (int i = 0; i < n; ++i)
154     {
155         NonterminalParser* nonterminal = nonterminals[i];
156         expandMap[nonterminal->Name()] = U"context->" + nonterminal->ValueFieldName();
157     }
158     n = int(inheritedAttributes.size());
159     for (int i = 0; i < n; ++i)
160     {
161         const AttrOrVariable& attr = inheritedAttributes[i];
162         expandMap[attr.Name()] = U"context->" + attr.Name();
163     }
164     n = int(localVariables.size());
165     for (int i = 0; i < n; ++i)
166     {
167         const AttrOrVariable& var = localVariables[i];
168         expandMap[var.Name()] = U"context->" + var.Name();
169     }
170     CodeExpandingVisitor visitor(expandMap);
171     n = int(actions.size());
172     for (int i = 0; i < n; ++i)
173     {
174         ActionParser* action = actions[i];
175         action->SuccessCode()->Accept(visitor);
176         if (action->FailCode())
177         {
178             action->FailCode()->Accept(visitor);
179         }
180     }
181     n = int(nonterminals.size());
182     for (int i = 0; i < n; ++i)
183     {
184         NonterminalParser* nonterminal = nonterminals[i];
185         if (!nonterminal->Arguments().empty())
186         {
187             int m = int(nonterminal->Arguments().size());
188             for (int j = 0; j < m; ++j)
189             {
190                 cmajor::codedom::CppObject* argument = nonterminal->Arguments()[j];
191                 argument->Accept(visitor);
192             }
193         }
194     }
195 }

197 Match Rule::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
198 {
199     bool writeToLog = !scanner.Skipping() && scanner.Log();
200     if (writeToLog)
201     {
202         scanner.Log()->WriteBeginRule(ToUtf8(Name()));
203         scanner.Log()->IncIndent();
204         scanner.Log()->WriteTry(scanner.RestOfLine(scanner.Log()->MaxLineLength()));
205         scanner.Log()->IncIndent();
206     }
207     int startIndex = scanner.GetSpan().Start();
208     Enter(stack, parsingData);
209     Match match = definition ? definition->Parse(scanner, stack, parsingData) : Match::Nothing();
210     Leave(stack, parsingData, match.Hit());
211     if (writeToLog)
212     {
213         scanner.Log()->DecIndent();
214         if (match.Hit())
215         {
216             std::string matched(scanner.Start() + startIndex, scanner.Start() + scanner.GetSpan().Start());
217             scanner.Log()->WriteSuccess(matched);
218         }
219         else
220         {
221             scanner.Log()->WriteFail();
222         }
223         scanner.Log()->DecIndent();
224         scanner.Log()->WriteEndRule(ToUtf8(Name()));
225     }
226     return match;
227 }

239 std::u32string GetPrefix(const std::u32string& fullName)
240 {
241     std::u32string::size_type dotPos = fullName.find('.');
242     if (dotPos != std::u32string::npos)
243     {
244         return fullName.substr(0, dotPos);
245     }
246     return fullName;
247 }

249 std::u32string GetCommonName(const std::u32string& qualifiedId)
250 {
251     std::u32string::size_type lastDotPos = qualifiedId.rfind('.');
252     if (lastDotPos != std::u32string::npos)
253     {
254         return qualifiedId.substr(lastDotPos + 1);
255     }
256     return qualifiedId;
257 }
