 37 void Grammar::AddRule(Rule* rule)
 38 {
 39     Own(rule);
 40     rule->SetGrammar(this);
 41     rule->GetScope()->SetEnclosingScope(GetScope());
 42     rule->SetEnclosingScope(GetScope());
 43     rules.push_back(rule);
 44     try
 45     {
 46         GetScope()->Add(rule);
 47     }
 48     catch (std::exception& ex;)
 49     {
 50         ThrowException(ex.what(), GetSpan());
 51     }
 52 }

 54 Rule* Grammar::GetRule(const std::u32string& ruleName) const
 55 {
 56     ParsingObject* object = GetScope()->Get(ruleName, ObjectKind::rule | ObjectKind::ruleLink);
 57     if (!object)
 58     {
 59         ThrowException("rule '" + ToUtf8(ruleName) + "' not found in grammar '" + ToUtf8(FullName()) + "'", GetSpan());
 60     }
 61     if (object->IsRule())
 62     {
 63         return static_cast<Rule*>(object);
 64     }
 65     else if (object->IsRuleLink())
 66     {
 67         RuleLink* ruleLink = static_cast<RuleLink*>(object);
 68         Rule* rule = ruleLink->GetRule();
 69         if (rule)
 70         {
 71             return rule;
 72         }
 73         else
 74         {
 75             ThrowException("rule link '" + ToUtf8(ruleName) + "' not bound in grammar '" + ToUtf8(FullName()) + "'", GetSpan());
 76         }
 77     }
 78     else
 79     {
 80         ThrowException("'" + ToUtf8(ruleName) + "' is not a rule or rule link", GetSpan());
 81     }
 82     return nullptr;
 83 }

 85 void Grammar::Accept(Visitor& visitor)
 86 {
 87     visitor.BeginVisit(*this);
 88     RuleLinkSetIt e = ruleLinks.cend();
 89     for (RuleLinkSetIt i = ruleLinks.cbegin(); i != e; ++i)
 90     {
 91         RuleLink* ruleLink = *i;
 92         ruleLink->Accept(visitor);
 93     }
 94     int n = int(rules.size());
 95     for (int i = 0; i < n; ++i)
 96     {
 97         Rule* rule = rules[i];
 98         rule->Accept(visitor);
 99     }
100     visitor.EndVisit(*this);
101 }

103 void Grammar::Parse(const char32_t* start, const char32_t* end, int fileIndex, const std::string& fileName, ParsingData* parsingData)
104 {
105     Scanner scanner(start, end, fileName, fileIndex, skipRule);
106     scanner.SetParsingData(parsingData);
107     std::unique_ptr<XmlLog> xmlLog;
108     if (log)
109     {
110         xmlLog.reset(new XmlLog(*log, maxLogLineLength));
111         scanner.SetLog(xmlLog.get());
112         xmlLog->WriteBeginRule("parse");
113         xmlLog->IncIndent();
114     }
115     ObjectStack stack;
116     Match match = Parse(scanner, stack, parsingData);
117     Span stop = scanner.GetSpan();
118     if (log)
119     {
120         xmlLog->DecIndent();
121         xmlLog->WriteEndRule("parse");
122     }
123     if (!match.Hit() || stop.Start() != int(end - start))
124     {
125         if (startRule)
126         {
127             throw ExpectationFailure(startRule->Info(), fileName, stop, start, end);
128         }
129         else
130         {
131             throw ParsingException("grammar '" + ToUtf8(FullName()) + "' has no start rule", fileName, stop, start, end);
132         }
133     }
134 }

136 Match Grammar::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
137 {
138     if (startRule)
139     {
140         if (!contentParser)
141         {
142             if (skipRule)
143             {
144                 contentParser = new SequenceParser(new SequenceParser(new OptionalParser(skipRule), startRule), new OptionalParser(skipRule));
145                 Own(contentParser);
146             }
147             else
148             {
149                 contentParser = startRule;
150             }
151         }
152         return contentParser->Parse(scanner, stack, parsingData);
153     }
154     return Match::Nothing();
155 }

169 void Grammar::ResolveStartRule()
170 {
171     if (startRuleName.empty())
172     {
173         if (!rules.empty())
174         {
175             startRule = rules.front();
176         }
177         else
178         {
179             ThrowException("cannot resolve start rule because grammar '" + ToUtf8(FullName()) + "' is empty", GetSpan());
180         }
181     }
182     else
183     {
184         startRule = GetRule(startRuleName);
185     }
186 }

188 void Grammar::ResolveSkipRule()
189 {
190     if (!skipRuleName.empty())
191     {
192         skipRule = GetRule(skipRuleName);
193     }
194 }

196 void Grammar::Link()
197 {
198     if (!linked && !linking)
199     {
200         linking = true;
201         GetReferencedGrammars();
202         GrammarSetIt e = grammarReferences.end();
203         for (GrammarSetIt i = grammarReferences.begin(); i != e; ++i)
204         {
205             Grammar* grammarReference = *i;
206             grammarReference->Link();
207         }
208         LinkerVisitor linkerVisitor;
209         Accept(linkerVisitor);
210         linking = false;
211         linked = true;
212     }
213 }
