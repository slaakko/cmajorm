27 class Rule : publicParser
28 {
29 public:
30     typedef std::vector<AttrOrVariable> AttrOrVariableVec;
31     typedef std::vector<ActionParser*> ActionVec;
32     typedef std::vector<NonterminalParser*> NonterminalVec;
33     Rule(const std::u32string& name_, Scope* enclosingScope_, int id_, Parser* definition_);
34     Rule(const std::u32string& name_, Scope* enclosingScope_, Parser* definition_);
35     Rule(const std::u32string& name_, Scope* enclosingScope_);
36     bool IsRule() const override { return true; }
37     Grammar* GetGrammar() const { return grammar; }
38     void SetGrammar(Grammar* grammar_) { grammar = grammar_; }
39     Parser* Definition() const { return definition; }
40     void SetDefinition(Parser* definition_);
41     const AttrOrVariableVec& InheritedAttributes() const { return inheritedAttributes; }
42     void AddInheritedAttribute(const AttrOrVariable& attr) { inheritedAttributes.push_back(attr); }
43     const AttrOrVariableVec& LocalVariables() const { return localVariables; }
44     void AddLocalVariable(const AttrOrVariable& var) { localVariables.push_back(var); }
45     const std::u32string& ValueTypeName() const { return valueTypeName; }
46     void SetValueTypeName(const std::u32string& valueTypeName_) { valueTypeName = valueTypeName_; }
47     int NumberOfParameters() const { return int(inheritedAttributes.size()); }
48     const ActionVec& Actions() const { return actions; }
49     void AddAction(ActionParser* action);
50     ActionParser* GetAction(const std::u32string& actionName) const;
51     const NonterminalVec& Nonterminals() const { return nonterminals; }
52     void AddNonterminal(NonterminalParser* nonterminal);
53     NonterminalParser* GetNonterminal(const std::u32string& nonterminalName) const;
54     bool TakesOrReturnsValuesOrHasLocals() const { return !inheritedAttributes.empty() || !localVariables.empty() || !valueTypeName.empty(); }
55     bool Specialized() const { return TakesOrReturnsValuesOrHasLocals() || !actions.empty(); }
56     std::u32string SpecializedTypeName() const { return Specialized() ? Name() + U"Rule" : U"Rule"; }
57     virtual void Link() {}
58     void ExpandCode();
59     Match Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData) override;
60     virtual void Enter(ObjectStack& stack) {}
61     virtual void Leave(ObjectStack& stack, bool matched) {}
62     virtual void Enter(ObjectStack& stack, ParsingData* parsingData) {}
63     virtual void Leave(ObjectStack& stack, ParsingData* parsingData, bool matched) {}
64     void Accept(Visitor& visitor) override;
65     int Id() const { return id; }
66 private:
67     int id;
68     Grammar* grammar;
69     Parser* definition;
70     AttrOrVariableVec inheritedAttributes;
71     AttrOrVariableVec localVariables;
72     std::u32string valueTypeName;
73     ActionVec actions;
74     NonterminalVec nonterminals;
75 };
