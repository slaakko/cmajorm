 9 namespace cmajor { namespace parsing {

11 class Match 
12 {
13 public:
14     Match(bool hit_, int length_): hit(hit_), length(length_) {}
15     bool Hit() const { return hit; }
16     int Length() const { return length; }
17     void Concatenate(const Match& that)
18     {
19         length += that.length;
20     }
21     static Match Empty()
22     {
23         return Match(true, 0);
24     }
25     static Match One()
26     {
27         return Match(true, 1);
28     }
29     static Match Nothing()
30     {
31         return Match(false, -1);
32     }
33 private:
34     bool hit;
35     int length;
36 };
