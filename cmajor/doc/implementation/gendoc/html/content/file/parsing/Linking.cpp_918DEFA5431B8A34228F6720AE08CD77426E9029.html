 24 LinkerVisitor::LinkerVisitor(): currentGrammar(0), currentRule(0)
 25 {
 26 }
 27 
 28 void LinkerVisitor::BeginVisit(Grammar& grammar)
 29 {
 30     if (!grammar.IsExternal())
 31     {
 32         currentGrammar = &grammar;
 33     }
 34     else
 35     {
 36         currentGrammar = 0;
 37     }
 38 }
 39 
 40 void LinkerVisitor::EndVisit(Grammar& grammar)
 41 {
 42     if (currentGrammar)
 43     {
 44         currentGrammar->ResolveStartRule();
 45         currentGrammar->ResolveSkipRule();
 46     }
 47 }
 48 
 49 void LinkerVisitor::Visit(RuleLink& link)
 50 {
 51     if (currentGrammar)
 52     {
 53         ParsingObject* object = currentGrammar->GetScope()->Get(link.LinkedRuleName(), ObjectKind::rule);
 54         if (object)
 55         {
 56             if (object->IsRule())
 57             {
 58                 Rule* rule = static_cast<Rule*>(object);
 59                 link.SetRule(rule);
 60                 currentGrammar->AddGrammarReference(rule->GetGrammar());
 61                 link.AddToScope();
 62             }
 63             else
 64             {
 65                 ThrowException("'" + ToUtf8(link.LinkedRuleName()) + "' is not a rule", link.GetSpan());
 66             }
 67         }
 68         else
 69         {
 70             ThrowException("rule '" + ToUtf8(link.LinkedRuleName()) + "' not found", link.GetSpan());
 71         }
 72     }
 73 }
 74 
 75 void LinkerVisitor::BeginVisit(Rule& rule)
 76 {
 77     if (currentGrammar)
 78     {
 79         currentRule = &rule;
 80         actionNumber = 0;
 81     }
 82 }
 83 
 84 void LinkerVisitor::EndVisit(Rule& rule)
 85 {
 86     if (currentGrammar)
 87     {
 88         rule.Link();
 89     }
 90 }

 92 void LinkerVisitor::BeginVisit(ActionParser& parser)
 93 {
 94     if (currentGrammar)
 95     {
 96         if (parser.Name().empty())
 97         {
 98             parser.SetName(U"A" + ToUtf32(std::to_string(actionNumber)));
 99             ++actionNumber;
100         }
101         currentRule->AddAction(static_cast<ActionParser*>(&parser));
102     }
103 }
104 
105 void LinkerVisitor::Visit(NonterminalParser& parser)
106 {
107     if (currentGrammar)
108     {
109         try
110         {
111             Rule* rule = currentGrammar->GetRule(parser.RuleName());
112             parser.SetRule(rule);
113             if (parser.Specialized())
114             {
115                 currentRule->AddNonterminal(static_cast<NonterminalParser*>(&parser));
116             }
117         }
118         catch (const ParsingException& ex;)
119         {
120             ThrowException(ex.Message(), parser.GetSpan());
121         }
122         catch (const std::exception& ex;)
123         {
124             ThrowException(ex.what(), parser.GetSpan());
125         }
126         if (parser.NumberOfArguments() != parser.GetRule()->NumberOfParameters())
127         {
128             ThrowException("rule '" + ToUtf8(parser.RuleName()) + "' takes " + std::to_string(parser.GetRule()->NumberOfParameters()) + " parameters (" + 
129                 std::to_string(parser.NumberOfArguments()) + " arguments supplied)", parser.GetSpan());
130         }
131     }
132 }
133 
134 void LinkerVisitor::Visit(KeywordParser& parser)
135 {
136     if (currentGrammar)
137     {
138         if (!parser.ContinuationRuleName().empty())
139         {
140             parser.SetContinuationRule(currentGrammar->GetRule(parser.ContinuationRuleName()));
141         }
142     }
143 }
144 
145 void LinkerVisitor::Visit(KeywordListParser& parser)
146 {
147     if (currentGrammar)
148     {
149         parser.SetSelectorRule(currentGrammar->GetRule(parser.SelectorRuleName()));
150     }
151 }

154 void Link(ParsingDomain* parsingDomain)
155 {
156     LinkerVisitor visitor;
157     parsingDomain->GlobalNamespace()->Accept(visitor);
158 }

160 class ExpandCodeVisitor : public Visitor
161 {
162 public:
163     virtual void BeginVisit(Grammar& grammar)
164     {
165         if (!grammar.IsExternal())
166         {
167             const Grammar::Grammar& rules = grammar.Rules();
168             int n = int(rules.size());
169             for (int i = 0; i < n; ++i)
170             {
171                 Rule* rule = rules[i];
172                 rule->ExpandCode();
173             }
174         }
175     }
176 };
177 
178 void ExpandCode(ParsingDomain* parsingDomain)
179 {
180     ExpandCodeVisitor visitor;
181     parsingDomain->GlobalNamespace()->Accept(visitor);
182 }
