12 Namespace::Namespace(const std::u32string& name_, Scope* enclosingScope_) : ParsingObject(name_, enclosingScope_, ObjectKind::ns)
13 {
14     SetScope(new Scope(Name(), EnclosingScope()));
15 }
16 
17 void Namespace::Accept(Visitor& visitor)
18 {
19     visitor.BeginVisit(*this);
20     GetScope()->Accept(visitor);
21     visitor.EndVisit(*this);
22 }
23 
24 UsingObject::UsingObject(cmajor::codedom::UsingObject* subject_, Scope* enclosingScope_) :
25     ParsingObject(subject_->Name(), enclosingScope_, ObjectKind::usingObject), subject(subject_)
26 {
27     if (!subject->IsOwned())
28     {
29         subject->SetOwned();
30         ownedSubject.reset(subject);
31     }
32 }
33 
34 void UsingObject::Accept(Visitor& visitor)
35 {
36     visitor.Visit(*this);
37 }
