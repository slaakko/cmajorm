  44 Match StringParser::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
  45 {
  46     Match match = Match::Empty();
  47     int i = 0;
  48     int n = int(s.length());
  49     while (i < n && !scanner.AtEnd() && scanner.GetChar() == s[i])
  50     {
  51         ++scanner;
  52         ++i;
  53         match.Concatenate(Match::One());
  54     }
  55     if (i == n)
  56     {
  57         return match;
  58     }
  59     return Match::Nothing();
  60 }

  77 void CharSetParser::InitRanges()
  78 {
  79     int i = 0;
  80     int n = int(s.length());
  81     while (i < n)
  82     {
  83         char32_t first = s[i];
  84         char32_t last = first;
  85         ++i;
  86         if (i < n)
  87         {
  88             if (s[i] == '-')
  89             {
  90                 ++i;
  91                 if (i < n)
  92                 {
  93                     last = s[i];
  94                     ++i;
  95                 }
  96                 else
  97                 {
  98                     ranges.push_back(CharRange(first, first));
  99                     first = '-';
 100                     last = '-';
 101                 }
 102             }
 103         }
 104         ranges.push_back(CharRange(first, last));
 105     }
 106 }

 108 Match CharSetParser::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
 109 {
 110     if (!scanner.AtEnd())
 111     {
 112         char32_t c = scanner.GetChar();
 113         bool found = false;
 114         for (const CharRange& range : ranges)
 115         {
 116             if (range.Includes(c))
 117             {
 118                 found = true;
 119                 break;
 120             }
 121         }
 122         if (found != inverse)
 123         {
 124             ++scanner;
 125             return Match::One();
 126         }
 127     }
 128     return Match::Nothing();
 129 }
