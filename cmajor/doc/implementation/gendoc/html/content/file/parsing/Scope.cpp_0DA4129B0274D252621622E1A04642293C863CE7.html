 22 std::u32string Scope::FullName() const
 23 {
 24     if (!fullNameComputed)
 25     {
 26         Scope* parentScope = EnclosingScope();
 27         std::u32string parentScopeFullName = parentScope ? parentScope->FullName() : U"";
 28         fullName = parentScopeFullName.empty() ? Name() : parentScopeFullName + U"." + Name();
 29         fullNameComputed = true;
 30     }
 31     return fullName;
 32 }

 34 void Scope::Accept(Visitor& visitor)
 35 {
 36     FullNameMapIt e = fullNameMap.end();
 37     for (FullNameMapIt i = fullNameMap.begin(); i != e; ++i)
 38     {
 39         ParsingObject* object = i->second;
 40         object->Accept(visitor);
 41     }
 42 }

 44 void Scope::Add(ParsingObject* object)
 45 {
 46     Own(object);
 47     std::u32string objectFullName = object->FullName();
 48     if (!fullNameMap.insert(std::make_pair(objectFullName, object)).second)
 49     {
 50         std::u32string msg = U"object '" + objectFullName + U"' already exists (detected in scope '" + FullName() + U"')";
 51         throw std::runtime_error(ToUtf8(msg));
 52     }
 53     std::vector<ParsingObject*>& objects = shortNameMap[object->Name()];
 54     for (ParsingObject* o : objects)
 55     {
 56         if (o == object) return;
 57         if ((o->Kind() & object->Kind()) != ObjectKind::none)
 58         {
 59             std::u32string msg = U"object '" + objectFullName + U"' already exists (detected in scope '" + FullName() + U"')";
 60             throw std::runtime_error(ToUtf8(msg));
 61         }
 62     }
 63     objects.push_back(object);
 64 }

 66 Scope* Scope::GetGlobalScope() const
 67 {
 68     Scope* globalScope = const_cast<Scope*>(this);
 69     while (globalScope->EnclosingScope())
 70     {
 71         globalScope = globalScope->EnclosingScope();
 72     }
 73     return globalScope;
 74 }

 76 void Scope::AddNamespace(Namespace* nsToAdd)
 77 {
 78     Own(nsToAdd);
 79     Namespace* parent = GetGlobalScope()->Ns();
 80     std::vector<std::u32string> nameComponents = cmajor::util::Split(nsToAdd->FullName(), '.');
 81     if (nameComponents.empty())
 82     {
 83         throw std::runtime_error("namespace components empty");
 84     }
 85     int n = int(nameComponents.size());
 86     for (int i = 0; i < n - 1; ++i)
 87     {
 88         const std::u32string& namespaceName = nameComponents[i];
 89         Scope* parentScope = parent->GetScope();
 90         ParsingObject* object = parentScope->Get(namespaceName, ObjectKind::ns);
 91         if (object)
 92         {
 93             if (object->IsNamespace())
 94             {
 95                 parent = static_cast<Namespace*>(object);
 96             }
 97             else
 98             {
 99                 throw std::runtime_error("object '" + ToUtf8(namespaceName) + "' does not denote a namespace");
100             }
101         }
102         else
103         {
104             parent = new Namespace(namespaceName, parentScope);
105             Own(parent);
106             parent->GetScope()->SetName(namespaceName);
107             parent->GetScope()->SetNs(parent);
108             parentScope->Add(parent);
109         }
110     }
111     std::u32string commonName = nameComponents[n - 1];
112     nsToAdd->SetName(commonName);
113     nsToAdd->GetScope()->SetName(commonName);
114     nsToAdd->SetEnclosingScope(parent->GetScope());
115     nsToAdd->GetScope()->SetEnclosingScope(parent->GetScope());
116     parent->GetScope()->Add(nsToAdd);
117 }

119 ParsingObject* Scope::GetQualifiedObject(const std::u32string& qualifiedObjectName, ObjectKind kind) const
120 {
121     std::vector<std::u32string> components = cmajor::util::Split(qualifiedObjectName, '.');
122     int n = int(components.size());
123     Scope* s = const_cast<Scope*>(this);
124     while (s)
125     {
126         Scope* subScope = s;
127         int i = 0;
128         ShortNameMapIt it = subScope->shortNameMap.find(components[i]);
129         while (it != subScope->shortNameMap.end())
130         {
131             if (i == n - 1)
132             {
133                 for (ParsingObject* object : it->second)
134                 {
135                     if ((object->Kind() & kind) != ObjectKind::none)
136                     {
137                         return object;
138                     }
139                 }
140             }
141             else
142             {
143                 bool found = false;
144                 for (ParsingObject* object : it->second)
145                 {
146                     if ((object->Kind() & ObjectKind::parent) != ObjectKind::none)
147                     {
148                         ++i;
149                         subScope = object->GetScope();
150                         it = subScope->shortNameMap.find(components[i]);
151                         found = true;
152                         break;
153                     }
154                 }
155                 if (!found)
156                 {
157                     it = subScope->shortNameMap.end();
158                 }
159             }
160         }
161         s = s->EnclosingScope();
162     }
163     return nullptr;
164 }

166 ParsingObject* Scope::Get(const std::u32string& objectName, ObjectKind kind) const
167 {
168     if (objectName.find('.') != std::string::npos)
169     {
170         FullNameMapIt i = fullNameMap.find(objectName);
171         if (i != fullNameMap.end())
172         {
173             return i->second;
174         }
175         else
176         {
177             ParsingObject* object = GetQualifiedObject(objectName, kind);
178             if (object)
179             {
180                 return object;
181             }
182         }
183     }
184     else
185     {
186         ShortNameMapIt i = shortNameMap.find(objectName);
187         if (i != shortNameMap.end())
188         {
189             for (ParsingObject* object : i->second)
190             {
191                 if ((object->Kind() & kind) != ObjectKind::none)
192                 {
193                     return object;
194                 }
195             }
196         }
197     }
198     return nullptr;
199 }

201 Namespace* Scope::GetNamespace(const std::u32string& fullNamespaceName) const
202 {
203     ParsingObject* object = GetQualifiedObject(fullNamespaceName, ObjectKind::ns);
204     if (object)
205     {
206         if (object->IsNamespace())
207         {
208             return static_cast<Namespace*>(object);
209         }
210         else
211         {
212             throw std::runtime_error("object '" + ToUtf8(fullNamespaceName) + "' is not a namespace");
213         }
214     }
215     return nullptr;
216 }
