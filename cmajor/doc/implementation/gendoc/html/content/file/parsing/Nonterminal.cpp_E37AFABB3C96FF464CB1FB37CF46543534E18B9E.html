33 void NonterminalParser::SetArguments(const ArgumentVector& arguments_)
34 {
35     arguments = arguments_;
36     for (cmajor::codedom::CppObject* arg : arguments)
37     {
38         if (!arg->IsOwned())
39         {
40             arg->SetOwned();
41             ownedArgs.push_back(     std::unique_ptr<cmajor::codedom::CppObject>(arg));
42         }
43     }
44 }

46 bool NonterminalParser::Specialized() const
47 {
48     return numberOfArguments > 0 || !arguments.empty() || 
49         (rule && rule->TakesOrReturnsValuesOrHasLocals());
50 }

52 Match NonterminalParser::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
53 {
54     if (rule)
55     {
56         if (preCall)
57         {
58             (*preCall)(stack, parsingData);
59         }
60         Match match = rule->Parse(scanner, stack, parsingData);
61         if (postCall)
62         {
63             (*postCall)(stack, parsingData, match.Hit());
64         }
65         return match;
66     }
67     return Match::Nothing();
68 }
