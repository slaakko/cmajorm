53 Match ActionParser::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
54 {
55     Span actionSpan = scanner.GetSpan();
56     Match match = Child()->Parse(scanner, stack, parsingData);
57     if (match.Hit())
58     {
59         if (action)
60         {
61             bool pass = true;
62             actionSpan.SetEnd(scanner.GetSpan().Start());
63             const char32_t* matchBegin = scanner.Start() + actionSpan.Start();
64             const char32_t* matchEnd = scanner.Start() + actionSpan.End();
65             (*action)(matchBegin, matchEnd, actionSpan, scanner.FileName(), parsingData, pass);
66             if (!pass)
67             {
68                 return Match::Nothing();
69             }
70         }
71     }
72     else if (failureAction)
73     {
74         (*failureAction)(parsingData);
75     }
76     return match;
77 }
