 22 class Span 
 23 {
 24 public:
 25     Span(): fileIndex(-1), lineNumber(-1), start(-1), end(-1) {}
 26     Span(int32_t fileIndex_): fileIndex(fileIndex_), lineNumber(1), start(0), end(0) {}
 27     Span(int32_t fileIndex_, int32_t lineNumber_, int32_t start_): fileIndex(fileIndex_), lineNumber(lineNumber_), start(start_), end(start + 1) {}
 28     Span(int32_t fileIndex_, int32_t lineNumber_, int32_t start_, int32_t end_): fileIndex(fileIndex_), lineNumber(lineNumber_), start(start_), end(end_) {}
 29     int32_t FileIndex() const { return fileIndex; }
 30     void SetFileIndex(int32_t fileIndex_) { fileIndex = fileIndex_; }
 31     int32_t LineNumber() const { return lineNumber; }
 32     int32_t Start() const { return start; }
 33     int32_t End() const { return end; }
 34     bool Valid() const { return start != -1; }
 35     void IncLineNumber() { ++lineNumber; }
 36     void operator++()
 37     {
 38         ++start;
 39         ++end;
 40     }
 41     void operator--()
 42     {
 43         --start;
 44         --end;
 45     }
 46     void SetStart(int32_t start_) { start = start_; }
 47     void SetEnd(int32_t end_) { end = end_; }
 48     bool IsNull() const { return fileIndex == 0 && lineNumber == 0 && start == 0 && end == 0; }
 49 private:
 50     int32_t fileIndex;
 51     int32_t lineNumber;
 52     int32_t start;
 53     int32_t end;
 54 };

 57 {
 58     return left.FileIndex() == right.FileIndex() && left.LineNumber() == right.LineNumber() && left.Start() == right.Start() && left.End() == right.End();
 59 }
