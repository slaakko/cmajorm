13 class Visitor;
14 class Scope;
15 
16 enum class ObjectKind : uint8_t 
17 {
18     none = 0, 
19     grammar = 1 << 0, 
20     ns = 1 << 1, 
21     usingObject = 1 << 2, 
22     parser = 1 << 3, 
23     parsingDomain = 1 << 4, 
24     rule = 1 << 5, 
25     ruleLink = 1 << 6, 
26     scope = 1 << 7, 
27     parent = grammar | ns | scope
28 };
29 
30 constexpr, inlineObjectKind operator|(ObjectKind left, ObjectKind right)
31 {
32     return  ObjectKind(uint8_t(left) | uint8_t(right));
33 }
34 
35 constexpr, inlineObjectKind operator&(ObjectKind left, ObjectKind right)
36 {
37     return  ObjectKind(uint8_t(left) & uint8_t(right));
38 }

40 class ParsingObject 
41 {
42 public:
43     static const int external = -1;
44     ParsingObject(const std::u32string& name_, ObjectKind kind_);
45     ParsingObject(const std::u32string& name_, Scope* enclosingScope_, ObjectKind kind_);
46     virtual ~ParsingObject();
47     virtual void Accept(Visitor& visitor) = 0;
48     void Own(ParsingObject* object);
49     bool IsOwned() const { return isOwned; }
50     void SetOwned() { isOwned = true; }
51     void ResetOwned() { isOwned = false; }
52     void SetOwner(int owner_) { owner = owner_; }
53     int Owner() const { return owner; }
54     void SetExternal() { owner = external; }
55     bool IsExternal() const { return owner == external; }
56     const std::u32string& Name() const { return name; }
57     void SetName(const std::u32string& name_) { name = name_; }
58     virtual std::u32string FullName() const;
59     Scope* EnclosingScope() const { return enclosingScope; }
60     void SetEnclosingScope(Scope* enclosingScope_) { enclosingScope = enclosingScope_; }
61     void SetScope(Scope* scope_);
62     Scope* GetScope() const { if (scope == nullptr) return enclosingScope; else return scope; }
63     virtual bool IsActionParser() const { return false; }
64     virtual bool IsNonterminalParser() const { return false; }
65     virtual bool IsRule() const { return false; }
66     virtual bool IsRuleLink() const { return false; }
67     virtual bool IsNamespace() const { return false; }
68     virtual void AddToScope();
69     void SetSpan(const Span& span_) { span = span_; }
70     const Span& GetSpan() const { return span; }
71     ObjectKind Kind() const { return kind; }
72 private:
73     ObjectKind kind;
74     std::u32string name;
75     std::vector<std::unique_ptr<ParsingObject>> ownedObjects;
76     int owner;
77     bool isOwned;
78     Scope* enclosingScope;
79     Scope* scope;
80     Span span;
81 };
