 8 #include <ostream>
 9 #include <string>

11 namespace cmajor { namespace parsing {
12 
13 class XmlLog 
14 {
15 public:
16     XmlLog(std::ostream& stream_, int maxLineLength_): stream(stream_), maxLineLength(maxLineLength_), indent(0) {}
17     void IncIndent() { ++indent; }
18     void DecIndent() { --indent; }
19     void WriteBeginRule(const std::string& ruleName) { Write("<" + ruleName + ">"); }
20     void WriteEndRule(const std::string& ruleName) { Write("</" + ruleName + ">"); }
21     void WriteTry(const std::string& s) { WriteElement("try", s); }
22     void WriteSuccess(const std::string& match) { WriteElement("success", match); }
23     void WriteFail() { Write("<fail/>"); }
24     void WriteElement(const std::string& elementName, const std::string& elementContent);
25     void Write(const std::string& s);
26     int MaxLineLength() const { return maxLineLength; }
27 private:
28     std::ostream& stream;
29     int maxLineLength;
30     int indent;
31 };
32 
33 } } // namespace cmajor::parsing
