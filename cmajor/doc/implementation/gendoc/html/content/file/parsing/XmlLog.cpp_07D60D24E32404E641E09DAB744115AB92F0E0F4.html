12 void XmlLog::WriteElement(const std::string& elementName, const std::string& elementContent)
13 {
14     std::string converted = XmlEscape(elementContent);
15     int convertedLength = int(converted.length());
16     int lineLength = indent + 2 * int(elementName.length()) + 5 + convertedLength;
17     std::string s = converted;
18     if (lineLength > maxLineLength)
19     {
20         lineLength += 3;
21         s = converted.substr(0, std::max(0, convertedLength - (lineLength - maxLineLength))) + "...";
22     }
23     Write("<" + elementName + ">" + s + "</" + elementName + ">");
24 }
25 
26 void XmlLog::Write(const std::string& s)
27 {
28     if (indent > 0)
29     {
30         stream << std::string(indent, ' ');
31     }
32     stream << s << std::endl;
33 }
