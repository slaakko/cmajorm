12 std::string HexEscape(char c)
13 {
14     std::stringstream s;
15     s << "\\x" << std::hex << (unsigned int)(unsigned char)c;
16     return s.str();
17 }
18 
19 std::string XmlCharStr(char c)
20 {
21     switch (c)
22     {
23         case '&': return "&amp;";
24         case '<': return "&lt;";
25         case '>': return "&gt;";
26         case '\a': return "\\a";
27         case '\b': return "\\b";
28         case '\f': return "\\f";
29         case '\n': return "\\n";
30         case '\r': return "\\r";
31         case '\t': return "\\t";
32         case '\v': return "\\v";
33         default:
34         {
35             if (((int)c >= 32 && (int)c <= 126) || std::isalnum(c) || std::ispunct(c))
36             {
37                 return std::string(1, c);
38             }
39             else
40             {
41                 return HexEscape(c);
42             }
43         }
44     }
45 }
46 
47 std::string XmlEscape(const std::string& s)
48 {
49     std::string result;
50     result.reserve(2 * s.length());
51     std::string::const_iterator e = s.end();
52     for (std::string::const_iterator i = s.begin(); i != e; ++i)
53     {
54         result.append(XmlCharStr(*i));
55     }
56     return result;
57 }
