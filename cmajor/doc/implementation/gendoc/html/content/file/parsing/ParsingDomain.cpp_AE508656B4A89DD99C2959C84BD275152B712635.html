 15 class ParsingDomainRepository 
 16 {
 17 public:
 18     static void Init();
 19     static void Done();
 20     static ParsingDomainRepository& Instance();
 21     void Register(ParsingDomain* parsingDomain);
 22 private:
 23     ParsingDomainRepository();
 24     static std::unique_ptr<ParsingDomainRepository> instance;
 25     static int initCount;
 26     typedef std::unordered_set<std::unique_ptr<ParsingDomain>> ParsingDomainSet;
 27     typedef ParsingDomainSet::const_iterator ParsingDomainSetIt;
 28     ParsingDomainSet parsingDomains;
 29 };

 35 void ParsingDomainRepository::Register(ParsingDomain* parsingDomain)
 36 {
 37     if (!parsingDomain->IsOwned())
 38     {
 39         parsingDomain->SetOwned();
 40         parsingDomains.insert(     std::unique_ptr<ParsingDomain>(parsingDomain));
 41     }
 42 }

 44 int ParsingDomainRepository::initCount = 0;
 45 
 46 void ParsingDomainRepository::Init()
 47 {
 48     if (initCount++ == 0)
 49     {
 50         instance.reset(new ParsingDomainRepository());
 51     }
 52 }

 54 void ParsingDomainRepository::Done()
 55 {
 56     if (--initCount == 0)
 57     {
 58         instance.reset();
 59     }
 60 }

 69 ParsingDomain::ParsingDomain(): ParsingObject(U"parsingDomain", ObjectKind::parsingDomain),

 82 void ParsingDomain::BeginNamespace(const std::u32string& ns)
 83 {
 84     std::u32string fullNamespaceName = currentNamespace->FullName().empty() ? ns : currentNamespace->FullName() + U"." + ns;
 85     namespaceStack.push(currentNamespace);
 86     NamespaceMapIt i = namespaceMap.find(fullNamespaceName);
 87     if (i != namespaceMap.end())
 88     {
 89         currentNamespace = i->second;
 90     }
 91     else
 92     {
 93         Namespace * newNs(new Namespace(fullNamespaceName, globalScope));
 94         Own(newNs);
 95         newNs->GetScope()->SetNs(newNs);
 96         namespaceMap[newNs->FullName()] = newNs;
 97         currentNamespace = newNs;
 98         globalScope->AddNamespace(newNs);
 99     }
100 }
101 
102 void ParsingDomain::EndNamespace()
103 {
104     currentNamespace = namespaceStack.top();
105     namespaceStack.pop();
106 }

108 Scope* ParsingDomain::GetNamespaceScope(const std::u32string& fullNamespaceName)
109 {
110     if (fullNamespaceName.empty())
111     {
112         return globalScope;
113     }
114     Namespace* ns = globalScope->GetNamespace(fullNamespaceName);
115     if (!ns)
116     {
117         ns = new Namespace(fullNamespaceName, globalScope);
118         Own(ns);
119         globalScope->AddNamespace(ns);
120     }
121     return ns->GetScope();
122 }

124 Grammar* ParsingDomain::GetGrammar(const std::u32string& grammarName)
125 {
126     GrammarMapIt i = grammarMap.find(grammarName);
127     if (i != grammarMap.end())
128     {
129         return i->second;
130     }
131     return nullptr;
132 }
133 
134 void ParsingDomain::AddGrammar(Grammar* grammar)
135 {
136     Own(grammar);
137     std::u32string grammarFullName = grammar->FullName();
138     GrammarMapIt i = grammarMap.find(grammarFullName);
139     if (i == grammarMap.end())
140     {
141         grammarMap[grammarFullName] = grammar;
142         grammar->SetNs(currentNamespace);
143         grammar->AddToScope();
144     }
145 }

147 void ParsingDomain::Accept(Visitor& visitor)
148 {
149     for (const std::pair<std::u32string, Grammar*>& p : grammarMap)
150     {
151         Grammar* grammar = p.second;
152         grammar->Accept(visitor);
153     }
154 }
