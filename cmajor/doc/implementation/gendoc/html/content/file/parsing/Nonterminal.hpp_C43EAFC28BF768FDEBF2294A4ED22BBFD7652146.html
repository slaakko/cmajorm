12 namespace cmajor { namespace parsing {
13 
14 class Rule;
15 class PreCall;
16 class PostCall;
17 
18 class NonterminalParser : publicParser
19 {
20 public:
21     typedef std::vector<cmajor::codedom::CppObject*> ArgumentVector;
22     NonterminalParser(const std::u32string& name_, const std::u32string& ruleName_);
23     NonterminalParser(const std::u32string& name_, const std::u32string& ruleName_, int numberOfArguments_);
24     const std::u32string& RuleName() const { return ruleName; }
25     const ArgumentVector& Arguments() const { return arguments; }
26     Rule* GetRule() const { return rule; }
27     void SetRule(Rule* rule_) { rule = rule_; }
28     void SetArguments(const ArgumentVector& arguments_);
29     int NumberOfArguments() const { return numberOfArguments != 0 ? numberOfArguments : int(arguments.size()); }
30     void SetPreCall(PreCall* preCall_);
31     void SetPostCall(PostCall* postCall_);
32     const Span& GetSpan() const { return span; }
33     void SetSpan(const Span& span_) { span = span_; }
34     std::u32string PreCallMethodName() const { return U"Pre" + name; }
35     std::u32string PostCallMethodName() const { return U"Post" + name; }
36     std::u32string ValueFieldName() const { return U"from" + name; }
37     std::u32string VariableName() const { return (!name.empty() ? std::u32string(1, std::tolower(name[0])) + name.substr(1) : name) + U"NonterminalParser"; }
38     bool Specialized() const;
39     Match Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData) override;
40     void Accept(Visitor& visitor) override;
41     bool IsNonterminalParser() const override { return true; }
42 private:
43     std::u32string name;
44     std::u32string ruleName;
45     int numberOfArguments;
46     ArgumentVector arguments;
47     std::vector<std::unique_ptr<cmajor::codedom::CppObject>> ownedArgs;
48     std::unique_ptr<PreCall> preCall;
49     std::unique_ptr<PostCall> postCall;
50     Span span;
51     Rule* rule;
52 };
53 
54 } } // namespace cmajor::parsing
