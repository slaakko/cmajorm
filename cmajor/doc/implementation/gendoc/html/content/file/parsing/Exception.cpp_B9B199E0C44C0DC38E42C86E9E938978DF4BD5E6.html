 23 const char32_t* LineStart(const char32_t* start, const char32_t* pos)
 24 {
 25     while (pos > start && *pos != '\n')
 26     {
 27         --pos;
 28     }
 29     if (*pos == '\n')
 30     {
 31         ++pos;
 32     }
 33     return pos;
 34 }
 35 
 36 const char32_t* LineEnd(const char32_t* end, const char32_t* pos)
 37 {
 38     while (pos < end && *pos != '\n')
 39     {
 40         ++pos;
 41     }
 42     return pos;
 43 }

 45 std::u32string GetErrorLines(const char32_t* start, const char32_t* end, const Span& span)
 46 {
 47     const char32_t* startPos = start + span.Start();
 48     if (startPos < start || startPos >= end)
 49     {
 50         return std::u32string();
 51     }
 52     const char32_t* lineStart = LineStart(start, startPos);
 53     int cols = static_cast<int>(startPos - lineStart);
 54     if (cols < 0)
 55     {
 56         cols = 0;
 57     }
 58     const char32_t* lineEnd = LineEnd(end, startPos);
 59     if (lineEnd < lineStart)
 60     {
 61         lineEnd = lineStart;
 62     }
 63     int lineLength = static_cast<int>(lineEnd - lineStart);
 64     std::u32string lines(NarrowString(lineStart, lineEnd));
 65     int spanCols = std::max(1, std::min(span.End() -  span.Start(), lineLength - cols));
 66     lines.append(1, '\n').append(std::u32string(cols, ' ')).append(spanCols, '^');
 67     return lines;
 68 }

 70 void GetColumns(const char32_t* start, const char32_t* end, const Span& span, int32_t& startCol, int32_t& endCol)
 71 {
 72     startCol = 0;
 73     endCol = 0;
 74     const char32_t* startPos = start + span.Start();
 75     if (startPos < start || startPos >= end)
 76     {
 77         return;
 78     }
 79     const char32_t* lineStart = LineStart(start, startPos);
 80     int cols = static_cast<int>(startPos - lineStart);
 81     if (cols < 0)
 82     {
 83         cols = 0;
 84     }
 85     startCol = cols + 1;
 86     const char32_t* lineEnd = LineEnd(end, startPos);
 87     if (lineEnd < lineStart)
 88     {
 89         lineEnd = lineStart;
 90     }
 91     int lineLength = static_cast<int>(lineEnd - lineStart);
 92     int spanCols = std::max(1, std::min(span.End() - span.Start(), lineLength - cols));
 93     endCol = startCol + spanCols;
 94 }

 96 void ThrowException(const std::string& message, const Span& span)
 97 {
 98     if (span.FileIndex() >= 0 && span.FileIndex() < int(files.size()))
 99     {
100         const std::string& filePath = files[span.FileIndex()];
101         cmajor::util::MappedInputFile file(filePath);
102         std::string s(file.Begin(), file.End());
103         std::u32string t(ToUtf32(s));
104         throw ParsingException(message, filePath, span, &t[0], &t[0] + t.length());
105     }
106     throw std::runtime_error(message);
107 }
