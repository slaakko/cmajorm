 14 class IdentifierCharsRule 
 15 {
 16 public:
 17     static void Init();
 18     static void Done();
 19     static IdentifierCharsRule& Instance();
 20     Rule* GetRule() const{ return rule.get(); }
 21 private:
 22     IdentifierCharsRule();
 23     std::unique_ptr<Rule> rule;
 24     static std::unique_ptr<IdentifierCharsRule> instance;
 25     static int initCount;
 26 };
 27 
 28 IdentifierCharsRule::IdentifierCharsRule():
 29     rule(new Rule(U"identifier_chars", nullptr, -1, 
 30             new PositiveParser(
 31                 new AlternativeParser(
 32                     new AlternativeParser(
 33                         new AlternativeParser(
 34                             new LetterParser(), 
 35                             new DigitParser()), 
 36                         new CharParser('_')), 
 37                     new CharParser('.')))))
 38 {
 39     rule->SetOwned();
 40 }

 42 void IdentifierCharsRule::Init()
 43 {
 44     if (initCount++ == 0)
 45     {
 46         instance.reset(new IdentifierCharsRule());
 47     }
 48 }

 50 void IdentifierCharsRule::Done()
 51 {
 52     if (--initCount == 0)
 53     {
 54         instance.reset();
 55     }
 56 }

 58 IdentifierCharsRule& IdentifierCharsRule::Instance()
 59 {

 61 }

 93 void KeywordParser::CreateKeywordRule()
 94 {
 95     keywordRule = new Rule(keyword, nullptr, -1, 
 96         new DifferenceParser(
 97             keywordStringParser, 
 98             new TokenParser(

101                     continuationRule))));
102     Own(keywordRule);
103 }

120 Match KeywordListParser::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
121 {
122     if (selectorRule)
123     {
124         Span save = scanner.GetSpan();
125         Match match = selectorRule->Parse(scanner, stack, parsingData);
126         if (match.Hit())
127         {
128             const char32_t* matchBegin = scanner.Start() + save.Start();
129             const char32_t* matchEnd = scanner.Start() + scanner.GetSpan().Start();
130             std::u32string keyword(matchBegin, matchEnd);
131             if (keywords.find(keyword) != keywords.end())
132             {
133                 return match;
134             }
135             scanner.SetSpan(save);
136         }
137     }
138     return Match::Nothing();
139 }
