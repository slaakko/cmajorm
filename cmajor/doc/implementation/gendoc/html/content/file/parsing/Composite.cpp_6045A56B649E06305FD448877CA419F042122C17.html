 45 Match PositiveParser::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
 46 {
 47     Match match = Child()->Parse(scanner, stack, parsingData);
 48     if (match.Hit())
 49     {
 50         for (;;)
 51         {
 52             Span save = scanner.GetSpan();
 53             scanner.Skip();
 54             Match next = Child()->Parse(scanner, stack, parsingData);
 55             if (next.Hit())
 56             {
 57                 match.Concatenate(next);
 58             }
 59             else
 60             {
 61                 scanner.SetSpan(save);
 62                 break;
 63             }
 64         }
 65     }
 66     return match;
 67 }

 80 Match KleeneStarParser::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
 81 {
 82     Match match = Match::Empty();
 83     bool first = true;
 84     for (;;)
 85     {
 86         Span save = scanner.GetSpan();
 87         if (first)
 88         {
 89             first = false;
 90         }
 91         else
 92         {
 93             scanner.Skip();
 94         }
 95         Match next = Child()->Parse(scanner, stack, parsingData);
 96         if (next.Hit())
 97         {
 98             match.Concatenate(next);
 99         }
100         else
101         {
102             scanner.SetSpan(save);
103             break;
104         }
105     }
106     return match;
107 }

120 Match ExpectationParser::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
121 {
122     Span expectationSpan = scanner.GetSpan();
123     Match match = Match::Nothing();
124     try
125     {
126         match = Child()->Parse(scanner, stack, parsingData);
127     }
128     catch (const ExpectationFailure& ex;)
129     {
130         throw ExpectationFailure(Child()->Info() + ex.Info(), ex.FileName(), ex.GetSpan(), ex.Start(), ex.End());
131     }
132     if (match.Hit())
133     {
134         return match;
135     }
136     else
137     {
138         throw ExpectationFailure(Child()->Info(), scanner.FileName(), expectationSpan, scanner.Start(), scanner.End());
139     }
140 }

195 Match SequenceParser::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
196 {
197     Match leftMatch = Left()->Parse(scanner, stack, parsingData);
198     if (leftMatch.Hit())
199     {
200         scanner.Skip();
201         Match rightMatch = Right()->Parse(scanner, stack, parsingData);
202         if (rightMatch.Hit())
203         {
204             leftMatch.Concatenate(rightMatch);
205             return leftMatch;
206         }
207     }
208     return Match::Nothing();
209 }

249 Match DifferenceParser::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
250 {
251     Span save = scanner.GetSpan();
252     Match leftMatch = Left()->Parse(scanner, stack, parsingData);
253     if (leftMatch.Hit())
254     {
255         Span tmp = scanner.GetSpan();
256         scanner.SetSpan(save);
257         save = tmp;
258         Match rightMatch = Right()->Parse(scanner, stack, parsingData);
259         if (!rightMatch.Hit() || rightMatch.Length() < leftMatch.Length())
260         {
261             scanner.SetSpan(save);
262             return leftMatch;
263         }
264     }
265     return Match::Nothing();
266 }

281 Match ExclusiveOrParser::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
282 {
283     Span save = scanner.GetSpan();
284     Match leftMatch = Left()->Parse(scanner, stack, parsingData);
285     Span temp = scanner.GetSpan();
286     scanner.SetSpan(save);
287     save = temp;
288     Match rightMatch = Right()->Parse(scanner, stack, parsingData);
289     bool match = leftMatch.Hit() ? !rightMatch.Hit() : rightMatch.Hit();
290     if (match)
291     {
292         if (leftMatch.Hit())
293         {
294             scanner.SetSpan(save);
295         }
296         return leftMatch.Hit() ? leftMatch : rightMatch;
297     }
298     return Match::Nothing();
299 }

314 Match IntersectionParser::Parse(Scanner& scanner, ObjectStack& stack, ParsingData* parsingData)
315 {

317     Match leftMatch = Left()->Parse(scanner, stack, parsingData);
318     if (leftMatch.Hit())
319     {
320         scanner.SetSpan(save);
321         Match rightMatch = Right()->Parse(scanner, stack, parsingData);
322         if (leftMatch.Length() == rightMatch.Length())
323         {
324             return leftMatch;
325         }
326     }
327     return Match::Nothing();
