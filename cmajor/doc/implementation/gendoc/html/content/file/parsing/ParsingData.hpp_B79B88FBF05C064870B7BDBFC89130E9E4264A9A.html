15 struct Context 
16 {
17     virtual ~Context();
18 };
19 
20 class ContextStack 
21 {
22 public:
23     void Push(Context* context) { stack.push(     std::unique_ptr<Context>(context));}
24     Context* Pop() { Assert(!stack.empty(), "context stack is empty"); Context* context = stack.top().release(); stack.pop(); return context; }
25 private:
26     std::stack::stack<std::unique_ptr<Context>>stack;
27 };
28 
29 struct RuleData 
30 {
31     std::unique_ptr<Context> context;
32     ContextStack contextStack;
33 };
34 
35 class ParsingData 
36 {
37 public:
38     ParsingData(int numRules);
39     void PushContext(int ruleId, Context* context)
40     {
41         Assert(ruleId >= 0 && ruleId < ruleData.size(), "invalid rule id");
42         ruleData[ruleId].contextStack.Push(ruleData[ruleId].context.release());
43         ruleData[ruleId].context.reset(context);
44     }
45     void PopContext(int ruleId)
46     {
47         Assert(ruleId >= 0 && ruleId < ruleData.size(), "invalid rule id");
48         ruleData[ruleId].context.reset(ruleData[ruleId].contextStack.Pop());
49     }
50     Context* GetContext(int ruleId) const
51     {
52         Assert(ruleId >= 0 && ruleId < ruleData.size(), "invalid rule id");
53         return ruleData[ruleId].context.get();
54     }
55 private:
56     std::vector<RuleData> ruleData;
57 };
