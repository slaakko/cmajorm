26 std::u32string ParsingObject::FullName() const
27 {
28     std::u32string scopeName = U"";
29     if (enclosingScope)
30     {
31         scopeName = enclosingScope->FullName();
32     }
33     if (scopeName.empty())
34     {
35         return name;
36     }
37     return scopeName + U"." + name;
38 }

40 void ParsingObject::Own(ParsingObject* object)
41 {
42     if (object)
43     {
44         if (!object->IsOwned())
45         {
46             object->SetOwned();
47             ownedObjects.push_back(     std::unique_ptr<ParsingObject>(object));
48         }
49     }
50 }

52 void ParsingObject::SetScope(Scope* scope_)
53 {
54     scope = scope_;
55     Own(scope);
56 }

58 void ParsingObject::AddToScope()
59 {
60     if (enclosingScope)
61     {
62         try
63         {
64             enclosingScope->Add(this);
65         }
66         catch (std::exception& ex;)
67         {
68             ThrowException(ex.what(), span);
69         }
70         if (scope)
71         {
72             scope->SetEnclosingScope(enclosingScope);
73         }
74     }
75 }
