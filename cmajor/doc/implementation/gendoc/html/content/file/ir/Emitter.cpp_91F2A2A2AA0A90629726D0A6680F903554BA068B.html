 61 llvm::DebugLoc Emitter::GetDebugLocation(const Span& span)
 62 {
 63     if (!diCompileUnit || !span.Valid()) return llvm::DebugLoc();

 68 void Emitter::SetCurrentDebugLocation(const Span& span)
 69 {
 70     if (!diCompileUnit) return;
 71     if (inPrologue || !span.Valid())
 72     {
 73         currentDebugLocation = llvm::DebugLoc();

 75     }
 76     else

 88 llvm::DIFile* Emitter::GetFile(int32_t fileIndex)
 89 {
 90     if (fileIndex == -1)
 91     {
 92         return diFile;
 93     }
 94     auto it = fileMap.find(fileIndex);
 95     if (it != fileMap.cend())
 96     {
 97         return it->second;
 98     }
 99     std::string sourceFilePath = GetSourceFilePath(fileIndex);
100     if (sourceFilePath.empty())
101     {
102         return diFile;
103     }
104     llvm::DIFile* file = diBuilder->createFile(Path::GetFileName(sourceFilePath), Path::GetDirectoryName(sourceFilePath));
105     fileMap[fileIndex] = file;
106     return file;
107 }

129 void Emitter::ReplaceForwardDeclarations()
130 {
131     std::unordered_map<llvm::DIType*, void*> currentFwdDeclarationMap;
132     std::swap(currentFwdDeclarationMap, fwdDeclarationMap);
133     while (!currentFwdDeclarationMap.empty())
134     {
135         for (const auto& p : currentFwdDeclarationMap)
136         {
137             llvm::DIType* fwdDeclaration = p.first;
138             void* type = p.second;
139             llvm::DIType* diType = GetDIType(type);
140             if (!diType)
141             {
142                 diType = CreateDIType(type);
143             }
144             fwdDeclaration->replaceAllUsesWith(diType);
145         }
146         currentFwdDeclarationMap.clear();
147         std::swap(currentFwdDeclarationMap, fwdDeclarationMap);
148     }
149 }
