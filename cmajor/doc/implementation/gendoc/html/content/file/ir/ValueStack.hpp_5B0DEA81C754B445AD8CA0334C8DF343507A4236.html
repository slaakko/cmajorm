 6 #ifndef CMAJOR_IR_VALUE_STACK_INCLUDED
 7 #define CMAJOR_IR_VALUE_STACK_INCLUDED
 8 #include <cmajor/util/Error.hpp>
 9 #include <llvm/IR/Value.h>
10 #include <vector>
11 
12 namespace cmajor { namespace ir {
13 
14 class ValueStack 
15 {
16 public:
17     void Push(llvm::Value* value)
18     {
19         s.push_back(value);
20     }
21     llvm::Value* Pop()
22     {
23         Assert(!s.empty(), "value stack is empty");
24         llvm::Value* top = s.back();
25         s.pop_back();
26         return top;
27     }
28     void Dup()
29     {
30         s.push_back(s.back());
31     }
32     void Swap()
33     {
34         std::swap(s.back(), s[s.size() - 2]);
35     }
36     void Rotate()
37     {
38         std::swap(s[s.size() - 3], s[s.size() - 2]);
39         std::swap(s.back(), s[s.size() - 2]);
40     }
41 private:
42     std::vector<llvm::Value*> s;
43 };
44 
45 } } // namespace cmajor::ir
