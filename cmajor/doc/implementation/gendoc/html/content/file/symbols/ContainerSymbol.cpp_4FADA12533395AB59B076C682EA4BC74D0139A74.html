 25 void ContainerSymbol::Write(SymbolWriter& writer)
 26 {
 27     Symbol::Write(writer);
 28     std::vector<Symbol*> exportSymbols;
 29     for (const std::unique_ptr<Symbol>& member : members)
 30     {
 31         if (member->IsExportSymbol())
 32         {
 33             exportSymbols.push_back(member.get());
 34         }
 35     }
 36     uint32_t n = uint32_t(exportSymbols.size());
 37     writer.GetBinaryWriter().WriteULEB128UInt(n);
 38     for (uint32_t i = 0; i < n; ++i)
 39     {
 40         writer.Write(exportSymbols[i]);
 41     }
 42 }

 44 void ContainerSymbol::Read(SymbolReader& reader)
 45 {
 46     Symbol::Read(reader);
 47     uint32_t n = reader.GetBinaryReader().ReadULEB128UInt();
 48     for (uint32_t i = 0; i < n; ++i)
 49     {
 50         Symbol* symbol = reader.ReadSymbol(this);
 51         AddMember(symbol);
 52     }
 53 }

 55 void ContainerSymbol::AddMember(Symbol* member)
 56 {
 57     member->SetSymbolTable(GetSymbolTable());
 58     if (GetModule())
 59     {
 60         member->SetModule(GetModule());
 61     }
 62     if (!member->GetOriginalModule() && GetOriginalModule())
 63     {
 64         member->SetOriginalModule(GetOriginalModule());
 65     }
 66     member->SetParent(this);
 67     members.push_back(     std::unique_ptr<Symbol>(member));
 68     if (member->IsFunctionSymbol())
 69     {
 70         FunctionSymbol* functionSymbol = static_cast<FunctionSymbol*>(member);
 71         FunctionGroupSymbol* functionGroupSymbol = MakeFunctionGroupSymbol(functionSymbol->GroupName(), functionSymbol->GetSpan());
 72         functionGroupSymbol->AddFunction(functionSymbol);
 73         functionSymbol->GetContainerScope()->SetParent(GetContainerScope());
 74         functionIndexMap[functionSymbol->GetIndex()] = functionSymbol;
 75     }
 76     else if (member->GetSymbolType() == SymbolType::conceptSymbol)
 77     {
 78         ConceptSymbol* conceptSymbol = static_cast<ConceptSymbol*>(member);
 79         ConceptGroupSymbol* conceptGroupSymbol = MakeConceptGroupSymbol(conceptSymbol->GroupName(), conceptSymbol->GetSpan());
 80         conceptGroupSymbol->AddConcept(conceptSymbol);
 81         conceptSymbol->GetContainerScope()->SetParent(GetContainerScope());
 82     }
 83     else if (member->GetSymbolType() == SymbolType::classTypeSymbol || member->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
 84     {
 85         ClassTypeSymbol* classTypeSymbol = static_cast<ClassTypeSymbol*>(member);
 86         ClassGroupTypeSymbol* classGroupTypeSymbol = MakeClassGroupTypeSymbol(classTypeSymbol->GroupName(), classTypeSymbol->GetSpan());
 87         classGroupTypeSymbol->AddClass(classTypeSymbol);
 88         classTypeSymbol->GetContainerScope()->SetParent(GetContainerScope());
 89     }
 90     else
 91     {
 92         containerScope.Install(member);
 93     }
 94 }

128 FunctionGroupSymbol* ContainerSymbol::MakeFunctionGroupSymbol(const std::u32string& groupName, const Span& span)
129 {
130     Symbol* symbol = containerScope.Lookup(groupName);
131     if (!symbol)
132     {
133         FunctionGroupSymbol* functionGroupSymbol = new FunctionGroupSymbol(span, groupName);
134         functionGroupSymbol->SetSymbolTable(GetSymbolTable());
135         functionGroupSymbol->SetModule(GetModule());
136         functionGroupSymbol->SetOriginalModule(GetOriginalModule());
137         AddMember(functionGroupSymbol);
138         return functionGroupSymbol;
139     }
140     if (symbol->GetSymbolType() == SymbolType::functionGroupSymbol)
141     {
142         return static_cast<FunctionGroupSymbol*>(symbol);
143     }
144     else
145     {
146         throw Exception(GetModule(), "name of symbol '" + ToUtf8(symbol->FullName()) + "' conflicts with a function group '" + ToUtf8(groupName) + "'", symbol->GetSpan(), span);
147     }
148 }

172 ClassGroupTypeSymbol* ContainerSymbol::MakeClassGroupTypeSymbol(const std::u32string& groupName, const Span& span)
173 {
174     Symbol* symbol = containerScope.Lookup(groupName);
175     if (!symbol)
176     {
177         ClassGroupTypeSymbol* classGroupTypeSymbol = new ClassGroupTypeSymbol(span, groupName);
178         classGroupTypeSymbol->SetSymbolTable(GetSymbolTable());
179         classGroupTypeSymbol->SetModule(GetModule());
180         classGroupTypeSymbol->SetOriginalModule(GetOriginalModule());
181         GetSymbolTable()->SetTypeIdFor(classGroupTypeSymbol);
182         AddMember(classGroupTypeSymbol);
183         return classGroupTypeSymbol;
184     }
185     if (symbol->GetSymbolType() == SymbolType::classGroupTypeSymbol)
186     {
187         return static_cast<ClassGroupTypeSymbol*>(symbol);
188     }
189     else
190     {
191         throw Exception(GetModule(), "name of symbol '" + ToUtf8(symbol->FullName()) + "' conflicts with a class group '" + ToUtf8(groupName) + "'", symbol->GetSpan(), span);
192     }

195 void ContainerSymbol::AppendChildElements(dom::Element* element, TypeMap& typeMap) const
196 {
197     for (const std::unique_ptr<Symbol>& member : members)
198     {
199         if (member->IsFunctionSymbol()) continue;
200         if (member->IsClassTypeSymbol()) continue;
201         if (member->GetSymbolType() == SymbolType::conceptSymbol) continue;
202         if (member->GetSymbolType() != SymbolType::namespaceSymbol && !member->IsProject()) continue;
203         if (member->GetSymbolType() == SymbolType::namespaceSymbol || 
204             member->GetSymbolType() == SymbolType::classGroupTypeSymbol || 
205             member->GetSymbolType() == SymbolType::functionGroupSymbol || 
206             member->GetSymbolType() == SymbolType::conceptGroupSymbol)
207         {
208             if (!member->HasProjectMembers()) continue;
209         }
210         std::unique_ptr<dom::Element> memberElement = member->ToDomElement(typeMap);
211         if (memberElement)
212         {
213             element->AppendChild(     std::unique_ptr<dom::Node>(memberElement.release()));
214         }
215     }

217 bool ContainerSymbol::HasProjectMembers() const
218 {
219     if (FullName() == U"System.Meta") return false;
220     for (const std::unique_ptr<Symbol>& member : members)
221     {
222         if (member->GetSymbolType() == SymbolType::namespaceSymbol || 
223             member->GetSymbolType() == SymbolType::classGroupTypeSymbol || 
224             member->GetSymbolType() == SymbolType::functionGroupSymbol || 
225             member->GetSymbolType() == SymbolType::conceptGroupSymbol)
226         {
227             if (member->HasProjectMembers()) return true;
228         }
229         else
230         {
231             if (member->IsFunctionSymbol())
232             {
233                 FunctionSymbol* fun = static_cast<FunctionSymbol*>(member.get());
234                 if (fun->IsTemplateSpecialization()) continue;
235             }
236             if (member->IsProject()) return true;
237         }
238     }
239     return false;
240 }
