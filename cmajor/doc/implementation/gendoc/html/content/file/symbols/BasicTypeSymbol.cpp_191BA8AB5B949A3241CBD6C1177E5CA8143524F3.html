 19 BasicTypeSymbol::BasicTypeSymbol(SymbolType symbolType_, const Span& span_, const std::u32string& name_) :
 20     TypeSymbol(symbolType_, span_, name_), 
 21     defaultConstructor(nullptr), copyConstructor(nullptr), moveConstructor(nullptr), copyAssignment(nullptr), moveAssignment(nullptr), returnFun(nullptr), equalityOp(nullptr)
 22 {
 23 }
 24 
 25 void BasicTypeSymbol::Write(SymbolWriter& writer)
 26 {
 27     TypeSymbol::Write(writer);
 28     boost::uuids::uuid defaultConstructorId = boost::uuids::nil_generator()();
 29     if (defaultConstructor)
 30     {
 31         defaultConstructorId = defaultConstructor->FunctionId();
 32     }
 33     writer.GetBinaryWriter().Write(defaultConstructorId);
 34     boost::uuids::uuid copyConstructorId = boost::uuids::nil_generator()();
 35     if (copyConstructor)
 36     {
 37         copyConstructorId = copyConstructor->FunctionId();
 38     }
 39     writer.GetBinaryWriter().Write(copyConstructorId);
 40     boost::uuids::uuid moveConstructorId = boost::uuids::nil_generator()();
 41     if (moveConstructor)
 42     {
 43         moveConstructorId = moveConstructor->FunctionId();
 44     }
 45     writer.GetBinaryWriter().Write(moveConstructorId);
 46     boost::uuids::uuid copyAssignmentId = boost::uuids::nil_generator()();
 47     if (copyAssignment)
 48     {
 49         copyAssignmentId = copyAssignment->FunctionId();
 50     }
 51     writer.GetBinaryWriter().Write(copyAssignmentId);
 52     boost::uuids::uuid moveAssignmentId = boost::uuids::nil_generator()();
 53     if (moveAssignment)
 54     {
 55         moveAssignmentId = moveAssignment->FunctionId();
 56     }
 57     writer.GetBinaryWriter().Write(moveAssignmentId);
 58     boost::uuids::uuid returnId = boost::uuids::nil_generator()();
 59     if (returnFun)
 60     {
 61         returnId = returnFun->FunctionId();
 62     }
 63     writer.GetBinaryWriter().Write(returnId);
 64     boost::uuids::uuid equalityOpId = boost::uuids::nil_generator()();
 65     if (equalityOp)
 66     {
 67         equalityOpId = equalityOp->FunctionId();
 68     }
 69     writer.GetBinaryWriter().Write(equalityOpId);
 70 }

 72 void BasicTypeSymbol::Read(SymbolReader& reader)
 73 {
 74     TypeSymbol::Read(reader);
 75     boost::uuids::uuid defaultConstructorId;
 76     reader.GetBinaryReader().ReadUuid(defaultConstructorId);
 77     if (!defaultConstructorId.is_nil())
 78     {
 79         GetSymbolTable()->EmplaceFunctionRequest(this, defaultConstructorId, 0);
 80     }
 81     boost::uuids::uuid copyConstructorId;
 82     reader.GetBinaryReader().ReadUuid(copyConstructorId);
 83     if (!copyConstructorId.is_nil())
 84     {
 85         GetSymbolTable()->EmplaceFunctionRequest(this, copyConstructorId, 1);
 86     }
 87     boost::uuids::uuid moveConstructorId;
 88     reader.GetBinaryReader().ReadUuid(moveConstructorId);
 89     if (!moveConstructorId.is_nil())
 90     {
 91         GetSymbolTable()->EmplaceFunctionRequest(this, moveConstructorId, 2);
 92     }
 93     boost::uuids::uuid copyAssignmentId;
 94     reader.GetBinaryReader().ReadUuid(copyAssignmentId);
 95     if (!copyAssignmentId.is_nil())
 96     {
 97         GetSymbolTable()->EmplaceFunctionRequest(this, copyAssignmentId, 3);
 98     }
 99     boost::uuids::uuid moveAssignmentId;
100     reader.GetBinaryReader().ReadUuid(moveAssignmentId);
101     if (!moveAssignmentId.is_nil())
102     {
103         GetSymbolTable()->EmplaceFunctionRequest(this, moveAssignmentId, 4);
104     }
105     boost::uuids::uuid returnId;
106     reader.GetBinaryReader().ReadUuid(returnId);
107     if (!returnId.is_nil())
108     {
109         GetSymbolTable()->EmplaceFunctionRequest(this, returnId, 5);
110     }
111     boost::uuids::uuid equalityOpId;
112     reader.GetBinaryReader().ReadUuid(equalityOpId);
113     if (!equalityOpId.is_nil())
114     {
115         GetSymbolTable()->EmplaceFunctionRequest(this, equalityOpId, 6);
116     }
117 }

119 void BasicTypeSymbol::EmplaceFunction(FunctionSymbol* functionSymbol, int index)
120 {
121     switch (index)
122     {
123         case 0: defaultConstructor = functionSymbol; break;
124         case 1: copyConstructor = functionSymbol; break;
125         case 2: moveConstructor = functionSymbol; break;
126         case 3: copyAssignment = functionSymbol; break;
127         case 4: moveAssignment = functionSymbol; break;
128         case 5: returnFun = functionSymbol; break;
129         case 6: equalityOp = functionSymbol; break;
130         default:
131         {
132             Assert(false, "invalid emplace function index");
133         }
134     }
135 }
