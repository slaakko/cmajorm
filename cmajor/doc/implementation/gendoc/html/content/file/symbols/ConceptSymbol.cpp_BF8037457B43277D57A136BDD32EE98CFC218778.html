 24 void ConceptGroupSymbol::AddConcept(ConceptSymbol* concept)
 25 {
 26     Assert(concept->GroupName() == Name(), "wrong concept group");
 27     int arity = concept->Arity();
 28     auto it = arityConceptMap.find(arity);
 29     if (it != arityConceptMap.cend())
 30     {
 31         throw Exception(GetModule(), "concept group '" + ToUtf8(FullName()) + "' already has concept with arity " + std::to_string(arity), GetSpan());
 32     }
 33     arityConceptMap[arity] = concept;
 34 }

 36 ConceptSymbol* ConceptGroupSymbol::GetConcept(int arity)
 37 {
 38     auto it = arityConceptMap.find(arity);
 39     if (it != arityConceptMap.cend())
 40     {
 41         return it->second;
 42     }
 43     else
 44     {
 45         throw Exception(GetModule(), "concept with arity " + std::to_string(arity) + " not found from concept group '" + ToUtf8(FullName()) + "'", GetSpan());
 46     }
 47 }

 49 bool ConceptGroupSymbol::HasProjectMembers() const
 50 {
 51     for (const auto& p : arityConceptMap)
 52     {
 53         ConceptSymbol* concept = p.second;
 54         if (concept->IsProject())
 55         {
 56             return true;
 57         }
 58     }
 59     return false;

 80 void ConceptSymbol::Write(SymbolWriter& writer)
 81 {
 82     ContainerSymbol::Write(writer);
 83     Assert(!typeId.is_nil(), "type id not initialized");
 84     writer.GetBinaryWriter().Write(typeId);
 85     writer.GetBinaryWriter().Write(groupName);
 86     boost::uuids::uuid refineConceptId = boost::uuids::nil_generator()();
 87     if (refinedConcept)
 88     {
 89         refineConceptId = refinedConcept->TypeId();
 90     }
 91     writer.GetBinaryWriter().Write(refineConceptId);
 92     uint32_t n = templateParameters.size();
 93     writer.GetBinaryWriter().WriteULEB128UInt(n);
 94     for (uint32_t i = 0; i < n; ++i)
 95     {
 96         TemplateParameterSymbol* templateParameter = templateParameters[i];
 97         Assert(!templateParameter->TypeId().is_nil(), "type id not initialized");
 98         writer.GetBinaryWriter().Write(templateParameter->TypeId());
 99     }
100     Node* node = GetSymbolTable()->GetNode(this);
101     Assert(node->IsConceptNode(), "concept node expected");
102     writer.GetAstWriter().Write(node);
103     writer.GetBinaryWriter().Write(hasSource);
104 }

106 void ConceptSymbol::Read(SymbolReader& reader)
107 {
108     ContainerSymbol::Read(reader);
109     reader.GetBinaryReader().ReadUuid(typeId);
110     GetSymbolTable()->AddTypeOrConceptSymbolToTypeIdMap(this);
111     groupName = reader.GetBinaryReader().ReadUtf32String();
112     boost::uuids::uuid refinedConcepId;
113     reader.GetBinaryReader().ReadUuid(refinedConcepId);
114     if (!refinedConcepId.is_nil())
115     {
116         GetSymbolTable()->EmplaceConceptRequest(this, refinedConcepId);
117     }
118     uint32_t n = reader.GetBinaryReader().ReadULEB128UInt();
119     templateParameters.resize(n);
120     for (uint32_t i = 0; i < n; ++i)
121     {
122         boost::uuids::uuid templateParameterId;
123         reader.GetBinaryReader().ReadUuid(templateParameterId);
124         GetSymbolTable()->EmplaceTypeRequest(this, templateParameterId, i);
125     }
126     conceptNode.reset(reader.GetAstReader().ReadConceptNode());
127     hasSource = reader.GetBinaryReader().ReadBool();
128 }

135 void ConceptSymbol::EmplaceType(TypeSymbol* typeSymbol, int index)
136 {
137     if (index >= 0 && index < templateParameters.size())
138     {
139         if (typeSymbol->GetSymbolType() == SymbolType::templateParameterSymbol)
140         {
141             TemplateParameterSymbol* templateParameter = static_cast<TemplateParameterSymbol*>(typeSymbol);
142             templateParameters[index] = templateParameter;
143         }
144         else
145         {
146             throw Exception(GetModule(), "invalid emplace type", GetSpan());
147         }
148     }
149     else
150     {
151         throw Exception(GetModule(), "invalid emplace type index", GetSpan());
152     }
153 }

185 void ConceptSymbol::ComputeName()
186 {
187     std::u32string name = groupName;
188     bool first = true;
189     name.append(1, '<');
190     for (TemplateParameterSymbol* templateParameter : templateParameters)
191     {
192         if (first)
193         {
194             first = false;
195         }
196         else
197         {
198             name.append(U", ");
199         }
200         name.append(templateParameter->Name());
201     }
202     name.append(1, '>');
203     SetName(name);
204     ComputeMangledName();
205 }

207 void ConceptSymbol::SetSpecifiers(Specifiers specifiers)
208 {
209     Specifiers accessSpecifiers = specifiers & Specifiers::access_;
210     SetAccess(accessSpecifiers);
211     if ((specifiers & Specifiers::static_) != Specifiers::none)
212     {
213         throw Exception(GetModule(), "concept symbol cannot be static", GetSpan());
214     }
215     if ((specifiers & Specifiers::virtual_) != Specifiers::none)
216     {
217         throw Exception(GetModule(), "concept symbol cannot be virtual", GetSpan());
218     }
219     if ((specifiers & Specifiers::override_) != Specifiers::none)
220     {
221         throw Exception(GetModule(), "concept symbol cannot be override", GetSpan());
222     }
223     if ((specifiers & Specifiers::abstract_) != Specifiers::none)
224     {
225         throw Exception(GetModule(), "concept symbol cannot be abstract", GetSpan());
226     }
227     if ((specifiers & Specifiers::inline_) != Specifiers::none)
228     {
229         throw Exception(GetModule(), "concept symbol cannot be inline", GetSpan());
230     }
231     if ((specifiers & Specifiers::explicit_) != Specifiers::none)
232     {
233         throw Exception(GetModule(), "concept symbol cannot be explicit", GetSpan());
234     }
235     if ((specifiers & Specifiers::external_) != Specifiers::none)
236     {
237         throw Exception(GetModule(), "concept symbol cannot be external", GetSpan());
238     }
239     if ((specifiers & Specifiers::suppress_) != Specifiers::none)
240     {
241         throw Exception(GetModule(), "concept symbol cannot be suppressed", GetSpan());
242     }
243     if ((specifiers & Specifiers::default_) != Specifiers::none)
244     {
245         throw Exception(GetModule(), "concept symbol cannot be default", GetSpan());
246     }
247     if ((specifiers & Specifiers::constexpr_) != Specifiers::none)
248     {
249         throw Exception(GetModule(), "concept symbol cannot be constexpr", GetSpan());
250     }
251     if ((specifiers & Specifiers::cdecl_) != Specifiers::none)
252     {
253         throw Exception(GetModule(), "concept symbol cannot be decl", GetSpan());
254     }
255     if ((specifiers & Specifiers::nothrow_) != Specifiers::none)
256     {
257         throw Exception(GetModule(), "concept symbol cannot be nothrow", GetSpan());
258     }
259     if ((specifiers & Specifiers::throw_) != Specifiers::none)
260     {
261         throw Exception(GetModule(), "concept symbol cannot be throw", GetSpan());
262     }
263     if ((specifiers & Specifiers::new_) != Specifiers::none)
264     {
265         throw Exception(GetModule(), "concept symbol cannot be new", GetSpan());
266     }
267     if ((specifiers & Specifiers::const_) != Specifiers::none)
268     {
269         throw Exception(GetModule(), "concept symbol cannot be const", GetSpan());
270     }
271     if ((specifiers & Specifiers::unit_test_) != Specifiers::none)
272     {
273         throw Exception(GetModule(), "concept symbol cannot be unit_test", GetSpan());
274     }
275 }
