 35 void SymbolCreatorVisitor::Visit(NamespaceNode& namespaceNode)
 36 {
 37     symbolTable.BeginNamespace(namespaceNode);
 38     NodeList<Node>& members = namespaceNode.Members();
 39     int n = members.Count();
 40     for (int i = 0; i < n; ++i)
 41     {
 42         Node* member = members[i];
 43         member->Accept(*this);
 44     }
 45     symbolTable.EndNamespace();
 46 }

 48 void SymbolCreatorVisitor::Visit(FunctionNode& functionNode)
 49 {
 50     symbolTable.BeginFunction(functionNode, functionIndex++);
 51     int nt = functionNode.TemplateParameters().Count();
 52     for (int i = 0; i < nt; ++i)
 53     {
 54         symbolTable.AddTemplateParameter(*functionNode.TemplateParameters()[i]);
 55     }
 56     int n = functionNode.Parameters().Count();
 57     for (int i = 0; i < n; ++i)
 58     {
 59         ParameterNode* parameterNode = functionNode.Parameters()[i];
 60         parameterNode->Accept(*this);
 61     }
 62     if (nt == 0)
 63     {
 64         if (functionNode.Body())
 65         {
 66             functionNode.Body()->Accept(*this);
 67         }
 68     }
 69     symbolTable.EndFunction();
 70 }

 77 void SymbolCreatorVisitor::Visit(ClassNode& classNode)
 78 {
 79     if (&classNode == classInstanceNode)
 80     {
 81         symbolTable.BeginClassTemplateSpecialization(*classInstanceNode, classTemplateSpecialization);
 82     }
 83     else
 84     {
 85         symbolTable.BeginClass(classNode);
 86     }
 87     int nt = classNode.TemplateParameters().Count();
 88     for (int i = 0; i < nt; ++i)
 89     {
 90         symbolTable.AddTemplateParameter(*classNode.TemplateParameters()[i]);
 91     }
 92     if (nt == 0)
 93     {
 94         int n = classNode.Members().Count();
 95         for (int i = 0; i < n; ++i)
 96         {
 97             Node* member = classNode.Members()[i];
 98             member->Accept(*this);
 99         }
100     }
101     if (&classNode == classInstanceNode)
102     {
103         symbolTable.EndClassTemplateSpecialization();
104     }
105     else
106     {
107         symbolTable.EndClass();
108     }
109 }

331 void SymbolCreatorVisitor::Visit(ConditionalCompilationDisjunctionNode& conditionalCompilationDisjunctionNode)
332 {
333     conditionalCompilationDisjunctionNode.Left()->Accept(*this);
334     bool left = conditionalCompilationStack.top();
335     conditionalCompilationStack.pop();
336     conditionalCompilationDisjunctionNode.Right()->Accept(*this);
337     bool right = conditionalCompilationStack.top();
338     conditionalCompilationStack.pop();
339     conditionalCompilationStack.push(left || right);
340 }

367 void SymbolCreatorVisitor::Visit(ConditionalCompilationStatementNode& conditionalCompilationStatementNode)
368 {
369     conditionalCompilationStatementNode.IfPart()->Accept(*this);
370     bool defined = conditionalCompilationStack.top();
371     conditionalCompilationStack.pop();
372     if (defined)
373     {
374         int n = conditionalCompilationStatementNode.IfPart()->Statements().Count();
375         for (int i = 0; i < n; ++i)
376         {
377             StatementNode* statement = conditionalCompilationStatementNode.IfPart()->Statements()[i];
378             statement->Accept(*this);
379         }
380     }
381     else
382     {
383         bool executed = false;
384         int n = conditionalCompilationStatementNode.ElifParts().Count();
385         for (int i = 0; i < n; ++i)
386         {
387             ConditionalCompilationPartNode* elifPart = conditionalCompilationStatementNode.ElifParts()[i];
388             elifPart->Accept(*this);
389             bool defined = conditionalCompilationStack.top();
390             conditionalCompilationStack.pop();
391             if (defined)
392             {
393                 int n = elifPart->Statements().Count();
394                 for (int i = 0; i < n; ++i)
395                 {
396                     StatementNode* statement = elifPart->Statements()[i];
397                     statement->Accept(*this);
398                 }
399                 executed = true;
400                 break;
401             }
402         }
403         if (!executed)
404         {
405             ConditionalCompilationPartNode* elsePart = conditionalCompilationStatementNode.ElsePart();
406             if (elsePart)
407             {
408                 int n = elsePart->Statements().Count();
409                 for (int i = 0; i < n; ++i)
410                 {
411                     StatementNode* statement = elsePart->Statements()[i];
412                     statement->Accept(*this);
413                 }
414             }
415         }
416     }
417 }
