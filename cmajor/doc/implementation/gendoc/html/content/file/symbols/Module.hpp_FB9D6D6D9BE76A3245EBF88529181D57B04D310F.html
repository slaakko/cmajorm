 33 enum class ModuleFlags : uint8_t 
 34 {
 35     none = 0, system = 1 << 0
 36 };
 37 
 38 inline ModuleFlags operator|(ModuleFlags left, ModuleFlags right)
 39 {
 40     return ModuleFlags(uint8_t(left) | uint8_t(right));
 41 }

 43 inline ModuleFlags operator&(ModuleFlags left, ModuleFlags right)
 44 {
 45     return ModuleFlags(uint8_t(left) & uint8_t(right));
