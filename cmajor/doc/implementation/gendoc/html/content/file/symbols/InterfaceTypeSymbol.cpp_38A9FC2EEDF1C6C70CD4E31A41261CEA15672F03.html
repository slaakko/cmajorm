 40 void InterfaceTypeSymbol::SetSpecifiers(Specifiers specifiers)
 41 {
 42     Specifiers accessSpecifiers = specifiers & Specifiers::access_;
 43     SetAccess(accessSpecifiers);
 44     if ((specifiers & Specifiers::static_) != Specifiers::none)
 45     {
 46         throw Exception(GetModule(), "interface cannot be static", GetSpan());
 47     }
 48     if ((specifiers & Specifiers::virtual_) != Specifiers::none)
 49     {
 50         throw Exception(GetModule(), "interface cannot be virtual", GetSpan());
 51     }
 52     if ((specifiers & Specifiers::override_) != Specifiers::none)
 53     {
 54         throw Exception(GetModule(), "interface cannot be override", GetSpan());
 55     }
 56     if ((specifiers & Specifiers::abstract_) != Specifiers::none)
 57     {
 58         throw Exception(GetModule(), "interface cannot be abstract", GetSpan());
 59     }
 60     if ((specifiers & Specifiers::inline_) != Specifiers::none)
 61     {
 62         throw Exception(GetModule(), "interface cannot be inline", GetSpan());
 63     }
 64     if ((specifiers & Specifiers::explicit_) != Specifiers::none)
 65     {
 66         throw Exception(GetModule(), "interface cannot be explicit", GetSpan());
 67     }
 68     if ((specifiers & Specifiers::external_) != Specifiers::none)
 69     {
 70         throw Exception(GetModule(), "interface cannot be external", GetSpan());
 71     }
 72     if ((specifiers & Specifiers::suppress_) != Specifiers::none)
 73     {
 74         throw Exception(GetModule(), "interface cannot be suppressed", GetSpan());
 75     }
 76     if ((specifiers & Specifiers::default_) != Specifiers::none)
 77     {
 78         throw Exception(GetModule(), "interface cannot be default", GetSpan());
 79     }
 80     if ((specifiers & Specifiers::constexpr_) != Specifiers::none)
 81     {
 82         throw Exception(GetModule(), "interface cannot be constexpr", GetSpan());
 83     }
 84     if ((specifiers & Specifiers::cdecl_) != Specifiers::none)
 85     {
 86         throw Exception(GetModule(), "interface cannot be cdecl", GetSpan());
 87     }
 88     if ((specifiers & Specifiers::nothrow_) != Specifiers::none)
 89     {
 90         throw Exception(GetModule(), "interface cannot be nothrow", GetSpan());
 91     }
 92     if ((specifiers & Specifiers::throw_) != Specifiers::none)
 93     {
 94         throw Exception(GetModule(), "interface cannot be throw", GetSpan());
 95     }
 96     if ((specifiers & Specifiers::new_) != Specifiers::none)
 97     {
 98         throw Exception(GetModule(), "interface cannot be new", GetSpan());
 99     }
100     if ((specifiers & Specifiers::const_) != Specifiers::none)
101     {
102         throw Exception(GetModule(), "interface cannot be const", GetSpan());
103     }
104     if ((specifiers & Specifiers::unit_test_) != Specifiers::none)
105     {
106         throw Exception(GetModule(), "interface cannot be unit_test", GetSpan());

131 void InterfaceTypeSymbol::GenerateCall(Emitter& emitter, std::vector<GenObject*>& genObjects, OperationFlags flags, MemberFunctionSymbol* interfaceMemberFunction, const Span& span)
132 {
133     TypeSymbol* type = static_cast<TypeSymbol*>(genObjects[0]->GetType());
134     if (type->GetSymbolType() == SymbolType::interfaceTypeSymbol)
135     {
136         genObjects[0]->Load(emitter, OperationFlags::addr);
137     }
138     else
139     {
140         genObjects[0]->Load(emitter, OperationFlags::none);
141     }
142     llvm::Value* interfaceTypePtr = emitter.Stack().Pop();
143     ArgVector objectIndeces;
144     objectIndeces.push_back(emitter.Builder().getInt32(0));
145     objectIndeces.push_back(emitter.Builder().getInt32(0));
146     llvm::Value* objectPtrPtr = emitter.Builder().CreateGEP(interfaceTypePtr, objectIndeces);
147     llvm::Value* objectPtr = emitter.Builder().CreateLoad(objectPtrPtr);
148     ArgVector interfaceIndeces;
149     interfaceIndeces.push_back(emitter.Builder().getInt32(0));
150     interfaceIndeces.push_back(emitter.Builder().getInt32(1));
151     llvm::Value* interfacePtrPtr = emitter.Builder().CreateGEP(interfaceTypePtr, interfaceIndeces);
152     llvm::Value* interfacePtr = emitter.Builder().CreateLoad(interfacePtrPtr);
153     llvm::Value* imtPtr = emitter.Builder().CreateBitCast(interfacePtr, llvm::PointerType::get(emitter.Builder().getInt8PtrTy(), 0));
154     ArgVector methodIndeces;
155     methodIndeces.push_back(emitter.Builder().getInt32(interfaceMemberFunction->ImtIndex()));
156     llvm::Value* methodPtrPtr = emitter.Builder().CreateGEP(imtPtr, methodIndeces);
157     llvm::Value* methodPtr = emitter.Builder().CreateLoad(methodPtrPtr);
158     llvm::Value* callee = emitter.Builder().CreateBitCast(methodPtr, llvm::PointerType::get(interfaceMemberFunction->IrType(emitter), 0));
159     int na = genObjects.size();
160     for (int i = 1; i < na; ++i)
161     {
162         GenObject* genObject = genObjects[i];
163         genObject->Load(emitter, flags & OperationFlags::functionCallFlags);
164     }
165     emitter.SetCurrentDebugLocation(span);
166     ArgVector args;
167     int n = interfaceMemberFunction->Parameters().size();
168     if (interfaceMemberFunction->ReturnsClassInterfaceOrClassDelegateByValue())
169     {
170         ++n;
171     }
172     args.resize(n);
173     args[0] = objectPtr;
174     for (int i = 0; i < n - 1; ++i)
175     {
176         llvm::Value* arg = emitter.Stack().Pop();
177         args[n - i - 1] = arg;
178     }
179     llvm::BasicBlock* handlerBlock = emitter.HandlerBlock();
180     llvm::BasicBlock* cleanupBlock = emitter.CleanupBlock();
181     bool newCleanupNeeded = emitter.NewCleanupNeeded();
182     Pad* currentPad = emitter.CurrentPad();
183     std::vector<llvm::OperandBundleDef> bundles;
184     if (currentPad != nullptr)
185     {
186         std::vector<llvm::Value*> inputs;
187         inputs.push_back(currentPad->value);
188         bundles.push_back(llvm::OperandBundleDef("funclet", inputs));
189     }
190     if (interfaceMemberFunction->ReturnType()->GetSymbolType() != SymbolType::voidTypeSymbol && !interfaceMemberFunction->ReturnsClassInterfaceOrClassDelegateByValue())
191     {
192         if (IsNothrow() || (!handlerBlock && !cleanupBlock && !newCleanupNeeded))
193         {
194             if (currentPad == nullptr)
195             {
196                 emitter.Stack().Push(emitter.Builder().CreateCall(callee, args));
197             }
198             else
199             {
200                 llvm::CallInst* callInst = llvm::CallInst::Create(callee, args, bundles, "", emitter.CurrentBasicBlock());
201                 if (emitter.DIBuilder())
202                 {
203                     callInst->setDebugLoc(emitter.GetDebugLocation(span));
204                 }
205                 emitter.Stack().Push(callInst);
206             }
207         }
208         else
209         {
210             llvm::BasicBlock* nextBlock = llvm::BasicBlock::Create(emitter.Context(), "next", emitter.Function());
211             if (newCleanupNeeded)
212             {
213                 emitter.CreateCleanup();
214                 cleanupBlock = emitter.CleanupBlock();
215             }
216             llvm::BasicBlock* unwindBlock = cleanupBlock;
217             if (unwindBlock == nullptr)
218             {
219                 unwindBlock = handlerBlock;
220                 Assert(unwindBlock, "no unwind block");
221             }
222             if (currentPad == nullptr)
223             {
224                 emitter.Stack().Push(emitter.Builder().CreateInvoke(callee, nextBlock, unwindBlock, args));
225             }
226             else
227             {
228                 llvm::InvokeInst* invokeInst = llvm::InvokeInst::Create(callee, nextBlock, unwindBlock, args, bundles, "", emitter.CurrentBasicBlock());
229                 if (emitter.DIBuilder())
230                 {
231                     invokeInst->setDebugLoc(emitter.GetDebugLocation(span));
232                 }
233                 emitter.Stack().Push(invokeInst);
234             }
235             emitter.SetCurrentBasicBlock(nextBlock);
236         }
237     }
238     else
239     {
240         if (IsNothrow() || (!handlerBlock && !cleanupBlock && !newCleanupNeeded))
241         {
242             if (currentPad == nullptr)
243             {
244                 emitter.Builder().CreateCall(callee, args);
245             }
246             else
247             {
248                 llvm::CallInst* callInst = llvm::CallInst::Create(callee, args, bundles, "", emitter.CurrentBasicBlock());
249                 if (emitter.DIBuilder())
250                 {
251                     callInst->setDebugLoc(emitter.GetDebugLocation(span));
252                 }
253             }
254         }
255         else
256         {
257             llvm::BasicBlock* nextBlock = llvm::BasicBlock::Create(emitter.Context(), "next", emitter.Function());
258             if (newCleanupNeeded)
259             {
260                 emitter.CreateCleanup();
261                 cleanupBlock = emitter.CleanupBlock();
262             }
263             llvm::BasicBlock* unwindBlock = cleanupBlock;
264             if (unwindBlock == nullptr)
265             {
266                 unwindBlock = handlerBlock;
267                 Assert(unwindBlock, "no unwind block");
268             }
269             if (currentPad == nullptr)
270             {
271                 emitter.Builder().CreateInvoke(callee, nextBlock, unwindBlock, args);
272             }
273             else
274             {
275                 llvm::InvokeInst* invokeInst = llvm::InvokeInst::Create(callee, nextBlock, unwindBlock, args, bundles, "", emitter.CurrentBasicBlock());
276                 if (emitter.DIBuilder())
277                 {
278                     invokeInst->setDebugLoc(emitter.GetDebugLocation(span));
279                 }
280             }
281             emitter.SetCurrentBasicBlock(nextBlock);
282         }
283     }
284 }
