 25 void ConstantSymbol::Write(SymbolWriter& writer)
 26 {
 27     Symbol::Write(writer);
 28     writer.GetBinaryWriter().Write(type->TypeId());
 29     bool hasComplexValue = value->IsComplexValue();
 30     writer.GetBinaryWriter().Write(hasComplexValue);
 31     if (hasComplexValue)
 32     {
 33         uint32_t sizePos = writer.GetBinaryWriter().Pos();
 34         uint32_t sizeOfValue = 0;
 35         writer.GetBinaryWriter().Write(sizeOfValue);
 36         uint32_t startPos = writer.GetBinaryWriter().Pos();
 37         value->Write(writer.GetBinaryWriter());
 38         uint32_t endPos = writer.GetBinaryWriter().Pos();
 39         sizeOfValue = endPos - startPos;
 40         writer.GetBinaryWriter().Seek(sizePos);
 41         writer.GetBinaryWriter().Write(sizeOfValue);
 42         writer.GetBinaryWriter().Seek(endPos);
 43     }
 44     else
 45     {
 46         WriteValue(value.get(), writer.GetBinaryWriter());
 47     }
 48     writer.GetBinaryWriter().Write(strValue);
 49 }

 51 void ConstantSymbol::Read(SymbolReader& reader)
 52 {
 53     Symbol::Read(reader);
 54     boost::uuids::uuid typeId;
 55     reader.GetBinaryReader().ReadUuid(typeId);
 56     GetSymbolTable()->EmplaceTypeRequest(this, typeId, 0);
 57     bool hasComplexValue = reader.GetBinaryReader().ReadBool();
 58     if (hasComplexValue)
 59     {
 60         sizeOfValue = reader.GetBinaryReader().ReadUInt();
 61         valuePos = reader.GetBinaryReader().Pos();
 62         reader.GetBinaryReader().Skip(sizeOfValue);
 63         filePathReadFrom = reader.GetBinaryReader().FileName();
 64     }
 65     else
 66     {
 67         value = ReadValue(reader.GetBinaryReader(), GetSpan(), GetModule());
 68     }
 69     strValue = reader.GetBinaryReader().ReadUtf32String();
 70 }

 72 Value* ConstantSymbol::GetValue()
 73 {
 74     if (!value)
 75     {
 76         if (filePathReadFrom.empty())
 77         {
 78             throw Exception(GetModule(), "internal error: could not read value: value file name not set", GetSpan());
 79         }
 80         BinaryReader reader(filePathReadFrom);
 81         reader.Skip(valuePos);
 82         value.reset(type->MakeValue());
 83         if (!value)
 84         {
 85             throw Exception(GetModule(), "internal error: could not read value because could not create value of type '" + ToUtf8(type->FullName()) + "'", GetSpan());
 86         }
 87         value->Read(reader);
 88     }
 89     return value.get();
 90 }

115 void ConstantSymbol::SetSpecifiers(Specifiers specifiers)
116 {
117     Specifiers accessSpecifiers = specifiers & Specifiers::access_;
118     SetAccess(accessSpecifiers);
119     if ((specifiers & Specifiers::static_) != Specifiers::none)
120     {
121         throw Exception(GetModule(), "constant cannot be static", GetSpan());
122     }
123     if ((specifiers & Specifiers::virtual_) != Specifiers::none)
124     {
125         throw Exception(GetModule(), "constant cannot be virtual", GetSpan());
126     }
127     if ((specifiers & Specifiers::override_) != Specifiers::none)
128     {
129         throw Exception(GetModule(), "constant cannot be override", GetSpan());
130     }
131     if ((specifiers & Specifiers::abstract_) != Specifiers::none)
132     {
133         throw Exception(GetModule(), "constant cannot be abstract", GetSpan());
134     }
135     if ((specifiers & Specifiers::inline_) != Specifiers::none)
136     {
137         throw Exception(GetModule(), "constant cannot be inline", GetSpan());
138     }
139     if ((specifiers & Specifiers::explicit_) != Specifiers::none)
140     {
141         throw Exception(GetModule(), "constant cannot be explicit", GetSpan());
142     }
143     if ((specifiers & Specifiers::external_) != Specifiers::none)
144     {
145         throw Exception(GetModule(), "constant cannot be external", GetSpan());
146     }
147     if ((specifiers & Specifiers::suppress_) != Specifiers::none)
148     {
149         throw Exception(GetModule(), "constant cannot be suppressed", GetSpan());
150     }
151     if ((specifiers & Specifiers::default_) != Specifiers::none)
152     {
153         throw Exception(GetModule(), "constant cannot be default", GetSpan());
154     }
155     if ((specifiers & Specifiers::constexpr_) != Specifiers::none)
156     {
157         throw Exception(GetModule(), "constant cannot be constexpr", GetSpan());
158     }
159     if ((specifiers & Specifiers::cdecl_) != Specifiers::none)
160     {
161         throw Exception(GetModule(), "constant cannot be cdecl", GetSpan());
162     }
163     if ((specifiers & Specifiers::nothrow_) != Specifiers::none)
164     {
165         throw Exception(GetModule(), "constant cannot be nothrow", GetSpan());
166     }
167     if ((specifiers & Specifiers::throw_) != Specifiers::none)
168     {
169         throw Exception(GetModule(), "constant cannot be throw", GetSpan());
170     }
171     if ((specifiers & Specifiers::new_) != Specifiers::none)
172     {
173         throw Exception(GetModule(), "constant cannot be new", GetSpan());
174     }
175     if ((specifiers & Specifiers::const_) != Specifiers::none)
176     {
177         throw Exception(GetModule(), "constant cannot be const", GetSpan());
178     }
179     if ((specifiers & Specifiers::unit_test_) != Specifiers::none)
180     {
181         throw Exception(GetModule(), "constant cannot be unit_test", GetSpan());
182     }
183 }

185 std::string ConstantSymbol::Syntax() const
186 {
187     std::string syntax = GetSpecifierStr();
188     if (!syntax.empty())
189     {
190         syntax.append(1, ' ');
191     }
192     syntax.append("const ");
193     syntax.append(ToUtf8(GetType()->DocName()));
194     syntax.append(1, ' ');
195     syntax.append(ToUtf8(DocName()));
196     if (value)
197     {
198         syntax.append(" = ");
199         std::string valueStr = value->ToString();
200         if (GetType()->IsUnsignedType())
201         {
202             valueStr.append(1, 'u');
203         }
204         syntax.append(valueStr);
205     }
206     syntax.append(1, ';');
207     return syntax;
208 }

215 llvm::Value* ConstantSymbol::ArrayIrObject(Emitter& emitter, bool create)
216 {
217     if (!type->IsArrayType())
218     {
219         throw Exception(GetModule(), "internal error: array object expected", GetSpan());
220     }
221     if (!value)
222     {
223         throw Exception(GetModule(), "internal error: array value missing", GetSpan());
224     }
225     if (value->GetValueType() != ValueType::arrayValue)
226     {
227         throw Exception(GetModule(), "internal error: array value expected", GetSpan());
228     }
229     ArrayValue* arrayValue = static_cast<ArrayValue*>(value.get());
230     llvm::ArrayType* irArrayType =       llvm::cast<llvm::ArrayType>(type->IrType(emitter));
231     llvm::Constant* irArrayObject = emitter.Module()->getOrInsertGlobal(ToUtf8(MangledName()), irArrayType);
232     if (create)
233     {
234         llvm::GlobalVariable* arrayObjectGlobal =       llvm::cast<llvm::GlobalVariable>(irArrayObject);
235         arrayObjectGlobal->setInitializer(      llvm::cast<llvm::Constant>(arrayValue->IrValue(emitter)));
236     }
237     return irArrayObject;
238 }

240 llvm::Value* ConstantSymbol::StructureIrObject(Emitter& emitter, bool create)
241 {
242     if (!type->IsClassTypeSymbol())
243     {
244         throw Exception(GetModule(), "internal error: class type object expected", GetSpan());
245     }
246     if (!value)
247     {
248         throw Exception(GetModule(), "internal error: structured value missing", GetSpan());
249     }
250     if (value->GetValueType() != ValueType::structuredValue)
251     {
252         throw Exception(GetModule(), "internal error: structured value expected", GetSpan());
253     }
254     StructuredValue* structuredValue = static_cast<StructuredValue*>(value.get());
255     llvm::StructType* irStructureType =       llvm::cast<llvm::StructType>(type->IrType(emitter));
256     llvm::Constant* irStructureObject = emitter.Module()->getOrInsertGlobal(ToUtf8(MangledName()), irStructureType);
257     if (create)
258     {
259         llvm::GlobalVariable* structureObjectGlobal =       llvm::cast<llvm::GlobalVariable>(irStructureObject);
260         structureObjectGlobal->setInitializer(      llvm::cast<llvm::Constant>(structuredValue->IrValue(emitter)));
261     }
262     return irStructureObject;
263 }
