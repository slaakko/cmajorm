14 TypeMap::TypeMap() : nextClassTemplateSpecializationId(1), nextOtherTypeId(1000000)
15 {
16 }

18 int TypeMap::GetOrInsertType(TypeSymbol* type)
19 {
20     auto it = typeMap.find(type);
21     if (it != typeMap.cend())
22     {
23         return it->second;
24     }
25     else
26     {
27         int typeId = 0;
28         if (type->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
29         {
30             typeId = nextClassTemplateSpecializationId++;
31         }
32         else
33         {
34             typeId = nextOtherTypeId++;
35         }
36         typeMap[type] = typeId;
37         std::unique_ptr<dom::Element> typeElement(new dom::Element(U"type"));
38         typeElement->SetAttribute(U"id", U"type_" + ToUtf32(std::to_string(typeId)));
39         typeElement->AppendChild(std::move(type->ToDomElement(*this)));
40         typeIdTypeElementMap[typeId] = std::move(typeElement);
41         return typeId;
42     }
43 }
44 
45 std::vector<std::unique_ptr<dom::Element>> TypeMap::TypeElements()
46 {
47     std::vector<std::unique_ptr<dom::Element>> typeElements;
48     for (auto& p : typeIdTypeElementMap)
49     {
50         typeElements.push_back(std::move(p.second));
51     }
52     return typeElements;
53 }
