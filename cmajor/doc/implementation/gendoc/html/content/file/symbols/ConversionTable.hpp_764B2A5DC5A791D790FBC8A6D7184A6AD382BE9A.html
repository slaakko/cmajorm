11 namespace cmajor { namespace symbols {

13 struct ConversionTableEntry 
14 {
15     ConversionTableEntry(TypeSymbol* sourceType_, TypeSymbol* targetType_) : sourceType(sourceType_), targetType(targetType_) {}
16     TypeSymbol* sourceType;
17     TypeSymbol* targetType;
18 };
19 
20 inline bool operator==(const ConversionTableEntry& left, const ConversionTableEntry& right)
21 {
22     return TypesEqual(left.sourceType, right.sourceType) && TypesEqual(left.targetType, right.targetType);
23 }
24 
25 struct ConversionTableEntryHash 
26 {
27     size_t operator()(const ConversionTableEntry& entry) const { return      std::hash<std::u32string>()(entry.sourceType->Name()) ^ std::hash<std::u32string>()(entry.targetType->Name());}
28 };
29 
30 class ConversionTable 
31 {
32 public:
33     void AddConversion(FunctionSymbol* conversion);
34     FunctionSymbol* GetConversion(TypeSymbol* sourceType, TypeSymbol* targetType, const Span& span) const;
35     void AddGeneratedConversion(std::unique_ptr<FunctionSymbol>&& generatedConversion);
36     void Add(ConversionTable& that);
37 private:
38     std::unordered_map<ConversionTableEntry, FunctionSymbol*, ConversionTableEntryHash> conversionMap;
39     std::vector<std::unique_ptr<FunctionSymbol>> generatedConversions;
40 };
