 87 void EnumTypeSymbol::SetSpecifiers(Specifiers specifiers)
 88 {
 89     Specifiers accessSpecifiers = specifiers & Specifiers::access_;
 90     SetAccess(accessSpecifiers);
 91     if ((specifiers & Specifiers::static_) != Specifiers::none)
 92     {
 93         throw Exception(GetModule(), "enumerated type cannot be static", GetSpan());
 94     }
 95     if ((specifiers & Specifiers::virtual_) != Specifiers::none)
 96     {
 97         throw Exception(GetModule(), "enumerated type cannot be virtual", GetSpan());
 98     }
 99     if ((specifiers & Specifiers::override_) != Specifiers::none)
100     {
101         throw Exception(GetModule(), "enumerated type cannot be override", GetSpan());
102     }
103     if ((specifiers & Specifiers::abstract_) != Specifiers::none)
104     {
105         throw Exception(GetModule(), "enumerated type cannot be abstract", GetSpan());
106     }
107     if ((specifiers & Specifiers::inline_) != Specifiers::none)
108     {
109         throw Exception(GetModule(), "enumerated type cannot be inline", GetSpan());
110     }
111     if ((specifiers & Specifiers::explicit_) != Specifiers::none)
112     {
113         throw Exception(GetModule(), "enumerated type cannot be explicit", GetSpan());
114     }
115     if ((specifiers & Specifiers::external_) != Specifiers::none)
116     {
117         throw Exception(GetModule(), "enumerated type cannot be external", GetSpan());
118     }
119     if ((specifiers & Specifiers::suppress_) != Specifiers::none)
120     {
121         throw Exception(GetModule(), "enumerated type cannot be suppressed", GetSpan());
122     }
123     if ((specifiers & Specifiers::default_) != Specifiers::none)
124     {
125         throw Exception(GetModule(), "enumerated type cannot be default", GetSpan());
126     }
127     if ((specifiers & Specifiers::constexpr_) != Specifiers::none)
128     {
129         throw Exception(GetModule(), "enumerated type cannot be constexpr", GetSpan());
130     }
131     if ((specifiers & Specifiers::cdecl_) != Specifiers::none)
132     {
133         throw Exception(GetModule(), "enumerated type cannot be cdecl", GetSpan());
134     }
135     if ((specifiers & Specifiers::nothrow_) != Specifiers::none)
136     {
137         throw Exception(GetModule(), "enumerated type cannot be nothrow", GetSpan());
138     }
139     if ((specifiers & Specifiers::throw_) != Specifiers::none)
140     {
141         throw Exception(GetModule(), "enumerated type cannot be throw", GetSpan());
142     }
143     if ((specifiers & Specifiers::new_) != Specifiers::none)
144     {
145         throw Exception(GetModule(), "enumerated type cannot be new", GetSpan());
146     }
147     if ((specifiers & Specifiers::const_) != Specifiers::none)
148     {
149         throw Exception(GetModule(), "enumerated type cannot be const", GetSpan());
150     }
151     if ((specifiers & Specifiers::unit_test_) != Specifiers::none)
152     {
153         throw Exception(GetModule(), "enumerated type cannot be unit_test", GetSpan());
154     }

179 llvm::DIType* EnumTypeSymbol::CreateDIType(Emitter& emitter)
180 {
181     uint64_t sizeInBits = SizeInBits(emitter);
182     uint32_t alignInBits = AlignmentInBits(emitter);
183     std::vector<llvm::Metadata*> elements;
184     std::vector<EnumConstantSymbol*> enumConstants = GetEnumConstants();
185     for (EnumConstantSymbol* enumConstant : enumConstants)
186     {
187         int64_t value = 0;
188         if (underlyingType->IsUnsignedType())
189         {
190             Value* val = enumConstant->GetValue()->As(GetSymbolTable()->GetTypeByName(U"ulong"), false, GetSpan(), true);
191             if (val)
192             {
193                 ULongValue* ulongValue = static_cast<ULongValue*>(val);
194                 value = static_cast<int64_t>(ulongValue->GetValue());
195             }
196         }
197         else
198         {
199             Value* val = enumConstant->GetValue()->As(GetSymbolTable()->GetTypeByName(U"long"), false, GetSpan(), true);
200             if (val)
201             {
202                 LongValue* longValue = static_cast<LongValue*>(val);
203                 value = longValue->GetValue();
204             }
205         }
206         elements.push_back(emitter.DIBuilder()->createEnumerator(ToUtf8(enumConstant->Name()), value));
207     }
208     return emitter.DIBuilder()->createEnumerationType(nullptr, ToUtf8(Name()), emitter.GetFile(GetSpan().FileIndex()), GetSpan().LineNumber(), sizeInBits, alignInBits, 
209         emitter.DIBuilder()->getOrCreateArray(elements), underlyingType->GetDIType(emitter), ToUtf8(MangledName()));
210 }
