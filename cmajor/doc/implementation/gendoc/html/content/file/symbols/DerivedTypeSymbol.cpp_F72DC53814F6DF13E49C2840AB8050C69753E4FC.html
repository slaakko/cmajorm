 16 std::u32string DerivationStr(Derivation derivation)
 17 {
 18     switch (derivation)
 19     {
 20         case Derivation::constDerivation: return U"C";
 21         case Derivation::lvalueRefDerivation: return U"R";
 22         case Derivation::rvalueRefDerivation: return U"RR";
 23         case Derivation::pointerDerivation: return U"P";
 24         default: return std::u32string();
 25     }
 26 }

 97 int CountPointerDerivations(const DerivationVec& derivations)
 98 {

136 TypeDerivationRec RemovePointerDerivation(const TypeDerivationRec& typeDerivationRec)
137 {
138     TypeDerivationRec pointerRemovedDerivationRec;
139     int numPointers = CountPointerDerivations(typeDerivationRec.derivations);
140     if (numPointers > 0)
141     {
142         --numPointers;
143     }
144     int ip = 0;
145     int n = typeDerivationRec.derivations.size();
146     for (int i = 0; i < n; ++i)
147     {
148         Derivation derivation = typeDerivationRec.derivations[i];
149         if (derivation == Derivation::pointerDerivation)
150         {
151             if (ip < numPointers)
152             {
153                 pointerRemovedDerivationRec.derivations.push_back(derivation);
154                 ++ip;
155             }
156         }
157         else
158         {
159             pointerRemovedDerivationRec.derivations.push_back(derivation);
160         }
161     }
162     return pointerRemovedDerivationRec;
163 }

220 TypeDerivationRec AddPointerDerivation(const TypeDerivationRec& typeDerivationRec)
221 {
222     TypeDerivationRec pointerAddedDerivationRec;
223     int numPointers = CountPointerDerivations(typeDerivationRec.derivations);
224     int np = 0;
225     for (Derivation derivation : typeDerivationRec.derivations)
226     {
227         pointerAddedDerivationRec.derivations.push_back(derivation);
228         if (derivation == Derivation::pointerDerivation)
229         {
230             ++np;
231         }
232         if (np == numPointers)
233         {
234             pointerAddedDerivationRec.derivations.push_back(Derivation::pointerDerivation);
235             ++np;
236         }
237     }
238     return pointerAddedDerivationRec;
239 }

241 TypeDerivationRec UnifyDerivations(const TypeDerivationRec& left, const TypeDerivationRec& right)
242 {
243     TypeDerivationRec result;
244     if (HasFrontConstDerivation(left.derivations) || HasFrontConstDerivation(right.derivations))
245     {
246         result.derivations.push_back(Derivation::constDerivation);
247     }
248     int pointerCount = CountPointerDerivations(left.derivations) + CountPointerDerivations(right.derivations);
249     for (int i = 0; i < pointerCount; ++i)
250     {
251         result.derivations.push_back(Derivation::pointerDerivation);
252     }
253     if (HasLvalueReferenceDerivation(left.derivations))
254     {
255         result.derivations.push_back(Derivation::lvalueRefDerivation);
256     }
257     else if (HasRvalueReferenceDerivation(left.derivations))
258     {
259         result.derivations.push_back(Derivation::rvalueRefDerivation);
260     }
261     else if (HasLvalueReferenceDerivation(right.derivations))
262     {
263         result.derivations.push_back(Derivation::lvalueRefDerivation);
264     }
265     else if (HasRvalueReferenceDerivation(right.derivations))
266     {
267         result.derivations.push_back(Derivation::rvalueRefDerivation);
268     }
269     return result;
270 }

272 std::u32string MakeDerivedTypeName(TypeSymbol* baseType, const TypeDerivationRec& derivationRec)
273 {
274     std::u32string derivedTypeName;
275     bool constAdded = false;
276     if (HasFrontConstDerivation(derivationRec.derivations))
277     {
278         derivedTypeName.append(U"const");
279         constAdded = true;
280     }
281     if (!derivedTypeName.empty())
282     {
283         derivedTypeName.append(1, U' ');
284     }
285     derivedTypeName.append(baseType->Name());
286     for (Derivation derivation : derivationRec.derivations)
287     {
288         switch (derivation)
289         {
290             case Derivation::constDerivation:
291             {
292                 if (constAdded)
293                 {
294                     constAdded = false;
295                 }
296                 else
297                 {
298                     if (!derivedTypeName.empty())
299                     {
300                         derivedTypeName.append(1, U' ');
301                     }
302                     derivedTypeName.append(U"const ");
303                 }
304                 break;
305             }
306             case Derivation::lvalueRefDerivation:
307             {
308                 derivedTypeName.append(U"&");
309                 break;
310             }
311             case Derivation::rvalueRefDerivation:
312             {
313                 derivedTypeName.append(U"&&");
314                 break;
315             }
316             case Derivation::pointerDerivation:
317             {
318                 derivedTypeName.append(U"*");
319                 break;
320             }
321         }
322     }
323     return derivedTypeName;
324 }

485 TypeSymbol* DerivedTypeSymbol::RemoveDerivations(const TypeDerivationRec& sourceDerivationRec, const Span& span)
486 {
487     TypeDerivationRec result;
488     const DerivationVec& sourceDerivations = sourceDerivationRec.derivations;
489     if (!HasFrontConstDerivation(sourceDerivations) && HasFrontConstDerivation(derivationRec.derivations))
490     {
491         result.derivations.push_back(Derivation::constDerivation);
492     }
493     int pointerDiff = CountPointerDerivations(derivationRec.derivations) - CountPointerDerivations(sourceDerivations);
494     if (pointerDiff != 0)
495     {
496         for (int i = 0; i < pointerDiff; ++i)
497         {
498             result.derivations.push_back(Derivation::pointerDerivation);
499         }
500     }
501     if (!HasLvalueReferenceDerivation(sourceDerivations) && HasLvalueReferenceDerivation(derivationRec.derivations))
502     {
503         result.derivations.push_back(Derivation::lvalueRefDerivation);
504     }
505     else if (!HasRvalueReferenceDerivation(sourceDerivations) && HasRvalueReferenceDerivation(derivationRec.derivations))
506     {
507         result.derivations.push_back(Derivation::rvalueRefDerivation);
508     }
509     return GetSymbolTable()->MakeDerivedType(baseType, result, span);
510 }

525 llvm::Type* DerivedTypeSymbol::IrType(Emitter& emitter)
526 {
527     if (!irType)
528     {
529         if (baseType->IsVoidType())
530         {
531             irType = emitter.Builder().getInt8Ty();
532         }
533         else
534         {
535             irType = baseType->IrType(emitter);
536         }
537         for (Derivation derivation : derivationRec.derivations)
538         {
539             switch (derivation)
540             {
541                 case Derivation::lvalueRefDerivation:
542                 case Derivation::rvalueRefDerivation:
543                 case Derivation::pointerDerivation:
544                 {
545                     irType = llvm::PointerType::get(irType, 0);
546                     break;
547                 }
548                 default:
549                 {
550                     // todo
551                     break;
552                 }
553             }
554         }
555     }
556     return irType;
557 }

571 llvm::DIType* DerivedTypeSymbol::CreateDIType(Emitter& emitter)
572 {
573     llvm::DIType* type = baseType->GetDIType(emitter);
574     for (Derivation derivation : derivationRec.derivations)
575     {
576         switch (derivation)
577         {
578             case Derivation::constDerivation:
579             {
580                 type = emitter.DIBuilder()->createQualifiedType(llvm::dwarf::DW_TAG_const_type, type);
581                 break;
582             }
583             case Derivation::lvalueRefDerivation:
584             {
585                 type = emitter.DIBuilder()->createReferenceType(llvm::dwarf::DW_TAG_reference_type, type);
586                 break;
587             }
588             case Derivation::rvalueRefDerivation:
589             {
590                 type = emitter.DIBuilder()->createReferenceType(llvm::dwarf::DW_TAG_rvalue_reference_type, type);
591                 break;
592             }
593             case Derivation::pointerDerivation:
594             {
595                 type = emitter.DIBuilder()->createPointerType(type, 64);
596                 break;
597             }
598         }
599     }
600     return type;
601 }
