15 void NamespaceSymbol::Import(NamespaceSymbol* that, SymbolTable& symbolTable)
16 {
17     symbolTable.BeginNamespace(that->Name(), that->GetSpan(), that->GetOriginalModule());
18     for (std::unique_ptr<Symbol>& symbol : that->Members())
19     {
20         if (symbol->GetSymbolType() == SymbolType::namespaceSymbol)
21         {
22             NamespaceSymbol* thatNs = static_cast<NamespaceSymbol*>(symbol.get());
23             Import(thatNs, symbolTable);
24         }
25         else if (symbol->GetSymbolType() != SymbolType::functionGroupSymbol && 
26             symbol->GetSymbolType() != SymbolType::conceptGroupSymbol && 
27             symbol->GetSymbolType() != SymbolType::classGroupTypeSymbol)
28         {
29             symbolTable.Container()->AddMember(symbol.release());
30         }
31     }
32     symbolTable.EndNamespace();
33 }

36 {

39         return      std::unique_ptr<dom::Element>(new dom::Element(U"NamespaceSymbol"));
40     }
41     else

43         return      std::unique_ptr<dom::Element>();
44     }
