 18 std::u32string MakeClassTemplateSpecializationName(ClassTypeSymbol* classTemplate, const std::vector<TypeSymbol*>& templateArgumentTypes)
 19 {
 20     std::u32string name = classTemplate->GroupName();
 21     name.append(1, '<');
 22     int n = templateArgumentTypes.size();
 23     for (int i = 0; i < n; ++i)
 24     {
 25         if (i > 0)
 26         {
 27             name.append(U", ");
 28         }
 29         name.append(templateArgumentTypes[i]->FullName());
 30     }
 31     name.append(1, '>');
 32     return name;
 33 }

 45 std::u32string ClassTemplateSpecializationSymbol::SimpleName() const
 46 {
 47     std::u32string simpleName = classTemplate->GroupName();
 48     int n = templateArgumentTypes.size();
 49     for (int i = 0; i < n; ++i)
 50     {
 51         simpleName.append(U"_").append(templateArgumentTypes[i]->SimpleName());
 52     }
 53     return simpleName;

 56 void ClassTemplateSpecializationSymbol::Write(SymbolWriter& writer)
 57 {
 58     ClassTypeSymbol::Write(writer);
 59     const boost::uuids::uuid& classTemplateId = classTemplate->TypeId();
 60     writer.GetBinaryWriter().Write(classTemplateId);
 61     uint32_t n = templateArgumentTypes.size();
 62     writer.GetBinaryWriter().WriteULEB128UInt(n);
 63     for (uint32_t i = 0; i < n; ++i)
 64     {
 65         TypeSymbol* templateArgumentType = templateArgumentTypes[i];
 66         const boost::uuids::uuid& templateArgumentTypeId = templateArgumentType->TypeId();
 67         writer.GetBinaryWriter().Write(templateArgumentTypeId);
 68     }
 69     writer.GetBinaryWriter().Write(uint8_t(flags));
 70 }

 72 void ClassTemplateSpecializationSymbol::Read(SymbolReader& reader)
 73 {
 74     ClassTypeSymbol::Read(reader);
 75     boost::uuids::uuid classTemplateId;
 76     reader.GetBinaryReader().ReadUuid(classTemplateId);
 77     GetSymbolTable()->EmplaceTypeRequest(this, classTemplateId, -1);
 78     uint32_t n = reader.GetBinaryReader().ReadULEB128UInt();
 79     templateArgumentTypes.resize(n);
 80     for (uint32_t i = 0; i < n; ++i)
 81     {
 82         boost::uuids::uuid typeArgumentId;
 83         reader.GetBinaryReader().ReadUuid(typeArgumentId);
 84         GetSymbolTable()->EmplaceTypeRequest(this, typeArgumentId, -2 - i);
 85     }
 86     flags = ClassTemplateSpecializationFlags(reader.GetBinaryReader().ReadByte());
 87 }

 89 void ClassTemplateSpecializationSymbol::EmplaceType(TypeSymbol* typeSymbol, int index)
 90 {
 91     if (index < 0)
 92     {
 93         if (index == -1)
 94         {
 95             Assert(typeSymbol->GetSymbolType() == SymbolType::classTypeSymbol, "class type symbol expected");
 96             classTemplate = static_cast<ClassTypeSymbol*>(typeSymbol);
 97         }
 98         else
 99         {
100             int typeArgumentIndex = -(index + 2);
101             if (typeArgumentIndex < 0 || typeArgumentIndex >= templateArgumentTypes.size())
102             {
103                 Assert(false, "invalid emplace type index in class template specialization");
104             }
105             templateArgumentTypes[typeArgumentIndex] = typeSymbol;
106         }
107     }
108     else
109     {
110         ClassTypeSymbol::EmplaceType(typeSymbol, index);
111     }
112 }

114 void ClassTemplateSpecializationSymbol::ComputeExportClosure()
115 {
116     if (IsProject())
117     {
118         ClassTypeSymbol::ComputeExportClosure();
119         if (!classTemplate->ExportComputed())
120         {
121             classTemplate->SetExportComputed();
122             classTemplate->ComputeExportClosure();
123         }
124         for (TypeSymbol* templateArgumentType : templateArgumentTypes)
125         {
126             if (!templateArgumentType->ExportComputed())
127             {
128                 templateArgumentType->SetExportComputed();
129                 templateArgumentType->ComputeExportClosure();
130             }
131         }
132         MarkExport();
133     }
134 }

156 TypeSymbol* ClassTemplateSpecializationSymbol::UnifyTemplateArgumentType(SymbolTable& symbolTable, const std::unordered_map<TemplateParameterSymbol*, TypeSymbol*>& templateParameterMap, const Span& span)
157 {
158     std::vector<TypeSymbol*> targetTemplateArgumentTypes;
159     for (int i = 0; i < templateArgumentTypes.size(); ++i)
160     {
161         TypeSymbol* templateArgumentType = templateArgumentTypes[i]->UnifyTemplateArgumentType(symbolTable, templateParameterMap, span);
162         if (templateArgumentType)
163         {
164             targetTemplateArgumentTypes.push_back(templateArgumentType);
165         }
166         else
167         {
168             return nullptr;
169         }
170     }
171     return symbolTable.MakeClassTemplateSpecialization(classTemplate, targetTemplateArgumentTypes, span);
172 }
