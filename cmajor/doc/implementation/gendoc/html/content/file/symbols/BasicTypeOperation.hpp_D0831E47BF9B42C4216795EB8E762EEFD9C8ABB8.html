 343 BasicTypeUnaryOperation<UnOp>::BasicTypeUnaryOperation(SymbolType symbolType, TypeSymbol* type) : FunctionSymbol(symbolType, Span(), UnOp::GroupName())
 344 {
 345     SetGroupName(UnOp::GroupName());

 347     ParameterSymbol* operandParam = new ParameterSymbol(Span(), U"operand");
 348     operandParam->SetType(type);

 350     SetReturnType(type);
 351     ComputeName();
 352 }

 354 template<typename UnOp>
 355 void BasicTypeUnaryOperation<UnOp>::GenerateCall(Emitter& emitter, std::vector<GenObject*>& genObjects, OperationFlags flags, const Span& span)
 356 {
 357     Assert(genObjects.size() == 1, "unary operation needs one object");
 358     genObjects[0]->Load(emitter, OperationFlags::none);
 359     llvm::Value* arg = emitter.Stack().Pop();
 360     emitter.SetCurrentDebugLocation(span);
 361     emitter.Stack().Push(UnOp::Generate(emitter.Builder(), arg));
 362 }

1008 BasicTypeComparisonOperation<ComparisonOp>::BasicTypeComparisonOperation(SymbolType symbolType, TypeSymbol* type, TypeSymbol* boolType) : FunctionSymbol(symbolType, Span(), ComparisonOp::GroupName())
1009 {
1010     SetGroupName(ComparisonOp::GroupName());
1011     SetAccess(SymbolAccess::public_);
1012     ParameterSymbol* leftParam = new ParameterSymbol(Span(), U"left");
1013     leftParam->SetType(type);
1014     AddMember(leftParam);
1015     ParameterSymbol* rightParam = new ParameterSymbol(Span(), U"right");
1016     rightParam->SetType(type);
1017     AddMember(rightParam);
1018     SetReturnType(boolType);
1019     ComputeName();
1020     if (ComparisonOp::IsIntegerOpEqual())
1021     {
1022         if (type->IsBasicTypeSymbol())
1023         {
1024             BasicTypeSymbol* basicTypeSymbol = static_cast<BasicTypeSymbol*>(type);
1025             basicTypeSymbol->SetEqualityOp(this);
1026         }
1027     }

1030 template<typename ComparisonOp>
1031 void BasicTypeComparisonOperation<ComparisonOp>::GenerateCall(Emitter& emitter, std::vector<GenObject*>& genObjects, OperationFlags flags, const Span& span)
1032 {
1033     Assert(genObjects.size() == 2, "comparison operation needs two objects");
1034     genObjects[0]->Load(emitter, OperationFlags::none);
1035     llvm::Value* left = emitter.Stack().Pop();
1036     genObjects[1]->Load(emitter, OperationFlags::none);
1037     llvm::Value* right = emitter.Stack().Pop();
1038     emitter.Stack().Push(ComparisonOp::Generate(emitter.Builder(), left, right));
1039 }
