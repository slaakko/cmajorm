  66 std::string DelegateTypeSymbol::Syntax() const
  67 {
  68     std::string syntax = GetSpecifierStr();
  69     if (!syntax.empty())
  70     {
  71         syntax.append(1, ' ');
  72     }
  73     syntax.append("delegate ");
  74     syntax.append(ToUtf8(ReturnType()->DocName()));
  75     syntax.append(1, ' ');
  76     syntax.append(ToUtf8(DocName()));
  77     syntax.append(1, '(');
  78     bool first = true;
  79     for (ParameterSymbol* param : parameters)
  80     {
  81         if (first)
  82         {
  83             first = false;
  84         }
  85         else
  86         {
  87             syntax.append(", ");
  88         }
  89         syntax.append(ToUtf8(param->GetType()->DocName()));
  90         syntax.append(1, ' ');
  91         syntax.append(ToUtf8(param->DocName()));
  92     }
  93     syntax.append(");");
  94     return syntax;
  95 }

 102 void DelegateTypeSymbol::ComputeExportClosure()
 103 {
 104     if (IsProject())
 105     {
 106         for (ParameterSymbol* parameter : parameters)
 107         {
 108             if (!parameter->ExportComputed())
 109             {
 110                 parameter->SetExportComputed();
 111                 parameter->ComputeExportClosure();
 112             }
 113         }
 114         if (returnParam)
 115         {
 116             if (!returnParam->ExportComputed())
 117             {
 118                 returnParam->SetExportComputed();
 119                 returnParam->ComputeExportClosure();
 120             }
 121         }
 122         if (returnType)
 123         {
 124             if (!returnType->ExportComputed())
 125             {
 126                 returnType->SetExportComputed();
 127                 returnType->ComputeExportClosure();
 128             }
 129         }
 130         AddConst(GetSpan())->AddLvalueReference(GetSpan())->ComputeExportClosure();
 131         AddRvalueReference(GetSpan())->ComputeExportClosure();
 132     }

 151 llvm::Type* DelegateTypeSymbol::IrType(Emitter& emitter)
 152 {
 153     if (!irType)
 154     {
 155         llvm::Type* retType = llvm::Type::getVoidTy(emitter.Context());
 156         if (!returnType->IsVoidType() && !ReturnsClassInterfaceOrClassDelegateByValue())
 157         {
 158             retType = returnType->IrType(emitter);
 159         }
 160         std::vector<llvm::Type*> paramTypes;
 161         int np = parameters.size();
 162         for (int i = 0; i < np; ++i)
 163         {
 164             ParameterSymbol* parameter = parameters[i];
 165             paramTypes.push_back(parameter->GetType()->IrType(emitter));
 166         }
 167         if (returnParam)
 168         {
 169             paramTypes.push_back(returnParam->GetType()->IrType(emitter));
 170         }
 171         irType = llvm::PointerType::get(llvm::FunctionType::get(retType, paramTypes, false), 0);
 172     }
 173     return irType;
 174 }

 181 void DelegateTypeSymbol::SetSpecifiers(Specifiers specifiers)
 182 {
 183     Specifiers accessSpecifiers = specifiers & Specifiers::access_;
 184     SetAccess(accessSpecifiers);
 185     if ((specifiers & Specifiers::static_) != Specifiers::none)
 186     {
 187         throw Exception(GetModule(), "delegate cannot be static", GetSpan());
 188     }
 189     if ((specifiers & Specifiers::virtual_) != Specifiers::none)
 190     {
 191         throw Exception(GetModule(), "delegate cannot be virtual", GetSpan());
 192     }
 193     if ((specifiers & Specifiers::override_) != Specifiers::none)
 194     {
 195         throw Exception(GetModule(), "delegate cannot be override", GetSpan());
 196     }
 197     if ((specifiers & Specifiers::abstract_) != Specifiers::none)
 198     {
 199         throw Exception(GetModule(), "delegate cannot be abstract", GetSpan());
 200     }
 201     if ((specifiers & Specifiers::inline_) != Specifiers::none)
 202     {
 203         throw Exception(GetModule(), "delegate cannot be inline", GetSpan());
 204     }
 205     if ((specifiers & Specifiers::explicit_) != Specifiers::none)
 206     {
 207         throw Exception(GetModule(), "delegate cannot be explicit", GetSpan());
 208     }
 209     if ((specifiers & Specifiers::external_) != Specifiers::none)
 210     {
 211         throw Exception(GetModule(), "delegate cannot be external", GetSpan());
 212     }
 213     if ((specifiers & Specifiers::suppress_) != Specifiers::none)
 214     {
 215         throw Exception(GetModule(), "delegate cannot be suppressed", GetSpan());
 216     }
 217     if ((specifiers & Specifiers::default_) != Specifiers::none)
 218     {
 219         throw Exception(GetModule(), "delegate cannot be default", GetSpan());
 220     }
 221     if ((specifiers & Specifiers::constexpr_) != Specifiers::none)
 222     {
 223         throw Exception(GetModule(), "delegate cannot be constexpr", GetSpan());
 224     }
 225     if ((specifiers & Specifiers::cdecl_) != Specifiers::none)
 226     {
 227         throw Exception(GetModule(), "delegate cannot be cdecl", GetSpan());
 228     }
 229     if ((specifiers & Specifiers::nothrow_) != Specifiers::none)
 230     {
 231         SetNothrow();
 232     }
 233     if ((specifiers & Specifiers::throw_) != Specifiers::none)
 234     {
 235         if (IsNothrow())
 236         {
 237             throw Exception(GetModule(), "delegate cannot be throw and nothrow at the same time", GetSpan());
 238         }
 239     }
 240     if ((specifiers & Specifiers::new_) != Specifiers::none)
 241     {
 242         throw Exception(GetModule(), "delegate cannot be new", GetSpan());
 243     }
 244     if ((specifiers & Specifiers::const_) != Specifiers::none)
 245     {
 246         throw Exception(GetModule(), "delegate cannot be const", GetSpan());
 247     }
 248     if ((specifiers & Specifiers::unit_test_) != Specifiers::none)
 249     {
 250         throw Exception(GetModule(), "delegate cannot be unit_test", GetSpan());
 251     }
 252 }

 264 void DelegateTypeSymbol::GenerateCall(Emitter& emitter, std::vector<GenObject*>& genObjects, OperationFlags flags, const Span& span)
 265 {
 266     llvm::Value* callee = nullptr;
 267     int na = genObjects.size();
 268     for (int i = 0; i < na; ++i)
 269     {
 270         GenObject* genObject = genObjects[i];
 271         genObject->Load(emitter, flags & OperationFlags::functionCallFlags);
 272         if (i == 0)
 273         {
 274             callee = emitter.Stack().Pop();
 275         }
 276     }
 277     ArgVector args;
 278     int n = parameters.size();
 279     if (ReturnsClassInterfaceOrClassDelegateByValue())
 280     {
 281         ++n;
 282     }
 283     args.resize(n);
 284     for (int i = 0; i < n; ++i)
 285     {
 286         llvm::Value* arg = emitter.Stack().Pop();
 287         args[n - i - 1] = arg;
 288     }
 289     emitter.SetCurrentDebugLocation(span);
 290     llvm::BasicBlock* handlerBlock = emitter.HandlerBlock();
 291     llvm::BasicBlock* cleanupBlock = emitter.CleanupBlock();
 292     bool newCleanupNeeded = emitter.NewCleanupNeeded();
 293     Pad* currentPad = emitter.CurrentPad();
 294     std::vector<llvm::OperandBundleDef> bundles;
 295     if (currentPad != nullptr)
 296     {
 297         std::vector<llvm::Value*> inputs;
 298         inputs.push_back(currentPad->value);
 299         bundles.push_back(llvm::OperandBundleDef("funclet", inputs));
 300     }
 301     if (returnType->GetSymbolType() != SymbolType::voidTypeSymbol && !ReturnsClassInterfaceOrClassDelegateByValue())
 302     {
 303         if (IsNothrow() || (!handlerBlock && !cleanupBlock && !newCleanupNeeded))
 304         {
 305             if (currentPad == nullptr)
 306             {
 307                 emitter.Stack().Push(emitter.Builder().CreateCall(callee, args));
 308             }
 309             else
 310             {
 311                 llvm::CallInst* callInst = llvm::CallInst::Create(callee, args, bundles, "", emitter.CurrentBasicBlock());
 312                 if (emitter.DIBuilder())
 313                 {
 314                     callInst->setDebugLoc(emitter.GetDebugLocation(span));
 315                 }
 316                 emitter.Stack().Push(callInst);
 317             }
 318         }
 319         else
 320         {
 321             llvm::BasicBlock* nextBlock = llvm::BasicBlock::Create(emitter.Context(), "next", emitter.Function());
 322             if (newCleanupNeeded)
 323             {
 324                 emitter.CreateCleanup();
 325                 cleanupBlock = emitter.CleanupBlock();
 326             }
 327             llvm::BasicBlock* unwindBlock = cleanupBlock;
 328             if (unwindBlock == nullptr)
 329             {
 330                 unwindBlock = handlerBlock;
 331                 Assert(unwindBlock, "no unwind block");
 332             }
 333             if (currentPad == nullptr)
 334             {
 335                 emitter.Stack().Push(emitter.Builder().CreateInvoke(callee, nextBlock, unwindBlock, args));
 336             }
 337             else
 338             {
 339                 llvm::InvokeInst* invokeInst = llvm::InvokeInst::Create(callee, nextBlock, unwindBlock, args, bundles, "", emitter.CurrentBasicBlock());
 340                 if (emitter.DIBuilder())
 341                 {
 342                     invokeInst->setDebugLoc(emitter.GetDebugLocation(span));
 343                 }
 344                 emitter.Stack().Push(invokeInst);
 345             }
 346             emitter.SetCurrentBasicBlock(nextBlock);
 347         }
 348     }
 349     else
 350     {
 351         if (IsNothrow() || (!handlerBlock && !cleanupBlock && !newCleanupNeeded))
 352         {
 353             if (currentPad == nullptr)
 354             {
 355                 emitter.Builder().CreateCall(callee, args);
 356             }
 357             else
 358             {
 359                 llvm::CallInst* callInst = llvm::CallInst::Create(callee, args, bundles, "", emitter.CurrentBasicBlock());
 360                 if (emitter.DIBuilder())
 361                 {
 362                     callInst->setDebugLoc(emitter.GetDebugLocation(span));
 363                 }
 364             }
 365         }
 366         else
 367         {
 368             llvm::BasicBlock* nextBlock = llvm::BasicBlock::Create(emitter.Context(), "next", emitter.Function());
 369             if (newCleanupNeeded)
 370             {
 371                 emitter.CreateCleanup();
 372                 cleanupBlock = emitter.CleanupBlock();
 373             }
 374             llvm::BasicBlock* unwindBlock = cleanupBlock;
 375             if (unwindBlock == nullptr)
 376             {
 377                 unwindBlock = handlerBlock;
 378                 Assert(unwindBlock, "no unwind block");
 379             }
 380             if (currentPad == nullptr)
 381             {
 382                 emitter.Builder().CreateInvoke(callee, nextBlock, unwindBlock, args);
 383             }
 384             else
 385             {
 386                 llvm::InvokeInst* invokeInst = llvm::InvokeInst::Create(callee, nextBlock, unwindBlock, args, bundles, "", emitter.CurrentBasicBlock());
 387                 if (emitter.DIBuilder())
 388                 {
 389                     invokeInst->setDebugLoc(emitter.GetDebugLocation(span));
 390                 }
 391             }
 392             emitter.SetCurrentBasicBlock(nextBlock);
 393         }
 394     }
 395 }
