 108 void ClassTypeSymbol::Write(SymbolWriter& writer)
 109 {
 110     TypeSymbol::Write(writer);
 111     writer.GetBinaryWriter().Write(groupName);
 112     writer.GetBinaryWriter().Write(static_cast<uint8_t>(flags & ~ClassTypeSymbolFlags::layoutsComputed));
 113     writer.GetBinaryWriter().Write(static_cast<int32_t>(minArity));
 114     if (IsClassTemplate())
 115     {
 116         uint32_t sizePos = writer.GetBinaryWriter().Pos();
 117         uint32_t sizeOfAstNodes = 0;
 118         writer.GetBinaryWriter().Write(sizeOfAstNodes);
 119         uint32_t startPos = writer.GetBinaryWriter().Pos();
 120         usingNodes.Write(writer.GetAstWriter());
 121         Node* node = GetSymbolTable()->GetNode(this);
 122         writer.GetAstWriter().Write(node);
 123         uint32_t endPos = writer.GetBinaryWriter().Pos();
 124         sizeOfAstNodes = endPos - startPos;
 125         writer.GetBinaryWriter().Seek(sizePos);
 126         writer.GetBinaryWriter().Write(sizeOfAstNodes);
 127         writer.GetBinaryWriter().Seek(endPos);
 128         bool hasConstraint = constraint != nullptr;
 129         writer.GetBinaryWriter().Write(hasConstraint);
 130         if (hasConstraint)
 131         {
 132             writer.GetAstWriter().Write(constraint.get());
 133         }
 134         bool hasPrototype = prototype != nullptr;
 135         writer.GetBinaryWriter().Write(hasPrototype);
 136         if (hasPrototype)
 137         {
 138             writer.GetBinaryWriter().Write(prototype->TypeId());
 139         }
 140     }
 141     else if (GetSymbolType() == SymbolType::classTypeSymbol)
 142     {
 143         boost::uuids::uuid baseClassId = boost::uuids::nil_generator()();
 144         if (baseClass)
 145         {
 146             baseClassId = baseClass->TypeId();
 147         }
 148         writer.GetBinaryWriter().Write(baseClassId);
 149         uint32_t n = uint32_t(implementedInterfaces.size());
 150         writer.GetBinaryWriter().WriteULEB128UInt(n);
 151         for (uint32_t i = 0; i < n; ++i)
 152         {
 153             InterfaceTypeSymbol* intf = implementedInterfaces[i];
 154             const boost::uuids::uuid& intfTypeId = intf->TypeId();
 155             writer.GetBinaryWriter().Write(intfTypeId);
 156         }
 157         uint32_t vmtSize = vmt.size();
 158         writer.GetBinaryWriter().WriteULEB128UInt(vmtSize);
 159         writer.GetBinaryWriter().Write(vmtPtrIndex);
 160         bool hasConstraint = constraint != nullptr;
 161         writer.GetBinaryWriter().Write(hasConstraint);
 162         if (hasConstraint)
 163         {
 164             writer.GetAstWriter().Write(constraint.get());
 165         }
 166     }
 167     else if (GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
 168     {
 169         bool hasConstraint = constraint != nullptr;
 170         writer.GetBinaryWriter().Write(hasConstraint);
 171         if (hasConstraint)
 172         {
 173             writer.GetAstWriter().Write(constraint.get());
 174         }
 175     }

 178 void ClassTypeSymbol::Read(SymbolReader& reader)
 179 {
 180     TypeSymbol::Read(reader);
 181     groupName = reader.GetBinaryReader().ReadUtf32String();
 182     flags = static_cast<ClassTypeSymbolFlags>(reader.GetBinaryReader().ReadByte());
 183     minArity = reader.GetBinaryReader().ReadInt();
 184     if (IsClassTemplate())
 185     {
 186         sizeOfAstNodes = reader.GetBinaryReader().ReadUInt();
 187         astNodesPos = reader.GetBinaryReader().Pos();
 188         reader.GetBinaryReader().Skip(sizeOfAstNodes);
 189         filePathReadFrom = reader.GetBinaryReader().FileName();
 190         bool hasConstraint = reader.GetBinaryReader().ReadBool();
 191         if (hasConstraint)
 192         {
 193             constraint.reset(reader.GetAstReader().ReadConstraintNode());
 194         }
 195         bool hasPrototype = reader.GetBinaryReader().ReadBool();
 196         if (hasPrototype)
 197         {
 198             boost::uuids::uuid prototypeId;
 199             reader.GetBinaryReader().ReadUuid(prototypeId);
 200             GetSymbolTable()->EmplaceTypeRequest(this, prototypeId, 1000000);
 201         }
 202     }
 203     else if (GetSymbolType() == SymbolType::classTypeSymbol)
 204     {
 205         boost::uuids::uuid baseClassId;
 206         reader.GetBinaryReader().ReadUuid(baseClassId);
 207         if (!baseClassId.is_nil())
 208         {
 209             GetSymbolTable()->EmplaceTypeRequest(this, baseClassId, 0);
 210         }
 211         uint32_t n = reader.GetBinaryReader().ReadULEB128UInt();
 212         implementedInterfaces.resize(n);
 213         for (uint32_t i = 0; i < n; ++i)
 214         {
 215             boost::uuids::uuid intfTypeId;
 216             reader.GetBinaryReader().ReadUuid(intfTypeId);
 217             GetSymbolTable()->EmplaceTypeRequest(this, intfTypeId, 1 + i);
 218         }
 219         uint32_t vmtSize = reader.GetBinaryReader().ReadULEB128UInt();
 220         vmt.resize(vmtSize);
 221         vmtPtrIndex = reader.GetBinaryReader().ReadInt();
 222         bool hasConstraint = reader.GetBinaryReader().ReadBool();
 223         if (hasConstraint)
 224         {
 225             constraint.reset(reader.GetAstReader().ReadConstraintNode());
 226         }
 227         if (destructor)
 228         {
 229             if (destructor->VmtIndex() != -1)
 230             {
 231                 Assert(destructor->VmtIndex() < vmt.size(), "invalid destructor vmt index");
 232                 vmt[destructor->VmtIndex()] = destructor;
 233             }
 234         }
 235         for (FunctionSymbol* memberFunction : memberFunctions)
 236         {
 237             if (memberFunction->VmtIndex() != -1)
 238             {
 239                 Assert(memberFunction->VmtIndex() < vmt.size(), "invalid member function vmt index");
 240                 vmt[memberFunction->VmtIndex()] = memberFunction;
 241             }
 242         }
 243         if (IsPolymorphic() && !IsPrototypeTemplateSpecialization())
 244         {
 245             GetSymbolTable()->AddPolymorphicClass(this);
 246         }
 247         if (StaticConstructor())
 248         {
 249             GetSymbolTable()->AddClassHavingStaticConstructor(this);
 250         }
 251         reader.AddClassType(this);
 252     }
 253     else if (GetSymbolType() == SymbolType::classTemplateSpecializationSymbol)
 254     {
 255         bool hasConstraint = reader.GetBinaryReader().ReadBool();
 256         if (hasConstraint)
 257         {
 258             constraint.reset(reader.GetAstReader().ReadConstraintNode());
 259         }
 260     }
 261 }

 291 void ClassTypeSymbol::EmplaceType(TypeSymbol* typeSymbol, int index)
 292 {
 293     if (index == 0)
 294     {
 295         Assert(typeSymbol->GetSymbolType() == SymbolType::classTypeSymbol || typeSymbol->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol, "class type symbol expected");
 296         baseClass = static_cast<ClassTypeSymbol*>(typeSymbol);
 297         GetContainerScope()->SetBase(baseClass->GetContainerScope());
 298     }
 299     else if (index >= 1)
 300     {
 301         if (index == 1000000)
 302         {
 303             Assert(typeSymbol->GetSymbolType() == SymbolType::classTemplateSpecializationSymbol, "class template specialization expected");
 304             SetPrototype(static_cast<ClassTemplateSpecializationSymbol*>(typeSymbol));
 305         }
 306         else
 307         {
 308             Assert(typeSymbol->GetSymbolType() == SymbolType::interfaceTypeSymbol, "interface type symbol expected");
 309             InterfaceTypeSymbol* interfaceTypeSymbol = static_cast<InterfaceTypeSymbol*>(typeSymbol);
 310             implementedInterfaces[index - 1] = interfaceTypeSymbol;
 311         }
 312     }

 315 void ClassTypeSymbol::AddMember(Symbol* member)
 316 {
 317     TypeSymbol::AddMember(member);
 318     switch (member->GetSymbolType())
 319     {
 320         case SymbolType::templateParameterSymbol:
 321         {
 322             templateParameters.push_back(static_cast<TemplateParameterSymbol*>(member));
 323             break;
 324         }
 325         case SymbolType::memberVariableSymbol:
 326         {
 327             if (member->IsStatic())
 328             {
 329                 staticMemberVariables.push_back(static_cast<MemberVariableSymbol*>(member));
 330             }
 331             else
 332             {
 333                 memberVariables.push_back(static_cast<MemberVariableSymbol*>(member));
 334             }
 335             break;
 336         }
 337         case SymbolType::staticConstructorSymbol:
 338         {
 339             if (staticConstructor)
 340             {
 341                 throw Exception(GetModule(), "already has a static constructor", member->GetSpan(), staticConstructor->GetSpan());
 342             }
 343             else
 344             {
 345                 staticConstructor = static_cast<StaticConstructorSymbol*>(member);
 346             }
 347             break;
 348         }
 349         case SymbolType::constructorSymbol:
 350         {
 351             ConstructorSymbol* constructor = static_cast<ConstructorSymbol*>(member);
 352             constructors.push_back(constructor);
 353             break;
 354         }
 355         case SymbolType::destructorSymbol:
 356         {
 357             if (destructor)
 358             {
 359                 throw Exception(GetModule(), "already has a destructor", member->GetSpan(), destructor->GetSpan());
 360             }
 361             else
 362             {
 363                 destructor = static_cast<DestructorSymbol*>(member);
 364             }
 365             break;
 366         }
 367         case SymbolType::memberFunctionSymbol:
 368         {
 369             memberFunctions.push_back(static_cast<MemberFunctionSymbol*>(member));
 370             break;
 371         }
 372     }
 373 }

 391 bool ClassTypeSymbol::HasNontrivialDestructor() const
 392 {
 393     if (destructor || IsPolymorphic()) return true;

 404     return false;

 407 void ClassTypeSymbol::Accept(SymbolCollector* collector)
 408 {

 410     {

 415             {

 424             {
 425                 collector->AddClass(this);

 427         }

 437 {

 444     }
 445     if (!implementedInterfaces.empty())

 457         formatter.WriteLine("constraint: " + constraint->ToString());
 458     }

 461     {
 462         formatter.WriteLine("vmt object name: " + VmtObjectName());

 465     {
 466         formatter.WriteLine("static object name: " + StaticObjectName());
 467     }
 468     formatter.IncIndent();
 469     SymbolCollector collector;
 470     TypeSymbol::Accept(&collector);
 471     if (!collector.Functions().empty())
 472     {
 473         formatter.WriteLine();
 474         formatter.WriteLine("MEMBER FUNCTIONS");
 475         for (FunctionSymbol* function : collector.Functions())
 476         {
 477             formatter.WriteLine();
 478             function->Dump(formatter);
 479         }
 480     }
 481     if (!collector.Classes().empty())
 482     {
 483         formatter.WriteLine();
 484         formatter.WriteLine("CLASSES");
 485         for (ClassTypeSymbol* class_ : collector.Classes())
 486         {
 487             formatter.WriteLine();
 488             class_->Dump(formatter);
 489         }
 490     }
 491     if (!collector.Interfaces().empty())
 492     {
 493         formatter.WriteLine();
 494         formatter.WriteLine("INTERFACES");
 495         for (InterfaceTypeSymbol* interface : collector.Interfaces())
 496         {
 497             formatter.WriteLine();
 498             interface->Dump(formatter);
 499         }
 500     }
 501     if (!collector.Typedefs().empty())
 502     {
 503         formatter.WriteLine();
 504         formatter.WriteLine("TYPEDEFS");
 505         for (TypedefSymbol* typedef_ : collector.Typedefs())
 506         {
 507             formatter.WriteLine();
 508             typedef_->Dump(formatter);
 509         }
 510     }
 511     if (!collector.Concepts().empty())
 512     {
 513         formatter.WriteLine();
 514         formatter.WriteLine("CONCEPTS");
 515         for (ConceptSymbol* concept_ : collector.Concepts())
 516         {
 517             formatter.WriteLine();
 518             concept_->Dump(formatter);
 519         }
 520     }
 521     if (!collector.Constants().empty())
 522     {
 523         formatter.WriteLine();
 524         formatter.WriteLine("CONSTANTS");
 525         for (ConstantSymbol* constant : collector.Constants())
 526         {
 527             formatter.WriteLine();
 528             constant->Dump(formatter);
 529         }
 530     }
 531     if (!collector.Delegates().empty())
 532     {
 533         formatter.WriteLine();
 534         formatter.WriteLine("DELEGATES");
 535         for (DelegateTypeSymbol* delegate_ : collector.Delegates())
 536         {
 537             formatter.WriteLine();
 538             delegate_->Dump(formatter);
 539         }
 540     }
 541     if (!collector.ClassDelegates().empty())
 542     {
 543         formatter.WriteLine();
 544         formatter.WriteLine("CLASS DELEGATES");
 545         for (ClassDelegateTypeSymbol* classDelegate : collector.ClassDelegates())
 546         {
 547             formatter.WriteLine();
 548             classDelegate->Dump(formatter);
 549         }
 550     }
 551     if (!collector.EnumeratedTypes().empty())
 552     {
 553         formatter.WriteLine();
 554         formatter.WriteLine("ENUMERATED TYPES");
 555         for (EnumTypeSymbol* enumeratedType : collector.EnumeratedTypes())
 556         {
 557             formatter.WriteLine();
 558             enumeratedType->Dump(formatter);
 559         }
 560     }
 561     if (!collector.MemberVariables().empty())
