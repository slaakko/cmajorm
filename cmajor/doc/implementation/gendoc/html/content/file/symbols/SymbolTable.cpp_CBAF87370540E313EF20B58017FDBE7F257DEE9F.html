  72 void SymbolTable::Write(SymbolWriter& writer)
  73 {
  74     globalNs.Write(writer);
  75     std::vector<ArrayTypeSymbol*> exportedArrayTypes;
  76     for (const std::unique_ptr<ArrayTypeSymbol>& arrayType : arrayTypes)
  77     {
  78         if (arrayType->IsProject())
  79         {
  80             exportedArrayTypes.push_back(arrayType.get());
  81         }
  82     }
  83     uint32_t na = exportedArrayTypes.size();
  84     writer.GetBinaryWriter().WriteULEB128UInt(na);
  85     for (ArrayTypeSymbol* exportedArrayType : exportedArrayTypes)
  86     {
  87         writer.Write(exportedArrayType);
  88     }
  89     globalNs.ComputeExportClosure();
  90     std::vector<TypeSymbol*> exportedDerivedTypes;
  91     for (const auto& derivedType : derivedTypes)
  92     {
  93         if (derivedType->MarkedExport())
  94         {
  95             exportedDerivedTypes.push_back(derivedType.get());
  96         }
  97     }
  98     uint32_t ned = exportedDerivedTypes.size();
  99     writer.GetBinaryWriter().WriteULEB128UInt(ned);
 100     for (TypeSymbol* exportedDerivedType : exportedDerivedTypes)
 101     {
 102         writer.Write(exportedDerivedType);
 103     }
 104     std::vector<TypeSymbol*> exportedClassTemplateSpecializations;
 105     for (const auto& classTemplateSpecialization : classTemplateSpecializations)
 106     {
 107         if (classTemplateSpecialization->MarkedExport())
 108         {
 109             exportedClassTemplateSpecializations.push_back(classTemplateSpecialization.get());
 110         }
 111     }
 112     uint32_t nec = exportedClassTemplateSpecializations.size();
 113     writer.GetBinaryWriter().WriteULEB128UInt(nec);
 114     for (TypeSymbol* classTemplateSpecialization : exportedClassTemplateSpecializations)
 115     {
 116         writer.Write(classTemplateSpecialization);
 117     }
 118     uint32_t nj = jsonClasses.size();
 119     writer.GetBinaryWriter().WriteULEB128UInt(nj);
 120     for (const std::u32string& jsonClass : jsonClasses)
 121     {
 122         writer.GetBinaryWriter().Write(jsonClass);
 123     }
 124     if (GetGlobalFlag(GlobalFlags::profile))
 125     {
 126         uint32_t n = profiledFunctionNameMap.size();
 127         writer.GetBinaryWriter().WriteULEB128UInt(n);
 128         for (const auto& p : profiledFunctionNameMap)
 129         {
 130             writer.GetBinaryWriter().Write(p.first);
 131             writer.GetBinaryWriter().Write(p.second);
 132         }
 133     }

 136 void SymbolTable::Read(SymbolReader& reader)
 137 {
 138     reader.SetSymbolTable(this);
 139     reader.GetAstReader().SetModuleId(reader.GetModule()->GetModuleId());
 140     globalNs.Read(reader);
 141     uint32_t na = reader.GetBinaryReader().ReadULEB128UInt();
 142     for (uint32_t i = 0; i < na; ++i)
 143     {
 144         ArrayTypeSymbol* arrayTypeSymbol = reader.ReadArrayTypeSymbol(&globalNs);
 145         arrayTypes.push_back(     std::unique_ptr<ArrayTypeSymbol>(arrayTypeSymbol));
 146     }
 147     uint32_t nd = reader.GetBinaryReader().ReadULEB128UInt();
 148     for (uint32_t i = 0; i < nd; ++i)
 149     {
 150         DerivedTypeSymbol* derivedTypeSymbol = reader.ReadDerivedTypeSymbol(&globalNs);
 151         derivedTypes.push_back(     std::unique_ptr<DerivedTypeSymbol>(derivedTypeSymbol));
 152     }
 153     uint32_t nc = reader.GetBinaryReader().ReadULEB128UInt();
 154     for (uint32_t i = 0; i < nc; ++i)
 155     {
 156         ClassTemplateSpecializationSymbol* classTemplateSpecialization = reader.ReadClassTemplateSpecializationSymbol(&globalNs);
 157         classTemplateSpecializations.push_back(     std::unique_ptr<ClassTemplateSpecializationSymbol>(classTemplateSpecialization));
 158         reader.AddClassTemplateSpecialization(classTemplateSpecialization);
 159     }
 160     uint32_t nj = reader.GetBinaryReader().ReadULEB128UInt();
 161     for (uint32_t i = 0; i < nj; ++i)
 162     {
 163         std::u32string jsonClass = reader.GetBinaryReader().ReadUtf32String();
 164         jsonClasses.insert(jsonClass);
 165     }
 166     if (GetGlobalFlag(GlobalFlags::profile))
 167     {
 168         uint32_t n = reader.GetBinaryReader().ReadULEB128UInt();
 169         for (uint32_t i = 0; i < n; ++i)
 170         {
 171             boost::uuids::uuid functionId;
 172             reader.GetBinaryReader().ReadUuid(functionId);
 173             std::u32string profiledFunctionName = reader.GetBinaryReader().ReadUtf32String();
 174             MapProfiledFunction(functionId, profiledFunctionName);
 175         }
 176     }
 177     ProcessTypeConceptAndFunctionRequests();
 178     for (FunctionSymbol* conversion : reader.Conversions())
 179     {
 180         AddConversion(conversion);
 181     }
 182 }

 184 void SymbolTable::Import(SymbolTable& symbolTable)
 185 {
 186     globalNs.Import(&symbolTable.globalNs, *this);
 187     for (const auto& pair : symbolTable.typeIdMap)
 188     {
 189         Symbol* typeOrConcept = pair.second;
 190         if (typeOrConcept->IsTypeSymbol())
 191         {
 192             TypeSymbol* type = static_cast<TypeSymbol*>(typeOrConcept);
 193             type->SetSymbolTable(this);
 194             type->SetModule(module);
 195             typeIdMap[type->TypeId()] = type;
 196             typeNameMap[type->FullName()] = type;
 197         }
 198         else if (typeOrConcept->GetSymbolType() == SymbolType::conceptSymbol)
 199         {
 200             ConceptSymbol* concept = static_cast<ConceptSymbol*>(typeOrConcept);
 201             concept->SetSymbolTable(this);
 202             concept->SetModule(module);
 203             typeIdMap[concept->TypeId()] = concept;
 204         }
 205         else
 206         {
 207             Assert(false, "type or concept symbol expected");
 208         }
 209     }
 210     for (const auto& pair : symbolTable.functionIdMap)
 211     {
 212         FunctionSymbol* function = pair.second;
 213         function->SetSymbolTable(this);
 214         function->SetModule(module);
 215         functionIdMap[function->FunctionId()] = function;
 216     }
 217     for (auto& arrayType : symbolTable.arrayTypes)
 218     {
 219         arrayType->SetSymbolTable(this);
 220         arrayType->SetModule(module);
 221         arrayType->SetParent(&globalNs);
 222         ArrayTypeSymbol* releasedArrayType = arrayType.release();
 223         arrayTypes.push_back(     std::unique_ptr<ArrayTypeSymbol>(releasedArrayType));
 224         ArrayKey key(releasedArrayType->ElementType(), releasedArrayType->Size());
 225         arrayTypeMap[key] = releasedArrayType;
 226     }
 227     for (auto& derivedType : symbolTable.derivedTypes)
 228     {
 229         derivedType->SetSymbolTable(this);
 230         derivedType->SetModule(module);
 231         derivedType->SetParent(&globalNs);
 232         derivedTypes.push_back(     std::unique_ptr<DerivedTypeSymbol>(derivedType.release()));
 233     }
 234     int nd = derivedTypes.size();
 235     for (int i = 0; i < nd; ++i)
 236     {
 237         DerivedTypeSymbol* derivedTypeSymbol = derivedTypes[i].get();
 238         std::vector<DerivedTypeSymbol*>& derivedTypeVec = derivedTypeMap[derivedTypeSymbol->BaseType()];
 239         int n = derivedTypeVec.size();
 240         bool found = false;
 241         for (int i = 0; i < n; ++i)
 242         {
 243             DerivedTypeSymbol* prevDerivedTypeSymbol = derivedTypeVec[i];
 244             if (prevDerivedTypeSymbol->DerivationRec() == derivedTypeSymbol->DerivationRec())
 245             {
 246                 found = true;
 247                 break;
 248             }
 249         }
 250         if (!found)
 251         {
 252             derivedTypeVec.push_back(derivedTypeSymbol);
 253         }
 254     }
 255     for (auto& s : symbolTable.classTemplateSpecializations)
 256     {
 257         classTemplateSpecializations.push_back(std::move(s));
 258     }
 259     int nc = classTemplateSpecializations.size();
 260     for (int i = 0; i < nc; ++i)
 261     {
 262         ClassTemplateSpecializationSymbol* classTemplateSpecialization = classTemplateSpecializations[i].get();
 263         classTemplateSpecialization->SetSymbolTable(this);
 264         classTemplateSpecialization->SetModule(module);
 265         classTemplateSpecialization->SetParent(&globalNs);
 266         ClassTemplateSpecializationKey key(classTemplateSpecialization->GetClassTemplate(), classTemplateSpecialization->TemplateArgumentTypes());
 267         auto it = classTemplateSpecializationMap.find(key);
 268         if (it == classTemplateSpecializationMap.cend())
 269         {
 270             classTemplateSpecializationMap[key] = classTemplateSpecialization;
 271         }
 272     }
 273     conversionTable.Add(symbolTable.conversionTable);
 274     for (ClassTypeSymbol* polymorphicClass : symbolTable.PolymorphicClasses())
 275     {
 276         AddPolymorphicClass(polymorphicClass);
 277     }
 278     for (ClassTypeSymbol* classHavingStaticConstructor : symbolTable.ClassesHavingStaticConstructor())
 279     {
 280         AddClassHavingStaticConstructor(classHavingStaticConstructor);
 281     }
 282     for (const std::u32string& jsonClass : symbolTable.JsonClasses())
 283     {
 284         AddJsonClass(jsonClass);
 285     }
 286     if (GetGlobalFlag(GlobalFlags::profile))
 287     {
 288         for (const auto& p : symbolTable.profiledFunctionNameMap)
 289         {
 290             const boost::uuids::uuid& functionId = p.first;
 291             const std::u32string& profiledFunctionName = p.second;
 292             MapProfiledFunction(functionId, profiledFunctionName);
 293         }
 294     }
 295     symbolTable.Clear();
 296 }

 364 void SymbolTable::BeginFunction(FunctionNode& functionNode, int32_t functionIndex)
 365 {
 366     FunctionSymbol* functionSymbol = new FunctionSymbol(functionNode.GetSpan(), functionNode.GroupId());
 367     SetFunctionIdFor(functionSymbol);
 368     functionSymbol->SetIndex(functionIndex);
 369     if ((functionNode.GetSpecifiers() & Specifiers::constexpr_) != Specifiers::none)
 370     {
 371         functionSymbol->SetConstExpr();
 372     }
 373     functionSymbol->SetHasSource();
 374     functionSymbol->SetCompileUnit(currentCompileUnit);
 375     functionSymbol->SetSymbolTable(this);
 376     functionSymbol->SetModule(module);
 377     functionSymbol->SetOriginalModule(module);
 378     functionSymbol->SetGroupName(functionNode.GroupId());
 379     if (functionNode.WhereConstraint())
 380     {
 381         CloneContext cloneContext;
 382         functionSymbol->SetConstraint(static_cast<WhereConstraintNode*>(functionNode.WhereConstraint()->Clone(cloneContext)));
 383     }
 384     if (functionSymbol->GroupName() == U"main")
 385     {
 386         if (functionNode.IsProgramMain())
 387         {
 388             functionSymbol->SetCDecl();
 389             functionSymbol->SetProgramMain();
 390         }
 391         else
 392         {
 393             if (mainFunctionSymbol)
 394             {
 395                 throw Exception(module, "already has main function", functionNode.GetSpan(), mainFunctionSymbol->GetSpan());
 396             }
 397             else
 398             {
 399                 mainFunctionSymbol = functionSymbol;
 400             }
 401         }
 402     }
 403     MapNode(&functionNode, functionSymbol);
 404     ContainerScope* functionScope = functionSymbol->GetContainerScope();
 405     ContainerScope* containerScope = container->GetContainerScope();
 406     functionScope->SetParent(containerScope);
 407     BeginContainer(functionSymbol);
 408     parameterIndex = 0;
 409     ResetDeclarationBlockIndex();
 410 }

1666             baseClassTypeId = polymorphicClass->BaseClass()->TypeId();
1667         }
1668         writer.Write(typeId);
1669         writer.Write(vmtObjectName);

1671     }
