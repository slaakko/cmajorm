 13 enum class Derivation : uint8_t 
 14 {
 15     none = 0, constDerivation = 1, lvalueRefDerivation = 2, rvalueRefDerivation = 3, pointerDerivation = 4

 20 typedef llvm::SmallVector<Derivation, 8> DerivationVec;
 21 
 22 struct TypeDerivationRec 
 23 {
 24     DerivationVec derivations;
 25     bool IsEmpty() const { return derivations.empty(); }
 26 };
 27 
 28 inline bool operator==(const TypeDerivationRec& left, const TypeDerivationRec& right)
 29 {
 30     return left.derivations == right.derivations;
 31 }
 32 
 33 inline bool operator!=(const TypeDerivationRec& left, const TypeDerivationRec& right)
 34 {
 35     return !(left == right);
 36 }
