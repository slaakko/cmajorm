 23 void TypedefSymbol::Write(SymbolWriter& writer)
 24 {

 26     writer.GetBinaryWriter().Write(type->TypeId());

 29 void TypedefSymbol::Read(SymbolReader& reader)
 30 {
 31     Symbol::Read(reader);
 32     boost::uuids::uuid typeId;
 33     reader.GetBinaryReader().ReadUuid(typeId);
 34     GetSymbolTable()->EmplaceTypeRequest(this, typeId, 0);
 35 }

 49 void TypedefSymbol::ComputeExportClosure()

 51     if (IsProject())
 52     {

 54         {
 55             type->SetExportComputed();
 56             type->ComputeExportClosure();
 57         }
 58     }
 59 }

 77 std::string TypedefSymbol::Syntax() const
 78 {
 79     std::string syntax = GetSpecifierStr();
 80     if (!syntax.empty())
 81     {
 82         syntax.append(1, ' ');
 83     }
 84     syntax.append("typedef ");
 85     syntax.append(ToUtf8(GetType()->DocName()));
 86     syntax.append(1, ' ');
 87     syntax.append(ToUtf8(DocName()));
 88     syntax.append(1, ';');
 89     return syntax;
 90 }

 92 void TypedefSymbol::SetSpecifiers(Specifiers specifiers)
 93 {
 94     Specifiers accessSpecifiers = specifiers & Specifiers::access_;
 95     SetAccess(accessSpecifiers);
 96     if ((specifiers & Specifiers::static_) != Specifiers::none)
 97     {
 98         throw Exception(GetModule(), "typedef cannot be static", GetSpan());
 99     }
100     if ((specifiers & Specifiers::virtual_) != Specifiers::none)
101     {
102         throw Exception(GetModule(), "typedef cannot be virtual", GetSpan());
103     }
104     if ((specifiers & Specifiers::override_) != Specifiers::none)
105     {
106         throw Exception(GetModule(), "typedef cannot be override", GetSpan());
107     }
108     if ((specifiers & Specifiers::abstract_) != Specifiers::none)
109     {
110         throw Exception(GetModule(), "typedef cannot be abstract", GetSpan());
111     }
112     if ((specifiers & Specifiers::inline_) != Specifiers::none)
113     {
114         throw Exception(GetModule(), "typedef cannot be inline", GetSpan());
115     }
116     if ((specifiers & Specifiers::explicit_) != Specifiers::none)
117     {
118         throw Exception(GetModule(), "typedef cannot be explicit", GetSpan());
119     }
120     if ((specifiers & Specifiers::external_) != Specifiers::none)
121     {
122         throw Exception(GetModule(), "typedef cannot be external", GetSpan());
123     }
124     if ((specifiers & Specifiers::suppress_) != Specifiers::none)
125     {
126         throw Exception(GetModule(), "typedef cannot be suppressed", GetSpan());
127     }
128     if ((specifiers & Specifiers::default_) != Specifiers::none)
129     {
130         throw Exception(GetModule(), "typedef cannot be default", GetSpan());
131     }
132     if ((specifiers & Specifiers::constexpr_) != Specifiers::none)
133     {
134         throw Exception(GetModule(), "typedef cannot be constexpr", GetSpan());
135     }
136     if ((specifiers & Specifiers::cdecl_) != Specifiers::none)
137     {
138         throw Exception(GetModule(), "typedef cannot be cdecl", GetSpan());
139     }
140     if ((specifiers & Specifiers::nothrow_) != Specifiers::none)
141     {
142         throw Exception(GetModule(), "typedef cannot be nothrow", GetSpan());
143     }
144     if ((specifiers & Specifiers::throw_) != Specifiers::none)
145     {
146         throw Exception(GetModule(), "typedef cannot be throw", GetSpan());
147     }
148     if ((specifiers & Specifiers::new_) != Specifiers::none)
149     {
150         throw Exception(GetModule(), "typedef cannot be new", GetSpan());
151     }
152     if ((specifiers & Specifiers::const_) != Specifiers::none)
153     {
154         throw Exception(GetModule(), "typedef cannot be const", GetSpan());
155     }
156     if ((specifiers & Specifiers::unit_test_) != Specifiers::none)
157     {
158         throw Exception(GetModule(), "typedef cannot be unit_test", GetSpan());
159     }
160 }

162 std::unique_ptr<dom::Element> TypedefSymbol::CreateDomElement(TypeMap& typeMap)
163 {
164     std::unique_ptr<dom::Element> element(new dom::Element(U"TypedefSymbol"));
165     if (type)
166     {
167         std::unique_ptr<dom::Element> typeElement(new dom::Element(U"type"));
168         int typeId = typeMap.GetOrInsertType(type);
169         typeElement->SetAttribute(U"ref", U"type_" + ToUtf32(std::to_string(typeId)));
170         element->AppendChild(     std::unique_ptr<dom::Node>(typeElement.release()));
171     }
172     return element;
173 }
