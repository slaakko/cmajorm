12 class TemplateParameterSymbol : public TypeSymbol
13 {
14 public:
15     TemplateParameterSymbol(const Span& span_, const std::u32string& name_);
16     std::u32string FullName() const override { return Name(); }
17     void Write(SymbolWriter& writer);
18     void Read(SymbolReader& reader);
19     void EmplaceType(TypeSymbol* typeSymbol, int index) override;
20     void ComputeExportClosure() override;
21     llvm::Type* IrType(Emitter& emitter) override { Assert(false, "tried to get ir type of template parameter"); return nullptr; }
22     llvm::Constant* CreateDefaultIrValue(Emitter& emitter) override { Assert(false, "tried to create defualt ir value of template parameter"); return nullptr; }
23     TypeSymbol* Unify(TypeSymbol* type, const Span& span) override;
24     bool ContainsTemplateParameter() const override { return true; }
25     bool HasDefault() const { return hasDefault; }
26     void SetHasDefault() { hasDefault = true; }
27     void SetDefaultType(TypeSymbol* defaultType_) { defaultType = defaultType_; }
28     TypeSymbol* DefaultType() { return defaultType; }
29     TypeSymbol* UnifyTemplateArgumentType(SymbolTable& symbolTable, const std::unordered_map<TemplateParameterSymbol*, TypeSymbol*>& templateParameterMap, const Span& span) override;
30     std::u32string Info() const override { return Name(); }
31     const char* ClassName() const override { return "TemplateParameterSymbol"; }
32 private:
33     bool hasDefault;
34     TypeSymbol* defaultType;
35 };
