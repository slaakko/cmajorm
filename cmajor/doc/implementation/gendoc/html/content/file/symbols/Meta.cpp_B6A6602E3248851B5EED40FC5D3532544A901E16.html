 25 std::unique_ptr<Value> IntrinsicFunction::Evaluate(const std::vector<std::unique_ptr<Value>>& arguments, const std::vector<TypeSymbol*>& templateArguments, const Span& span)
 26 {
 27     if (arguments.size() != Arity())
 28     {
 29         throw Exception(module, "wrong number of parameters for intrinsic " + std::string(GroupName()), span);
 30     }
 31     if (templateArguments.size() != NumberOfTypeParameters())
 32     {
 33         throw Exception(module, "wrong number of template type arguments for intrinsic " + std::string(GroupName()), span);
 34     }
 35     return DoEvaluate(arguments, templateArguments, span);
 36 }

 38 FunctionSymbol* CreateIntrinsic(IntrinsicFunction* intrinsic, SymbolTable& symbolTable)
 39 {
 40     FunctionSymbol* fun = new FunctionSymbol(Span(), ToUtf32(intrinsic->GroupName()));
 41     fun->SetSymbolTable(&symbolTable);
 42     fun->SetModule(symbolTable.GetModule());
 43     fun->SetOriginalModule(symbolTable.GetModule());
 44     fun->SetGroupName(ToUtf32(intrinsic->GroupName()));
 45     fun->SetIntrinsic(intrinsic);
 46     fun->SetAccess(SymbolAccess::public_);
 47     fun->SetReturnType(intrinsic->ReturnType(symbolTable));
 48     int n = intrinsic->NumberOfTypeParameters();
 49     for (int i = 0; i < n; ++i)
 50     {
 51         std::u32string p = U"T";
 52         if (i > 0)
 53         {
 54             p.append(ToUtf32(std::to_string(i)));
 55         }
 56         TemplateParameterSymbol* s = new TemplateParameterSymbol(Span(), p);
 57         symbolTable.SetTypeIdFor(s);
 58         s->SetModule(symbolTable.GetModule());
 59         s->SetOriginalModule(symbolTable.GetModule());
 60         s->SetSymbolTable(&symbolTable);
 61         fun->AddMember(s);
 62     }
 63     fun->ComputeName();
 64     return fun;
 65 }
