 34 class Value 
 35 {
 36 public:
 37     Value(const Span& span_, ValueType valueType_);
 38     virtual ~Value();
 39     virtual Value* Clone() const = 0;
 40     virtual Value* As(TypeSymbol* targetType, bool cast, const Span& span, bool dontThrow) const = 0;
 41     virtual llvm::Value* IrValue(Emitter& emitter) = 0;
 42     virtual void Write(BinaryWriter& writer) = 0;
 43     virtual void Read(BinaryReader& reader) = 0;
 44     virtual bool IsComplete() const { return true; }
 45     virtual bool IsScopedValue() const { return false; }
 46     virtual bool IsFunctionGroupValue() const { return false; }
 47     virtual bool IsArrayReferenceValue() const { return false; }
 48     virtual bool IsStructuredReferenceValue() const { return false; }
 49     virtual bool IsComplexValue() const { return false; }
 50     virtual std::string ToString() const { return std::string(); }
 51     virtual TypeSymbol* GetType(SymbolTable* symbolTable) = 0;
 52     const Span& GetSpan() const { return span; }
 53     ValueType GetValueType() const { return valueType; }
 54     std::unique_ptr<dom::Element> ToDomElement();
 55     virtual const char* ClassName() const { return "Value";  }
 56 private:
 57     Span span;
 58     ValueType valueType;
 59 };
