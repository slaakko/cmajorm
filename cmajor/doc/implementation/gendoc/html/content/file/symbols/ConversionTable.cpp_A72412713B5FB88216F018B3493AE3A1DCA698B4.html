11 void ConversionTable::AddConversion(FunctionSymbol* conversion)
12 {
13     ConversionTableEntry entry(conversion->ConversionSourceType()->PlainType(conversion->GetSpan()), conversion->ConversionTargetType()->PlainType(conversion->GetSpan()));
14     conversionMap.insert(std::make_pair(entry, conversion));
15 }
16 
17 void ConversionTable::Add(ConversionTable& that)
18 {
19     for (std::unique_ptr<FunctionSymbol>& conversion : that.generatedConversions)
20     {
21         generatedConversions.push_back(std::move(conversion));
22     }
23     for (const auto& p : that.conversionMap)
24     {
25         conversionMap[p.first] = p.second;
26     }
27 }
28 
29 FunctionSymbol* ConversionTable::GetConversion(TypeSymbol* sourceType, TypeSymbol* targetType, const Span& span) const
30 {
31     TypeSymbol* sourcePlainType = sourceType->PlainType(span);
32     TypeSymbol* targetPlainType = targetType->PlainType(span);
33     ConversionTableEntry entry(sourcePlainType, targetPlainType);
34     auto it = conversionMap.find(entry);
35     if (it != conversionMap.cend())
36     {
37         return it->second;
38     }
39     return nullptr;
40 }
41 
42 void ConversionTable::AddGeneratedConversion(std::unique_ptr<FunctionSymbol>&& generatedConversion)
43 {
44     generatedConversions.push_back(std::move(generatedConversion));
45 }
