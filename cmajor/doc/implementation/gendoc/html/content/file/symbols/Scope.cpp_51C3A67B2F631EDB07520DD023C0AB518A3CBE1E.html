 29 void ContainerScope::Install(Symbol* symbol)
 30 {
 31     auto it = symbolMap.find(symbol->Name());
 32     if (symbol->GetSymbolType() != SymbolType::namespaceSymbol && symbol->GetSymbolType() != SymbolType::declarationBlock && it != symbolMap.cend())
 33     {
 34         Symbol* prev = it->second;
 35         if (prev != symbol)
 36         {
 37             const Span& defined = symbol->GetSpan();
 38             const Span& referenced = prev->GetSpan();
 39             throw Exception(symbol->GetModule(), "symbol '" + ToUtf8(symbol->Name()) + "' already defined", defined, referenced);
 40         }
 41     }
 42     else
 43     {
 44         symbolMap[symbol->Name()] = symbol;
 45         if (symbol->IsContainerSymbol())
 46         {
 47             ContainerSymbol* containerSymbol = static_cast<ContainerSymbol*>(symbol);
 48             containerSymbol->GetContainerScope()->SetParent(this);
 49         }
 50     }
 51 }

 58 int CountQualifiedNameComponents(const std::u32string& qualifiedName)
 59 {
 60     int numComponents = 0;
 61     int componentSize = 0;
 62     int state = 0;
 63     int angleBracketCount = 0;
 64     for (char32_t c : qualifiedName)
 65     {
 66         switch (state)
 67         {
 68             case 0:
 69             {
 70                 if (c == '.')
 71                 {
 72                     ++numComponents;
 73                     componentSize = 0;
 74                 }
 75                 else if (c == '<')
 76                 {
 77                     ++componentSize;
 78                     angleBracketCount = 1;
 79                     state = 1;
 80                 }
 81                 else
 82                 {
 83                     ++componentSize;
 84                 }
 85                 break;
 86             }
 87             case 1:
 88             {
 89                 ++componentSize;
 90                 if (c == '<')
 91                 {
 92                     ++angleBracketCount;
 93                 }
 94                 else if (c == '>')
 95                 {
 96                     --angleBracketCount;
 97                     if (angleBracketCount == 0)
 98                     {
 99                         state = 0;
100                     }
101                 }
102                 break;
103             }
104         }
105     }
106     if (componentSize > 0)
107     {
108         ++numComponents;
109     }
110     return numComponents;
111 }

113 std::vector<std::u32string> ParseQualifiedName(const std::u32string& qualifiedName)
114 {
115     std::vector<std::u32string> components;
116     int state = 0;
117     std::u32string component;
118     int angleBracketCount = 0;
119     for (char32_t c : qualifiedName)
120     {
121         switch (state)
122         {
123             case 0:
124             {
125                 if (c == '.')
126                 {
127                     components.push_back(component);
128                     component.clear();
129                 }
130                 else if (c == '<')
131                 {
132                     component.append(1, c);
133                     angleBracketCount = 1;
134                     state = 1;
135                 }
136                 else
137                 {
138                     component.append(1, c);
139                 }
140                 break;
141             }
142             case 1:
143             {
144                 component.append(1, c);
145                 if (c == '<')
146                 {
147                     ++angleBracketCount;
148                 }
149                 else if (c == '>')
150                 {
151                     --angleBracketCount;
152                     if (angleBracketCount == 0)
153                     {
154                         state = 0;
155                     }
156                 }
157                 break;
158             }
159         }
160     }
161     if (!component.empty())
162     {
163         components.push_back(component);
164     }
165     return components;
166 }

168 Symbol* ContainerScope::Lookup(const std::u32string& name, ScopeLookup lookup) const
169 {
170     int numQualifiedNameComponents = CountQualifiedNameComponents(name);
171     if (numQualifiedNameComponents > 1)
172     {
173         std::vector<std::u32string> components = ParseQualifiedName(name);
174         return LookupQualified(components, lookup);
175     }
176     else
177     {
178         auto it = symbolMap.find(name);
179         if (it != symbolMap.end())
180         {
181             Symbol* s = it->second;
182             return s;
183         }
184         if ((lookup & ScopeLookup::base) != ScopeLookup::none)
185         {
186             if (base)
187             {
188                 Symbol* s = base->Lookup(name, lookup);
189                 if (s)
190                 {
191                     return s;
192                 }
193             }
194         }
195         if ((lookup & ScopeLookup::parent) != ScopeLookup::none)
196         {
197             if (parent)
198             {
199                 Symbol* s = parent->Lookup(name, lookup);
200                 if (s)
201                 {
202                     return s;
203                 }
204             }
205         }
206         return nullptr;
207     }
208 }

210 Symbol* ContainerScope::LookupQualified(const std::vector<std::u32string>& components, ScopeLookup lookup) const
211 {
212     const ContainerScope* scope = this;
213     Symbol* s = nullptr;
214     int n = int(components.size());
215     bool allComponentsMatched = true;
216     for (int i = 0; i < n; ++i)
217     {
218         const std::u32string& component = components[i];
219         if (scope)
220         {
221             s = scope->Lookup(component, ScopeLookup::this_);
222             if (s)
223             {
224                 scope = s->GetContainerScope();
225             }
226             else
227             {
228                 allComponentsMatched = false;
229             }
230         }
231     }
232     if (!s || !allComponentsMatched)
233     {
234         if ((lookup & ScopeLookup::parent) != ScopeLookup::none)
235         {
236             if (parent)
237             {
238                 return parent->LookupQualified(components, lookup);
239             }
240             else
241             {
242                 return nullptr;
243             }
244         }
245     }
246     return s;
247 }

264 NamespaceSymbol* ContainerScope::CreateNamespace(const std::u32string& qualifiedNsName, const Span& span, Module* originalModule)
265 {
266     ContainerScope* scope = this;
267     NamespaceSymbol* parentNs = scope->Ns();
268     std::vector<std::u32string> components = Split(qualifiedNsName, '.');
269     for (const std::u32string& component : components)
270     {
271         Symbol* s = scope->Lookup(component);
272         if (s)
273         {
274             if (s->GetSymbolType() == SymbolType::namespaceSymbol)
275             {
276                 scope = s->GetContainerScope();
277                 parentNs = scope->Ns();
278             }
279             else
280             {
281                 throw Exception(s->GetModule(), "symbol '" + ToUtf8(s->Name()) + "' does not denote a namespace", s->GetSpan());
282             }
283         }
284         else
285         {
286             NamespaceSymbol* newNs = new NamespaceSymbol(span, component);
287             newNs->SetSymbolTable(container->GetSymbolTable());
288             newNs->SetModule(container->GetModule());
289             newNs->SetOriginalModule(originalModule);
290             scope = newNs->GetContainerScope();
291             scope->SetParent(parentNs->GetContainerScope());
292             parentNs->AddMember(newNs);
293             parentNs = newNs;
294         }
295     }
296     return parentNs;
297 }

299 void ContainerScope::CollectViableFunctions(int arity, const std::u32string& groupName, std::unordered_set<ContainerScope*>& scopesLookedUp, ScopeLookup scopeLookup, 
300     std::unordered_set<FunctionSymbol*>& viableFunctions)
301 {
302     if ((scopeLookup & ScopeLookup::this_) != ScopeLookup::none)
303     {
304         if (scopesLookedUp.find(this) == scopesLookedUp.cend())
305         {
306             scopesLookedUp.insert(this);
307             Symbol* symbol = Lookup(groupName);
308             if (symbol)
309             {
310                 if (symbol->GetSymbolType() == SymbolType::functionGroupSymbol)
311                 {
312                     FunctionGroupSymbol* functionGroupSymbol = static_cast<FunctionGroupSymbol*>(symbol);
313                     functionGroupSymbol->CollectViableFunctions(arity, viableFunctions);
314                 }
315             }
316         }
317     }
318     if ((scopeLookup & ScopeLookup::base) != ScopeLookup::none)
319     {
320         if (base)
321         {
322             base->CollectViableFunctions(arity, groupName, scopesLookedUp, scopeLookup, viableFunctions);
323         }
324     }
325     if ((scopeLookup & ScopeLookup::parent) != ScopeLookup::none)
326     {
327         if (parent)
328         {
329             parent->CollectViableFunctions(arity, groupName, scopesLookedUp, scopeLookup, viableFunctions);
330         }
331     }
332 }

362 void FileScope::InstallNamespaceImport(ContainerScope* containerScope, NamespaceImportNode* namespaceImportNode)
363 {
364     try
365     {
366         Assert(containerScope, "container scope is null");
367         std::u32string importedNamespaceName = namespaceImportNode->Ns()->Str();
368         Symbol* symbol = containerScope->Lookup(importedNamespaceName, ScopeLookup::this_and_parent);
369         if (symbol)
370         {
371             if (symbol->GetSymbolType() == SymbolType::namespaceSymbol)
372             {
373                 ContainerScope* symbolContainerScope = symbol->GetContainerScope();
374                 if (std::find(containerScopes.cbegin(), containerScopes.cend(), symbolContainerScope) == containerScopes.cend())
375                 {
376                     containerScopes.push_back(symbolContainerScope);
377                 }
378             }
379             else
380             {
381                 throw Exception(module, "'" + ToUtf8(namespaceImportNode->Ns()->Str()) + "' does not denote a namespace", namespaceImportNode->Ns()->GetSpan());
382             }
383         }
384         else
385         {
386             throw Exception(module, "referred namespace symbol '" + ToUtf8(namespaceImportNode->Ns()->Str()) + "' not found", namespaceImportNode->Ns()->GetSpan());
387         }
388     }

390     {
391         throw ;

400 Symbol* FileScope::Lookup(const std::u32string& name, ScopeLookup lookup) const
401 {
402     if (lookup != ScopeLookup::this_)
403     {
404         throw std::runtime_error("file scope supports only this scope lookup");
405     }
406     std::unordered_set<Symbol*> foundSymbols;
407     auto it = aliasSymbolMap.find(name);
408     if (it != aliasSymbolMap.cend())
409     {
410         Symbol* symbol = it->second;
411         foundSymbols.insert(symbol);
412     }
413     else
414     {
415         for (ContainerScope* containerScope : containerScopes)
416         {
417             Symbol* symbol = containerScope->Lookup(name, ScopeLookup::this_);
418             if (symbol)
419             {
420                 foundSymbols.insert(symbol);
421             }
422         }
423     }
424     if (foundSymbols.empty())
425     {
426         return nullptr;
427     }
428     else if (foundSymbols.size() > 1)
429     {
430         std::string message("reference to object '" + ToUtf8(name) + "' is ambiguous: ");
431         bool first = true;
432         Span span;
433         for (Symbol* symbol : foundSymbols)
434         {
435             if (first)
436             {
437                 first = false;
438                 span = symbol->GetSpan();
439             }
440             else
441             {
442                 message.append(" or ");
443             }
444             message.append(ToUtf8(symbol->FullName()));
445         }
446         throw Exception(module, message, span);
447     }
448     else
449     {
450         return *foundSymbols.begin();
451     }
452 }
