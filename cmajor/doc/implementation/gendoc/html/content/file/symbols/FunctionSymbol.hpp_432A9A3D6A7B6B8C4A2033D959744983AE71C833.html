 53     explicit_ = 1 << 6,

 56     abstract_ = 1 << 9,

 59     conversion = 1 << 12, 
 60     linkOnceOdrLinkage = 1 << 13, 
 61     templateSpecialization = 1 << 14, 
 62     hasTry = 1 << 15,

 67 inline FunctionSymbolFlags operator|(FunctionSymbolFlags left, FunctionSymbolFlags right)
 68 {
 69     return FunctionSymbolFlags(uint32_t(left) | uint32_t(right));
 70 }
 71 
 72 inline FunctionSymbolFlags operator&(FunctionSymbolFlags left, FunctionSymbolFlags right)
 73 {
 74     return FunctionSymbolFlags(uint32_t(left) & uint32_t(right));
 75 }
