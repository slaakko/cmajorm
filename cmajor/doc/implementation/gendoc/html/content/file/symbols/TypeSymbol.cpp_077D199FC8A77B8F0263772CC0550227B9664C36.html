 82 bool TypeSymbol::IsRecursive(TypeSymbol* type, std::unordered_set<TypeSymbol*>& tested)
 83 {
 84     if (tested.find(this) != tested.cend()) return type == this;
 85     tested.insert(this);
 86     return TypesEqual(type, this);
 87 }

 99 llvm::DIType* TypeSymbol::GetDIType(Emitter& emitter)
100 {
101     if (!diType || compileUnitIndex != emitter.CompileUnitIndex())
102     {
103         compileUnitIndex = emitter.CompileUnitIndex();
104         diType = emitter.GetDIType(this);
105         if (!diType)
106         {
107             if (IsClassTypeSymbol())
108             {
109                 ClassTypeSymbol* classTypeSymbol = static_cast<ClassTypeSymbol*>(this);
110                 diType = classTypeSymbol->CreateDIForwardDeclaration(emitter);
111                 emitter.MapFwdDeclaration(diType, this);
112                 emitter.SetDIType(this, diType);
113             }
114             diType = CreateDIType(emitter);
115             emitter.SetDIType(this, diType);
116         }
117     }
118     return diType;
119 }

131 bool CompareTypesForEquality(const TypeSymbol* left, const TypeSymbol* right)
132 {
133     if (left->GetSymbolType() == SymbolType::derivedTypeSymbol && right->GetSymbolType() == SymbolType::derivedTypeSymbol)
134     {
135         const DerivedTypeSymbol* derivedLeft = static_cast<const DerivedTypeSymbol*>(left);
136         const DerivedTypeSymbol* derivedRight = static_cast<const DerivedTypeSymbol*>(right);
137         if (TypesEqual(derivedLeft->BaseType(), derivedRight->BaseType()) && derivedLeft->DerivationRec() == derivedRight->DerivationRec())
138         {
139             return true;
140         }
141     }
142     return false;
143 }
