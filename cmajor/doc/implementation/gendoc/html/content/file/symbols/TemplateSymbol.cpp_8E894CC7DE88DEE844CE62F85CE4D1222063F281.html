 18 TemplateParameterSymbol::TemplateParameterSymbol(const Span& span_, const std::u32string& name_) :
 19     TypeSymbol(SymbolType::templateParameterSymbol, span_, name_), hasDefault(false), defaultType(nullptr)
 20 {
 21 }

 23 void TemplateParameterSymbol::Write(SymbolWriter& writer)
 24 {
 25     TypeSymbol::Write(writer);
 26     writer.GetBinaryWriter().Write(hasDefault);
 27     if (hasDefault)
 28     {
 29         boost::uuids::uuid defaultTypeId = boost::uuids::nil_generator()();
 30         if (defaultType != nullptr)
 31         {
 32             defaultTypeId = defaultType->TypeId();
 33         }
 34         writer.GetBinaryWriter().Write(defaultTypeId);
 35     }
 36 }
 37 
 38 void TemplateParameterSymbol::Read(SymbolReader& reader)
 39 {
 40     TypeSymbol::Read(reader);
 41     hasDefault = reader.GetBinaryReader().ReadBool();
 42     if (hasDefault)
 43     {
 44         boost::uuids::uuid defaultTypeId;
 45         reader.GetBinaryReader().ReadUuid(defaultTypeId);
 46         if (!defaultTypeId.is_nil())
 47         {
 48             GetSymbolTable()->EmplaceTypeRequest(this, defaultTypeId, 0);
 49         }
 50     }
 51 }

 53 void TemplateParameterSymbol::EmplaceType(TypeSymbol* typeSymbol, int index)
 54 {
 55     Assert(index == 0, "invalid emplace type index");
 56     defaultType = typeSymbol;
 57 }

 59 void TemplateParameterSymbol::ComputeExportClosure()
 60 {
 61     if (IsProject())
 62     {
 63         TypeSymbol::ComputeExportClosure();
 64         if (defaultType)
 65         {
 66             if (!defaultType->ExportComputed())
 67             {
 68                 defaultType->SetExportComputed();
 69                 defaultType->ComputeExportClosure();
 70             }
 71         }
 72     }
 73 }

 80 TypeSymbol* TemplateParameterSymbol::UnifyTemplateArgumentType(SymbolTable& symbolTable, const std::unordered_map<TemplateParameterSymbol*, TypeSymbol*>& templateParameterMap, const Span& span)
 81 {
 82     auto it = templateParameterMap.find(this);
 83     if (it != templateParameterMap.cend())
 84     {
 85         return it->second;
 86     }
 87     else
 88     {
 89         return nullptr;
 90     }
 91 }

 97 std::unique_ptr<dom::Element> BoundTemplateParameterSymbol::CreateDomElement(TypeMap& typeMap)
 98 {
 99     std::unique_ptr<dom::Element> element(new dom::Element(U"BoundTemplateParameterSymbol"));
100     if (type)
101     {
102         std::unique_ptr<dom::Element> typeElement(new dom::Element(U"type"));
103         int typeId = typeMap.GetOrInsertType(type);
104         typeElement->SetAttribute(U"ref", U"type_" + ToUtf32(std::to_string(typeId)));
105         element->AppendChild(     std::unique_ptr<dom::Node>(typeElement.release()));
106     }
107     return element;
108 }
