 49 struct ClassTemplateSpecializationKey 
 50 {
 51     ClassTemplateSpecializationKey(ClassTypeSymbol* classTemplate_, const std::vector<TypeSymbol*>& templateArgumentTypes_) :
 52         classTemplate(classTemplate_), templateArgumentTypes(templateArgumentTypes_)
 53     {
 54     }
 55     ClassTypeSymbol* classTemplate;
 56     std::vector<TypeSymbol*> templateArgumentTypes;
 57 };
 58 
 59 struct ClassTemplateSpecializationKeyHash 
 60 {
 61     size_t operator()(const ClassTemplateSpecializationKey& key) const
 62     {
 63         size_t x =      std::hash<ClassTypeSymbol*>()(key.classTemplate);
 64         int n = key.templateArgumentTypes.size();
 65         for (int i = 0; i < n; ++i)
 66         {
 67             x = x ^      std::hash<TypeSymbol*>()(key.templateArgumentTypes[i]);
 68         }
 69         return x;
 70     }
 71 };

 86 struct ArrayKeyHash 
 87 {
 88     size_t operator()(const ArrayKey& key) const
 89     {
 90         size_t x =      std::hash<TypeSymbol*>()(key.elementType);
 91         x = x ^      std::hash<int64_t>()(key.size);
 92         return x;
 93     }
