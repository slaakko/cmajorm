17 Symbol* SymbolReader::ReadSymbol(Symbol* parent)
18 {
19     SymbolType symbolType = static_cast<SymbolType>(astReader.GetBinaryReader().ReadByte());
20     Span span = astReader.ReadSpan();
21     std::u32string name = astReader.GetBinaryReader().ReadUtf32String();
22     Symbol* symbol = SymbolFactory::Instance().CreateSymbol(symbolType, span, name);
23     symbol->SetSymbolTable(symbolTable);
24     symbol->SetModule(module);
25     symbol->SetOriginalModule(module);
26     symbol->SetParent(parent);
27     symbol->Read(*this);
28     return symbol;
29 }

31 ArrayTypeSymbol* SymbolReader::ReadArrayTypeSymbol(Symbol* parent)
32 {
33     Symbol* symbol = ReadSymbol(parent);
34     if (symbol->GetSymbolType() == SymbolType::arrayTypeSymbol)
35     {
36         return static_cast<ArrayTypeSymbol*>(symbol);
37     }
38     else
39     {
40         throw std::runtime_error("internal error: array type symbol expected");
41     }
42 }
