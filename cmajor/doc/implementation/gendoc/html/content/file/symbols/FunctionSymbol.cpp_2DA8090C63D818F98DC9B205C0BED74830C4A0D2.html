 267 void FunctionSymbol::Write(SymbolWriter& writer)
 268 {
 269     ContainerSymbol::Write(writer);
 270     Assert(!functionId.is_nil(), "function id not initialized");
 271     writer.GetBinaryWriter().Write(functionId);
 272     writer.GetBinaryWriter().Write(index);
 273     writer.GetBinaryWriter().Write(groupName);
 274     writer.GetBinaryWriter().Write(static_cast<uint32_t>(flags));
 275     boost::uuids::uuid functionTemplateId = boost::uuids::nil_generator()();
 276     if (functionTemplate)
 277     {
 278         functionTemplateId = functionTemplate->FunctionId();
 279     }
 280     writer.GetBinaryWriter().Write(functionTemplateId);
 281     if (IsFunctionTemplate() || (GetGlobalFlag(GlobalFlags::release) && IsInline()) || IsConstExpr())
 282     {
 283         uint32_t sizePos = writer.GetBinaryWriter().Pos();
 284         uint32_t sizeOfAstNodes = 0;
 285         writer.GetBinaryWriter().Write(sizeOfAstNodes);
 286         uint32_t startPos = writer.GetBinaryWriter().Pos();
 287         usingNodes.Write(writer.GetAstWriter());
 288         Node* node = GetSymbolTable()->GetNode(this);
 289         writer.GetAstWriter().Write(node);
 290         uint32_t endPos = writer.GetBinaryWriter().Pos();
 291         sizeOfAstNodes = endPos - startPos;
 292         writer.GetBinaryWriter().Seek(sizePos);
 293         writer.GetBinaryWriter().Write(sizeOfAstNodes);
 294         writer.GetBinaryWriter().Seek(endPos);
 295     }
 296     boost::uuids::uuid returnTypeId = boost::uuids::nil_generator()();
 297     if (returnType)
 298     {
 299         returnTypeId = returnType->TypeId();
 300     }
 301     writer.GetBinaryWriter().Write(returnTypeId);
 302     writer.GetBinaryWriter().Write(vmtIndex);
 303     writer.GetBinaryWriter().Write(imtIndex);
 304     bool hasReturnParam = returnParam != nullptr;
 305     writer.GetBinaryWriter().Write(hasReturnParam);
 306     if (hasReturnParam)
 307     {
 308         writer.Write(returnParam.get());
 309     }
 310     if (constraint)
 311     {
 312         writer.GetBinaryWriter().Write(true);
 313         writer.GetAstWriter().Write(constraint.get());
 314     }
 315     else
 316     {
 317         writer.GetBinaryWriter().Write(false);
 318     }
 319     uint32_t n = templateArgumentTypes.size();
 320     writer.GetBinaryWriter().WriteULEB128UInt(n);
 321     for (uint32_t i = 0; i < n; ++i)
 322     {
 323         TypeSymbol* templateArgumentType = templateArgumentTypes[i];
 324         writer.GetBinaryWriter().Write(templateArgumentType->TypeId());
 325     }
 326 }

 328 void FunctionSymbol::Read(SymbolReader& reader)
 329 {
 330     ContainerSymbol::Read(reader);
 331     reader.GetBinaryReader().ReadUuid(functionId);
 332     index = reader.GetBinaryReader().ReadInt();
 333     GetSymbolTable()->AddFunctionSymbolToFunctionIdMap(this);
 334     groupName = reader.GetBinaryReader().ReadUtf32String();
 335     flags = static_cast<FunctionSymbolFlags>(reader.GetBinaryReader().ReadUInt());
 336     boost::uuids::uuid functionTemplateId;
 337     reader.GetBinaryReader().ReadUuid(functionTemplateId);
 338     if (!functionTemplateId.is_nil())
 339     {
 340         GetSymbolTable()->EmplaceFunctionRequest(this, functionTemplateId, 0);
 341     }
 342     if (IsFunctionTemplate() || (GetGlobalFlag(GlobalFlags::release) && IsInline()) || IsConstExpr())
 343     {
 344         sizeOfAstNodes = reader.GetBinaryReader().ReadUInt();
 345         astNodesPos = reader.GetBinaryReader().Pos();
 346         reader.GetBinaryReader().Skip(sizeOfAstNodes);
 347         filePathReadFrom = reader.GetBinaryReader().FileName();
 348     }
 349     boost::uuids::uuid returnTypeId;
 350     reader.GetBinaryReader().ReadUuid(returnTypeId);
 351     if (!returnTypeId.is_nil())
 352     {
 353         GetSymbolTable()->EmplaceTypeRequest(this, returnTypeId, 0);
 354     }
 355     vmtIndex = reader.GetBinaryReader().ReadInt();
 356     imtIndex = reader.GetBinaryReader().ReadInt();
 357     bool hasReturnParam = reader.GetBinaryReader().ReadBool();
 358     if (hasReturnParam)
 359     {
 360         returnParam.reset(reader.ReadParameterSymbol(this));
 361     }
 362     bool hasConstraint = reader.GetBinaryReader().ReadBool();
 363     if (hasConstraint)
 364     {
 365         constraint.reset(reader.GetAstReader().ReadConstraintNode());
 366     }
 367     if (IsConversion())
 368     {
 369         reader.AddConversion(this);
 370     }
 371     uint32_t n = reader.GetBinaryReader().ReadULEB128UInt();
 372     templateArgumentTypes.resize(n);
 373     for (uint32_t i = 0; i < n; ++i)
 374     {
 375         boost::uuids::uuid templateArgumentId;
 376         reader.GetBinaryReader().ReadUuid(templateArgumentId);
 377         GetSymbolTable()->EmplaceTypeRequest(this, templateArgumentId, -1 - i);
 378     }
 379 }

 522 void FunctionSymbol::ComputeMangledName()
 523 {
 524     if (IsCDecl())
 525     {
 526         SetMangledName(GroupName());
 527         return;
 528     }
 529     std::u32string mangledName = ToUtf32(TypeString());
 530     if (groupName.find(U"operator") != std::u32string::npos)
 531     {
 532         mangledName.append(1, U'_').append(OperatorMangleMap::Instance().Mangle(GroupName()));
 533     }
 534     else if (groupName.find(U'@') == std::u32string::npos)
 535     {
 536         mangledName.append(1, U'_').append(GroupName());
 537     }
 538     SymbolType symbolType = GetSymbolType();
 539     switch (symbolType)
 540     {
 541         case SymbolType::staticConstructorSymbol: case SymbolType::constructorSymbol: case SymbolType::destructorSymbol: case SymbolType::memberFunctionSymbol:
 542         {
 543             Symbol* parentClass = Parent();
 544             mangledName.append(1, U'_').append(parentClass->SimpleName());
 545         }
 546     }
 547     std::string templateArgumentString;
 548     for (TypeSymbol* templateArgumentType : templateArgumentTypes)
 549     {
 550         templateArgumentString.append("_").append(ToUtf8(templateArgumentType->FullName()));
 551     }
 552     std::string constraintString;
 553     if (Constraint())
 554     {
