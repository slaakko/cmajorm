392 void ArrayTypeDefaultConstructor::GenerateCall(Emitter& emitter, std::vector<GenObject*>& genObjects, OperationFlags flags, const Span& span)
393 {
394     Assert(genObjects.size() == 1, "default constructor needs one object");
395     genObjects[0]->Load(emitter, OperationFlags::addr);
396     llvm::Value* ptr = emitter.Stack().Pop();
397     emitter.Builder().CreateStore(emitter.Builder().getInt64(0), loopVar->IrObject());
398     llvm::Value* size = emitter.Builder().getInt64(arrayType->Size());
399     llvm::BasicBlock* loop = llvm::BasicBlock::Create(emitter.Context(), "loop", emitter.Function());
400     llvm::BasicBlock* init = llvm::BasicBlock::Create(emitter.Context(), "init", emitter.Function());
401     llvm::BasicBlock* next = llvm::BasicBlock::Create(emitter.Context(), "next", emitter.Function());
402     emitter.Builder().CreateBr(loop);
403     emitter.SetCurrentBasicBlock(loop);
404     llvm::Value* index = emitter.Builder().CreateLoad(loopVar->IrObject());
405     llvm::Value* less = emitter.Builder().CreateICmpULT(index, size);
406     emitter.Builder().CreateCondBr(less, init, next);
407     emitter.SetCurrentBasicBlock(init);
408     ArgVector elementIndeces;
409     elementIndeces.push_back(emitter.Builder().getInt64(0));
410     elementIndeces.push_back(index);
411     llvm::Value* elementPtr = emitter.Builder().CreateGEP(ptr, elementIndeces);
412     LlvmValue elementPtrValue(elementPtr);
413     std::vector<GenObject*> elementGenObjects;
414     elementGenObjects.push_back(&elementPtrValue);
415     elementTypeDefaultConstructor->GenerateCall(emitter, elementGenObjects, OperationFlags::none, span);
416     llvm::Value* nextI = emitter.Builder().CreateAdd(index, emitter.Builder().getInt64(1));
417     emitter.Builder().CreateStore(nextI, loopVar->IrObject());
418     emitter.Builder().CreateBr(loop);
419     emitter.SetCurrentBasicBlock(next);
420 }

436 void ArrayTypeCopyConstructor::GenerateCall(Emitter& emitter, std::vector<GenObject*>& genObjects, OperationFlags flags, const Span& span)
437 {
438     Assert(genObjects.size() == 2, "copy constructor needs two objects");
439     genObjects[0]->Load(emitter, OperationFlags::addr);
440     llvm::Value* ptr = emitter.Stack().Pop();
441     genObjects[1]->Load(emitter, OperationFlags::none);
442     llvm::Value* sourcePtr = emitter.Stack().Pop();
443     emitter.Builder().CreateStore(emitter.Builder().getInt64(0), loopVar->IrObject());
444     llvm::Value* size = emitter.Builder().getInt64(arrayType->Size());
445     llvm::BasicBlock* loop = llvm::BasicBlock::Create(emitter.Context(), "loop", emitter.Function());
446     llvm::BasicBlock* init = llvm::BasicBlock::Create(emitter.Context(), "init", emitter.Function());
447     llvm::BasicBlock* next = llvm::BasicBlock::Create(emitter.Context(), "next", emitter.Function());
448     emitter.Builder().CreateBr(loop);
449     emitter.SetCurrentBasicBlock(loop);
450     llvm::Value* index = emitter.Builder().CreateLoad(loopVar->IrObject());
451     llvm::Value* less = emitter.Builder().CreateICmpULT(index, size);
452     emitter.Builder().CreateCondBr(less, init, next);
453     emitter.SetCurrentBasicBlock(init);
454     ArgVector elementIndeces;
455     elementIndeces.push_back(emitter.Builder().getInt64(0));
456     elementIndeces.push_back(index);
457     llvm::Value* elementPtr = emitter.Builder().CreateGEP(ptr, elementIndeces);
458     LlvmValue elementPtrValue(elementPtr);
459     std::vector<GenObject*> elementGenObjects;
460     elementGenObjects.push_back(&elementPtrValue);
461     llvm::Value* sourceElementPtr = emitter.Builder().CreateGEP(sourcePtr, elementIndeces);
462     llvm::Value* sourceElementValue = sourceElementPtr;
463     TypeSymbol* elementType = arrayType->ElementType();
464     if (elementType->IsBasicTypeSymbol() || elementType->IsPointerType() || elementType->GetSymbolType() == SymbolType::delegateTypeSymbol)
465     {
466         sourceElementValue = emitter.Builder().CreateLoad(sourceElementPtr);
467     }
468     LlvmValue sourceValue(sourceElementValue);
469     elementGenObjects.push_back(&sourceValue);
470     elementTypeCopyConstructor->GenerateCall(emitter, elementGenObjects, OperationFlags::none, span);
471     llvm::Value* nextI = emitter.Builder().CreateAdd(index, emitter.Builder().getInt64(1));
472     emitter.Builder().CreateStore(nextI, loopVar->IrObject());
473     emitter.Builder().CreateBr(loop);
474     emitter.SetCurrentBasicBlock(next);
475 }

668     {
