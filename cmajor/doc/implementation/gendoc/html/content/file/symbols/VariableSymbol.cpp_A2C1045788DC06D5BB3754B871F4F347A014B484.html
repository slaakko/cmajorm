161 std::string MemberVariableSymbol::Syntax() const
162 {
163     std::string syntax = GetSpecifierStr();
164     if (!syntax.empty())
165     {
166         syntax.append(1, ' ');
167     }
168     syntax.append(ToUtf8(GetType()->DocName()));
169     syntax.append(1, ' ');
170     syntax.append(ToUtf8(DocName()));
171     syntax.append(1, ';');
172     return syntax;

175 void MemberVariableSymbol::SetSpecifiers(Specifiers specifiers)
176 {
177     Specifiers accessSpecifiers = specifiers & Specifiers::access_;
178     SetAccess(accessSpecifiers);
179     if ((specifiers & Specifiers::static_) != Specifiers::none)
180     {
181         SetStatic();
182     }
183     if ((specifiers & Specifiers::virtual_) != Specifiers::none)
184     {
185         throw Exception(GetModule(), "member variable cannot be virtual", GetSpan());
186     }
187     if ((specifiers & Specifiers::override_) != Specifiers::none)
188     {
189         throw Exception(GetModule(), "member variable cannot be override", GetSpan());
190     }
191     if ((specifiers & Specifiers::abstract_) != Specifiers::none)
192     {
193         throw Exception(GetModule(), "member variable cannot be abstract", GetSpan());
194     }
195     if ((specifiers & Specifiers::inline_) != Specifiers::none)
196     {
197         throw Exception(GetModule(), "member variable cannot be inline", GetSpan());
198     }
199     if ((specifiers & Specifiers::explicit_) != Specifiers::none)
200     {
201         throw Exception(GetModule(), "member variable cannot be explicit", GetSpan());
202     }
203     if ((specifiers & Specifiers::external_) != Specifiers::none)
204     {
205         throw Exception(GetModule(), "member variable cannot be external", GetSpan());
206     }
207     if ((specifiers & Specifiers::suppress_) != Specifiers::none)
208     {
209         throw Exception(GetModule(), "member variable cannot be suppressed", GetSpan());
210     }
211     if ((specifiers & Specifiers::default_) != Specifiers::none)
212     {
213         throw Exception(GetModule(), "member variable cannot be default", GetSpan());
214     }
215     if ((specifiers & Specifiers::constexpr_) != Specifiers::none)
216     {
217         throw Exception(GetModule(), "member variable cannot be constexpr", GetSpan());
218     }
219     if ((specifiers & Specifiers::cdecl_) != Specifiers::none)
220     {
221         throw Exception(GetModule(), "member variable cannot be cdecl", GetSpan());
222     }
223     if ((specifiers & Specifiers::nothrow_) != Specifiers::none)
224     {
225         throw Exception(GetModule(), "member variable cannot be nothrow", GetSpan());
226     }
227     if ((specifiers & Specifiers::throw_) != Specifiers::none)
228     {
229         throw Exception(GetModule(), "member variable cannot be throw", GetSpan());
230     }
231     if ((specifiers & Specifiers::new_) != Specifiers::none)
232     {
233         throw Exception(GetModule(), "member variable cannot be new", GetSpan());
234     }
235     if ((specifiers & Specifiers::const_) != Specifiers::none)
236     {
237         throw Exception(GetModule(), "member variable cannot be const", GetSpan());
238     }
239     if ((specifiers & Specifiers::unit_test_) != Specifiers::none)
240     {
241         throw Exception(GetModule(), "member variable cannot be unit_test", GetSpan());
242     }
243 }

245 llvm::DIDerivedType* MemberVariableSymbol::GetDIMemberType(Emitter& emitter, uint64_t offsetInBits)
246 {
247     if (!diMemberType || compileUnitIndex != emitter.CompileUnitIndex())
248     {
249         uint64_t sizeInBits = GetType()->SizeInBits(emitter);
250         uint32_t alignInBits = GetType()->AlignmentInBits(emitter);
251         llvm::DINode::DIFlags flags = llvm::DINode::DIFlags::FlagZero;
252         llvm::DIType* scope = nullptr;
253         Symbol* parent = Parent();
254         if (parent->IsClassTypeSymbol())
255         {
256             ClassTypeSymbol* cls = static_cast<ClassTypeSymbol*>(parent);
257             scope = cls->GetDIType(emitter);
258         }
259         diMemberType = emitter.DIBuilder()->createMemberType(scope, ToUtf8(Name()), emitter.GetFile(GetSpan().FileIndex()), GetSpan().LineNumber(), sizeInBits, alignInBits, offsetInBits, flags, 
260             GetType()->GetDIType(emitter));
261         compileUnitIndex = emitter.CompileUnitIndex();
262     }
263     return diMemberType;
264 }
