 42 enum class ClassTypeSymbolFlags : uint8_t 
 43 {
 44     none = 0, 
 45     abstract_ = 1 << 0, 
 46     polymorphic = 1 << 1, 
 47     vmtInitialized = 1 << 2, 
 48     imtsInitialized = 1 << 3, 
 49     layoutsComputed = 1 << 4, 
 50     vmtObjectCreated = 1 << 5, 
 51     staticObjectCreated = 1 << 6, 
 52     statementsNotBound = 1 << 7
 53 };
 54 
 55 inline ClassTypeSymbolFlags operator|(ClassTypeSymbolFlags left, ClassTypeSymbolFlags right)
 56 {
 57     return ClassTypeSymbolFlags(uint8_t(left) | uint8_t(right));
 58 }
 59 
 60 inline ClassTypeSymbolFlags operator&(ClassTypeSymbolFlags left, ClassTypeSymbolFlags right)
 61 {
 62     return ClassTypeSymbolFlags(uint8_t(left) & uint8_t(right));
 63 }
 64 
 65 inline ClassTypeSymbolFlags operator~(ClassTypeSymbolFlags operand)
 66 {
 67     return ClassTypeSymbolFlags(~uint8_t(operand));
 68 }
