185             }
186         }
187         writer.Write(filePath);

235 void Visit(std::vector<Module*>& finishReadOrder, Module* module, std::unordered_set<Module*>& visited, std::unordered_set<Module*>& tempVisit, 
236     std::unordered_map<Module*, ModuleDependency*>& dependencyMap, const Module* rootModule)
237 {
238     if (tempVisit.find(module) == tempVisit.cend())
239     {
240         if (visited.find(module) == visited.cend())
241         {
242             tempVisit.insert(module);
243             auto i = dependencyMap.find(module);
244             if (i != dependencyMap.cend())
245             {
246                 ModuleDependency* dependency = i->second;
247                 for (Module* dependentAssembly : dependency->ReferencedModules())
248                 {
249                     Visit(finishReadOrder, dependentAssembly, visited, tempVisit, dependencyMap, rootModule);
250                 }
251                 tempVisit.erase(module);
252                 visited.insert(module);
253                 finishReadOrder.push_back(module);
254             }
255             else
256             {
257                 throw std::runtime_error("module '" + ToUtf8(module->Name()) + "' not found in dependencies of module '" + ToUtf8(rootModule->Name()) + "'");
258             }
259         }
260     }
261     else
262     {
263         throw std::runtime_error("circular module dependency '" + ToUtf8(module->Name()) + "' detected in dependencies of module '" + ToUtf8(rootModule->Name()) + "'");
264     }
265 }

267 std::vector<Module*> CreateFinishReadOrder(std::vector<Module*>& modules, std::unordered_map<Module*, ModuleDependency*>& dependencyMap, const Module* rootModule)
268 {
269     std::vector<Module*> finishReadOrder;
270     std::unordered_set<Module*> visited;
271     std::unordered_set<Module*> tempVisit;
272     for (Module* module : modules)
273     {
274         if (visited.find(module) == visited.cend())
275         {
276             Visit(finishReadOrder, module, visited, tempVisit, dependencyMap, rootModule);
277         }
278     }
279     return finishReadOrder;
280 }

288 Module::Module(const std::string& filePath, std::vector<ClassTypeSymbol*>& classTypes, std::vector<ClassTemplateSpecializationSymbol*>& classTemplateSpecializations)  :
289     format(currentModuleFormat), flags(ModuleFlags::none), name(), originalFilePath(), filePathReadFrom(), referenceFilePaths(), moduleDependency(this), symbolTablePos(0), symbolTable(this), 
290     directoryPath(), libraryFilePaths(), moduleId(-1), rootModule(nullptr), logStreamId(0)
291 {
292     SymbolReader reader(filePath);
293     ModuleTag expectedTag;
294     ModuleTag tag;
295     tag.Read(reader);
296     for (int i = 0; i < 3; ++i)
297     {
298         if (tag.bytes[i] != expectedTag.bytes[i])
299         {
300             throw std::runtime_error("Invalid Cmajor module tag read from file '" + reader.GetBinaryReader().FileName() + "', please rebuild module from sources");
301         }
302     }
303     if (tag.bytes[3] != expectedTag.bytes[3])
304     {
305         throw std::runtime_error("Cmajor module format version mismatch reading from file '" + reader.GetBinaryReader().FileName() + 
306             "': format " + std::string(1, expectedTag.bytes[3]) + " expected, format " + std::string(1, tag.bytes[3]) + " read, please rebuild module from sources");
307     }
308     flags = ModuleFlags(reader.GetBinaryReader().ReadByte());
309     name = reader.GetBinaryReader().ReadUtf32String();
310     std::unordered_set<std::string> importSet;
311     Module* rootModule = this;
312     std::vector<Module*> modules;
313     std::unordered_map<std::string, ModuleDependency*> moduleDependencyMap;
314     std::unordered_map<std::string, Module*> readMap;
315     if (SystemModuleSet::Instance().IsSystemModule(name)) SetSystemModule();
316     SymbolReader reader2(filePath);
317     ReadHeader(reader2, rootModule, importSet, modules, moduleDependencyMap, readMap);
318     moduleDependencyMap[originalFilePath] = &moduleDependency;
319     std::unordered_map<Module*, ModuleDependency*> dependencyMap;
320     for (const auto& p : moduleDependencyMap)
321     {
322         dependencyMap[p.second->GetModule()] = p.second;
323     }
324     std::vector<Module*> finishReadOrder = CreateFinishReadOrder(modules, dependencyMap, rootModule);
325     if (!fileTable.IsEmpty())
326     {
327 #ifdef _WIN32
328         libraryFilePath = GetFullPath(boost::filesystem::path(originalFilePath).replace_extension(".lib").generic_string());
329 #else
330         libraryFilePath = GetFullPath(boost::filesystem::path(originalFilePath).replace_extension(".a").generic_string());
331 #endif
332     }
333     for (Module* module : finishReadOrder)
334     {
335         if (!module->LibraryFilePath().empty() && !module->fileTable.IsEmpty())
336         {
337             libraryFilePaths.push_back(module->LibraryFilePath());
338         }
339     }
340     FinishReads(rootModule, finishReadOrder, finishReadOrder.size() - 2, classTypes, classTemplateSpecializations);
341     SymbolReader reader3(filePathReadFrom);
342     reader3.SetProjectBitForSymbols();
343     reader3.SetModule(this);
344     reader3.GetBinaryReader().Skip(symbolTablePos);
345     SymbolTable st(this);
346     st.Copy(symbolTable);
347     st.Read(reader3);
348     symbolTable.Import(st);
349     classTypes.insert(classTypes.end(), reader3.ClassTypes().begin(), reader3.ClassTypes().end());
350     classTemplateSpecializations.insert(classTemplateSpecializations.end(), reader3.ClassTemplateSpecializations().begin(), reader3.ClassTemplateSpecializations().end());
351 }

363 void Module::PrepareForCompilation(const std::vector<std::string>& references, std::vector<ClassTypeSymbol*>& classTypes, 
364     std::vector<ClassTemplateSpecializationSymbol*>& classTemplateSpecializations)
365 {
366     boost::filesystem::path mfd = originalFilePath;
367     mfd.remove_filename();
368     boost::filesystem::create_directories(mfd);
369     SetDirectoryPath(GetFullPath(mfd.generic_string()));
370     if (name == U"System.Core")
371     {
372         InitCoreSymbolTable(symbolTable);
373     }
374     std::unordered_set<std::string> importSet;
375     Module* rootModule = this;
376     moduleId = rootModule->RegisterFileTable(&fileTable);
377     std::vector<Module*> modules;
378     std::unordered_map<std::string, ModuleDependency*> moduleDependencyMap;
379     std::unordered_map<std::string, Module*> readMap;
380     ImportModules(references, importSet, rootModule, modules, moduleDependencyMap, readMap);
381     modules.push_back(this);
382     moduleDependencyMap[originalFilePath] = &moduleDependency;
383     std::unordered_map<Module*, ModuleDependency*> dependencyMap;
384     for (const auto& p : moduleDependencyMap)
385     {
386         dependencyMap[p.second->GetModule()] = p.second;
387     }
388     std::vector<Module*> finishReadOrder = CreateFinishReadOrder(modules, dependencyMap, rootModule);
389     if (!this->fileTable.IsEmpty())
390     {
391 #ifdef _WIN32
392         libraryFilePath = GetFullPath(boost::filesystem::path(originalFilePath).replace_extension(".lib").generic_string());
393 #else
394         libraryFilePath = GetFullPath(boost::filesystem::path(originalFilePath).replace_extension(".a").generic_string());
395 #endif
396     }
397     for (Module* module : finishReadOrder)
398     {
399         if (!module->LibraryFilePath().empty())
400         {
401             libraryFilePaths.push_back(module->LibraryFilePath());
402         }
403     }
404     FinishReads(rootModule, finishReadOrder, finishReadOrder.size() - 2, classTypes, classTemplateSpecializations);
405 }

434 void Module::Write(SymbolWriter& writer)
435 {
436     ModuleTag tag;
437     tag.Write(writer);
438     writer.GetBinaryWriter().Write(static_cast<uint8_t>(flags));
439     writer.GetBinaryWriter().Write(name);
440     writer.GetBinaryWriter().Write(originalFilePath);
441     uint32_t nr = referencedModules.size();
442     writer.GetBinaryWriter().WriteULEB128UInt(nr);
443     for (uint32_t i = 0; i < nr; ++i)
444     {
445         const std::unique_ptr<Module>& referencedModule = referencedModules[i];
446         writer.GetBinaryWriter().Write(referencedModule->OriginalFilePath());
447     }
448     fileTable.Write(writer.GetBinaryWriter(), IsSystemModule());
449     uint32_t efn = exportedFunctions.size();
450     writer.GetBinaryWriter().WriteULEB128UInt(efn);
451     for (uint32_t i = 0; i < efn; ++i)
452     {
453         writer.GetBinaryWriter().Write(exportedFunctions[i]);
454     }
455     uint32_t edn = exportedData.size();
456     writer.GetBinaryWriter().WriteULEB128UInt(edn);
457     for (uint32_t i = 0; i < edn; ++i)
458     {
459         writer.GetBinaryWriter().Write(exportedData[i]);
460     }
461     symbolTable.Write(writer);
462 }

464 void Module::ReadHeader(SymbolReader& reader, Module* rootModule, std::unordered_set<std::string>& importSet, std::vector<Module*>& modules, 
465     std::unordered_map<std::string, ModuleDependency*>& dependencyMap, std::unordered_map<std::string, Module*>& readMap)
466 {
467     ModuleTag expectedTag;
468     ModuleTag tag;
469     tag.Read(reader);
470     for (int i = 0; i < 3; ++i)
471     {
472         if (tag.bytes[i] != expectedTag.bytes[i])
473         {
474             throw std::runtime_error("Invalid Cmajor module tag read from file '" + reader.GetBinaryReader().FileName() + "', please rebuild module from sources");
475         }
476     }
477     if (tag.bytes[3] != expectedTag.bytes[3])
478     {
479         throw std::runtime_error("Cmajor module format version mismatch reading from file '" + reader.GetBinaryReader().FileName() + 
480             "': format " + std::string(1, expectedTag.bytes[3]) + " expected, format " + std::string(1, tag.bytes[3]) + " read, please rebuild module from sources");
481     }
482     moduleId = rootModule->RegisterFileTable(&fileTable);
483     flags = ModuleFlags(reader.GetBinaryReader().ReadByte());
484     name = reader.GetBinaryReader().ReadUtf32String();
485     originalFilePath = reader.GetBinaryReader().ReadUtf8String();
486     if (dependencyMap.find(originalFilePath) == dependencyMap.cend())
487     {
488         modules.push_back(this);
489         dependencyMap[originalFilePath] = &moduleDependency;
490     }
491     filePathReadFrom = GetFullPath(reader.GetBinaryReader().FileName());
492     uint32_t nr = reader.GetBinaryReader().ReadULEB128UInt();
493     for (uint32_t i = 0; i < nr; ++i)
494     {
495         referenceFilePaths.push_back(reader.GetBinaryReader().ReadUtf8String());
496     }
497     fileTable.Read(reader.GetBinaryReader(), IsSystemModule());
498     if (!fileTable.IsEmpty())
499     {
500 #ifdef _WIN32
501         libraryFilePath = GetFullPath(boost::filesystem::path(filePathReadFrom).replace_extension(".lib").generic_string());
502 #else
503         libraryFilePath = GetFullPath(boost::filesystem::path(filePathReadFrom).replace_extension(".a").generic_string());
504 #endif
505     }
506     uint32_t efn = reader.GetBinaryReader().ReadULEB128UInt();
507     for (uint32_t i = 0; i < efn; ++i)
508     {
509         exportedFunctions.push_back(reader.GetBinaryReader().ReadUtf8String());
510     }
511     for (const std::string& exportedFunction : exportedFunctions)
512     {
513         rootModule->allExportedFunctions.push_back(exportedFunction);
514     }
515     uint32_t edn = reader.GetBinaryReader().ReadULEB128UInt();
516     for (uint32_t i = 0; i < edn; ++i)
517     {
518         exportedData.push_back(reader.GetBinaryReader().ReadUtf8String());
519     }
520     for (const std::string& data : exportedData)
521     {
522         rootModule->allExportedData.push_back(data);
523     }
524     CheckUpToDate();
525     symbolTablePos = reader.GetBinaryReader().Pos();
526     ImportModules(rootModule, importSet, modules, dependencyMap, readMap);
527 }

529 void Module::FinishReads(Module* rootModule, std::vector<Module*>& finishReadOrder, int prevModuleIndex, 
530     std::vector<ClassTypeSymbol*>& classTypes, std::vector<ClassTemplateSpecializationSymbol*>& classTemplateSpecializations)
531 {
532     Module* prevModule = nullptr;
533     if (prevModuleIndex >= 0)
534     {
535         prevModule = finishReadOrder[prevModuleIndex];
536     }
537     if (prevModule)
538     {
539         prevModule->FinishReads(rootModule, finishReadOrder, prevModuleIndex - 1, classTypes, classTemplateSpecializations);
540         symbolTable.Import(prevModule->symbolTable);
541     }
542     if (this != rootModule)
543     {
544         SymbolReader reader(filePathReadFrom);
545         reader.SetModule(this);
546         reader.GetBinaryReader().Skip(symbolTablePos);
547         symbolTable.Read(reader);
548         classTypes.insert(classTypes.end(), reader.ClassTypes().begin(), reader.ClassTypes().end());
549         classTemplateSpecializations.insert(classTemplateSpecializations.end(), reader.ClassTemplateSpecializations().begin(), reader.ClassTemplateSpecializations().end());
550     }

575 void Module::Import(const std::vector<std::string>& references, std::unordered_set<std::string>& importSet, Module* rootModule, std::vector<Module*>& modules, 
576     std::unordered_map<std::string, ModuleDependency*>& moduleDependencyMap, std::unordered_map<std::string, Module*>& readMap)
577 {
578     for (const std::string& reference : references)
579     {
580         if (importSet.find(reference) == importSet.cend())
581         {
582             importSet.insert(reference);
583             std::unique_ptr<Module> referencedModule(new Module());
584             referencedModule->SetRootModule(rootModule);
585             rootModule->allRefModules.push_back(referencedModule.get());
586             std::string config = GetConfig();
587             boost::filesystem::path mfn = boost::filesystem::path(reference).filename();
588             boost::filesystem::path mfp;
589             std::string searchedDirectories;
590             if (!rootModule->IsSystemModule())
591             {
592                 mfp = CmajorSystemLibDir(config);
593                 searchedDirectories.append("\n").append(mfp.generic_string());
594                 mfp /= mfn;
595                 if (!boost::filesystem::exists(mfp))
596                 {
597                     mfp = reference;
598                     if (!boost::filesystem::exists(mfp))
599                     {
600                         boost::filesystem::path mrd = mfp;
601                         mrd.remove_filename();
602                         searchedDirectories.append("\n").append(mrd.generic_string());
603                         throw std::runtime_error("Could not find module reference '" + mfn.generic_string() + "'.\nDirectories searched:\n" + searchedDirectories);
604                     }
605                 }
606             }
607             else
608             {
609                 mfp = reference;
610                 if (!boost::filesystem::exists(mfp))
611                 {
612                     boost::filesystem::path mrd = mfp;
613                     mrd.remove_filename();
614                     searchedDirectories.append("\n").append(mrd.generic_string());
615                     throw std::runtime_error("Could not find module reference '" + mfn.generic_string() + "'.\nDirectories searched:\n" + searchedDirectories);
616                 }
617             }
618             std::string moduleFilePath = GetFullPath(mfp.generic_string());
619             readMap[moduleFilePath] = referencedModule.get();
620             importSet.insert(moduleFilePath);
621             SymbolReader reader(moduleFilePath);
622             referencedModule->ReadHeader(reader, rootModule, importSet, modules, moduleDependencyMap, readMap);
623             Module* refModule = referencedModule.get();
624             moduleDependency.AddReferencedModule(refModule);
625             referencedModules.push_back(std::move(referencedModule));
626             Import(refModule->referenceFilePaths, importSet, rootModule, modules, moduleDependencyMap, readMap);
627         }
628         else
629         {
630             std::string config = GetConfig();
631             boost::filesystem::path mfn = boost::filesystem::path(reference).filename();
632             boost::filesystem::path mfp;
633             std::string searchedDirectories;
634             if (!rootModule->IsSystemModule())
635             {
636                 mfp = CmajorSystemLibDir(config);
637                 mfp /= mfn;
638                 if (!boost::filesystem::exists(mfp))
639                 {
640                     mfp = reference;
641                     if (!boost::filesystem::exists(mfp))
642                     {
643                         boost::filesystem::path mrd = mfp;
644                         mrd.remove_filename();
645                         searchedDirectories.append("\n").append(mrd.generic_string());
646                         throw std::runtime_error("Could not find module reference '" + mfn.generic_string() + "'.\nDirectories searched:\n" + searchedDirectories);
647                     }
648                 }
649             }
650             else
651             {
652                 mfp = reference;
653                 if (!boost::filesystem::exists(mfp))
654                 {
655                     boost::filesystem::path mrd = mfp;
656                     mrd.remove_filename();
657                     searchedDirectories.append("\n").append(mrd.generic_string());
658                     throw std::runtime_error("Could not find module reference '" + mfn.generic_string() + "'.\nDirectories searched:\n" + searchedDirectories);
659                 }
660             }
661             std::string moduleFilePath = GetFullPath(mfp.generic_string());
662             auto it = readMap.find(moduleFilePath);
663             if (it != readMap.cend())
664             {
665                 Module* referencedModule = it->second;
666                 moduleDependency.AddReferencedModule(referencedModule);
667             }
668             else
669             {
670                 throw std::runtime_error("module file path '" + moduleFilePath + "' not found from module read map for module '" + ToUtf8(name) + "'");
671             }
672         }
673     }

696     formatter.WriteLine("file path read from: " + filePathReadFrom);

699         formatter.WriteLine("library file path: " + libraryFilePath);

729         }
730     }

755         for (FunctionSymbol* function : collector.Functions())
756         {
757             formatter.WriteLine();
758             function->Dump(formatter);
759         }
760     }

762     {
763         formatter.WriteLine();

823 void Module::CheckUpToDate()
824 {
825     if (fileTable.IsEmpty()) return;
826     boost::filesystem::path libDirPath = boost::filesystem::path(originalFilePath).parent_path();
827     int n = fileTable.NumFilePaths();
828     for (int i = 0; i < n; ++i)
829     {
830         boost::filesystem::path sfp(fileTable.GetFilePath(i));
831         if (boost::filesystem::exists(sfp))
832         {
833 #ifdef _WIN32
834             boost::filesystem::path objectFilePath = libDirPath / sfp.filename().replace_extension(".obj");
835 #else
836             boost::filesystem::path objectFilePath = libDirPath / sfp.filename().replace_extension(".o");
837 #endif
838             if (boost::filesystem::exists(objectFilePath))
839             {
840                 if (boost::filesystem::last_write_time(sfp) > boost::filesystem::last_write_time(objectFilePath))
841                 {
842                     Warning warning(name, "source file '" + GetFullPath(sfp.generic_string()) + "' is more recent than object file '" + 
843                         GetFullPath(objectFilePath.generic_string()) + "'");
844                     bool found = false;
845                     for (const Warning& prev : warnings.Warnings())
846                     {
847                         if (prev.Message() == warning.Message())
848                         {
849                             found = true;
850                             break;
851                         }
852                     }
853                     if (!found)
854                     {
855                         warnings.AddWarning(warning);
856                     }
857                 }
858             }
859         }
860     }
