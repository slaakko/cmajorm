 43 std::string Expand(Module* module, const std::string& errorMessage, const Span& span, const std::vector<Span>& references, const std::string& title)
 44 {
 45     std::vector<Span> referenceSpans = references;
 46     referenceSpans.erase(std::unique(referenceSpans.begin(), referenceSpans.end()), referenceSpans.end());
 47     std::string expandedMessage = title + ": " + errorMessage;
 48     if (span.Valid())
 49     {
 50         if (!module)
 51         {
 52             throw std::runtime_error("module not set");
 53         }
 54         std::string fileName = module->GetFilePath(span.FileIndex());
 55         if (!fileName.empty())
 56         {
 57             expandedMessage.append(" (file '" + fileName + "', line " + std::to_string(span.LineNumber()) + ")");
 58             MappedInputFile file(fileName);
 59             std::string s(file.Begin(), file.End());
 60             std::u32string t(ToUtf32(s));
 61             expandedMessage.append(":\n").append(ToUtf8(cmajor::parsing::GetErrorLines(&t[0], &t[0] + t.length(), span)));
 62         }
 63     }
 64     for (const Span& referenceSpan : referenceSpans)
 65     {
 66         if (!referenceSpan.Valid()) continue;
 67         if (referenceSpan == span) continue;
 68         if (!module)
 69         {
 70             throw std::runtime_error("module not set");
 71         }
 72         std::string fileName = module->GetFilePath(referenceSpan.FileIndex());
 73         if (!fileName.empty())
 74         {
 75             expandedMessage.append("\nsee reference to file '" + fileName + "', line " + std::to_string(referenceSpan.LineNumber()));
 76             MappedInputFile file(fileName);
 77             std::string s(file.Begin(), file.End());
 78             std::u32string t(ToUtf32(s));
 79             expandedMessage.append(":\n").append(ToUtf8(cmajor::parsing::GetErrorLines(&t[0], &t[0] + t.length(), referenceSpan)));
 80         }
 81     }
 82     return expandedMessage;
 83 }

 85 std::unique_ptr<JsonObject> SpanToJson(Module* module, const Span& span)
 86 {
 87     if (!span.Valid()) return      std::unique_ptr<JsonObject>();
 88     if (!module)
 89     {
 90         throw std::runtime_error("module not set");
 91     }
 92     const std::string& fileName = module->GetFilePath(span.FileIndex());
 93     if (fileName.empty()) return      std::unique_ptr<JsonObject>();
 94     std::unique_ptr<JsonObject> json(new JsonObject());
 95     json->AddField(U"file",      std::unique_ptr<JsonValue>(new JsonString(ToUtf32(fileName))));
 96     json->AddField(U"line",      std::unique_ptr<JsonValue>(new JsonString(ToUtf32(std::to_string(span.LineNumber())))));
 97     MappedInputFile file(fileName);
 98     std::string s(file.Begin(), file.End());
 99     std::u32string t(ToUtf32(s));
100     std::u32string text = cmajor::parsing::GetErrorLines(&t[0], &t[0] + t.length(), span);
101     int32_t startCol = 0;
102     int32_t endCol = 0;
103     GetColumns(&t[0], &t[0] + t.length(), span, startCol, endCol);
104     json->AddField(U"startCol",      std::unique_ptr<JsonValue>(new JsonString(ToUtf32(std::to_string(startCol)))));
105     json->AddField(U"endCol",      std::unique_ptr<JsonValue>(new JsonString(ToUtf32(std::to_string(endCol)))));
106     json->AddField(U"text",      std::unique_ptr<JsonValue>(new JsonString(text)));
107     return json;
108 }

110 std::unique_ptr<cmajor::dom::Element> SpanToDomElement(Module* module, const Span& span)
111 {
112     if (!span.Valid()) return      std::unique_ptr<cmajor::dom::Element>();
113     if (!module)
114     {
115         throw std::runtime_error("module not set");
116     }
117     const std::string& fileName = module->GetFilePath(span.FileIndex());
118     if (fileName.empty()) return      std::unique_ptr<cmajor::dom::Element>();
119     std::unique_ptr<cmajor::dom::Element> spanElement(new cmajor::dom::Element(U"span"));
120     std::unique_ptr<cmajor::dom::Element> fileElement(new cmajor::dom::Element(U"file"));
121     std::unique_ptr<cmajor::dom::Text> fileText(new cmajor::dom::Text(ToUtf32(fileName)));
122     fileElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(fileText.release()));
123     std::unique_ptr<cmajor::dom::Element> lineElement(new cmajor::dom::Element(U"line"));
124     std::unique_ptr<cmajor::dom::Text> lineText(new cmajor::dom::Text(ToUtf32(std::to_string(span.LineNumber()))));
125     lineElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(lineText.release()));
126     MappedInputFile file(fileName);
127     std::string s(file.Begin(), file.End());
128     std::u32string t(ToUtf32(s));
129     std::u32string text = cmajor::parsing::GetErrorLines(&t[0], &t[0] + t.length(), span);
130     int32_t startCol = 0;
131     int32_t endCol = 0;
132     GetColumns(&t[0], &t[0] + t.length(), span, startCol, endCol);
133     spanElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(fileElement.release()));
134     spanElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(lineElement.release()));
135     std::unique_ptr<cmajor::dom::Element> startColElement(new cmajor::dom::Element(U"startCol"));
136     std::unique_ptr<cmajor::dom::Text> startColText(new cmajor::dom::Text(ToUtf32(std::to_string(startCol))));
137     startColElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(startColText.release()));
138     spanElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(startColElement.release()));
139     std::unique_ptr<cmajor::dom::Element> endColElement(new cmajor::dom::Element(U"endCol"));
140     std::unique_ptr<cmajor::dom::Text> endColText(new cmajor::dom::Text(ToUtf32(std::to_string(endCol))));
141     endColElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(endColText.release()));
142     spanElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(endColElement.release()));
143     std::unique_ptr<cmajor::dom::Element> textElement(new cmajor::dom::Element(U"text"));
144     std::unique_ptr<cmajor::dom::Text> textText(new cmajor::dom::Text(text));
145     textElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(textText.release()));
146     spanElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(textElement.release()));
147     return spanElement;
148 }

168 std::unique_ptr<JsonValue> Exception::ToJson() const
169 {
170     std::unique_ptr<JsonObject> json(new JsonObject());
171     json->AddField(U"tool",      std::unique_ptr<JsonValue>(new JsonString(module->GetCurrentToolName())));
172     json->AddField(U"kind",      std::unique_ptr<JsonValue>(new JsonString(U"error")));
173     json->AddField(U"project",      std::unique_ptr<JsonValue>(new JsonString(module->GetCurrentProjectName())));
174     json->AddField(U"message",      std::unique_ptr<JsonValue>(new JsonString(ToUtf32(message))));
175     std::unique_ptr<JsonArray> refs(new JsonArray());
176     std::unique_ptr<JsonObject> ref = SpanToJson(module, defined);
177     if (ref)
178     {
179         refs->AddItem(std::move(ref));
180     }
181     std::vector<Span> referenceSpans = references;
182     referenceSpans.erase(std::unique(referenceSpans.begin(), referenceSpans.end()), referenceSpans.end());
183     for (const Span& referenceSpan : referenceSpans)
184     {
185         if (!referenceSpan.Valid()) continue;
186         if (referenceSpan == defined) continue;
187         std::unique_ptr<JsonObject> ref = SpanToJson(module, referenceSpan);
188         if (ref)
189         {
190             refs->AddItem(std::move(ref));
191         }
192     }
193     json->AddField(U"references", std::move(refs));
194     return      std::unique_ptr<JsonValue>(json.release());
195 }

197 void Exception::AddToDiagnosticsElement(cmajor::dom::Element* diagnosticsElement) const
198 {
199     std::unique_ptr<cmajor::dom::Element> diagnosticElement(new cmajor::dom::Element(U"diagnostic"));
200     std::unique_ptr<cmajor::dom::Element> categoryElement(new cmajor::dom::Element(U"category"));
201     std::unique_ptr<cmajor::dom::Text> categoryText(new cmajor::dom::Text(U"error"));
202     categoryElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(categoryText.release()));
203     std::unique_ptr<cmajor::dom::Element> subcategoryElement(new cmajor::dom::Element(U"subcategory"));
204     std::unique_ptr<cmajor::dom::Text> subcategoryText(new cmajor::dom::Text(U"error"));
205     subcategoryElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(subcategoryText.release()));
206     std::unique_ptr<cmajor::dom::Element> messageElement(new cmajor::dom::Element(U"message"));
207     std::unique_ptr<cmajor::dom::Text> messageText(new cmajor::dom::Text(ToUtf32(message)));
208     messageElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(messageText.release()));
209     diagnosticElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(categoryElement.release()));
210     diagnosticElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(subcategoryElement.release()));
211     diagnosticElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(messageElement.release()));
212     std::unique_ptr<cmajor::dom::Element> spanElement = SpanToDomElement(module, defined);
213     if (spanElement)
214     {
215         diagnosticElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(spanElement.release()));
216     }
217     diagnosticsElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(diagnosticElement.release()));
218     for (const Span& span : references)
219     {
220         if (!span.Valid()) continue;
221         std::unique_ptr<cmajor::dom::Element> diagnosticElement(new cmajor::dom::Element(U"diagnostic"));
222         std::unique_ptr<cmajor::dom::Element> categoryElement(new cmajor::dom::Element(U"category"));
223         std::unique_ptr<cmajor::dom::Text> categoryText(new cmajor::dom::Text(U"info"));
224         categoryElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(categoryText.release()));
225         std::unique_ptr<cmajor::dom::Element> messageElement(new cmajor::dom::Element(U"message"));
226         std::unique_ptr<cmajor::dom::Text> messageText(new cmajor::dom::Text(ToUtf32("see reference to")));
227         messageElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(messageText.release()));
228         diagnosticElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(categoryElement.release()));
229         diagnosticElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(messageElement.release()));
230         std::unique_ptr<cmajor::dom::Element> spanElement = SpanToDomElement(module, span);
231         if (spanElement)
232         {
233             diagnosticElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(spanElement.release()));
234             diagnosticsElement->AppendChild(     std::unique_ptr<cmajor::dom::Node>(diagnosticElement.release()));
235         }
236     }
237 
238 }
