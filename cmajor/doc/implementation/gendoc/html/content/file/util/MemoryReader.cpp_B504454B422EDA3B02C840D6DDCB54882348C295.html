11 MemoryReader::MemoryReader(const uint8_t* begin_, const uint8_t* end_) : begin(begin_), end(end_), pos(0)
12 {
13 }
14 
15 uint8_t MemoryReader::GetByte()
16 {
17     if (begin >= end)
18     {
19         throw std::runtime_error("MemoryReader: unexpected end of data");
20     }
21     ++pos;
22     return *begin++;
23 }
24 
25 int8_t MemoryReader::GetSByte()
26 {
27     uint8_t x = GetByte();
28     return static_cast<int8_t>(x);
29 }
30 
31 uint16_t MemoryReader::GetUShort()
32 {
33     uint8_t x[2];
34     x[0] = GetByte();
35     x[1] = GetByte();
36     return *reinterpret_cast<uint16_t*>(&x[0]);
37 }
38 
39 int16_t MemoryReader::GetShort()
40 {
41     uint16_t x = GetUShort();
42     return static_cast<int16_t>(x);
43 }
44 
45 uint32_t MemoryReader::GetUInt()

50     x[2] = GetByte();

53 }

55 int32_t MemoryReader::GetInt()
