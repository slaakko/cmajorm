15 enum class JsonValueType
16 {
17     object, array, string, number, boolean, null
18 };
19 
20 class JsonValue 
21 {
22 public:
23     JsonValue(JsonValueType type_);
24     virtual ~JsonValue();
25     JsonValueType Type() const { return type; }
26     virtual std::string ToString() const = 0;
27 private:
28     JsonValueType type;
29 };
30 
31 class JsonString : public JsonValue
32 {
33 public:
34     JsonString();
35     JsonString(const std::u32string& value_);
36     void Append(char32_t c);
37     const std::u32string& Value() const { return value; }
38     std::u16string JsonCharStr(char32_t c) const;
39     std::string ToString() const override;
40 private:
41     std::u32string value;
42 };

44 class JsonNumber : public JsonValue
45 {
46 public:

48     JsonNumber(double value_);
49     double Value() const { return value; }
50     std::string ToString() const override;
51 private:

53 };

61     std::string ToString() const override;

66 class JsonObject : public JsonValue
67 {
68 public:
69     JsonObject();
70     void AddField(const std::u32string& fieldName, std::unique_ptr<JsonValue>&& fieldValue);
71     JsonValue* GetField(const std::u32string& fieldName);
72     std::string ToString() const override;
73 private:

75     std::map<std::u32string, JsonValue*> fieldMap;
76 };
