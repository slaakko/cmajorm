11 void CodeFormatter::Write(const std::string& text)
12 {
13     if (atBeginningOfLine)
14     {
15         if (indent != 0)
16         {
17             stream << std::string(indentSize * indent, ' ');
18             atBeginningOfLine = false;
19         }
20     }
21     stream << text;
22 }
23 
24 void CodeFormatter::WriteLine(const std::string& text)
25 {
26     Write(text);
27     NewLine();
28 }
29 
30 void CodeFormatter::NewLine()
31 {
32     stream << "\n";
33     atBeginningOfLine = true;
34     ++line;
35 }
