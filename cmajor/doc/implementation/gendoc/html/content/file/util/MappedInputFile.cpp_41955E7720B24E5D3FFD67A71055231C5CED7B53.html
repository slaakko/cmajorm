 6 #include <cmajor/util/MappedInputFile.hpp>
 7 #include <boost/iostreams/device/mapped_file.hpp>
 8 
 9 namespace cmajor { namespace util {
10 
11 class MappedInputFileImpl 
12 {
13 public:
14     MappedInputFileImpl(const std::string& fileName_);
15     const char* Data() const { return mappedFile.const_data(); }
16     boost::iostreams::mapped_file_source::size_type Size() const { return mappedFile.size(); }
17 private:
18     boost::iostreams::mapped_file mappedFile;
19 };
20 
21 MappedInputFileImpl::MappedInputFileImpl(const std::string& fileName_) : mappedFile()
22 {
23     try
24     {
25         mappedFile.open(fileName_, boost::iostreams::mapped_file::mapmode::readonly);
26     }
27     catch (std::exception& ex;)
28     {
29         throw std::runtime_error("error opening mapped file '" + fileName_ + "': " + ex.what());
30     }
31     catch (...)
32     {
33         throw std::runtime_error("error opening mapped file '" + fileName_ + "'");
34     }
35 }
36 
37 MappedInputFile::MappedInputFile(const std::string& fileName_) : impl(new MappedInputFileImpl(fileName_))
38 {
39 }
40 
41 MappedInputFile::~MappedInputFile()
42 {
43     delete impl;
44 }
45 
46 const char* MappedInputFile::Begin() const
47 {
48     const char* start = impl->Data();
49     if (impl->Size() >= 3)
50     {
51         if ((unsigned char)start[0] == (unsigned char)0xEF && 
52             (unsigned char)start[1] == (unsigned char)0xBB && 
53             (unsigned char)start[2] == (unsigned char)0xBF)
54         {
55             start += 3;
56         }
57     }
58     return start;
59 }
60 
61 const char* MappedInputFile::End() const
62 {
63     return impl->Data() + impl->Size();
64 }
65 
66 std::string ReadFile(const std::string& fileName)
67 {
68     MappedInputFile mappedFile(fileName);
69     return std::string(mappedFile.Begin(), mappedFile.End());
70 }
