13 class Sha1 
14 {
15 public:
16     Sha1();
17     void Reset();
18     void Process(uint8_t x)
19     {
20         ProcessByte(x);
21         bitCount = bitCount + 8u;
22     }
23     void Process(void* begin, void* end);
24     void Process(void* buf, int count)
25     {
26         uint8_t* b = static_cast<uint8_t*>(buf);
27         Process(b, b + count);
28     }
29     std::string GetDigest();
30 private:
31     void ProcessByte(uint8_t x)
32     {
33         block[byteIndex++] = x;
34         if (byteIndex == 64u)
35         {
36             byteIndex = 0u;
37             ProcessBlock();
38         }
39     }
40     void ProcessBlock();
41     uint32_t digest[5];
42     uint8_t block[64];
43     uint8_t byteIndex;
44     uint64_t bitCount;
45 };
