 6 #include <cmajor/util/Log.hpp>
 7 #include <iostream>
 8 #include <mutex>
 9 
10 namespace cmajor { namespace util {
11 
12 std::mutex logMutex;
13 
14 void LogMessage(int logStreamId, const std::string& message)
15 {
16     std::lock_guard<std::mutex> lock(logMutex);
17     if (logStreamId == -1)
18     {
19         std::cout << message << std::endl;
20     }
21     else
22     {
23         std::cout << logStreamId << ">" << message << std::endl;
24     }
25 }

27 } } // namespace cmajor::util
