 35 std::u16string JsonString::JsonCharStr(char32_t c) const
 36 {
 37     std::u16string result;
 38     std::u32string s(1, c);
 39     std::u16string t = ToUtf16(s);
 40     for (char16_t x : t)
 41     {
 42         switch (x)
 43         {
 44             case '"': result.append(u"\\\""); break;
 45             case '\\': result.append(u"\\\\"); break;
 46             case '/': result.append(u"\\/"); break;
 47             case '\b': result.append(u"\\b"); break;
 48             case '\f': result.append(u"\\f"); break;
 49             case '\n': result.append(u"\\n"); break;
 50             case '\r': result.append(u"\\r"); break;
 51             case '\t': result.append(u"\\t"); break;
 52             default:
 53             {
 54                 if (x >= 32 && x <= 126)
 55                 {
 56                     result.append(1, x);
 57                 }
 58                 else
 59                 {
 60                     result.append(u"\\u").append(ToUtf16(ToHexString(static_cast<uint16_t>(x))));
 61                 }
 62             }
 63         }
 64     }
 65     return result;
 66 }

 68 std::string JsonString::ToString() const
 69 {
 70     std::u16string s;
 71     for (char32_t c : value)
 72     {
 73         s.append(JsonCharStr(c));
 74     }
 75     return "\"" + ToUtf8(s) + "\"";
 76 }

108 void JsonObject::AddField(const std::u32string& fieldName, std::unique_ptr<JsonValue>&& fieldValue)
109 {
110     fieldMap[fieldName] = fieldValue.get();
111     fieldValues.push_back(std::move(fieldValue));
112 }
113 
114 JsonValue* JsonObject::GetField(const std::u32string& fieldName)
115 {
116     auto it = fieldMap.find(fieldName);
117     if (it != fieldMap.cend())
118     {
119         return it->second;
120     }
121     else
122     {
123         return nullptr;
124     }
125 }
126 
127 std::string JsonObject::ToString() const
128 {
129     std::string str = "{";
130     bool first = true;
131     for (const auto& p : fieldMap)
132     {
133         JsonString s(p.first);
134         JsonValue * v(p.second);
135         if (first)
136         {
137             first = false;
138         }
139         else
140         {
141             str.append(", ");
142         }
143         str.append(s.ToString()).append(":").append(v->ToString());
144     }
145     str.append("}");
146     return str;
147 }
