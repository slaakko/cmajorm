 16 std::string Date::ToString(bool omitDashes) const
 17 {
 18     std::string date;
 19     date.append(1, static_cast<char>(static_cast<int16_t>('0') + ((year / 1000) % 10)));
 20     date.append(1, static_cast<char>(static_cast<int16_t>('0') + ((year / 100) % 10)));
 21     date.append(1, static_cast<char>(static_cast<int16_t>('0') + ((year / 10) % 10)));
 22     date.append(1, static_cast<char>(static_cast<int16_t>('0') + (year % 10)));
 23     if (!omitDashes)
 24     {
 25         date.append(1, '-');
 26     }
 27     date.append(1, static_cast<char>(static_cast<int8_t>('0') + ((static_cast<int8_t>(month) / 10) % 10)));
 28     date.append(1, static_cast<char>(static_cast<int8_t>('0') + (static_cast<int8_t>(month) % 10)));
 29     if (!omitDashes)
 30     {
 31         date.append(1, '-');
 32     }
 33     date.append(1, static_cast<char>(static_cast<int8_t>('0') + ((day / 10) % 10)));
 34     date.append(1, static_cast<char>(static_cast<int8_t>('0') + (day % 10)));
 35     return date;
 36 }

 38 Date GetCurrentDate()

 42     struct tm* localTime = nullptr;
 43     localTime = std::localtime(&currentTime);

 52 std::string DateTime::ToString(bool omitDashes, bool omitColons, bool omitMins, bool omitSecs) const
 53 {
 54     std::string dateTime;
 55     dateTime.append(date.ToString(omitDashes));
 56     dateTime.append(1, 'T');
 57     int32_t hh = Hours() % 24;
 58     int32_t mm = Minutes() % 60;
 59     int32_t ss = Seconds() % 60;
 60     dateTime.append(1, static_cast<char>(static_cast<int32_t>('0') + ((hh / 10) % 10)));
 61     dateTime.append(1, static_cast<char>(static_cast<int32_t>('0') + (hh % 10)));
 62     if (!omitMins)
 63     {
 64         if (!omitColons)
 65         {
 66             dateTime.append(1, ':');
 67         }
 68         dateTime.append(1, static_cast<char>(static_cast<int32_t>('0') + ((mm / 10) % 10)));
 69         dateTime.append(1, static_cast<char>(static_cast<int>('0') + (mm % 10)));
 70         if (!omitSecs)
 71         {
 72             if (!omitColons)
 73             {
 74                 dateTime.append(1, ':');
 75             }
 76             dateTime.append(1, static_cast<char>(static_cast<int32_t>('0') + ((ss / 10) % 10)));
 77             dateTime.append(1, static_cast<char>(static_cast<int32_t>('0') + (ss % 10)));
 78         }
 79     }
 80     return dateTime;
 81 }

 83 std::string FormatTimeMs(int32_t milliseconds)
 84 {
 85     int32_t hh = milliseconds / 3600000;
 86     int32_t mm = milliseconds / 60000 % 60;
 87     int32_t ss = milliseconds / 1000 % 60;
 88     int32_t ms = milliseconds % 1000;
 89     std::string time;
 90     time.append(1, static_cast<char>(static_cast<int32_t>('0') + ((hh / 10) % 10)));
 91     time.append(1, static_cast<char>(static_cast<int32_t>('0') + (hh % 10)));
 92     time.append(1, ':');
 93     time.append(1, static_cast<char>(static_cast<int32_t>('0') + ((mm / 10) % 10)));
 94     time.append(1, static_cast<char>(static_cast<int>('0') + (mm % 10)));
 95     time.append(1, ':');
 96     time.append(1, static_cast<char>(static_cast<int32_t>('0') + ((ss / 10) % 10)));
 97     time.append(1, static_cast<char>(static_cast<int32_t>('0') + (ss % 10)));
 98     time.append(1, '.');
 99     time.append(1, static_cast<char>(static_cast<int32_t>('0') + ((ms / 100) % 10)));
100     time.append(1, static_cast<char>(static_cast<int32_t>('0') + ((ms / 10) % 10)));
101     time.append(1, static_cast<char>(static_cast<int32_t>('0') + (ms % 10)));
102     return time;
103 }

112 }
