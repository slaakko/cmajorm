12 
13 class CodeFormatter 
14 {
15 public:
16     CodeFormatter(std::ostream& stream_) : stream(stream_), indent(0), indentSize(4), atBeginningOfLine(true), line(1), start(false), preserveSpace(false) {}
17     int Indent() const { return indent; }
18     int IndentSize() const { return indentSize; }
19     void SetIndentSize(int indentSize_) { indentSize = indentSize_; }
20     int CurrentIndent() const { return indentSize * indent; }
21     void Write(const std::string& text);
22     void WriteLine(const std::string& text);
23     void NewLine();
24     void WriteLine() { NewLine(); }
25     void IncIndent()
26     {
27         ++indent;
28     }
29     void DecIndent()
30     {
31         --indent;
32     }
33     int Line() const { return line; }
34     void SetLine(int line_) { line = line_; }
35     bool Start() const { return start; }
36     void SetStart() { start = true; }
37     void ResetStart() { start = false; }
38     void SetStartText(const std::string& startText_) { startText = startText_; }
39     const std::string& StartText() const { return startText; }
40     bool PreserveSpace() const { return preserveSpace; }
41     void SetPreserveSpace(bool preserveSpace_) { preserveSpace = preserveSpace_; }
42 private:
43     std::ostream& stream;
44     int indent;
45     int indentSize;
46     bool atBeginningOfLine;
47     int line;
48     bool start;
49     std::string startText;
50     bool preserveSpace;
51 };
