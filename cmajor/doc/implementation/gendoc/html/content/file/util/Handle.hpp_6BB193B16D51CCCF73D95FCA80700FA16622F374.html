 6 #ifndef CMAJOR_UTIL_HANDLE_INCLUDED
 7 #define CMAJOR_UTIL_HANDLE_INCLUDED
 8 #if defined(_WIN32)
 9 #include <io.h>
10 #elif defined(__linux) || defined(__posix) || defined(__unix)
11 #include <unistd.h>
12 #else
13 #error unknown platform
14 #endif
15 
16 namespace cmajor { namespace util {
17 
18 class Handle 
19 {
20 public:
21     Handle(int handle_) : handle(handle_)
22     {
23     }
24     ~Handle()
25     {
26         if (handle != -1)
27         {
28             close(handle);
29         }
30     }
31     Handle(Handle&& that)
32     {
33         handle = that.handle;
34         that.handle = -1;
35     }
36     void operator=(Handle&& that)
37     {
38         if (handle != -1)
39         {
40             close(handle);
41         }
42         handle = that.handle;
43         that.handle = -1;
44     }
45     operator int() const { return handle; }
46 private:
47     int handle;
48 };
49 
50 } } // namespace cmajor::util
51 
52 #endif // CMAJOR_UTIL_HANDLE_INCLUDED
