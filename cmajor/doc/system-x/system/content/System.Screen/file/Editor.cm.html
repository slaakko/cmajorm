 377         public override void OnKeyPressed(KeyEventArgs& args)

 380             Rect updateRect;
 381             int prevXOffset = xOffset;
 382             int prevYOffset = yOffset;
 383             uchar key = args.Key();
 384             if (key >= ' ' && key < specialKeyStart)
 385             {
 386                 args.SetHandled();
 387                 DeleteSelection();
 388                 while (caretLine >= lines.Count())
 389                 {
 390                     lines.Add(ustring());
 391                 }
 392                 ustring& line = lines[caretLine];
 393                 bool inserted = false;
 394                 if (caretCol < line.Length())
 395                 {
 396                     line.Insert(caretCol, key);
 397                     inserted = true;
 398                 }
 399                 else
 400                 {
 401                     line.Append(key);
 402                 }
 403                 Point start = ScreenPoint(caretCol, caretLine);
 404                 SetCaretPos(caretLine, caretCol + 1);
 405                 Point end = ScreenPoint(caretCol, caretLine + 1);
 406                 if (inserted)
 407                 {
 408                     end = ScreenPoint(cast<int>(line.Length()), caretLine + 1);
 409                 }
 410                 updateRect = ScreenRect(start, end);
 411                 SetXOffset();
 412                 SetYOffset();

 414             }

 611         private void WriteScreenNoSelection(const Rect& updateRect)
 612         {
 613             Terminal.Out() << SetColors(ForeColor(), BackColor());
 614             for (int y = 0; y < updateRect.size.h; ++y;)
 615             {
 616                 ustring updateLine;
 617                 for (int x = 0; x < updateRect.size.w; ++x;)
 618                 {
 619                     uchar c = CharAt(updateRect.location.x + x, updateRect.location.y + y);
 620                     if (c == '\0')
 621                     {
 622                         break;
 623                     }
 624                     updateLine.Append(c);
 625                 }
 626                 SetCursorPos(updateRect.location.x, updateRect.location.y + y);
 627                 Terminal.Out() << updateLine;
 628             }
 629         }
 630         private void WriteScreenWithSelection(const Rect& updateRect)
 631         {
 632             MakeCanonicalSelection();
 633             for (int y = 0; y < updateRect.size.h; ++y;)
 634             {
 635                 for (int x = 0; x < updateRect.size.w; ++x;)
 636                 {
 637                     uchar c = CharAt(updateRect.location.x + x, updateRect.location.y + y);
 638                     if (c != '\0')
 639                     {
 640                         int line = 0;
