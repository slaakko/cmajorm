 29     public class DirectoryItemLess : Rel<DirectoryItem>
 30     {
 31         public nothrow bool operator()(const DirectoryItem& left, const DirectoryItem& right) const
 32         {
 33             if (left.status.fileType == FileType.directory)
 34             {
 35                 if (right.status.fileType == FileType.directory)
 36                 {
 37                     return left.entry.name < right.entry.name;
 38                 }
 39                 else
 40                 {
 41                     return true;
 42                 }
 43             }
 44             else if (left.status.fileType == FileType.regular)
 45             {
 46                 if (right.status.fileType == FileType.directory)
 47                 {
 48                     return false;
 49                 }
 50                 else if (right.status.fileType == FileType.regular)
 51                 {
 52                     return left.entry.name < right.entry.name;
 53                 }
 54                 else
 55                 {
 56                     return true;
 57                 }
 58             }
 59             else if (left.status.fileType == FileType.fifo)
 60             {
 61                 if (right.status.fileType == FileType.directory)
 62                 {
 63                     return false;
 64                 }
 65                 else if (right.status.fileType == FileType.regular)
 66                 {
 67                     return false;
 68                 }
 69                 else
 70                 {
 71                     return left.entry.name < right.entry.name;
 72                 }
 73             }
 74             return false;
 75         }
 76     }

 78     public DirectoryItem MakeItem(const string& directory, const DirectoryEntry& entry)
 79     {
 80         string path = Path.Combine(directory, entry.name);
 81         FileStatus status;
 82         Stat(path.Chars(), status);
 83         return DirectoryItem(entry, status);
 84     }

122     public class FileDialog : Window
123     {
124         public nothrow FileDialog(FileDialogCreateParams& createParams) : 
125             base(createParams.controlCreateParams), 
126             kind(createParams.kind), 
127             caption(createParams.caption), 
128             directory(createParams.directory), 
129             requireFileExists(createParams.requireFileExists)
130         {
131             InvalidateGuard guard(this, true);
132             if (caption.IsEmpty())
133             {
134                 switch (kind)
135                 {
136                     case FileDialogKind.openDialog:
137                     {
138                         caption = "Open File";
139                         break;
140                     }
141                     case FileDialogKind.saveDialog:
142                     {
143                         caption = "Save File";
144                         break;
145                     }
146                 }
147             }
148             bool setLoc = false;
149             if (Location().IsDefault())
150             {
151                 SetLocation(Point(0, 0));
152                 setLoc = true;
153             }
154             if (GetSize().IsDefault())
155             {
156                 SetSize(Size(60, 20));
157             }
158             if (ForeColor() == ConsoleColor.defaultColor)
159             {
160                 SetForeColor(ConsoleColor.black);
161             }
162             if (BackColor() == ConsoleColor.defaultColor)
163             {
164                 SetBackColor(ConsoleColor.gray);
165             }
166             Measure(setLoc);
167             Point loc = Location();
168             Size sz = GetSize();
169             Label* label = new Label(LabelCreateParams().SetLocation(Point(loc.x + 2, loc.y + 1)).SetText("Directory:"));
170             AddChild(label);
171             directoryLabel = new Label(LabelCreateParams().SetLocation(Point(loc.x + 2, loc.y + 2)).SetText(directory));
172             AddChild(directoryLabel);
173             listBox = new ListBox(ListBoxCreateParams().SetLocation(Point(loc.x + 2, loc.y + 4)).SetSize(Size(sz.w - 4, 13)));
174             listBox->SelectedIndexChangedEvent().AddHandler(ListBoxSelectedIndexChanged);
175             listBox->ItemSelectedEvent().AddHandler(ListBoxItemSelected);
176             AddChild(listBox);
177             textBox = new TextBox(TextBoxCreateParams().SetLocation(Point(loc.x + 2, loc.y + 18)).SetSize(Size(sz.w - 4, 1)));
178             textBox->TextChangedEvent().AddHandler(TextBoxTextChanged);
179             textBox->TextEnteredEvent().AddHandler(TextBoxTextEntered);
180             AddChild(textBox);
181             string buttonText;
182             switch (kind)
183             {
184                 case FileDialogKind.openDialog:
185                 {
186                     buttonText = "[ &Open ]";
187                     break;
188                 }
189                 case FileDialogKind.saveDialog:
190                 {
191                     buttonText = "[ &Save ]";
192                     break;
193                 }
194             }
195             defaultButton = new Button(ButtonCreateParams().Text(buttonText));
196             defaultButton->SetDialogResult(DialogResult.ok);
197             defaultButton->SetDefault();
198             AddChild(defaultButton);
199             Button* cancelButton = Button.Cancel();
200             AddChild(cancelButton);
201             cancelButton->SetLocation(Point(loc.x + sz.w - cancelButton->GetSize().w - 1, loc.y + sz.h - 2));
202             defaultButton->SetLocation(Point(cancelButton->Location().x - defaultButton->GetSize().w - 1, cancelButton->Location().y));
203             MakeListBoxContent();
204             Control* firstFocusableControl = FirstFocusabledControl();
205             if (firstFocusableControl != null)
206             {
207                 firstFocusableControl->SetFocus();
208             }
209             SetDefaultButtonStatus();
210         }
211         public string FilePath() const
212         {
213             if (!textBox->Text().IsEmpty())
214             {
215                 return Path.Combine(directory, ToUtf8(textBox->Text()));
216             }
217             else if (listBox->SelectedIndex() >= 0 && listBox->SelectedIndex() < listBox->ItemCount())
218             {
219                 if (items[listBox->SelectedIndex()].status.fileType == FileType.regular)
220                 {
221                     return Path.Combine(directory, items[listBox->SelectedIndex()].entry.name);
222                 }
223             }
224             return string();
225         }

236         private void ListBoxItemSelected()
237         {
238             if (items[listBox->SelectedIndex()].status.fileType == FileType.directory)
239             {
240                 directory = GetFullPath(Path.Combine(directory, items[listBox->SelectedIndex()].entry.name));
241                 directoryLabel->SetText(directory);
242                 MakeListBoxContent();
243                 Invalidate();
244             }
245             else if (items[listBox->SelectedIndex()].status.fileType == FileType.regular)
246             {
247                 SetDefaultButtonStatus();
248                 defaultButton->Press();
249             }
250         }
251         private void ListBoxSelectedIndexChanged()
252         {
253             if (requireFileExists)
254             {
255                 if (items[listBox->SelectedIndex()].status.fileType == FileType.directory)
256                 {
257                     defaultButton->SetEnabled();
258                 }
259                 else if (items[listBox->SelectedIndex()].status.fileType == FileType.regular)
260                 {
261                     SetDefaultButtonStatus();
262                 }
263                 else if (items[listBox->SelectedIndex()].status.fileType == FileType.fifo)
264                 {
265                     defaultButton->SetDisabled();
266                 }
267             }
268         }
269         private void TextBoxTextChanged()
270         {
271             if (!textBox->Text().IsEmpty())
272             {
273                 SetDefaultButtonStatus();
274             }
275             else
276             {
277                 defaultButton->SetDisabled();
278             }
279         }
280         private void TextBoxTextEntered()
281         {
282             if (!textBox->Text().IsEmpty())
283             {
284                 SetDefaultButtonStatus();
285                 defaultButton->Press();
286             }
287         }
288         private void SetDefaultButtonStatus()
289         {
290             if (requireFileExists)
291             {

293                 if (!filePath.IsEmpty() && File.Exists(filePath))

295                     defaultButton->SetEnabled();

297                 else
298                 {
299                     defaultButton->SetDisabled();

301             }

303         private void Measure(bool setLoc)
304         {
305             Rect rect = GetRect();
306             rect.Inflate(1, 1);
307             SetLocation(rect.location);
308             SetSize(rect.size);
309             Size sz = GetSize();
310             if (setLoc)
311             {
312                 int x = Max(cast<int>(0), (TerminalWindowWidth() - sz.w) / 2);
313                 int y = Max(cast<int>(0), (TerminalWindowHeight() - sz.h) / 2);
314                 SetLocation(Point(x, y));
315             }
316         }
317         private void WriteCaption(const ustring& captionStr)
318         {
319             Point loc = Location();
320             Size sz = GetSize();
321             int offset = Max(cast<int>(0), cast<int>(sz.w - captionStr.Length()));
322             int x = loc.x + offset / 2;
323             int y = loc.y;
324             SetCursorPos(x, y);
325             Terminal.Out() << captionStr;
326         }
327         private void MakeListBoxContent()
328         {
329             ReadDirectory();
330             SortItems();
331             AddItemsToListBox();
332         }
333         private void AddItemsToListBox()
334         {
335             listBox->Clear();
336             Size sz = listBox->GetSize();
337             int width = sz.w;
338             for (const DirectoryItem& item : items)
339             {
340                 ustring itemStr;
341                 if (item.status.fileType == FileType.directory)
342                 {
343                     itemStr.Append(' ').Append(folderSymbol).Append(' ');
344                 }
345                 else if (item.status.fileType == FileType.regular)
346                 {
347                     itemStr.Append(' ').Append(emptyDocumentSymbol).Append(' ');
348                 }
349                 else if (item.status.fileType == FileType.fifo)
350                 {
351                     itemStr.Append(' ').Append(emptyDocumentSymbol).Append(' ');
352                 }
353                 ustring entryName = ToUtf32(item.entry.name);
354                 int entryWidth = width - 3;
355                 itemStr.Append(entryName.Substring(0, entryWidth));
356                 int spaceWidth = Max(cast<int>(0), cast<int>(entryWidth - entryName.Length()));
357                 if (spaceWidth > 0)
358                 {
359                     itemStr.Append(ustring(' ', spaceWidth));
360                 }
361                 listBox->AddItem(itemStr);
362             }
363             listBox->SetSelectedIndex(0);
364         }
365         private void ReadDirectory()
366         {
367             items.Clear();
368             DirectoryReader reader(directory);
369             DirectoryEntry entry;
370             while (reader.Read(entry))
371             {
372                 if (entry.IsDot())
373                 {
374                     continue;
375                 }
376                 else if (entry.IsDotDot())
377                 {
378                     if (directory != "/")
379                     {
380                         items.Add(MakeItem(directory, entry));
381                     }
382                 }
383                 else
384                 {
385                     items.Add(MakeItem(directory, entry));
386                 }
387             }
388         }
389         private void SortItems()
390         {
391             Sort(items.Begin(), items.End(), DirectoryItemLess());
392         }
