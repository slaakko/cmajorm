  6 using System;
  7 using System.Os;
  8 
  9 namespace System.Screen
 10 {
 11     public class Application
 12     {
 13         static Application() : instance(new Application())
 14         {
 15         }
 16         public ~Application()
 17         {
 18             try
 19             {
 20                 UnbindTerminal();
 21                 CloseMsgQ(msgQ);
 22             }
 23             catch (const Exception& ex)
 24             {
 25             }
 26         }
 27         public static Application& Instance()
 28         {
 29             return *instance;
 30         }
 31         private Application() : msgQ(MsgQ("system.screen.msg.queue")), menuBar(null), focusedControl(null), lastDefaultWriteScreenMessageWindow(-1)
 32         {
 33             RegisterMessage<KeyPressedMessage>(keyPressedMessageId);
 34             BindTerminal(msgQ);
 35         }
 36         public nothrow Window* MainWindow()
 37         {
 38             return mainWindow;
 39         }
 40         public nothrow MenuBar* GetMenuBar() const
 41         {
 42             return menuBar;
 43         }
 44         public nothrow void SetMenuBar(MenuBar* menuBar_)
 45         {
 46             menuBar = menuBar_;
 47             menuBarWindowHandle = menuBar->Handle();
 48         }
 49         public WindowManager& GetWindowManager() const
 50         {
 51             return windowManager;
 52         }
 53         public void Run(Window& mainWindow_)
 54         {
 55             mainWindow = &mainWindow_;
 56             mainWindow->SetMainWindow();
 57             mainWindow->SetLocation(Point(0, 0));
 58             mainWindow->SetSize(Size(TerminalWindowWidth(), TerminalWindowHeight()));
 59             mainWindow->Invalidate();
 60             UniquePtr<Message> message = ReadMessage();
 61             while (!(message.Get() is QuitMessage*))
 62             {
 63                 bool handled = false;
 64                 if (message->TargetWindowHandle() != -1)
 65                 {
 66                     Control* control = windowManager.GetWindow(message->TargetWindowHandle());
 67                     if (control != null)
 68                     {
 69                         handled = control->HandleMessage(message.Get());
 70                     }
 71                 }
 72                 if (!handled)
 73                 {
 74                     handled = mainWindow->HandleMessage(message.Get());
 75                 }
 76                 message = ReadMessage();
 77             }
 78             Terminal.Out() << ResetColors();
 79             ClearScreen(TerminalWindowWidth(), TerminalWindowHeight());
 80             SetCursorPos(0, 0);
 81         }
 82         public void RunDialog(Window* window)
 83         {
 84             WriteScreenMessage writeScreenMessage(Rect.Default());
 85             mainWindow->HandleWriteScreen(&writeScreenMessage);
 86             UniquePtr<Message> message = ReadMessage();
 87             while (!(message.Get() is QuitMessage*))
 88             {
 89                 bool handled = window->HandleMessage(message.Get());
 90                 if (handled)
 91                 {
 92                     if (window->GetDialogResult() != DialogResult.none)
 93                     {
 94                         WriteScreenMessage writeScreenMessage(window->GetRect());
 95                         mainWindow->HandleWriteScreen(&writeScreenMessage);
 96                         return;
 97                     }
 98                 }
 99                 message = ReadMessage();
100             }
101         }
102         public void Exit()
103         {
104             QuitMessage quitMessage;
105             PostMessage(quitMessage);
106         }
107         public void PostMessage(Message& message)
108         {
109             if (message.TargetWindowHandle() != menuBarWindowHandle)
110             {
111                 if (message.IsDefaultWriteScreenMessage())
112                 {
113                     if (lastDefaultWriteScreenMessageWindow == message.TargetWindowHandle())
114                     {
115                         return;
116                     }
117                     else
118                     {
119                         lastDefaultWriteScreenMessageWindow = message.TargetWindowHandle();
120                     }
121                 }
122                 else
123                 {
124                     lastDefaultWriteScreenMessageWindow = -1;
125                 }
126             }
127             int msgSize = message.Size();
128             UniquePtr<byte> msg(cast<byte*>(MemAlloc(msgSize)));
129             MemoryWriter writer(msg.Get(), msgSize);
130             message.Write(writer);
131             PutMsg(msgQ, msg.Get(), msgSize);
132         }
133         public UniquePtr<Message> ReadMessage()
134         {
135             WaitMsg(msgQ);
136             return GetMessage();
137         }
138         public void Timer(const Duration& duration, Control* control, int timerId)
139         {
140             TimerMessage message(timerId);
141             message.SetTargetWindowHandle(control->Handle());
142             int msgSize = message.Size();
143             UniquePtr<byte> msg(cast<byte*>(MemAlloc(msgSize)));
144             MemoryWriter writer(msg.Get(), msgSize);
145             message.Write(writer);
146             TimerMsg(duration, msgQ, msg.Get(), msgSize);
147         }
148         public nothrow Control* FocusedControl() const
149         {
150             return focusedControl;
151         }
152         public nothrow void SetFocusedControl(Control* focusedControl_)
153         {
154             focusedControl = focusedControl_;
155         }
156         private UniquePtr<Message> GetMessage()
157         {
158             int qlen = GetMsgQLength(msgQ);
159             if (qlen > 0)
160             {
161                 int msgSize = GetMsgSize(msgQ);
162                 if (msgSize > 0)
163                 {
164                     UniquePtr<byte> msgData(cast<byte*>(MemAlloc(msgSize)));
165                     GetMsg(msgQ, msgData.Get());
166                     UniquePtr<Message> message(ReadMessage(msgData.Get(), msgSize));
167                     return message;
168                 }
169                 else
170                 {
171                     throw Exception("invalid message size 0");
172                 }
173             }
174             else
175             {
176                 throw Exception("message queue " + ToString(msgQ) + " is empty");
177             }
178         }
179         private static UniquePtr<Application> instance;
180         private int msgQ;
181         private Window* mainWindow;
182         private WindowManager windowManager;
183         private MenuBar* menuBar;
184         private Control* focusedControl;
185         private int lastDefaultWriteScreenMessageWindow;
186         private int menuBarWindowHandle;
187     }
188 }
