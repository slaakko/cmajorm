 10     public class InvalidateGuard
 11     {
 12         public nothrow InvalidateGuard(Control* control_, bool invalidate_) : control(control_), invalidate(invalidate_)
 13         {
 14             control->BeginInvalidate(invalidate);
 15         }
 16         public ~InvalidateGuard()
 17         {
 18             control->EndInvalidate(invalidate);
 19         }
 20         private Control* control;
 21         private bool invalidate;
 22     }

149         public nothrow Window* GetWindow() const
150         {
151             Container* container = GetContainer();
152             if (container != null)
153             {
154                 Component* parent = container->Parent();
155                 if (parent != null)
156                 {
157                     if (parent is Window*)
158                     {
159                         return cast<Window*>(parent);
160                     }
161                     else if (parent is Control*)
162                     {
163                         return cast<Control*>(parent)->GetWindow();
164                     }
165                 }
166             }
167             return null;
168         }

267         public nothrow void BeginInvalidate(bool invalidate)
268         {
269             Control* parentControl = ParentControl();
270             if (parentControl != null && parentControl->InvalidateCount() > 0)
271             {
272                 return;
273             }
274             if (invalidateCount == 0)
275             {
276                 updateRect = Rect();
277             }
278             ++invalidateCount;
279         }
280         public nothrow void EndInvalidate(bool invalidate)
281         {
282             Control* parentControl = ParentControl();
283             if (parentControl != null && parentControl->InvalidateCount() > 0)
284             {
285                 return;
286             }
287             --invalidateCount;
288             if (invalidateCount == 0 && invalidate)
289             {
290                 if (!updateRect.IsEmpty())
291                 {
292                     WriteScreenMessage message(updateRect);
293                     message.SetTargetWindowHandle(handle);
294                     Application.Instance().PostMessage(message);
295                 }
296             }
297         }
298         public nothrow int InvalidateCount() const
299         {
300             return invalidateCount;
301         }
302         public void Invalidate()
303         {
304             Invalidate(Rect.Default());
305         }
306         public void Invalidate(const Rect& rect)
307         {
308             InvalidateGuard guard(this, true);
309             if (updateRect.IsEmpty())
310             {
311                 updateRect = rect;
312             }
313             else if (rect.IsDefault())
314             {
315                 updateRect = Rect.Default();
316             }
317             else
318             {
319                 updateRect = Rect.Union(updateRect, rect);
320             }
321         }

342         public virtual void SetFocus()
343         {
344             Control* prevFocusedControl = Application.Instance().FocusedControl();
345             if (prevFocusedControl != this)
346             {
347                 if (prevFocusedControl != null)
348                 {
349                     prevFocusedControl->OnLostFocus();
350                 }
351                 Application.Instance().SetFocusedControl(this);
352                 OnGotFocus();
353                 if (prevFocusedControl != null)
354                 {
355                     prevFocusedControl->Invalidate(prevFocusedControl->FocusRect());
356                 }
357                 Invalidate(FocusRect());
358             }

364         public virtual bool HandleKeyPressed(KeyPressedMessage* message)
365         {
366             KeyEventArgs args(message->Key());
367             MenuBar* menuBar = Application.Instance().GetMenuBar();
368             if (menuBar != null)
369             {
370                 menuBar->OnKeyPressed(args);
371                 if (menuBar->IsOpen())
372                 {
373                     return true;
374                 }
375             }
376             if (args.Handled())
377             {
378                 return true;
379             }
380             Control* focusedControl = Application.Instance().FocusedControl();
381             if (focusedControl != null)
382             {
383                 focusedControl->OnKeyPressed(args);
384                 if (args.Handled())
385                 {
386                     return true;
387                 }
388             }
389             OnKeyPressed(args);
390             return args.Handled();
391         }
