 11     const int quitMessageId = 0;

 16     public class QuitMessage : Message
 17     {
 18         public override nothrow int Id() const
 19         {
 20             return quitMessageId;
 21         }
 22         public override nothrow string Name() const
 23         {
 24             return "quit";
 25         }

 30     }

 32     public class KeyPressedMessage : Message
 33     {
 34         public override nothrow int Id() const
 35         {
 36             return keyPressedMessageId;
 37         }
 38         public override nothrow int Size() const
 39         {
 40             return cast<int>(base->Size() + sizeof(uchar));
 41         }
 42         public override nothrow string Name() const
 43         {
 44             return "keyPressed";
 45         }
 46         static nothrow KeyPressedMessage() : Message()
 47         {
 48             RegisterMessage<KeyPressedMessage>(keyPressedMessageId);
 49         }
 50         public nothrow KeyPressedMessage(uchar key_) : key(key_)
 51         {
 52         }
 53         public override void Write(MemoryWriter& writer)
 54         {
 55             base->Write(writer);
 56             writer.Write(cast<int>(key));
 57         }
 58         public override void Read(MemoryReader& reader)
 59         {
 60             base->Read(reader);
 61             key = cast<uchar>(reader.ReadInt());
 62         }
 63         public override bool Dispatch(Control* control)
 64         {
 65             return control->HandleKeyPressed(this);
 66         }
 67         public nothrow uchar Key() const
 68         {
 69             return key;
 70         }
 71         private uchar key;
 72     }

 74     public class WriteScreenMessage : Message
 75     {
 76         public override nothrow int Id() const
 77         {
 78             return writeScreenMessageId;
 79         }
 80         public override nothrow int Size() const
 81         {
 82             return cast<int>(base->Size() + sizeof(Rect));
 83         }
 84         public override nothrow string Name() const
 85         {
 86             return "writeScreen";
 87         }
 88         static nothrow WriteScreenMessage() : Message()
 89         {
 90             RegisterMessage<WriteScreenMessage>(writeScreenMessageId);
 91         }
 92         public nothrow WriteScreenMessage(const Rect& rect_) : rect(rect_)
 93         {
 94         }
 95         public override void Write(MemoryWriter& writer)
 96         {
 97             base->Write(writer);
 98             writer.Write(rect.location.x);
 99             writer.Write(rect.location.y);
100             writer.Write(rect.size.w);
101             writer.Write(rect.size.h);
102         }
103         public override void Read(MemoryReader& reader)
104         {
105             base->Read(reader);
106             rect.location.x = reader.ReadInt();
107             rect.location.y = reader.ReadInt();
108             rect.size.w = reader.ReadInt();
109             rect.size.h = reader.ReadInt();
110         }
111         public override bool Dispatch(Control* control)
112         {
113             return control->HandleWriteScreen(this);
114         }
115         public nothrow const Rect& GetRect() const
116         {
117             return rect;
118         }
119         private Rect rect;
120     }

162     }
