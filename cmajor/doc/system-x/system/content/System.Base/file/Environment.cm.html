  6 using System.Collections;
  7 
  8 namespace System
  9 {
 10     private char** environment = null;
 11 
 12     public class Environment
 13     {
 14         static Environment() : instance(new Environment())
 15         {
 16         }
 17         private Environment() : variables()
 18         {
 19             if (environment != null)
 20             {
 21                 const char** envp = environment;
 22                 while (*envp != null)
 23                 {
 24                     string envVar = *envp;
 25                     long eqPos = envVar.Find('=');
 26                     if (eqPos != -1)
 27                     {
 28                         Set(envVar.Substring(0, eqPos), envVar.Substring(eqPos + 1));
 29                     }
 30                     ++envp;
 31                 }
 32             }
 33         }
 34         public static nothrow Environment& Instance()
 35         {
 36             return *instance;
 37         }
 38         public nothrow bool Has(const string& envVarName) const
 39         {
 40             return variables.CFind(envVarName) != variables.CEnd();
 41         }
 42         public nothrow string Get(const string& envVarName) const
 43         {
 44             Map<string, string>.ConstIterator it = variables.CFind(envVarName);
 45             if (it != variables.CEnd())
 46             {
 47                 return it->second;
 48             }
 49             else
 50             {
 51                 return string();
 52             }
 53         }
 54         public void Set(const string& envVarName, const string& envVarValue)
 55         {
 56             if (envVarName.Find('=') != -1)
 57             {
 58                 throw SystemError(EPARAM, "environment variable name cannot contain '=' character");
 59             }
 60             variables[envVarName] = envVarValue;
 61         }
 62         public nothrow void Remove(const string& envVarName)
 63         {
 64             variables.Remove(envVarName);
 65         }
 66         public nothrow void Clear()
 67         {
 68             variables.Clear();
 69         }
 70         public nothrow long Size() const
 71         {
 72             long size = 0;
 73             for (const Pair<string, string>& v : variables)
 74             {
 75                 size = size + v.first.Length() + 1 + v.second.Length() + 1 + 16;
 76             }
 77             return size;
 78         }
 79         public nothrow const Map<string, string>& Variables() const
 80         {
 81             return variables;
 82         }
 83         private static UniquePtr<Environment> instance;
 84         private Map<string, string> variables;
 85     }
 86 
 87     public nothrow bool HasEnv(const string& envVarName)
 88     {
 89         return Environment.Instance().Has(envVarName);
 90     }
 91     
 92     public nothrow string GetEnv(const string& envVarName)
 93     {
 94         return Environment.Instance().Get(envVarName);
 95     }
 96     
 97     public void SetEnv(const string& envVarName, const string& envVarValue)
 98     {
 99         Environment.Instance().Set(envVarName, envVarValue);
100     }
101     
102     public nothrow void RemoveEnv(const string& envVarName)
103     {
104         Environment.Instance().Remove(envVarName);
105     }
106     
107     public nothrow void ClearEnvironment()
108     {
109         Environment.Instance().Clear();
110     }
111 
112     public cdecl nothrow void StartupSetupEnvironment(const char** envp)
113     {
114         environment = envp;
115     }
116 }
