 6 namespace System
 7 {
 8     public const int EFAIL = 1;
 9     public const int EBADF = 2;
10     public const int EPARAM = 3;
11     public const int EMEMORYACCESS = 4;
12     public const int ENOTFOUND = 5;
13     public const int EALREADYEXISTS = 6;
14     public const int ELIMITEXCEEDED = 7;
15     public const int ENOCHILD = 8;
16  public const int EHOST = 9;
17  public const int EPERMISSION = 10;
18     
19     public class SystemError : Exception
20     {
21         public nothrow SystemError(int errorCode_, const string& message_) : base(message_), errorCode(errorCode_)
22         {
23         }
24         public int errorCode;
25     }
26     
27     public SystemError GetSystemError()
28     {
29         int errorCode;
30         UniquePtr<byte> buffer(cast<byte*>(MemAlloc(4096)));
31         if (get_system_error(&errorCode, buffer.Get(), 4096) == 0)
32         {
33             string errorMessage;
34             byte* p = buffer.Get();
35             while (*p != 0u)
36             {
37                 errorMessage.Append(cast<char>(*p));
38                 ++p;
39             }
40             return SystemError(errorCode, errorMessage);
41         }
42         return SystemError(EFAIL, "could not get system error");
43     }
44     
45     public void ThrowSystemError()
46     {
47         throw GetSystemError();
48     }
49 }
