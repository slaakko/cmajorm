 41     public void Stat(const char* path, FileStatus& status)
 42     {
 43         byte[statBufSize] statBuf;
 44         int result = stat(path, &statBuf[0], statBufSize);
 45         if (result == -1)
 46         {
 47             ThrowSystemError();
 48         }
 49         GetFileStatus(&statBuf[0], status);
 50     }

 52     public void GetFileStatus(byte* statBuf, FileStatus& status)
 53     {
 54         MemoryReader reader(statBuf, statBufSize);
 55         status.fileSystemNumber = reader.ReadInt();
 56         status.inodeNumber = reader.ReadInt();
 57         status.fileType = cast<FileType>(reader.ReadInt());
 58         status.ownerAccess = cast<Access>(reader.ReadInt());
 59         status.groupAccess = cast<Access>(reader.ReadInt());
 60         status.otherAccess = cast<Access>(reader.ReadInt());
 61         status.nlinks = reader.ReadInt();
 62         status.uid = reader.ReadInt();
 63         status.gid = reader.ReadInt();
 64         status.fileSize = reader.ReadLong();
 65         status.ctime = reader.ReadDateTime();
 66         status.mtime = reader.ReadDateTime();
 67         status.atime = reader.ReadDateTime();
 68     }

 70     public void UTime(const char* path, const DateTime& atime, const DateTime& mtime)
 71     {
 72         byte[16] timeBuf;
 73         MemoryWriter writer(&timeBuf[0], 16);
 74         writer.Write(atime);
 75         writer.Write(mtime);
 76         UTime(path, &timeBuf[0], 16);
 77     }

163         public static string ReadAllText(const string& filePath)
164         {
165             StreamReader reader = OpenRead(filePath);
166             string content = reader.ReadToEnd();
167             if (content.Length() >= 3 && cast<byte>(content[0]) == 0xEFu && cast<byte>(content[1]) == 0xBBu && cast<byte>(content[2]) == 0xBFu)
168             {
169                 return content.Substring(3);
170             }
171             else
172             {
173                 return content;
174             }
175         }
176         public static List<string> ReadAllLines(const string& filePath)
177         {
178             List<string> lines;
179             bool start = true;
180             StreamReader reader = OpenRead(filePath);
181             string line = reader.ReadLine();
182             while (!reader.Eof())
183             {
184                 if (start)
185                 {
186                     if (line.Length() >= 3 && cast<byte>(line[0]) == 0xEFu && cast<byte>(line[1]) == 0xBBu && cast<byte>(line[2]) == 0xBFu)
187                     {
188                         line = line.Substring(3);
189                     }
190                     start = false;
191                 }
192                 lines.Add(line);
193                 line = reader.ReadLine();
194             }
195             if (!line.IsEmpty())
196             {
197                 lines.Add(line);
198             }
199             return lines;
200         }

203     public bool FileExists(const string& filePath)
204     {
205         byte[statBufSize] statBuf;
206         int result = stat(filePath.Chars(), &statBuf[0], statBufSize);
207         if (result == -1)
208         {
209             SystemError systemError = GetSystemError();
210             if (systemError.errorCode == ENOTFOUND)
211             {
212                 return false;
213             }
214             else
215             {
216                 ThrowSystemError();
217             }
218         }
219         else
220         {
221             FileStatus status;
222             GetFileStatus(&statBuf[0], status);
223             if (status.fileType == FileType.regular)
224             {
225                 return true;
226             }
227             else
228             {
229                 throw FileSystemException("path '" + filePath + "' is not a regular file path");
230             }
231         }
232     }

295     public void CopyFile(const string& sourceFilePath, const string& destFilePath, FileCopyOptions options)
296     {
297         DateTime sourceAccessTime;
298         DateTime sourceModificationTime;
299         GetFileTimes(sourceFilePath, sourceAccessTime, sourceModificationTime);
300         bool destinationExists = File.Exists(destFilePath);
301         bool copy = false;
302         if ((options & FileCopyOptions.update) != FileCopyOptions.none)
303         {
304             if (!destinationExists)
305             {
306                 copy = true;
307             }
308             else
309             {
310                 DateTime destModificationTime = GetFileModificationTime(destFilePath);
311                 if (sourceModificationTime > destModificationTime)
312                 {
313                     copy = true;
314                 }
315             }
316         }
317         else
318         {
319             copy = true;
320         }
321         if (copy)
322         {
323             bool addCrs = (options & FileCopyOptions.addCrs) != FileCopyOptions.none;
324             bool removeCrs = (options & FileCopyOptions.removeCrs) != FileCopyOptions.none;
325             if (addCrs && removeCrs)
326             {
327                 throw SystemError(EPARAM, "cannot have both 'FileCopyOptions.addCrs' and 'FileCopyOptions.removeCrs' option");
328             }
329             long size = File.Size(sourceFilePath);
330             bool bufferedCopy = (options & FileCopyOptions.noBufferedCopy) == FileCopyOptions.none && !addCrs && !removeCrs;
331             if (bufferedCopy)
332             {
333                 BufferedCopy(sourceFilePath, destFilePath, size);
334             }
335             else
336             {
337                 BinaryReader reader = File.OpenBinary(sourceFilePath);
338                 BinaryWriter writer = File.CreateBinary(destFilePath);
339                 for (long i = 0; i < size; ++i;)
340                 {
341                     byte x = reader.ReadByte();
342                     if (removeCrs && x == cast<byte>('\r'))
343                     {
344                         continue;
345                     }
346                     if (addCrs && x == cast<byte>('\n'))
347                     {
348                         writer.Write(cast<byte>('\r'));
349                     }
350                     writer.Write(x);
351                 }
352             }
353         }
354         if (copy)
355         {
356             if ((options & FileCopyOptions.dontPreserveTimestamps) == FileCopyOptions.none)
357             {
358                 File.SetTimes(destFilePath, sourceAccessTime, sourceModificationTime);
359             }
360             if ((options & FileCopyOptions.verbose) != FileCopyOptions.none)
361             {
362                 // DEBUG PID: Console.WriteLine(ToString(GetPid()) + " : " + GetFullPath(sourceFilePath) + " -> " + GetFullPath(destFilePath));
363                 Console.WriteLine(GetFullPath(sourceFilePath) + " -> " + GetFullPath(destFilePath));
364             }
365         }
366     }

368     public void BufferedCopy(const string& sourceFilePath, const string& destFilePath, long size)
369     {
370         FileStream sourceFileStream(sourceFilePath, OpenFlags.read);
371         FileStream destFileStream(destFilePath, cast<OpenFlags>(OpenFlags.write | OpenFlags.create | OpenFlags.truncate));
372         byte[4096] buffer;
373         while (size > 0)
374         {
375             long bytesRead = sourceFileStream.Read(&buffer[0], 4096);
376             destFileStream.Write(&buffer[0], bytesRead);
377             size = size - bytesRead;
378         }
379     }
