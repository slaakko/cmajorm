251 class SortBy : Rel<FileInfo>
252 {
253     public SortBy(SortField sortField_) : sortField(sortField_)
254     {
255     }
256     public bool operator()(const FileInfo& left, const FileInfo& right) const
257     {
258         switch (sortField)
259         {
260             case SortField.inode:
261             {
262                 if (left.fileStatus.inodeNumber < right.fileStatus.inodeNumber) return true;
263                 if (left.fileStatus.inodeNumber > right.fileStatus.inodeNumber) return false;
264                 break;
265             }
266             case SortField.nlinks:
267             {
268                 if (left.fileStatus.nlinks < right.fileStatus.nlinks) return true;
269                 if (left.fileStatus.nlinks > right.fileStatus.nlinks) return false;
270                 break;
271             }
272             case SortField.user:
273             {
274                 if (left.userName < right.userName) return true;
275     if (left.userName > right.userName) return false;
276                 break;
277             }
278             case SortField.size:
279             {
280                 if (left.fileStatus.fileSize > right.fileStatus.fileSize) return true;
281                 if (left.fileStatus.fileSize < right.fileStatus.fileSize) return false;
282                 break;
283             }
284             case SortField.time:
285             {
286                 if (left.fileStatus.mtime < right.fileStatus.mtime) return true;
287                 if (left.fileStatus.mtime > right.fileStatus.mtime) return false;
288                 break;
289             }
290         }
291         return left.fieldValues[left.nameFieldIndex].text < right.fieldValues[right.nameFieldIndex].text;
292     }
293     private SortField sortField;
294 }

530 string MakeLine(const List<FileInfo>& fileInfos, int lineNr, int numberOfLines, int numberOfColumns, int numberOfFields, bool hasColors, const Map<Pair<int, int>, int>& fieldWidths)
531 {

533     for (int c = 0; c < numberOfColumns; ++c;)
534     {
535         if (c > 0)
536         {
537             line.Append("  ");
538         }
539         int fileInfoIndex = c * numberOfLines + lineNr;
540         if (fileInfoIndex < fileInfos.Count())
541         {
542             const FileInfo& fileInfo = fileInfos[fileInfoIndex];
543             for (int f = 0; f < numberOfFields; ++f;)
544             {
545                 if (f > 0)
546                 {
547                     line.Append(' ');
548                 }
549                 const FieldValue& fieldValue = fileInfo.GetFieldValue(f);
550                 FormatJustify justify = GetFieldJustification(fieldValue.field);
551                 ustring text = Format(fieldValue.text, fieldWidths[MakePair(c, f)], justify);
552                 if (fieldValue.field == Fields.name)
553                 {
554                     line.Append(GetColorString(fileInfo, hasColors)).Append(ToUtf8(text)).Append(ResetColorString(hasColors));
555                 }
556                 else

558                     line.Append(ToUtf8(text));
559                 }
560             }
561         }
562     }
563     return line;
564 }

566 List<string> MakeLines(List<FileInfo>& fileInfos, SortField sortField, Fields fields, Flags flags, int screenWidth)
567 {
568     MakeFileInfos(fileInfos, fields, (flags & Flags.humanReadableSize) != Flags.none);
569     Sort(fileInfos, sortField);
570     if ((flags & Flags.reverse) != Flags.none)
571     {
572         Reverse(fileInfos.Begin(), fileInfos.End());
573     }
574     int numberOfFields = NumberOfFields(fields);
575     int numberOfColumns = 1;
576     int lineLength = 0;
577     int ncols = numberOfColumns;
578     if (screenWidth > 0 && (flags & Flags.longFormat) == Flags.none)
579     {
580         while (ncols < screenWidth)
581         {
582             int len = CalculateLineLength(fileInfos, ncols, numberOfFields, screenWidth);
583             if (len >= screenWidth)
584             {
585                 break;
586             }
587             else
588             {
589                 numberOfColumns = ncols;
590                 lineLength = len;
591                 ++ncols;
592             }
593         }
594     }
595     int numberOfLines = NumberOfLines(cast<int>(fileInfos.Count()), numberOfColumns);
596     Map<Pair<int, int>, int> fieldWidths;
597     for (int c = 0; c < numberOfColumns; ++c;)
598     {
599         for (int f = 0; f < numberOfFields; ++f;)
600         {
601             int w = CalculateFieldWidth(fileInfos, c, f, numberOfLines);
602             fieldWidths[MakePair(c, f)] = w;
603         }
604     }
605     List<string> lines;
606     for (int i = 0; i < numberOfLines; ++i;)
607     {
608         lines.Add(MakeLine(fileInfos, i, numberOfLines, numberOfColumns, numberOfFields, (flags & Flags.hasColors) != Flags.none, fieldWidths));
609     }
610     return lines;
611 }

624     if ((flags & Flags.longFormat) != Flags.none)
625     {
626         fields = LongFields();
627     }
628     else
629     {
630         fields = ShortFields();
631     }
632     fields = cast<Fields>(fields | additionalFields);
633     List<FileInfo> fileInfos;
634     DirectoryEntry dirEntry;
635     for (const string& file : files)
636     {
637         FileStatus fileStatus;
638         Stat(file.Chars(), fileStatus);
639         DirectoryEntry entry(fileStatus.inodeNumber, file);
640         if (fileStatus.fileType == FileType.regular)
641         {
642             fileInfos.Add(FileInfo(fileStatus, entry));
643         }
644         else if (fileStatus.fileType == FileType.directory)
645         {
646             List<FileInfo> dirFileInfos;
647             DirectoryReader reader(file);
648             while (reader.Read(dirEntry))
649             {
650                 if (dirEntry.name.StartsWith("."))
651                 {
652                     if ((flags & Flags.all) == Flags.none)
653                     {
654                         continue;
655                     }
656                 }
657                 string path = Path.Combine(file, dirEntry.name);
658                 FileStatus status;
659                 Stat(path.Chars(), status);
660                 dirFileInfos.Add(FileInfo(status, dirEntry));
661             }
662             if (!dirFileInfos.IsEmpty())
663             {
664                 List<string> lines = MakeLines(dirFileInfos, sortField, fields, flags, screenWidth);
665                 PrintLines(lines);
666             }
667         }
668     }
669     if (!fileInfos.IsEmpty())
670     {
671         List<string> lines = MakeLines(fileInfos, sortField, fields, flags, screenWidth);
672         PrintLines(lines);
673     }
674 }
