 35 bool CheckPassword(int uid, const string& hash)
 36 {
 37     StreamReader reader = File.OpenRead("/etc/passwd_hash");
 38     int lineNumber = 0;
 39     while (!reader.EndOfStream())
 40     {
 41         string line = reader.ReadLine();
 42         if (!line.IsEmpty() && !line.StartsWith("#"))
 43         {
 44             List<string> fields = line.Split(':');
 45             if (fields.Count() != 2)
 46             {
 47                 throw SecurityException("invalid line " + ToString(lineNumber) + " in /etc/passwd_hash: wrong number of fields (" + ToString(fields.Count()) + "), should be 2.");
 48             }
 49             int puid = ParseInt(fields[0]);
 50             if (puid == uid)
 51             {
 52                 if (hash == fields[1])
 53                 {
 54                     return true;
 55                 }
 56                 else
 57                 {
 58                     return false;
 59                 }
 60             }
 61         }
 62         ++lineNumber;
 63     }
 64     throw SecurityException("no matching user account for UID " + ToString(uid) + " found from /etc/passwd_hash");
 65 }

 67 int main(int argc, const char** argv)
 68 {
 69     try
 70     {
 71         int euid = GetEUID();
 72         if (euid != 0)
 73         {
 74             throw SecurityException("cannot login: login needs effective rights (EUID) of root account");
 75         }
 76         string userName;
 77         bool prevWasUser = false;
 78         for (int i = 1; i < argc; ++i;)
 79         {
 80             string arg = argv[i];
 81             if (arg.StartsWith("--"))
 82             {
 83                 if (arg == "--help")
 84                 {
 85                     PrintHelp();
 86                     return 1;
 87                 }
 88                 else if (arg == "--user")
 89                 {
 90                     prevWasUser = true;
 91                 }
 92                 else
 93                 {
 94                     throw Exception("unknown option '" + arg + "'");
 95                 }
 96             }
 97             else if (arg.StartsWith("-"))
 98             {
 99                 string options = arg.Substring(1);
100                 if (options.IsEmpty())
101                 {
102                     throw Exception("unknown argument '" + arg + "'");
103                 }
104                 else
105                 {
106                     bool unknown = false;
107                     string uo;
108                     for (char o : options)
109                     {
110                         switch (o)
111                         {
112                             case 'h':
113                             {
114                                 PrintHelp();
115                                 return 1;
116                             }
117                             case 'u':
118                             {
119                                 prevWasUser = true;
120                                 break;
121                             }
122                             default:
123                             {
124                                 unknown = true;
125                                 uo.Append(o);
126                                 break;
127                             }
128                         }
129                         if (unknown)
130                         {
131                             throw Exception("unknown option '-" + uo + "'");
132                         }
133                     }
134                 }
135             }
136             else if (prevWasUser)
137             {
138                 prevWasUser = false;
139                 userName = arg;
140             }
141         }
142         User* user = null;
143         Users users = GetUsers();
144         if (userName.IsEmpty())
145         {
146             int uid = GetUID();
147             user = users.GetUser(uid);
148         }
149         else
150         {
151             user = users.GetUser(userName);
152         }
153         if (user == null)
154         {
155             throw SecurityException("user '" + userName + "' not found");
156         }
157         if (!user->HasPassword())
158         {
159             return 0;
160         }
161         int secs = 5;
162         for (int i = 0; i < 3; ++i;)
163         {
164             Console.Out() << "password for " << user->Name() << ": ";
165             EchoSetter echoSetter;
166             string password = Console.ReadLine();
167             string hash = GetSha1MessageDigest(password);
168             if (CheckPassword(user->UID(), hash))
169             {
170                 return 0;
171             }
172             else
173             {
174                 Sleep(Duration.FromSeconds(secs));
175                 secs = 2 * secs;
176                 if (i < 2)
177                 {
178                     Console.Out() << "incorrect password, try again" << endl();
179                 }
180             }
181         }
182         return 1;
183     }
184     catch (const Exception& ex)
185     {
186         Console.Error() << ex.ToString() << endl();
187         return 1;
188     }
189     return 0;
190 }
