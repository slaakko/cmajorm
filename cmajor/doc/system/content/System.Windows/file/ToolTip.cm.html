  6 using System;
  7 using System.Collections;
  8 
  9 namespace System.Windows
 10 {
 11     public class ToolTip : Control
 12     {
 13         private enum Flags : sbyte
 14         {
 15             none = 0, changed = 1 << 0
 16         }
 17         public ToolTip(const Color& backgroundColor, const Color& textColor, const Color& frameColor, const string& text_, const Point& location_, const Font& font_) : 
 18             base("System.Windows.ToolTip", DefaultWindowClassStyle(), HiddenChildWindowStyle(), DefaultExtendedWindowStyle(), backgroundColor, text_, location_, Size(0, 0), Dock.none, Anchors.none), 
 19             framePen(frameColor), font(font_), textBrush(textColor), format(StringAlignment.near, StringAlignment.near), textHeight(0)
 20         {
 21             SetChanged();
 22         }
 23         public ToolTip() : this(Color.FloralWhite(), Color.Black(), Color.Black(), "", Point(0, 0), Font(FontFamily("Segoe UI"), 9.0f))
 24         {
 25         }
 26         public void SetTextColor(const Color& textColor)
 27         {
 28             textBrush = SolidBrush(textColor);
 29             Invalidate();
 30         }
 31         public void SetFrameColor(const Color& frameColor)
 32         {
 33             framePen = Pen(frameColor);
 34             Invalidate();
 35         }
 36         public void SetFont(const Font& font_)
 37         {
 38             font = font_;
 39             SetChanged();
 40             Invalidate();
 41         }
 42         public void MeasureExtent()
 43         {
 44             Graphics graphics = Graphics.FromWindowHandle(Handle());
 45             Measure(graphics);
 46         }
 47         protected override void OnTextChanged()
 48         {
 49             base->OnTextChanged();
 50             lines = SplitIntoLines(Text());
 51             Invalidate();
 52         }
 53         protected override void OnPaint(PaintEventArgs& args)
 54         {
 55             if (Changed())
 56             {
 57                 ResetChanged();
 58                 Measure(args.graphics);
 59             }
 60             Rect r(Point(), GetSize());
 61             r.size.w = r.size.w - 1;
 62             r.size.h = r.size.h - 1;
 63             args.graphics.Clear(BackgroundColor());
 64             args.graphics.DrawRectangleChecked(framePen, r);
 65             PointF pt(1, 1);
 66             for (const string& line : lines)
 67             {
 68                 args.graphics.DrawStringChecked(line, font, pt, textBrush);
 69                 pt.y = pt.y + textHeight;
 70             }
 71             base->OnPaint(args);
 72         }
 73         private void Measure(Graphics& graphics)
 74         {
 75             Size size;
 76             for (const string& line : lines)
 77             {
 78                 RectF textRect = graphics.MeasureStringChecked(line, font, PointF(0, 0), format);
 79                 textHeight = Max(textHeight, cast<int>(textRect.size.h));
 80                 size.w = Max(size.w, cast<int>(textRect.size.w));
 81             }
 82             size.h = cast<int>(lines.Count()) * textHeight;
 83             size.w = size.w + 1;
 84             size.h = size.h + 1;
 85             SetSize(size);
 86         }
 87         private inline nothrow bool Changed() const
 88         {
 89             return (flags & Flags.changed) != Flags.none;
 90         }
 91         private inline nothrow void SetChanged()
 92         {
 93             flags = cast<Flags>(flags | Flags.changed);
 94         }
 95         private inline nothrow void ResetChanged()
 96         {
 97             flags = cast<Flags>(flags & ~Flags.changed);
 98         }
 99         private Flags flags;
100         private Pen framePen;
101         private Font font;
102         private SolidBrush textBrush;
103         private StringFormat format;
104         private List<string> lines;
105         private int textHeight;
106     }
107 
108     public nothrow List<string> SplitIntoLines(const string& text)
109     {
110         List<string> lines;
111         string line;
112         int state = 0;
113         for (char c : text)
114         {
115             switch (state)
116             {
117                 case 0:
118                 {
119                     switch (c)
120                     {
121                         case '\n':
122                         {
123                             lines.Add(line);
124                             line.Clear();
125                             break;
126                         }
127                         case '\r':
128                         {
129                             state = 1;
130                             break;
131                         }
132                         default:
133                         {
134                             line.Append(c);
135                             break;
136                         }
137                     }
138                     break;
139                 }
140                 case 1:
141                 {
142                     switch (c)
143                     {
144                         case '\n':
145                         {
146                             lines.Add(line);
147                             line.Clear();
148                             state = 0;
149                             break;
150                         }
151                     }
152                     break;
153                 }
154             }
155         }
156         if (!line.IsEmpty())
157         {
158             lines.Add(line);
159         }
160         return lines;
161     }
