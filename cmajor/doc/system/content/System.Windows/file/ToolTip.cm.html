  6 using System;
  7 using System.Collections;
  8 
  9 namespace System.Windows
 10 {
 11     public class ToolTip : Control
 12     {
 13         private enum Flags : sbyte
 14         {
 15             none = 0, changed = 1 << 0
 16         }
 17         public ToolTip(const Color& backgroundColor, const Color& textColor, const Color& frameColor, const string& text_, const Point& location_, const Font& font_) : 
 18             base("System.Windows.ToolTip", DefaultWindowClassStyle(), HiddenChildWindowStyle(), DefaultExtendedWindowStyle(), backgroundColor, text_, location_, Size(0, 0), Dock.none, Anchors.none), 
 19             framePen(frameColor), font(font_), textBrush(textColor), format(StringAlignment.near, StringAlignment.near), textHeight(0)
 20         {
 21             SetChanged();
 22         }
 23         public ToolTip(const Color& backgroundColor) : this(backgroundColor, Color.Black(), Color.Black(), "", Point(0, 0), Font(FontFamily("Segoe UI"), 9.0f))
 24         {
 25         }
 26         public ToolTip() : this(Color.FloralWhite())
 27         {
 28         }
 29         public void SetTextColor(const Color& textColor)
 30         {
 31             textBrush = SolidBrush(textColor);
 32             Invalidate();
 33         }
 34         public void SetFrameColor(const Color& frameColor)
 35         {
 36             framePen = Pen(frameColor);
 37             Invalidate();
 38         }
 39         public void SetFont(const Font& font_)
 40         {
 41             font = font_;
 42             SetChanged();
 43             Invalidate();
 44         }
 45         public void MeasureExtent()
 46         {
 47             Graphics graphics = Graphics.FromWindowHandle(Handle());
 48             Measure(graphics);
 49         }
 50         protected override void OnTextChanged()
 51         {
 52             base->OnTextChanged();
 53             lines = SplitIntoLines(Text());
 54             Invalidate();
 55         }
 56         protected override void OnPaint(PaintEventArgs& args)
 57         {
 58             if (Changed())
 59             {
 60                 ResetChanged();
 61                 Measure(args.graphics);
 62             }
 63             Rect r(Point(), GetSize());
 64             r.size.w = r.size.w - 1;
 65             r.size.h = r.size.h - 1;
 66             args.graphics.Clear(BackgroundColor());
 67             args.graphics.DrawRectangleChecked(framePen, r);
 68             PointF pt(1, 1);
 69             for (const string& line : lines)
 70             {
 71                 args.graphics.DrawStringChecked(line, font, pt, textBrush);
 72                 pt.y = pt.y + textHeight;
 73             }
 74             base->OnPaint(args);
 75         }
 76         private void Measure(Graphics& graphics)
 77         {
 78             Size size;
 79             for (const string& line : lines)
 80             {
 81                 RectF textRect = graphics.MeasureStringChecked(line, font, PointF(0, 0), format);
 82                 textHeight = Max(textHeight, cast<int>(textRect.size.h));
 83                 size.w = Max(size.w, cast<int>(textRect.size.w));
 84             }
 85             size.h = cast<int>(lines.Count()) * textHeight;
 86             size.w = size.w + 1;
 87             size.h = size.h + 1;
 88             SetSize(size);
 89         }
 90         private inline nothrow bool Changed() const
 91         {
 92             return (flags & Flags.changed) != Flags.none;
 93         }
 94         private inline nothrow void SetChanged()
 95         {
 96             flags = cast<Flags>(flags | Flags.changed);
 97         }
 98         private inline nothrow void ResetChanged()
 99         {
100             flags = cast<Flags>(flags & ~Flags.changed);
101         }
102         private Flags flags;
103         private Pen framePen;
104         private Font font;
105         private SolidBrush textBrush;
106         private StringFormat format;
107         private List<string> lines;
108         private int textHeight;
109     }
110 
111     public nothrow List<string> SplitIntoLines(const string& text)
112     {
113         List<string> lines;
114         string line;
115         int state = 0;
116         for (char c : text)
117         {
118             switch (state)
119             {
120                 case 0:
121                 {
122                     switch (c)
123                     {
124                         case '\n':
125                         {
126                             lines.Add(line);
127                             line.Clear();
128                             break;
129                         }
130                         case '\r':
131                         {
132                             state = 1;
133                             break;
134                         }
135                         default:
136                         {
137                             line.Append(c);
138                             break;
139                         }
140                     }
141                     break;
142                 }
143                 case 1:
144                 {
145                     switch (c)
146                     {
147                         case '\n':
148                         {
149                             lines.Add(line);
150                             line.Clear();
151                             state = 0;
152                             break;
153                         }
154                     }
155                     break;
156                 }
157             }
158         }
159         if (!line.IsEmpty())
160         {
161             lines.Add(line);
162         }
163         return lines;
164     }
