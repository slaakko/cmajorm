 47         internal void CreateWindowInternal()
 48         {
 49             if (handle != null) return;
 50             Control* parentControl = null;
 51             Container* container = GetContainer();
 52             if (container != null)
 53             {
 54                 Component* parent = container->Parent();
 55                 if (parent != null && (parent is Control*))
 56                 {
 57                     parentControl = cast<Control*>(parent);
 58                 }
 59             }
 60             if (Application.GetWindowManager().IsSystemClassName(windowClassName))
 61             {
 62                 handle = CreateWindowByClassName(windowClassName, text.Chars(), style, exStyle, location, size, parentControl);
 63             }
 64             else
 65             {
 66                 ushort windowClass = Application.GetWindowManager().RegisterWindowClass(windowClassName, windowClassStyle, 0);
 67                 handle = CreateWindowByClassAtom(windowClass, text.Chars(), style, exStyle, location, size, parentControl);

 69             Application.GetWindowManager().AddWindow(this);
 70             WinRect clientRect = GetClientRect();
 71             if ((style & WindowStyle.WS_CHILD) == 0)
 72             {
 73                 WinRect windowRect = GetWindowRect();
 74                 location.x = windowRect.left;
 75                 location.y = windowRect.top;
 76             }
 77             size.w = clientRect.right - clientRect.left;
 78             size.h = clientRect.bottom - clientRect.top;

215         internal void DockWindow(Rect& parentRect)
216         {
217             if (dock == Dock.none) return;
218             Point parentLoc = parentRect.location;
219             Size parentSize = parentRect.size;
220             Point newLocation = location;
221             Size newSize = size;
222             switch (dock)
223             {
224                 case Dock.left:
225                 {
226                     newLocation = parentLoc;
227                     newSize.h = parentSize.h;
228                     parentRect.location = Point(parentLoc.x + size.w, parentLoc.y);
229                     parentRect.size = Size(parentSize.w - size.w, parentSize.h);
230                     break;
231                 }
232                 case Dock.top:
233                 {
234                     newLocation = parentLoc;
235                     newSize.w = parentSize.w;
236                     parentRect.location = Point(parentLoc.x, parentLoc.y + size.h);
237                     parentRect.size = Size(parentSize.w, parentSize.h - size.h);
238                     break;
239                 }
240                 case Dock.right:
241                 {
242                     newLocation = Point(parentLoc.x + parentSize.w - size.w, parentLoc.y);
243                     newSize.h = parentSize.h;
244                     parentRect.size = Size(parentSize.w - size.w, parentSize.h);
245                     break;
246                 }
247                 case Dock.bottom:
248                 {
249                     newLocation = Point(parentLoc.x, parentLoc.y + parentSize.h - size.h);
250                     newSize.w = parentSize.w;
251                     parentRect.size = Size(parentSize.w, parentSize.h - size.h);
252                     break;
253                 }
254                 case Dock.fill:
255                 {
256                     newLocation = parentLoc;
257                     newSize = parentSize;
258                     parentRect.location = Point();
259                     parentRect.size = Size();
260                     break;
261                 }
262             }
263             if (location != newLocation || size != newSize)
264             {
265                 location = newLocation;
266                 size = newSize;
267                 MoveWindow(handle, location, size, true);
268             }

270         internal void MoveWindow(int dx, int dy)

273             Size newSize = size;

276             if (!leftAnchored)
277             {
278                 newLocation.x = newLocation.x + dx;
279             }
280             else if (rightAnchored)
281             {
282                 newSize.w = newSize.w + dx;
283             }
284             bool topAnchored = cast<Anchors>(anchors & Anchors.top) != Anchors.none;
285             bool bottomAnchored = cast<Anchors>(anchors & Anchors.bottom) != Anchors.none;
286             if (!topAnchored)
287             {
288                 newLocation.y = newLocation.y + dy;
289             }
290             else if (bottomAnchored)
291             {
292                 newSize.h = newSize.h + dy;
293             }
294             if (location != newLocation || size != newSize)
295             {
296                 location = newLocation;
297                 size = newSize;
298                 MoveWindow(handle, location, size, true);
299             }

333         protected virtual bool ProcessMessage(Message& message)
334         {

346                     DoMouseMove(args);
347                     message.result = 0;
348                     return true;
349                 }
350                 case WM_MOUSELEAVE:
351                 {
352                     DoMouseLeave();
353                     message.result = 0;
354                     return true;
355                 }
356                 case WM_LBUTTONDOWN:
357                 {
358                     MouseEventArgs args(Point(message.LParamX(), message.LParamY()), cast<MouseButtons>(message.wparam | MouseButtons.lbutton));
359                     DoMouseDown(args);
360                     message.result = 0;
361                     return true;
362                 }

366                     DoMouseUp(args);
367                     message.result = 0;
368                     return true;

374                     DoKeyPress(args);
375                     if (args.handled)
376                     {

378                         return true;
379                     }
380                     break;
381                 }
382                 case WM_KEYDOWN:

384                     int virtualKeyCode = cast<int>(message.wparam);
385                     if (DoKeyDown(virtualKeyCode))
386                     {
387                         message.result = 0;
388                         return true;
389                     }
390                     break;
391                 }
392                 case WM_KEYUP:
393                 {
394                     int virtualKeyCode = cast<int>(message.wparam);
395                     if (DoKeyUp(virtualKeyCode))
396                     {
397                         message.result = 0;
398                         return true;
399                     }
400                     break;
401                 }

403                 {
404                     if (DoSysCommand(message.wparam, message.lparam))
405                     {
406                         message.result = 0;
407                         return true;
408                     }
409                     break;
410                 }
411             }
412             return false;

434                     }
