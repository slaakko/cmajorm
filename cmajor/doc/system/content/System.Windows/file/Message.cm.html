 99     public class Message
100     {
101         public nothrow Message(void* windowHandle_, uint msg_, uint wparam_, long lparam_, long result_) : windowHandle(windowHandle_), msg(msg_), wparam(wparam_), lparam(lparam_), result(result_)
102         {
103         }
104         public inline nothrow ushort WParamLoWord() const
105         {
106             return cast<ushort>(wparam & 0xFFFFu);
107         }
108         public inline nothrow ushort WParamHiWord() const
109         {
110             return cast<ushort>((wparam >> 16u) & 0xFFFFu);
111         }
112         public inline nothrow void* LParamHandle() const
113         {
114             return cast<void*>(cast<ulong>(lparam));
115         }
116         public inline nothrow uint LParamLoDWord() const
117         {
118             return cast<uint>(lparam);
119         }
120         public inline nothrow uint LParamHiDWord() const
121         {
122             return cast<uint>((lparam >> 32u) & 0xFFFFFFFFu);
123         }
124         public inline nothrow int LParamX() const
125         {
126             return cast<int>(cast<short>(LParamLoDWord()));
127         }
128         public inline nothrow int LParamY() const
129         {
130             return cast<int>(cast<short>(LParamLoDWord() >> 16u));
131         }
132         public inline nothrow Size LParamSize() const
133         {
134             uint s = LParamLoDWord();
135             ushort sw = cast<ushort>(s);
136             ushort sh = cast<ushort>(s >> 16u);
137             return Size(cast<int>(sw), cast<int>(sh));
138         }
139         public void* windowHandle;
140         public uint msg;
141         public uint wparam;
142         public long lparam;
143         public long result;
144     }
