 6 using System;
 7 using System.Collections;
 8 
 9 namespace System.Windows
10 {
11     public abstract class Resource
12     {
13         public virtual default ~Resource();
14     }
15 
16     public class ResourceManager
17     {
18         public Resource* GetResource(const string& resourceName)
19         {
20             HashMap<string, Resource*>.ConstIterator it = resourceMap.CFind(resourceName);
21             if (it != resourceMap.CEnd())
22             {
23                 return it->second;
24             }
25             return null;
26         }
27         public Cursor& GetCursor(const string& cursorName)
28         {
29             Resource* result = GetResource(cursorName);
30             if (result != null && result is Cursor*)
31             {
32                 return *cast<Cursor*>(result);
33             }
34             UniquePtr<Resource> resource(new Cursor(LoadCursor(cursorName)));
35             Cursor* cursor = cast<Cursor*>(resource.Get());
36             resources.Add(Rvalue(resource));
37             return *cursor;
38         }
39         public Icon& GetIcon(const string& iconName)
40         {
41             Resource* result = GetResource(iconName);
42             if (result != null && result is Icon*)
43             {
44                 return *cast<Icon*>(result);
45             }
46             UniquePtr<Resource> resource(new Icon(LoadIcon(iconName)));
47             Icon* icon = cast<Icon*>(resource.Get());
48             resources.Add(Rvalue(resource));
49             return *icon;
50         }
51         public WinBitmap& GetBitmap(const string& bitmapName)
52         {
53             Resource* result = GetResource(bitmapName);
54             if (result != null && result is WinBitmap*)
55             {
56                 return *cast<WinBitmap*>(result);
57             }
58             UniquePtr<Resource> resource(new WinBitmap(LoadBitmap(bitmapName)));
59             WinBitmap* bitmap = cast<WinBitmap*>(resource.Get());
60             resources.Add(Rvalue(resource));
61             return *bitmap;
62         }
63         private HashMap<string, Resource*> resourceMap;
64         private List<UniquePtr<Resource>> resources;
65     }
66 }
