503         private void SetVisibility(Graphics& graphics)
504         {
505             Component* component = tabPages.FirstChild();
506             TabPage* firstVisibleTabPage = cast<TabPage*>(component);
507             int width = GetSize().w;
508             int sum = leadingWidth;
509             bool selectedPassed = false;
510             while (component != null)
511             {
512                 TabPage* tabPage = cast<TabPage*>(component);
513                 int w = tabPage->tab.width;
514                 sum = sum + w;
515                 if (tabPage == selectedTabPage)
516                 {
517                     if (sum < width)
518                     {
519                         firstVisibleTabPage->tab.visible = true;
520                         while (firstVisibleTabPage != selectedTabPage)
521                         {
522                             firstVisibleTabPage = cast<TabPage*>(firstVisibleTabPage->NextSibling());
523                             firstVisibleTabPage->tab.visible = true;
524                         }
525                         selectedPassed = true;
526                     }
527                     else
528                     {
529                         sum = sum - firstVisibleTabPage->tab.width;
530                         firstVisibleTabPage->tab.visible = false;
531                         component = firstVisibleTabPage;
532                     }
533                 }
534                 else if (selectedPassed)
535                 {
536                     if (sum < width)
537                     {
538                         tabPage->tab.visible = true;
539                     }
540                     else
541                     {
542                         tabPage->tab.visible = false;
543                     }
544                 }
545                 sum = sum - overlapWidth;
546                 component = component->NextSibling();
547             }
548         }

781         internal void MeasureWidthAndHeight(Graphics& graphics, TabControl* tabControl)
782         {
783             if (tab.width == 0)
784             {
785                 RectF textRect = graphics.MeasureStringChecked(Text(), tabControl->GetFont(), PointF(), tabControl->GetStringFormat());
786                 RectF closeRect = graphics.MeasureStringChecked("x", tabControl->GetFont(), PointF(), tabControl->GetStringFormat());
787                 tab.textHeight = textRect.size.h;
788                 tab.textWidth = textRect.size.w;
789                 tab.closeBoxWidth = closeRect.size.w;
790                 tab.height = cast<int>(tabControl->TabPadding().Vertical() + tabControl->TabCloseBoxPadding().Vertical() + tab.textHeight);
791                 tab.width = cast<int>(tabControl->TabPadding().Horizontal() + tab.textWidth + tabControl->OverlapWidth() + 
792                     tabControl->TabCloseBoxPadding().Horizontal() + tab.closeBoxWidth);
793             }
794             tabControl->SetHeaderHeight(Max(tabControl->HeaderHeight(), tab.height + tabControl->TopMarginHeight()));
795         }
796         internal void CalculateMetrics(Graphics& graphics, TabControl* tabControl, int& left)
797         {
798             if (!tab.visible) return;
799             float roundingRadius = tabControl->RoundingRadius();
800             int topMarginHeight = tabControl->TopMarginHeight();
801             tab.left = left;
802             tab.leftRoundingRect = RectF(PointF(), SizeF(2 * roundingRadius, 2 * roundingRadius));
803             tab.leftRoundingRect.Offset(left, topMarginHeight);
804             tab.rightRoundingRect = RectF(PointF(), SizeF(2 * roundingRadius, 2 * roundingRadius));
805             tab.rightRoundingRect.Offset(left + tab.width - 2 * roundingRadius, topMarginHeight);
806             tab.topRect = RectF(PointF(), SizeF(tab.width - 2 * roundingRadius, roundingRadius));
807             tab.topRect.Offset(left + roundingRadius, topMarginHeight);
808             tab.bottomRect = RectF(PointF(), SizeF(tab.width, tab.height - roundingRadius + 1));
809             tab.bottomRect.Offset(left, topMarginHeight + roundingRadius);
810             tab.textRect = RectF(PointF(), SizeF(tab.width - (tabControl->TabCloseBoxPadding().Horizontal() + tab.closeBoxWidth), tab.height));
811             tab.textRect.Offset(left, topMarginHeight);
812             tab.selectRect = Rect(Point(tab.left, topMarginHeight), 
813                 Size(cast<int>(tab.width - (tab.closeBoxWidth + tabControl->TabCloseBoxPadding().right)), tab.height));
814             tab.closeBoxRect = RectF(
815                 PointF(tab.left + tab.selectRect.size.w, 
816                     topMarginHeight + tabControl->TabPadding().top + tabControl->TabCloseBoxPadding().top), 
817                 SizeF(tab.closeBoxWidth, tab.closeBoxWidth));
818             tab.closeRect = Rect(Point(cast<int>(tab.closeBoxRect.location.x), cast<int>(tab.closeBoxRect.location.y)), 
819                 Size(cast<int>(tab.closeBoxRect.size.w), cast<int>(tab.closeBoxRect.size.h)));
820             tab.closeRect.Inflate(3, 3);
821             tab.closeBoxRect.Inflate(-1, -1);
822             left = left + tab.width - tabControl->OverlapWidth();
823         }
824         internal void DrawTab(Graphics& graphics, TabControl* tabControl)
825         {
826             if (!tab.visible) return;
827             Brush* backgroundBrush = null;
828             if (this == tabControl->SelectedTabPage())
829             {
830                 const Brush& brush = tabControl->TabSelectedBackgroundBrush();
831                 backgroundBrush = &brush;
832             }
833             else
834             {
835                 const Brush& brush = tabControl->TabNormalBackgroundBrush();
836                 backgroundBrush = &brush;
837             }
838             const Pen& framePen = tabControl->FramePen();
839             const Pen& closeBoxPen = tabControl->CloseBoxPen();
840             int topMarginHeight = tabControl->TopMarginHeight();
841             float roundingRadius = tabControl->RoundingRadius();
842             graphics.FillEllipseChecked(*backgroundBrush, tab.leftRoundingRect);
843             graphics.FillEllipseChecked(*backgroundBrush, tab.rightRoundingRect);
844             graphics.FillRectangleChecked(*backgroundBrush, tab.topRect);
845             graphics.FillRectangleChecked(*backgroundBrush, tab.bottomRect);
846             graphics.DrawLineChecked(framePen, PointF(tab.left, tab.bottomRect.location.y + tab.bottomRect.size.h), PointF(tab.left, tab.bottomRect.location.y));
847             graphics.DrawArcChecked(framePen, tab.leftRoundingRect, -180.0f, 90.0f);
848             graphics.DrawStringChecked(Text(), tabControl->GetFont(), tab.textRect, tabControl->CenterFormat(), tabControl->TextBrush());
849             graphics.DrawLineChecked(framePen, 
850                 PointF(tab.leftRoundingRect.location.x + roundingRadius, topMarginHeight), 
851                 PointF(tab.rightRoundingRect.location.x + roundingRadius, topMarginHeight));
852                 graphics.DrawArcChecked(framePen, tab.rightRoundingRect, -90.0f, 90.0f);
853             graphics.DrawLineChecked(framePen, 
854                 PointF(tab.left + tab.width, tab.bottomRect.location.y), 
855                 PointF(tab.left + tab.width, tab.bottomRect.location.y + tab.bottomRect.size.h - 1));
856             if (tab.state == Tab.State.normal)
857             {
858                 RectF r = tab.closeBoxRect;
859                 r.Inflate(3, 3);
860                 graphics.FillRectangleChecked(*backgroundBrush, r);
861             }
862             else if (tab.state == Tab.State.closeBoxSelected)
863             {
864                 RectF r = tab.closeBoxRect;
865                 r.Inflate(3, 3);
866                 const Brush& selectedBrush = tabControl->CloseBoxSelectedBrush();
867                 graphics.FillRectangleChecked(selectedBrush, r);
868             }
869             graphics.DrawLineChecked(closeBoxPen, 
870                 tab.closeBoxRect.location, PointF(tab.closeBoxRect.location.x + tab.closeBoxRect.size.w, tab.closeBoxRect.location.y + tab.closeBoxRect.size.h));
871             graphics.DrawLineChecked(closeBoxPen, 
872                 PointF(tab.closeBoxRect.location.x, tab.closeBoxRect.location.y + tab.closeBoxRect.size.h), 
873                 PointF(tab.closeBoxRect.location.x + tab.closeBoxRect.size.w, tab.closeBoxRect.location.y));
874         }
