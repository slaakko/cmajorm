 6 using System;
 7 using System.Collections;
 8 using System.Windows.API;
 9 
10 namespace System.Windows
11 {
12     public class LogView : TextView
13     {
14         public LogView(const FontFamily& fontFamily, float fontSize, const Color& backgroundColor, const Color& textColor, const Point& location, const Size& size, 
15             Dock dock, Anchors anchors) : 
16             base(fontFamily, fontSize, backgroundColor, textColor, location, size, dock, anchors)
17         {
18             SetReadOnly();
19         }
20         public LogView(const Point& location, const Size& size, Dock dock, Anchors anchors) : 
21             this(FontFamily("Consolas"), 9.0f, Color.White(), Color.Black(), location, size, dock, anchors)
22         {
23         }
24         public LogView() : this(Point(0, 0), Size(0, 0), Dock.bottom, Anchors.none)
25         {
26         }
27         public virtual void WriteLine(const string& text)
28         {
29             int startLineNumber = cast<int>(Lines().Count() - 1);
30             List<ustring> lines = SplitTextIntoLines(ToUtf32(text));
31             for (const ustring& line : lines)
32             {
33                 AddLine(line);
34             }
35             SetCaretLineCol(cast<int>(Lines().Count()), 1);
36             ScrollToCaret();
37             SetChanged();
38             InvalidateLines(startLineNumber, cast<int>(startLineNumber + lines.Count()));
39         }
40         protected override void OnPaint(PaintEventArgs& args)
41         {
42             base->OnPaint(args);
43         }
44         protected override void OnMouseDown(MouseEventArgs& args)
45         {
46             base->OnMouseDown(args);
47         }
48         protected override void OnKeyDown(KeyEventArgs& args)
49         {
50             base->OnKeyDown(args);
51             if (CaretLine() == Lines().Count())
52             {
53                 Invalidate();
54             }
55         }
56     }
57 }
