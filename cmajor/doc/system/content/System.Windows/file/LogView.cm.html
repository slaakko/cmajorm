 6 using System;
 7 using System.Collections;
 8 
 9 namespace System.Windows
10 {
11     public class LogView : TextView
12     {
13         private enum Flags : sbyte
14         {
15             none = 0, frozen = 1 << 0
16         }
17         public LogView(const FontFamily& fontFamily, float fontSize, const Color& backgroundColor, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
18             base(fontFamily, fontSize, backgroundColor, location, size, dock, anchors), flags(Flags.none)
19         {
20         }
21         public LogView(const Point& location, const Size& size, Dock dock, Anchors anchors) : this(FontFamily("Consolas"), 9.0f, Color(223u, 232u, 247u), location, size, dock, anchors)
22         {
23         }
24         public LogView() : this(Point(0, 0), Size(0, 0), Dock.bottom, Anchors.none)
25         {
26         }
27         public void Clear()
28         {
29             Lines().Clear();
30             Invalidate();
31         }
32         public virtual void WriteLine(const string& text)
33         {
34             List<ustring> lines = SplitTextIntoLines(ToUtf32(text));
35             for (const ustring& line : lines)
36             {
37                 AddLine(line);
38             }
39             if (!Frozen())
40             {
41                 SetCaretLineCol(cast<int>(Lines().Count()), 1);
42                 ScrollToCaret();
43                 SetChanged();
44                 Invalidate();
45             }
46         }
47         protected override void OnPaint(PaintEventArgs& args)
48         {
49             base->OnPaint(args);
50         }
51         protected override void OnMouseDown(MouseEventArgs& args)
52         {
53             SetFrozen();
54             base->OnMouseDown(args);
55         }
56         protected override void OnKeyDown(KeyEventArgs& args)
57         {
58             base->OnKeyDown(args);
59             if (CaretLine() == Lines().Count())
60             {
61                 ResetFrozen();
62                 Invalidate();
63             }
64             else
65             {
66                 SetFrozen();
67             }
68         }
69         private inline nothrow bool Frozen() const
70         {
71             return (flags & Flags.frozen) != Flags.none;
72         }
73         private inline nothrow void SetFrozen()
74         {
75             flags = cast<Flags>(flags | Flags.frozen);
76         }
77         private inline nothrow void ResetFrozen()
78         {
79             flags = cast<Flags>(flags & ~Flags.frozen);
80         }
81         private Flags flags;
82     }
83 }
