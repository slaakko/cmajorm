 6 using System;
 7 
 8 namespace System.Windows
 9 {
10     public abstract class Component
11     {
12         public nothrow Component() : name(), container(null), nextSibling(null), prevSibling(null)
13         {
14         }
15         public virtual ~Component()
16         {
17         }
18         public inline nothrow Container* GetContainer() const
19         {
20             return container;
21         }
22         internal nothrow void SetContainer(Container* container_)
23         {
24             container = container_;
25         }
26         public inline nothrow Component* NextSibling() const
27         {
28             return nextSibling;
29         }
30         internal nothrow void SetNextSibling(Component* nextSibling_)
31         {
32             nextSibling = nextSibling_;
33         }
34         public inline nothrow Component* PrevSibling() const
35         {
36             return prevSibling;
37         }
38         internal nothrow void SetPrevSibling(Component* prevSibling_)
39         {
40             prevSibling = prevSibling_;
41         }
42         internal nothrow void LinkBefore(Component* component)
43         {
44             component->prevSibling = prevSibling;
45             component->nextSibling = this;
46             prevSibling = component;
47         }
48         internal nothrow void LinkAfter(Component* component)
49         {
50             component->prevSibling = this;
51             component->nextSibling = nextSibling;
52             nextSibling = component;
53         }
54         internal nothrow void Unlink()
55         {
56             if (prevSibling != null)
57             {
58                 prevSibling->nextSibling = nextSibling;
59             }
60             if (nextSibling != null)
61             {
62                 nextSibling->prevSibling = prevSibling;
63             }
64         }
65         private string name;
66         private Container* container;
67         private Component* nextSibling;
68         private Component* prevSibling;
69     }
70 }
