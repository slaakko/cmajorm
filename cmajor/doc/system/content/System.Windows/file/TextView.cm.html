532                         }

538                         if (lineNumber < lines.Count())
539                         {
540                             ++lineNumber;
541                             int lineLength = GetLineLength(lineNumber);
542                             columnNumber = Min(columnNumber, lineLength + 1 + LineNumberFieldLength());

567                         lineNumber = lineNumber + windowLines;

569                         args.handled = true;
570                         break;
571                     }
572                     case Keys.pageUp:

575                         int windowLines = cast<int>(size.h / charHeight);

578                         args.handled = true;

582                     {

651         public void ScrollToCaret()
652         {
653             if (lines.IsEmpty() || charHeight == 0) return;
654             Size size = GetSize();
655             int windowLines = cast<int>(size.h / charHeight);
656             int windowCols = cast<int>(size.w / charWidth);
657             Point currentOrigin = ContentLocation();
658             int topLineNumber = cast<int>(currentOrigin.y / charHeight + 1);
659             int leftColNumber = cast<int>(currentOrigin.x / charWidth + 1);
660             int oldTopLineNumber = topLineNumber;
661             if (caretLine > topLineNumber)
662             {
663                 while (caretLine - topLineNumber + 1 >= windowLines)
664                 {
665                     ++topLineNumber;
666                 }
667             }
668             else if (caretLine < topLineNumber)
669             {
670                 topLineNumber = caretLine;
671             }
672             else
673             {
674                 topLineNumber = Max(cast<int>(1), caretLine - 1);
675             }
676             int oldLeftColNumber = leftColNumber;
677             if (caretColumn >= leftColNumber)
678             {
679                 while (caretColumn - leftColNumber + 2 >= windowCols)
680                 {
681                     ++leftColNumber;
682                 }
683             }
684             else
685             {
686                 if (caretColumn == 1 + LineNumberFieldLength())
687                 {
688                     leftColNumber = 1;
689                 }
690                 else
691                 {
692                     leftColNumber = caretColumn;
693                 }
694             }
695             if (update || caretLine == 1 || topLineNumber != oldTopLineNumber || leftColNumber != oldLeftColNumber)
696             {
697                 update = false;
698                 Point newOrigin(cast<int>((leftColNumber - 1) * charWidth), cast<int>((topLineNumber - 1) * charHeight));
699                 topLine = topLineNumber;
700                 topLineDiff = 0.0f;
701                 leftCol = leftColNumber;
702                 leftColDiff = 0.0f;
703                 SetContentLocation(newOrigin);
704                 SetCaretLocation();
705                 Invalidate();
706             }

904                 }

913                     }

915                 }
916             }
