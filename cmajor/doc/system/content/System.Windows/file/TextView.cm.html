548         public void ScrollToCaret()
549         {
550             if (lines.IsEmpty() || charHeight == 0) return;
551             Size size = GetSize();
552             int windowLines = cast<int>(size.h / charHeight);
553             int windowCols = cast<int>(size.w / charWidth);
554             Point currentOrigin = ContentLocation();
555             int topLineNumber = cast<int>(currentOrigin.y / charHeight + 1);
556             int leftColNumber = cast<int>(currentOrigin.x / charWidth + 1);
557             int oldTopLineNumber = topLineNumber;
558             if (caretLine > topLineNumber)
559             {
560                 while (caretLine - topLineNumber + 1 >= windowLines)
561                 {
562                     ++topLineNumber;
563                 }
564             }
565             else if (caretLine < topLineNumber)
566             {
567                 topLineNumber = caretLine;
568             }
569             else
570             {
571                 topLineNumber = Max(cast<int>(1), caretLine - 1);
572             }
573             int oldLeftColNumber = leftColNumber;
574             if (caretColumn >= leftColNumber)
575             {
576                 while (caretColumn - leftColNumber + 2 >= windowCols)
577                 {
578                     ++leftColNumber;
579                 }
580             }
581             else
582             {
583                 if (caretColumn == 1 + LineNumberFieldLength())
584                 {
585                     leftColNumber = 1;
586                 }
587                 else
588                 {
589                     leftColNumber = caretColumn;
590                 }
591             }
592             if (update || caretLine == 1 || topLineNumber != oldTopLineNumber || leftColNumber != oldLeftColNumber)

734     {

736         ustring line;
737         int state = 0;
738         for (uchar c : text)
739         {
740             switch (state)
741             {
742                 case 0:
743                 {

746                         case '\n':
747                         {
748                             lines.Add(TrimEnd(line));
749                             line.Clear();
750                             break;
751                         }
752                         case '\r':
753                         {
754                             state = 1;
755                             break;
756                         }
757                         default:
758                         {
759                             line.Append(c);
760                             break;
761                         }
762                     }
763                     break;
764                 }
765                 case 1:
766                 {
767                     if (c == '\n')
768                     {
769                         lines.Add(TrimEnd(line));
770                         line.Clear();
771                         state = 0;
772                         break;
773                     }
774                     break;
775                 }
776             }
777         }
778         if (!line.IsEmpty())
779         {
780             lines.Add(TrimEnd(line));
