  8 namespace System.Windows

 10     public class delegate void WindowClosingEventHandler(bool& cancelClose);
 11     
 12     public class delegate void WindowClosedEventHandler();
 13

 20         {
 21         }
 22         public Window(const char* windowClassName, WindowStyle style, ExtendedWindowStyle exStyle, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) :

 28             this("System.Windows.Window", WindowStyle.WS_OVERLAPPEDWINDOW, ExtendedWindowStyle.WS_EX_DEFAULT, caption, Point.Default(), Size.Default(), Dock.none, Anchors.none)
 29         {
 30         }
 31         protected override bool ProcessMessage(Message& message)
 32         {
 33             switch (message.msg)
 34             {
 35                 case WM_CLOSE:
 36                 {
 37                     bool cancelClose = false;
 38                     OnWindowClosing(cancelClose);
 39                     if (cancelClose)
 40                     {
 41                         message.result = 0;
 42                         return true;
 43                     }
 44                     break;
 45                 }
 46                 case WM_DESTROY:
 47                 {
 48                     bool processed = false;
 49                     OnWindowClosed(processed);
 50                     if (processed)
 51                     {
 52                         message.result = 0;
 53                         return true;
 54                     }
 55                     break;
 56                 }
 57                 case WM_COMMAND: case WM_SIZE: case WM_PAINT: case WM_MOUSEMOVE: case WM_MOUSELEAVE: case WM_LBUTTONDOWN: case WM_LBUTTONUP: case WM_CHAR: case WM_KEYDOWN: case WM_KEYUP:
 58                 case WM_SYSCOMMAND:
 59                 {
 60                     return base->ProcessMessage(message);
 61                 }
 62             }
 63             return false;
 64         }
 65         public void Close()
 66         {
 67             PostMessage(Handle(), WM_CLOSE, 0u, 0);
 68         }
 69         protected virtual void OnWindowClosing(bool& cancelClose)
 70         {
 71             windowClosingEvent.Fire(cancelClose);
 72         }
 73         protected virtual void OnWindowClosed(bool& processed)
 74         {
 75             if (windowClosedEvent.Handlers().IsEmpty())
 76             {
 77                 if (isMainWindow)
 78                 {
 79                     Application.Exit();
 80                     processed = true;
 81                 }
 82             }
 83             else
 84             {
 85                 windowClosedEvent.Fire();
 86                 processed = true;
 87             }
 88         }
 89         protected override void OnControlAdded(ControlEventArgs& args)
 90         {
 91             base->OnControlAdded(args);
 92             Control* control = args.control;
 93             if (control is MenuBar*)
 94             {
 95                 menuBar = cast<MenuBar*>(control);
 96             }
 97         }
 98         protected override void OnControlRemoved(ControlEventArgs& args)
 99         {
100             base->OnControlRemoved(args);
101             Control* control = args.control;
102             if (control == menuBar)
103             {
104                 menuBar = null;
105             }
106         }
107         protected override void OnMouseMove(MouseEventArgs& args)
108         {
109             base->OnMouseMove(args);
110             if (menuBar != null)
111             {
112                 menuBar->MouseMoveInternal(args);
113             }
114         }
115         protected override void OnMouseDown(MouseEventArgs& args)
116         {
117             base->OnMouseDown(args);
118             if (menuBar != null)
119             {
120                 menuBar->MouseDownInternal(args);
121             }
122         }
123         protected override void OnMouseUp(MouseEventArgs& args)
124         {
125             base->OnMouseUp(args);
126             if (menuBar != null)
127             {
128                 menuBar->MouseUpInternal(args);
129             }
130         }

141             return state;
142         }
143         public void SetWindowState(WindowState newState)
144         {
145             if (state != newState)
146             {
147                 switch (state)
148                 {
149                     case WindowState.normal:
150                     {
151                         switch (newState)
152                         {
153                             case WindowState.minimized:
154                             {
155                                 ShowWindow(ShowCommand.SW_MINIMIZE);
156                                 break;
157                             }
158                             case WindowState.maximized:
159                             {
160                                 ShowWindow(ShowCommand.SW_MAXIMIZE);
161                                 break;
162                             }
163                         }
164                         break;
165                     }
166                     case WindowState.minimized: case WindowState.maximized:
167                     {
168                         switch (newState)
169                         {
170                             case WindowState.normal:
171                             {
172                                 ShowWindow(ShowCommand.SW_RESTORE);
173                                 break;
174                             }
175                             case WindowState.minimized:
176                             {
177                                 ShowWindow(ShowCommand.SW_MINIMIZE);
178                                 break;
179                             }
180                             case WindowState.maximized:
181                             {
182                                 ShowWindow(ShowCommand.SW_MAXIMIZE);
183                                 break;
184                             }
185                         }
186                         break;
187                     }
188                 }
189                 state = newState;
190             }
191         }
