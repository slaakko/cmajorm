142         protected override bool ProcessMessage(Message& message)
143         {
144             switch (message.msg)
145             {
146                 case WM_CLOSE:
147                 {
148                     bool cancelClose = false;
149                     OnWindowClosing(cancelClose);
150                     if (cancelClose)
151                     {
152                         message.result = 0;
153                         return true;
154                     }
155                     break;
156                 }
157                 case WM_DESTROY:
158                 {
159                     bool processed = false;
160                     OnWindowClosed(processed);
161                     if (processed)
162                     {
163                         message.result = 0;
164                         return true;
165                     }
166                     break;
167                 }
168                 case WM_COMMAND: case WM_MOVE: case WM_SIZE: case WM_PAINT: case WM_MOUSEMOVE: case WM_MOUSELEAVE: case WM_LBUTTONDOWN: case WM_LBUTTONUP: case WM_LBUTTONDBLCLK: case WM_RBUTTONDOWN: case WM_RBUTTONUP:
169                 case WM_CHAR: case WM_KEYDOWN: case WM_KEYUP: case WM_SYSCOMMAND: case WM_HSCROLL: case WM_VSCROLL: case WM_MOUSEWHEEL: case WM_TIMER: case WM_SETFOCUS: case WM_KILLFOCUS:
170                 case WM_SHOWWINDOW: case WM_CLIPBOARDUPDATE:
171                 {
172                     return base->ProcessMessage(message);
173                 }
174             }
175             return false;
176         }

235         public void DefaultKeyPreview(Keys key, bool shift, bool control, bool alt, bool keyDown, bool& handled)
236         {
237             if (keyDown)
238             {
239                 Keys modifierKeys = Keys.none;
240                 if (shift)
241                 {
242                     modifierKeys = cast<Keys>(modifierKeys | Keys.shiftModifier);
243                 }
244                 if (control)
245                 {
246                     modifierKeys = cast<Keys>(modifierKeys | Keys.controlModifier);
247                 }
248                 if (alt)
249                 {
250                     modifierKeys = cast<Keys>(modifierKeys | Keys.altModifier);
251                 }
252                 key = cast<Keys>(key | modifierKeys);
253                 switch (key)
254                 {
255                     case Keys.enter:
256                     {
257                         if (ShowingDialog())
258                         {
259                             if (focusedControl != null)
260                             {
261                                 if (focusedControl is TextBox*)
262                                 {
263                                     TextBox* textBox = cast<TextBox*>(focusedControl);
264                                     if (textBox->IsMultiline())
265                                     {
266                                         break;
267                                     }
268                                 }
269                             }
270                             if (focusedControl != null)
271                             {
272                                 if (focusedControl is Button*)
273                                 {
274                                     Button* button = cast<Button*>(focusedControl);
275                                     button->OnClickInternal();
276                                     handled = true;
277                                 }
278                             }
279                             if (!handled)
280                             {
281                                 if (defaultButton != null && defaultButton->IsEnabled())
282                                 {
283                                     defaultButton->OnClickInternal();
284                                     handled = true;
285                                 }
286                             }
287                         }
288                         break;
289                     }
290                     case Keys.escape:
291                     {
292                         if (ShowingDialog())
293                         {
294                             if (cancelButton != null)
295                             {
296                                 cancelButton->OnClickInternal();
297                                 handled = true;
298                             }
299                         }
300                         break;
301                     }
302                     case Keys.tab:
303                     {
304                         FocusNext();
305                         handled = true;
306                         break;
307                     }
308                     case cast<Keys>(Keys.shiftModifier | Keys.tab):
309                     {
310                         FocusPrev();
311                         handled = true;
312                         break;
313                     }
314                 }
315             }
316         }

525             Size contextMenuSize = contextMenu->GetSize();
526             if (ptInScreenCoords.x + contextMenuSize.w >= screenWindowEdgeLoc.x)

535             contextMenu->SetLocation(loc);

574         public void FocusNext()
575         {
576             LogView* logView = Application.GetLogView();
577             if (focusedControl == null)
578             {
579                 Component* child = Children().FirstChild();
580                 while (child != null)
581                 {
582                     if (child is Control*)
583                     {
584                         Control* control = cast<Control*>(child);
585                         Control* tabStopChild = control->GetFirstEnabledTabStopControl();
586                         if (tabStopChild != null)
587                         {
588                             tabStopChild->SetFocus();
589                             focusedControl = tabStopChild;
590                             if (logView != null)
591                             {
592                                 logView->WriteLine("focused: " + string(typename(*focusedControl)) + ": " + focusedControl->Text());
593                             }
594                             return;
595                         }
596                     }
597                     child = child->NextSibling();
598                 }
599             }
600             else
601             {
602                 Component* parent = null;
603                 Component* child = focusedControl;
604                 Container* container = child->GetContainer();
605                 if (container != null)
606                 {
607                     parent = container->Parent();
608                 }
