 97         public DialogResult ShowDialog(Window& parentWindow)
 98         {
 99             DialogGuard guard(this);
100             Point parentLoc = parentWindow.Location();
101             Size parentSize = parentWindow.GetSize();
102             Size size = GetSize();
103             SetLocation(Point(Max(0, parentLoc.x + (parentSize.w - size.w) / 2), Max(0, parentLoc.y + (parentSize.h - size.h) / 2)));
104             SetDialogResult(DialogResult.none);
105             GetDialogResultFunction dialogResultFunction = System.Windows.GetDialogResult;
106             DialogWindowKeyPreviewFunction keyPreviewFunction = System.Windows.DialogWindowKeyPreview;
107             int result = WinMessageLoop(Handle(), parentWindow.Handle(), cast<void*>(dialogResultFunction), cast<void*>(keyPreviewFunction), this);
108             return cast<DialogResult>(result);
109         }

141         protected override bool ProcessMessage(Message& message)
142         {
143             switch (message.msg)
144             {
145                 case WM_CLOSE:
146                 {
147                     bool cancelClose = false;
148                     OnWindowClosing(cancelClose);
149                     if (cancelClose)
150                     {
151                         message.result = 0;
152                         return true;
153                     }
154                     break;
155                 }
156                 case WM_DESTROY:
157                 {
158                     bool processed = false;
159                     OnWindowClosed(processed);
160                     if (processed)
161                     {
162                         message.result = 0;
163                         return true;
164                     }
165                     break;
166                 }
167                 case WM_COMMAND: case WM_SIZE: case WM_PAINT: case WM_MOUSEMOVE: case WM_MOUSELEAVE: case WM_LBUTTONDOWN: case WM_LBUTTONUP: case WM_LBUTTONDBLCLK: case WM_RBUTTONDOWN: case WM_RBUTTONUP:
168                 case WM_CHAR: case WM_KEYDOWN: case WM_KEYUP: case WM_SYSCOMMAND: case WM_HSCROLL: case WM_VSCROLL: case WM_MOUSEWHEEL: case WM_TIMER: case WM_SETFOCUS: case WM_KILLFOCUS:
169                 case WM_SHOWWINDOW: case WM_CLIPBOARDUPDATE:
170                 {
171                     return base->ProcessMessage(message);
172                 }
173             }
174             return false;
175         }

234         public void DefaultKeyPreview(Keys key, bool keyDown, bool& handled)
235         {
236             switch (key)
237             {
238                 case Keys.controlKey:
239                 {
240                     if (keyDown)
241                     {
242                         currentKeyPreviewModifierKeys = cast<Keys>(currentKeyPreviewModifierKeys | Keys.controlModifier);
243                     }
244                     else
245                     {
246                         currentKeyPreviewModifierKeys = cast<Keys>(currentKeyPreviewModifierKeys & ~Keys.controlModifier);
247                     }
248                     break;
249                 }
250                 case Keys.shiftKey:
251                 {
252                     if (keyDown)
253                     {
254                         currentKeyPreviewModifierKeys = cast<Keys>(currentKeyPreviewModifierKeys | Keys.shiftModifier);
255                     }
256                     else
257                     {
258                         currentKeyPreviewModifierKeys = cast<Keys>(currentKeyPreviewModifierKeys & ~Keys.shiftModifier);
259                     }
260                     break;
261                 }
262             }
263             if (keyDown)
264             {
265                 key = cast<Keys>(key | currentKeyPreviewModifierKeys);
266                 switch (key)
267                 {
268                     case Keys.enter:
269                     {
270                         if (ShowingDialog())
271                         {
272                             if (focusedControl != null)
273                             {
274                                 if (focusedControl is TextBox*)
275                                 {
276                                     TextBox* textBox = cast<TextBox*>(focusedControl);
277                                     if (textBox->IsMultiline())
278                                     {
279                                         break;
280                                     }
281                                 }
282                             }
283                             if (focusedControl != null)
284                             {
285                                 if (focusedControl is Button*)
286                                 {
287                                     Button* button = cast<Button*>(focusedControl);
288                                     button->OnClickInternal();
289                                     handled = true;
290                                 }
291                             }
292                             if (!handled)
293                             {
294                                 if (defaultButton != null && defaultButton->IsEnabled())
295                                 {
296                                     defaultButton->OnClickInternal();
297                                     handled = true;
298                                 }
299                             }
300                         }
301                         break;
302                     }
303                     case Keys.escape:
304                     {
305                         if (ShowingDialog())
306                         {
307                             if (cancelButton != null)
308                             {
309                                 cancelButton->OnClickInternal();
310                                 handled = true;
311                             }
312                         }
313                         break;
314                     }
315                     case Keys.tab:
316                     {
317                         FocusNext();
318                         handled = true;
319                         break;
320                     }
321                     case cast<Keys>(Keys.shiftModifier | Keys.tab):
322                     {
323                         FocusPrev();
324                         handled = true;
325                         break;
326                     }
327                 }
328             }
329         }

574         public void FocusNext()
575         {
576             LogView* logView = Application.GetLogView();
577             if (focusedControl == null)
578             {
579                 Component* child = Children().FirstChild();
580                 while (child != null)
581                 {
582                     if (child is Control*)
583                     {
584                         Control* control = cast<Control*>(child);
585                         Control* tabStopChild = control->GetFirstEnabledTabStopControl();
586                         if (tabStopChild != null)
587                         {
588                             tabStopChild->SetFocus();
589                             focusedControl = tabStopChild;
590                             if (logView != null)
591                             {
592                                 logView->WriteLine("focused: " + string(typename(*focusedControl)) + ": " + focusedControl->Text());
593                             }
594                             return;
595                         }
596                     }
597                     child = child->NextSibling();
598                 }
599             }
600             else
601             {
602                 Component* parent = null;
603                 Component* child = focusedControl;
604                 Container* container = child->GetContainer();
605                 if (container != null)
606                 {
607                     parent = container->Parent();
608                 }
609                 while (child != null)
610                 {
611                     child = child->NextSibling();
612                     while (child != null)
613                     {
614                         if (child is Control*)
615                         {
616                             Control* control = cast<Control*>(child);
617                             Control* tabStopChild = control->GetFirstEnabledTabStopControl();
618                             if (tabStopChild != null)
619                             {
620                                 tabStopChild->SetFocus();
621                                 focusedControl = tabStopChild;
622                                 return;
623                             }

645         public void FocusPrev()
646         {
647             LogView* logView = Application.GetLogView();
648             if (focusedControl == null)
649             {
650                 Component* child = Children().LastChild();

652                 {
653                     if (child is Control*)
654                     {
655                         Control* control = cast<Control*>(child);

657                         if (tabStopChild != null)
658                         {
659                             tabStopChild->SetFocus();
660                             focusedControl = tabStopChild;
661                             if (logView != null)
662                             {
663                                 logView->WriteLine("focused: " + string(typename(*focusedControl)) + ": " + focusedControl->Text());
664                             }
665                             return;
666                         }
667                     }
668                     child = child->PrevSibling();
669                 }
670             }
671             else
672             {
673                 Component* parent = null;
674                 Component* child = focusedControl;
675                 Container* container = child->GetContainer();
676                 if (container != null)
677                 {
678                     parent = container->Parent();
679                 }
680                 while (child != null)
681                 {
682                     child = child->PrevSibling();
683                     while (child != null)
684                     {
685                         if (child is Control*)
686                         {
687                             Control* control = cast<Control*>(child);
688                             Control* tabStopChild = control->GetLastEnabledTabStopControl();
689                             if (tabStopChild != null)
690                             {
691                                 tabStopChild->SetFocus();
692                                 focusedControl = tabStopChild;
693                                 return;
694                             }
695                         }
696                         child = child->PrevSibling();
697                     }
698                     child = parent;
699                     if (parent != null)
700                     {
701                         container = parent->GetContainer();
702                         if (container != null)
703                         {
704                             parent = container->Parent();
705                         }
706                         else
707                         {
708                             parent = null;
709                         }
710                     }
711                 }
712                 focusedControl = null;
713                 FocusPrev();
714             }
