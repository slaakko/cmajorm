  6 using System;
  7 
  8 namespace System.Windows
  9 {
 10     public class GroupBox : ButtonBase
 11     {
 12         public GroupBox(const Color& backgroundColor, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 13             base("BUTTON", DefaultWindowClassStyle(), cast<WindowStyle>(DefaultChildWindowStyle() | ButtonStyle.BS_GROUPBOX), DefaultExtendedWindowStyle(), 
 14             backgroundColor, text, location, size, dock, anchors), children(this), checkedRadioButton(null)
 15         {
 16         }
 17         public GroupBox(const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : this(DefaultControlBackgroundColor(), text, location, size, dock, anchors)
 18         {
 19         }
 20         public void AddChild(Control* child)
 21         {
 22             children.AddChild(child);
 23             ControlEventArgs args(child);
 24             OnControlAdded(args);
 25         }
 26         public UniquePtr<Control> RemoveChild(Control* child)
 27         {
 28             UniquePtr<Component> childComponent = children.RemoveChild(child);
 29             ControlEventArgs args(child);
 30             OnControlRemoved(args);
 31             return UniquePtr<Control>(childComponent.Release() as Control*);
 32         }
 33         public void InsertChildBefore(Control* child, Control* before)
 34         {
 35             children.InsertBefore(child, before);
 36             ControlEventArgs args(child);
 37             OnControlAdded(args);
 38         }
 39         public void InsertChildAfter(Control* child, Control* after)
 40         {
 41             children.InsertAfter(child, after);
 42             ControlEventArgs args(child);
 43             OnControlAdded(args);
 44         }
 45         public nothrow const Container& Children() const
 46         {
 47             return children;
 48         }
 49         public inline nothrow RadioButton* CheckedRadioButton() const
 50         {
 51             return checkedRadioButton;
 52         }
 53         public void SetCheckedRadioButton(RadioButton* radioButton)
 54         {
 55             if (radioButton != checkedRadioButton)
 56             {
 57                 if (checkedRadioButton != null)
 58                 {
 59                     checkedRadioButton->SetChecked(false);
 60                 }
 61                 checkedRadioButton = radioButton;
 62                 if (checkedRadioButton != null)
 63                 {
 64                     checkedRadioButton->SetChecked(true);
 65                 }
 66             }
 67         }
 68         internal inline nothrow void ResetCheckedRadioButton()
 69         {
 70             checkedRadioButton = null;
 71         }
 72         protected override void OnCreated()
 73         {
 74             base->OnCreated();
 75             Graphics graphics = Graphics.FromWindowHandle(Handle());
 76             const FontHandle& fontHandle = GetFontHandle(graphics);
 77             if (!fontHandle.IsNull())
 78             {
 79                 SendSetFontMessage(fontHandle);
 80             }
 81             SubClassCommandWndProc();
 82         }
 83         protected override void OnControlAdded(ControlEventArgs& args)
 84         {
 85             base->OnControlAdded(args);
 86             if (args.control is RadioButton*)
 87             {
 88                 if (checkedRadioButton == null)
 89                 {
 90                     SetCheckedRadioButton(cast<RadioButton*>(args.control));
 91                 }
 92             }
 93         }
 94         protected override void OnControlRemoved(ControlEventArgs& args)
 95         {
 96             base->OnControlRemoved(args);
 97             if (args.control is RadioButton*)
 98             {
 99                 if (checkedRadioButton == cast<RadioButton*>(args.control))
100                 {
101                     ResetCheckedRadioButton();
102                 }
103             }
104         }
105         protected override void OnChildGotFocus(ControlEventArgs& args)
106         {
107             base->OnChildGotFocus(args);
108             Control* parentControl = ParentControl();
109             if (parentControl != null)
110             {
111                 parentControl->OnChildGotFocus(args);
112             }
113         }
114         protected override void OnChildLostFocus(ControlEventArgs& args)
115         {
116             base->OnChildLostFocus(args);
117             Control* parentControl = ParentControl();
118             if (parentControl != null)
119             {
120                 parentControl->OnChildLostFocus(args);
121             }
122         }
123         internal override nothrow Control* GetFirstEnabledTabStopControl() const
124         {
125             Component* child = children.FirstChild();
126             while (child != null)
127             {
128                 if (child is Control*)
129                 {
130                     Control* control = cast<Control*>(child);
131                     Control* tabStopChild = control->GetFirstEnabledTabStopControl();
132                     if (tabStopChild != null)
133                     {
134                         return tabStopChild;
135                     }
136                 }
137                 child = child->NextSibling();
138             }
139             return null;
140         }
141         internal override nothrow Control* GetLastEnabledTabStopControl() const
142         {
143             Component* child = children.LastChild();
144             while (child != null)
145             {
146                 if (child is Control*)
147                 {
148                     Control* control = cast<Control*>(child);
149                     Control* tabStopChild = control->GetLastEnabledTabStopControl();
150                     if (tabStopChild != null)
151                     {
152                         return tabStopChild;
153                     }
154                 }
155                 child = child->PrevSibling();
156             }
157             return null;
158         }
159         private Container children;
160         private RadioButton* checkedRadioButton;
161     }
162 }
