  6 using System;
  7 
  8 namespace System.Windows
  9 {
 10     public class GroupBox : ButtonBase
 11     {
 12         public GroupBox(const Color& backgroundColor, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 13             base("BUTTON", DefaultWindowClassStyle(), cast<WindowStyle>(DefaultChildWindowStyle() | ButtonStyle.BS_GROUPBOX), DefaultExtendedWindowStyle(), 
 14             backgroundColor, text, location, size, dock, anchors), children(this), checkedRadioButton(null)
 15         {
 16         }
 17         public GroupBox(const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : this(DefaultControlBackgroundColor(), text, location, size, dock, anchors)
 18         {
 19         }
 20         public void AddChild(Control* child)
 21         {
 22             children.AddChild(child);
 23             ControlEventArgs args(child);
 24             OnControlAdded(args);
 25         }
 26         public UniquePtr<Control> RemoveChild(Control* child)
 27         {
 28             UniquePtr<Component> childComponent = children.RemoveChild(child);
 29             ControlEventArgs args(child);
 30             OnControlRemoved(args);
 31             return UniquePtr<Control>(childComponent.Release() as Control*);
 32         }
 33         public void InsertChildAfter(Control* child, Control* after)
 34         {
 35             children.InsertAfter(child, after);
 36             ControlEventArgs args(child);
 37             OnControlAdded(args);
 38         }
 39         public nothrow const Container& Children() const
 40         {
 41             return children;
 42         }
 43         public inline nothrow RadioButton* CheckedRadioButton() const
 44         {
 45             return checkedRadioButton;
 46         }
 47         public void SetCheckedRadioButton(RadioButton* radioButton)
 48         {
 49             if (radioButton != checkedRadioButton)
 50             {
 51                 if (checkedRadioButton != null)
 52                 {
 53                     checkedRadioButton->SetChecked(false);
 54                 }
 55                 checkedRadioButton = radioButton;
 56                 if (checkedRadioButton != null)
 57                 {
 58                     checkedRadioButton->SetChecked(true);
 59                 }
 60             }
 61         }
 62         internal inline nothrow void ResetCheckedRadioButton()
 63         {
 64             checkedRadioButton = null;
 65         }
 66         protected override void OnCreated()
 67         {
 68             base->OnCreated();
 69             Graphics graphics = Graphics.FromWindowHandle(Handle());
 70             const FontHandle& fontHandle = GetFontHandle(graphics);
 71             if (!fontHandle.IsNull())
 72             {
 73                 SendSetFontMessage(fontHandle);
 74             }
 75             SubClassCommandWndProc();
 76         }
 77         protected override void OnControlAdded(ControlEventArgs& args)
 78         {
 79             base->OnControlAdded(args);
 80             if (args.control is RadioButton*)
 81             {
 82                 if (checkedRadioButton == null)
 83                 {
 84                     SetCheckedRadioButton(cast<RadioButton*>(args.control));
 85                 }
 86             }
 87         }
 88         protected override void OnControlRemoved(ControlEventArgs& args)
 89         {
 90             base->OnControlRemoved(args);
 91             if (args.control is RadioButton*)
 92             {
 93                 if (checkedRadioButton == cast<RadioButton*>(args.control))
 94                 {
 95                     ResetCheckedRadioButton();
 96                 }
 97             }
 98         }
 99         protected override void OnChildGotFocus(ControlEventArgs& args)
100         {
101             base->OnChildGotFocus(args);
102             Control* parentControl = ParentControl();
103             if (parentControl != null)
104             {
105                 parentControl->OnChildGotFocus(args);
106             }
107         }
108         protected override void OnChildLostFocus(ControlEventArgs& args)
109         {
110             base->OnChildLostFocus(args);
111             Control* parentControl = ParentControl();
112             if (parentControl != null)
113             {
114                 parentControl->OnChildLostFocus(args);
115             }
116         }
117         internal override nothrow Control* GetFirstEnabledTabStopControl() const
118         {
119             Component* child = children.FirstChild();
120             while (child != null)
121             {
122                 if (child is Control*)
123                 {
124                     Control* control = cast<Control*>(child);
125                     Control* tabStopChild = control->GetFirstEnabledTabStopControl();
126                     if (tabStopChild != null)
127                     {
128                         return tabStopChild;
129                     }
130                 }
131                 child = child->NextSibling();
132             }
133             return null;
134         }
135         internal override nothrow Control* GetLastEnabledTabStopControl() const
136         {
137             Component* child = children.LastChild();
138             while (child != null)
139             {
140                 if (child is Control*)
141                 {
142                     Control* control = cast<Control*>(child);
143                     Control* tabStopChild = control->GetLastEnabledTabStopControl();
144                     if (tabStopChild != null)
145                     {
146                         return tabStopChild;
147                     }
148                 }
149                 child = child->PrevSibling();
150             }
151             return null;
152         }
153         private Container children;
154         private RadioButton* checkedRadioButton;
155     }
156 }
