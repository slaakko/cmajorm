 11     public string CmajorRootDir()
 12     {
 13         string cmajorRootDir;
 14         const char* cmajorRootEnv = RtGetEnvironmentVariable("CMAJOR_ROOT");
 15         if (cmajorRootEnv != null && *cmajorRootEnv != '\0')
 16         {
 17             cmajorRootDir = cmajorRootEnv;
 18         }
 19         if (cmajorRootDir.IsEmpty())
 20         {
 21             throw Exception("please set 'CMAJOR_ROOT' environment variable to contain /path/to/cmajor directory.");
 22         }
 23         return cmajorRootDir;
 24     }

 53     public static class Application
 54     {
 55         static Application() : logView(null), activeWindow(null), runningModal(false)
 56         {
 57             MessageProcessorFunction messageProcessorFunction = ProcessMessage;
 58             ModelessWindowKeyPreviewFunction keyPreviewFunction = ModelessWindowKeyPreview;
 59             void* messageProcessorFunctionAddress = cast<void*>(messageProcessorFunction);
 60             void* keyPreviewFunctionAddress = cast<void*>(keyPreviewFunction);
 61             int result = WinInit(messageProcessorFunctionAddress, keyPreviewFunctionAddress);
 62             CheckGraphicsStatus(cast<GraphicsStatus>(result));
 63         }

 67             {

 69             }

 71         public static int Run(Window& mainWindow)
 72         {
 73             mainWindow.SetAsMainWindow();
 74             SetMainWindow(&mainWindow);
 75             switch (mainWindow.GetWindowState())
 76             {
 77                 case WindowState.normal:
 78                 {
 79                     mainWindow.ShowWindow(ShowCommand.SW_SHOWNORMAL);
 80                     break;
 81                 }
 82                 case WindowState.minimized:
 83                 {
 84                     mainWindow.ShowWindow(ShowCommand.SW_MINIMIZE);
 85                     break;
 86                 }
 87                 case WindowState.maximized:
 88                 {
 89                     mainWindow.ShowWindow(ShowCommand.SW_MAXIMIZE);
 90                     break;
 91                 }
 92             }
 93             mainWindow.Show();
 94             mainWindow.Update();
 95             int exitCode = WinRun();
 96             return exitCode;
 97         }

119         private static bool ProcessMessage(void* windowHandle, uint msg, uint wparam, long lparam, long& result, void*& originalWndProc)
120         {
121             Control* window = windowManager.GetWindow(windowHandle);
122             if (window != null)
123             {
124                 if (Debug.Messages())
125                 {
126                     System.IO.LogWriter writer(ApplicationLogFilePath());
127                     writer << GetCurrentTimestamp().ToString() << endl();
128                     writer << ">" << MessageNameMap.Instance().GetMessageName(msg) << endl();
129                 }
130                 Message message(windowHandle, msg, wparam, lparam, result);
131                 bool handled = window->ProcessMessageInternal(message);
132                 if (message.originalWndProc != null)
133                 {
134                     originalWndProc = message.originalWndProc;
135                 }
136                 if (handled)
137                 {
138                     result = message.result;
139                 }
140                 if (Debug.Messages())
141                 {
142                     System.IO.LogWriter writer(ApplicationLogFilePath());
143                     writer << "<" << MessageNameMap.Instance().GetMessageName(msg) << endl();
144                 }
145                 return handled;
146             }
147             return false;
148         }
