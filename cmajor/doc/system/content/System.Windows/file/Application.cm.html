 11     public delegate bool MessageProcessorFunction(void* windowHandle, uint msg, uint wparam, long lparam, long& result, void*& originalWndProc);
 12 
 13     public static class Application
 14     {
 15         static Application() : logView(null), activeWindow(null)
 16         {
 17             MessageProcessorFunction messageProcessorFunction = ProcessMessage;
 18             ModelessWindowKeyPreviewFunction keyPreviewFunction = ModelessWindowKeyPreview;
 19             void* messageProcessorFunctionAddress = cast<void*>(messageProcessorFunction);
 20             void* keyPreviewFunctionAddress = cast<void*>(keyPreviewFunction);
 21             int result = WinInit(messageProcessorFunctionAddress, keyPreviewFunctionAddress);
 22             CheckGraphicsStatus(cast<GraphicsStatus>(result));
 23         }
 24         public static void Init()
 25         {
 26         }
 27         public static int Run(Window& mainWindow)
 28         {
 29             mainWindow.SetAsMainWindow();
 30             switch (mainWindow.GetWindowState())
 31             {
 32                 case WindowState.normal:
 33                 {
 34                     mainWindow.ShowWindow(ShowCommand.SW_SHOWNORMAL);
 35                     break;
 36                 }
 37                 case WindowState.minimized:
 38                 {
 39                     mainWindow.ShowWindow(ShowCommand.SW_MINIMIZE);
 40                     break;
 41                 }
 42                 case WindowState.maximized:
 43                 {
 44                     mainWindow.ShowWindow(ShowCommand.SW_MAXIMIZE);
 45                     break;
 46                 }
 47             }
 48             mainWindow.Show();
 49             mainWindow.Update();
 50             int exitCode = WinRun();
 51             return exitCode;
 52         }
 53         public static void Exit(int exitCode)

 56         }
 57         public static void Exit()
 58         {
 59             Exit(0);
 60         }
 61         private static bool ProcessMessage(void* windowHandle, uint msg, uint wparam, long lparam, long& result, void*& originalWndProc)
 62         {
 63             Control* window = windowManager.GetWindow(windowHandle);
 64             if (window != null)
 65             {
 66                 Message message(windowHandle, msg, wparam, lparam, result);
 67                 bool handled = window->ProcessMessageInternal(message);
 68                 if (message.originalWndProc != null)
 69                 {
 70                     originalWndProc = message.originalWndProc;
 71                 }
 72                 if (handled)
 73                 {
 74                     result = message.result;
 75                 }
 76                 return handled;
 77             }
 78             return false;
 79         }
