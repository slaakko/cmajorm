19         public nothrow void AddHandler(Handler handler)
20         {
21             if (Find(handlers.CBegin(), handlers.CEnd(), handler) == handlers.CEnd())

25         }

33         }
34         public void Fire() where Args is EmptyEventArgs
35         {
36             for (Handler handler : handlers)

38                 handler();
39             }
40         }
