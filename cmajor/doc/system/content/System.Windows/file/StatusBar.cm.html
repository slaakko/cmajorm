252         private void Measure(Graphics& graphics)
253         {
254             LogView* logView = Application.GetLogView();
255             Padding padding = DefaultPadding();
256             int statusBarItemBorderWidth = StatusBarItem.BorderWidth();
257             int statusBarItemHorizontalPadding = StatusBarItem.HorizontalPadding();
258             int textHeight = 0;
259             long n = statusBarItems.Count();
260             for (long i = 0; i < n; ++i;)
261             {
262                 StatusBarItem* item = statusBarItems[i].Get();
263                 string s = item->Text();
264                 ustring u(ToUtf32(s));
265                 string z('0', u.Length());
266                 RectF rect = graphics.MeasureStringChecked(z, font, PointF(0, 0), stringFormat);
267                 textHeight = Max(textHeight, cast<int>(rect.size.h));
268                 item->SetTextWidth(cast<int>(rect.size.w));
269             }
270             if (textHeight != 0)
271             {
272                 int height = topLineWidth + padding.Vertical() + 2 * statusBarItemBorderWidth + textHeight;
273                 Size size = GetSize();
274                 size.h = height;
275                 SetSize(size);
276                 DockWindow();
277                 Point leftOrigin(padding.left, topLineWidth + padding.top);
278                 long springIndex = -1;
279                 for (long i = 0; i < n; ++i;)
280                 {
281                     StatusBarItem* item = statusBarItems[i].Get();
282                     if (item->Spring())
283                     {
284                         springIndex = i;
285                         break;
286                     }
287                     item->SetLocation(leftOrigin);
288                     item->SetSize(Size(2 * statusBarItemBorderWidth + statusBarItemHorizontalPadding + item->TextWidth(), 2 * statusBarItemBorderWidth + textHeight));
289                     leftOrigin.x = leftOrigin.x + item->GetSize().w;
290                 }
291                 Point rightOrigin(size.w - 1 - padding.right, topLineWidth + padding.top);
292                 if (springIndex != -1)
293                 {
294                     for (long i = n - 1; i > springIndex; --i;)
295                     {
296                         StatusBarItem* item = statusBarItems[i].Get();
297                         if (item->Spring())
298                         {
299                             break;
300                         }
301                         item->SetSize(Size(2 * statusBarItemBorderWidth + statusBarItemHorizontalPadding + item->TextWidth(), 2 * statusBarItemBorderWidth + textHeight));
302                         rightOrigin.x = rightOrigin.x - item->GetSize().w;
303                         item->SetLocation(rightOrigin);
304                     }
305                     StatusBarItem* springItem = statusBarItems[springIndex].Get();
306                     springItem->SetLocation(leftOrigin);
307                     springItem->SetSize(Size(rightOrigin.x - leftOrigin.x + 1, 2 * statusBarItemBorderWidth + textHeight));
308                 }
309             }
310         }
