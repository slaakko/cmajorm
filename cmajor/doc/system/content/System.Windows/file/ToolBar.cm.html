 367         internal void ShowToolTipWindow(ToolButton* toolButton)
 368         {
 369             if (toolTipWindow == null || !ToolTipWindowAdded()) return;
 370             if (ToolTipWindowVisible()) toolTipWindow->Hide();
 371             ResetToolTipWindowVisible();
 372             Point loc = toolButton->Location();
 373             Size size = toolButton->GetSize();
 374             toolTipWindow->SetText(toolButton->GetToolTip());
 375             toolTipWindow->MeasureExtent();
 376             Point pt(loc.x, loc.y + size.h + 8);
 377             pt = ClientToScreen(pt);
 378             Window* window = GetWindow();
 379             if (window != null)
 380             {
 381                 pt = window->ScreenToClient(pt);
 382                 Size sz = toolTipWindow->GetSize();
 383                 Point wloc = window->Location();
 384                 Size wsz = window->GetSize();
 385                 if (pt.y + sz.h > wloc.y + wsz.h)
 386                 {
 387                     pt.y = wloc.y + wsz.h - sz.h;
 388                 }
 389                 if (pt.x + sz.w > wloc.x + wsz.w)
 390                 {
 391                     pt.x = wloc.x + wsz.w - sz.w;
 392                 }
 393                 toolTipWindow->SetLocation(pt);
 394                 toolTipWindow->BringToFront();
 395                 toolTipWindow->Show();
 396                 toolTipWindow->Invalidate();
 397                 SetToolTipWindowVisible();
 398             }

 427         private Pair<Bitmap*, Bitmap*> GetOrCreateBitmaps(const string& bitmapName)
 428         {
 429             Pair<Bitmap*, Bitmap*> bitmapPair;
 430             if (bitmapName.IsEmpty()) return bitmapPair;
 431             try
 432             {
 433                 HashMap<string, Pair<Bitmap*, Bitmap*>>.ConstIterator it = bitmapMap.CFind(bitmapName);
 434                 if (it != bitmapMap.CEnd())
 435                 {
 436                     return it->second;
 437                 }
 438                 UniquePtr<Bitmap> bitmapPtr(new Bitmap(Bitmap.FromResource(bitmapName)));
 439                 Bitmap* bitmap = bitmapPtr.Get();
 440                 UniquePtr<Bitmap> disabledBitmapPtr(new Bitmap(bitmap->ToGrayBitmap()));
 441                 Bitmap* disabledBitmap = disabledBitmapPtr.Get();
 442                 bitmapPair = MakePair(bitmap, disabledBitmap);
 443                 bitmapMap[bitmapName] = bitmapPair;
 444                 bitmaps.Add(Rvalue(bitmapPtr));
 445                 bitmaps.Add(Rvalue(disabledBitmapPtr));
 446                 return bitmapPair;
 447             }
 448             catch (const Exception& ex)
 449             {
 450                 throw Exception("could not load bitmap '" + bitmapName + "': " + ex.Message());
 451             }
 452             return bitmapPair;
 453         }
 454         private nothrow ToolButton* GetToolButtonAt(const Point& location) const
 455         {
 456             Size size = GetSize();
 457             if (location.y >= 0 && location.y < size.h && location.x >= 0 && location.x < size.w)
 458             {
 459                 Component* component = toolButtons.FirstChild();
 460                 while (component != null)
 461                 {
 462                     if (component is ToolButton*)
 463                     {
 464                         ToolButton* toolButton = cast<ToolButton*>(component);
 465                         if (location.x >= toolButton->Location().x && location.x <= toolButton->Location().x + toolButton->GetSize().w)
 466                         {
 467                             return toolButton;
 468                         }
 469                     }
 470                     component = component->NextSibling();
 471                 }
 472             }
 473             return null;
 474         }
