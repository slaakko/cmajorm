257         internal void ShowToolTipWindow(ToolButton* toolButton)
258         {
259             toolTipWindow->Hide();
260             Point loc = toolButton->Location();
261             Size size = toolButton->GetSize();
262             toolTipWindow->SetText(toolButton->GetToolTip());
263             toolTipWindow->MeasureExtent();
264             Point pt(loc.x + size.w + toolButton->GetPadding().left, loc.y + (size.h - toolTipWindow->GetSize().h) / 2);
265             pt = ClientToScreen(pt);
266             Window* window = GetWindow();
267             if (window != null)
268             {
269                 pt = window->ScreenToClient(pt);
270                 Size sz = toolTipWindow->GetSize();
271                 Point wloc = window->Location();
272                 Size wsz = window->GetSize();
273                 if (pt.y + sz.h > wloc.y + wsz.h)
274                 {
275                     pt.y = wloc.y + wsz.h - sz.h;
276                 }
277                 if (pt.x + sz.w > wloc.x + wsz.w)
278                 {
279                     pt.x = wloc.x + wsz.w - sz.w;
280                 }
281                 toolTipWindow->SetLocation(pt);
282                 toolTipWindow->BringToFront();
283                 toolTipWindow->Show();
284                 toolTipWindow->Invalidate();
285                 toolTipWindow->Update();
286             }

297         private Bitmap* GetOrCreateBitmap(const string& bitmapName)
298         {
299             Bitmap* bitmap = null;
300             try
301             {
302                 HashMap<string, Bitmap*>.ConstIterator it = bitmapMap.CFind(bitmapName);
303                 if (it != bitmapMap.CEnd())
304                 {
305                     return it->second;
306                 }
307                 UniquePtr<Bitmap> bitmapPtr(new Bitmap(Bitmap.FromResource(bitmapName)));
308                 bitmap = bitmapPtr.Get();
309                 bitmapMap[bitmapName] = bitmap;
310                 bitmaps.Add(Rvalue(bitmapPtr));
311                 return bitmap;
312             }
313             catch (const Exception& ex)
314             {
315                 throw Exception("could not load bitmap '" + bitmapName + "': " + ex.Message());
316             }
317             return bitmap;
318         }
