 29     public string GetWindowsAPIErrorMessage(ulong errorCode)
 30     {
 31         char[4096] buffer;
 32         OsFormatMessage(errorCode, &buffer[0]);
 33         return string(&buffer[0]);
 34     }

127     public int WaitForMultipleObjects(const List<void*>& handles)
128     {
129         int result = WinWaitForMultipleObjects(cast<uint>(handles.Count()), handles.Begin().Ptr());
130         if (result >= 0 && result <= handles.Count())
131         {
132             return result;
133         }
134         else
135         {
136             throw WindowsAPIException(WinGetLastError());
137         }
138     }

140     public void ShellExecute(const string& filePath)
141     {
142         long errorCode = 0;
143         bool succeeded = WinShellExecute(filePath.Chars(), errorCode);
144         if (!succeeded)
145         {
146             string errorMessage = "shell execution error for file '" + filePath + "': ";
147             string error = "out of resources";
148             switch (errorCode)
149             {
150                 case ERROR_DUPLICATE_CODE_MAPPING:
151                 {
152                     error = "duplicate code mapping detected"; break;
153                 }
154                 case ERROR_CODE_NOT_FOUND:
155                 {
156                     error = "code not found"; break;
157                 }
158                 case ERROR_OUT_OF_RESOURCES:
159                 {
160                     error = "out of resources"; break;
161                 }
162                 case ERROR_FILE_NOT_FOUND:
163                 {
164                     error = "file not found"; break;
165                 }
166                 case ERROR_PATH_NOT_FOUND:
167                 {
168                     error = "path not found"; break;
169                 }
170                 case ERROR_BAD_FORMAT:
171                 {
172                     error = "invalid executable"; break;
173                 }
174                 case SE_ERR_ACCESSDENIED:
175                 {
176                     error = "access denied"; break;
177                 }
178                 case SE_ERR_ASSOCINCOMPLETE:
179                 {
180                     error = "invalid file association"; break;
181                 }
182                 case SE_ERR_DDEBUSY:
183                 {
184                     error = "DDE busy"; break;
185                 }
186                 case SE_ERR_DDEFAIL:
187                 {
188                     error = "DDE failed"; break;
189                 }
190                 case SE_ERR_DDETIMEOUT:
191                 {
192                     error = "DDE timeout"; break;
193                 }
194                 case SE_ERR_DLLNOTFOUND:
195                 {
196                     error = "DLL not found"; break;
197                 }
198                 case SE_ERR_FNF:
199                 {
200                     error = "file not found"; break;
201                 }
202                 case SE_ERR_NOASSOC:
203                 {
204                     error = "no association for this type of file"; break;
205                 }
206                 case SE_ERR_OOM:
207                 {
208                     error = "not enough memory"; break;
209                 }
210                 case SE_ERR_PNF:
211                 {
212                     error = "path not found"; break;
213                 }
214                 case SE_ERR_SHARE:
215                 {
216                     error = "sharing violation"; break;
217                 }
218             }
219             errorMessage.Append(error);
220             throw Exception(errorMessage);
221         }
222     }
