600     public bool GetOpenFileName(void* windowHandle, const List<Pair<string, string>>& descriptionFilterPairs, const string& initialDirectory, const string& defaultFilePath, 
601         const string& defaultExtension, uint flags, string& currentDirectory, List<string>& fileNames)
602     {
603         long filterBufferSize = 0;
604         for (const Pair<string, string>& descriptionFilterPair : descriptionFilterPairs)
605         {
606             wstring description = ToUtf16(descriptionFilterPair.first);
607             wstring filter = ToUtf16(descriptionFilterPair.second);
608             filterBufferSize = filterBufferSize + description.Length() + 1 + filter.Length() + 1;
609         }
610         filterBufferSize = filterBufferSize + 1 + 1;
611         UniquePtr<wchar> filterBuffer = cast<wchar*>(MemAlloc(sizeof(wchar) * filterBufferSize));
612         wchar* filterBufferPtr = filterBuffer.Get();
613         for (const Pair<string, string>& descriptionFilterPair : descriptionFilterPairs)
614         {
615             wstring description = ToUtf16(descriptionFilterPair.first);
616             long descLen = StrCopy(filterBufferPtr, description.Chars(), description.Length());
617             filterBufferPtr = filterBufferPtr + descLen + 1;
618             wstring filter = ToUtf16(descriptionFilterPair.second);
619             long filterLen = StrCopy(filterBufferPtr, filter.Chars(), filter.Length());
620             filterBufferPtr = filterBufferPtr + filterLen + 1;
621         }
622         *filterBufferPtr++ = '\0';
623         *filterBufferPtr++ = '\0';
624         long fileNameBufferSize = 16 * 1024;
625         UniquePtr<wchar> fileNameBuffer = cast<wchar*>(MemAlloc(sizeof(wchar) * fileNameBufferSize));
626         wstring defFilePath = ToUtf16(defaultFilePath);
627         if (defaultFilePath.IsEmpty())
628         {
629             *fileNameBuffer.Get() = '\0';
630         }
631         else
632         {
633             StrCopy(fileNameBuffer.Get(), defFilePath.Chars(), defFilePath.Length());
634         }
635         wstring initialDir = ToUtf16(initialDirectory);
636         wstring defExt = ToUtf16(defaultExtension);
637         bool success = WinGetOpenFileName(windowHandle, filterBuffer.Get(), initialDir.Chars(), fileNameBuffer.Get(), cast<uint>(fileNameBufferSize), flags, defExt.Chars());
638         if (success)
639         {
640             long firstPathLength = StrLen(fileNameBuffer.Get());
641             string firstPath = ToUtf8(wstring(fileNameBuffer.Get(), firstPathLength));
642             if ((flags & OFN_ALLOWMULTISELECT) == 0)
643             {
644                 currentDirectory = Path.GetDirectoryName(firstPath);
645                 fileNames.Add(Path.GetFileName(firstPath));
646             }
647             else
648             {
649                 currentDirectory = firstPath;
650                 long pathStart = firstPathLength + 1;
651                 wchar* wpath = fileNameBuffer.Get() + pathStart;
652                 while (*wpath != '\0')
653                 {
654                     long pathLength = StrLen(wpath);
655                     string path = ToUtf8(wstring(wpath, pathLength));
656                     fileNames.Add(path);
657                     wpath = wpath + pathLength + 1;
658                 }
659             }
660             return true;
661         }
662         else
663         {
664             uint errorCode = WinCommDlgExtendedError();
665             if (errorCode == 0u)
666             {
667                 return false;
668             }
669             else
670             {
671                 ThrowCommonDialogError(errorCode);
672             }
673         }
674         return false;
675     }
