 11     public enum ButtonStyle : long
 12     {
 13         BS_PUSHBUTTON = 0x0, 
 14         BS_DEFPUSHBUTTON = 0x1, 
 15         BS_3STATE = 0x5, 
 16         BS_AUTO3STATE = 0x6, 
 17         BS_AUTOCHECKBOX = 0x3, 
 18         BS_AUTORADIOBUTTON = 0x9, 
 19         BS_BITMAP = 0x80, 
 20         BS_BOTTOM = 0x800, 
 21         BS_CENTER = 0x300, 
 22         BS_CHECKBOX = 0x2, 
 23         BS_GROUPBOX = 0x7, 
 24         BS_ICON = 0x40, 
 25         BS_FLAT = 0x8000, 
 26         BS_LEFT = 0x100, 
 27         BS_LEFTTEXT = 0x20, 
 28         BS_MULTILINE = 0x2000, 
 29         BS_NOTIFY = 0x4000, 
 30         BS_OWNERDRAW = 0x0B, 
 31         BS_PUSHLIKE = 0x1000, 
 32         BS_RADIOBUTTON = 0x4, 
 33         BS_RIGHT = 0x200, 
 34         BS_TOP = 0x400, 
 35         BS_TYPEMASK = 0xF, 
 36         BS_VCENTER = 0xC00
 37     }

 39     public abstract class ButtonBase : Control
 40     {
 41         public ButtonBase(const char* windowClassName, WindowClassStyle windowClassStyle, WindowStyle style, ExtendedWindowStyle exStyle, 
 42             const Color& backgroundColor, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 43             base(windowClassName, windowClassStyle, style, exStyle, backgroundColor, text, location, size, dock, anchors)
 44         {
 45         }
 46         internal void OnClickInternal()
 47         {
 48             OnClick();
 49         }
 50         public void DoClick()
 51         {
 52             OnClick();
 53         }
 54     }

 56     public class Button : ButtonBase
 57     {
 58         private enum Flags : sbyte
 59         {
 60             none = 0, defaultButton = 1 << 0
 61         }
 62         public Button(ButtonStyle buttonStyle, const Color& backgroundColor, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 63             base("BUTTON", DefaultWindowClassStyle(), cast<WindowStyle>(DefaultChildWindowStyle() | WindowStyle.WS_TABSTOP | buttonStyle), DefaultExtendedWindowStyle(), 
 64             backgroundColor, text, location, size, dock, anchors), flags(Flags.none), dialogResult(DialogResult.none)
 65         {
 66             SetFlagsFromButtonStyle(buttonStyle);
 67         }
 68         public Button(const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 69             this(cast<ButtonStyle>(ButtonStyle.BS_PUSHBUTTON | ButtonStyle.BS_NOTIFY), GetSystemColor(SystemColor.COLOR_BTNFACE), text, location, size, dock, anchors)
 70         {
 71         }
 72         public Button(ButtonStyle buttonStyle, const string& text, const Point& location, const Size& size) : 
 73             this(buttonStyle, GetSystemColor(SystemColor.COLOR_BTNFACE), text, location, size, Dock.none, cast<Anchors>(Anchors.top | Anchors.left))
 74         {
 75         }
 76         public Button(const string& text, const Point& location, const Size& size) : 
 77             this(cast<ButtonStyle>(ButtonStyle.BS_PUSHBUTTON | ButtonStyle.BS_NOTIFY), text, location, size)
 78         {
 79         }
 80         protected override void OnCreated()
 81         {
 82             base->OnCreated();
 83             Graphics graphics = Graphics.FromWindowHandle(Handle());
 84             const FontHandle& fontHandle = GetFontHandle(graphics);
 85             if (!fontHandle.IsNull())
 86             {
 87                 SendSetFontMessage(fontHandle);
 88             }
 89         }
 90         protected override void OnGotFocus()
 91         {
 92             base->OnGotFocus();
 93             if (!IsDefault())
 94             {
 95                 Window* window = GetWindow();
 96                 if (window != null)
 97                 {
 98                     Button* defaultButton = window->DefaultButton();
 99                     if (defaultButton != null)
100                     {
101                         defaultButton->ResetDefaultButtonStyle();
102                     }
103                 }
104                 SetDefaultButtonStyle();
105             }
106         }
107         protected override void OnLostFocus()
108         {
109             base->OnLostFocus();
110             if (!IsDefault())
111             {
112                 ResetDefaultButtonStyle();
113                 Window* window = GetWindow();
114                 if (window != null)
115                 {
116                     Button* defaultButton = window->DefaultButton();
117                     if (defaultButton != null)
118                     {
119                         defaultButton->SetDefaultButtonStyle();
120                     }
121                 }
122             }
123         }
124         protected override void OnClick()
125         {
126             base->OnClick();
127             if (dialogResult != DialogResult.none)
128             {
129                 Window* window = GetWindow();
130                 if (window != null)
131                 {
132                     window->SetDialogResult(dialogResult);
133                 }
134             }
135         }
136         protected override void OnKeyDown(KeyEventArgs& args)
137         {
138             base->OnKeyDown(args);
139             if (!args.handled)
140             {
141                 switch (args.keyData)
142                 {
143                     case Keys.enter:
144                     {
145                         OnClick();
146                         break;
147                     }
148                 }
149             }
150         }
151         public override void PrintWindowTree(int level)
152         {
153             LogView* log = Application.GetLogView();
154             if (log != null)
155             {
156                 log->WriteLine(string(' ', level) + "Button." + Text() + ".handle=" + ToHexString(cast<ulong>(Handle())) + " " + ParentText() + "[" + Rect(Point(), GetSize()).ToString() + "]");
157             }
158         }
159         public inline nothrow DialogResult GetDialogResult() const
160         {
161             return dialogResult;
162         }
163         public inline nothrow void SetDialogResult(DialogResult dialogResult_)
164         {
165             dialogResult = dialogResult_;
166         }
167         private nothrow void SetFlagsFromButtonStyle(ButtonStyle buttonStyle)
168         {
169             if ((buttonStyle & ButtonStyle.BS_DEFPUSHBUTTON) != 0)
170             {
171                 flags = cast<Flags>(flags | Flags.defaultButton);
172             }
173         }
174         public inline nothrow bool IsDefault() const
175         {
176             return (flags & Flags.defaultButton) != Flags.none;
177         }
178         public void SetDefault()
179         {
180             flags = cast<Flags>(flags | Flags.defaultButton);
181             SetDefaultButtonStyle();
182         }
183         public void ResetDefault()
184         {
185             flags = cast<Flags>(flags & ~Flags.defaultButton);
186             ResetDefaultButtonStyle();
187         }
188         private void SetDefaultButtonStyle()
189         {
190             if (Handle() != null)
191             {
192                 WinSendMessage(Handle(), BM_SETSTYLE, cast<uint>(GetWindowStyle() | ButtonStyle.BS_DEFPUSHBUTTON), cast<long>(true));
193             }
194         }
195         private void ResetDefaultButtonStyle()
196         {
197             if (Handle() != null)
198             {
199                 WinSendMessage(Handle(), BM_SETSTYLE, cast<uint>(GetWindowStyle() & ~ButtonStyle.BS_DEFPUSHBUTTON), cast<long>(true));
200             }
201         }
202         private Flags flags;
203         private DialogResult dialogResult;
204     }
