 11     public enum ButtonStyle : long
 12     {
 13         BS_PUSHBUTTON = 0x0, 
 14         BS_DEFPUSHBUTTON = 0x1, 
 15         BS_3STATE = 0x5, 
 16         BS_AUTO3STATE = 0x6, 
 17         BS_AUTOCHECKBOX = 0x3, 
 18         BS_AUTORADIOBUTTON = 0x9, 
 19         BS_BITMAP = 0x80, 
 20         BS_BOTTOM = 0x800, 
 21         BS_CENTER = 0x300, 
 22         BS_CHECKBOX = 0x2, 
 23         BS_GROUPBOX = 0x7, 
 24         BS_ICON = 0x40, 
 25         BS_FLAT = 0x8000, 
 26         BS_LEFT = 0x100, 
 27         BS_LEFTTEXT = 0x20, 
 28         BS_MULTILINE = 0x2000, 
 29         BS_NOTIFY = 0x4000, 
 30         BS_OWNERDRAW = 0x0B, 
 31         BS_PUSHLIKE = 0x1000, 
 32         BS_RADIOBUTTON = 0x4, 
 33         BS_RIGHT = 0x200, 
 34         BS_TOP = 0x400, 
 35         BS_TYPEMASK = 0xF, 
 36         BS_VCENTER = 0xC00
 37     }

 39     public abstract class ButtonBase : Control
 40     {
 41         public ButtonBase(const char* windowClassName, WindowClassStyle windowClassStyle, WindowStyle style, ExtendedWindowStyle exStyle, 
 42             const Color& backgroundColor, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 43             base(windowClassName, windowClassStyle, style, exStyle, backgroundColor, text, location, size, dock, anchors)
 44         {
 45         }
 46         internal void OnClickInternal()
 47         {
 48             OnClick();
 49         }
 50     }

 52     public class Button : ButtonBase
 53     {
 54         private enum Flags : sbyte
 55         {
 56             none = 0, defaultButton = 1 << 0
 57         }
 58         public Button(ButtonStyle buttonStyle, const Color& backgroundColor, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 59             base("BUTTON", DefaultWindowClassStyle(), cast<WindowStyle>(DefaultChildWindowStyle() | WindowStyle.WS_TABSTOP | buttonStyle), DefaultExtendedWindowStyle(), 
 60             backgroundColor, text, location, size, dock, anchors), flags(Flags.none), dialogResult(DialogResult.none)
 61         {
 62             SetFlagsFromButtonStyle(buttonStyle);
 63         }
 64         public Button(const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 65             this(cast<ButtonStyle>(ButtonStyle.BS_PUSHBUTTON | ButtonStyle.BS_NOTIFY), GetSystemColor(SystemColor.COLOR_BTNFACE), text, location, size, dock, anchors)
 66         {
 67         }
 68         public Button(ButtonStyle buttonStyle, const string& text, const Point& location, const Size& size) : 
 69             this(buttonStyle, GetSystemColor(SystemColor.COLOR_BTNFACE), text, location, size, Dock.none, cast<Anchors>(Anchors.top | Anchors.left))
 70         {
 71         }
 72         public Button(const string& text, const Point& location, const Size& size) : 
 73             this(cast<ButtonStyle>(ButtonStyle.BS_PUSHBUTTON | ButtonStyle.BS_NOTIFY), text, location, size)
 74         {
 75         }
 76         protected override void OnCreated()
 77         {
 78             base->OnCreated();
 79             Graphics graphics = Graphics.FromWindowHandle(Handle());
 80             const FontHandle& fontHandle = GetFontHandle(graphics);
 81             if (!fontHandle.IsNull())
 82             {
 83                 SendSetFontMessage(fontHandle);
 84             }
 85         }
 86         protected override void OnGotFocus()
 87         {
 88             base->OnGotFocus();
 89             if (!IsDefault())
 90             {
 91                 Window* window = GetWindow();
 92                 if (window != null)
 93                 {
 94                     Button* defaultButton = window->DefaultButton();
 95                     if (defaultButton != null)
 96                     {
 97                         defaultButton->ResetDefaultButtonStyle();
 98                     }
 99                 }
100                 SetDefaultButtonStyle();
101             }
102         }
103         protected override void OnLostFocus()
104         {
105             base->OnLostFocus();
106             if (!IsDefault())
107             {
108                 ResetDefaultButtonStyle();
109                 Window* window = GetWindow();
110                 if (window != null)
111                 {
112                     Button* defaultButton = window->DefaultButton();
113                     if (defaultButton != null)
114                     {
115                         defaultButton->SetDefaultButtonStyle();
116                     }
117                 }
118             }
119         }
120         protected override void OnClick()
121         {
122             base->OnClick();
123             if (dialogResult != DialogResult.none)
124             {
125                 Window* window = GetWindow();
126                 if (window != null)
127                 {
128                     window->SetDialogResult(dialogResult);
129                 }
130             }
131         }
132         protected override void OnKeyDown(KeyEventArgs& args)
133         {
134             base->OnKeyDown(args);
135             if (!args.handled)
136             {
137                 switch (args.keyData)
138                 {
139                     case Keys.enter:
140                     {
141                         OnClick();
142                         break;
143                     }
144                 }
145             }
146         }
147         public override void PrintWindowTree(int level)
148         {
149             LogView* log = Application.GetLogView();
150             if (log != null)
151             {
152                 log->WriteLine(string(' ', level) + "Button." + Text() + ".handle=" + ToHexString(cast<ulong>(Handle())) + " " + ParentText() + "[" + Rect(Point(), GetSize()).ToString() + "]");
153             }
154         }
155         public inline nothrow DialogResult GetDialogResult() const
156         {
157             return dialogResult;
158         }
159         public inline nothrow void SetDialogResult(DialogResult dialogResult_)
160         {
161             dialogResult = dialogResult_;
162         }
163         private nothrow void SetFlagsFromButtonStyle(ButtonStyle buttonStyle)
164         {
165             if ((buttonStyle & ButtonStyle.BS_DEFPUSHBUTTON) != 0)
166             {
167                 flags = cast<Flags>(flags | Flags.defaultButton);
168             }
169         }
170         public inline nothrow bool IsDefault() const
171         {
172             return (flags & Flags.defaultButton) != Flags.none;
173         }
174         public void SetDefault()
175         {
176             flags = cast<Flags>(flags | Flags.defaultButton);
177             SetDefaultButtonStyle();
178         }
179         public void ResetDefault()
180         {
181             flags = cast<Flags>(flags & ~Flags.defaultButton);
182             ResetDefaultButtonStyle();
183         }
184         private void SetDefaultButtonStyle()
185         {
186             if (Handle() != null)
187             {
188                 WinSendMessage(Handle(), BM_SETSTYLE, cast<uint>(GetWindowStyle() | ButtonStyle.BS_DEFPUSHBUTTON), cast<long>(true));
189             }
190         }
191         private void ResetDefaultButtonStyle()
192         {
193             if (Handle() != null)
194             {
195                 WinSendMessage(Handle(), BM_SETSTYLE, cast<uint>(GetWindowStyle() & ~ButtonStyle.BS_DEFPUSHBUTTON), cast<long>(true));
196             }
197         }
198         private Flags flags;
199         private DialogResult dialogResult;
200     }
