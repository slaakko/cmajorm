  6 using System;
  7 
  8 namespace System.Windows
  9 {
 10     public class delegate void CheckedChangedEventHandler();
 11 
 12     public class CheckBox : ButtonBase
 13     {
 14         private enum Flags : sbyte
 15         {
 16             none = 0, checked = 1 << 0, autoSize = 1 << 1, autoSized = 1 << 2
 17         }
 18         public CheckBox(const Color& backgroundColor, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors, bool autoSize) : 
 19             base("BUTTON", DefaultWindowClassStyle(), cast<WindowStyle>(DefaultChildWindowStyle() | WindowStyle.WS_TABSTOP | ButtonStyle.BS_CHECKBOX | ButtonStyle.BS_NOTIFY), DefaultExtendedWindowStyle(), 
 20             backgroundColor, text, location, size, dock, anchors), flags(Flags.none)
 21         {
 22             if (autoSize)
 23             {
 24                 SetAutoSizeFlag();
 25             }
 26             else
 27             {
 28                 ResetAutoSizeFlag();
 29             }
 30         }
 31         public CheckBox(const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors, bool autoSize) : 
 32             this(DefaultControlBackgroundColor(), text, location, size, dock, anchors, autoSize)
 33         {
 34         }
 35         public bool Checked()
 36         {
 37             return GetCheckedFlag();
 38         }
 39         public void SetChecked(bool checked)
 40         {
 41             if (checked)
 42             {
 43                 SetCheckedFlag();
 44                 if (Handle() != null)
 45                 {
 46                     WinSendMessage(Handle(), BM_SETCHECK, BST_CHECKED, 0);
 47                 }
 48             }
 49             else
 50             {
 51                 ResetCheckedFlag();
 52                 if (Handle() != null)
 53                 {
 54                     WinSendMessage(Handle(), BM_SETCHECK, BST_UNCHECKED, 0);
 55                 }
 56             }
 57         }
 58         protected override void OnCreated()
 59         {
 60             base->OnCreated();
 61             Graphics graphics = Graphics.FromWindowHandle(Handle());
 62             const FontHandle& fontHandle = GetFontHandle(graphics);
 63             if (!fontHandle.IsNull())
 64             {
 65                 SendSetFontMessage(fontHandle);
 66             }
 67             if (GetCheckedFlag())
 68             {
 69                 SetChecked(true);
 70             }
 71             else
 72             {
 73                 SetChecked(false);
 74             }
 75             if (GetAutoSizeFlag())
 76             {
 77                 DoAutoSize();
 78             }
 79         }
 80         protected override void OnClick()
 81         {
 82             base->OnClick();
 83             SetChecked(!GetCheckedFlag());
 84         }
 85         protected override void OnTextChanged()
 86         {
 87             base->OnTextChanged();
 88             if (GetAutoSizeFlag())
 89             {
 90                 ResetAutoSized();
 91                 DoAutoSize();
 92             }
 93         }
 94         protected virtual void OnCheckedChanged()
 95         {
 96             checkedChangedEvent.Fire();
 97         }
 98         public nothrow Event<CheckedChangedEventHandler>& CheckedChangedEvent() const
 99         {
100             return checkedChangedEvent;
101         }
102         private void RetrieveCheckedState()
103         {
104             long result = WinSendMessage(Handle(), BM_GETCHECK, 0u, 0);
105             if (result == BST_CHECKED)
106             {
107                 SetCheckedFlag();
108             }
109             else
110             {
111                 ResetCheckedFlag();
112             }
113         }
114         private void DoAutoSize()
115         {
116             if (AutoSized()) return;
117             if (Handle() == null) return;
118             Graphics graphics = Graphics.FromWindowHandle(Handle());
119             const Font& font = GetFont();
120             StringFormat stringFormat;
121             RectF r = graphics.MeasureStringChecked(Text(), font, PointF(0, 0), stringFormat);
122             Size checkSize = GetCheckSize();
123             Size borderSize = GetBorderSize();
124             r.size.w = r.size.w + checkSize.w + borderSize.w;
125             r.size.h = Max(r.size.h, checkSize.h + borderSize.h);
126             SetSize(Size(cast<int>(r.size.w), cast<int>(r.size.h)));
127             SetAutoSized();
128         }
129         private nothrow Size GetCheckSize()
130         {
131             int x = GetSystemMetrics(SystemMetricsId.SM_CXMENUCHECK);
132             int y = GetSystemMetrics(SystemMetricsId.SM_CYMENUCHECK);
133             return Size(x, y);
134         }
135         private nothrow Size GetBorderSize()
136         {
137             int x = GetSystemMetrics(SystemMetricsId.SM_CXBORDER);
138             int y = GetSystemMetrics(SystemMetricsId.SM_CYBORDER);
139             return Size(x, y);
140         }
141         private inline nothrow bool GetCheckedFlag() const
142         {
143             return (flags & Flags.checked) != Flags.none;
144         }
145         private void SetCheckedFlag()
146         {
147             if (!GetCheckedFlag())
148             {
149                 flags = cast<Flags>(flags | Flags.checked);
150                 OnCheckedChanged();
151             }
152         }
153         private void ResetCheckedFlag()
154         {
155             if (GetCheckedFlag())
156             {
157                 flags = cast<Flags>(flags & ~Flags.checked);
158                 OnCheckedChanged();
159             }
160         }
161         private inline nothrow bool GetAutoSizeFlag() const
162         {
163             return (flags & Flags.autoSize) != Flags.none;
164         }
165         private inline nothrow void SetAutoSizeFlag()
166         {
167             flags = cast<Flags>(flags | Flags.autoSize);
168         }
169         private inline nothrow void ResetAutoSizeFlag()
170         {
171             flags = cast<Flags>(flags & ~Flags.autoSize);
172         }
173         private inline nothrow bool AutoSized() const
174         {
175             return (flags & Flags.autoSized) != Flags.none;
176         }
177         private inline nothrow void SetAutoSized()
178         {
179             flags = cast<Flags>(flags | Flags.autoSized);
180         }
181         private inline nothrow void ResetAutoSized()
182         {
183             flags = cast<Flags>(flags & ~Flags.autoSized);
184         }
185         private Flags flags;
186         private Event<CheckedChangedEventHandler> checkedChangedEvent;
187     }
188 }
