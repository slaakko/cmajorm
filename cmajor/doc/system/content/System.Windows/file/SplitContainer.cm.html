 16     public nothrow Dock SplitDock(Orientation orientation)
 17     {
 18         switch (orientation)
 19         {
 20             case Orientation.horizontal: return Dock.left;
 21             case Orientation.vertical: return Dock.top;
 22         }
 23         return Dock.none;
 24     }

 27     {
 28         switch (orientation)
 29         {
 30             case Orientation.horizontal: return Size(width, 0);
 31             case Orientation.vertical: return Size(0, width);
 32         }
 33         return Size();
 34     }

 38     public class Splitter : Control

 58         public override void PrintWindowTree(int level)
 59         {
 60             LogView* log = Application.GetLogView();
 61             if (log != null)
 62             {
 63                 log->WriteLine(string(' ', level) + "Splitter." + Text() + ".handle=" + ToHexString(cast<ulong>(Handle())) + " " + ParentText() + "[" + Rect(Point(), GetSize()).ToString() + "]");

 66         protected override void OnPaint(PaintEventArgs& args)
 67         {
 68             try
 69             {
 70                 if (Debug.Paint())
 71                 {
 72                     Rect r(Point(), GetSize());
 73                     LogView* log = Application.GetLogView();
 74                     if (log != null)
 75                     {
 76                         log->WriteLine("Splitter.OnPaint: " + r.ToString());
 77                     }
 78                 }
 79                 base->OnPaint(args);
 80                 args.graphics.Clear(BackgroundColor());
 81             }
 82             catch (const Exception& ex)
 83             {
 84                 MessageBox.Show(ex.Message());
 85             }
 86         }
 87         protected override void SetCursor()

 92                 case Orientation.vertical: SetCursor(verticalSplitterCursor); break;
 93             }
 94         }
 95         protected override void OnMouseDown(MouseEventArgs& args)
 96         {
 97             try
 98             {
 99                 base->OnMouseDown(args);
100                 if (args.buttons == MouseButtons.lbutton)
101                 {
102                     SetFlag(Flags.moveSplitter);
103                     switch (orientation)
104                     {
105                         case Orientation.horizontal:
106                         {
107                             x = args.location.x;
108                             break;
109                         }
110                         case Orientation.vertical:
111                         {
112                             y = args.location.y;
113                             break;
114                         }
115                     }
116                     WinSetCapture(Handle());
117                 }
118             }
119             catch (const Exception& ex)
120             {
121                 MessageBox.Show(ex.Message());
122             }
123         }
124         protected override void OnMouseMove(MouseEventArgs& args)
125         {
126             try
127             {
128                 if (GetFlag(Flags.moveSplitter))
129                 {
130                     switch (orientation)
131                     {
132                         case Orientation.horizontal:
133                         {
134                             int dx = args.location.x - x;
135                             container->SetSplitterDistance(container->SplitterDistance() + dx);
136                             break;
137                         }
138                         case Orientation.vertical:
139                         {
140                             int dy = args.location.y - y;
141                             container->SetSplitterDistance(container->SplitterDistance() + dy);
142                             break;
143                         }
144                     }
145                 }
146             }
147             catch (const Exception& ex)
148             {
149                 MessageBox.Show(ex.Message());
150             }
151         }
152         protected override void OnMouseUp(MouseEventArgs& args)
153         {
154             try
155             {
156                 base->OnMouseUp(args);
157                 if (GetFlag(Flags.moveSplitter))
158                 {
159                     ResetFlag(Flags.moveSplitter);
160                     WinReleaseCapture();
161                     switch (orientation)
162                     {
163                         case Orientation.horizontal:
164                         {
165                             int dx = args.location.x - x;
166                             container->SetSplitterDistance(container->SplitterDistance() + dx);
167                             break;
168                         }
169                         case Orientation.vertical:
170                         {
171                             int dy = args.location.y - y;
172                             container->SetSplitterDistance(container->SplitterDistance() + dy);
173                             break;
174                         }
175                     }
176                 }
177             }
178             catch (const Exception& ex)
179             {
180                 MessageBox.Show(ex.Message());
181             }
182         }
183         private inline nothrow void SetFlag(Flags flag)
184         {
185             flags = cast<Flags>(flags | flag);
186         }
187         private inline nothrow void ResetFlag(Flags flag)
188         {
189             flags = cast<Flags>(flags & ~flag);
190         }
191         private inline nothrow bool GetFlag(Flags flag) const
192         {
193             return (flags & flag) != 0;
194         }
195         private Orientation orientation;
196         private Cursor& horizontalSplitterCursor;
197         private Cursor& verticalSplitterCursor;
198         private Flags flags;
199         private int x;
200         private int y;
201         private SplitContainer* container;
202     }

204     public class SplitContainer : ContainerControl
205     {
206         public SplitContainer(Orientation orientation_, Control* pane1_, Splitter* splitter_, Control* pane2_, int splitterWidth, int splitterDistance_, 
207             const Point& location, const Size& size, Dock dock, Anchors anchors) : 
208             base("System.Windows.SplitContainer", DefaultWindowClassStyle(), DefaultChildWindowStyle(), DefaultExtendedWindowStyle(), 
209                 DefaultControlBackgroundColor(), "splitContainer", location, size, dock, anchors), 
210                 orientation(orientation_), pane1(pane1_), splitter(splitter_), pane2(pane2_), splitterDistance(splitterDistance_)
211         {
212             pane1->SetSize(SplitSize(orientation, splitterDistance));
213             pane1->SetDock(SplitDock(orientation));
214             AddChild(pane1);
215             splitter->SetContainer(this);
216             AddChild(splitter);
217             pane2->SetLocation(location);
218             pane2->SetDock(Dock.fill);
219             AddChild(pane2);
220         }
221         public SplitContainer(Orientation orientation, int splitterDistance, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
222             this(orientation, 
223                 new Panel(location, SplitSize(orientation, splitterDistance), SplitDock(orientation), Anchors.none), 
224                 new Splitter(orientation, location), 
225                 new Panel(location, SplitSize(orientation, splitterDistance), SplitDock(orientation), Anchors.none), 
226                 defaultSplitterWidth, splitterDistance, location, size, dock, anchors)
227         {
228         }
229         public override void PrintWindowTree(int level)
230         {
231             LogView* log = Application.GetLogView();
232             if (log != null)
233             {
234                 log->WriteLine(string(' ', level) + "SplitContainer." + Text() + ".handle=" + ToHexString(cast<ulong>(Handle())) + " " + ParentText() + "[" + Rect(Point(), GetSize()).ToString() + "]");
235             }
236             Component* child = Children().FirstChild();
237             while (child != null)
238             {
239                 if (child is Control*)
240                 {
241                     Control* childControl = cast<Control*>(child);
242                     childControl->PrintWindowTree(level + 1);
243                 }
244                 child = child->NextSibling();
245             }
246         }

267         public void SetSplitterDistance(int splitterDistance_)
268         {
269             splitterDistance = splitterDistance_;
270             switch (orientation)
271             {
272                 case Orientation.horizontal: pane1->SetSize(Size(splitterDistance, pane1->GetSize().h)); break;
273                 case Orientation.vertical: pane1->SetSize(Size(pane1->GetSize().w, splitterDistance)); break;
274             }
275             DockChildren();
276             Invalidate();
277         }
