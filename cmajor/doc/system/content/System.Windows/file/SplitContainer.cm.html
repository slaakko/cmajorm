 75         protected override void OnPaint(PaintEventArgs& args)
 76         {
 77             try
 78             {
 79                 if (Debug.Paint())
 80                 {
 81                     Rect r(Point(), GetSize());
 82                     LogView* log = Application.GetLogView();
 83                     if (log != null)
 84                     {
 85                         log->WriteLine("Splitter.OnPaint: " + r.ToString());
 86                     }
 87                 }
 88                 args.graphics.Clear(BackgroundColor());
 89                 Size size = GetSize();
 90                 LogView* log = Application.GetLogView();
 91                 switch (orientation)
 92                 {
 93                     case Orientation.horizontal:
 94                     {
 95                         args.graphics.DrawLineChecked(edgePen, Point(0, 0), Point(0, size.h - 1));
 96                         args.graphics.DrawLineChecked(edgePen, Point(size.w - 1, 0), Point(size.w - 1, size.h - 1));
 97                         break;
 98                     }
 99                     case Orientation.vertical:
100                     {
101                         args.graphics.DrawLineChecked(edgePen, Point(0, 0), Point(size.w - 1, 0));
102                         args.graphics.DrawLineChecked(edgePen, Point(0, size.h - 1), Point(size.w - 1, size.h - 1));
103                         break;
104                     }
105                 }
106                 base->OnPaint(args);
107             }
108             catch (const Exception& ex)
109             {
110                 MessageBox.Show(ex.Message());
111             }
112         }

121         protected override void OnMouseDown(MouseEventArgs& args)
122         {
123             try
124             {
125                 base->OnMouseDown(args);
126                 if (args.buttons == MouseButtons.lbutton)
127                 {
128                     SetFlag(Flags.moveSplitter);
129                     switch (orientation)
130                     {
131                         case Orientation.horizontal:
132                         {
133                             x = args.location.x;
134                             break;
135                         }
136                         case Orientation.vertical:
137                         {
138                             y = args.location.y;
139                             break;
140                         }
141                     }
142                     WinSetCapture(Handle());
143                 }
144             }
145             catch (const Exception& ex)
146             {
147                 MessageBox.Show(ex.Message());
148             }
149         }
150         protected override void OnMouseMove(MouseEventArgs& args)
151         {
152             try
153             {
154                 if (GetFlag(Flags.moveSplitter))
155                 {
156                     switch (orientation)
157                     {
158                         case Orientation.horizontal:
159                         {
160                             int dx = args.location.x - x;
161                             container->SetSplitterDistance(container->SplitterDistance() + dx);
162                             break;
163                         }
164                         case Orientation.vertical:
165                         {
166                             int dy = args.location.y - y;
167                             container->SetSplitterDistance(container->SplitterDistance() + dy);
168                             break;
169                         }
170                     }
171                 }
172             }
173             catch (const Exception& ex)
174             {
175                 MessageBox.Show(ex.Message());
176             }
177         }
178         protected override void OnMouseUp(MouseEventArgs& args)
179         {
180             try
181             {
182                 base->OnMouseUp(args);
183                 if (GetFlag(Flags.moveSplitter))
184                 {
185                     ResetFlag(Flags.moveSplitter);
186                     WinReleaseCapture();
187                     switch (orientation)
188                     {
189                         case Orientation.horizontal:
190                         {
191                             int dx = args.location.x - x;
192                             container->SetSplitterDistance(container->SplitterDistance() + dx);
193                             break;
194                         }
195                         case Orientation.vertical:
196                         {
197                             int dy = args.location.y - y;
198                             container->SetSplitterDistance(container->SplitterDistance() + dy);
199                             break;
200                         }
201                     }
202                 }
203             }
204             catch (const Exception& ex)
205             {
206                 MessageBox.Show(ex.Message());
207             }
208         }
209         private inline nothrow void SetFlag(Flags flag)
210         {
211             flags = cast<Flags>(flags | flag);
212         }
213         private inline nothrow void ResetFlag(Flags flag)
214         {
215             flags = cast<Flags>(flags & ~flag);
216         }
217         private inline nothrow bool GetFlag(Flags flag) const
218         {
219             return (flags & flag) != 0;
220         }

232     public class SplitContainer : ContainerControl
233     {
234         public SplitContainer(Orientation orientation_, Control* pane1_, Splitter* splitter_, Control* pane2_, int splitterWidth, int splitterDistance_, 
235             const Point& location, const Size& size, Dock dock, Anchors anchors) : 
236             base("System.Windows.SplitContainer", DefaultWindowClassStyle(), DefaultChildWindowStyle(), DefaultExtendedWindowStyle(), 
237                 DefaultControlBackgroundColor(), "splitContainer", location, size, dock, anchors), 
238                 orientation(orientation_), pane1(pane1_), splitter(splitter_), pane2(pane2_), splitterDistance(splitterDistance_)
239         {
240             pane1->SetSize(SplitSize(orientation, splitterDistance));
241             pane1->SetDock(SplitDock(orientation));
242             AddChild(pane1);
243             splitter->SetContainer(this);
244             AddChild(splitter);
245             pane2->SetLocation(location);
246             pane2->SetDock(Dock.fill);
247             AddChild(pane2);
248         }
249         public SplitContainer(Orientation orientation, int splitterDistance, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
250             this(orientation, 
251                 new Panel(location, SplitSize(orientation, splitterDistance), SplitDock(orientation), Anchors.none), 
252                 new Splitter(orientation, location), 
253                 new Panel(location, SplitSize(orientation, splitterDistance), SplitDock(orientation), Anchors.none), 
254                 defaultSplitterWidth, splitterDistance, location, size, dock, anchors)
255         {
256         }
257         public override void OnPaint(PaintEventArgs& args)
258         {
259             Size size = GetSize();
260             if (splitterDistance == 0)
261             {
262                 int s = 0;
263                 switch (orientation)
264                 {
265                     case Orientation.horizontal: s = size.w; break;
266                     case Orientation.vertical: s = size.h; break;
267                 }
268                 int d = cast<int>((2.0 / 3.0) * s);
269                 SetSplitterDistance(d);
270             }
271             base->OnPaint(args);
272         }
273         public override void PrintWindowTree(int level)
274         {
275             LogView* log = Application.GetLogView();
276             if (log != null)
277             {
278                 log->WriteLine(string(' ', level) + "SplitContainer." + Text() + ".handle=" + ToHexString(cast<ulong>(Handle())) + " " + ParentText() + "[" + Rect(Point(), GetSize()).ToString() + "]");
279             }
280             Component* child = Children().FirstChild();
281             while (child != null)
282             {
283                 if (child is Control*)
284                 {
285                     Control* childControl = cast<Control*>(child);
286                     childControl->PrintWindowTree(level + 1);
287                 }
288                 child = child->NextSibling();
289             }
290         }

311         public void SetSplitterDistance(int splitterDistance_)
312         {
313             int prevSplitterDistance = splitterDistance;
314             splitterDistance = splitterDistance_;
315             switch (orientation)
316             {
317                 case Orientation.horizontal: pane1->SetSize(Size(splitterDistance, pane1->GetSize().h)); break;
318                 case Orientation.vertical: pane1->SetSize(Size(pane1->GetSize().w, splitterDistance)); break;
319             }
320             DockChildren();
321             Invalidate();
322             if (prevSplitterDistance != splitterDistance)
323             {
324                 OnSplitterDistanceChanged();
325             }
326         }
327         public virtual void OnSplitterDistanceChanged()
328         {
329             splitterDistanceChangedEvent.Fire();
330         }

332         {
333             return splitterDistanceChangedEvent;
334         }
