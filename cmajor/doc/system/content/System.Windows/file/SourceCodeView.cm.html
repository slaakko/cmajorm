128         protected override void OnLinesChanged()
129         {
130             base->OnLinesChanged();
131             tokenLines.Clear();
132             int state = 0;
133             int n = cast<int>(Lines().Count());
134             numLineNumberDigits = Log10(n + 1);
135             for (int i = 0; i < n; ++i;)
136             {
137                 const ustring& line = Lines()[i];
138                 TokenLine tokenLine = TokenizeLine(line, i + 1, state);
139                 state = tokenLine.endState;
140                 tokenLines.Add(Rvalue(tokenLine));
141             }
142         }
143         protected override void OnLineChanged(LineEventArgs& args)
144         {
145             base->OnLineChanged(args);
146             int lineIndex = args.lineIndex;
147             const ustring& line = Lines()[lineIndex];
148             int state = 0;
149             if (lineIndex > 0 && !tokenLines.IsEmpty())
150             {
151                 state = tokenLines[lineIndex - 1].endState;
152             }
153             while (lineIndex >= tokenLines.Count())
154             {
155                 tokenLines.Add(TokenLine());
156             }
157             tokenLines[lineIndex] = TokenizeLine(line, lineIndex + 1, state);
158         }

165         private nothrow bool IsBeginBlockLine(int lineIndex) const
166         {
167             if (lineIndex >= 0 && lineIndex < tokenLines.Count())
168             {
169                 const TokenLine& tokenLine = tokenLines[lineIndex];
170                 for (const Token& token : tokenLine.tokens)
171                 {
172                     if (GetTokenKind(token) == SourceCodeTokenKind.beginBlock)
173                     {
174                         return true;
175                     }
176                 }
177             }
178             return false;
179         }
180         private nothrow bool IsEndBlockLine(int lineIndex) const
181         {
182             if (lineIndex >= 0 && lineIndex < tokenLines.Count())
183             {
184                 const TokenLine& tokenLine = tokenLines[lineIndex];
185                 for (const Token& token : tokenLine.tokens)
186                 {
187                     if (GetTokenKind(token) == SourceCodeTokenKind.endBlock)
188                     {
189                         return true;
190                     }
191                 }
192             }
193             return false;

210         protected override nothrow int GetIndent(const ustring& line, int lineIndex)
211         {
212             for (int i = 0; i < line.Length(); ++i;)
213             {
214                 if (line[i] != ' ')
215                 {
216                     if (IsBeginBlockLine(lineIndex))
217                     {
218                         return i + indentSize;
219                     }
220                     else
221                     {
222                         return i;
223                     }
224                 }
225             }
226             return 0;
227         }
228         protected override void OnLineInserted(LineEventArgs& args)
229         {
230             base->OnLineInserted(args);
231             int lineIndex = args.lineIndex;
232             ustring& line = Lines()[lineIndex];
233             int indent = 0;
234             if (args.indentLineIndex >= 0 && args.indentLineIndex < Lines().Count())
235             {
236                 indent = GetIndent(Lines()[args.indentLineIndex], args.indentLineIndex);
237             }
238             if (indent != 0)
239             {
240                 line.Insert(0, ustring(' ', indent));
241             }
242             int state = 0;
243             while (lineIndex >= tokenLines.Count())
244             {
245                 tokenLines.Add(TokenLine());
246             }
247             if (lineIndex > 0)
248             {
249                 state = tokenLines[lineIndex - 1].endState;
250             }
251             tokenLines.Insert(tokenLines.Begin() + lineIndex, TokenizeLine(line, lineIndex + 1, state));
252             Invalidate();
253         }
254         public nothrow ustring GetText(const SourceSpan& span) const
255         {
256             if (span.line >= 1 && span.line <= Lines().Count())
257             {
258                 const ustring& line = Lines()[span.line - 1];
259                 if (span.scol >= 1 && span.scol <= line.Length())
260                 {
261                     int n = span.ecol - span.scol;
262                     if (n > 0)
263                     {
264                         return line.Substring(span.scol - 1, n);
265                     }
266                 }
267             }
268             return ustring();
269         }

275         public nothrow ustring GetTokenText(int lineNumber, short columnNumber) const
276         {
277             if (lineNumber >= 1 && lineNumber <= tokenLines.Count())
278             {

280                 int tokenIndex = tokenLine.TokenIndex(columnNumber);

284                     if (GetTokenKind(token) == SourceCodeTokenKind.identifier)

288                 }
289             }

300         protected override void DrawLine(Graphics& graphics, int lineIndex, const PointF& origin)
301         {
302             int lineNumber = lineIndex + 1;
303             string lineNumberStr = System.ToString(lineNumber);
304             string lineNumberText = Format(lineNumberStr, numLineNumberDigits, FormatJustify.right);
305             PointF pt(origin);
306             SourceCodeTextStyle* lineNumberTextStyle = GetTextStyle(SourceCodeTokenKind.lineNumber);
307             graphics.DrawStringChecked(lineNumberText, *lineNumberTextStyle->font, pt, *lineNumberTextStyle->brush);
308             pt.x = pt.x + CharWidth() * (numLineNumberDigits + 1);
309             const TokenLine& tokenLine = tokenLines[lineIndex];
310             int startState = tokenLine.startState;
311             for (const Token& token : tokenLine.tokens)
312             {
313                 SourceCodeTokenKind tokenKind = GetTokenKind(token);
314                 SourceCodeTextStyle* tokenTextStyle = GetTextStyle(tokenKind);
315                 ustring tokenStr = token.match.ToString();
316                 string s(ToUtf8(tokenStr));
317                 graphics.DrawStringChecked(s, *tokenTextStyle->font, pt, *tokenTextStyle->brush);
318                 pt.x = pt.x + CharWidth() * tokenStr.Length();
319                 startState = -1;
320             }
321             PointF hiliteOrigin(origin);
322             hiliteOrigin.x = hiliteOrigin.x + CharWidth() * LineNumberFieldLength();
323             DrawHilites(graphics, lineIndex, hiliteOrigin);
324         }

328         private SourceCodeTextStyle* GetTextStyle(SourceCodeTokenKind tokenKind)
329         {

331             SourceCodeTokenStyle* tokenStyle = null;
332             HashMap<SourceCodeTokenKind, SourceCodeTokenStyle*>.ConstIterator it = tokenKindMap.CFind(tokenKind);
333             if (it != tokenKindMap.CEnd())
334             {
335                 tokenStyle = it->second;
336             }
337             else
338             {
339                 throw Exception("source code token style not found");
340             }
341             HashMap<SourceCodeTokenStyle*, SourceCodeTextStyle*>.ConstIterator it2 = tokenStyleTextStyleMap.CFind(tokenStyle);
342             if (it2 != tokenStyleTextStyleMap.CEnd())
343             {
344                 textStyle = it2->second;
345             }
346             else
347             {
348                 throw Exception("source code text style not found");
349             }
350             return textStyle;

375         private SourceCodeTokenStyle* GetOrInsertTokenStyle(const Color& color, FontStyle fontStyle)
376         {
377             SourceCodeTokenStyle style(color, fontStyle);
378             HashMap<SourceCodeTokenStyle, SourceCodeTokenStyle*>.ConstIterator it = tokenStyleMap.CFind(style);
379             if (it != tokenStyleMap.CEnd())
380             {
381                 return it->second;
382             }
383             SourceCodeTokenStyle* sourceCodeTokenStyle = new SourceCodeTokenStyle(color, fontStyle);
384             sourceCodeTokenStyles.Add(UniquePtr<SourceCodeTokenStyle>(sourceCodeTokenStyle));
385             tokenStyleMap[style] = sourceCodeTokenStyle;
386             Brush* brush = GetOrInsertBrush(color);

388             SourceCodeTextStyle* textStyle = GetOrInsertTextStyle(brush, font);
389             tokenStyleTextStyleMap[sourceCodeTokenStyle] = textStyle;
390             return sourceCodeTokenStyle;
391         }

429     public TokenLine DefaultTokenizeLine(const ustring& line, int lineNumber, int startState)
430     {
431         Token token;
432         TokenLine tokenLine;
433         uchar* begin = line.Chars();
434         uchar* end = line.Chars() + line.Length();
435         token.match.begin = begin;
436         token.match.end = end;
437         token.line = lineNumber;
438         tokenLine.tokens.Add(token);
439         tokenLine.endState = 0;
440         return tokenLine;
441     }
