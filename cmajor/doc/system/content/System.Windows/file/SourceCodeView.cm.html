 74         protected override void OnFontChanged()
 75         {
 76             fontStyleFontMap.Clear();
 77             textStyleMap.Clear();
 78             tokenStyleTextStyleMap.Clear();
 79             for (UniquePtr<SourceCodeTokenStyle>& tokenStyle : sourceCodeTokenStyles)
 80             {
 81                 Brush* brush = GetOrInsertBrush(tokenStyle->color);
 82                 Font* font = GetOrInsertFont(tokenStyle->fontStyle);
 83                 SourceCodeTextStyle* textStyle = GetOrInsertTextStyle(brush, font);
 84                 tokenStyleTextStyleMap[tokenStyle.Get()] = textStyle;
 85             }
 86         }

 92         protected override nothrow int LineNumberFieldLength() const
 93         {
 94             return numLineNumberDigits + 1;
 95         }
 96         protected override nothrow void SetLineNumberFieldLength(int lineCount)
 97         {
 98             numLineNumberDigits = Log10(lineCount + 1);
 99         }
100         protected override void OnLinesChanged()
101         {
102             base->OnLinesChanged();
103             tokenLines.Clear();
104             int state = 0;
105             int n = cast<int>(Lines().Count());
106             numLineNumberDigits = Log10(n + 1);
107             for (int i = 0; i < n; ++i;)
108             {
109                 const ustring& line = Lines()[i];
110                 TokenLine tokenLine = TokenizeLine(line, i + 1, state);
111                 state = tokenLine.endState;
112                 tokenLines.Add(Rvalue(tokenLine));
113             }
114         }

118         }

123         protected override void DrawLine(Graphics& graphics, int lineIndex, const PointF& origin)
124         {
125             int lineNumber = lineIndex + 1;
126             string lineNumberStr = System.ToString(lineNumber);
127             string lineNumberText = Format(lineNumberStr, numLineNumberDigits, FormatJustify.right);
128             PointF pt(origin);
129             SourceCodeTextStyle* lineNumberTextStyle = GetTextStyle(SourceCodeTokenKind.lineNumber);
130             graphics.DrawStringChecked(lineNumberText, *lineNumberTextStyle->font, pt, *lineNumberTextStyle->brush);
131             pt.x = pt.x + CharWidth() * (numLineNumberDigits + 1);
132             const TokenLine& tokenLine = tokenLines[lineIndex];
133             int startState = tokenLine.startState;
134             for (const Token& token : tokenLine.tokens)
135             {
136                 SourceCodeTokenKind tokenKind = GetTokenKind(token);
137                 SourceCodeTextStyle* tokenTextStyle = GetTextStyle(tokenKind);
138                 ustring tokenStr = token.match.ToString();
139                 string s(ToUtf8(tokenStr));
140                 graphics.DrawStringChecked(s, *tokenTextStyle->font, pt, *tokenTextStyle->brush);
141                 pt.x = pt.x + CharWidth() * tokenStr.Length();
142                 startState = -1;
143             }
144         }
145         private SourceCodeTextStyle* GetTextStyle(SourceCodeTokenKind tokenKind)
146         {
147             SourceCodeTextStyle* textStyle = null;
148             SourceCodeTokenStyle* tokenStyle = null;
149             HashMap<SourceCodeTokenKind, SourceCodeTokenStyle*>.ConstIterator it = tokenKindMap.CFind(tokenKind);
150             if (it != tokenKindMap.CEnd())
151             {
152                 tokenStyle = it->second;
153             }
154             else
155             {
156                 throw Exception("source code token style not found");
157             }
158             HashMap<SourceCodeTokenStyle*, SourceCodeTextStyle*>.ConstIterator it2 = tokenStyleTextStyleMap.CFind(tokenStyle);
159             if (it2 != tokenStyleTextStyleMap.CEnd())
160             {
161                 textStyle = it2->second;
162             }
163             else
164             {
165                 throw Exception("source code text style not found");
166             }
167             return textStyle;
168         }
169         private void BuildDefaultStyles()
170         {
171             SourceCodeTokenStyle* plainStyle = GetOrInsertTokenStyle(Color.Black(), FontStyle.regular);
172             tokenKindMap[SourceCodeTokenKind.plain] = plainStyle;
173             SourceCodeTokenStyle* keywordStyle = GetOrInsertTokenStyle(Color.Blue(), FontStyle.regular);
174             tokenKindMap[SourceCodeTokenKind.keyword] = keywordStyle;
175             SourceCodeTokenStyle* identifierStyle = GetOrInsertTokenStyle(Color.Black(), FontStyle.regular);
176             tokenKindMap[SourceCodeTokenKind.identifier] = identifierStyle;
177             SourceCodeTokenStyle* stringStyle = GetOrInsertTokenStyle(Color(163u, 21u, 21u), FontStyle.regular);
178             tokenKindMap[SourceCodeTokenKind.string] = stringStyle;
179             SourceCodeTokenStyle* characterStyle = GetOrInsertTokenStyle(Color(163u, 21u, 21u), FontStyle.regular);
180             tokenKindMap[SourceCodeTokenKind.character] = characterStyle;
181             SourceCodeTokenStyle* numberStyle = GetOrInsertTokenStyle(Color.Black(), FontStyle.regular);
182             tokenKindMap[SourceCodeTokenKind.number] = numberStyle;
183             SourceCodeTokenStyle* commentStyle = GetOrInsertTokenStyle(Color(0u, 128u, 0u), FontStyle.regular);
184             tokenKindMap[SourceCodeTokenKind.comment] = commentStyle;
185             SourceCodeTokenStyle* lineNumberStyle = GetOrInsertTokenStyle(Color(43u, 145u, 175u), FontStyle.regular);
186             tokenKindMap[SourceCodeTokenKind.lineNumber] = lineNumberStyle;
187         }
188         private SourceCodeTokenStyle* GetOrInsertTokenStyle(const Color& color, FontStyle fontStyle)
189         {
190             SourceCodeTokenStyle style(color, fontStyle);
191             HashMap<SourceCodeTokenStyle, SourceCodeTokenStyle*>.ConstIterator it = tokenStyleMap.CFind(style);
192             if (it != tokenStyleMap.CEnd())
193             {
194                 return it->second;
195             }
196             SourceCodeTokenStyle* sourceCodeTokenStyle = new SourceCodeTokenStyle(color, fontStyle);
197             sourceCodeTokenStyles.Add(UniquePtr<SourceCodeTokenStyle>(sourceCodeTokenStyle));
198             tokenStyleMap[style] = sourceCodeTokenStyle;
199             Brush* brush = GetOrInsertBrush(color);
200             Font* font = GetOrInsertFont(fontStyle);
201             SourceCodeTextStyle* textStyle = GetOrInsertTextStyle(brush, font);
202             tokenStyleTextStyleMap[sourceCodeTokenStyle] = textStyle;
203             return sourceCodeTokenStyle;
204         }
205         private Brush* GetOrInsertBrush(const Color& color)
206         {
207             HashMap<Color, Brush*>.ConstIterator it = colorBrushMap.CFind(color);
208             if (it != colorBrushMap.CEnd())
209             {
210                 return it->second;
211             }
212             Brush* brush = new SolidBrush(color);
213             brushes.Add(UniquePtr<Brush>(brush));
214             colorBrushMap[color] = brush;
215             return brush;
216         }
217         private Font* GetOrInsertFont(FontStyle fontStyle)
218         {
219             Map<FontStyle, Font*>.ConstIterator it = fontStyleFontMap.CFind(fontStyle);
220             if (it != fontStyleFontMap.CEnd())
221             {
222                 return it->second;
223             }
224             Font* font = new Font(GetFontFamily(), FontSize(), fontStyle, Unit.point);
225             Fonts().Add(UniquePtr<Font>(font));
226             fontStyleFontMap[fontStyle] = font;
227             return font;
228         }
229         private SourceCodeTextStyle* GetOrInsertTextStyle(Brush* brush, Font* font)
230         {
231             SourceCodeTextStyle style(font, brush);
232             HashMap<SourceCodeTextStyle, SourceCodeTextStyle*>.ConstIterator it = textStyleMap.CFind(style);
233             if (it != textStyleMap.CEnd())
234             {
235                 return it->second;
236             }
237             SourceCodeTextStyle* textStyle = new SourceCodeTextStyle(font, brush);
238             textStyles.Add(UniquePtr<SourceCodeTextStyle>(textStyle));
239             textStyleMap[style] = textStyle;
240             return textStyle;
241         }

255     public TokenLine DefaultTokenizeLine(const ustring& line, int lineNumber, int startState)
256     {
257         Token token;
258         TokenLine tokenLine;
259         uchar* begin = line.Chars();
260         uchar* end = line.Chars() + line.Length();
261         token.match.begin = begin;
262         token.match.end = end;
263         token.line = lineNumber;
264         tokenLine.tokens.Add(token);
265         tokenLine.endState = 0;
266         return tokenLine;
267     }
