 6 using System;
 7 using System.Collections;
 8 
 9 namespace System.Windows
10 {
11     public class LogView : Control
12     {
13         private enum Flags : sbyte
14         {
15             none = 0, changed = 1 << 0
16         }
17 
18         public LogView(const Font& font_, const Color& backgroundColor, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
19             base("System.Windows.LogView", WindowClassStyle.CS_DEFAULT, cast<WindowStyle>(WindowStyle.WS_CHILD | WindowStyle.WS_VISIBLE), 
20             ExtendedWindowStyle.WS_EX_DEFAULT, backgroundColor, string(), location, size, dock, anchors), font(font_), textBrush(Color.Black()), lines(), textHeight(10), flags(Flags.none)
21         {
22             SetChanged();
23         }
24         public LogView(const Font& font, const Size& size) : this(font, Color(230u, 231u, 232u), Point(), size, Dock.bottom, Anchors.none)
25         {
26         }
27         public LogView(const Size& size) : this(Font(FontFamily.GenericSansSerif(), 9.0f), size)
28         {
29         }
30         public void Clear()
31         {
32             lines.Clear();
33             Invalidate();
34         }
35         public virtual void WriteLine(const string& line)
36         {
37             lines.Add(line);
38             Invalidate();
39         }
40         public override void OnPaint(PaintEventArgs& args)
41         {
42             base->OnPaint(args);
43             if (Changed())
44             {
45                 ResetChanged();
46                 textHeight = cast<int>(font.GetHeight(args.graphics));
47             }
48             args.graphics.Clear(BackgroundColor());
49             long linesCount = lines.Count();
50             long numLines = Min(linesCount, GetSize().h / textHeight);
51             PointF origin;
52             for (long i = linesCount - numLines; i < linesCount; ++i;)
53             {
54                 const string& line = lines[i];
55                 args.graphics.DrawStringChecked(line, font, origin, textBrush);
56                 origin.y = origin.y + textHeight;
57             }
58         }
59         private inline nothrow bool Changed() const
60         {
61             return (flags & Flags.changed) != Flags.none;
62         }
63         private nothrow void SetChanged()
64         {
65             flags = cast<Flags>(flags | Flags.changed);
66         }
67         private nothrow void ResetChanged()
68         {
69             flags = cast<Flags>(flags & ~Flags.changed);
70         }
71         private Font font;
72         private SolidBrush textBrush;
73         private List<string> lines;
74         private int textHeight;
75         private Flags flags;
76     }
77 }
