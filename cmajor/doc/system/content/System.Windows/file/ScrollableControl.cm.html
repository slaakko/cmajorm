 12     public class ScrollableControl : Control

 14         public ScrollableControl(Control* child_, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 15             base("System.Windows.ScrollableControl", DefaultWindowClassStyle(), DefaultChildWindowStyle(), DefaultExtendedWindowStyle(), 
 16             child_->BackgroundColor(), "scrollableControl", location, size, dock, anchors), container(this), child(child_), 
 17             verticalScrollUnit(0), horizontalScrollUnit(0), vpage(0u), vpos(0), vmin(0), vmax(0), hpage(0u), hpos(0), hmin(0), hmax(0), verticalScrollBarShown(false), horizontalScrollBarShown(false)
 18         {
 19             container.AddChild(child);
 20         }

 37         protected override void OnChildContentChanged(ControlEventArgs& args)
 38         {
 39             base->OnChildContentChanged(args);
 40             hpos = 0;
 41             vpos = 0;
 42         }
 43         protected override void OnChildContentLocationChanged(ControlEventArgs& args)
 44         {
 45             base->OnChildContentLocationChanged(args);
 46             scrolledChild = args.control;
 47             Point childContentLocation = scrolledChild->ContentLocation();
 48             Pair<int, int> scrollUnits = scrolledChild->GetScrollUnits();
 49             int verticalScrollUnit = scrollUnits.first;
 50             int horizontalScrollUnit = scrollUnits.second;
 51             vpos = childContentLocation.y / verticalScrollUnit;
 52             SetScrollInfo(Handle(), ScrollBar.SB_VERT, ScrollInfoMask.SIF_POS, true, 0u, vpos, 0, 0);
 53             hpos = childContentLocation.x / horizontalScrollUnit;
 54             SetScrollInfo(Handle(), ScrollBar.SB_HORZ, ScrollInfoMask.SIF_POS, true, 0u, hpos, 0, 0);
 55         }
 56         protected override void OnChildContentSizeChanged(ControlEventArgs& args)
 57         {
 58             base->OnChildContentSizeChanged(args);
 59             scrolledChild = args.control;
 60             Pair<int, int> scrollUnits = scrolledChild->GetScrollUnits();
 61             verticalScrollUnit = scrollUnits.first;
 62             horizontalScrollUnit = scrollUnits.second;
 63             Size scrolledChildClientSize = scrolledChild->GetSize();
 64             Size scrolledChildContentSize = scrolledChild->ContentSize();
 65             if (scrolledChildContentSize.h > scrolledChildClientSize.h)
 66             {
 67                 vmin = 0;
 68                 vmax = scrolledChildContentSize.h / verticalScrollUnit;
 69                 vpage = cast<uint>(scrolledChildClientSize.h / verticalScrollUnit);
 70                 SetScrollInfo(Handle(), ScrollBar.SB_VERT, cast<ScrollInfoMask>(ScrollInfoMask.SIF_POS | ScrollInfoMask.SIF_PAGE | ScrollInfoMask.SIF_RANGE), true, 
 71                     vpage, vpos, vmin, vmax);
 72                 ShowScrollBar(Handle(), ScrollBar.SB_VERT, true);
 73                 verticalScrollBarShown = true;
 74             }
 75             else
 76             {
 77                 ShowScrollBar(Handle(), ScrollBar.SB_VERT, false);
 78                 verticalScrollBarShown = false;
 79             }
 80             if (scrolledChildContentSize.w > scrolledChildClientSize.w)
 81             {
 82                 hmin = 0;
 83                 hmax = scrolledChildContentSize.w / horizontalScrollUnit;
 84                 hpage = cast<uint>(scrolledChildClientSize.w / horizontalScrollUnit);
 85                 SetScrollInfo(Handle(), ScrollBar.SB_HORZ, cast<ScrollInfoMask>(ScrollInfoMask.SIF_POS | ScrollInfoMask.SIF_PAGE | ScrollInfoMask.SIF_RANGE), true, 
 86                     hpage, hpos, hmin, hmax);
 87                 ShowScrollBar(Handle(), ScrollBar.SB_HORZ, true);
 88                 horizontalScrollBarShown = true;
 89             }
 90             else
 91             {
 92                 ShowScrollBar(Handle(), ScrollBar.SB_HORZ, false);
 93                 horizontalScrollBarShown = false;
 94             }
 95         }

114         protected override void OnHScroll(ScrollEventArgs& args)
115         {
116             base->OnHScroll(args);
117             int trackPos;
118             GetScrollInfo(Handle(), ScrollBar.SB_HORZ, hpage, hpos, hmin, hmax, trackPos);
119             int prevHPos = hpos;
120             switch (args.request)
121             {
122                 case SB_LINELEFT:
123                 {
124                     --hpos;
125                     break;
126                 }
127                 case SB_LINERIGHT:
128                 {
129                     ++hpos;
130                     break;
131                 }
132                 case SB_PAGELEFT:
133                 {
134                     hpos = hpos - cast<int>(hpage);
135                     break;
136                 }
137                 case SB_PAGERIGHT:
138                 {
139                     hpos = hpos + cast<int>(hpage);
140                     break;
141                 }
142                 case SB_THUMBTRACK:
143                 {
144                     hpos = trackPos;
145                     break;
146                 }
147             }
148             SetScrollInfo(Handle(), ScrollBar.SB_HORZ, ScrollInfoMask.SIF_POS, true, 0u, hpos, 0, 0);
149             GetScrollInfo(Handle(), ScrollBar.SB_HORZ, hpage, hpos, hmin, hmax, trackPos);
150             if (prevHPos != hpos)
151             {
152                 scrolledChild->SetContentLocationInternal(Point(horizontalScrollUnit * hpos, verticalScrollUnit * vpos));
153                 ScrollWindow(scrolledChild->Handle(), horizontalScrollUnit * (prevHPos - hpos), 0, null, null);
154                 scrolledChild->Update();
155             }
156         }
157         public override void ScrollLineDown()
158         {
159             int trackPos;
160             GetScrollInfo(Handle(), ScrollBar.SB_VERT, vpage, vpos, vmin, vmax, trackPos);
161             int prevVPos = vpos;
162             if (vpos != vmax)
163             {
164                 ++vpos;
165                 SetScrollInfo(Handle(), ScrollBar.SB_VERT, ScrollInfoMask.SIF_POS, true, 0u, vpos, 0, 0);
166                 GetScrollInfo(Handle(), ScrollBar.SB_VERT, vpage, vpos, vmin, vmax, trackPos);
167                 if (prevVPos != vpos)
168                 {
169                     scrolledChild->SetContentLocationInternal(Point(horizontalScrollUnit * hpos, verticalScrollUnit * vpos));
170                     ScrollWindow(scrolledChild->Handle(), 0, verticalScrollUnit * (prevVPos - vpos), null, null);
171                     scrolledChild->Update();
172                 }
173             }
174         }

181             {
182                 --vpos;
183                 SetScrollInfo(Handle(), ScrollBar.SB_VERT, ScrollInfoMask.SIF_POS, true, 0u, vpos, 0, 0);
184                 GetScrollInfo(Handle(), ScrollBar.SB_VERT, vpage, vpos, vmin, vmax, trackPos);
185                 if (prevVPos != vpos)
186                 {
187                     scrolledChild->SetContentLocationInternal(Point(horizontalScrollUnit * hpos, verticalScrollUnit * vpos));

246         protected override void OnMouseWheel(MouseWheelEventArgs& args)
247         {
248             base->OnMouseWheel(args);
249             if (!args.handled)
250             {
251                 if (verticalScrollBarShown)
252                 {
253                     int trackPos;
254                     GetScrollInfo(Handle(), ScrollBar.SB_VERT, vpage, vpos, vmin, vmax, trackPos);
255                     int prevVPos = vpos;
256                     vpos = cast<int>(vpos - args.distance / (2.0 * verticalScrollUnit));
257                     SetScrollInfo(Handle(), ScrollBar.SB_VERT, ScrollInfoMask.SIF_POS, true, 0u, vpos, 0, 0);
258                     GetScrollInfo(Handle(), ScrollBar.SB_VERT, vpage, vpos, vmin, vmax, trackPos);
259                     if (prevVPos != vpos)
260                     {
261                         scrolledChild->SetContentLocationInternal(Point(horizontalScrollUnit * hpos, verticalScrollUnit * vpos));
262                         ScrollWindow(scrolledChild->Handle(), 0, verticalScrollUnit * (prevVPos - vpos), null, null);
263                         scrolledChild->Update();
264                     }
265                     args.handled = true;
266                 }
267             }
268         }
269         protected override void TranslateChildGraphics(Graphics& graphics)
270         {
271             int dx = -hpos * horizontalScrollUnit;
272             int dy = -vpos * verticalScrollUnit;
273             if (dx != 0 || dy != 0)
274             {
275                 graphics.TranslateTransformChecked(dx, dy);
276             }
277         }
278         protected override void TranslateMousePos(Point& location)
279         {
280             int dx = hpos * horizontalScrollUnit;
281             int dy = vpos * verticalScrollUnit;
282             location.x = location.x + dx;
283             location.y = location.y + dy;
284         }
285         protected override void TranslateContentLocation(Point& location)
286         {
287             int dx = hpos * horizontalScrollUnit;
288             int dy = vpos * verticalScrollUnit;
289             location.x = location.x - dx;
290             location.y = location.y - dy;
291         }
