 25     public nothrow int GetBorderWidth(BorderStyle borderStyle)
 26     {

 41         public BorderedControl(Control* child_, BorderStyle borderStyle_, const Color& normalBorderColor_, const Color& focusedBorderColor_, 
 42             const Point& location, const Size& size, Dock dock, Anchors anchors) :

150         public override void PrintWindowTree(int level)
151         {
152             LogView* log = Application.GetLogView();
153             if (log != null)
154             {
155                 log->WriteLine(string(' ', level) + "BorderedControl." + Text() + ".handle=" + ToHexString(cast<ulong>(Handle())) + " " + ParentText() + "[" + Rect(Location(), GetSize()).ToString() + "]");
156             }
157             Component* child = container.FirstChild();
158             while (child != null)
159             {
160                 if (child is Control*)
161                 {
162                     Control* childControl = cast<Control*>(child);
163                     childControl->PrintWindowTree(level + 1);
164                 }
165                 child = child->NextSibling();
166             }
167         }

178         protected override void OnPaint(PaintEventArgs& args)
179         {
180             if (Debug.Paint())
181             {

183                 LogView* log = Application.GetLogView();
184                 if (log != null)
185                 {
186                     log->WriteLine("BorderedControl.OnPaint: " + r.ToString());
187                 }
188             }
189             DrawBorder(args.graphics);
190             base->OnPaint(args);
191         }
192         private void SetChildPos()
193         {
194             Point loc;
195             Size size = GetSize();
196             Rect childRect(loc, size);
197             childRect.Inflate(-borderWidth, -borderWidth);
198             child->SetLocation(childRect.location);
199             child->SetSize(childRect.size);
200         }
201         private void DrawBorder(Graphics& graphics)
202         {
203             switch (borderStyle)
204             {
205                 case BorderStyle.single: DrawSingleBorder(graphics); break;
206                 case BorderStyle.style3D: Draw3DBorder(graphics); break;
207             }
208         }
209         private void DrawSingleBorder(Graphics& graphics)
210         {
211             Rect r(Point(), GetSize());
212             r.size.w = r.size.w - 1;
213             r.size.h = r.size.h - 1;
214             if (ChildFocused())
215             {
216                 Pen pen(focusedBorderColor, 1);
217                 graphics.DrawRectangleChecked(pen, r);
218             }
219             else
220             {
221                 Pen pen(normalBorderColor, 1);
222                 graphics.DrawRectangleChecked(pen, r);
223             }
224         }
225         private void Draw3DBorder(Graphics& graphics)
226         {
227             Rect r(Point(), GetSize());
228             r.size.w = r.size.w - 1;
229             r.size.h = r.size.h - 1;
230             Pen outerLeftTopEdgePen(Color(160u, 160u, 160u), 1);
231             graphics.DrawLineChecked(outerLeftTopEdgePen, r.location, Point(r.location.x + r.size.w - 1, r.location.y));
232             graphics.DrawLineChecked(outerLeftTopEdgePen, r.location, Point(r.location.x, r.location.y + r.size.h - 1));
233             Pen outerRightBottomEdgePen(Color.White(), 1);
234             graphics.DrawLineChecked(outerRightBottomEdgePen, Point(r.location.x + r.size.w, r.location.y), Point(r.location.x + r.size.w, r.location.y + r.size.h));
235             graphics.DrawLineChecked(outerRightBottomEdgePen, Point(r.location.x, r.location.y + r.size.h), Point(r.location.x + r.size.w, r.location.y + r.size.h));
236             r.Inflate(-1, -1);
237             Pen innerLeftTopEdgePen(Color(105u, 105u, 105u), 1);
238             graphics.DrawLineChecked(innerLeftTopEdgePen, r.location, Point(r.location.x + r.size.w - 1, r.location.y));
239             graphics.DrawLineChecked(innerLeftTopEdgePen, r.location, Point(r.location.x, r.location.y + r.size.h - 1));
240             Pen innerRightBottomEdgePen(Color(227u, 227u, 227u), 1);
241             graphics.DrawLineChecked(innerRightBottomEdgePen, Point(r.location.x + r.size.w, r.location.y), Point(r.location.x + r.size.w, r.location.y + r.size.h));
242             graphics.DrawLineChecked(innerRightBottomEdgePen, Point(r.location.x, r.location.y + r.size.h), Point(r.location.x + r.size.w, r.location.y + r.size.h));
243         }
244         private inline nothrow bool ChildFocused() const
245         {
246             return (flags & Flags.childFocused) != Flags.none;
247         }
248         private inline nothrow void SetChildFocused()
249         {
250             flags = cast<Flags>(flags | Flags.childFocused);
251         }
252         private inline nothrow void ResetChildFocused()
253         {
254             flags = cast<Flags>(flags & ~Flags.childFocused);
255         }
