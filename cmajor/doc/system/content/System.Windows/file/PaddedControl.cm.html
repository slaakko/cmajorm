  6 using System;
  7 using System.Collections;
  8 
  9 namespace System.Windows
 10 {
 11     public inline nothrow int DefaultControlPadding()
 12     {
 13         return 3;
 14     }
 15 
 16     public class PaddedControl : Control
 17     {
 18         public PaddedControl(Control* child_, const Padding& padding_, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 19             base("System.Windows.PaddedControl", DefaultWindowClassStyle(), DefaultChildWindowStyle(), DefaultExtendedWindowStyle(), 
 20             child_->BackgroundColor(), "paddedControl", location, size, dock, anchors), container(this), child(child_), padding(padding_)
 21         {
 22             container.AddChild(child);
 23             SetChildPos();
 24         }
 25         public PaddedControl(Control* child) : this(child, Padding(DefaultControlPadding(), DefaultControlPadding(), DefaultControlPadding(), DefaultControlPadding()), 
 26             Point(), Size(), Dock.none, Anchors.none)
 27         {
 28         }
 29         protected override nothrow bool IsDecoratorControl() const
 30         {
 31             return true;
 32         }
 33         public override nothrow Padding DefaultPadding() const
 34         {
 35             return padding;
 36         }
 37         protected override void OnLocationChanged()
 38         {
 39             base->OnLocationChanged();
 40             SetChildPos();
 41         }
 42         protected override void OnSizeChanged()
 43         {
 44             base->OnSizeChanged();
 45             SetChildPos();
 46         }
 47         protected override void OnChildContentChanged(ControlEventArgs& args)
 48         {
 49             base->OnChildContentChanged(args);
 50             Control* parentControl = ParentControl();
 51             if (parentControl != null)
 52             {
 53                 parentControl->OnChildContentChanged(args);
 54             }
 55         }
 56         protected override void OnChildContentLocationChanged(ControlEventArgs& args)
 57         {
 58             base->OnChildContentLocationChanged(args);
 59             Control* parentControl = ParentControl();
 60             if (parentControl != null)
 61             {
 62                 parentControl->OnChildContentLocationChanged(args);
 63             }
 64         }
 65         protected override void OnChildContentSizeChanged(ControlEventArgs& args)
 66         {
 67             base->OnChildContentSizeChanged(args);
 68             Control* parentControl = ParentControl();
 69             if (parentControl != null)
 70             {
 71                 parentControl->OnChildContentSizeChanged(args);
 72             }
 73         }
 74         protected override void OnChildGotFocus(ControlEventArgs& args)
 75         {
 76             base->OnChildGotFocus(args);
 77             Control* parentControl = ParentControl();
 78             if (parentControl != null)
 79             {
 80                 parentControl->OnChildGotFocus(args);
 81             }
 82         }
 83         protected override void OnChildLostFocus(ControlEventArgs& args)
 84         {
 85             base->OnChildLostFocus(args);
 86             Control* parentControl = ParentControl();
 87             if (parentControl != null)
 88             {
 89                 parentControl->OnChildLostFocus(args);
 90             }
 91         }
 92         protected override void OnPaint(PaintEventArgs& args)
 93         {
 94             args.graphics.Clear(BackgroundColor());
 95             base->OnPaint(args);
 96         }
 97         internal override nothrow Control* GetFirstEnabledTabStopControl() const
 98         {
 99             return child->GetFirstEnabledTabStopControl();
100         }
101         internal override nothrow Control* GetLastEnabledTabStopControl() const
102         {
103             return child->GetLastEnabledTabStopControl();
104         }
105         private void SetChildPos()
106         {
107             Point loc;
108             Size size = GetSize();
109             Rect childRect(loc, size);
110             childRect.location.x = loc.x + padding.left;
111             childRect.location.y = loc.x + padding.top;
112             childRect.size.w = size.w - padding.Horizontal();
113             childRect.size.h = size.h - padding.Vertical();
114             child->SetLocation(childRect.location);
115             child->SetSize(childRect.size);
116         }
117         private Container container;
118         private Control* child;
119         private Padding padding;
120     }
121 }
