 431         private void Measure(Graphics& graphics)
 432         {
 433             string s = "RootNode";
 434             RectF textRect = graphics.MeasureStringChecked(s, font, PointF(0, 0), format);

 439         private void MakeStateIndicatorBitmaps(Graphics& graphics)
 440         {
 441             SolidBrush stateIndicatorBrush(stateIndicatorColor);
 442             stateIndicatorBitmaps.Clear();
 443             PointF center(textHeight / 2, textHeight / 2);
 444             float w = cast<float>((Sqrt(3.0) / 2) * stateIndicatorHeight);
 445             PointF pt1(center.x - w / 2, center.y - stateIndicatorHeight / 2);
 446             PointF pt2(center.x - w / 2, center.y + stateIndicatorHeight / 2);
 447             PointF pt3(center.x + w / 2, center.y);
 448             for (int i = 9; i >= 0; --i;)
 449             {
 450                 float angle = i * 10.0f;
 451                 List<PointF> triangle;
 452                 triangle.Add(pt1);
 453                 triangle.Add(pt2);
 454                 triangle.Add(pt3);
 455                 Bitmap bm(cast<int>(textHeight), cast<int>(textHeight), graphics);
 456                 Graphics bmGraphics = Graphics.FromImage(bm);
 457                 bmGraphics.Clear(BackgroundColor());
 458                 if (i > 0)
 459                 {
 460                     Matrix id;
 461                     Matrix rm = RotateAt(id, angle, center);
 462                     rm.TransformPointsChecked(triangle);
 463                 }
 464                 List<Point> indicatorTriangle;
 465                 for (const PointF& pt : triangle)
 466                 {
 467                     indicatorTriangle.Add(Point(cast<int>(pt.x), cast<int>(pt.y)));
 468                 }
 469                 bmGraphics.SetSmoothingModeChecked(SmoothingMode.highQuality);
 470                 bmGraphics.FillPolygonChecked(stateIndicatorBrush, 3, indicatorTriangle.Begin().Ptr());
 471                 stateIndicatorBitmaps.Add(Rvalue(bm));
 472             }
