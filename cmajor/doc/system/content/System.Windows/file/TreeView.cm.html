 508         private void MakeStateIndicatorBitmaps(Graphics& graphics)
 509         {
 510             SolidBrush stateIndicatorBrush(stateIndicatorColor);
 511             stateIndicatorBitmaps.Clear();
 512             PointF center(textHeight / 2, textHeight / 2);
 513             float w = cast<float>((Sqrt(3.0) / 2) * stateIndicatorHeight);
 514             PointF pt1(center.x - w / 2, center.y - stateIndicatorHeight / 2);
 515             PointF pt2(center.x - w / 2, center.y + stateIndicatorHeight / 2);
 516             PointF pt3(center.x + w / 2, center.y);
 517             for (int i = 9; i >= 0; --i;)
 518             {
 519                 float angle = i * 10.0f;
 520                 List<PointF> triangle;
 521                 triangle.Add(pt1);
 522                 triangle.Add(pt2);
 523                 triangle.Add(pt3);
 524                 Bitmap bm(cast<int>(textHeight), cast<int>(textHeight), graphics);
 525                 Graphics bmGraphics = Graphics.FromImage(bm);
 526                 bmGraphics.Clear(BackgroundColor());
 527                 if (i > 0)
 528                 {
 529                     Matrix id;
 530                     Matrix rm = RotateAt(id, angle, center);
 531                     rm.TransformPointsChecked(triangle);
 532                 }
 533                 List<Point> indicatorTriangle;
 534                 for (const PointF& pt : triangle)
 535                 {
 536                     indicatorTriangle.Add(Point(cast<int>(pt.x), cast<int>(pt.y)));
 537                 }
 538                 bmGraphics.SetSmoothingModeChecked(SmoothingMode.highQuality);
 539                 bmGraphics.FillPolygonChecked(stateIndicatorBrush, 3, indicatorTriangle.Begin().Ptr());
 540                 stateIndicatorBitmaps.Add(Rvalue(bm));
 541             }
