 33     public class ListBox : Control

 35         public ListBox(ListBoxStyle listBoxStyle, const Color& backgroundColor, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) :

 37             backgroundColor, text, location, size, dock, anchors)
 38         {
 39         }
 40         public ListBox(const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 41             this(cast<ListBoxStyle>(ListBoxStyle.LBS_NOTIFY | ListBoxStyle.LBS_HASSTRINGS | WindowStyle.WS_VSCROLL), Color.White(), string(), location, size, dock, anchors)
 42         {
 43         }
 44         public void AddItem(const string& item)
 45         {
 46             if (Handle() != null)
 47             {
 48                 wstring wstr(ToUtf16(item));
 49                 WinSendMessage(Handle(), LB_ADDSTRING, 0u, cast<long>(cast<ulong>(cast<void*>(wstr.Chars()))));
 50             }
 51         }
 52         public void DeleteItem(int itemIndex)
 53         {
 54             if (Handle() != null)
 55             {
 56                 WinSendMessage(Handle(), LB_DELETESTRING, cast<uint>(itemIndex), 0);
 57             }
 58         }
 59         public void InsertItem(int index, const string& item)
 60         {
 61             if (Handle() != null)
 62             {
 63                 wstring wstr(ToUtf16(item));
 64                 WinSendMessage(Handle(), LB_INSERTSTRING, cast<uint>(index), cast<long>(cast<ulong>(cast<void*>(wstr.Chars()))));
 65             }
 66         }
 67         public string GetItem(int index)
 68         {
 69             if (Handle() != null)
 70             {
 71                 int size = cast<int>(WinSendMessage(Handle(), LB_GETTEXTLEN, cast<uint>(index), 0));
 72                 UniquePtr<wchar> buf(cast<wchar*>(MemAlloc(size * sizeof(wchar))));
 73                 WinSendMessage(Handle(), LB_GETTEXT, cast<uint>(index), cast<long>(cast<ulong>(cast<void*>(buf.Get()))));
 74                 return ToUtf8(buf.Get());
 75             }
 76             return string();
 77         }
 78         public void SetItem(int index, const string& item)
 79         {
 80             DeleteItem(index);
 81             InsertItem(index, item);
 82         }

130         protected override void OnCreated()
131         {
132             base->OnCreated();
133             Graphics graphics = Graphics.FromWindowHandle(Handle());
134             const FontHandle& fontHandle = GetFontHandle(graphics);
135             if (!fontHandle.IsNull())
136             {
137                 SendSetFontMessage(fontHandle);
138             }
