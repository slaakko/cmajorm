 59         public void Write(int handle, const string& text)
 60         {
 61             if (text.IsEmpty()) return;
 62             ColorCharOutputMethod outputCharMethod = OutputChar;
 63             AnsiProcess(handle, text, outputCharMethod);
 64             const List<ustring>& lines = Lines();
 65             if (!lines.IsEmpty())
 66             {
 67                 int line = cast<int>(lines.Count());
 68                 int col = cast<int>(lines.Back().Length() + 1);
 69                 SetCaretLineCol(line, col);
 70                 ScrollToCaret();
 71                 Invalidate();
 72             }
 73         }

 80         protected override void OnKeyDown(KeyEventArgs& args)
 81         {
 82             int caretCol = CaretColumn();
 83             int caretLine = CaretLine();
 84             switch (args.keyData)
 85             {
 86                 case cast<Keys>(Keys.controlModifier | Keys.z):
 87                 {
 88                     eof = true;
 89                     Write(1, "^Z\n");
 90                     OnConsoleInputReady();
 91                     args.handled = true;
 92                     break;
 93                 }
 94                 case Keys.enter:
 95                 {
 96                     eof = false;
 97                     Write(1, "\n");
 98                     OnConsoleInputReady();
 99                     args.handled = true;
100                     break;
101                 }
102                 case Keys.delete_:
103                 {
104                     int index = caretCol - startInputCol;
105                     if (index < inputLine.Length())
106                     {
107                         DeleteChar();
108                     }
109                     args.handled = true;
110                     break;
111                 }
112                 case Keys.back:
113                 {
114                     if (caretCol > startInputCol)
115                     {
116                         --caretCol;
117                         SetCaretLineCol(caretLine, caretCol);
118                         DeleteChar();
119                     }
120                     args.handled = true;
121                     break;
122                 }
123                 case Keys.left:
124                 {
125                     if (caretCol > startInputCol)
126                     {
127                         --caretCol;
128                         SetCaretLineCol(caretLine, caretCol);
129                         Invalidate();
130                     }
131                     args.handled = true;
132                     break;
133                 }
134                 case Keys.right:
135                 {
136                     int index = caretCol - startInputCol;
137                     if (index < inputLine.Length())
138                     {
139                         ++caretCol;
140                         SetCaretLineCol(caretLine, caretCol);
141                         Invalidate();
142                     }
143                     args.handled = true;
144                     break;
145                 }
146                 case Keys.home:
147                 {
148                     caretCol = startInputCol;
149                     SetCaretLineCol(caretLine, caretCol);
150                     Invalidate();
151                     args.handled = true;
152                     break;
153                 }
154                 case Keys.end:
155                 {
156                     caretCol = startInputCol + cast<int>(inputLine.Length());
157                     SetCaretLineCol(caretLine, caretCol);
158                     Invalidate();
159                     args.handled = true;
160                     break;
161                 }
162             }
163         }
164         protected override void OnKeyPress(KeyPressEventArgs& keyPressEventArgs)
165         {
166             if (eof) return;
167             base->OnKeyPress(keyPressEventArgs);
168             if (!keyPressEventArgs.handled)
169             {
170                 uchar ch = keyPressEventArgs.keyChar;
171                 InsertChar(ch);
172                 keyPressEventArgs.handled = true;
173             }

175         private nothrow void InsertChar(uchar ch)
176         {
177             int caretCol = CaretColumn();
178             int caretLine = CaretLine();
179             int index = caretCol - startInputCol;
180             if (index < inputLine.Length())
181             {
182                 inputLine = inputLine.Substring(0, index) + ustring(ch) + inputLine.Substring(index);
183             }
184             else
185             {
186                 inputLine.Append(ch);
187             }
188             List<ustring>& lines = Lines();
189             ustring line = lines[caretLine - 1];
190             if (caretCol < line.Length())
191             {
192                 line = line.Substring(0, caretCol - 1) + ustring(ch) + line.Substring(caretCol - 1);
193             }
194             else
195             {
196                 line.Append(ch);
197             }
198             lines[caretLine - 1] = line;
199             IncrementCaretColorCount();
200             ++caretCol;
201             SetCaretLineCol(caretLine, caretCol);
202             Invalidate();
203         }
204         private nothrow void DeleteChar()
205         {
206             int caretCol = CaretColumn();
207             int caretLine = CaretLine();
208             int index = caretCol - startInputCol;
209             inputLine = inputLine.Substring(0, index) + inputLine.Substring(index + 1);
210             List<ustring>& lines = Lines();
211             ustring line = lines[caretLine - 1];
212             line = line.Substring(0, caretCol - 1) + line.Substring(caretCol);
213             lines[caretLine - 1] = line;
214             DecrementCaretColorCount();
215             Invalidate();
216         }

223         protected override void PaintContent(Graphics& graphics, const Rect& clipRect)
224         {
225             TextRenderingHint prevRenderingHint = graphics.GetTextRenderingHint();
226             graphics.SetTextRenderingHintChecked(TextRenderingHint.clearTypeGridFit);
227             if (Changed())
228             {
229                 ResetChanged();
230                 SetMaxLineLength();
231                 Measure(graphics);
232             }
233             graphics.ClearChecked(BackgroundColor());
234             const List<ustring>& lines = Lines();
235             int n = cast<int>(lines.Count());
236             Padding padding = GetPadding();
237             PointF origin(padding.left, padding.top);
238             for (int i = 0; i < n; ++i;)
239             {
240                 if (IsLinePartiallyVisible(i + 1))
241                 {
242                     List<ColorCount>& backColorLine = backColorLines[i];
243                     PaintLineBackground(graphics, backColorLine, origin);
244                     List<ColorCount>& textColorLine = textColorLines[i];
245                     const ustring& line = lines[i];
246                     DrawLineText(graphics, line, textColorLine, origin);
247                 }
248                 origin.y = origin.y + CharHeight();
249             }
250             graphics.SetTextRenderingHintChecked(prevRenderingHint);
251         }
252         private void PaintLineBackground(Graphics& graphics, const List<ColorCount>& backColorLine, const PointF& origin)
253         {
254             int n = cast<int>(backColorLine.Count());
255             if (n == 1 && backColorLine[0].color == defaultBackColor) return;
256             PointF loc = origin;
257             for (int i = 0; i < n; ++i;)
258             {
259                 const ColorCount& colorCount = backColorLine[i];
260                 SizeF size(colorCount.count * CharWidth(), CharHeight());
261                 if (colorCount.color != defaultBackColor)
262                 {
263                     SolidBrush* brush = GetOrInsertBrush(colorCount.color);
264                     RectF rect(loc, size);
265                     graphics.FillRectangleChecked(*brush, rect);
266                 }
267                 loc.x = loc.x + size.w;
268             }
269         }
270         private void DrawLineText(Graphics& graphics, const ustring& line, const List<ColorCount>& textColorLine, const PointF& origin)
271         {
272             PointF loc = origin;
273             long start = 0;
274             int n = cast<int>(textColorLine.Count());
275             for (int i = 0; i < n; ++i;)
276             {
277                 const ColorCount& colorCount = textColorLine[i];
278                 SolidBrush* brush = GetOrInsertBrush(colorCount.color);
279                 long length = colorCount.count;
280                 ustring s = line.Substring(start, length);
281                 graphics.DrawStringChecked(ToUtf8(s), *Fonts()[0], loc, *brush);
282                 loc.x = loc.x + length * CharWidth();
283                 start = start + length;
284             }
285         }
286         private nothrow void IncrementCaretColorCount()
287         {
288             int caretCol = CaretColumn();
289             int caretLine = CaretLine();
290             List<ColorCount>& textColorLine = textColorLines[caretLine - 1];
291             IncrementColorCount(textColorLine, caretCol);
292             List<ColorCount>& backColorLine = backColorLines[caretLine - 1];
293             IncrementColorCount(backColorLine, caretCol);
294         }
295         private nothrow void IncrementColorCount(List<ColorCount>& colorLine, int caretCol)
296         {
297             int count = 0;
298             bool incremented = false;
299             for (ColorCount& colorCount : colorLine)
300             {
301                 if (caretCol >= count && caretCol < count + colorCount.count)
302                 {
303                     colorCount.IncrementCount();
304                     incremented = true;
305                     break;
306                 }
307                 count = count + colorCount.count;
308             }
309             if (!incremented)
310             {
311                 colorLine.Back().IncrementCount();
312             }
313         }
314         private nothrow void DecrementCaretColorCount()
315         {
316             int caretCol = CaretColumn();
317             int caretLine = CaretLine();
318             List<ColorCount>& textColorLine = textColorLines[caretLine - 1];
319             DecrementColorCount(textColorLine, caretCol);
320             List<ColorCount>& backColorLine = backColorLines[caretLine - 1];
321             DecrementColorCount(backColorLine, caretCol);
322         }
323         private nothrow void DecrementColorCount(List<ColorCount>& colorLine, int caretCol)
324         {
325             int count = 0;
326             bool decremented = false;
327             for (ColorCount& colorCount : colorLine)
328             {
329                 if (caretCol >= count && caretCol < count + colorCount.count)
330                 {
331                     colorCount.DecrementCount();
332                     decremented = true;
333                     break;
334                 }
335                 count = count + colorCount.count;
336             }
337             if (!decremented)
338             {
339                 colorLine.Back().DecrementCount();
340             }
341         }
342         private void OutputChar(System.Color.Constant textColor, System.Color.Constant backColor, int handle, uchar c)
343         {
344             List<ustring>& lines = Lines();
345             if (lines.IsEmpty() || c == '\n')
346             {
347                 ustring line;
348                 AddLine(line);
349                 List<ColorCount> colorLine;
350                 textColorLines.Add(colorLine);
351                 backColorLines.Add(colorLine);
352             }
353             if (c != '\n')
354             {
355                 ustring& lastLine = lines.Back();
356                 lastLine.Append(c);
357                 AddColor(textColor, textColorLines.Back());
358                 AddColor(backColor, backColorLines.Back());
359                 if (lastLine.Length() > MaxLineLength())
360                 {
361                     SetMaxLineLength();
362                 }
363             }
364         }
365         private nothrow void AddColor(System.Color.Constant color, List<ColorCount>& colorLine)
366         {
367             if (colorLine.IsEmpty() || color != colorLine.Back().color)
368             {
369                 ColorCount colorCount(color, 1);
370                 colorLine.Add(colorCount);
371             }
372             else
373             {
374                 colorLine.Back().IncrementCount();
375             }
376         }
377         private SolidBrush* GetOrInsertBrush(System.Color.Constant color)
378         {
379             HashMap<sbyte, SolidBrush*>.ConstIterator it = brushMap.CFind(cast<sbyte>(color));
380             if (it != brushMap.CEnd())
381             {
382                 return it->second;
383             }
384             else
385             {
386                 SolidBrush* solidBrush = new SolidBrush(GetColor(color));
387                 brushes.Add(UniquePtr<SolidBrush>(solidBrush));
388                 brushMap[cast<sbyte>(color)] = solidBrush;
389                 return solidBrush;
390             }
391         }
