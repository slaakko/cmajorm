 25     public class MemoryHandle
 26     {
 27         public nothrow MemoryHandle() : handle(null)
 28         {
 29         }
 30         public MemoryHandle(void* handle_) : handle(handle_), memory(GlobalLock(handle))
 31         {
 32         }
 33         suppress MemoryHandle(const MemoryHandle&);
 34         suppress void operator=(const MemoryHandle&);
 35         public MemoryHandle(MemoryHandle&& that) : handle(that.handle)
 36         {
 37             that.handle = null;
 38         }
 39         public default nothrow void operator=(MemoryHandle&&);
 40         public ~MemoryHandle()
 41         {
 42             if (handle != null)
 43             {
 44                 GlobalUnlock(handle);
 45             }
 46         }
 47         public static MemoryHandle AllocateMemory(ulong size)
 48         {
 49             return MemoryHandle(GlobalAlloc(GlobalAllocFlags.GHND, size));
 50         }
 51         public nothrow void* Memory() const
 52         {
 53             return memory;
 54         }
 55         public ulong Size() const
 56         {
 57             return GlobalSize(handle);
 58         }
 59         public nothrow void* Handle() const
 60         {
 61             return handle;
 62         }
 63         private void* handle;
 64         private void* memory;
 65     }

104         public MemoryHandle GetData(const ClipboardFormat& clipboardFormat)
105         {
106             return MemoryHandle(GetClipboardData(clipboardFormat.Number()));
107         }
108         public void SetData(const ClipboardFormat& clipboardFormat, const MemoryHandle& memoryHandle)
109         {
110             SetEmpty();
111             SetClipboardData(clipboardFormat.Number(), memoryHandle.Handle());
112         }
113         public ustring GetStringData(const ClipboardFormat& clipboardFormat)
114         {
115             MemoryHandle memoryHandle = GetData(clipboardFormat);
116             ulong size = memoryHandle.Size();
117             UniquePtr<uchar> ptr(cast<uchar*>(MemAlloc(cast<long>(size))));
118             StrCopy(ptr.Get(), cast<uchar*>(memoryHandle.Memory()));
119             ustring str(ptr.Get());
120             return str;
121         }
122         public void SetStringData(const ClipboardFormat& clipboardFormat, const ustring& str)
123         {
124             MemoryHandle memoryHandle = MemoryHandle.AllocateMemory(cast<ulong>((str.Length() + 1) * sizeof(uchar)));
125             StrCopy(cast<uchar*>(memoryHandle.Memory()), str.Chars());
126             SetData(clipboardFormat, memoryHandle);
127         }
128         private void* windowHandle;
129     }
