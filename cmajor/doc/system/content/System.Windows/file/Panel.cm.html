 6 using System;
 7 using System.Collections;
 8 
 9 namespace System.Windows
10 {
11     public class Panel : ContainerControl
12     {
13         public Panel(const Point& location, const Size& size, Dock dock, Anchors anchors) : 
14             base("System.Windows.Panel", DefaultWindowClassStyle(), DefaultChildWindowStyle(), DefaultExtendedWindowStyle(), DefaultControlBackgroundColor(), "panel", location, size, dock, anchors)
15         {
16         }
17         public override void PrintWindowTree(int level)
18         {
19             LogView* log = Application.GetLogView();
20             if (log != null)
21             {
22                 log->WriteLine(string(' ', level) + "Panel." + Text() + ".handle=" + ToHexString(cast<ulong>(Handle())) + " " + ParentText() + "[" + Rect(Point(), GetSize()).ToString() + "]");
23             }
24             Component* child = Children().FirstChild();
25             while (child != null)
26             {
27                 if (child is Control*)
28                 {
29                     Control* childControl = cast<Control*>(child);
30                     childControl->PrintWindowTree(level + 1);
31                 }
32                 child = child->NextSibling();
33             }
34         }
35         protected override void OnPaint(PaintEventArgs& args)
36         {
37             try
38             {
39                 if (Debug.Paint())
40                 {
41                     Rect r(Location(), GetSize());
42                     LogView* log = Application.GetLogView();
43                     if (log != null)
44                     {
45                         log->WriteLine("Panel.OnPaint: " + r.ToString());
46                     }
47                 }
48                 args.graphics.Clear(BackgroundColor());
49                 base->OnPaint(args);
50             }
51             catch (const Exception& ex)
52             {
53                 MessageBox.Show(ex.Message());
54             }
55         }
56     }
57 }
