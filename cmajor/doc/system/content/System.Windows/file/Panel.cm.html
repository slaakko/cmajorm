 6 using System;
 7 using System.Collections;
 8 
 9 namespace System.Windows
10 {
11     public class Panel : ContainerControl
12     {
13         private enum Flags : sbyte
14         {
15             none = 0, dontPaint = 1 << 0
16         }
17         public Panel(const char* windowClassName, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors, const Color& backgroundColor) : 
18             base(windowClassName, DefaultWindowClassStyle(), DefaultChildWindowStyle(), DefaultExtendedWindowStyle(), backgroundColor, text, location, size, dock, anchors)
19         {
20         }
21         public Panel(const Point& location, const Size& size, Dock dock, Anchors anchors) : 
22             this("System.Windows.Panel", "panel", location, size, dock, anchors, DefaultControlBackgroundColor())
23         {
24         }
25         public override void PrintWindowTree(int level)
26         {
27             LogView* log = Application.GetLogView();
28             if (log != null)
29             {
30                 log->WriteLine(string(' ', level) + "Panel." + Text() + ".handle=" + ToHexString(cast<ulong>(Handle())) + " " + ParentText() + "[" + Rect(Point(), GetSize()).ToString() + "]");
31             }
32             Component* child = Children().FirstChild();
33             while (child != null)
34             {
35                 if (child is Control*)
36                 {
37                     Control* childControl = cast<Control*>(child);
38                     childControl->PrintWindowTree(level + 1);
39                 }
40                 child = child->NextSibling();
41             }
42         }
43         protected override void OnPaint(PaintEventArgs& args)
44         {
45             try
46             {
47                 if (DontPaint()) return;
48                 if (Debug.Paint())
49                 {
50                     Rect r(Location(), GetSize());
51                     LogView* log = Application.GetLogView();
52                     if (log != null)
53                     {
54                         log->WriteLine("Panel.OnPaint: " + r.ToString());
55                     }
56                 }
57                 args.graphics.Clear(BackgroundColor());
58                 base->OnPaint(args);
59             }
60             catch (const Exception& ex)
61             {
62                 MessageBox.Show(ex.Message());
63             }
64         }
65         public inline nothrow bool DontPaint() const
66         {
67             return (flags & Flags.dontPaint) != Flags.none;
68         }
69         public nothrow void SetDontPaint()
70         {
71             flags = cast<Flags>(flags | Flags.dontPaint);
72         }
73         public nothrow void ResetDontPaint()
74         {
75             flags = cast<Flags>(flags & ~Flags.dontPaint);
76         }
77         private Flags flags;
78     }
79 }
