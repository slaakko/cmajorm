 51         internal override nothrow Control* GetFirstEnabledTabStopControl() const

 54             while (child != null)
 55             {
 56                 if (child is Control*)
 57                 {
 58                     Control* control = cast<Control*>(child);
 59                     Control* tabStopChild = control->GetFirstEnabledTabStopControl();
 60                     if (tabStopChild != null)
 61                     {
 62                         return tabStopChild;
 63                     }
 64                 }
 65                 child = child->NextSibling();
 66             }
 67             return null;
 68         }

132         public override void PrintWindowTree(int level)
133         {
134             LogView* log = Application.GetLogView();
135             if (log != null)
136             {
137                 log->WriteLine(string(' ', level) + "ContainerControl." + Text() + ".handle=" + ToHexString(cast<ulong>(Handle())) + " " + ParentText() + "[" + Rect(Point(), GetSize()).ToString() + "]");
138             }
139             Component* child = children.FirstChild();

150         protected override bool ProcessMessage(Message& message)
151         {
152             switch (message.msg)
153             {
154                 case WM_SIZE:
155                 {
156                     Size newSize = message.LParamSize();
157                     Size oldSize = GetSize();
158                     if (newSize != oldSize)
159                     {
160                         Rect parentRect(Point(), newSize);
161                         int dx = newSize.w - oldSize.w;
162                         int dy = newSize.h - oldSize.h;
163                         Component* child = children.FirstChild();
164                         while (child != null)
165                         {
166                             if (child is Control*)
167                             {
168                                 Control* childControl = cast<Control*>(child);
169                                 if (childControl->GetDock() == Dock.none)
170                                 {
171                                     childControl->MoveWindow(dx, dy);
172                                 }
173                             }
174                             child = child->NextSibling();
175                         }
176                         SetSizeInternal(newSize);
177                         DockChildren();
178                     }
179                     message.result = 0;
180                     return true;
181                 }
182                 case WM_PAINT: case WM_MOUSEMOVE: case WM_MOUSELEAVE: case WM_LBUTTONDOWN: case WM_LBUTTONUP: case WM_LBUTTONDBLCLK: case WM_RBUTTONDOWN: case WM_RBUTTONUP:
183                 case WM_CHAR: case WM_KEYDOWN: case WM_KEYUP: case WM_SYSCOMMAND: case WM_HSCROLL: case WM_VSCROLL: case WM_MOUSEWHEEL:
184                 case WM_SETFOCUS: case WM_KILLFOCUS: case WM_TIMER: case WM_COMMAND: case WM_SHOWWINDOW: case WM_CLIPBOARDUPDATE:
185                 {
186                     return base->ProcessMessage(message);
187                 }
188             }
189             return false;
190         }
191         internal void DockChildren()
192         {
193             Rect parentRect(Point(), GetSize());
194             DockChildren(parentRect);
195         }
196         internal void DockChildren(Rect& parentRect)
197         {
198             List<Control*> filledControls;
199             List<ContainerControl*> containers;
200             Component* child = children.FirstChild();
201             while (child != null)
202             {
203                 if (child is Control*)
204                 {
205                     Control* childControl = cast<Control*>(child);
206                     switch (childControl->GetDock())
207                     {
208                         case Dock.none: break;
209                         case Dock.fill: filledControls.Add(childControl); break;
210                         default:
211                         {
212                             childControl->DockWindow(parentRect);
213                             break;
214                         }
215                     }
216                     if (childControl is ContainerControl*)
217                     {
218                         ContainerControl* containerChild = cast<ContainerControl*>(childControl);
219                         containers.Add(containerChild);
220                     }
221                 }
222                 child = child->NextSibling();
223             }
224             for (Control* filledControl : filledControls)
225             {
226                 filledControl->DockWindow(parentRect);
227             }
228             for (ContainerControl* container : containers)
229             {
230                 container->DockChildren();
231             }
232         }
233         internal override void PaintAll(PaintEventArgs& args, bool skipMenuBar)
234         {
235             try
236             {
237                 base->PaintAll(args, skipMenuBar);
238                 Control* topControl = TopControl();
239                 Component* child = children.FirstChild();
240                 while (child != null)
241                 {
242                     if (child is Control*)
243                     {
244                         Control* childControl = cast<Control*>(child);
245                         if (childControl != topControl)
246                         {
247                             bool skip = false;
248                             if (skipMenuBar)
249                             {
250                                 if (childControl is MenuBar*)
251                                 {
252                                     skip = true;
253                                 }
254                                 else if (childControl is MenuBox*)
255                                 {
256                                     MenuBox* menuBox = cast<MenuBox*>(childControl);
257                                     if (!menuBox->PaintThisMenuBox())
258                                     {
259                                         skip = true;
260                                     }
261                                 }
262                             }
263                             if (!skip)
264                             {
265                                 GraphicsState state = args.graphics.SaveChecked();
266                                 Point loc = childControl->Location();
267                                 args.graphics.SetClipChecked(Rect(childControl->Location(), childControl->GetSize()));
268                                 args.graphics.TranslateTransformChecked(loc.x, loc.y);
269                                 childControl->PaintAll(args, skipMenuBar);
270                                 args.graphics.RestoreChecked(state);
271                             }
272                         }
273                     }
274                     child = child->NextSibling();
275                 }
276                 if (topControl != null)
277                 {
278                     GraphicsState state = args.graphics.SaveChecked();
279                     Point loc = topControl->Location();
280                     args.graphics.SetClipChecked(Rect(topControl->Location(), topControl->GetSize()));
281                     args.graphics.TranslateTransformChecked(loc.x, loc.y);
282                     topControl->PaintAll(args, skipMenuBar);
283                     args.graphics.RestoreChecked(state);
284                 }
285             }
286             catch (const Exception& ex)
287             {
288                 MessageBox.Show(ex.Message());
289             }
290         }
291         private Container children;
