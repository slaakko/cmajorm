 51         internal override nothrow Control* GetFirstEnabledTabStopControl() const
 52         {

 54             while (child != null)
 55             {
 56                 if (child is Control*)
 57                 {
 58                     Control* control = cast<Control*>(child);
 59                     Control* tabStopChild = control->GetFirstEnabledTabStopControl();
 60                     if (tabStopChild != null)
 61                     {
 62                         return tabStopChild;
 63                     }
 64                 }
 65                 child = child->NextSibling();
 66             }
 67             return null;
 68         }

150         protected override bool ProcessMessage(Message& message)
151         {
152             switch (message.msg)
153             {
154                 case WM_SIZE:
155                 {
156                     Size newSize = message.LParamSize();
157                     Size oldSize = GetSize();
158                     if (newSize != oldSize)
159                     {
160                         Rect parentRect(Point(), newSize);
161                         int dx = newSize.w - oldSize.w;
162                         int dy = newSize.h - oldSize.h;
163                         Component* child = children.FirstChild();
164                         while (child != null)
165                         {
166                             if (child is Control*)
167                             {
168                                 Control* childControl = cast<Control*>(child);
169                                 if (childControl->GetDock() == Dock.none)
170                                 {
171                                     childControl->MoveWindow(dx, dy);
172                                 }
173                             }
174                             child = child->NextSibling();
175                         }
176                         uint windowState = message.wparam;
177                         SetSizeInternal(newSize, windowState);
178                         DockChildren();
179                         OnSizeChanged(windowState);
180                     }
181                     message.result = 0;
182                     return true;
183                 }
184                 case WM_PAINT: case WM_MOVE: case WM_MOUSEMOVE: case WM_MOUSELEAVE: case WM_LBUTTONDOWN: case WM_LBUTTONUP: case WM_LBUTTONDBLCLK: case WM_RBUTTONDOWN: case WM_RBUTTONUP:
185                 case WM_CHAR: case WM_KEYDOWN: case WM_KEYUP: case WM_SYSCOMMAND: case WM_HSCROLL: case WM_VSCROLL: case WM_MOUSEWHEEL:
186                 case WM_SETFOCUS: case WM_KILLFOCUS: case WM_TIMER: case WM_COMMAND: case WM_SHOWWINDOW: case WM_CLIPBOARDUPDATE:
187                 {
188                     return base->ProcessMessage(message);
189                 }
190             }
191             return false;
192         }
193         internal void DockChildren()
194         {
195             Rect parentRect(Point(), GetSize());
196             DockChildren(parentRect);
197         }
198         internal void DockChildren(Rect& parentRect)
199         {
200             List<Control*> filledControls;
201             List<ContainerControl*> containers;
202             Component* child = children.FirstChild();
203             while (child != null)
204             {
205                 if (child is Control*)
206                 {
207                     Control* childControl = cast<Control*>(child);
208                     switch (childControl->GetDock())
209                     {
210                         case Dock.none: break;
211                         case Dock.fill: filledControls.Add(childControl); break;
212                         default:
213                         {
214                             childControl->DockWindow(parentRect);
215                             break;
216                         }
217                     }
218                     if (childControl is ContainerControl*)
219                     {
220                         ContainerControl* containerChild = cast<ContainerControl*>(childControl);
221                         containers.Add(containerChild);
222                     }
223                 }
224                 child = child->NextSibling();
225             }
226             for (Control* filledControl : filledControls)
227             {
228                 filledControl->DockWindow(parentRect);
229             }
230             for (ContainerControl* container : containers)
231             {
232                 container->DockChildren();
233             }
234         }
235         internal override void PaintAll(PaintEventArgs& args, bool skipMenuBar)
236         {
237             try
238             {
239                 base->PaintAll(args, skipMenuBar);
240                 Control* topControl = TopControl();
241                 Component* child = children.FirstChild();
242                 while (child != null)
243                 {
244                     if (child is Control*)
245                     {
246                         Control* childControl = cast<Control*>(child);
247                         if (childControl != topControl)
248                         {
249                             bool skip = false;
250                             if (skipMenuBar)
251                             {
252                                 if (childControl is MenuBar*)
253                                 {
254                                     skip = true;
255                                 }
256                                 else if (childControl is MenuBox*)
257                                 {
258                                     MenuBox* menuBox = cast<MenuBox*>(childControl);
259                                     if (!menuBox->PaintThisMenuBox())
260                                     {
261                                         skip = true;
262                                     }
263                                 }
264                             }
265                             if (!skip)
266                             {
267                                 GraphicsState state = args.graphics.SaveChecked();
268                                 Point loc = childControl->Location();
269                                 args.graphics.SetClipChecked(Rect(childControl->Location(), childControl->GetSize()));
270                                 args.graphics.TranslateTransformChecked(loc.x, loc.y);
271                                 childControl->PaintAll(args, skipMenuBar);
272                                 args.graphics.RestoreChecked(state);
273                             }
274                         }
275                     }
276                     child = child->NextSibling();
277                 }
278                 if (topControl != null)
279                 {
280                     GraphicsState state = args.graphics.SaveChecked();
281                     Point loc = topControl->Location();
282                     args.graphics.SetClipChecked(Rect(topControl->Location(), topControl->GetSize()));
283                     args.graphics.TranslateTransformChecked(loc.x, loc.y);
284                     topControl->PaintAll(args, skipMenuBar);
285                     args.graphics.RestoreChecked(state);
286                 }
287             }
288             catch (const Exception& ex)
289             {
290                 MessageBox.Show(ex.Message());
291             }
292         }
293         private Container children;
