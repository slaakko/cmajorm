  6 using System;
  7 using System.Collections;

  9 namespace System.Windows
 10 {
 11     public class delegate void ControlEventHandler(ControlEventArgs& args);
 12 
 13     public class ContainerControl : Control
 14     {
 15         public ContainerControl(const char* windowClassName, WindowClassStyle windowClassStyle, WindowStyle style, ExtendedWindowStyle exStyle, 
 16             const Color& backgroundColor, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 17             base(windowClassName, windowClassStyle, style, exStyle, backgroundColor, text, location, size, dock, anchors), children(this)
 18         {
 19         }
 20         public void AddChild(Control* child)
 21         {
 22             children.AddChild(child);
 23             ControlEventArgs args(child);
 24             OnControlAdded(args);
 25         }
 26         public UniquePtr<Control> RemoveChild(Control* child)
 27         {
 28             UniquePtr<Component> childComponent = children.RemoveChild(child);
 29             ControlEventArgs args(child);
 30             OnControlRemoved(args);
 31             return UniquePtr<Control>(childComponent.Release() as Control*);
 32         }
 33         protected override bool ProcessMessage(Message& message)
 34         {
 35             switch (message.msg)
 36             {
 37                 case WM_COMMAND:
 38                 {
 39                     void* childWindowHandle = message.LParamHandle();
 40                     ushort notificationCode = message.WParamHiWord();
 41                     ushort controlId = message.WParamLoWord();
 42                     Control* child = Application.GetWindowManager().GetWindow(childWindowHandle);
 43                     if (child != null)
 44                     {
 45                         switch (notificationCode)
 46                         {
 47                             case BN_CLICKED:
 48                             {
 49                                 if (child is ButtonBase*)
 50                                 {
 51                                     ButtonBase* button = cast<ButtonBase*>(child);
 52                                     button->OnClickInternal();
 53                                     message.result = 0;
 54                                     return true;
 55                                 }
 56                                 break;
 57                             }
 58                         }
 59                     }
 60                     break;
 61                 }
 62                 case WM_SIZE:
 63                 {
 64                     Size newSize = message.LParamSize();
 65                     Size oldSize = GetSize();
 66                     if (newSize != oldSize)
 67                     {
 68                         Rect parentRect(Point(), newSize);
 69                         int dx = newSize.w - oldSize.w;
 70                         int dy = newSize.h - oldSize.h;
 71                         Component* child = children.FirstChild();
 72                         while (child != null)
 73                         {
 74                             if (child is Control*)
 75                             {
 76                                 Control* childControl = cast<Control*>(child);
 77                                 if (childControl->GetDock() != Dock.none)
 78                                 {
 79                                     childControl->DockWindow(parentRect);
 80                                 }
 81                                 else
 82                                 {
 83                                     childControl->MoveWindow(dx, dy);
 84                                 }
 85                             }
 86                             child = child->NextSibling();
 87                         }
 88                         SetSizeInternal(newSize);
 89                     }
 90                     message.result = 0;
 91                     return true;
 92                 }
 93                 case WM_PAINT: case WM_MOUSEMOVE: case WM_MOUSELEAVE: case WM_LBUTTONDOWN: case WM_LBUTTONUP: case WM_CHAR: case WM_KEYDOWN: case WM_KEYUP: case WM_SYSCOMMAND:
 94                 {
 95                     return base->ProcessMessage(message);
 96                 }
 97             }
 98             return false;
 99         }
100         internal void DockChildren()
101         {
102             Rect parentRect(Point(), GetSize());
103             Component* child = children.FirstChild();
104             while (child != null)
105             {
106                 if (child is Control*)
107                 {
108                     Control* childControl = cast<Control*>(child);
109                     childControl->DockWindow(parentRect);
110                 }
111                 child = child->NextSibling();
112             }
113         }
114         protected virtual void OnControlAdded(ControlEventArgs& args)
115         {
116             controlAddedEvent.Fire(args);
117         }
118         public virtual void OnControlRemoved(ControlEventArgs& args)
119         {
120             controlRemovedEvent.Fire(args);
121         }
122         private Container children;
123         private Event<ControlEventHandler, ControlEventArgs> controlAddedEvent;
124         private Event<ControlEventHandler, ControlEventArgs> controlRemovedEvent;
125     }
