  6 using System;
  7 
  8 namespace System.Windows
  9 {
 10     public enum TextBoxStyle : long
 11     {
 12         ES_AUTOHSCROLL = 0x0080, 
 13         ES_AUTOVSCROLL = 0x0040, 
 14         ES_CENTER = 0x0001, 
 15         ES_LEFT = 0x0000, 
 16         ES_LOWERCASE = 0x0010, 
 17         ES_MULTILINE = 0x0004, 
 18         ES_NOHIDESEL = 0x0100, 
 19         ES_NUMBER = 0x2000, 
 20         ES_OEMCONVERT = 0x0400, 
 21         ES_PASSWORD = 0x0020, 
 22         ES_READONLY = 0x0800, 
 23         ES_RIGHT = 0x0002, 
 24         ES_UPPERCASE = 0x0008, 
 25         ES_WANTRETURN = 0x1000
 26     }
 27 
 28     public class TextBox : Control
 29     {
 30         private enum Flags : sbyte
 31         {
 32             none = 0, selectAll = 1 << 0, multiline = 1 << 1
 33         }
 34 
 35         public TextBox(TextBoxStyle textBoxStyle, const Color& backgroundColor, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 36             base("EDIT", DefaultWindowClassStyle(), cast<WindowStyle>(DefaultChildWindowStyle() | WindowStyle.WS_TABSTOP | textBoxStyle), DefaultExtendedWindowStyle(), backgroundColor, 
 37             text, location, size, dock, anchors)
 38         {
 39             if ((textBoxStyle & TextBoxStyle.ES_MULTILINE) != 0)
 40             {
 41                 SetMultiline();
 42             }
 43         }
 44         public TextBox(const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors) : 
 45             this(cast<TextBoxStyle>(TextBoxStyle.ES_LEFT | TextBoxStyle.ES_AUTOHSCROLL), Color.White(), text, location, size, dock, anchors)
 46         {
 47         }
 48         public TextBox(const string& text) : this(text, Point(), Size(), Dock.none, Anchors.none)
 49         {
 50         }
 51         public inline nothrow bool IsMultiline() const
 52         {
 53             return (flags & Flags.multiline) != Flags.none;
 54         }
 55         private inline nothrow void SetMultiline()
 56         {
 57             flags = cast<Flags>(flags | Flags.multiline);
 58         }
 59         private inline nothrow void ResetMultiline()
 60         {
 61             flags = cast<Flags>(flags & ~Flags.multiline);
 62         }
 63         public void SelectAll()
 64         {
 65             if (Handle() != null)
 66             {
 67                 WinSendMessage(Handle(), EM_SETSEL, 0u, -1);
 68             }
 69             else
 70             {
 71                 SetSelectAllFlag();
 72             }
 73         }
 74         public void Clear()
 75         {
 76             SetText(string());
 77         }
 78         protected override void OnCreated()
 79         {
 80             base->OnCreated();
 81             Graphics graphics = Graphics.FromWindowHandle(Handle());
 82             const FontHandle& fontHandle = GetFontHandle(graphics);
 83             if (!fontHandle.IsNull())
 84             {
 85                 SendSetFontMessage(fontHandle);
 86             }
 87             if (SelectAllFlagSet())
 88             {
 89                 ResetSelectAllFlag();
 90                 SelectAll();
 91             }
 92         }
 93         protected override void OnGotFocus()
 94         {
 95             base->OnGotFocus();
 96             SelectAll();
 97         }
 98         private inline nothrow bool SelectAllFlagSet() const
 99         {
100             return (flags & Flags.selectAll) != Flags.none;
101         }
102         private inline nothrow void SetSelectAllFlag()
103         {
104             flags = cast<Flags>(flags | Flags.selectAll);
105         }
106         private inline nothrow void ResetSelectAllFlag()
107         {
108             flags = cast<Flags>(flags & ~Flags.selectAll);
109         }
110         private Flags flags;
111     }
112 }
