 45     public class Label : Control
 46     {
 47         private enum Flags : sbyte
 48         {
 49             none = 0, autoSize = 1 << 0, autoSized = 1 << 1
 50         }
 51 
 52         public Label(LabelStyle labelStyle, const Color& backgroundColor, const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors, bool autoSize) : 
 53             base("STATIC", DefaultWindowClassStyle(), cast<WindowStyle>(DefaultChildWindowStyle() | labelStyle), DefaultExtendedWindowStyle(), backgroundColor, 
 54             text, location, size, dock, anchors), flags(Flags.none)
 55         {
 56             if (autoSize)
 57             {
 58                 SetAutoSizeFlag();
 59                 DoAutoSize();
 60             }
 61         }
 62         public Label(const string& text, const Point& location, const Size& size, Dock dock, Anchors anchors, bool autoSize) : 
 63             this(LabelStyle.SS_LEFT, DefaultControlBackgroundColor(), text, location, size, dock, anchors, autoSize)
 64         {
 65         }
 66         protected override void OnCreated()
 67         {
 68             base->OnCreated();
 69             Graphics graphics = Graphics.FromWindowHandle(Handle());
 70             const FontHandle& fontHandle = GetFontHandle(graphics);
 71             if (!fontHandle.IsNull())
 72             {
 73                 SendSetFontMessage(fontHandle);
 74             }
 75             if (AutoSize())
 76             {
 77                 DoAutoSize();
 78             }
 79         }
 80         protected override void OnTextChanged()
 81         {
 82             base->OnTextChanged();
 83             if (AutoSize())
 84             {
 85                 ResetAutoSized();
 86                 DoAutoSize();
 87             }
 88         }
 89         public inline nothrow bool AutoSize() const
 90         {
 91             return (flags & Flags.autoSize) != Flags.none;
 92         }
 93         public void SetAutoSize(bool autoSize)
 94         {
 95             if (autoSize)
 96             {
 97                 if (!AutoSize())
 98                 {
 99                     SetAutoSizeFlag();
100                     if (!AutoSized())
101                     {
102                         DoAutoSize();
103                     }
104                 }
105             }
106             else
107             {
108                 ResetAutoSizeFlag();
109                 ResetAutoSized();
110             }
111         }
112         private void DoAutoSize()
113         {
114             if (AutoSized()) return;
115             if (Handle() == null) return;
116             Graphics graphics = Graphics.FromWindowHandle(Handle());
117             const Font& font = GetFont();
118             StringFormat stringFormat;
119             RectF r = graphics.MeasureStringChecked(Text(), font, PointF(0, 0), stringFormat);
120             SetSize(Size(cast<int>(r.size.w), cast<int>(r.size.h)));
121             SetAutoSized();
122         }
123         private inline nothrow void SetAutoSizeFlag()
124         {
125             flags = cast<Flags>(flags | Flags.autoSize);
126         }
127         private inline nothrow void ResetAutoSizeFlag()
128         {
129             flags = cast<Flags>(flags & ~Flags.autoSize);
130         }
131         private inline nothrow bool AutoSized() const
132         {
133             return (flags & Flags.autoSized) != Flags.none;
134         }
135         private inline nothrow void SetAutoSized()
136         {
137             flags = cast<Flags>(flags | Flags.autoSized);
138         }
139         private inline nothrow void ResetAutoSized()
140         {
141             flags = cast<Flags>(flags & ~Flags.autoSized);
142         }
143         private Flags flags;
144     }
