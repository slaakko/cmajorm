 46     public class IconListViewItem : Component
 47     {
 48         public enum Flags : sbyte
 49         {
 50             none = 0, selected = 1 << 0
 51         }
 52         public nothrow IconListViewItem(const string& iconName_, const string& itemName_) : index(-1), iconName(iconName_), itemName(itemName_), location(), size()
 53         {
 54         }
 55         public void Measure(Graphics& graphics, const Padding& iconPadding, const Padding& textPadding, const Font& font)
 56         {
 57             iconSize = Size(cast<int>(bitmap->GetWidth()), cast<int>(bitmap->GetHeight()));
 58             PointF origin;
 59             StringFormat format(StringAlignment.near, StringAlignment.near);
 60             RectF textRect = graphics.MeasureStringChecked(itemName, font, origin, format);
 61             textSize = textRect.size;
 62             size.w = Max(cast<int>(iconSize.w + iconPadding.Horizontal()), cast<int>(textSize.w + textPadding.Horizontal()));
 63             size.h = cast<int>(iconSize.h + iconPadding.Vertical() + textSize.h + textPadding.Vertical());
 64         }
 65         public void SetLocations(const Point& loc, const Padding& iconPadding, const Padding& textPadding)
 66         {
 67             location = loc;
 68             if (textSize.w > iconSize.w)
 69             {
 70                 iconLocation = Point(cast<int>(location.x + iconPadding.left + ((textSize.w + textPadding.Horizontal()) - (iconSize.w + iconPadding.Horizontal())) / 2), location.y + iconPadding.top);
 71                 textLocation = PointF(location.x + textPadding.left, location.y + (size.h - textSize.h - textPadding.Vertical()) + textPadding.top);
 72             }
 73             else
 74             {
 75                 iconLocation = Point(location.x + iconPadding.left, location.y + iconPadding.top);
 76                 textLocation = PointF(location.x + textPadding.left + ((iconSize.w + iconPadding.Horizontal()) - (textSize.w + textPadding.Horizontal())) / 2, location.y + (size.h - textSize.h - textPadding.Vertical()) + textPadding.top);
 77             }
 78         }
 79         public void Draw(Graphics& graphics, IconListView& iconListView)
 80         {
 81             if (IsSelected())
 82             {
 83                 SolidBrush& selectedBrush = iconListView.GetSelectedBrush();
 84                 RectF rect(PointF(location.x, location.y), SizeF(size.w, size.h));
 85                 graphics.FillRectangleChecked(selectedBrush, rect);
 86             }
 87             Rect r(iconLocation, iconSize);
 88             graphics.DrawImageChecked(*bitmap, r, 0, 0, iconSize.w, iconSize.h, Unit.pixel, iconListView.GetImageAttributes());
 89             RectF textRect(textLocation, textSize);
 90             graphics.DrawStringChecked(itemName, iconListView.GetFont(), textRect, iconListView.GetStringFormat(), iconListView.GetTextBrush());
 91         }

139         public void AddItem(const string& iconName, const string& itemName)
140         {
141             UniquePtr<IconListViewItem> item(new IconListViewItem(iconName, itemName));
142             item->index = numberOfItems;
143             item->bitmap = GetOrInsertBitmap(iconName);
144             items.AddChild(item.Release());
145             ++numberOfItems;
146             SetChanged();
147             Invalidate();
148         }
149         protected override void OnMouseDown(MouseEventArgs& args)
150         {
151             if (selectedItem != null)
152             {
153                 selectedItem->ResetSelected();
154             }
155             Point loc = args.location;
156             Component* component = items.FirstChild();
157             while (component != null)
158             {
159                 if (component is IconListViewItem*)
160                 {
161                     IconListViewItem* item = cast<IconListViewItem*>(component);
162                     Rect rect(item->location, item->size);
163                     if (rect.Contains(loc))
164                     {
165                         item->SetSelected();
166                         selectedItem = item;
167                         OnSelectedIndexChanged();
168                         break;
169                     }
170                 }
171                 component = component->NextSibling();
172             }
173             Invalidate();
174         }
175         protected override void OnMouseDoubleClick(MouseEventArgs& args)
176         {
177             Point loc = args.location;
178             Component* component = items.FirstChild();
179             while (component != null)
180             {
181                 if (component is IconListViewItem*)
182                 {
183                     IconListViewItem* item = cast<IconListViewItem*>(component);
184                     Rect rect(item->location, item->size);
185                     if (rect.Contains(loc))
186                     {
187                         item->SetSelected();
188                         selectedItem = item;
189                         OnDoubleClicked();
190                         break;
191                     }
192                 }
193                 component = component->NextSibling();
194             }
195             Invalidate();
196         }

262         protected override void OnPaint(PaintEventArgs& args)
263         {
264             try
265             {
266                 if (Changed())
267                 {
268                     ResetChanged();
269                     Measure(args.graphics);
270                 }
271                 args.graphics.Clear(BackgroundColor());
272                 Component* component = items.FirstChild();
273                 while (component != null)
274                 {
275                     if (component is IconListViewItem*)
276                     {
277                         IconListViewItem* item = cast<IconListViewItem*>(component);
278                         item->Draw(args.graphics, *this);
279                     }
280                     component = component->NextSibling();
281                 }
282             }
283             catch (const Exception& ex)
284             {
285                 MessageBox.Show(ex.Message());
286             }
287             base->OnPaint(args);
288         }
289         private void Measure(Graphics& graphics)
290         {
291             Point loc(internalPadding.left, internalPadding.top);
292             Size size;
293             Component* component = items.FirstChild();
294             while (component != null)
295             {
296                 if (component is IconListViewItem*)
297                 {
298                     IconListViewItem* item = cast<IconListViewItem*>(component);
299                     item->Measure(graphics, iconPadding, textPadding, font);
300                     item->SetLocations(loc, iconPadding, textPadding);
301                     loc.x = loc.x + item->size.w;
302                     size.w = size.w + item->size.w;
303                     size.h = Max(size.h, item->size.h);
304                 }
305                 component = component->NextSibling();
306             }
307             size.w = size.w + internalPadding.Horizontal();
308             size.h = size.h + internalPadding.Vertical();
309             SetSize(size);
310         }

323         private Bitmap* GetOrInsertBitmap(const string& bitmapName)
324         {
325             try
326             {
327                 HashMap<string, Bitmap*>.ConstIterator it = bitmapMap.CFind(bitmapName);
328                 if (it != bitmapMap.CEnd())
329                 {
330                     return it->second;
331                 }
332                 UniquePtr<Bitmap> bitmapPtr(new Bitmap(Bitmap.FromResource(bitmapName)));
333                 Bitmap* bitmap = bitmapPtr.Get();
334                 bitmapMap[bitmapName] = bitmap;
335                 bitmaps.Add(Rvalue(bitmapPtr));
336                 return bitmap;
337             }
338             catch (const Exception& ex)
339             {
340                 throw Exception("could not load bitmap '" + bitmapName + "': " + ex.Message());
341             }
342             return null;
343         }
344         private SolidBrush* GetOrInsertSolidBrush(const System.Windows.Color& color)
345         {
346             HashMap<System.Windows.Color, SolidBrush*>.ConstIterator it = brushMap.CFind(color);
347             if (it != brushMap.CEnd())
348             {
349                 return it->second;
350             }
351             else
352             {
353                 UniquePtr<SolidBrush> solidBrushPtr(new SolidBrush(color));
354                 SolidBrush* solidBrush = solidBrushPtr.Get();
355                 brushMap[color] = solidBrush;
356                 brushes.Add(Rvalue(solidBrushPtr));
357                 return solidBrush;
358             }
