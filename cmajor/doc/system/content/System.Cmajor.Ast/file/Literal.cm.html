838     public LiteralNode* CreateIntegerLiteralNode(const System.Lex.Span& span, ulong value, bool unsignedSuffix)
839     {
840         if (unsignedSuffix)
841         {
842             if (value <= MaxValue<byte>()) return new ByteLiteralNode(span, cast<byte>(value));
843             if (value <= MaxValue<ushort>()) return new UShortLiteralNode(span, cast<ushort>(value));
844             if (value <= MaxValue<uint>()) return new UIntLiteralNode(span, cast<uint>(value));
845             return new ULongLiteralNode(span, value);
846         }
847         else
848         {
849             if (value <= MaxValue<sbyte>()) return new SByteLiteralNode(span, cast<sbyte>(value));
850             if (value <= MaxValue<byte>()) return new ByteLiteralNode(span, cast<byte>(value));
851             if (value <= MaxValue<short>()) return new ShortLiteralNode(span, cast<short>(value));
852             if (value <= MaxValue<ushort>()) return new UShortLiteralNode(span, cast<ushort>(value));
853             if (value <= MaxValue<int>()) return new IntLiteralNode(span, cast<int>(value));
854             if (value <= MaxValue<uint>()) return new UIntLiteralNode(span, cast<uint>(value));
855             if (value <= MaxValue<long>()) return new LongLiteralNode(span, cast<long>(value));
856             return new ULongLiteralNode(span, value);
857         }
858     }
