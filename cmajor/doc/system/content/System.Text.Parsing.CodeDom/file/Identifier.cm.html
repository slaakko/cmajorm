 6 using System;
 7 using System.Collections;

 9 namespace System.Text.Parsing.CodeDom
10 {
11     public class Identifier : DomObject
12     {
13         public nothrow Identifier(const ustring& name_) : base(name_)
14         {
15         }
16         public override nothrow int Rank() const
17         {
18             return 20;
19         }
20         public override void Accept(Visitor& visitor)
21         {
22             visitor.Visit(*this);
23         }
24     }
25     
26     public class TemplateId : DomObject
27     {
28         public nothrow TemplateId(const ustring& name_) : base(name_)
29         {
30         }
31         public override nothrow ustring Name() const
32         {
33             ustring name = base->Name();
34             name.Append( u"<");
35             bool first = true;
36             for (const UniquePtr<DomObject>& templateArg : templateArgs)
37             {
38                 if (first)
39                 {
40                     first = false;
41                 }
42                 else
43                 {
44                     name.Append( u", ");
45                 }
46                 name.Append(templateArg->Name());
47             }
48             name.Append( u">");
49             return name;
50         }
51         public void AddTemplateArg(DomObject* templateArg)
52         {
53             templateArgs.Add(UniquePtr<DomObject>(templateArg));
54         }
55         public override void Print(CodeFormatter& formatter)
56         {
57             if (Parens())
58             {
59                 formatter.Write('(');
60             }
61             formatter.Write(ToUtf8(Name()));
62             if (Parens())
63             {
64                 formatter.Write(')');
65             }
66         }
67         public override nothrow int Rank() const
68         {
69             return 20;
70         }
71         public override void Accept(Visitor& visitor)
72         {
73             visitor.Visit(*this);
74             for (const UniquePtr<DomObject>& templateArg : templateArgs)
75             {
76                 templateArg->Accept(visitor);
77             }
78         }
79         private List<UniquePtr<DomObject>> templateArgs;
80     }
81 }
