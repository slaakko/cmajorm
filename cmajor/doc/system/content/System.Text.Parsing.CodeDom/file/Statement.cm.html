 42         {
 43             statements.Add(UniquePtr<Statement>(statement));
 44         }
 45         public override void Print(CodeFormatter& formatter)
 46         {
 47             formatter.WriteLine('{');
 48             formatter.IncIndent();
 49             if (formatter.Start())
 50             {
 51                 formatter.WriteLine(formatter.StartText());
 52                 formatter.SetStart(false);
 53             }
 54             for (const UniquePtr<Statement>& statement : statements)
 55             {
 56                 statement->Print(formatter);
 57             }
 58             formatter.DecIndent();
 59             formatter.WriteLine('}');
 60         }
 61         public override void Accept(Visitor& visitor)
 62         {
 63             visitor.Visit(*this);
 64             for (const UniquePtr<Statement>& statement : statements)
 65             {
 66                 statement->Accept(visitor);
 67             }
 68         }

105         {
106         }
107         public override void Print(CodeFormatter& formatter)
108         {
109             formatter.Write("if (");
110             condition->Print(formatter);
111             formatter.Write(')');
112             if (thenS.Get() is CompoundStatement*)
113             {
114                 formatter.WriteLine();
115             }
116             else
117             {
118                 formatter.Write(' ');
119                 thenS->SetNoNewLine();
120             }
121             thenS->Print(formatter);
122             if (!elseS.IsNull())
123             {
124                 if (elseS.Get() is CompoundStatement*)
125                 {
126                     if (thenS->NoNewLine())
127                     {
128                         formatter.WriteLine();
129                     }
130                     formatter.WriteLine("else");
131                 }
132                 else
133                 {
134                     if (thenS->NoNewLine())
135                     {
136                         formatter.Write(' ');
137                     }
138                     formatter.Write("else ");
139                 }
140                 elseS->Print(formatter);
141             }
142             else if (thenS->NoNewLine())
143             {
144                 formatter.WriteLine();
145             }
146         }
147         public override void Accept(Visitor& visitor)
148         {
149             visitor.Visit(*this);
150             condition->Accept(visitor);
151             thenS->Accept(visitor);
152             if (!elseS.IsNull())
153             {
154                 elseS->Accept(visitor);
155             }
156         }
157         private UniquePtr<DomObject> condition;
158         private UniquePtr<Statement> thenS;
159         private UniquePtr<Statement> elseS;

351         public ConstructionStatement(DomObject* typeExpr_, DomObject* id_) : base( u"construction"), typeExpr(typeExpr_), id(id_), parens(false)

356             arguments.Add(UniquePtr<DomObject>(argument));
357         }
358         public override void Print(CodeFormatter& formatter)
359         {
360             typeExpr->Print(formatter);
361             formatter.Write(' ');
362             id->Print(formatter);
363             if (!arguments.IsEmpty())
364             {
365                 if (parens)
366                 {
367                     formatter.Write('(');
368                 }
369                 else
370                 {
371                     formatter.Write(" = ");
372                 }
373                 bool first = true;
374                 for (const UniquePtr<DomObject>& argument : arguments)
375                 {
376                     if (first)
377                     {
378                         first = false;
379                     }
380                     else
381                     {
382                         formatter.Write(", ");
383                     }
384                     argument->Print(formatter);
385                 }
386                 if (parens)
387                 {
388                     formatter.Write(')');
389                 }
390             }
391             formatter.Write(';');
392             if (!NoNewLine())
393             {
394                 formatter.WriteLine();
395             }
396         }
397         public override void Accept(Visitor& visitor)
398         {
399             visitor.Visit(*this);
400             typeExpr->Accept(visitor);
401             id->Accept(visitor);
402             for (const UniquePtr<DomObject>& argument : arguments)
403             {
404                 argument->Accept(visitor);
405             }
406         }
407         public nothrow void SetParens()
408         {
409             parens = true;
410         }
411         private UniquePtr<DomObject> typeExpr;
412         private UniquePtr<DomObject> id;
