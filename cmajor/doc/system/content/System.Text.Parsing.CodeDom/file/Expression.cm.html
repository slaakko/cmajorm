297     public abstract class BinaryExpression : Expression
298     {
299         public BinaryExpression(Operator op_, DomObject* left_, DomObject* right_) :

303         public override void Print(CodeFormatter& formatter)
304         {
305             if (Parens())
306             {
307                 formatter.Write('(');
308             }
309             if (left->Rank() < Rank())
310             {
311                 left->SetParens();
312             }
313             left->Print(formatter);
314             formatter.Write(' ');
315             formatter.Write(ToUtf8(Name()));
316             formatter.Write(' ');
317             if (right->Rank() < Rank())
318             {
319                 right->SetParens();
320             }
321             right->Print(formatter);
322             if (Parens())
323             {
324                 formatter.Write(')');
325             }
326         }
327         public override void Accept(Visitor& visitor)
328         {
329             left->Accept(visitor);
330             right->Accept(visitor);
331         }

610     public class Invoke : Expression
611     {
612         public Invoke(DomObject* subject_) : base( u"()", 11), subject(subject_)
613         {
614         }
615         public override void AddArgument(DomObject* argument)
616         {
617             arguments.Add(UniquePtr<DomObject>(argument));
618         }
619         public override void Print(CodeFormatter& formatter)
620         {
621             if (Parens())
622             {
623                 formatter.Write('(');
624             }
625             subject->Print(formatter);
626             formatter.Write('(');
627             bool first = true;
628             for (const UniquePtr<DomObject>& argument : arguments)
629             {
630                 if (first)
631                 {
632                     first = false;
633                 }
634                 else
635                 {
636                     formatter.Write(", ");
637                 }
638                 argument->Print(formatter);
639             }
640             formatter.Write(')');
641             if (Parens())
642             {
643                 formatter.Write(')');
644             }
645         }
646         public override void Accept(Visitor& visitor)
647         {
648             visitor.Visit(*this);
649             subject->Accept(visitor);
650             for (const UniquePtr<DomObject>& argument : arguments)
651             {
652                 argument->Accept(visitor);
653             }
654         }
655         private UniquePtr<DomObject> subject;
656         private List<UniquePtr<DomObject>> arguments;
