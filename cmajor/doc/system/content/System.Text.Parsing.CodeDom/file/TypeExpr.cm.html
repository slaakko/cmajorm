  8 namespace System.Text.Parsing.CodeDom

 10     public class Const : DomObject
 11     {
 12         public nothrow Const(DomObject* subject_) : base(), subject(subject_)
 13         {
 14         }
 15         public override nothrow ustring Name() const
 16         {
 17             return  u"const " + subject->Name();
 18         }
 19         public override void Print(CodeFormatter& formatter)
 20         {
 21             if (Parens())
 22             {
 23                 formatter.Write('(');
 24             }
 25             formatter.Write("const ");
 26             subject->Print(formatter);
 27             if (Parens())
 28             {
 29                 formatter.Write(')');
 30             }
 31         }
 32         public override void Accept(Visitor& visitor)
 33         {
 34             visitor.Visit(*this);
 35             subject->Accept(visitor);
 36         }
 37         private UniquePtr<DomObject> subject;
 38     }

130     public class Array : DomObject
131     {
132         public nothrow Array(DomObject* subject_, DomObject* size_) : subject(subject_), size(size_)
133         {
134         }
135         public override nothrow ustring Name() const
136         {
137             return subject->Name() +  u"[" + size->Name() + u"]";
138         }
139         public override void Print(CodeFormatter& formatter)
140         {
141             if (Parens())
142             {
143                 formatter.Write('(');
144             }
145             subject->Print(formatter);
146             formatter.Write('[');
147             if (!size.IsNull())
148             {
149                 size->Print(formatter);
150             }
151             formatter.Write(']');
152             if (Parens())
153             {
154                 formatter.Write(')');
155             }
156         }
157         public override void Accept(Visitor& visitor)
158         {
159             visitor.Visit(*this);
160             subject->Accept(visitor);
161             size->Accept(visitor);
162         }
163         private UniquePtr<DomObject> subject;
164         private UniquePtr<DomObject> size;
165     }
