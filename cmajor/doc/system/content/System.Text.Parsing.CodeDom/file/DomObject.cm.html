 6 using System;
 7 using System.Text;
 8 
 9 namespace System.Text.Parsing.CodeDom
10 {
11     public abstract class DomObject
12     {
13         public nothrow DomObject(const ustring& name_) : name(name_), parens(false)
14         {
15         }
16         public virtual nothrow ustring Name() const
17         {
18             return name;
19         }
20         public nothrow void SetName(const ustring& name_)
21         {
22             name = name_;
23         }
24         public inline nothrow bool Parens() const
25         {
26             return parens;
27         }
28         public inline nothrow void SetParens()
29         {
30             parens = true;
31         }
32         public virtual void AddArgument(DomObject* argument)
33         {
34             throw Exception("AddArgument override not implemented");
35         }
36         public virtual void Print(CodeFormatter& formatter)
37         {
38             if (parens)
39             {
40                 formatter.Write('(');
41             }
42             formatter.Write(ToUtf8(name));
43             if (parens)
44             {
45                 formatter.Write(')');
46             }
47         }
48         public virtual nothrow int Rank() const
49         {
50             return 100;
51         }
52         public abstract void Accept(Visitor& visitor);
53         private ustring name;
54         private bool parens;
55     }
56 }
