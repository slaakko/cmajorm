310         public virtual DomNode* InsertBefore(UniquePtr<DomNode>&& newChild, DomNode* refChild)
311         {
312             if (refChild == null)
313             {
314                 return AppendChild(newChild);
315             }
316             else
317             {
318                 if (OwnerDocument() != null)
319                 {
320                     OwnerDocument()->InternalInvalidateIndex();
321                 }
322                 if (newChild.Get() is DomDocumentFragment*)
323                 {
324                     DomDocumentFragment* documentFragment = cast<DomDocumentFragment*>(newChild.Get());
325                     DomNode* docFragChild = documentFragment->FirstChild();
326                     DomNode* nodeInserted = null;
327                     while (docFragChild != null)
328                     {
329                         UniquePtr<DomNode> docFragChildRemoved = documentFragment->RemoveChild(docFragChild);
330                         nodeInserted = InsertBefore(Rvalue(docFragChildRemoved), refChild);
331                         docFragChild = documentFragment->FirstChild();
332                     }
333                     return nodeInserted;
334                 }
335                 else
336                 {
337                     if (newChild->Parent() != null)
338                     {
339                         newChild.Reset(newChild->Parent()->RemoveChild(newChild.Get()).Release());
340                     }
341                     if (firstChild == refChild)
342                     {
343                         firstChild = newChild.Get();
344                     }
345                     newChild->InternalSetParent(this);
346                     newChild->InternalSetOwnerDocument(OwnerDocument());
347                     DomNode* nodeInserted = newChild.Get();
348                     refChild->InternalLinkBefore(newChild.Release());
349                     return nodeInserted;
350                 }
351             }
352         }
353         public virtual UniquePtr<DomNode> RemoveChild(DomNode* oldChild)
354         {
355             if (oldChild == null)
356             {
357                 throw DomException("could not remove node: given old child is null");
358             }
359             if (oldChild->Parent() != this)
360             {
361                 throw DomException("could not remove node: given old child is not child of this node");
362             }
363             if (OwnerDocument() != null)
364             {
365                 OwnerDocument()->InternalInvalidateIndex();
366             }
367             oldChild->InternalUnlink();
368             if (oldChild == firstChild)
369             {
370                 firstChild = oldChild->NextSibling();
371             }
372             if (oldChild == lastChild)
373             {
374                 lastChild = oldChild->PreviousSibling();
375             }
376             UniquePtr<DomNode> removedNode = UniquePtr<DomNode>(oldChild);
377             removedNode->InternalSetOwnerDocument(null);
378             removedNode->InternalSetParent(null);
379             return removedNode;
380         }
381         public virtual UniquePtr<DomNode> ReplaceChild(UniquePtr<DomNode>&& newChild, DomNode* oldChild)
382         {
383             if (oldChild == null)
384             {
385                 throw DomException("could not replace node: given old child is null");
386             }
387             if (oldChild->Parent() != this)
388             {
389                 throw DomException("could not replace node: given old child is not child of this node");
390             }
391             if (OwnerDocument() != null)
392             {
393                 OwnerDocument()->InternalInvalidateIndex();
394             }
395             if (newChild->Parent() != null)
396             {
397                 newChild.Reset(newChild->Parent()->RemoveChild(newChild.Get()).Release());
398             }
399             InsertBefore(newChild, oldChild);
400             return RemoveChild(oldChild);
401         }
402         public virtual DomNode* AppendChild(UniquePtr<DomNode>&& newChild)
403         {
404             if (OwnerDocument() != null)
405             {
406                 OwnerDocument()->InternalInvalidateIndex();
407             }
408             if (newChild->Parent() != null)
409             {
410                 newChild.Reset(newChild->Parent()->RemoveChild(newChild.Get()).Release());
411             }
412             if (newChild.Get() is DomDocumentFragment*)
413             {
414                 DomDocumentFragment* documentFragment = cast<DomDocumentFragment*>(newChild.Get());
415                 DomNode* docFragChild = documentFragment->FirstChild();
416                 DomNode* nodeInserted = null;
417                 while (docFragChild != null)
418                 {
419                     UniquePtr<DomNode> docFragChildRemoved = documentFragment->RemoveChild(docFragChild);
420                     nodeInserted = AppendChild(Rvalue(docFragChildRemoved));
421                     docFragChild = documentFragment->FirstChild();
422                 }
423                 return nodeInserted;
424             }
425             else
426             {
427                 if (lastChild != null)
428                 {
429                     lastChild->InternalLinkAfter(newChild.Get());
430                 }
431                 if (firstChild == null)
432                 {
433                     firstChild = newChild.Get();
434                 }
435                 newChild->InternalSetOwnerDocument(OwnerDocument());
436                 newChild->InternalSetParent(this);
437                 lastChild = newChild.Release();
438                 return lastChild;
439             }
440         }
