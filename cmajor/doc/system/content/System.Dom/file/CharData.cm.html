192     public nothrow ustring XmlCharDataEscape(const ustring& charData)
193     {
194         ustring result;
195         for (uchar c : charData)
196         {
197             switch (c)
198             {
199                 case u'<': result.Append(u"&lt;"); break;
200                 case u'&': result.Append(u"&amp;"); break;
201                 default: result.Append(c); break;
202             }
203         }
204         return result;
205     }
