 12     public nothrow ustring AttrValueEscape(const ustring& attributeValue, uchar delimiter)
 13     {
 14         ustring result;
 15         for (uchar c : attributeValue)
 16         {
 17             switch (c)
 18             {
 19                 case  u'<':result.Append( u"&lt;");break;
 20                 case  u'&':result.Append( u"&amp;");break;
 21                 case  u'"':
 22                 {
 23                     if (delimiter ==  u'"')
 24                     {
 25                         result.Append( u"&quot;");
 26                     }
 27                     else
 28                     {
 29                         result.Append( u'"');
 30                     }
 31                     break;
 32                 }
 33                 case  u'\'':
 34                 {
 35                     if (delimiter ==  u'\'')
 36                     {
 37                         result.Append( u"&apos;");
 38                     }
 39                     else
 40                     {
 41                         result.Append( u'\'');
 42                     }
 43                     break;
 44                 }
 45                 default:
 46                 {
 47                     result.Append(c);
 48                     break;
 49                 }
 50             }
 51         }
 52         return result;
 53     }

 55     public nothrow ustring MakeXmlAttrValue(const ustring& attrValue)
 56     {
 57         ustring result;
 58         if (attrValue.Find( u'"') == -1)
 59         {
 60             result.Append( u'"');
 61             result.Append(AttrValueEscape(attrValue,  u'"'));
 62             result.Append( u'"');
 63         }
 64         else if (attrValue.Find( u'\'') == -1)
 65         {
 66             result.Append( u'\'');
 67             result.Append(AttrValueEscape(attrValue,  u'\''));
 68             result.Append( u'\'');
 69         }
 70         else
 71         {
 72             result.Append( u'"');
 73             result.Append(AttrValueEscape(attrValue,  u'"'));
 74             result.Append( u'"');
 75         }
 76         return result;
 77     }

157         public override UniquePtr<DomNode> CloneNode(bool deep) const
158         {
159             UniquePtr<DomNode> clone(new DomElement(Name()));
160             DomParentNode* cloneAsParent = cast<DomParentNode*>(clone.Get());
161             List<UniquePtr<DomAttr>> clonedAttrs;
162             Map<ustring, DomAttr*> clonedAttrMap;
163             for (const Pair<ustring, DomAttr*>& p : attrMap)
164             {
165                 DomAttr* attr = p.second;
166                 UniquePtr<DomNode> clonedAttr = attr->CloneNode(deep);
167                 clonedAttr->InternalSetParent(cloneAsParent);
168                 clonedAttrMap[attr->Name()] = cast<DomAttr*>(clonedAttr.Get());
169                 clonedAttrs.Add(UniquePtr<DomAttr>(cast<DomAttr*>(clonedAttr.Release())));
170             }
171             DomElement* cloneAsElement = cast<DomElement*>(clone.Get());
172             cloneAsElement->attrs = Rvalue(clonedAttrs);
173             cloneAsElement->attrMap = Rvalue(clonedAttrMap);
174             if (deep)
175             {
176                 CloneChildrenTo(cloneAsParent);
177             }
178             return clone;
179         }

221         public override void Write(CodeFormatter& formatter)
222         {
223             if (HasChildNodes())
224             {
225                 if (attrMap.IsEmpty())
226                 {
227                     formatter.Write("<" + ToUtf8(Name()) + ">");
228                 }
229                 else
230                 {
231                     formatter.Write("<" + ToUtf8(Name()));
232                     WriteAttributes(formatter);
233                     formatter.Write(">");
234                 }
235                 bool prevPreserveSpace = formatter.PreserveSpace();
236                 if (GetAttribute( u"xml:space") == u"preserve")
237                 {
238                     formatter.SetPreserveSpace(true);
239                 }
240                 bool preserveSpace = formatter.PreserveSpace() || !HasMultilineContent();
241                 if (!preserveSpace)
242                 {
243                     formatter.WriteLine();
244                     formatter.IncIndent();
245                 }
246                 base->Write(formatter);
247                 if (!preserveSpace)
248                 {
249                     formatter.DecIndent();
250                     formatter.WriteLine("</" + ToUtf8(Name()) + ">");
251                 }
252                 else if (prevPreserveSpace)
253                 {
254                     formatter.Write("</" + ToUtf8(Name()) + ">");
255                 }
256                 else
257                 {
258                     formatter.WriteLine("</" + ToUtf8(Name()) + ">");
259                 }
260                 formatter.SetPreserveSpace(prevPreserveSpace);
261             }
262             else
263             {
264                 if (attrMap.IsEmpty())
265                 {
266                     formatter.WriteLine("<" + ToUtf8(Name()) + "/>");
267                 }
268                 else
269                 {
270                     formatter.Write("<" + ToUtf8(Name()));
271                     WriteAttributes(formatter);
272                     formatter.WriteLine("/>");
273                 }
274             }

291         public void SetAttribute(const ustring& attrName, const ustring& attrValue)
292         {
293             RemoveAttribute(attrName);
294             DomAttr* attr = new DomAttr(attrName, attrValue);
295             attrMap[attrName] = attr;
296             attrs.Add(UniquePtr<DomAttr>(attr));
297         }
298         public void RemoveAttribute(const ustring& attrName)
299         {
300             Map<ustring, DomAttr*>.Iterator it = attrMap.Find(attrName);
301             if (it != attrMap.End())
302             {
303                 DomAttr* toRemove = it->second;
304                 List<UniquePtr<DomAttr>>.Iterator e = attrs.End();
305                 for (List<UniquePtr<DomAttr>>.Iterator i = attrs.Begin(); i != e; ++i;)
306                 {
307                     DomAttr* attr = (*i).Get();
308                     if (attr == toRemove)
309                     {
310                         attrs.Remove(i);
311                         break;
312                     }
313                 }
314                 attrMap.Remove(it);
315             }
316         }
317         public override void WalkAttribute(NodeOp& nodeOp)
318         {
319             for (const Pair<ustring, DomAttr*>& p : attrMap)
320             {
321                 DomAttr* attr = p.second;
322                 nodeOp.Apply(attr);
323             }
324         }

332         private void WriteAttributes(CodeFormatter& formatter)
333         {
334             for (const Pair<ustring, DomAttr*>& p : attrMap)
335             {
336                 p.second->Write(formatter);
337             }
338         }
339         private nothrow bool HasMultilineContent() const
340         {
341             if (FirstChild() != LastChild()) return true;
342             DomNode* child = FirstChild();
343             if (child != null)
344             {
345                 if ((child is DomElement*) || (child is DomDocument*))
346                 {
347                     return true;
348                 }
349                 if (child->ValueContainsNewLine()) return true;
350             }
351             return false;
352         }
