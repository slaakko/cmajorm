 12     public class DomDocument : DomParentNode
 13     {
 14         public nothrow DomDocument() : base( u"document")
 15         {
 16         }
 17         suppress DomDocument(const DomDocument&);
 18         suppress void operator=(const DomDocument&);
 19         suppress DomDocument(DomDocument&&);
 20         suppress void operator=(DomDocument&&);
 21         public inline nothrow DomElement* DocumentElement() const
 22         {
 23             return documentElement;
 24         }
 25         public override UniquePtr<DomNode> CloneNode(bool deep) const
 26         {
 27             UniquePtr<DomNode> clonedDocument(new DomDocument());
 28             if (deep)
 29             {
 30                 DomParentNode* parentNode = cast<DomParentNode*>(clonedDocument.Get());
 31                 CloneChildrenTo(parentNode);
 32             }
 33             return clonedDocument;
 34         }
 35         public override DomNode* InsertBefore(UniquePtr<DomNode>&& newChild, DomNode* refChild)
 36         {
 37             CheckValidInsert(newChild.Get(), refChild);
 38             if (newChild.Get() is DomElement*)
 39             {
 40                 documentElement = cast<DomElement*>(newChild.Get());
 41             }
 42             return base->InsertBefore(newChild, refChild);
 43         }
 44         public override UniquePtr<DomNode> RemoveChild(DomNode* oldChild)
 45         {
 46             if (oldChild == null)
 47             {
 48                 throw DomException("could not remove node: given old child is null");
 49             }
 50             if (oldChild->Parent() != this)
 51             {
 52                 throw DomException("could not remove node: given old child is not child of this node");
 53             }
 54             if (oldChild is DomElement*)
 55             {
 56                 documentElement = null;
 57             }
 58             return base->RemoveChild(oldChild);
 59         }
 60         public override UniquePtr<DomNode> ReplaceChild(UniquePtr<DomNode>&& newChild, DomNode* oldChild)
 61         {
 62             if (oldChild == null)
 63             {
 64                 throw DomException("could not replace node: given old child is null");
 65             }
 66             if (oldChild->Parent() != this)
 67             {
 68                 throw DomException("could not replace node: given old child is not child of this node");
 69             }
 70             CheckValidInsert(newChild.Get(), null);
 71             if (newChild.Get() is DomElement*)
 72             {
 73                 UniquePtr<DomNode> removed = RemoveChild(oldChild);
 74                 AppendChild(newChild);
 75                 return removed;
 76             }
 77             else
 78             {
 79                 return base->ReplaceChild(newChild, oldChild);
 80             }
 81         }
 82         public override DomNode* AppendChild(UniquePtr<DomNode>&& newChild)
 83         {
 84             CheckValidInsert(newChild.Get(), null);
 85             if (newChild.Get() is DomElement*)
 86             {
 87                 documentElement = cast<DomElement*>(newChild.Get());
 88             }
 89             return base->AppendChild(newChild);
 90         }
 91         public DomElement* GetElementById(const ustring& elementId)
 92         {
 93             if (!indexValid)
 94             {
 95                 elementsByIdMap.Clear();
 96                 BuildIndexVisitor visitor(elementsByIdMap);
 97                 Accept(visitor);
 98                 indexValid = true;
 99             }
100             HashMap<ustring, DomElement*>.ConstIterator it = elementsByIdMap.CFind(elementId);
101             if (it != elementsByIdMap.CEnd())
102             {
103                 DomElement* element = it->second;
104                 return element;
105             }
106             return null;
107         }

139         {

142         private void CheckValidInsert(DomNode* node, DomNode* refNode)
143         {
144             if (node is DomElement*)
145             {
146                 if (refNode != null || documentElement != null)
147                 {
148                     throw DomException("attempt to insert a second element to a document");
149                 }
150             }
151         }
152         private DomElement* documentElement;
153         private HashMap<ustring, DomElement*> elementsByIdMap;
154         private bool indexValid;
155         private bool xmlStandalone;
156         private ustring xmlVersion;
157         private ustring xmlEncoding;
158     }

160     public class BuildIndexVisitor : Visitor
161     {
162         public nothrow BuildIndexVisitor(HashMap<ustring, DomElement*>& elementsByIdMap_) : elementsByIdMap(elementsByIdMap_)
163         {
164         }
165         public override void BeginVisit(DomElement* element)
166         {
167             const ustring& id = element->GetAttribute( u"id");
168             if (!id.IsEmpty())
169             {
170                 elementsByIdMap[id] = element;
171             }
172         }
173         private HashMap<ustring, DomElement*>& elementsByIdMap;
174     }
