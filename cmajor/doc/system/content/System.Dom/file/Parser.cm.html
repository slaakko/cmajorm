 43         public override void Comment(const ustring& comment)
 44         {
 45             AddTextContent();
 46             UniquePtr<DomNode> commentNode(new DomComment(comment));
 47             if (!currentElement.IsNull())
 48             {
 49                 currentElement->AppendChild(Rvalue(commentNode));
 50             }
 51             else
 52             {
 53                 document->AppendChild(Rvalue(commentNode));
 54             }
 55         }

 61             {

 81         public override void StartElement(const ustring& namespaceUri, const ustring& localName, const ustring& qualifiedName, const System.Xml.Attributes& attributes)
 82         {
 83             AddTextContent();
 84             elementStack.Push(Rvalue(currentElement));
 85             List<UniquePtr<DomAttr>> attrs;
 86             Map<ustring, DomAttr*> attrMap;
 87             for (const System.Xml.Attribute& attribute : attributes)
 88             {
 89                 DomAttr* attr = new DomAttr(attribute.QualifiedName(), attribute.Value());
 90                 attrs.Add(UniquePtr<DomAttr>(attr));
 91                 attrMap[attribute.QualifiedName()] = attr;
 92             }
 93             currentElement.Reset(new DomElement(qualifiedName, Rvalue(attrs), Rvalue(attrMap)));
 94             if (!namespaceUri.IsEmpty())
 95             {
 96                 currentElement->InternalSetNamespaceUri(namespaceUri);
 97             }
 98         }
 99         public override void EndElement(const ustring& namespaceUri, const ustring& localName, const ustring& qualifiedName)
100         {
101             AddTextContent();
102             UniquePtr<DomElement> parentElement = elementStack.Pop();
103             if (!parentElement.IsNull())
104             {
105                 parentElement->AppendChild(UniquePtr<DomNode>(currentElement.Release()));
106                 currentElement = Rvalue(parentElement);
107             }
108             else
109             {
110                 document->AppendChild(UniquePtr<DomNode>(currentElement.Release()));
111             }
112         }
113         public override void SkippedEntity(const ustring& entityName)
114         {
115         }
116         private void AddTextContent()
117         {
118             if (!currentElement.IsNull())
119             {
120                 ustring text = TrimAll(textContent);
121                 if (!text.IsEmpty())
122                 {
123                     UniquePtr<DomNode> textNode(new DomText( u"text", text));
124                     currentElement->AppendChild(Rvalue(textNode));
125                 }
126             }
127             textContent.Clear();
128         }
129         private UniquePtr<DomDocument> document;
130         private UniquePtr<DomElement> currentElement;
131         private Stack<UniquePtr<DomElement>> elementStack;
132         private ustring textContent;
133     }
