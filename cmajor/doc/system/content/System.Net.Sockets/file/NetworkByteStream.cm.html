 6 using System;
 7 using System.IO;
 8 
 9 namespace System.Net.Sockets
10 {
11     public class NetworkByteStream : ByteStream
12     {
13         public nothrow NetworkByteStream(TcpSocket&& socket_) : socket(socket_)
14         {
15         }
16         public override int ReadByte()
17         {
18             byte buf;
19             long bytesRead = Read(&buf, 1);
20             if (bytesRead == 0)
21             {
22                 return -1;
23             }
24             else
25             {
26                 return buf;
27             }
28         }
29         public override long Read(byte* buf, long count)
30         {
31             #assert(buf != null);
32             #assert(count >= 0);
33             return socket.Receive(buf, cast<int>(count));
34         }
35         public override void Write(byte x)
36         {
37             Write(&x, 1);
38         }
39         public override void Write(byte* buf, long count)
40         {
41             #assert(buf != null);
42             #assert(count >= 0);
43             socket.Send(buf, cast<int>(count));
44         }
45         public TcpSocket& ContainedSocket()
46         {
47             return socket;
48         }
49         private TcpSocket socket;
50     }
51 }
