  6 using System;
  7 
  8 namespace System.Net.Sockets
  9 {
 10     public enum ShutdownMode : int
 11     {
 12         receive = 0, send = 1, both = 2
 13     }
 14     
 15     public enum ConnectOptions : int
 16     {
 17         none = 0, useTls = 1 << 0
 18     }
 19     
 20     public class SocketException : Exception
 21     {
 22         public nothrow SocketException(const string& message_) : base(message_)
 23         {
 24         }
 25     }
 26     
 27     public class TcpSocket
 28     {
 29         public TcpSocket() : handle(RtCreateSocket()), connected(false), shutdown(false)
 30         {
 31             if (handle < 0)
 32             {
 33                 string errorMessage = RtGetError(handle);
 34                 RtDisposeError(handle);
 35                 throw SocketException(errorMessage);
 36             }
 37         }
 38         public TcpSocket(const string& node_, const string& service_) : this(node_, service_, ConnectOptions.none)
 39         {
 40         }
 41         public TcpSocket(const string& node_, const string& service_, ConnectOptions options_) : 
 42             handle(RtConnectSocket(node_.Chars(), service_.Chars(), cast<int>(options_))), connected(false), shutdown(false)
 43         {
 44             if (handle < 0)
 45             {
 46                 string errorMessage = RtGetError(handle);
 47                 RtDisposeError(handle);
 48                 throw SocketException(errorMessage);
 49             }
 50             else
 51             {
 52                 connected = true;
 53             }
 54         }
 55         public nothrow TcpSocket(int handle_) : handle(handle_)
 56         {
 57         }
 58         suppress TcpSocket(const TcpSocket&);
 59         suppress void operator=(const TcpSocket&);
 60         public nothrow TcpSocket(TcpSocket&& that) : handle(that.handle)
 61         {
 62             that.handle = -1;
 63             that.connected = false;
 64             that.shutdown = false;
 65         }
 66         public nothrow void operator=(TcpSocket&& that)
 67         {
 68             Swap(handle, that.handle);
 69         }
 70         public ~TcpSocket()
 71         {
 72             if (handle > 0)
 73             {
 74                 if (connected && !shutdown)
 75                 {
 76                     try
 77                     {
 78                         Shutdown(ShutdownMode.both);
 79                     }
 80                     catch (const Exception& ex)
 81                     {
 82                     }
 83                 }
 84                 RtCloseSocket(handle);
 85             }
 86         }
 87         public void Close()
 88         {
 89             if (handle > 0)
 90             {
 91                 if (connected && !shutdown)
 92                 {
 93                     Shutdown(ShutdownMode.both);
 94                 }
 95                 int result = RtCloseSocket(handle);
 96                 if (result < 0)
 97                 {
 98                     string errorMessage = RtGetError(result);
 99                     RtDisposeError(result);
100                     throw SocketException(errorMessage);
101                 }
102                 handle = -1;
103                 connected = false;
104             }
105         }
106         public void Connect(const string& node, const string& service)
107         {
108             Connect(node, service, ConnectOptions.none);
109         }
110         public void Connect(const string& node, const string& service, ConnectOptions options)
111         {
112             Close();
113             handle = RtConnectSocket(node.Chars(), service.Chars(), cast<int>(options));
114             if (handle < 0)
115             {
116                 string errorMessage = RtGetError(handle);
117                 RtDisposeError(handle);
118                 throw SocketException(errorMessage);
119             }
120             else
121             {
122                 connected = true;
123             }
124         }
125         public void Bind(int port)
126         {
127             int result = RtBindSocket(handle, port);
128             if (result < 0)
129             {
130                 string errorMessage = RtGetError(result);
131                 RtDisposeError(result);
132                 throw SocketException(errorMessage);
133             }
134         }
135         public void Listen(int backLog)
136         {
137             int result = RtListenSocket(handle, backLog);
138             if (result < 0)
139             {
140                 string errorMessage = RtGetError(result);
141                 RtDisposeError(result);
142                 throw SocketException(errorMessage);
143             }
144         }
145         public TcpSocket Accept()
146         {
147             int acceptedHandle = RtAcceptSocket(handle);
148             if (acceptedHandle < 0)
149             {
150                 string errorMessage = RtGetError(acceptedHandle);
151                 RtDisposeError(acceptedHandle);
152                 throw SocketException(errorMessage);
153             }
154             return TcpSocket(acceptedHandle);
155         }
156         public void Shutdown(ShutdownMode mode)
157         {
158             shutdown = true;
159             int result = RtShutdownSocket(handle, cast<int>(mode));
160             if (result < 0)
161             {
162                 string errorMessage = RtGetError(result);
163                 RtDisposeError(result);
164                 throw SocketException(errorMessage);
165             }
166         }
167         public int Send(byte* buffer, int count)
168         {
169             int result = RtSendSocket(handle, buffer, count, 0);
170             if (result < 0)
171             {
172                 string errorMessage = RtGetError(result);
173                 RtDisposeError(result);
174                 throw SocketException(errorMessage);
175             }
176             return result;
177         }
178         public int Receive(byte* buffer, int count)
179         {
180             int result = RtReceiveSocket(handle, buffer, count, 0);
181             if (result < 0)
182             {
183                 string errorMessage = RtGetError(result);
184                 RtDisposeError(result);
185                 throw SocketException(errorMessage);
186             }
187             return result;
188         }
189         private int handle;
190         private bool connected;
191         private bool shutdown;
192     }
