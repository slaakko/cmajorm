 41     public class XPathFunctionLibrary
 42     {
 43         static XPathFunctionLibrary() : instance(new XPathFunctionLibrary())
 44         {
 45         }
 46         public static XPathFunctionLibrary& Instance()
 47         {
 48             return *instance;
 49         }
 50         public XPathFunction* GetFunction(const ustring& functionName) const
 51         {
 52             Map<ustring, XPathFunction*>.ConstIterator it = functionMap.CFind(functionName);
 53             if (it != functionMap.CEnd())
 54             {
 55                 return it->second;
 56             }
 57             else
 58             {
 59                 throw Exception("xpath function '" + ToUtf8(functionName) + "' not found");
 60             }
 61         }
 62         private XPathFunctionLibrary()
 63         {
 64             functions.Add(UniquePtr<XPathFunction>(new XPathBooleanFunction()));
 65             functions.Add(UniquePtr<XPathFunction>(new XPathNumberFunction()));
 66             functions.Add(UniquePtr<XPathFunction>(new XPathStringFunction()));
 67             functions.Add(UniquePtr<XPathFunction>(new XPathLastFunction()));
 68             functions.Add(UniquePtr<XPathFunction>(new XPathPositionFunction()));
 69             functions.Add(UniquePtr<XPathFunction>(new XPathCountFunction()));
 70             for (const UniquePtr<XPathFunction>& function : functions)
 71             {
 72                 functionMap[function->Name()] = function.Get();
 73             }
 74         }
 75         private static UniquePtr<XPathFunctionLibrary> instance;
 76         private Map<ustring, XPathFunction*> functionMap;
 77         private List<UniquePtr<XPathFunction>> functions;
 78     }

 80     public class XPathBooleanFunction : XPathFunction
 81     {
 82         public nothrow XPathBooleanFunction() : base( u"boolean", 1, 1)
 83         {
 84         }
 85         public override UniquePtr<XPathObject> Evaluate(XPathContext& context, List<XPathObject*>& arguments)
 86         {
 87             if (arguments.Count() != 1)
 88             {
 89                 throw XPathException("boolean function requires one argument");
 90             }
 91             XPathObject* argument = arguments[0];
 92             return UniquePtr<XPathObject>(new XPathBoolean(argument->ToBool()));
 93         }
 94     }

 96     public class XPathNumberFunction : XPathFunction
 97     {
 98         public nothrow XPathNumberFunction() : base( u"number", 0, 1)
 99         {
100         }
101         public override UniquePtr<XPathObject> Evaluate(XPathContext& context, List<XPathObject*>& arguments)
102         {
103             XPathObject* argument = null;
104             XPathNodeSet nodeSet;
105             if (arguments.IsEmpty())
106             {
107                 nodeSet.Add(context.Node());
108                 argument = &nodeSet;
109             }
110             else
111             {
112                 if (arguments.Count() != 1)
113                 {
114                     throw XPathException("boolean function requires zero or one arguments");
115                 }
116                 argument = arguments[0];
117             }
118             return UniquePtr<XPathObject>(new XPathNumber(argument->ToDouble(context)));
119         }
120     }

178     public class XPathCountFunction : XPathFunction
179     {
180         public nothrow XPathCountFunction() : base( u"count", 1, 1)
181         {
182         }
183         public override UniquePtr<XPathObject> Evaluate(XPathContext& context, List<XPathObject*>& arguments)
184         {
185             if (arguments.Count() != 1)
186             {
187                 throw XPathException("position function requires one argument");
188             }
189             XPathObject* argument = arguments[0];
190             if (!(argument is XPathNodeSet*))
191             {
192                 throw XPathException("position function requires one node set argument");
193             }
194             XPathNodeSet* nodeSet = cast<XPathNodeSet*>(argument);
195             return UniquePtr<XPathObject>(new XPathNumber(nodeSet->Length()));
196         }

199     public class ElementAndDocumentStringValueComputer : System.Dom.Visitor
200     {
201         public override void Visit(DomText* text)
202         {
203             stringValue.Append(text->Data());
204         }
205         public override void Visit(DomCDataSection* cdataSection)
206         {
207             stringValue.Append(cdataSection->Data());
208         }
209         public ustring GetStringValue()
210         {
211             return Rvalue(stringValue);
212         }
213         private ustring stringValue;
214     }
215     
216     public ustring StringValue(DomNode* node)
217     {
218         if ((node is DomDocument*) || (node is DomElement*))
219         {
220             ElementAndDocumentStringValueComputer stringValueComputer;
221             node->Accept(stringValueComputer);
222             return stringValueComputer.GetStringValue();
223         }
224         else if (node is DomAttr*)
225         {
226             DomAttr* attr = cast<DomAttr*>(node);
227             return attr->Value();
228         }
229         else if (node is DomCharacterData*)
230         {
231             DomCharacterData* characterData = cast<DomCharacterData*>(node);
232             return characterData->Data();
233         }
234         else if (node is DomProcessingInstruction*)
235         {
236             DomProcessingInstruction* pi = cast<DomProcessingInstruction*>(node);
237             return pi->Data();
238         }
239         else
240         {
241             return ustring();
242         }
243     }
