 10 namespace System.XPath
 11 {
 12     public abstract class XPathObject
 13     {
 14         public virtual default ~XPathObject();
 15         public abstract nothrow bool ToBool() const;
 16         public abstract double ToDouble(XPathContext& context) const;
 17         public abstract ustring ToString() const;
 18         public abstract UniquePtr<DomNode> ToDom() const;
 19     }
 20     
 21     public class XPathNodeSet : XPathObject
 22     {
 23         public void Add(DomNode* node)
 24         {
 25             nodes.InternalAddNode(node);
 26         }
 27         public inline nothrow DomNode* operator[](long index) const
 28         {
 29             return nodes[index];
 30         }
 31         public inline nothrow long Length() const
 32         {
 33             return nodes.Length();
 34         }
 35         public override nothrow bool ToBool() const
 36         {
 37             return nodes.Length() > 0;
 38         }
 39         public override double ToDouble(XPathContext& context) const
 40         {
 41             XPathFunction* stringFunction = GetXPathLibraryFunction( u"string");
 42             List<XPathObject*> stringArgs;
 43             XPathObject* arg = this;
 44             stringArgs.Add(arg);
 45             UniquePtr<XPathObject> asStringObject = stringFunction->Evaluate(context, stringArgs);
 46             if (!(asStringObject.Get() is XPathString*))
 47             {
 48                 throw XPathException("string result expected");
 49             }
 50             XPathString* asString = cast<XPathString*>(asStringObject.Get());
 51             return ParseDouble(ToUtf8(Trim(asString->Value())));
 52         }
 53         public override ustring ToString() const
 54         {
 55             if (nodes.Length() == 0)
 56             {
 57                 return ustring();
 58             }
 59             else
 60             {
 61                 DomNode* node = nodes[0];
 62                 return StringValue(node);
 63             }
 64         }
 65         public override UniquePtr<DomNode> ToDom() const
 66         {
 67             UniquePtr<DomElement> result(new DomElement( u"nodeSet"));
 68             result->SetAttribute( u"length", ToUtf32(ToString(nodes.Length())));
 69             long n = nodes.Length();
 70             for (long i = 0; i < n; ++i;)
 71             {
 72                 DomNode* node = nodes[i];
 73                 if (node is DomAttr*)
 74                 {
 75                     UniquePtr<DomElement> element(new DomElement( u"attribute"));
 76                     UniquePtr<DomNode> clonedAttrNode = node->CloneNode(false);
 77                     UniquePtr<DomAttr> clonedAttr(cast<DomAttr*>(clonedAttrNode.Release()));
 78                     element->AddAttribute(Rvalue(clonedAttr));
 79                     result->AppendChild(UniquePtr<DomNode>(element.Release()));
 80                 }
 81                 else
 82                 {
 83                     ustring nodeName;
 84                     if (node is DomDocument*)
 85                     {
 86                         nodeName =  u"document";
 87                     }
 88                     else if (node is DomElement*)
 89                     {
 90                          nodeName =  u"element";
 91                     }
 92                     else if (node is DomText*)
 93                     {
 94                         nodeName =  u"text";
 95                     }
 96                     else if (node is DomCDataSection*)
 97                     {
 98                         nodeName =  u"cdataSection";
 99                     }
100                     else if (node is DomComment*)
101                     {
102                         nodeName =  u"comment";
103                     }
104                     else if (node is DomProcessingInstruction*)
105                     {
106                         nodeName =  u"processingInstruction";
107                     }
108                     UniquePtr<DomElement> element(new DomElement(nodeName));
109                     element->AppendChild(node->CloneNode(true));
110                     result->AppendChild(UniquePtr<DomNode>(element.Release()));
111                 }
112             }
113             return UniquePtr<DomNode>(result.Release());
114         }
115         private NodeList nodes;
116     }

118     public class XPathBoolean : XPathObject
119     {
120         public nothrow XPathBoolean(bool value_) : value(value_)
121         {
122         }
123         public inline nothrow bool Value() const
124         {
125             return value;
126         }
127         public override nothrow bool ToBool() const
128         {
129             return value;
130         }
131         public override double ToDouble(XPathContext& context) const
132         {
133             if (value)
134             {
135                 return 1;
136             }
137             return 0;
138         }
139         public override ustring ToString() const
140         {
141             if (value)
142             {
143                 return  u"true";
144             }
145             return  u"false";
146         }
147         public override UniquePtr<DomNode> ToDom() const
148         {
149             UniquePtr<DomElement> result(new DomElement( u"boolean"));
150             result->SetAttribute( u"value", ToUtf32(ToString(value)));
151             return UniquePtr<DomNode>(result.Release());
152         }
153         private bool value;
154     }
