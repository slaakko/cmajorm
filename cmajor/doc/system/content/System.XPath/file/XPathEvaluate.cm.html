 6 using System;
 7 using System.Dom;
 8 using System.Text;
 9 
10 namespace System.XPath
11 {
12     public class XPathExpressionParser
13     {
14         static XPathExpressionParser() : instance(new XPathExpressionParser())
15         {
16         }
17         public static XPathExpressionParser& Instance()
18         {
19             return *instance;
20         }
21         public UniquePtr<XPathExpr> Parse(const ustring& xpathExpression)
22         {
23             if (DebugParsing())
24             {
25                 xpathExpressionGrammar->SetLog(&Console.Out());
26             }
27             TimePoint start = Now();
28             UniquePtr<XPathExpr> expr = xpathExpressionGrammar->Parse(xpathExpression.Chars(), xpathExpression.Chars() + xpathExpression.Length(), 0, "");
29             TimePoint end = Now();
30             if (DebugQuery())
31             {
32                 UniquePtr<DomNode> queryDom = expr->ToDom();
33                 SetQueryDom(Rvalue(queryDom));
34                 SetQueryDuration(end - start);
35             }
36             return expr;
37         }
38         private XPathExpressionParser() : xpathExpressionGrammar(new XPathExpressionGrammar())
39         {
40         }
41         private static UniquePtr<XPathExpressionParser> instance;
42         private XPathExpressionGrammar* xpathExpressionGrammar;
43     }
44     
45     public UniquePtr<XPathObject> Evaluate(const ustring& xpathExpression, DomNode* node)
46     {
47         UniquePtr<XPathExpr> xpathExpr = XPathExpressionParser.Instance().Parse(xpathExpression);
48         TimePoint start = Now();
49         XPathContext context(node, 1, 1);
50         UniquePtr<XPathObject> result = xpathExpr->Evaluate(context);
51         TimePoint end = Now();
52         if (DebugQuery())
53         {
54             SetExecutionDuration(end - start);
55         }
56         return result;
57     }
58     public UniquePtr<XPathObject> Evaluate(const ustring& xpathExpression, DomDocument* document)
59     {
60         return Evaluate(xpathExpression, cast<DomNode*>(document));
61     }
62 }
