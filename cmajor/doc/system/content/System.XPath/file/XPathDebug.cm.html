 6 using System;
 7 using System.Dom;
 8 
 9 namespace System.XPath
10 {
11     public class XPathDebug
12     {
13         static nothrow XPathDebug() : instance(new XPathDebug())
14         {
15         }
16         public static nothrow XPathDebug& Instance()
17         {
18             return *instance;
19         }
20         public inline nothrow bool DebugParsing() const
21         {
22             return debugParsing;
23         }
24         public inline nothrow void SetDebugParsing()
25         {
26             debugParsing = true;
27         }
28         public inline nothrow bool DebugQuery() const
29         {
30             return debugQuery;
31         }
32         public inline nothrow void SetDebugQuery()
33         {
34             debugQuery = true;
35         }
36         public nothrow void SetQueryDom(UniquePtr<DomNode>&& queryDom_)
37         {
38             queryDom = queryDom_;
39         }
40         public nothrow UniquePtr<DomNode> GetQueryDom()
41         {
42             return Rvalue(queryDom);
43         }
44         public nothrow void SetQueryDuration(Duration queryDuration_)
45         {
46             queryDuration = queryDuration_;
47         }
48         public nothrow Duration GetQueryDuration()
49         {
50             return queryDuration;
51         }
52         public nothrow void SetExecutionDuration(Duration executionDuration_)
53         {
54             executionDuration = executionDuration_;
55         }
56         public nothrow Duration GetExecutionDuration()
57         {
58             return executionDuration;
59         }
60         private nothrow XPathDebug() : debugParsing(false), debugQuery(false)
61         {
62         }
63         private static UniquePtr<XPathDebug> instance;
64         private bool debugParsing;
65         private bool debugQuery;
66         private UniquePtr<DomNode> queryDom;
67         private Duration queryDuration;
68         private Duration executionDuration;
69     }
70 
71     public nothrow bool DebugParsing()
72     {
73         return XPathDebug.Instance().DebugParsing();
74     }
75     
76     public nothrow bool DebugQuery()
77     {
78         return XPathDebug.Instance().DebugQuery();
79     }
80     
81     public nothrow void SetQueryDom(UniquePtr<DomNode>&& queryDom)
82     {
83         XPathDebug.Instance().SetQueryDom(queryDom);
84     }
85     
86     public nothrow void SetQueryDuration(Duration queryDuration)
87     {
88         XPathDebug.Instance().SetQueryDuration(queryDuration);
89     }
90     
91     public nothrow void SetExecutionDuration(Duration executionDuration)
92     {
93         XPathDebug.Instance().SetExecutionDuration(executionDuration);
94     }
95 }
