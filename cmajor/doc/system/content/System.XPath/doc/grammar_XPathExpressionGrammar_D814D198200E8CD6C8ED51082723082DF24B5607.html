XPathExpressionGrammar Grammar

Expr → OrExpr
OrExpr → AndExpr (keyword("or") AndExpr)*
AndExpr → EqualityExpr (keyword("and") EqualityExpr)*
EqualityExpr → RelationalExpr (('=' | "!=") RelationalExpr)*
RelationalExpr → AdditiveExpr (("<=" | ">=" | '<' | '>') AdditiveExpr)*
AdditiveExpr → MultiplicativeExpr (('+' | '-') MultiplicativeExpr)*
MultiplicativeExpr → UnaryExpr (('*' | "div" | "mod") UnaryExpr)*
UnaryExpr → '-' UnaryExpr | UnionExpr
UnionExpr → PathExpr ('|' PathExpr)*
PathExpr → (LocationPath − FunctionCall) | FilterExpr (("//" | "/") RelativeLocationPath)?
FilterExpr → PrimaryExpr (Predicate)*
LocationPath → AbsoluteLocationPath | RelativeLocationPath
AbsoluteLocationPath → AbbreviatedAbsoluteLocationPath | '/' (RelativeLocationPath)?
RelativeLocationPath → Step (("//" | '/') Step)*
AbbreviatedAbsoluteLocationPath → "//" RelativeLocationPath
Step → AxisSpecifier NodeTest (Predicate)* | AbbreviatedStep
AbbreviatedStep → ".." | '.'
AxisSpecifier → (AxisName "::") | AbbreviatedAxisSpecifier
AxisName → keyword_list(identifier, ["namespace", "descendant", "parent", "preceding", "ancestor", "preceding-sibling", "self", "following", "following-sibling", "descendant-or-self", "child", "attribute", "ancestor-or-self"])
AbbreviatedAxisSpecifier → '@' | empty
Predicate → '[' PredicateExpr ']'
PredicateExpr → Expr
NodeTest → (keyword("processing-instruction") '(' Literal ')') | (NodeType '(' ')') | NameTest
NameTest → '*' | (NCName ':' '*') | QName
NodeType → keyword("comment") | keyword("text") | keyword("processing-instruction") | keyword("node")
PrimaryExpr → FunctionCall | VariableReference | '(' Expr ')' | Literal | Number
VariableReference → token('$' QName)
Literal → token('\"' ([^\"]*) '\"') | token('\'' ([^']*) '\'')
Number → token(Digits ('.' Digits?)?) | token('.' Digits)
Digits → token([0-9]+)
FunctionCall → token(FunctionName '(') (Argument (',' Argument)*)? ')'
Argument → Expr
FunctionName → (QName − NodeType)
QName → PrefixedName | UnprefixedName
PrefixedName → token(Prefix ':' LocalPart)
UnprefixedName → LocalPart
Prefix → NCName
LocalPart → NCName
NCName → token((NameStartChar − ':') (NameChar − ':')*)
