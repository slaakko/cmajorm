  6 using System;
  7 using System.IO;
  8 
  9 namespace System.Numerics.Multiprecision
 10 {
 11     public class BigInt
 12     {
 13         public nothrow BigInt() : handle(RtCreateBigInt())
 14         {
 15         }
 16         public nothrow BigInt(int that) : handle(RtCreateBigInt())
 17         {
 18             RtAssignBigIntInt(handle, that);
 19         }
 20         public nothrow BigInt(uint that) : handle(RtCreateBigInt())
 21         {
 22             RtAssignBigIntUInt(handle, that);
 23         }
 24         public BigInt(const string& str) : this(str, 10)
 25         {
 26         }
 27         public BigInt(const string& str, int base_) : handle(RtCreateBigInt())
 28         {
 29             int result = RtAssignBigIntStr(handle, str.Chars(), base_);
 30             if (result != 0)
 31             {
 32                 ThrowConversionException("invalid digit string for initializing BigInt");
 33             }
 34         }
 35         public ~BigInt()
 36         {
 37             if (handle != null)
 38             {
 39                 RtDestroyBigInt(handle);
 40             }
 41         }
 42         public nothrow BigInt(const BigInt& that) : handle(RtCreateBigInt())
 43         {
 44             RtAssignBigIntBigInt(handle, that.handle);
 45         }
 46         public nothrow BigInt(BigInt&& that) : handle(that.handle)
 47         {
 48             that.handle = null;
 49         }
 50         public nothrow void operator=(const BigInt& that)
 51         {
 52             if (handle != null)
 53             {
 54                 RtDestroyBigInt(handle);
 55             }
 56             handle = RtCreateBigInt();
 57             RtAssignBigIntBigInt(handle, that.handle);
 58         }
 59         public default nothrow void operator=(BigInt&& that);
 60         public nothrow void operator=(int that)
 61         {
 62             if (handle != null)
 63             {
 64                 RtDestroyBigInt(handle);
 65             }
 66             handle = RtCreateBigInt();
 67             RtAssignBigIntInt(handle, that);
 68         }
 69         public nothrow void operator=(uint that)
 70         {
 71             if (handle != null)
 72             {
 73                 RtDestroyBigInt(handle);
 74             }
 75             handle = RtCreateBigInt();
 76             RtAssignBigIntUInt(handle, that);
 77         }
 78         public nothrow string ToString() const
 79         {
 80             return this->ToString(10);
 81         }
 82         public nothrow string ToString(int base_) const
 83         {
 84             int strHandle = RtGetBigIntStrHandle(handle, base_);
 85             string s = RtGetBigIntStr(strHandle);
 86             RtFreeBigIntStr(strHandle);
 87             return s;
 88         }
 89         public inline nothrow void* Handle() const
 90         {
 91             return handle;
 92         }
 93         private void* handle;
 94     }
