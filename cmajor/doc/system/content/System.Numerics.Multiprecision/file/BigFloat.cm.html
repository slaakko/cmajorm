161         public nothrow string ToString(int base_, uint numDigits) const
162         {
163             long exponent = 0;
164             int strHandle = RtGetBigFloatStrHandle(handle, base_, numDigits, &exponent);
165             string s = RtGetBigFloatStr(strHandle);
166             RtFreeBigFloatStr(strHandle);
167             string t;
168             if (!s.IsEmpty() && s[0] == '-')
169             {
170                 t.Append('-');
171                 s = s.Substring(1);
172             }
173             t.Append("0.").Append(s).Append("e").Append(System.ToString(exponent));
174             return t;
175         }

179         }
