  6 using System;
  7 using System.IO;
  8 
  9 namespace System.Numerics.Multiprecision
 10 {
 11     public class BigRational
 12     {
 13         public nothrow BigRational() : handle(RtCreateBigRational())
 14         {
 15         }
 16         public nothrow BigRational(int that) : handle(RtCreateBigRational())
 17         {
 18             RtSetBigRationalInt(handle, that, 1);
 19         }
 20         public nothrow BigRational(uint that) : handle(RtCreateBigRational())
 21         {
 22             RtSetBigRationalUInt(handle, that, 1u);
 23         }
 24         public nothrow BigRational(const BigInt& that) : handle(RtCreateBigRational())
 25         {
 26             RtSetBigRationalBigInt(handle, that.Handle());
 27         }
 28         public BigRational(const string& str) : this(str, 10)
 29         {
 30         }
 31         public BigRational(const string& str, int base_) : handle(RtCreateBigRational())
 32         {
 33             int result = RtAssignBigRationalStr(handle, str.Chars(), base_);
 34             if (result != 0)
 35             {
 36                 ThrowConversionException("invalid digit string for initializing BigRational");
 37             }
 38         }
 39         public ~BigRational()
 40         {
 41             if (handle != null)
 42             {
 43                 RtDestroyBigRational(handle);
 44             }
 45         }
 46         public nothrow BigRational(const BigRational& that) : handle(RtCreateBigRational())
 47         {
 48             RtAssignBigRational(handle, that.handle);
 49         }
 50         public nothrow BigRational(BigRational&& that) : handle(that.handle)
 51         {
 52             that.handle = null;
 53         }
 54         public nothrow void operator=(const BigRational& that)
 55         {
 56             if (handle != null)
 57             {
 58                 RtDestroyBigRational(handle);
 59             }
 60             handle = RtCreateBigRational();
 61             RtAssignBigRational(handle, that.handle);
 62         }
 63         public default nothrow void operator=(BigRational&& that);
 64         public nothrow BigInt Numerator() const
 65         {
 66             BigInt numerator;
 67             RtGetNumeratorBigRational(numerator.Handle(), handle);
 68             return numerator;
 69         }
 70         public nothrow BigInt Denominator() const
 71         {
 72             BigInt denominator;
 73             RtGetDenominatorBigRational(denominator.Handle(), handle);
 74             return denominator;
 75         }
 76         public nothrow string ToString() const
 77         {
 78             return this->ToString(10);
 79         }
 80         public nothrow string ToString(int base_) const
 81         {
 82             int strHandle = RtGetBigRationalStrHandle(handle, base_);
 83             string s = RtGetBigRationalStr(strHandle);
 84             RtFreeBigRationalStr(strHandle);
 85             return s;
 86         }
 87         public inline nothrow void* Handle() const
 88         {
 89             return handle;
 90         }
 91         private void* handle;
 92     }
 93     
 94     public nothrow BigRational operator-(const BigRational& x)
 95     {
 96         BigRational result;
 97         RtNegBigRational(result.Handle(), x.Handle());
 98         return result;
 99     }
100     
101     public nothrow BigRational Abs(const BigRational& x)
102     {
103         BigRational result;
104         RtAbsBigRational(result.Handle(), x.Handle());
105         return result;
106     }
107 
108     public nothrow BigRational operator+(const BigRational& left, const BigRational& right)
109     {
110         BigRational result;
111         RtAddBigRational(result.Handle(), left.Handle(), right.Handle());
112         return result;
113     }
114     
115     public nothrow BigRational operator-(const BigRational& left, const BigRational& right)
116     {
117         BigRational result;
118         RtSubBigRational(result.Handle(), left.Handle(), right.Handle());
119         return result;
120     }
121     
122     public nothrow BigRational operator*(const BigRational& left, const BigRational& right)
123     {
124         BigRational result;
125         RtMulBigRational(result.Handle(), left.Handle(), right.Handle());
126         return result;
127     }
128     
129     public nothrow BigRational operator/(const BigRational& left, const BigRational& right)
130     {
131         BigRational result;
132         RtDivBigRational(result.Handle(), left.Handle(), right.Handle());
133         return result;
134     }
135     
136     public nothrow bool operator==(const BigRational& left, const BigRational& right)
137     {
138         return RtEqualBigRational(left.Handle(), right.Handle()) != 0;
139     }
140 
141     public nothrow bool operator<(const BigRational& left, const BigRational& right)
142     {
143         return RtCmpBigRational(left.Handle(), right.Handle()) < 0;
144     }
145     
146     public OutputStream& operator<<(OutputStream& s, const BigRational& x)
147     {
148         return s << x.ToString();
149     }
150     
151     public StreamWriter& operator<<(StreamWriter& writer, const BigRational& x)
152     {
153         return writer << x.ToString();
154     }
155 }
