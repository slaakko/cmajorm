 93         public void BeginStartTag(const ustring& tagName)
 94         {
 95             tagStack.Push(Rvalue(currentTagName));
 96             currentTagName = tagName;
 97             namespaceUriStack.Push(currentNamespaceUri);
 98             namespacePrefixStack.Push(currentNamespacePrefix);
 99             attributes.Clear();
100         }
101         public void EndStartTag(const Span& span, const string& systemId)
102         {
103             ustring localName;
104             ustring prefix;
105             ParseQualifiedName(currentTagName, localName, prefix, span, systemId);
106             if (prefix ==  u"xmlns")
107             {
108                 throw XmlProcessingException(GetErrorLocationStr(systemId, span) + ": 'xmlns' prefix cannot be declared for an element");
109             }
110             contentHandler->StartElement(GetNamespaceUri(prefix, span, systemId), localName, currentTagName, attributes);
111         }
112         public void EndTag(const ustring& tagName, const Span& span, const string& systemId)
113         {
114             if (tagStack.IsEmpty())
115             {
116                 throw XmlProcessingException(GetErrorLocationStr(systemId, span) + ": end tag '" + ToUtf8(tagName) + "' has no corresponding start tag");
117             }
118             if (tagName != currentTagName)
119             {
120                 throw XmlProcessingException(GetErrorLocationStr(systemId, span) + ": end tag '" + ToUtf8(tagName) + "' does not match start tag '" + ToUtf8(currentTagName) + "'");
121             }
122             ustring localName;
123             ustring prefix;
124             ParseQualifiedName(currentTagName, localName, prefix, span, systemId);
125             if (prefix ==  u"xmlns")
126             {
127                 throw XmlProcessingException(GetErrorLocationStr(systemId, span) + ": 'xmlns' prefix cannot be declared for an element");
128             }
129             contentHandler->EndElement(GetNamespaceUri(prefix, span, systemId), localName, currentTagName);
130             if (namespaceUriStack.IsEmpty())
131             {
132                 throw XmlProcessingException("namespace URI stack is empty");
133             }
134             currentNamespaceUri = namespaceUriStack.Pop();
135             namespacePrefixMap.Remove(currentNamespacePrefix);
136             if (namespacePrefixStack.IsEmpty())
137             {
138                 throw XmlProcessingException("namespace prefix stack is empty");
139             }
140             currentNamespacePrefix = namespacePrefixStack.Pop();
141             namespacePrefixMap[currentNamespacePrefix] = currentNamespaceUri;
142             currentTagName = tagStack.Pop();
143         }
144         public void AddAttribute(const ustring& attName, const ustring& attValue, const Span& span, const string& systemId)
145         {
146             ustring localName;
147             ustring prefix;
148             ParseQualifiedName(attName, localName, prefix, span, systemId);
149             if (prefix ==  u"xmlns")
150             {
151                 currentNamespacePrefix = localName;
152                 currentNamespaceUri = attValue;
153                 namespacePrefixMap[currentNamespacePrefix] = currentNamespaceUri;
154             }
155             else if (localName ==  u"xmlns")
156             {
157                 currentNamespacePrefix.Clear();
158                 currentNamespaceUri = attValue;
159             }
160             else
161             {
162                 attributes.Add(Attribute(GetNamespaceUri(prefix, span, systemId), localName, attName, attValue));
163             }
164         }
165         public void EntityRef(const ustring& entityName, const Span& span, const string& systemId)
166         {
167             HashMap<ustring, Entity*>.ConstIterator it = entityMap.CFind(entityName);
168             if (it != entityMap.CEnd())
169             {
170                 Entity* entity = it->second;
171                 if (entity is InternalEntity*)
172                 {
173                     InternalEntity* internalEntity = cast<InternalEntity*>(entity);
174                     const ustring& entityValue = internalEntity->Value();
175                     if (attValue != null)
176                     {
177                         attValue->Append(entityValue);
178                     }
179                     else
180                     {
181                         Text(entityValue);
182                     }
183                 }
184                 else
185                 {
186                     contentHandler->SkippedEntity(entityName);
187                 }
188             }
189             else
190             {
191                 contentHandler->SkippedEntity(entityName);
192             }
193         }

205             attValue = attValueStack.Pop();
206         }

222         private nothrow int GetErrorColumn(int index) const
223         {
224             int errorColumn = 0;
225             while (index > 0 && contentStart[index] != '\n' && contentStart[index] != '\r')
226             {
227                 ++errorColumn;
228                 --index;
229             }
230             if (errorColumn == 0)
231             {
232                 errorColumn = 1;
233             }
234             return errorColumn;
235         }
236         private string GetErrorLocationStr(const string& systemId, const Span& span) const
237         {
238             int errorColumn = GetErrorColumn(span.start);
239             return "error in '" + systemId + "' at line " + ToString(span.lineNumber) + " column " + ToString(errorColumn) + " (index " + ToString(span.start) + ")";
240         }
241         private void ParseQualifiedName(const ustring& qualifiedName, ustring& localName, ustring& prefix, const Span& span, const string& systemId)
242         {
243             List<ustring> parts = qualifiedName.Split(':');
244             if (parts.Count() > 2)
245             {
246                 throw XmlProcessingException(GetErrorLocationStr(systemId, span) + ": qualified name '" + ToUtf8(qualifiedName) + "' has more than one ':' character");
247             }
248             else if (parts.Count() == 2)
249             {
250                 prefix = parts[0];
251                 localName = parts[1];
252             }
253             else
254             {
255                 prefix.Clear();
256                 localName = qualifiedName;
257             }
258         }
259         private ustring GetNamespaceUri(const ustring& namespacePrefix, const Span& span, const string& systemId)
260         {
261             if (namespacePrefix.IsEmpty())
262             {
263                 return currentNamespaceUri;
264             }
265             HashMap<ustring, ustring>.ConstIterator it = namespacePrefixMap.CFind(namespacePrefix);
266             if (it != namespacePrefixMap.CEnd())
267             {
268                 return it->second;
269             }
270             else
271             {
272                 throw XmlProcessingException(GetErrorLocationStr(systemId, span) + ": namespace prefix '" + ToUtf8(namespacePrefix) + "' not bound to any namespace URI");
273             }
274         }
