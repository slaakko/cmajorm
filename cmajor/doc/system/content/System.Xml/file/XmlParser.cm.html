 6 using System;
 7 using System.IO;
 8 
 9 namespace System.Xml
10 {
11     public class XmlParser
12     {
13         static XmlParser() : instance(new XmlParser())
14         {
15         }
16         public static XmlParser& Instance()
17         {
18             return *instance;
19         }
20         private XmlParser() : xmlGrammar(new XmlGrammar())
21         {
22         }
23         public void Parse(const uchar* contentStart, const uchar* contentEnd, const string& systemId, XmlProcessor* processor)
24         {
25             xmlGrammar->Parse(contentStart, contentEnd, 0, systemId, processor);
26         }
27         private static UniquePtr<XmlParser> instance;
28         private XmlGrammar* xmlGrammar;
29     }
30     
31     public void ParseXmlFile(const string& xmlFileName, XmlContentHandler* contentHandler)
32     {
33         string xmlContent = File.ReadAllText(xmlFileName);
34         ParseXmlContent(xmlContent, xmlFileName, contentHandler);
35     }
36     
37     public void ParseXmlContent(const string& xmlContent, const string& systemId, XmlContentHandler* contentHandler)
38     {
39         ParseXmlContent(ToUtf32(xmlContent), systemId, contentHandler);
40     }
41     
42     public void ParseXmlContent(const wstring& xmlContent, const string& systemId, XmlContentHandler* contentHandler)
43     {
44         ParseXmlContent(ToUtf32(xmlContent), systemId, contentHandler);
45     }
46     
47     public void ParseXmlContent(const ustring& xmlContent, const string& systemId, XmlContentHandler* contentHandler)
48     {
49         const uchar* contentStart = xmlContent.Chars();
50         const uchar* contentEnd = xmlContent.Chars() + xmlContent.Length();
51         XmlProcessor xmlProcessor(contentStart, contentEnd, contentHandler);
52         XmlParser.Instance().Parse(contentStart, contentEnd, systemId, &xmlProcessor);
53     }
