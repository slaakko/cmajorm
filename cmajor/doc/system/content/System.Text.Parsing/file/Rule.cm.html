171         public override Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
172         {
173             bool writeToLog = !scanner.Skipping() && scanner.Log() != null;
174             if (writeToLog)
175             {
176                 XmlLog* log = scanner.Log();
177                 log->WriteBeginRule(Name());
178                 log->IncIndent();
179                 log->WriteTry(scanner.RestOfLine(log->MaxLineLength()));
180                 log->IncIndent();
181             }
182             int startIndex = scanner.GetSpan().start;
183             Enter(stack, parsingData);
184             Match match = Match.Nothing();
185             if (definition != null)
186             {
187                 match = definition->Parse(scanner, stack, parsingData);
188             }
189             Leave(stack, parsingData, match.Hit());
190             if (writeToLog)
191             {
192                 XmlLog* log = scanner.Log();
193                 log->DecIndent();
194                 if (match.Hit())
195                 {
196                     ustring matched(scanner.Start() + startIndex, scanner.Start() + scanner.GetSpan().start);
197                     log->WriteSuccess(matched);
198                 }
199                 else
200                 {
201                     log->WriteFail();
202                 }
203                 log->DecIndent();
204                 log->WriteEndRule(Name());
205             }
206             return match;
207         }

214             }

217         public void ExpandCode()
218         {
219             HashMap<ustring, ustring> expandMap;
220             expandMap[ u"value"] = u"context->value";
221             for (NonterminalParser* nonterminal : nonterminals)
222             {
223                 expandMap[nonterminal->Name()] =  u"context->" + nonterminal->ValueFieldName();
224             }
225             for (const AttrOrVar& attr : inheritedAttributes)
226             {
227                 expandMap[attr.Name()] =  u"context->" + attr.Name();
228             }
229             for (const AttrOrVar& var : localVariables)
230             {
231                 expandMap[var.Name()] =  u"context->" + var.Name();
232             }
233             CodeExpandingVisitor visitor(expandMap);
234             for (ActionParser* action : actions)
235             {
236                 action->SuccessCode()->Accept(visitor);
237                 if (action->FailureCode() != null)
238                 {
239                     action->FailureCode()->Accept(visitor);
240                 }
241             }
242             for (NonterminalParser* nonterminal : nonterminals)
243             {
244                 for (const UniquePtr<System.Text.Parsing.CodeDom.DomObject>& argument : nonterminal->Arguments())
245                 {
246                     argument->Accept(visitor);
247                 }
248             }
249         }

260     public nothrow ustring GetPrefix(const ustring& fullName)
261     {
262         long dotPos = fullName.Find( u'.');
263         if (dotPos != -1)
264         {
265             return fullName.Substring(0, dotPos);
266         }
267         return fullName;
268     }
269     
270     public nothrow ustring CommonName(const ustring& qualifiedName)
271     {
272         long lastDotPos = qualifiedName.RFind( u'.');
273         if (lastDotPos != -1)
274         {
275             return qualifiedName.Substring(lastDotPos + 1);
276         }
277         return qualifiedName;
278     }
