  6 using System;
  7 using System.Collections;
  8 
  9 namespace System.Text.Parsing
 10 {
 11     public class LinkerVisitor : Visitor
 12     {
 13         public LinkerVisitor() : currentGrammar(null), currentRule(null), actionNumber(0)
 14         {
 15         }
 16         public override void BeginVisit(Grammar& grammar)
 17         {
 18             if (!grammar.IsExternal())
 19             {
 20                 currentGrammar = &grammar;
 21             }
 22             else
 23             {
 24                 currentGrammar = null;
 25             }
 26         }
 27         public override void EndVisit(Grammar& grammar)
 28         {
 29             if (currentGrammar != null)
 30             {
 31                 currentGrammar->ResolveStartRule();
 32                 currentGrammar->ResolveSkipRule();
 33             }
 34         }
 35         public override void Visit(RuleLink& ruleLink)
 36         {
 37             if (currentGrammar == null)
 38             {
 39                 return;
 40             }
 41             ParsingObject* o = currentGrammar->GetScope()->Get(ruleLink.LinkedRuleName());
 42             if (o != null)
 43             {
 44                 if (o is RuleParser*)
 45                 {
 46                     RuleParser* rule = cast<RuleParser*>(o);
 47                     ruleLink.SetRule(rule);
 48                     currentGrammar->AddGrammarReference(rule->GetGrammar());
 49                     ruleLink.AddToScope();
 50                 }
 51                 else
 52                 {
 53                     ThrowException("'" + ToUtf8(ruleLink.LinkedRuleName()) + "' is not a rule", ruleLink.GetSpan());
 54                 }
 55             }
 56             else
 57             {
 58                 ThrowException("rule '" + ToUtf8(ruleLink.LinkedRuleName()) + "' not found", ruleLink.GetSpan());
 59             }
 60         }
 61         public override void BeginVisit(RuleParser& rule)
 62         {
 63             if (currentGrammar != null)
 64             {
 65                 currentRule = &rule;
 66                 actionNumber = 0;
 67             }
 68             else
 69             {
 70                 currentRule = null;
 71             }
 72         }
 73         public override void EndVisit(RuleParser& rule)
 74         {
 75             if (currentGrammar != null)
 76             {
 77                 rule.Link();
 78             }
 79         }
 80         public override void BeginVisit(ActionParser& parser)
 81         {
 82             if (currentGrammar != null)
 83             {
 84                 if (parser.Name().IsEmpty())
 85                 {
 86                     parser.SetName( u"A" + ToUtf32(ToString(actionNumber++)));
 87                 }
 88                 currentRule->AddAction(&parser);
 89             }
 90         }
 91         public override void Visit(NonterminalParser& parser)
 92         {
 93             if (currentGrammar != null)
 94             {
 95                 try
 96                 {
 97                     RuleParser* rule = currentGrammar->GetRule(parser.RuleName());
 98                     parser.SetRule(rule);
 99                     if (parser.IsSpecialized())
100                     {
101                         currentRule->AddNonterminal(&parser);
102                     }
103                     if (parser.NumberOfArguments() != parser.Rule()->NumberOfParameters())
104                     {
105                         ThrowException("rule '" + ToUtf8(parser.RuleName()) + "' takes " + ToString(parser.Rule()->NumberOfParameters()) + " parameters (" + 
106                             ToString(parser.NumberOfArguments()) + " arguments supplied)", parser.GetSpan());
107                     }
108                 }
109                 catch (const ParsingException& ex)
110                 {
111                     ThrowException(ex.Message(), parser.GetSpan());
112                 }
113                 catch (const Exception& ex)
114                 {
115                     ThrowException(ex.Message(), parser.GetSpan());
116                 }
117             }
118         }
119         public override void Visit(KeywordParser& parser)
120         {
121             if (currentGrammar != null)
122             {
123                 if (!parser.ContinuationRuleName().IsEmpty())
124                 {
125                     parser.SetContinuationRule(currentGrammar->GetRule(parser.ContinuationRuleName()));
126                 }
127             }
128         }
129         public override void Visit(KeywordListParser& parser)
130         {
131             if (currentGrammar != null)
132             {
133                 parser.SetSelectorRule(currentGrammar->GetRule(parser.SelectorRuleName()));
134             }
135         }
136         private Grammar* currentGrammar;
137         private RuleParser* currentRule;
138         private int actionNumber;
139     }
140     
141     public void Link(ParsingDomain* parsingDomain)
142     {
143         LinkerVisitor visitor;
144         parsingDomain->GlobalNamespace()->Accept(visitor);
145     }
146     
147     public class ExpandCodeVisitor : Visitor
148     {
149         public override void BeginVisit(Grammar& grammar)
150         {
151             if (!grammar.IsExternal())
152             {
153                 for (RuleParser* rule : grammar.Rules())
154                 {
155                     rule->ExpandCode();
156                 }
157             }
158         }
159     }
160     
161     public void ExpandCode(ParsingDomain* parsingDomain)
162     {
163         ExpandCodeVisitor visitor;
164         parsingDomain->GlobalNamespace()->Accept(visitor);
165     }
166 }
