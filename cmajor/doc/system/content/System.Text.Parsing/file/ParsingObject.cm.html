  6 using System;
  7 using System.Collections;
  8 
  9 namespace System.Text.Parsing
 10 {
 11     public abstract class ParsingObject
 12     {
 13         public nothrow ParsingObject(const ustring& name_) : this(name_, null)
 14         {
 15         }
 16         public nothrow ParsingObject(const ustring& name_, Scope* enclosingScope_) : 
 17             owned(false), name(name_), enclosingScope(enclosingScope_), scope(null), owner(external)
 18         {
 19         }
 20         public virtual default ~ParsingObject();
 21         public abstract void Accept(Visitor& visitor);
 22         public inline nothrow const ustring& Name() const
 23         {
 24             return name;
 25         }
 26         public nothrow void SetName(const ustring& name_)
 27         {
 28             name = name_;
 29         }
 30         public nothrow ustring FullName() const
 31         {
 32             ustring enclosingScopeFullName;
 33             if (enclosingScope != null)
 34             {
 35                 enclosingScopeFullName = enclosingScope->FullName();
 36             }
 37             if (enclosingScopeFullName.IsEmpty())
 38             {
 39                 return name;
 40             }
 41             else
 42             {
 43                 return enclosingScopeFullName +  u"." + name;
 44             }
 45         }
 46         public inline nothrow Scope* EnclosingScope() const
 47         {
 48             return enclosingScope;
 49         }
 50         public nothrow void SetEnclosingScope(Scope* enclosingScope_)
 51         {
 52             enclosingScope = enclosingScope_;
 53         }
 54         public inline nothrow Scope* ContainerScope() const
 55         {
 56             return scope;
 57         }
 58         public nothrow void SetContainerScope(Scope* scope_)
 59         {
 60             scope = scope_;
 61         }
 62         public inline nothrow Scope* GetScope() const
 63         {
 64             if (scope != null)
 65             {
 66                 return scope;
 67             }
 68             return enclosingScope;
 69         }
 70         public inline nothrow const Span& GetSpan() const
 71         {
 72             return span;
 73         }
 74         public nothrow void SetSpan(const Span& span_)
 75         {
 76             span = span_;
 77         }
 78         public inline nothrow int Owner() const
 79         {
 80             return owner;
 81         }
 82         public inline nothrow void SetOwner(int owner_)
 83         {
 84             owner = owner_;
 85         }
 86         public inline nothrow bool IsExternal() const
 87         {
 88             return owner == external;
 89         }
 90         public virtual void AddToScope()
 91         {
 92             Scope* es = enclosingScope;
 93             if (es != null)
 94             {
 95                 try
 96                 {
 97                     es->Add(this);
 98                     if (scope != null)
 99                     {
100                         scope->SetEnclosingScope(es);
101                     }
102                 }
103                 catch (const Exception& ex)
104                 {
105                     ThrowException(ex.Message(), span);
106                 }
107             }
108         }
109         public void Own(ParsingObject* object)
110         {
111             if (object != null)
112             {
113                 if (!object->owned)
114                 {
115                     object->owned = true;
116                     ownedObjects.Add(UniquePtr<ParsingObject>(object));
117                 }
118             }
119         }
120         public inline nothrow bool IsOwned() const
121         {
122             return owned;
123         }
124         public inline nothrow void SetOwned()
125         {
126             owned = true;
127         }
128         public void Dump(CodeFormatter& formatter)
129         {
130             long n = ownedObjects.Count();
131             formatter.WriteLine(string(typename(*this)) + " : " + ToUtf8(name) + " : " + ToString(n));
132             formatter.WriteLine("{");
133             formatter.IncIndent();
134             for (long i = 0; i < n; ++i;)
135             {
136                 ParsingObject* o = ownedObjects[i].Get();
137                 o->Dump(formatter);
138             }
139             formatter.DecIndent();
140             formatter.WriteLine("}");
141         }
142         private const int external = -1;
143         private bool owned;
144         private List<UniquePtr<ParsingObject>> ownedObjects;
145         private ustring name;
146         private Scope* enclosingScope;
147         private Scope* scope;
148         private Span span;
149         private int owner;
150     }
151 }
