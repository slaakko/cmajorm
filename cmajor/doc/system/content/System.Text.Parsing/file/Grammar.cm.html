 36         public void Link()
 37         {
 38             if (!linking && !linked)
 39             {
 40                 linking = true;
 41                 GetReferencedGrammars();
 42                 for (Grammar* referencedGrammar : referencedGrammars)
 43                 {
 44                     referencedGrammar->Link();
 45                 }
 46                 LinkerVisitor visitor;
 47                 Accept(visitor);
 48                 linking = false;
 49                 linked = true;
 50             }
 51         }

 62         public void Parse(const uchar* start, const uchar* end, int fileIndex, const string& fileName, ParsingData* parsingData)
 63         {
 64             Scanner scanner(start, end, fileName, fileIndex, skipRule);
 65             scanner.SetParsingData(parsingData);
 66             UniquePtr<XmlLog> xmlLog;
 67             if (log != null)
 68             {
 69                 xmlLog.Reset(new XmlLog(*log, maxLogLineLength));
 70                 scanner.SetLog(xmlLog.Get());
 71                 xmlLog->WriteBeginRule( u"parse");
 72                 xmlLog->IncIndent();
 73             }
 74             Stack<UniquePtr<Object>> stack;
 75             Match match = Parse(scanner, stack, parsingData);
 76             Span stop = scanner.GetSpan();
 77             if (log != null)
 78             {
 79                 xmlLog->DecIndent();
 80                 xmlLog->WriteEndRule( u"parse");
 81             }
 82             if (!match.Hit() || stop.start != cast<int>(end - start))
 83             {
 84                 if (startRule != null)
 85                 {
 86                     throw ExpectationFailure(startRule->Info(), fileName, stop, start, end);
 87                 }
 88                 else
 89                 {
 90                     throw ParsingException("grammar '" + ToUtf8(FullName()) + "' has no start rule", fileName, stop, start, end);
 91                 }
 92             }
 93         }
 94         public Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
 95         {
 96             if (startRule != null)
 97             {
 98                 if (contentParser == null)
 99                 {
100                     if (skipRule != null)
101                     {
102                         contentParser = new SequenceParser(new SequenceParser(new OptionalParser(skipRule), startRule), new OptionalParser(skipRule));
103                         Own(contentParser);
104                     }
105                     else
106                     {
107                         contentParser = startRule;
108                     }
109                 }
110                 return contentParser->Parse(scanner, stack, parsingData);
111             }
112             return Match.Nothing();
113         }
114         public override void Accept(Visitor& visitor)
115         {
116             visitor.BeginVisit(*this);
117             for (RuleLink* ruleLink : ruleLinks)
118             {
119                 ruleLink->Accept(visitor);
120             }
121             for (RuleParser* rule : rules)
122             {
123                 rule->Accept(visitor);
124             }
125             visitor.EndVisit(*this);
126         }
127         public void AddRule(RuleParser* rule)
128         {
129             Own(rule);
130             rule->SetGrammar(this);
131             rule->GetScope()->SetEnclosingScope(GetScope());
132             rule->SetEnclosingScope(GetScope());
133             rules.Add(rule);
134             try
135             {
136                 GetScope()->Add(rule);
137             }
138             catch (const Exception& ex)
139             {
140                 ThrowException(ex.Message(), GetSpan());
141             }
142         }
143         public RuleParser* GetRule(const ustring& ruleName) const
144         {
145             ParsingObject* o = GetScope()->Get(ruleName);
146             if (o == null)
147             {
148                 ThrowException("rule '" + ToUtf8(ruleName) + "' not found in grammar '" + ToUtf8(FullName()) + "'", GetSpan());
149             }
150             else if (o is RuleParser*)
151             {
152                 return cast<RuleParser*>(o);
153             }
154             else
155             {
156                 if (o is RuleLink*)
157                 {
158                     RuleLink* ruleLink = cast<RuleLink*>(o);
159                     RuleParser* rule = ruleLink->Rule();
160                     if (rule == null)
161                     {
162                         ThrowException("rule link '" + ToUtf8(ruleName) + "' not bound in grammar '" + ToUtf8(FullName()) + "'", GetSpan());
163                     }
164                     return rule;
165                 }
166                 else
167                 {
168                     ThrowException("object '" + ToUtf8(ruleName) + "' (" + ToUtf8(o->FullName()) + ") is not a rule or rule link", GetSpan());
169                 }
170             }
171             return null;
172         }
173         public void AddRuleLink(RuleLink* ruleLink)
174         {
175             if (ruleLink->IsAddedToGrammar()) return;
176             ruleLink->SetAddedToGrammar();
177             Own(ruleLink);
178             ruleLinks.Add(ruleLink);
179         }
180         public void ResolveStartRule()
181         {
182             if (startRuleName.IsEmpty())
183             {
184                 if (rules.Count() > 0)
185                 {
186                     startRule = rules[0];
187                 }
188                 else
189                 {
190                     ThrowException("cannot resolve start rule because grammar '" + ToUtf8(FullName()) + "' is empty", GetSpan());
191                 }
192             }
193             else
194             {
195                 startRule = GetRule(startRuleName);
196             }
197         }
198         public void ResolveSkipRule()
199         {
200             if (skipRuleName.IsEmpty()) return;
201             skipRule = GetRule(skipRuleName);
202         }
