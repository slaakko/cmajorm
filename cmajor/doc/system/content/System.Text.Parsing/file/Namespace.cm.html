 6 using System;
 7 using System.Collections;
 8 
 9 namespace System.Text.Parsing
10 {
11     public class Namespace : ParsingObject
12     {
13         public Namespace(const ustring& qualifiedName_, Scope* enclosingScope_) : base(qualifiedName_, enclosingScope_)
14         {
15             Scope* containerScope = new Scope(Name(), EnclosingScope());
16             Own(containerScope);
17             SetContainerScope(containerScope);
18         }
19         public override void Accept(Visitor& visitor)
20         {
21             visitor.BeginVisit(*this);
22             GetScope()->Accept(visitor);
23             visitor.EndVisit(*this);
24         }
25         public void Import(Namespace* that, ParsingDomain* targetDomain, bool importExternals)
26         {
27             targetDomain->BeginNamespace(that->Name());
28             Scope* containerScope = that->ContainerScope();
29             for (const Pair<ustring, ParsingObject*>& p : containerScope->FullNameMap())
30             {
31                 ParsingObject* o = p.second;
32                 if (o is Grammar*)
33                 {
34                     if (!importExternals && o->IsExternal()) continue;
35                     Grammar* grammar = cast<Grammar*>(o);
36                     grammar->SetEnclosingScope(targetDomain->CurrentScope());
37                     targetDomain->AddGrammar(grammar);
38                 }
39                 else if (o is Namespace*)
40                 {
41                     Import(cast<Namespace*>(o), targetDomain, importExternals);
42                 }
43             }
44             targetDomain->EndNamespace();
45         }
46     }
47 }
