 6 using System;
 7 
 8 namespace System.Text.Parsing
 9 {
10     public class Match
11     {
12         public nothrow Match(bool hit_, int length_) : hit(hit_), length(length_)
13         {
14         }
15         public inline nothrow bool Hit() const
16         {
17             return hit;
18         }
19         public inline nothrow int Length() const
20         {
21             return length;
22         }
23         public inline nothrow void Concatenate(const Match& that)
24         {
25             length = length + that.length;
26         }
27         public static inline nothrow Match Empty()
28         {
29             return Match(true, 0);
30         }
31         public static inline nothrow Match One()
32         {
33             return Match(true, 1);
34         }
35         public static inline nothrow Match Nothing()
36         {
37             return Match(false, -1);
38         }
39         private bool hit;
40         private int length;
41     }
42 }
