 32     public class NonterminalParser : Parser

108         public ustring VariableName() const
109         {
110             ustring name = Name();
111             if (name.IsEmpty())
112             {
113                 return  u"NonterminalParser";
114             }
115             else
116             {
117                 return ToLower(name.Substring(0, 1)) + name.Substring(1) +  u"NonterminalParser";
118             }
119         }
120         public nothrow bool IsSpecialized() const
121         {
122             return numberOfArguments > 0 || arguments.Count() > 0 || rule != null && rule->TakesOrReturnsValuesOrHasLocals();
123         }
124         public override Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
125         {
126             if (rule != null)
127             {
128                 if (preCall != PreCall())
129                 {
130                     preCall(stack, parsingData);
131                 }
132                 Match match = rule->Parse(scanner, stack, parsingData);
133                 if (postCall != PostCall())
134                 {
135                     postCall(stack, parsingData, match.Hit());
136                 }
137                 return match;
138             }
139             return Match.Nothing();
140         }
141         public override void Accept(Visitor& visitor)
142         {
143             visitor.Visit(*this);
144         }

149         private PreCall preCall;
150         private PostCall postCall;
151     }
