  39     public class StringParser : Parser
  40     {
  41         public nothrow StringParser(const ustring& s_) : base( u"string", u"\"" + s_ + u"\""), s(s_)
  42         {
  43         }
  44         public inline nothrow const ustring& String() const
  45         {
  46             return s;
  47         }
  48         public override Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
  49         {
  50             Match match = Match.Empty();
  51             long i = 0;
  52             long n = s.Length();
  53             while (i < n && !scanner.AtEnd() && scanner.Char() == s[i])
  54             {
  55                 ++scanner;
  56                 ++i;
  57                 match.Concatenate(Match.One());
  58             }
  59             if (i == n)
  60             {
  61                 return match;
  62             }
  63             return Match.Nothing();
  64         }
  65         public override void Accept(Visitor& visitor)
  66         {
  67             visitor.Visit(*this);
  68         }
  69         private ustring s;
  70     }

 110     public class CharSetParser : Parser
 111     {
 112         public CharSetParser(const ustring& s_) : base( u"charset", u"[" + s_ + u"]"), s(s_), inverse(false), ranges()
 113         {
 114             InitRanges();
 115         }
 116         public CharSetParser(const ustring& s_, bool inverse_) : base( u"charset", u"[" + InverseStr(inverse_) + u"]"), s(s_), inverse(inverse_), ranges()
 117         {
 118             InitRanges();
 119         }
 120         public inline nothrow const ustring& Set() const
 121         {
 122             return s;
 123         }
 124         public inline nothrow bool Inverse() const
 125         {
 126             return inverse;
 127         }
 128         public override Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
 129         {
 130             if (!scanner.AtEnd())
 131             {
 132                 uchar c = scanner.Char();
 133                 bool found = false;
 134                 long n = ranges.Count();
 135                 for (long i = 0; i < n; ++i;)
 136                 {
 137                     const CharRange& range = ranges[i];
 138                     if (range.Includes(c))
 139                     {
 140                         found = true;
 141                         break;
 142                     }
 143                 }
 144                 bool matched = found != inverse;
 145                 if (matched)
 146                 {
 147                     ++scanner;
 148                     return Match.One();
 149                 }
 150             }
 151             return Match.Nothing();
 152         }
 153         public override void Accept(Visitor& visitor)
 154         {
 155             visitor.Visit(*this);
 156         }
 157         private void InitRanges()
 158         {
 159             long i = 0;
 160             long n = s.Length();
 161             while (i < n)
 162             {
 163                 uchar first = s[i];
 164                 uchar last = first;
 165                 ++i;
 166                 if (i < n)
 167                 {
 168                     if (s[i] ==  u'-')
 169                     {
 170                         ++i;
 171                         if (i < n)
 172                         {
 173                             last = s[i];
 174                             ++i;
 175                         }
 176                         else
 177                         {
 178                             ranges.Add(CharRange(first));
 179                             first =  u'-';
 180                             last = first;
 181                         }
 182                     }
 183                 }
 184                 ranges.Add(CharRange(first, last));
 185             }
 186         }
 187         private ustring s;
 188         private bool inverse;
 189         private List<CharRange> ranges;
 190     }
