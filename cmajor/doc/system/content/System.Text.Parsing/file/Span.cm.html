 6 using System;
 7 
 8 namespace System.Text.Parsing
 9 {
10     public class Span
11     {
12         public nothrow Span() : fileIndex(-1), lineNumber(0), start(-1), end(-1)
13         {
14         }
15         public nothrow Span(int fileIndex_) : fileIndex(fileIndex_), lineNumber(1), start(0), end(0)
16         {
17         }
18         public nothrow Span(int fileIndex_, int lineNumber_, int start_, int end_) : fileIndex(fileIndex_), lineNumber(lineNumber_), start(start_), end(end_)
19         {
20         }
21         public inline nothrow bool IsValid() const
22         {
23             return fileIndex != -1;
24         }
25         public inline nothrow void operator++()
26         {
27             ++start;
28             ++end;
29         }
30         public int fileIndex;
31         public int lineNumber;
32         public int start;
33         public int end;
34     }
