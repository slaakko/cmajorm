 11     public class delegate void ParsingAction(const uchar* matchBegin, const uchar* matchEnd, const Span& span, const string& fileName, ParsingData* parsingData, bool& pass);
 12     
 13     public class delegate void FailureAction(ParsingData* parsingData);

 15     public class ActionParser : UnaryParser
 16     {
 17         public ActionParser(const ustring& name_, Parser* child_) : base(name_, child_, child_->Info())
 18         {
 19         }
 20         public ustring MethodName()
 21         {
 22             return Name() +  u"Action";
 23         }
 24         public ustring VariableName()
 25         {
 26             ustring name = Name();
 27             if (name.IsEmpty())
 28             {
 29                 return  u"ActionParser";
 30             }
 31             return ToLower(name.Substring(0, 1)) + name.Substring(1) +  u"ActionParser";
 32         }
 33         public ParsingAction Action() const
 34         {
 35             return action;
 36         }
 37         public void SetAction(ParsingAction action_)
 38         {
 39             action = action_;
 40         }
 41         public FailureAction Failure() const
 42         {
 43             return failureAction;
 44         }
 45         public void SetFailure(FailureAction failureAction_)

 48         }
 49         public System.Text.Parsing.CodeDom.CompoundStatement* SuccessCode() const
 50         {

 56         }
 57         public System.Text.Parsing.CodeDom.CompoundStatement* FailureCode() const
 58         {

 60         }

 64         }
 65         public override Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
 66         {
 67             Span actionSpan = scanner.GetSpan();
 68             Match match = Child()->Parse(scanner, stack, parsingData);
 69             if (match.Hit())
 70             {
 71                 if (action != ParsingAction())
 72                 {
 73                     bool pass = true;
 74                     actionSpan.end = scanner.GetSpan().start;
 75                     const uchar* matchBegin = scanner.Start() + actionSpan.start;
 76                     const uchar* matchEnd = scanner.Start() + actionSpan.end;
 77                     action(matchBegin, matchEnd, actionSpan, scanner.FileName(), parsingData, pass);
 78                     if (!pass)
 79                     {
 80                         return Match.Nothing();
 81                     }
 82                 }
 83             }
 84             else if (failureAction != FailureAction())
 85             {
 86                 failureAction(parsingData);
 87             }
 88             return match;
 89         }
 90         public override void Accept(Visitor& visitor)
 91         {
 92             visitor.BeginVisit(*this);
 93             Child()->Accept(visitor);
 94             visitor.EndVisit(*this);
 95         }
 96         private ParsingAction action;
 97         private FailureAction failureAction;
 98         private UniquePtr<System.Text.Parsing.CodeDom.CompoundStatement> successCode;
 99         private UniquePtr<System.Text.Parsing.CodeDom.CompoundStatement> failureCode;
100     }

102     public class CodeExpandingVisitor : System.Text.Parsing.CodeDom.Visitor
103     {
104         public CodeExpandingVisitor(HashMap<ustring, ustring>& expandMap_) : expandMap(expandMap_)
105         {
106         }
107         public override void Visit(System.Text.Parsing.CodeDom.Identifier& id)
108         {
109             HashMap<ustring, ustring>.ConstIterator it = expandMap.CFind(id.Name());
110             if (it != expandMap.CEnd())
111             {
112                 const ustring& expandedName = it->second;
113                 id.SetName(expandedName);
114             }
115         }
116         private HashMap<ustring, ustring>& expandMap;
117     }
