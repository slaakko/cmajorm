  6 using System;
  7 using System.Collections;
  8 
  9 namespace System.Text.Parsing
 10 {
 11     public class ParsingDomain : ParsingObject
 12     {
 13         public ParsingDomain() : 
 14             base( u"parsingDomain"), grammarMap(), globalNamespace(new Namespace(u"", null)), globalScope(globalNamespace->GetScope()), currentNamespace(globalNamespace), nextRuleId(0)
 15         {
 16             Own(globalNamespace);
 17             globalScope->SetNs(globalNamespace);
 18             namespaceMap[currentNamespace->FullName()] = currentNamespace;
 19         }
 20         public Grammar* GetGrammar(const ustring& grammarName) const
 21         {
 22             HashMap<ustring, Grammar*>.ConstIterator it = grammarMap.CFind(grammarName);
 23             if (it != grammarMap.CEnd())
 24             {
 25                 return it->second;
 26             }
 27             return null;
 28         }
 29         public void AddGrammar(Grammar* grammar)
 30         {
 31             Own(grammar);
 32             ustring grammarFullName = grammar->FullName();
 33             HashMap<ustring, Grammar*>.ConstIterator it = grammarMap.CFind(grammarFullName);
 34             if (it == grammarMap.CEnd())
 35             {
 36                 grammarMap[grammarFullName] = grammar;
 37                 grammar->SetNs(currentNamespace);
 38                 grammar->AddToScope();
 39             }
 40             else
 41             {
 42                 Console.WriteLine("ParsingDomain.AddGrammar : grammar already found");
 43             }
 44         }
 45         public override void Accept(Visitor& visitor)
 46         {
 47             for (const Pair<ustring, Grammar*>& p : grammarMap)
 48             {
 49                 Grammar* grammar = p.second;
 50                 grammar->Accept(visitor);
 51             }
 52         }
 53         public Scope* GetNamespaceScope(const ustring& fullNamespaceName) const
 54         {
 55             if (fullNamespaceName.IsEmpty())
 56             {
 57                 return globalScope;
 58             }
 59             Namespace* ns = globalScope->GetNamespace(fullNamespaceName);
 60             if (ns == null)
 61             {
 62                 ns = new Namespace(fullNamespaceName, globalScope);
 63                 Own(ns);
 64                 globalScope->AddNamespace(ns);
 65             }
 66             return ns->GetScope();
 67         }
 68         public void BeginNamespace(const ustring& ns)
 69         {
 70             ustring fullNamespaceName = ns;
 71             if (!currentNamespace->FullName().IsEmpty())
 72             {
 73                 fullNamespaceName = currentNamespace->FullName() +  u"." + ns;
 74             }
 75             namespaceStack.Push(currentNamespace);
 76             HashMap<ustring, Namespace*>.ConstIterator it = namespaceMap.CFind(fullNamespaceName);
 77             if (it != namespaceMap.CEnd())
 78             {
 79                 currentNamespace = it->second;
 80             }
 81             else
 82             {
 83                 Namespace* newNs = new Namespace(fullNamespaceName, globalScope);
 84                 Own(newNs);
 85                 newNs->GetScope()->SetNs(newNs);
 86                 namespaceMap[newNs->FullName()] = newNs;
 87                 currentNamespace = newNs;
 88                 if (globalScope->GetNamespace(newNs->FullName()) == null)
 89                 {
 90                     globalScope->AddNamespace(newNs);
 91                 }
 92             }
 93         }
 94         public void EndNamespace()
 95         {
 96             currentNamespace = namespaceStack.Pop();
 97         }
 98         public inline nothrow int GetNextRuleId()
 99         {
100             return nextRuleId++;
101         }
102         public inline nothrow int NumRules() const
103         {
104             return nextRuleId;
105         }
106         public inline nothrow Namespace* GlobalNamespace() const
107         {
108             return globalNamespace;
109         }
110         public inline nothrow Namespace* CurrentNamespace() const
111         {
112             return currentNamespace;
113         }
114         public inline nothrow Scope* CurrentScope() const
115         {
116             return currentNamespace->GetScope();
117         }
118         public void Import(ParsingDomain* pd, bool importExternals)
119         {
120             globalNamespace->Import(pd->GlobalNamespace(), this, importExternals);
121         }
122         private HashMap<ustring, Grammar*> grammarMap;
123         private Namespace* globalNamespace;
124         private Scope* globalScope;
125         private Namespace* currentNamespace;
126         private HashMap<ustring, Namespace*> namespaceMap;
127         private Stack<Namespace*> namespaceStack;
128         private int nextRuleId;
129     }
130     
131     public class ParsingDomainRepository
132     {
133         static ParsingDomainRepository() : instance(new ParsingDomainRepository())
134         {
135         }
136         public static ParsingDomainRepository& Instance()
137         {
138             return *instance;
139         }
140         private ParsingDomainRepository()
141         {
142         }
143         public ~ParsingDomainRepository()
144         {
145         }
146         public void Register(ParsingDomain* parsingDomain)
147         {
148             if (!parsingDomain->IsOwned())
149             {
150                 parsingDomain->SetOwned();
151                 parsingDomains.Add(UniquePtr<ParsingDomain>(parsingDomain));
152             }
153         }
154         private static UniquePtr<ParsingDomainRepository> instance;
155         private List<UniquePtr<ParsingDomain>> parsingDomains;
156     }
157     
158     public void RegisterParsingDomain(ParsingDomain* parsingDomain)
159     {
160         ParsingDomainRepository.Instance().Register(parsingDomain);
161     }
