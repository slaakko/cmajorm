  6 using System;
  7 using System.Collections;
  8 
  9 namespace System.Text.Parsing
 10 {
 11     public class Scope : ParsingObject
 12     {
 13         public nothrow Scope(const ustring& name_, Scope* enclosingScope_) : base(name_, enclosingScope_), ns(null), fullNameMap(), shortNameMap()
 14         {
 15         }
 16         public override void Accept(Visitor& visitor)
 17         {
 18             for (const Pair<ustring, ParsingObject*>& p : fullNameMap)
 19             {
 20                 p.second->Accept(visitor);
 21             }
 22         }
 23         public inline nothrow Namespace* Ns() const
 24         {
 25             return ns;
 26         }
 27         public inline nothrow void SetNs(Namespace* ns_)
 28         {
 29             ns = ns_;
 30         }
 31         public void AddNamespace(Namespace* nsToAdd)
 32         {
 33             Own(nsToAdd);
 34             Namespace* parent = GetGlobalScope()->Ns();
 35             List<ustring> components = nsToAdd->FullName().Split( u'.');
 36             if (components.IsEmpty())
 37             {
 38                 throw Exception("Scope.AddNamespace: name components empty");
 39             }
 40             long n = components.Count();
 41             for (long i = 0; i < n - 1; ++i;)
 42             {
 43                 ustring namespaceName = components[i];
 44                 Scope* enclosingScope = parent->GetScope();
 45                 ParsingObject* o = enclosingScope->Get(namespaceName);
 46                 if (o != null)
 47                 {
 48                     if (o is Namespace*)
 49                     {
 50                         parent = cast<Namespace*>(o);
 51                     }
 52                     else
 53                     {
 54                         throw Exception("object '" + ToUtf8(namespaceName) + "' is not a namespace");
 55                     }
 56                 }
 57                 else
 58                 {
 59                     parent = new Namespace(namespaceName, enclosingScope);
 60                     parent->GetScope()->SetName(namespaceName);
 61                     parent->GetScope()->SetNs(parent);
 62                     enclosingScope->Add(parent);
 63                     Own(parent);
 64                 }
 65             }
 66             ustring name = components[n - 1];
 67             nsToAdd->SetName(name);
 68             nsToAdd->GetScope()->SetName(name);
 69             nsToAdd->SetEnclosingScope(parent->GetScope());
 70             nsToAdd->GetScope()->SetEnclosingScope(parent->GetScope());
 71             parent->GetScope()->Add(nsToAdd);
 72         }
 73         public Namespace* GetNamespace(const ustring& fullNamespaceName) const
 74         {
 75             ParsingObject* o = GetQualifiedObject(fullNamespaceName);
 76             if (o != null)
 77             {
 78                 if (o is Namespace*)
 79                 {
 80                     return cast<Namespace*>(o);
 81                 }
 82                 else
 83                 {
 84                     throw Exception("object '" + ToUtf8(fullNamespaceName) + "' is not a namespace");
 85                 }
 86             }
 87             return null;
 88         }
 89         public void Add(ParsingObject* o)
 90         {
 91             Own(o);
 92             ustring fullName = o->FullName();
 93             if (!fullNameMap.Insert(MakePair(fullName, o)).second)
 94             {
 95                 string msg = "object '" + ToUtf8(fullName) + "' already exists in full name map (detected in scope '" + ToUtf8(FullName()) + "')";
 96                 throw Exception(msg);
 97             }
 98             if (!shortNameMap.Insert(MakePair(o->Name(), o)).second)
 99             {
100                 string msg = "object '" + ToUtf8(fullName) + "' ('" + ToUtf8(o->Name()) + "') already exists in short name map (detected in scope '" + ToUtf8(FullName()) + "')";
101                 throw Exception(msg);
102             }
103         }
104         public ParsingObject* Get(const ustring& objectName) const
105         {
106             if (objectName.Find( u'.') != -1)
107             {
108                 HashMap<ustring, ParsingObject*>.ConstIterator it = fullNameMap.CFind(objectName);
109                 if (it != fullNameMap.CEnd())
110                 {
111                     return it->second;
112                 }
113                 ParsingObject* o = GetQualifiedObject(objectName);
114                 if (o != null)
115                 {
116                     return o;
117                 }
118             }
119             else
120             {
121                 HashMap<ustring, ParsingObject*>.ConstIterator it = shortNameMap.CFind(objectName);
122                 if (it != shortNameMap.CEnd())
123                 {
124                     return it->second;
125                 }
126             }
127             return null;
128         }
129         public HashMap<ustring, ParsingObject*>& FullNameMap()
130         {
131             return fullNameMap;
132         }
133         public void Dump(CodeFormatter& formatter)
134         {
135             formatter.WriteLine("Scope '" + ToUtf8(FullName()) + "'");
136             formatter.WriteLine("{");
137             formatter.IncIndent();
138             for (const Pair<ustring, ParsingObject*>& p : fullNameMap)
139             {
140                 formatter.WriteLine(ToUtf8(p.first));
141             }
142             formatter.WriteLine("---");
143             for (const Pair<ustring, ParsingObject*>& p : shortNameMap)
144             {
145                 formatter.WriteLine(ToUtf8(p.first));
146             }
147             formatter.DecIndent();
148             formatter.WriteLine("}");
149         }
150         private nothrow Scope* GetGlobalScope() const
151         {
152             Scope* globalScope = this;
153             while (globalScope->EnclosingScope() != null)
154             {
155                 globalScope = globalScope->EnclosingScope();
156             }
157             return globalScope;
158         }
159         private ParsingObject* GetQualifiedObject(const ustring& qualifiedObjectName) const
160         {
161             List<ustring> components = qualifiedObjectName.Split( u'.');
162             long n = components.Count();
163             Scope* scope = this;
164             while (scope != null)
165             {
166                 Scope* subScope = scope;
167                 long i = 0;
168                 HashMap<ustring, ParsingObject*>.ConstIterator it = subScope->shortNameMap.CFind(components[i]);
169                 while (it != subScope->shortNameMap.CEnd())
170                 {
171                     if (i == n - 1)
172                     {
173                         return it->second;
174                     }
175                     else
176                     {
177                         ++i;
178                         subScope = it->second->GetScope();
179                     }
180                     it = subScope->shortNameMap.CFind(components[i]);
181                 }
182                 scope = scope->EnclosingScope();
183             }
184             return null;
185         }
186         private Namespace* ns;
187         private HashMap<ustring, ParsingObject*> fullNameMap;
188         private HashMap<ustring, ParsingObject*> shortNameMap;
189     }
190 }
