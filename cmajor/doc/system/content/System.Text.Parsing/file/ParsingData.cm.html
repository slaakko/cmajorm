 6 using System;
 7 using System.Collections;
 8 
 9 namespace System.Text.Parsing
10 {
11     public abstract class Context
12     {
13         public virtual default ~Context();
14     }
15     
16     public class ContextStack
17     {
18         public void Push(Context* context)
19         {
20             stack.Push(UniquePtr<Context>(context));
21         }
22         public Context* Pop()
23         {
24             if (stack.IsEmpty())
25             {
26                 throw Exception("context stack is empty");
27             }
28             UniquePtr<Context> context = stack.Pop();
29             return context.Release();
30         }
31         private Stack<UniquePtr<Context>> stack;
32     }
33     
34     public class RuleData
35     {
36         public UniquePtr<Context> context;
37         public ContextStack contextStack;
38     }
39     
40     public class ParsingData
41     {
42         public ParsingData(int numRules)
43         {
44             ruleData.Resize(numRules);
45         }
46         public default ~ParsingData();
47         public void PushContext(int ruleId, Context* context)
48         {
49             #assert(ruleId >= 0 && ruleId < ruleData.Count());
50             #assert(context != null);
51             ruleData[ruleId].contextStack.Push(ruleData[ruleId].context.Release());
52             ruleData[ruleId].context.Reset(context);
53         }
54         public void PopContext(int ruleId)
55         {
56             #assert(ruleId >= 0 && ruleId < ruleData.Count());
57             ruleData[ruleId].context.Reset(ruleData[ruleId].contextStack.Pop());
58         }
59         public Context* GetContext(int ruleId) const
60         {
61             #assert(ruleId >= 0 && ruleId < ruleData.Count());
62             return ruleData[ruleId].context.Get();
63         }
64         private List<RuleData> ruleData;
65     }
66 }
