 11     public abstract class UnaryParser : Parser
 12     {
 13         public UnaryParser(const ustring& name_, Parser* child_, const ustring& info_) : base(name_, info_), child(child_)
 14         {
 15             Own(child);
 16         }
 17         public inline nothrow Parser* Child() const
 18         {
 19             return child;
 20         }
 21         private Parser* child;
 22     }

 28         }
 29         public override Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
 30         {
 31             Span save = scanner.GetSpan();
 32             Match match = Child()->Parse(scanner, stack, parsingData);
 33             if (match.Hit())
 34             {
 35                 return match;
 36             }
 37             scanner.SetSpan(save);
 38             return Match.Empty();
 39         }

 41         {
 42             visitor.BeginVisit(*this);

 48     public class PositiveParser : UnaryParser
 49     {
 50         public PositiveParser(Parser* child_) : base( u"positive", child_, child_->Info() + u"+")
 51         {
 52         }
 53         public override Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
 54         {
 55             Match match = Child()->Parse(scanner, stack, parsingData);
 56             if (match.Hit())
 57             {
 58                 for (;;;)
 59                 {
 60                     Span save = scanner.GetSpan();
 61                     scanner.Skip();
 62                     Match next = Child()->Parse(scanner, stack, parsingData);
 63                     if (next.Hit())
 64                     {
 65                         match.Concatenate(next);
 66                     }
 67                     else
 68                     {
 69                         scanner.SetSpan(save);
 70                         break;
 71                     }
 72                 }
 73             }
 74             return match;
 75         }
 76         public override void Accept(Visitor& visitor)
 77         {
 78             visitor.BeginVisit(*this);
 79             Child()->Accept(visitor);
 80             visitor.EndVisit(*this);
 81         }
 82     }

 92             bool first = true;

 99                 }
100                 else
101                 {
102                     scanner.Skip();
103                 }
104                 Match next = Child()->Parse(scanner, stack, parsingData);
105                 if (next.Hit())
106                 {
107                     match.Concatenate(next);
108                 }
109                 else
110                 {
111                     scanner.SetSpan(save);
112                     break;
113                 }
114             }
115             return match;
116         }

196     public abstract class BinaryParser : Parser
197     {
198         public BinaryParser(const ustring& name_, Parser* left_, Parser* right_, const ustring& info_) : base(name_, info_), left(left_), right(right_)
199         {
200             Own(left);
201             Own(right);
202         }
203         public inline nothrow Parser* Left() const
204         {
205             return left;
206         }
207         public inline nothrow Parser* Right() const
208         {
209             return right;
210         }
211         private Parser* left;
212         private Parser* right;
213     }

215     public class SequenceParser : BinaryParser
216     {
217         public SequenceParser(Parser* left_, Parser* right_) : base( u"sequence", left_, right_, left_->Info())
218         {
219         }
220         public override Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
221         {
222             Match leftMatch = Left()->Parse(scanner, stack, parsingData);
223             if (leftMatch.Hit())
224             {
225                 scanner.Skip();
226                 Match rightMatch = Right()->Parse(scanner, stack, parsingData);
227                 if (rightMatch.Hit())
228                 {
229                     leftMatch.Concatenate(rightMatch);
230                     return leftMatch;
231                 }
232             }
233             return Match.Nothing();
234         }
235         public override void Accept(Visitor& visitor)
236         {
237             visitor.BeginVisit(*this);
238             Left()->Accept(visitor);
239             visitor.Visit(*this);
240             Right()->Accept(visitor);
241             visitor.EndVisit(*this);
242         }
243     }

271     public class DifferenceParser : BinaryParser
272     {
273         public DifferenceParser(Parser* left_, Parser* right_) : base( u"difference", left_, right_, left_->Info() + u" - " + right_->Info())
274         {
275         }
276         public override Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
277         {
278             Span save = scanner.GetSpan();
279             Match leftMatch = Left()->Parse(scanner, stack, parsingData);
280             if (leftMatch.Hit())
281             {
282                 Span tmp = scanner.GetSpan();
283                 scanner.SetSpan(save);
284                 save = tmp;
285                 Match rightMatch = Right()->Parse(scanner, stack, parsingData);
286                 if (!rightMatch.Hit() || rightMatch.Length() < leftMatch.Length())
287                 {
288                     scanner.SetSpan(save);
289                     return leftMatch;
290                 }
291             }
292             return Match.Nothing();
293         }
294         public override void Accept(Visitor& visitor)
295         {
296             visitor.BeginVisit(*this);
297             Left()->Accept(visitor);
298             visitor.Visit(*this);
299             Right()->Accept(visitor);
300             visitor.EndVisit(*this);
301         }

304     public class ExclusiveOrParser : BinaryParser
305     {
306         public ExclusiveOrParser(Parser* left_, Parser* right_) : base( u"exclusive_or", left_, right_, left_->Info() + u" ^ " + right_->Info())
307         {
308         }
309         public override Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
310         {
311             Span save = scanner.GetSpan();
312             Match leftMatch = Left()->Parse(scanner, stack, parsingData);
313             Span temp = scanner.GetSpan();
314             scanner.SetSpan(save);
315             save = temp;
316             Match rightMatch = Right()->Parse(scanner, stack, parsingData);
317             bool match = rightMatch.Hit();
318             if (leftMatch.Hit())
319             {
320                 match = !rightMatch.Hit();
321             }
322             if (match)
323             {
324                 if (leftMatch.Hit())
325                 {
326                     scanner.SetSpan(save);
327                 }
328                 if (leftMatch.Hit())
329                 {
330                     return leftMatch;
331                 }
332                 else
333                 {
334                     return rightMatch;
335                 }
336             }
337             return Match.Nothing();
338         }
339         public override void Accept(Visitor& visitor)
340         {
341             visitor.BeginVisit(*this);
342             Left()->Accept(visitor);
343             visitor.Visit(*this);
344             Right()->Accept(visitor);
345             visitor.EndVisit(*this);
346         }
347     }
