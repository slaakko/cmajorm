 47     public nothrow const uchar* LineStart(const uchar* start, const uchar* pos)
 48     {
 49         while (pos > start && *pos !=  u'\n')
 50         {
 51             --pos;
 52         }
 53         if (*pos ==  u'\n')
 54         {
 55             ++pos;
 56         }
 57         return pos;
 58     }

 60     public nothrow const uchar* LineEnd(const uchar* end, const uchar* pos)
 61     {
 62         while (pos < end && *pos !=  u'\n')
 63         {
 64             ++pos;
 65         }
 66         return pos;
 67     }

 69     public nothrow ustring GetErrorLines(const uchar* start, const uchar* end, const Span& span)
 70     {
 71         const uchar* startPos = start + span.start;
 72         if (startPos < start || startPos >= end)
 73         {
 74             return ustring();
 75         }
 76         const uchar* lineStart = LineStart(start, startPos);
 77         int cols = cast<int>(startPos - lineStart);
 78         if (cols < 0)
 79         {
 80             cols = 0;
 81         }
 82         const uchar* lineEnd = LineEnd(end, startPos);
 83         if (lineEnd < lineStart)
 84         {
 85             lineEnd = lineStart;
 86         }
 87         int lineLength = cast<int>(lineEnd - lineStart);
 88         ustring lines(lineStart, lineEnd);
 89         int spanCols = Max(cast<int>(1), Min(span.end - span.start, lineLength - cols));
 90         lines.Append( u'\n').Append(ustring(u' ', cols)).Append(ustring(u'^', spanCols));
 91         return lines;
 92     }

162     public void ThrowException(const string& message, const Span& span)
163     {
164         if (span.IsValid())
165         {
166             string filePath = FileInfo.Instance().GetFilePath(span.fileIndex);
167             if (!filePath.IsEmpty())
168             {
169                 string content = File.ReadAllText(filePath);
170                 ustring uc = ToUtf32(content);
171                 throw ParsingException(message, filePath, span, &uc[0], &uc[0] + uc.Length());
172             }
173         }
174         throw Exception(message);
175     }
