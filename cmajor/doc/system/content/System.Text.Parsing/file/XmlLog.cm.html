  6 using System;
  7 using System.IO;

  9 namespace System.Text.Parsing

 11     public ustring XmlCharStr(uchar c)
 12     {
 13         switch (c)
 14         {
 15             case  u'&':return  u"&amp;";
 16             case  u'<':return  u"&lt;";
 17             case  u'>':return  u"&gt;";
 18             case  u'\a':return  u"\\a";
 19             case  u'\b':return  u"\\b";
 20             case  u'\f':return  u"\\f";
 21             case  u'\n':return  u"\\n";
 22             case  u'\r':return  u"\\r";
 23             case  u'\t':return  u"\\t";
 24             case  u'\v':return  u"\\v";
 25             default:
 26             {
 27                 uint x = cast<uint>(c);
 28                 if (x >= 32u && x <= 126u)
 29                 {
 30                     return ustring(c, 1);
 31                 }
 32                 else
 33                 {
 34                     return  u"\\x" + ToUtf32(ToHexString(x));
 35                 }
 36             }
 37         }
 38         return ustring();
 39     }
 40     
 41     public ustring XmlEscape(const ustring& s)
 42     {
 43         ustring result;
 44         for (uchar c : s)
 45         {
 46             result.Append(XmlCharStr(c));
 47         }
 48         return result;
 49     }

 51     public class XmlLog
 52     {
 53         public nothrow XmlLog(StreamWriter& writer_, int maxLineLength_) : writer(writer_), maxLineLength(maxLineLength_), indent(0)
 54         {
 55         }
 56         public inline nothrow void IncIndent()
 57         {
 58             ++indent;
 59         }
 60         public inline nothrow void DecIndent()
 61         {
 62             --indent;
 63         }
 64         public void WriteBeginRule(const ustring& ruleName)
 65         {
 66             Write( u"<" + ruleName + u">");
 67         }
 68         public void WriteEndRule(const ustring& ruleName)
 69         {
 70             Write( u"</" + ruleName + u">");
 71         }
 72         public void WriteTry(const ustring& s)
 73         {
 74             WriteElement( u"try", s);
 75         }
 76         public void WriteSuccess(const ustring& match)
 77         {
 78             WriteElement( u"success", match);
 79         }
 80         public void WriteFail()
 81         {
 82             Write( u"<fail/>");
 83         }
 84         public void WriteElement(const ustring& elementName, const ustring& elementContent)
 85         {
 86             ustring converted = XmlEscape(elementContent);
 87             int convertedLength = cast<int>(converted.Length());
 88             int lineLength = indent + 2 * cast<int>(elementName.Length()) + 5 + convertedLength;
 89             ustring s = converted;
 90             if (maxLineLength != 0 && lineLength > maxLineLength)
 91             {
 92                 s = converted.Substring(0, Max(0, convertedLength - (lineLength - maxLineLength))) +  u"...";
 93             }
 94             Write( u"<" + elementName + u">" + s + u"</" + elementName + u">");
 95         }
 96         public void Write(const ustring& s)
 97         {
 98             if (indent > 0)
 99             {
100                 writer.Write(string(' ', indent));
101             }
102             writer.WriteLine(s);
103         }
104         public inline nothrow int MaxLineLength() const
105         {
106             return maxLineLength;
107         }
108         private StreamWriter& writer;
109         private int maxLineLength;
110         private int indent;
111     }
