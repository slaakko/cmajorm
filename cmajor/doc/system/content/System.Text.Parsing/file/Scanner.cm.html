 11     public class Scanner
 12     {
 13         public nothrow Scanner(const uchar* start_, const uchar* end_, const string& fileName_, int fileIndex_, Parser* skipper_) : 
 14             start(start_), end(end_), fileName(fileName_), span(fileIndex_), skipper(skipper_), skipping(false), tokenCounter(0), atBeginningOfLine(true), parsingData(null), log(null)
 15         {
 16         }
 17         public inline nothrow const uchar* Start() const
 18         {
 19             return start;
 20         }
 21         public inline nothrow const uchar* End() const
 22         {
 23             return end;
 24         }
 25         public inline nothrow uchar Char() const
 26         {
 27             return start[span.start];
 28         }
 29         public nothrow void operator++()
 30         {
 31             uchar c = Char();
 32             ++span;
 33             if (c ==  u'\n')
 34             {
 35                 atBeginningOfLine = true;
 36                 ++span.lineNumber;
 37             }
 38         }
 39         public inline nothrow bool AtEnd() const
 40         {
 41             return start + span.start >= end;
 42         }
 43         public inline nothrow void BeginToken()
 44         {
 45             ++tokenCounter;
 46         }
 47         public inline nothrow void EndToken()
 48         {
 49             --tokenCounter;
 50         }
 51         public void Skip()
 52         {
 53             if (tokenCounter == 0 && skipper != null)
 54             {
 55                 Span save = span;
 56                 BeginToken();
 57                 skipping = true;
 58                 Stack<UniquePtr<Object>> stack;
 59                 Match match = skipper->Parse(*this, stack, parsingData);
 60                 skipping = false;
 61                 EndToken();
 62                 if (!match.Hit())
 63                 {
 64                     span = save;
 65                 }
 66             }
 67         }
 68         public inline nothrow const string& FileName() const
 69         {
 70             return fileName;
 71         }
 72         public inline nothrow const Span& GetSpan() const
 73         {
 74             return span;
 75         }
 76         public inline nothrow void SetSpan(const Span& span_)
 77         {

 79         }
 80         public inline nothrow Parser* Skipper() const
 81         {
 82             return skipper;
 83         }
 84         public inline nothrow bool Skipping() const
 85         {

 87         }

 91         }
 92         public inline nothrow bool AtBeginningOfLine() const
 93         {

 95         }
 96         public inline nothrow void SetParsingData(ParsingData* parsingData_)
 97         {

 99         }
100         public nothrow int LineEndIndex(int maxLineLength) const
101         {
102             int lineEndIndex = span.start;
103             int contentLength = cast<int>(end - start);
104             int end_ = contentLength;
105             if (maxLineLength != 0)
106             {
107                 end_ = Min(contentLength, span.start + maxLineLength);
108             }
109             while (lineEndIndex < end_ && (start[lineEndIndex] != '\r' && start[lineEndIndex] != '\n'))
110             {
111                 ++lineEndIndex;
112             }
113             return lineEndIndex;
114         }
115         public nothrow ustring RestOfLine(int maxLineLength)
116         {
117             ustring restOfLine(start + span.start, start + LineEndIndex(maxLineLength));
118             return restOfLine;
119         }

121         {
122             return log;
123         }

127         }
