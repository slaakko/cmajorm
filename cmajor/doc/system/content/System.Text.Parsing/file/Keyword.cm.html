 42     public class KeywordParser : Parser
 43     {
 44         public KeywordParser(const ustring& keyword_) : base( u"keyword", u"\"" + keyword_ + u"\""), keyword(keyword_), continuationRuleName()
 45         {
 46             continuationRule = IdentifierCharsRule.Instance().GetRule();
 47             keywordStringParser = new StringParser(keyword);
 48             Own(keywordStringParser);
 49             CreateKeywordRule();
 50         }
 51         public KeywordParser(const ustring& keyword_, const ustring& continuationRuleName_) : 
 52             base( u"keyword", u"\"" + keyword_ + u"\""), keyword(keyword_), continuationRuleName(continuationRuleName_)
 53         {
 54             continuationRule = IdentifierCharsRule.Instance().GetRule();
 55             keywordStringParser = new StringParser(keyword);
 56             Own(keywordStringParser);
 57             CreateKeywordRule();
 58         }
 59         private void CreateKeywordRule()
 60         {
 61             keywordRule = new RuleParser(keyword, null, -1, 
 62                 new DifferenceParser(
 63                     keywordStringParser, 
 64                     new TokenParser(
 65                         new SequenceParser(
 66                             keywordStringParser, 
 67                             continuationRule))));
 68             Own(keywordRule);
 69         }
 70         public override Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
 71         {
 72             return keywordRule->Parse(scanner, stack, parsingData);
 73         }
 74         public override void Accept(Visitor& visitor)
 75         {
 76             visitor.Visit(*this);
 77         }
 78         public inline nothrow const ustring& Keyword() const
 79         {
 80             return keyword;
 81         }
 82         public inline nothrow const ustring& ContinuationRuleName() const
 83         {
 84             return continuationRuleName;
 85         }
 86         public nothrow void SetContinuationRuleName(const ustring& continuationRuleName_)
 87         {
 88             continuationRuleName = continuationRuleName_;
 89         }
 90         public inline nothrow RuleParser* ContinuationRule() const
 91         {
 92             return continuationRule;
 93         }
 94         public void SetContinuationRule(RuleParser* continuationRule_)
 95         {
 96             continuationRule = continuationRule_;
 97             Own(continuationRule);

 99         }

107     public class KeywordListParser : Parser
108     {
109         public KeywordListParser(const ustring& selectorRuleName_, const List<ustring>& keywords_) : 
110             base( u"keywordList", u"keyword_list"), selectorRuleName(selectorRuleName_), selectorRule(null)
111         {
112             for (const ustring& keyword : keywords_)
113             {
114                 keywords.Insert(keyword);
115             }
116         }
117         public inline nothrow const ustring& SelectorRuleName() const
118         {
119             return selectorRuleName;
120         }
121         public inline nothrow const ustring& KeywordListName() const
122         {
123             return keywordListName;
124         }
125         public nothrow void SetKeywordListName(const ustring& keywordListName_)
126         {
127             keywordListName = keywordListName_;
128         }
129         public inline nothrow const HashSet<ustring>& Keywords() const
130         {
131             return keywords;
132         }
133         public inline nothrow RuleParser* SelectorRule() const
134         {
135             return selectorRule;
136         }
137         public nothrow void SetSelectorRule(RuleParser* selectorRule_)
138         {
139             selectorRule = selectorRule_;
140         }
141         public override Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData)
142         {
143             if (selectorRule != null)
144             {
145                 Span save = scanner.GetSpan();
146                 Match match = selectorRule->Parse(scanner, stack, parsingData);
147                 if (match.Hit())
148                 {
149                     const uchar* matchBegin = scanner.Start() + save.start;
150                     const uchar* matchEnd = scanner.Start() + scanner.GetSpan().start;
151                     ustring keyword(matchBegin, matchEnd);
152                     if (keywords.CFind(keyword) != keywords.CEnd())
153                     {
154                         return match;
155                     }
156                     scanner.SetSpan(save);
157                 }
158             }
159             return Match.Nothing();
160         }
161         public override void Accept(Visitor& visitor)
162         {
163             visitor.Visit(*this);
164         }
165         private ustring selectorRuleName;
166         private ustring keywordListName;
167         private HashSet<ustring> keywords;
168         private RuleParser* selectorRule;
169     }
