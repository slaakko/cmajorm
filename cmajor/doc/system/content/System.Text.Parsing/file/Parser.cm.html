 6 using System;
 7 using System.Collections;
 8 
 9 namespace System.Text.Parsing
10 {
11     public abstract class Object
12     {
13         public virtual default ~Object();
14     }
15     
16     public class ValueObject<ValueType> : Object
17     {
18         public default ValueObject();
19         public ValueObject(const ValueType& value_) : value(value_)
20         {
21         }
22         public void operator=(const ValueType& value_)
23         {
24             value = value_;
25         }
26         public operator ValueType() const
27         {
28             return value;
29         }
30         private ValueType value;
31     }
32     
33     public abstract class Parser : ParsingObject
34     {
35         public nothrow Parser(const ustring& name_, const ustring& info_) : base(name_), info(info_)
36         {
37         }
38         public inline nothrow const ustring& Info() const
39         {
40             return info;
41         }
42         public abstract Match Parse(Scanner& scanner, Stack<UniquePtr<Object>>& stack, ParsingData* parsingData);
43         private ustring info;
44     }
