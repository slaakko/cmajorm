stdlib Grammar

spaces → space+
newline → "\r\n" | "\n" | "\r"
comment → line_comment | block_comment
line_comment → "//" [^\r\n]* newline
block_comment → "/*" (string | char | (anychar − "*/"))* "*/"
spaces_and_comments → (space | comment)+
digit_sequence → token(digit+)
sign → '+' | '-'
sbyte → token(sign? digit_sequence)
byte → token(digit_sequence)
short → token(sign? digit_sequence)
ushort → token(digit_sequence)
int → token(sign? digit_sequence)
uint → token(digit_sequence)
long → token(sign? digit_sequence)
ulong → token(digit_sequence)
hexbyte → token(hexdigit+)
hexword → token(hexdigit+)
hexdword → token(hexdigit+)
hexqword → token(hexdigit+)
hex → hexqword
hex_literal → token(("0x" | "0X") hex!)
octal_digit → [0-7]
float → token(sign? fractional_real exponent_part? | sign? digit_sequence exponent_part)
double → token(sign? fractional_real exponent_part? | sign? digit_sequence exponent_part)
ureal → token(fractional_real exponent_part? | digit_sequence exponent_part)
fractional_real → token(digit_sequence? '.' digit_sequence) | token(digit_sequence '.')
exponent_part → token([eE] sign? digit_sequence)
num → double | int
bool → keyword("true") | keyword("false")
identifier → token((idstart | '_') idcont*)
qualified_id → token(identifier ('.' identifier)*)
escape → token('\\' ([xX] hexdword | (octal_digit octal_digit octal_digit) | [dD] uint | [^dDxX]))
char → token('\'' ([^\\\r\n] | escape) '\''!)
string → token('\"' (([^\"\\\r\n]+) | escape)* '\"'!)
