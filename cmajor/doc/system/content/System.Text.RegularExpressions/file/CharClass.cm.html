  6 using System;
  7 using System.Collections;
  8 using System.IO;
  9 
 10 namespace System.Text.RegularExpressions
 11 {
 12     public enum Class
 13     {
 14         alnum, alpha, blank, cntrl, digit, graph, lower, print, punct, space, upper, word, xdigit
 15     }
 16 
 17     public class CharClass
 18     {
 19         public nothrow CharClass() : ranges(), inverse(false), bits()
 20         {
 21             try
 22             {
 23                 bits.Resize(256);
 24             }
 25             catch (const Exception& ex)
 26             {
 27                 try
 28                 {
 29                     Console.Error() << ex.ToString() << endl();
 30                 }
 31                 catch (const Exception&)
 32                 {
 33                 }
 34                 RtExit(255);
 35             }
 36         }
 37         public CharClass(char c) : ranges(c), inverse(false), bits(256)
 38         {
 39             bits.Set(cast<byte>(c));
 40         }
 41         public CharClass(char first, char last) : ranges(string(first) + "-" + string(last)), inverse(false), bits(256)
 42         {
 43             InitBits();
 44         }
 45         public CharClass(const string& ranges_, bool inverse_) : ranges(ranges_), inverse(inverse_), bits(256)
 46         {
 47             InitBits();
 48         }
 49         public nothrow void Clear()
 50         {
 51             ranges.Clear();
 52             inverse = false;
 53             bits.Clear();
 54         }
 55         public string Ranges() const
 56         {
 57             string rep;
 58             for (char c : ranges)
 59             {
 60                 rep.Append(CharStr(c));
 61             }
 62             return rep;
 63         }
 64         public nothrow bool Inverse() const
 65         {
 66             return inverse;
 67         }
 68         public nothrow const Bitset& Bits() const
 69         {
 70             return bits;
 71         }
 72         public bool Contains(char c) const
 73         {
 74             return bits.Test(cast<byte>(c)) != inverse;
 75         }
 76         private void InitBits()
 77         {
 78             long i = 0;
 79             long n = ranges.Length();
 80             while (i < n)
 81             {
 82                 byte first = cast<byte>(ranges[i]);
 83                 byte last = first;
 84                 ++i;
 85                 if (i < n)
 86                 {
 87                     if (ranges[i] == '-')
 88                     {
 89                         ++i;
 90                         if (i < n)
 91                         {
 92                             last = cast<byte>(ranges[i]);
 93                             ++i;
 94                         }
 95                         else
 96                         {
 97                             bits.Set(first);
 98                             first = cast<byte>('-');
 99                             last = first;
100                         }
101                     }
102                 }
103                 for (byte b = first; b <= last; ++b;)
104                 {
105                     bits.Set(b);
106                 }
107             }
108         }
109         private string ranges;
110         private bool inverse;
111         private Bitset bits;
112     }

114     public string ToRanges(Class cls)
115     {
116         switch (cls)
117         {
118             case Class.alnum:   return "a-zA-Z0-9";
119             case Class.alpha:   return "a-zA-Z";
120             case Class.blank:   return "\x9\xb\xc\x20";
121             case Class.cntrl:   return "\x1-\x1f\x7f";
122             case Class.digit:   return "0-9";
123             case Class.graph:   return "\x21-x7e";
124             case Class.lower:   return "a-z";
125             case Class.print:   return "\x20-x7e";
126             case Class.punct:   return "\x21-x2f\x3a-\x40\x5b-\x60";
127             case Class.space:   return "\x9-\xd\x20";
128             case Class.upper:   return "A-Z";
129             case Class.word:    return "a-zA-Z0-9_";
130             case Class.xdigit:  return "0-9a-fA-F";
131         }
132         return string();
133     }

135     public OutputStream& operator<<(OutputStream& s, const CharClass& cls)
136     {
137         s << '[';
138         if (cls.Inverse())
139         {
140             s << '^';
141         }
142         s << cls.Ranges() << ']';
143         return s;
144     }

146     public StreamWriter& operator<<(StreamWriter& s, const CharClass& cls)
147     {
148         s << '[';
149         if (cls.Inverse())
150         {
151             s << '^';
152         }
153         s << cls.Ranges() << ']';
154         return s;
155     }
