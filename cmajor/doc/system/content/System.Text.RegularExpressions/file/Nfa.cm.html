 12     public class NfaStateNumberSet
 13     {
 14         public nothrow NfaStateNumberSet()
 15         {
 16         }
 17         public NfaStateNumberSet(int s)
 18         {
 19             stateNumbers.Add(s);
 20         }
 21         public nothrow bool Contains(int stateNumber)
 22         {
 23             for (int s : stateNumbers)
 24             {
 25                 if (s == stateNumber) return true;
 26             }
 27             return false;
 28         }
 29         public nothrow bool CanonicalContains(int stateNumber)
 30         {
 31             List<int>.ConstIterator i = LowerBound(stateNumbers.CBegin(), stateNumbers.CEnd(), stateNumber);
 32             return i != stateNumbers.CEnd() && *i == stateNumber;
 33         }
 34         public void Add(int stateNumber)
 35         {
 36             if (!Contains(stateNumber))
 37             {
 38                 stateNumbers.Add(stateNumber);
 39             }
 40         }
 41         public void Merge(const NfaStateNumberSet& x)
 42         {
 43             for (int s : x.stateNumbers)
 44             {
 45                 Add(s);
 46             }
 47         }
 48         public void MakeCanonical()
 49         {
 50             Sort(stateNumbers);
 51         }
 52         public inline nothrow bool IsEmpty() const
 53         {
 54             return stateNumbers.IsEmpty();
 55         }
 56         public string ToString() const
 57         {
 58             string s = "{";
 59             bool first = true;
 60             for (int n : stateNumbers)
 61             {
 62                 if (first)
 63                 {
 64                     first = false;
 65                 }
 66                 else
 67                 {
 68                     s.Append(", ");
 69                 }
 70                 s.Append(ToString(n));
 71             }
 72             s.Append("}");
 73             return s;
 74         }
 75         public nothrow const List<int>& Rep() const
 76         {
 77             return stateNumbers;
 78         }
 79         private List<int> stateNumbers;
 80     }

106         }

120         {

122         }
123         public nothrow const NfaStateNumberSet& Next(char c) const

126         }

136     public OutputStream& operator<<(OutputStream& s, const NfaState& state)
137     {
138         s << state.Id() << ": ";
139         for (int i = 0; i < 256; ++i;)
140         {
141             char c = cast<char>(i);
142             const NfaStateNumberSet& next = state.Next(c);
143             if (!next.IsEmpty())
144             {
145                 s << "(" << CharStr(c) << " -> " << next << ")";
146             }
147         }
148         return s << endl();
149     }
