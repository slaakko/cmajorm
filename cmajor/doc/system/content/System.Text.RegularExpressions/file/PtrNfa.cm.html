 14     public class PtrNfaState
 15     {
 16         public PtrNfaState() : id(-1), edges(), accepting(false)
 17         {
 18         }
 19         suppress PtrNfaState(const PtrNfaState&);
 20         suppress void operator=(const PtrNfaState&);
 21         public inline nothrow int Id() const
 22         {
 23             return id;
 24         }
 25         public inline nothrow void SetId(int id_)
 26         {
 27             id = id_;
 28         }
 29         public void AddEdge(const PtrNfaEdge& edge)
 30         {
 31             edges.Add(edge);
 32         }
 33         public inline nothrow const List<PtrNfaEdge>& Edges() const
 34         {
 35             return edges;
 36         }
 37         public void SetEdges(const List<PtrNfaEdge>& edges_)
 38         {
 39             edges = edges_;
 40         }
 41         public List<PtrNfaState*> Next(char c)
 42         {
 43             List<PtrNfaState*> next;
 44             for (const PtrNfaEdge& edge : edges)
 45             {
 46                 if (edge.Chars().Contains(c))
 47                 {
 48                     next.Add(edge.Next());
 49                 }
 50             }
 51             return next;
 52         }
 53         public nothrow void Clear()
 54         {
 55             id = -1;
 56             edges.Clear();
 57             accepting = false;
 58         }
 59         public inline nothrow bool Accepting() const
 60         {
 61             return accepting;
 62         }
 63         public inline nothrow void SetAccepting()
 64         {
 65             accepting = true;
 66         }
 67         public inline nothrow void ResetAccepting()
 68         {
 69             accepting = false;
 70         }
 71         private int id;
 72         private List<PtrNfaEdge> edges;
 73         private bool accepting;
 74     }

137     public class PtrNfa
138     {
139         public PtrNfa() : start(null), end(null), states()
140         {
141         }
142         public PtrNfa(PtrNfaState* start_, PtrNfaState* end_) : start(start_), end(end_)
143         {
144             states.Add(start);
145             states.Add(end);
146         }
147         public inline nothrow PtrNfaState* Start() const
148         {
149             return start;
150         }
151         public inline void SetStart(PtrNfaState* start_)
152         {
153             start = start_;
154             states.Add(start);
155         }
156         public inline nothrow PtrNfaState* End() const
157         {
158             return end;
159         }
160         public inline void SetEnd(PtrNfaState* end_)
161         {
162             end = end_;
163             states.Add(end);
164         }
165         public nothrow const List<PtrNfaState*>& States() const
166         {
167             return states;
168         }
169         public void AddState(PtrNfaState* state)
170         {
171             states.Add(state);
172         }
173         public nothrow void NumberStates()
174         {
175             long n = states.Count();
176             for (long i = 0; i < n; ++i;)
177             {
178                 PtrNfaState* state = states[i];
179                 state->SetId(cast<int>(i));
180             }
181         }
182         private PtrNfaState* start;
183         private PtrNfaState* end;
184         private List<PtrNfaState*> states;
185     }

187     public OutputStream& operator<<(OutputStream& s, const PtrNfa& nfa)
188     {
189         for (PtrNfaState* state : nfa.States())
190         {
191             string a;
192             if (state->Accepting())
193             {
194                 a.Append('*');
195             }
196             s << a << state->Id() << ": ";
197             for (const PtrNfaEdge& edge : state->Edges())
198             {
199                 s << edge;
200             }
201             s << endl();
202         }
203         return s;
204     }
