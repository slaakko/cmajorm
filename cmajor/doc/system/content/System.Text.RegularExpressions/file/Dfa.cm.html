 12     public class DfaState
 13     {
 14         public DfaState() : id(-1), accepting(false), next(256, -1)
 15         {
 16         }
 17         public DfaState(int id_) : id(id_), accepting(false), next(256, -1)
 18         {
 19         }
 20         public inline nothrow int Id() const
 21         {
 22             return id;
 23         }
 24         public inline nothrow bool Accepting() const
 25         {
 26             return accepting;
 27         }
 28         public inline nothrow void SetAccepting()
 29         {
 30             accepting = true;
 31         }
 32         public inline nothrow int Next(char c) const
 33         {
 34             return next[cast<byte>(c)];
 35         }
 36         public inline nothrow void SetNext(char c, int n)
 37         {
 38             next[cast<byte>(c)] = n;
 39         }
 40         private int id;
 41         private bool accepting;
 42         private List<int> next;
 43     }
 44     
 45     public OutputStream& operator<<(OutputStream& s, const DfaState& state)
 46     {
 47         if (state.Accepting())
 48         {
 49             s << '*';
 50         }
 51         s << state.Id() << ": ";
 52         for (int i = 1; i < 256; ++i;)
 53         {
 54             char c = cast<char>(i);
 55             int n = state.Next(c);
 56             if (n != -1)
 57             {
 58                 s << "(" << CharStr(c) << "->" << n << ")";
 59             }
 60         }
 61         return s << endl();
 62     }

 83     public class Dfa
 84     {
 85         public Dfa()
 86         {
 87         }
 88         public void AddState(const DfaState& state)
 89         {
 90             states.Add(state);
 91         }
 92         public inline nothrow const DfaState& GetState(int i) const
 93         {
 94             return states[i];
 95         }
 96         public inline nothrow long NumStates() const
 97         {
 98             return states.Count();
 99         }
100         public nothrow void SetNext(char c, int from, int to)
101         {
102             DfaState& s = states[from];
103             s.SetNext(c, to);
104         }
105         public inline nothrow int Next(char c, int s)
106         {
107             return states[s].Next(c);
108         }
109         public void SetAccepting(int s)
110         {
111             states[s].SetAccepting();
112         }
113         private List<DfaState> states;
114     }
115     
116     public OutputStream& operator<<(OutputStream& s, const Dfa& dfa)
117     {
118         for (int i = 0; i < dfa.NumStates(); ++i;)
119         {
120             const DfaState& state = dfa.GetState(i);
121             s << state;
122         }
123         return s;
124     }
