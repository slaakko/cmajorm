 11     public Nfa ToNfa(const PtrNfa& ptrNfa)
 12     {
 13         Nfa nfa;
 14         for (PtrNfaState* ptrNfaState : ptrNfa.States())
 15         {
 16             NfaState nfaState(ptrNfaState->Id());
 17             if (ptrNfaState->Accepting())
 18             {
 19                 nfaState.SetAccepting();
 20             }
 21             for (int i = 0; i < 256; ++i;)
 22             {
 23                 char c = cast<char>(i);
 24                 NfaStateNumberSet s;
 25                 List<PtrNfaState*> next = ptrNfaState->Next(c);
 26                 for (PtrNfaState* n : next)
 27                 {
 28                     s.Add(n->Id());
 29                 }
 30                 s.MakeCanonical();
 31                 nfaState.SetNext(c, s);
 32             }
 33             nfa.AddState(nfaState);
 34         }
 35         return nfa;
 36     }

 46     public NfaStateNumberSet Move(const NfaStateNumberSet& states, char c, const Nfa& nfa)
 47     {
 48         NfaStateNumberSet next;
 49         for (int s : states.Rep())
 50         {
 51             const NfaState& state = nfa.GetState(s);
 52             next.Merge(state.Next(c));
 53         }
 54         return next;
 55     }

 57     public NfaStateNumberSet EpsilonClosure(const NfaStateNumberSet& states, const Nfa& nfa)
 58     {
 59         Stack<int> stack;
 60         for (int s : states.Rep())
 61         {
 62             stack.Push(s);
 63         }
 64         NfaStateNumberSet epsilonClosure = states;
 65         while (!stack.IsEmpty())
 66         {
 67             int t = stack.Pop();
 68             const NfaState& s = nfa.GetState(t);
 69             NfaStateNumberSet u = s.Epsilon();
 70             for (int v : u.Rep())
 71             {
 72                 if (!epsilonClosure.Contains(v))
 73                 {
 74                     epsilonClosure.Add(v);
 75                     stack.Push(v);
 76                 }
 77             }
 78         }
 79         epsilonClosure.MakeCanonical();
 80         return epsilonClosure;
 81     }

 83     public NfaStateNumberSet EpsilonClosure(int s, const Nfa& nfa)
 84     {
 85         NfaStateNumberSet states(s);
 86         return EpsilonClosure(states, nfa);
 87     }

 89     public bool MatchesSlow(const string& s, const Nfa& nfa)
 90     {
 91         NfaStateNumberSet states = EpsilonClosure(0, nfa);
 92         for (char c : s)
 93         {
 94             states = EpsilonClosure(Move(states, c, nfa), nfa);
 95         }
 96         for (int s : states.Rep())
 97         {
 98             const NfaState& state = nfa.GetState(s);
 99             if (state.Accepting())
100             {
101                 return true;
102             }
103         }
104         return false;
105     }

107     public void AddState(int s, List<int>& newStates, Bitset& alreadyOn, const Nfa& nfa)
108     {
109         newStates.Add(s);
110         alreadyOn.Set(s);
111         const NfaState& state = nfa.GetState(s);
112         for (int t : state.Epsilon().Rep())
113         {
114             if (!alreadyOn.Test(t))
115             {
116                 AddState(t, newStates, alreadyOn, nfa);
117             }
118         }
119     }

121     public void Transform(char c, List<int>& oldStates, List<int>& newStates, Bitset& alreadyOn, const Nfa& nfa)
122     {
123         while (!oldStates.IsEmpty())
124         {
125             int s = oldStates.Back();
126             const NfaState& state = nfa.GetState(s);
127             for (int t : state.Next(c).Rep())
128             {
129                 if (!alreadyOn.Test(t))
130                 {
131                     AddState(t, newStates, alreadyOn, nfa);
132                 }
133             }
134             oldStates.RemoveLast();
135         }
136         while (!newStates.IsEmpty())
137         {
138             int s = newStates.Back();
139             newStates.RemoveLast();
140             oldStates.Add(s);
141             alreadyOn.Reset(s);
142         }
143     }

145     public bool Matches(const string& s, const Nfa& nfa)
146     {
147         Bitset alreadyOn(nfa.NumStates());
148         NfaStateNumberSet states = EpsilonClosure(0, nfa);
149         List<int> oldStates = states.Rep();
150         List<int> newStates;
151         for (char c : s)
152         {
153             Transform(c, oldStates, newStates, alreadyOn, nfa);
154         }
155         for (int s : oldStates)
156         {
157             const NfaState& state = nfa.GetState(s);
158             if (state.Accepting())
159             {
160                 return true;
161             }
162         }
163         return false;
164     }

166     public Dfa ToDfa(const Nfa& nfa)
167     {
168         Dfa dfa;
169         DfaState start(0);
170         dfa.AddState(start);
171         List<NfaStateNumberSet> dstates;
172         dstates.Add(EpsilonClosure(0, nfa));
173         Stack<int> unmarked;
174         unmarked.Push(0);
175         while (!unmarked.IsEmpty())
176         {
177             int t = unmarked.Pop();
178             for (int i = 1; i < 256; ++i;)
179             {
180                 char c = cast<char>(i);
181                 NfaStateNumberSet u = EpsilonClosure(Move(dstates[t], c, nfa), nfa);
182                 if (!u.IsEmpty())
183                 {
184                     int n = FindState(dstates.CBegin(), dstates.CEnd(), u);
185                     if (n == -1)
186                     {
187                         n = cast<int>(dstates.Count());
188                         DfaState s(n);
189                         dfa.AddState(s);
190                         unmarked.Push(n);
191                         dstates.Add(u);
192                     }
193                     dfa.SetNext(c, t, n);
194                 }
195             }
196         }
197         for (int i = 0; i < dstates.Count(); ++i;)
198         {
199             for (int s : dstates[i].Rep())
200             {
201                 if (nfa.GetState(s).Accepting())
202                 {
203                     dfa.SetAccepting(i);
204                     break;
205                 }
206             }
207         }
208         return dfa;
209     }

211     public nothrow bool Matches(const string& s, const Dfa& dfa)
212     {
213         int state = 0;
214         for (char c : s)
215         {
216             state = dfa.Next(c, state);
217             if (state == -1)
218             {
219                 return false;
220             }
221         }
222         return dfa.GetState(state).Accepting();
223     }
