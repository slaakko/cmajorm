 29     public PtrNfa Cat(const PtrNfa& left, const PtrNfa& right)
 30     {
 31         PtrNfa cat(left);
 32         PtrNfaState* leftEnd = cat.End();
 33         leftEnd->ResetAccepting();
 34         PtrNfaState* rightStart = right.Start();
 35         leftEnd->SetEdges(rightStart->Edges());
 36         long n = right.States().Count();
 37         for (long i = 1; i < n - 1; ++i;)
 38         {
 39             cat.AddState(right.States()[i]);
 40         }
 41         cat.SetEnd(right.End());
 42         return cat;
 43     }

 45     public PtrNfa Alt(PtrNfaStateFactory* factory, const PtrNfa& left, const PtrNfa& right)
 46     {
 47         PtrNfaState* leftStart = left.Start();
 48         PtrNfaState* leftEnd = left.End();
 49         PtrNfaState* rightStart = right.Start();
 50         PtrNfaState* rightEnd = right.End();
 51         PtrNfaState* start = factory->MakeState();
 52         PtrNfaState* end = factory->MakeState();
 53         end->SetAccepting();
 54         start->AddEdge(PtrNfaEdge(CharClass(eps), leftStart));
 55         start->AddEdge(PtrNfaEdge(CharClass(eps), rightStart));
 56         PtrNfa alt;
 57         alt.SetStart(start);
 58         for (PtrNfaState* state : left.States())
 59         {
 60             alt.AddState(state);
 61         }
 62         for (PtrNfaState* state : right.States())
 63         {
 64             alt.AddState(state);
 65         }
 66         leftEnd->ResetAccepting();
 67         leftEnd->AddEdge(PtrNfaEdge(CharClass(eps), end));
 68         rightEnd->ResetAccepting();
 69         rightEnd->AddEdge(PtrNfaEdge(CharClass(eps), end));
 70         alt.SetEnd(end);
 71         return alt;
 72     }

 78         PtrNfaState* end = factory->MakeState();
 79         end->SetAccepting();
 80         start->AddEdge(PtrNfaEdge(CharClass(eps), end));
 81         kleene.SetStart(start);
 82         PtrNfaState* subjectStart = subject.Start();
 83         start->AddEdge(PtrNfaEdge(CharClass(eps), subjectStart));
 84         for (PtrNfaState* state : subject.States())
 85         {
 86             kleene.AddState(state);
 87         }
 88         PtrNfaState* subjectEnd = subject.End();
 89         subjectEnd->ResetAccepting();
 90         subjectEnd->AddEdge(PtrNfaEdge(CharClass(eps), end));
 91         subjectEnd->AddEdge(PtrNfaEdge(CharClass(eps), subjectStart));
 92         kleene.SetEnd(end);
 93         return kleene;

124     public class RegExParser
125     {
126         static RegExParser() : instance(new RegExParser())
127         {
128         }
129         public RegExParser() : grammar(new RegularExpressionGrammar())
130         {
131         }
132         suppress RegExParser(const RegExParser&);
133         suppress void operator=(const RegExParser&);
134         suppress RegExParser(RegExParser&&);
135         suppress void operator=(RegExParser&&);
136         public static RegExParser& Instance()
137         {
138             return *instance;
139         }
140         public PtrNfa Parse(PtrNfaStateFactory* factory, const string& regEx)
141         {
142             ustring re = ToUtf32(regEx);
143             PtrNfa ptrNfa = grammar->Parse(re.Chars(), re.Chars() + re.Length(), 0, "", factory);
144             ptrNfa.NumberStates();
145             return ptrNfa;
146         }
147         private static UniquePtr<RegExParser> instance;
148         private RegularExpressionGrammar* grammar;
149     }

151     public PtrNfa ParseRegEx(PtrNfaStateFactory* factory, const string& pattern, RegExFlags flags)
152     {
153         if ((flags & RegExFlags.matchLine) != RegExFlags.none)
154         {
155             string linePattern = pattern;
156             if (!pattern.IsEmpty())
157             {
158                 if (pattern[0] == '^')
159                 {
160                     linePattern = linePattern.Substring(1);
161                 }
162                 else
163                 {
164                     linePattern = ".*" + linePattern;
165                 }
166                 if (pattern[pattern.Length() - 1] == '$')
167                 {
168                     linePattern = linePattern.Substring(0, linePattern.Length() - 1);
169                 }
170                 else
171                 {
172                     linePattern.Append(".*");
173                 }
174             }
175             return RegExParser.Instance().Parse(factory, linePattern);
176         }
177         return RegExParser.Instance().Parse(factory, pattern);
178     }
