 6 using System;
 7 
 8 namespace System.Text.RegularExpressions
 9 {
10     public enum RegExFlags
11     {
12         none = 0, compiled = 1 << 0, matchLine = 1 << 1
13     }
14     
15     public class RegEx
16     {
17         public RegEx(const string& pattern_) : this(pattern_, RegExFlags.none)
18         {
19         }
20         public RegEx(const string& pattern_, RegExFlags flags) : pattern(pattern_), nfa(ToNfa(pattern, flags)), dfa(), compiled(false)
21         {
22             if ((flags & RegExFlags.compiled) != RegExFlags.none)
23             {
24                 dfa = ToDfa(nfa);
25                 compiled = true;
26             }
27         }
28         public inline nothrow const string& Pattern() const
29         {
30             return pattern;
31         }
32         public inline nothrow const Nfa& GetNfa() const
33         {
34             return nfa;
35         }
36         public inline nothrow const Dfa& GetDfa() const
37         {
38             return dfa;
39         }
40         public inline nothrow bool Compiled() const
41         {
42             return compiled;
43         }
44         public bool Matches(const string& s)
45         {
46             if (compiled)
47             {
48                 return Matches(s, dfa);
49             }
50             else
51             {
52                 return Matches(s, nfa);
53             }
54         }
55         private string pattern;
56         private Nfa nfa;
57         private Dfa dfa;
58         private bool compiled;
59     }
60 }
