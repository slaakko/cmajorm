 47     public class LinkedListNodeIterator<T, R, P>
 48     {
 49         public typedef T ValueType;
 50         public typedef R ReferenceType;
 51         public typedef P PointerType;
 52         private typedef LinkedListNodeIterator<ValueType, ReferenceType, PointerType> Self;
 53 
 54         public nothrow LinkedListNodeIterator() : list(null), node(null)
 55         {
 56         }
 57         public nothrow LinkedListNodeIterator(LinkedList<ValueType>* list_, LinkedListNode<ValueType>* node_) : list(list_), node(node_)
 58         {
 59         }
 60         public inline nothrow ReferenceType operator*()
 61         {
 62             #assert(node != null);
 63             return node->Value();
 64         }
 65         public inline nothrow PointerType operator->()
 66         {
 67             #assert(node != null);
 68             return &(node->Value());
 69         }
 70         public inline nothrow Self& operator++()
 71         {
 72             #assert(node != null);
 73             node = node->Next();
 74             return *this;
 75         }
 76         public inline nothrow Self& operator--()
 77         {
 78             if (node == null)
 79             {
 80                 #assert(list != null);
 81                 node = list->Tail();
 82             }
 83             else
 84             {
 85                 node = node->Prev();
 86             }
 87             return *this;
 88         }
 89         public inline nothrow LinkedListNode<ValueType>* Node() const
 90         {
 91             return node;
 92         }
 93         private LinkedList<ValueType>* list;
 94         private LinkedListNode<ValueType>* node;
 95     }

 97     public inline nothrow bool operator==<T, R, P>(const LinkedListNodeIterator<T, R, P>& left, const LinkedListNodeIterator<T, R, P>& right)
 98     {

102     public class LinkedList<T> where T is Regular
103     {
104         public typedef T ValueType;
105         public typedef LinkedListNodeIterator<ValueType, ValueType&, ValueType*> Iterator;
106         public typedef LinkedListNodeIterator<ValueType, const ValueType&, const ValueType*> ConstIterator;
107 
108         public nothrow LinkedList() : head(null), tail(null), count(0)
109         {
110         }
111         public LinkedList(const LinkedList<ValueType>& that) : head(null), tail(null), count(0)
112         {
113             CopyFrom(that);
114         }
115         public nothrow LinkedList(LinkedList<ValueType>&& that) : head(that.head), tail(that.tail), count(that.count)
116         {
117             that.head = null;
118             that.tail = null;
119             that.count = 0;
120         }
121         public void operator=(const LinkedList<ValueType>& that)
122         {
123             Clear();
124             CopyFrom(that);
125         }
126         public void operator=(LinkedList<ValueType>&& that)
127         {
128             Swap(head, that.head);
129             Swap(tail, that.tail);
130             Swap(count, that.count);
131         }
132         public ~LinkedList()
133         {
134             Clear();
135         }
136         public inline nothrow Iterator Begin()
137         {
138             return Iterator(this, head);
139         }
140         public inline nothrow ConstIterator Begin() const
141         {
142             return ConstIterator(this, head);
143         }
144         public inline nothrow ConstIterator CBegin() const
145         {
146             return ConstIterator(this, head);
147         }
148         public inline nothrow Iterator End()
149         {
150             return Iterator(this, null);
151         }
152         public inline nothrow ConstIterator End() const
153         {
154             return ConstIterator(this, null);
155         }
156         public inline nothrow ConstIterator CEnd() const
157         {
158             return ConstIterator(this, null);
159         }
160         public inline nothrow long Count() const
161         {
162             return count;
163         }
164         public inline nothrow bool IsEmpty() const
165         {
166             return count == 0;
167         }
168         public nothrow void Clear()
169         {
170             LinkedListNode<ValueType>* n = head;
171             while (n != null)
172             {
173                 LinkedListNode<ValueType>* next = n->Next();
174                 delete n;
175                 n = next;
176             }
177             head = null;
178             tail = null;
179             count = 0;
180         }
181         public Iterator InsertFront(const ValueType& value)
182         {
183             if (head == null)
184             {
185                 head = new LinkedListNode<ValueType>(value, null, null);
186                 tail = head;
187             }
188             else
189             {
190                 head = new LinkedListNode<ValueType>(value, null, head);
191                 head->Next()->SetPrev(head);
192             }
193             ++count;
194             return Iterator(this, head);
195         }
196         public Iterator Insert(Iterator pos, const ValueType& value)
197         {
198             LinkedListNode<ValueType>* next = pos.Node();
199             if (next != null)
200             {
201                 LinkedListNode<ValueType>* prev = next->Prev();
202                 LinkedListNode<ValueType>* n = new LinkedListNode<ValueType>(value, prev, next);
203                 next->SetPrev(n);
204                 if (prev != null)
205                 {
206                     prev->SetNext(n);
207                 }
208                 else
209                 {
210                     head = n;
211                 }
212                 ++count;
213                 return Iterator(this, n);
214             }
215             else
216             {
217                 Add(value);
218                 return Iterator(this, tail);
219             }
220         }
221         public void Add(const ValueType& value)
222         {
223             if (tail == null)
224             {
225                 tail = new LinkedListNode<ValueType>(value, null, null);
226                 head = tail;
227             }
228             else
229             {
230                 tail = new LinkedListNode<ValueType>(value, tail, null);
231                 tail->Prev()->SetNext(tail);
232             }
233             ++count;
234         }
235         public nothrow void RemoveFirst()
236         {
237             #assert(head != null);
238             LinkedListNode<ValueType>* n = head;
239             head = head->Next();
240             if (head != null)
241             {
242                 head->SetPrev(null);
243             }
244             else
245             {
246                 tail = null;
247             }
248             delete n;
249             --count;
250         }
251         public nothrow void RemoveLast()
252         {
253             #assert(tail != null);
254             LinkedListNode<ValueType>* n = tail;
255             tail = tail->Prev();
256             if (tail != null)
257             {
258                 tail->SetNext(null);
259             }
260             else
261             {
262                 head = null;
263             }
264             delete n;
265             --count;
266         }
267         public nothrow Iterator Remove(Iterator pos)
268         {
269             LinkedListNode<ValueType>* n = pos.Node();
270             #assert(n != null);
271             LinkedListNode<ValueType>* prev = n->Prev();
272             LinkedListNode<ValueType>* next = n->Next();
273             Iterator nxt(this, next);
274             if (prev != null)
275             {
276                 prev->SetNext(next);
277             }
278             else
279             {
280                 head = next;
281             }
282             if (next != null)
283             {
284                 next->SetPrev(prev);
285             }
286             else
287             {
288                 tail = prev;
289             }
290             delete n;
291             --count;
292             return nxt;
293         }
294         public nothrow void Remove(const ValueType& value)
295         {
296             Iterator i = Begin();
297             Iterator e = End();
298             while (i != e)
299             {
300                 if (*i == value)
301                 {
302                     Iterator r = i;
303                     ++i;
304                     Remove(r);
305                 }
306                 else
307                 {
308                     ++i;
309                 }
310             }
311         }
312         public inline nothrow const ValueType& Front() const
313         {
314             #assert(head != null);
315             return head->Value();
316         }
317         public inline nothrow const ValueType& Back() const
318         {
319             #assert(tail != null);
320             return tail->Value();
321         }
322         public inline nothrow LinkedListNode<ValueType>* Tail()
323         {
324             return tail;
325         }
326         private void CopyFrom(const LinkedList<ValueType>& that)
327         {
328             ConstIterator e = that.CEnd();
329             for (ConstIterator i = that.CBegin(); i != e; ++i;)
330             {
331                 Add(*i);
332             }
333         }
334         private LinkedListNode<ValueType>* head;
335         private LinkedListNode<ValueType>* tail;
336         private long count;
337     }

339     public nothrow bool operator==<T>(const LinkedList<T>& left, const LinkedList<T>& right) where T is Regular
340     {
341         if (left.Count() != right.Count())
342         {
343             return false;
344         }
345         return Equal(left.CBegin(), left.CEnd(), right.CBegin(), right.CEnd());
346     }

348     public nothrow bool operator<<T>(const LinkedList<T>& left, const LinkedList<T>& right) where T is TotallyOrdered
349     {
350         return LexicographicalCompare(left.CBegin(), left.CEnd(), right.CBegin(), right.CEnd());
351     }
