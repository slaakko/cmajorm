116         public void WriteULEB128UInt(uint x)
117         {
118             do
119             {
120                 byte b = cast<byte>(x & 0x7Fu);
121                 x = x >> 7u;
122                 if (x != 0u)
123                 {
124                     b = b | 0x80u;
125                 }
126                 Write(b);
127             }
128             while (x != 0u);
129         }

144         public void WriteSLEB128Int(int x)
145         {
146             bool more = true;
147             bool negative = x < 0;
148             while (more)
149             {
150                 byte b = cast<byte>(x & 0x7F);
151                 x = x >> 7;
152                 if (negative)
153                 {
154                     x = x | (~int(0) << (32 - 7));
155                 }
156                 if (x == 0 && (b & 0x40u) == 0u || x == -1 && (b & 0x40u) != 0u)
157                 {
158                     more = false;
159                 }
160                 else
161                 {
162                     b = b | 0x80u;
163                 }
164                 Write(b);
165             }
166         }
