  6 using System;

  9 namespace System.IO

 11     public static class File
 12     {
 13         public static nothrow bool Exists(const string& filePath)
 14         {
 15             return RtFileExists(filePath.Chars());
 16         }
 17         public static ulong Size(const string& filePath)
 18         {
 19             ulong fileSize = 0u;
 20             int result = RtGetFileSize(filePath.Chars(), &fileSize);
 21             if (result < 0)
 22             {
 23                 string errorMessage = RtGetError(result);
 24                 RtDisposeError(result);
 25                 throw FileSystemException("could not get size of file '" + filePath + "': " + errorMessage);
 26             }
 27             else
 28             {
 29                 return fileSize;
 30             }
 31         }
 32         public static nothrow bool LastWriteTimeLess(const string& filePath1, const string& filePath2)
 33         {
 34             return RtLastWriteTimeLess(filePath1.Chars(), filePath2.Chars());
 35         }

 54                 new FileByteStream(filePath, OpenMode.read)))));

 61         public static string ReadAllText(const string& filePath)
 62         {
 63             StreamReader reader = OpenRead(filePath);
 64             string content = reader.ReadToEnd();
 65             if (content.Length() >= 3 && cast<byte>(content[0]) == 0xEFu && cast<byte>(content[1]) == 0xBBu && cast<byte>(content[2]) == 0xBFu)
 66             {
 67                 return content.Substring(3);
 68             }
 69             else
 70             {
 71                 return content;
 72             }
 73         }
 74         public static List<string> ReadAllLines(const string& filePath)
 75         {
 76             List<string> lines;
 77             bool start = true;
 78             StreamReader reader = OpenRead(filePath);
 79             string line = reader.ReadLine();
 80             while (!reader.EndOfStream())
 81             {
 82                 if (start)
 83                 {
 84                     if (line.Length() >= 3 && cast<byte>(line[0]) == 0xEFu && cast<byte>(line[1]) == 0xBBu && cast<byte>(line[2]) == 0xBFu)
 85                     {
 86                         line = line.Substring(3);
 87                     }
 88                     start = false;
 89                 }
 90                 lines.Add(line);
 91                 line = reader.ReadLine();
 92             }
 93             if (!line.IsEmpty())
 94             {
 95                 lines.Add(line);
 96             }
 97             return lines;
 98         }
 99         public static void Remove(const string& filePath)
100         {
101             if (Exists(filePath))
102             {
103                 int result = RtRemoveFile(filePath.Chars());
104                 if ( result < 0)
105                 {
106                     string errorMessage = RtGetError(result);
107                     RtDisposeError(result);
108                     throw FileSystemException("could not remove file '" + filePath + "': " + errorMessage);
109                 }
110             }
111             else
112             {
113                 throw FileSystemException("could not remove file '" + filePath + "': file not found");
114             }
115         }
116         public static void Copy(const string& sourceFilePath, const string& targetFilePath)
117         {
118             int result = RtCopyFile(sourceFilePath.Chars(), targetFilePath.Chars());
119             if ( result < 0)
120             {
121                 string errorMessage = RtGetError(result);
122                 RtDisposeError(result);
123                 throw FileSystemException("could not copy file '" + sourceFilePath + "' to file '" + targetFilePath + "': " + errorMessage);
124             }
125         }
