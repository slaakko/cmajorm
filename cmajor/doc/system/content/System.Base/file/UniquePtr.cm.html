  6 namespace System
  7 {
  8     public class UniquePtr<T>
  9     {
 10         private typedef UniquePtr<T> Self;
 11 
 12         public nothrow UniquePtr() : ptr(null)
 13         {
 14         }
 15         public explicit nothrow UniquePtr(T* ptr_) : ptr(ptr_)
 16         {
 17         }
 18         suppress UniquePtr(const Self&);
 19         public nothrow UniquePtr(Self&& that) : ptr(that.Release())
 20         {
 21         }
 22         public nothrow void operator=(T* ptr_)
 23         {
 24             if (ptr != null)
 25             {
 26                 delete ptr;
 27             }
 28             ptr = ptr_;
 29         }
 30         suppress void operator=(const Self&);
 31         public nothrow void operator=(Self&& that)
 32         {
 33             if (ptr != null)
 34             {
 35                 delete ptr;
 36             }
 37             ptr = that.Release();
 38         }
 39         public ~UniquePtr()
 40         {
 41             if (ptr != null)
 42             {
 43                 delete ptr;
 44             }
 45         }
 46         public nothrow void Reset()
 47         {
 48             if (ptr != null)
 49             {
 50                 delete ptr;
 51                 ptr = null;
 52             }
 53         }
 54         public nothrow void Reset(T* ptr_)
 55         {
 56             if (ptr != null)
 57             {
 58                 delete ptr;
 59             }
 60             ptr = ptr_;
 61         }
 62         public inline nothrow T* Release()
 63         {
 64             T* ptr_ = ptr;
 65             ptr = null;
 66             return ptr_;
 67         }
 68         public inline nothrow T* GetPtr()
 69         {
 70             return ptr;
 71         }
 72         public inline nothrow T* Get()
 73         {
 74             return ptr;
 75         }
 76         public inline nothrow bool IsNull() const
 77         {
 78             return ptr == null;
 79         }
 80         public inline nothrow T* operator->()
 81         {
 82             #assert(ptr != null);
 83             return ptr;
 84         }
 85         public inline nothrow const T* operator->() const
 86         {
 87             #assert(ptr != null);
 88             return ptr;
 89         }
 90         public inline nothrow T& operator*()
 91         {
 92             #assert(ptr != null);
 93             return *ptr;
 94         }
 95         public inline nothrow const T& operator*() const
 96         {
 97             #assert(ptr != null);
 98             return *ptr;
 99         }
100         public nothrow void Swap(Self& that)
101         {
102             Swap(ptr, that.ptr);
103         }
104         private T* ptr;
105     }
106 
107     public inline nothrow bool operator==<T>(const UniquePtr<T>& left, const UniquePtr<T>& right)
108     {
109         return left.Get() == right.Get();
110     }
111 
112     public inline nothrow bool operator<<T>(const UniquePtr<T>& left, const UniquePtr<T>& right)
113     {
114         return left.Get() < right.Get();
115     }
116 }
