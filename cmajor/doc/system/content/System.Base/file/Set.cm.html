 6 using System;
 7 using System.Concepts;
 8 
 9 namespace System.Collections
10 {
11     public class Set<T, C = Less<T>>
12         where T is Semiregular and C is Relation and C.Domain is T
13     {
14         public typedef T ValueType;
15         public typedef T KeyType;
16         public typedef C Compare;
17         public typedef TreeType.ConstIterator ConstIterator;
18         public typedef TreeType.Iterator Iterator;
19         private typedef Set<ValueType, Compare> Self;
20         private typedef RedBlackTree<KeyType, ValueType, Identity<ValueType>, Compare> TreeType;
21 
22         public inline nothrow Iterator Begin()
23         {
24             return tree.Begin();
25         }
26         public inline nothrow ConstIterator Begin() const
27         {
28             return tree.CBegin();
29         }
30         public inline nothrow ConstIterator CBegin() const
31         {
32             return tree.CBegin();
33         }
34         public inline nothrow Iterator End()
35         {
36             return tree.End();
37         }
38         public inline nothrow ConstIterator End() const
39         {
40             return tree.CEnd();
41         }
42         public inline nothrow ConstIterator CEnd() const
43         {
44             return tree.CEnd();
45         }
46         public inline nothrow long Count() const
47         {
48             return tree.Count();
49         }
50         public inline nothrow bool IsEmpty() const
51         {
52             return tree.IsEmpty();
53         }
54         public nothrow void Clear()
55         {
56             tree.Clear();
57         }
58         public inline nothrow Iterator Find(const KeyType& key)
59         {
60             return tree.Find(key);
61         }
62         public inline nothrow ConstIterator Find(const KeyType& key) const
63         {
64             return tree.CFind(key);
65         }
66         public inline nothrow ConstIterator CFind(const KeyType& key) const
67         {
68             return tree.CFind(key);
69         }
70         public inline Pair<Iterator, bool> Insert(const ValueType& value)
71             where T is Copyable
72         {
73             return tree.Insert(value);
74         }
75         public inline nothrow bool Remove(const KeyType& key)
76         {
77             return tree.Remove(key);
78         }
79         public inline nothrow void Remove(Iterator pos)
80         {
81             tree.Remove(pos);
82         }
83         private TreeType tree;
84     }
85 
86     public inline nothrow bool operator==<T, C>(const Set<T, C>& left, const Set<T, C>& right)
87         where T is Regular and C is Relation and C.Domain is T
88     {
89         return left.Count() == right.Count() && Equal(left.CBegin(), left.CEnd(), right.CBegin(), right.CEnd(), EqualTo<T>());
90     }
91 
92     public inline nothrow bool operator<<T, C>(const Set<T, C>& left, const Set<T, C>& right)
93         where T is Semiregular and C is Relation and C.Domain is T
94     {
95         return LexicographicalCompare(left.CBegin(), left.CEnd(), right.CBegin(), right.CEnd(), C());
96     }
