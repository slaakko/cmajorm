124         public void Open(const string& filePath_, OpenMode openMode)
125         {
126             if (opened)
127             {
128                 Close();
129             }
130             filePath = filePath_;
131             if (filePath.IsEmpty())
132             {
133                 throw FileSystemException("provided file name is empty");
134             }
135             int result = RtOpen(filePath.Chars(), openMode);
136             if (result < 0)
137             {
138                 string errorMessage = RtGetError(result);
139                 RtDisposeError(result);
140                 throw FileSystemException(errorMessage);
141             }
142             else
143             {
144                 fileHandle = result;
145             }
146             opened = true;
147         }
148         public void Close()
149         {
150             if (opened)
151             {
152                 opened = false;
153                 int result = RtClose(fileHandle);
154                 if (result < 0)
155                 {
156                     string errorMessage = RtGetError(result);
157                     RtDisposeError(result);
158                     throw FileSystemException(errorMessage);
159                 }
160                 fileHandle = -1;
161             }
162             else
163             {
164                 throw FileSystemException("file not opened");
165             }
166         }
167         public override int Read()
168         {
169             return RtReadByte(fileHandle);
170         }
171         public override string ReadLine()
172         {
173             string line;
174             int x = RtReadByte(fileHandle);
175             bool prevWasCR = false;
176             while (x != -1 && cast<char>(x) != '\n')
177             {
178                 char c = cast<char>(x);
179                 if (c == '\r')
180                 {
181                     if (prevWasCR)
182                     {
183                         line.Append(c);
184                     }
185                     prevWasCR = true;
186                 }
187                 else
188                 {
189                     if (prevWasCR)
190                     {
191                         line.Append('\r');
192                         prevWasCR = false;
193                     }
194                     line.Append(c);
195                 }
196                 x = RtReadByte(fileHandle);
197             }
198             endOfStream = x == -1;
199             return line;
200         }
201         public override string ReadToEnd()
202         {
203             string content;
204             while (!endOfStream)
205             {
206                 long result = RtRead(fileHandle, buffer.Mem(), buffer.Size());
207                 if (result < 0)
208                 {
209                     int res = cast<int>(result);
210                     string errorMessage = RtGetError(res);
211                     RtDisposeError(res);
212                     throw FileSystemException(errorMessage);
213                 }
214                 else if (result == 0)
215                 {
216                     endOfStream = true;
217                 }
218                 else
219                 {
220                     content.Append(cast<const char*>(buffer.Mem()), result);
221                 }
222             }
223             return content;
224         }
