129         public void Open(const string& filePath_, OpenMode openMode)
130         {
131             if (opened)
132             {
133                 Close();
134             }
135             filePath = filePath_;
136             if (filePath.IsEmpty())
137             {
138                 throw FileSystemException("provided file name is empty");
139             }
140             int result = RtOpen(filePath.Chars(), openMode);
141             if (result < 0)
142             {
143                 string errorMessage = RtGetError(result);
144                 RtDisposeError(result);
145                 throw FileSystemException(errorMessage);
146             }
147             else
148             {
149                 fileHandle = result;
150             }
151             opened = true;
152         }
153         public void Close()
154         {
155             if (opened)
156             {
157                 opened = false;
158                 int result = RtClose(fileHandle);
159                 if (result < 0)
160                 {
161                     string errorMessage = RtGetError(result);
162                     RtDisposeError(result);
163                     throw FileSystemException(errorMessage);
164                 }
165                 fileHandle = -1;
166             }
167             else
168             {
169                 throw FileSystemException("file not opened");
170             }
171         }
172         public override int Read()
173         {
174             return RtReadByte(fileHandle);
175         }
176         public override string ReadLine()
177         {
178             string line;
179             int x = RtReadByte(fileHandle);
180             bool prevWasCR = false;
181             while (x != -1 && cast<char>(x) != '\n')
182             {
183                 char c = cast<char>(x);
184                 if (c == '\r')
185                 {
186                     if (prevWasCR)
187                     {
188                         line.Append(c);
189                     }
190                     prevWasCR = true;
191                 }
192                 else
193                 {
194                     if (prevWasCR)
195                     {
196                         line.Append('\r');
197                         prevWasCR = false;
198                     }
199                     line.Append(c);
200                 }
201                 x = RtReadByte(fileHandle);
202             }
203             endOfStream = x == -1;
204             return line;
205         }
206         public override string ReadToEnd()
207         {
208             string content;
209             while (!endOfStream)
210             {
211                 long result = RtRead(fileHandle, buffer.Mem(), buffer.Size());
212                 if (result < 0)
213                 {
214                     int res = cast<int>(result);
215                     string errorMessage = RtGetError(res);
216                     RtDisposeError(res);
217                     throw FileSystemException(errorMessage);
218                 }
219                 else if (result == 0)
220                 {
221                     endOfStream = true;
222                 }
223                 else
224                 {
225                     content.Append(cast<const char*>(buffer.Mem()), result);
226                 }
227             }
228             return content;
229         }
