289         public Pair<Iterator, bool> Insert(const ValueType& value)
290         {
291             if (buckets.Count() == 0)
292             {
293                 buckets.Resize(GetNextHashtablePrime(0));
294             }
295             const KeyType& key = KeyOf(value);
296             long index = Hash(key);
297             #assert(index != -1);
298             Bucket<ValueType>* bucket = buckets[index];
299             while (bucket != null)
300             {
301                 if (KeysEqual(KeyOf(bucket->Value()), key))
302                 {
303                     return MakePair(Iterator(this, bucket), false);
304                 }
305                 bucket = bucket->Next();
306             }
307             bucket = new Bucket<ValueType>(value, buckets[index]);
308             buckets[index] = bucket;
309             ++count;
310             SetLoadFactor();
311             CheckForRehash();
312             return MakePair(Iterator(this, bucket), true);
313         }
314         public nothrow void Remove(const KeyType& key)
315         {
316             long index = Hash(key);
317             if (index == -1) return;
318             Bucket<ValueType>* bucket = buckets[index];
319             Bucket<ValueType>* prev = null;
320             while (bucket != null)
321             {
322                 if (KeysEqual(KeyOf(bucket->Value()), key))
323                 {
324                     if (prev != null)
325                     {
326                         prev->SetNext(bucket->Next());
327                     }
328                     else
329                     {
330                         buckets[index] = bucket->Next();
331                     }
332                     delete bucket;
333                     --count;
334                     SetLoadFactor();
335                     return;
336                 }
337                 prev = bucket;
338                 bucket = bucket->Next();
339             }
340         }
341         public nothrow void Remove(Iterator pos)
342         {
343             Bucket<ValueType>* bucket = pos.GetBucket();
344             if (bucket != null)
345             {
346                 long index = Hash(KeyOf(bucket->Value()));
347                 if (index == -1) return;
348                 Bucket<ValueType>* b = buckets[index];
349                 Bucket<ValueType>* prev = null;
350                 while (b != bucket && b != null)
351                 {
352                     prev = b;
353                     b = b->Next();
354                 }
355                 if (b == bucket)
356                 {
357                     if (prev != null)
358                     {
359                         prev->SetNext(b->Next());
360                     }
361                     else
362                     {
363                         buckets[index] = b->Next();
364                     }
365                     delete bucket;
366                     --count;
367                     SetLoadFactor();
368                 }
369             }
370         }

434         private inline nothrow void SetLoadFactor()

447         private inline void CheckForRehash()

454         private void Rehash()
455         {
456             List<Bucket<ValueType>*> b;
457             Swap(buckets, b);
458             long n = b.Count();
459             buckets.Resize(GetNextHashtablePrime(n + 1));
460             for (long i = 0; i < n; ++i;)
461             {
462                 Bucket<ValueType>* bucket = b[i];
463                 while (bucket != null)
464                 {
465                     const KeyType& key = KeyOf(bucket->Value());
466                     long index = Hash(key);
467                     #assert(index != -1);
468                     Bucket<ValueType>* next = bucket->Next();
469                     bucket->SetNext(buckets[index]);
470                     buckets[index] = bucket;
471                     bucket = next;
472                 }
473             }
474         }
