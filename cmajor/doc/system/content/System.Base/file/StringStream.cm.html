 10     public class InputStringStream : InputStream
 11     {
 12         public nothrow InputStringStream() : str(), pos(0)
 13         {
 14         }
 15         public nothrow InputStringStream(const string& str_) : str(str_), pos(0)
 16         {
 17         }
 18         suppress InputStringStream(const InputStringStream&);
 19         suppress void operator=(const InputStringStream&);
 20         public default nothrow InputStringStream(InputStringStream&&);
 21         public default nothrow void operator=(InputStringStream&&);
 22         public inline nothrow const string& Str() const
 23         {
 24             return str;
 25         }
 26         public nothrow void SetStr(const string& str_)
 27         {
 28             str = str_;
 29         }
 30         public override int Read()
 31         {
 32             if (pos < str.Length())
 33             {
 34                 return cast<int>(str[pos++]);
 35             }
 36             return -1;
 37         }
 38         public override string ReadLine()
 39         {
 40             string line;
 41             while (pos < str.Length())
 42             {
 43                 char c = str[pos++];
 44                 if (c == '\n')
 45                 {
 46                     return line;
 47                 }
 48                 line.Append(c);
 49             }
 50             return line;
 51         }
 52         public override string ReadToEnd()
 53         {
 54             long p = pos;
 55             pos = str.Length();
 56             return str.Substring(p);
 57         }
 58         public override nothrow bool EndOfStream() const
 59         {
 60             return pos == str.Length();
 61         }
 62         private string str;
 63         private long pos;
 64     }
