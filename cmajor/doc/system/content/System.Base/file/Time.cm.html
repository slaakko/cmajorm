409     {
410         static nothrow TimestampProvider() : instance(new TimestampProvider())
411         {
412         }
413         public static nothrow TimestampProvider& Instance()
414         {
415             return *instance;
416         }
417         private nothrow TimestampProvider()
418         {
419             Reset();
420         }
421         public nothrow Timestamp GetCurrentTimestamp()
422         {
423             if (GetCurrentDate() != startDateTime.GetDate())
424             {
425                 Reset();
426             }
427             Duration elapsed = Now() - startTimePoint;
428             long elapsedNanosecs = elapsed.Nanoseconds();
429             int secs = cast<int>(elapsedNanosecs / cast<long>(1000000000));
430             int nanosecs = cast<int>(elapsedNanosecs % cast<long>(1000000000));
431             Timestamp timestamp(DateTime(startDateTime.GetDate(), startDateTime.Seconds() + secs), nanosecs);
432             return timestamp;
433         }
434         private nothrow void Reset()
435         {
436             startDateTime = GetCurrentDateTime();
437             startTimePoint = Now();
438         }
439         private static UniquePtr<TimestampProvider> instance;
440         private DateTime startDateTime;
441         private TimePoint startTimePoint;
442     }
