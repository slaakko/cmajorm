  6 using System;
  7 
  8 namespace System.IO
  9 {
 10     public class BufferedByteStream : ByteStream
 11     {
 12         public nothrow BufferedByteStream(const SharedPtr<ByteStream>& baseStream_) : this(baseStream_, 4096)
 13         {
 14         }
 15         public nothrow BufferedByteStream(const SharedPtr<ByteStream>& baseStream_, long bufferSize_) : 
 16             baseStream(baseStream_), buffer(bufferSize_), pos(buffer.Size()), bytesAvailable(0), end(0)
 17         {
 18         }
 19         suppress BufferedByteStream(const BufferedByteStream&);
 20         suppress void operator=(const BufferedByteStream&);
 21         public override ~BufferedByteStream()
 22         {
 23             try
 24             {
 25                 if (!baseStream.IsNull())
 26                 {
 27                     Flush();
 28                 }
 29             }
 30             catch (const Exception&)
 31             {
 32             }
 33         }
 34         public override int ReadByte()
 35         {
 36             Flush();
 37             if (bytesAvailable == 0)
 38             {
 39                 FillBuf();
 40                 if (bytesAvailable == 0)
 41                 {
 42                     return -1;
 43                 }
 44             }
 45             byte value = buffer[pos++];
 46             --bytesAvailable;
 47             return value;
 48         }
 49         public override long Read(byte* buf, long count)
 50         {
 51             #assert(count >= 0);
 52             Flush();
 53             if (bytesAvailable == 0)
 54             {
 55                 FillBuf();
 56             }
 57             long bytesRead = 0;
 58             long n = Min(bytesAvailable, count);
 59             for (long i = 0; i < n; ++i;)
 60             {
 61                 buf[i] = buffer[pos++];
 62                 ++bytesRead;
 63                 --bytesAvailable;
 64             }
 65             return bytesRead;
 66         }
 67         public override void Write(byte x)
 68         {
 69             if (end >= buffer.Size())
 70             {
 71                 Flush();
 72             }
 73             buffer[end++] = x;
 74         }
 75         public override void Write(byte* buf, long count)
 76         {
 77             #assert(count >= 0);
 78             for (long i = 0; i < count; ++i;)
 79             {
 80                 Write(buf[i]);
 81             }
 82         }
 83         public override void Flush()
 84         {
 85             if (end != 0)
 86             {
 87                 baseStream->Write(cast<byte*>(buffer.Mem()), end);
 88                 end = 0;
 89             }
 90         }
 91         public override void Seek(long pos, Origin origin)
 92         {
 93             Flush();
 94             bytesAvailable = 0;
 95             baseStream->Seek(pos, origin);
 96         }
 97         public override long Tell()
 98         {
 99             Flush();
100             return baseStream->Tell() - bytesAvailable;
101         }
102         public nothrow const SharedPtr<ByteStream>& ContainedStream()
103         {
104             return baseStream;
105         }
106         private void FillBuf()
107         {
108             bytesAvailable = baseStream->Read(cast<byte*>(buffer.Mem()), buffer.Size());
109             pos = 0;
110         }
111         private SharedPtr<ByteStream> baseStream;
112         private IOBuffer buffer;
113         private long pos;
114         private long bytesAvailable;
115         private long end;
116     }
117 }
