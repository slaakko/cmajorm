 6 using System;
 7 
 8 namespace System.Threading
 9 {
10     public class Process
11     {
12         public enum StdHandle
13         {
14             std_out = 1, std_err = 2
15         }
16         public Process(const string& command)
17         {
18             int errorStringHandle = -1;
19             nativeProcess = RtCreateProcess(command.Chars(), errorStringHandle);
20             if (nativeProcess == null)
21             {
22                 string errorString = RtGetString(errorStringHandle);
23                 RtDisposeString(errorStringHandle);
24                 throw ThreadingException(errorString);
25             }
26         }
27         public ~Process()
28         {
29             if (nativeProcess != null)
30             {
31                 RtDeleteProcess(nativeProcess);
32             }
33         }
34         public bool Running()
35         {
36             return RtProcessRunning(nativeProcess);
37         }
38         public void WaitForExit()
39         {
40             RtProcessWaitForExit(nativeProcess);
41         }
42         public int ExitCode()
43         {
44             return RtProcessExitCode(nativeProcess);
45         }
46         public bool Eof(StdHandle handle)
47         {
48             return RtProcessEof(nativeProcess, cast<int>(handle));
49         }
50         public string ReadLine(StdHandle handle)
51         {
52             int stringHandle = RtProcessReadLine(nativeProcess, cast<int>(handle));
53             string line = RtGetString(stringHandle);
54             RtDisposeString(stringHandle);
55             return line;
56         }
57         public string ReadToEnd(StdHandle handle)
58         {
59             int stringHandle = RtProcessReadToEnd(nativeProcess, cast<int>(handle));
60             string content = RtGetString(stringHandle);
61             RtDisposeString(stringHandle);
62             return content;
63         }
64         public void WriteLine(const string& line)
65         {
66             RtProcessWriteLine(nativeProcess, line.Chars());
67         }
68         private void* nativeProcess;
69     }
70 }
