  6 using System;
  7 using System.IO;
  8 
  9 namespace System.Text
 10 {
 11     public class CodeFormatter
 12     {
 13         public nothrow CodeFormatter(StreamWriter& writer_) : 
 14             writer(writer_), indent(0), indentSize(4), atBeginningOfLine(true), line(1), start(false), startText(), preserveSpace(false)
 15         {
 16         }
 17         public void Write(const string& text)
 18         {
 19             if (atBeginningOfLine)
 20             {
 21                 if (indent != 0)
 22                 {
 23                     writer.Write(string(' ', indentSize * indent));
 24                     atBeginningOfLine = false;
 25                 }
 26             }
 27             writer.Write(text);
 28         }
 29         public void WriteLine(const string& text)
 30         {
 31             Write(text);
 32             WriteLine();
 33         }
 34         public void WriteLine()
 35         {
 36             writer.WriteLine();
 37             atBeginningOfLine = true;
 38             ++line;
 39         }
 40         public inline nothrow int Indent() const
 41         {
 42             return indent;
 43         }
 44         public inline nothrow void IncIndent()
 45         {
 46             ++indent;
 47         }
 48         public inline nothrow void DecIndent()
 49         {
 50             --indent;
 51         }
 52         public inline nothrow int IndentSize() const
 53         {
 54             return indentSize;
 55         }
 56         public inline nothrow void SetIndentSize(int indentSize_)
 57         {
 58             indentSize = indentSize_;
 59         }
 60         public inline nothrow int CurrentIndent() const
 61         {
 62             return indent * indentSize;
 63         }
 64         public inline nothrow bool AtBeginningOfLine() const
 65         {
 66             return atBeginningOfLine;
 67         }
 68         public inline nothrow int Line() const
 69         {
 70             return line;
 71         }
 72         public inline nothrow void SetLine(int line_)
 73         {
 74             line = line_;
 75         }
 76         public inline nothrow bool Start() const
 77         {
 78             return start;
 79         }
 80         public inline nothrow void SetStart(bool start_)
 81         {
 82             start = start_;
 83         }
 84         public inline nothrow const string& StartText() const
 85         {
 86             return startText;
 87         }
 88         public nothrow void SetStartText(const string& startText_)
 89         {
 90             startText = startText_;
 91         }
 92         public inline nothrow StreamWriter& Writer()
 93         {
 94             return writer;
 95         }
 96         public inline nothrow bool PreserveSpace() const
 97         {
 98             return preserveSpace;
 99         }
100         public inline nothrow void SetPreserveSpace(bool preserveSpace_)
101         {
102             preserveSpace = preserveSpace_;
103         }
104         private StreamWriter& writer;
105         private int indent;
106         private int indentSize;
107         private bool atBeginningOfLine;
108         private int line;
109         private bool start;
110         private string startText;
111         private bool preserveSpace;
112     }
113 }
