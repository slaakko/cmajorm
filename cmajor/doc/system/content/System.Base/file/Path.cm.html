 18     public string GetCurrentWorkingDirectory()
 19     {
 20         int handle = RtGetCurrentWorkingDirectoryHandle();
 21         if (handle < 0)
 22         {
 23             string errorMessage = RtGetError(handle);
 24             RtDisposeError(handle);
 25             throw FileSystemException(errorMessage);
 26         }
 27         string currentWorkingDirectory = Path.MakeCanonical(RtGetCurrentWorkingDirectory(handle));
 28         RtFreeCurrentWorkingDirectoryHandle(handle);
 29         return currentWorkingDirectory;
 30     }

 32     public string GetFullPath(const string& path)
 33     {
 34         string p = Path.MakeCanonical(path);
 35         if (Path.IsRelative(p))
 36         {
 37             p = GetCurrentWorkingDirectory();
 38             p.Append('/');
 39             p.Append(Path.MakeCanonical(path));
 40         }
 41         List<string> components = p.Split('/');
 42         long w = 0;
 43         long n = components.Count();
 44         for (long i = 0; i < n; ++i;)
 45         {
 46             string c = components[i];
 47             if (i == 0 || (!c.IsEmpty() && c != "."))
 48             {
 49                 if (c == "..")
 50                 {
 51                     --w;
 52                     if (w < 0)
 53                     {
 54                         throw InvalidPathException("path '" + path + "' is invalid");
 55                     }
 56                 }
 57                 else
 58                 {
 59                     if (w != i)
 60                     {
 61                         components[w] = components[i];
 62                     }
 63                     ++w;
 64                 }
 65             }
 66         }
 67         if (w == 0)
 68         {
 69             return "/";
 70         }
 71         else if (w == 1)
 72         {
 73             string p = components[0];
 74             if (p.Length() == 2 && IsAlpha(p[0]) && p[1] == ':')
 75             {
 76                 return p + "/";
 77             }
 78         }
 79         string result;
 80         for (long i = 0; i < w; ++i;)
 81         {
 82             if (i != 0)
 83             {
 84                 result.Append('/');
 85             }
 86             result.Append(components[i]);
 87         }
 88         if (result.IsEmpty())
 89         {
 90             return "/";
 91         }
 92         else
 93         {
 94             return result;
 95         }
 96     }

 98     public static class Path
 99     {
100         public static nothrow string MakeCanonical(const string& path)
101         {
102             bool startsWithDriveLetter = false;
103             #if (WINDOWS)
104                 if (path.Length() >= 2 && IsAlpha(path[0]) && path[1] == ':')
105                 {
106                     startsWithDriveLetter = true;
107                 }
108             #endif
109             string result;
110             char prev = ' ';
111             bool first = true;
112             for (char c : path)
113             {
114                 if (first)
115                 {
116                     first = false;
117                     if (startsWithDriveLetter)
118                     {
119                         c = AsciiToUpper(c);
120                     }
121                 }
122                 if (c == '\\')
123                 {
124                     c = '/';
125                 }
126                 if (c == '/')
127                 {
128                     if (prev != '/')
129                     {
130                         result.Append(c);
131                     }
132                 }
133                 else
134                 {
135                     result.Append(c);
136                 }
137                 prev = c;
138             }
139             if (result.Length() == 3 && IsAlpha(result[0]) && result[1] == ':' && result[2] == '/')
140             {
141                 return result;
142             }
143             if (result == "/")
144             {
145                 return result;
146             }
147             if (!result.IsEmpty())
148             {
149                 if (result[result.Length() - 1] == '/')
150                 {
151                     return result.Substring(0, result.Length() - 1);
152                 }
153             }
154             return result;
155         }
156         public static nothrow string ChangeExtension(const string& path, const string& extension)
157         {
158             long lastDotPos = path.RFind('.');
159             long lastSlashPos = path.RFind('/');
160             if (lastSlashPos != -1 && lastDotPos < lastSlashPos)
161             {
162                 lastDotPos = -1;
163             }
164             if (extension.IsEmpty())
165             {
166                 if (lastDotPos != -1)
167                 {
168                     return path.Substring(0, lastDotPos);
169                 }
170                 else
171                 {
172                     return path;
173                 }
174             }
175             else
176             {
177                 if (lastDotPos == -1)
178                 {
179                     if (extension[0] == '.')
180                     {
181                         return path + extension;
182                     }
183                     else
184                     {
185                         return path + "." + extension;
186                     }
187                 }
188                 else
189                 {
190                     if (extension[0] == '.')
191                     {
192                         return path.Substring(0, lastDotPos) + extension;
193                     }
194                     else
195                     {
196                         return path.Substring(0,  lastDotPos + 1) + extension;
197                     }
198                 }
199             }
200         }
201         public static nothrow bool HasExtension(const string& path)
202         {
203             string p = MakeCanonical(path);
204             long lastDotPos = p.RFind('.');
205             if (lastDotPos != -1)
206             {
207                 long lastColon = p.Find(':', lastDotPos + 1);
208                 long lastDirSep = p.Find('/', lastDotPos + 1);
209                 if (lastColon > lastDotPos || lastDirSep > lastDotPos)
210                 {
211                     return false;
212                 }
213                 else if (lastDotPos < p.Length() - 1)
214                 {
215                     return true;
216                 }
217                 else
218                 {
219                     return false;
220                 }
221             }
222             else
223             {
224                 return false;
225             }
226         }
227         public static nothrow string GetExtension(const string& path)
228         {
229             string p = MakeCanonical(path);
230             long lastDotPos = p.RFind('.');
231             if (lastDotPos != -1)
232             {
233                 if (p.Find('/', lastDotPos + 1) != -1)
234                 {
235                     return string();
236                 }
237                 else
238                 {
239                     return p.Substring(lastDotPos);
240                 }
241             }
242             else
243             {
244                 return string();
245             }
246         }
247         public static nothrow string GetFileName(const string& path)
248         {
249             if (path.IsEmpty())
250             {
251                 return string();
252             }
253             else
254             {
255                 string p = MakeCanonical(path);
256                 char lastChar = p[p.Length() - 1];
257                 if (lastChar == '/' || lastChar == ':')
258                 {
259                     return string();
260                 }
261                 else
262                 {
263                     long lastDirSepPos = p.RFind('/');
264                     if (lastDirSepPos != -1)
265                     {
266                         return p.Substring(lastDirSepPos + 1);
267                     }
268                     else
269                     {
270                         return p;
271                     }
272                 }
273             }

312         public static nothrow string Combine(const string& path1, const string& path2)
313         {
314             string p1 = MakeCanonical(path1);
315             string p2 = MakeCanonical(path2);
316             if (p1.IsEmpty())
317             {
318                 return p2;
319             }
320             else if (p2.IsEmpty())
321             {
322                 return p1;
323             }
324             else
325             {
326                 if (IsAbsolute(p2))
327                 {
328                     return p2;
329                 }
330                 else
331                 {
332                     string result = p1;
333                     if (p1[p1.Length() - 1] != '/')
334                     {
335                         result.Append('/');
336                     }
337                     result.Append(p2);
338                     return result;
339                 }
340             }

342         public static nothrow bool IsAbsolute(const string& path)
343         {
344             if (path.IsEmpty())
345             {
346                 return false;
347             }
348             else
349             {
350                 string p = MakeCanonical(path);
351                 if (p[0] == '/')
352                 {
353                     return true;
354                 }
355                 else if (p.Length() > 2 && IsAlpha(p[0]) && p[1] == ':' && p[2] == '/')
356                 {
357                     return true;
358                 }
359                 else
360                 {
361                     return false;
362                 }
363             }
