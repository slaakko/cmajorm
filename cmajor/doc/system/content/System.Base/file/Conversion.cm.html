383         {

387         int start = 0;
388         if (s[0] == '+')

392         else if (s[0] == '-')

397         int state = 0;

402         for (long i = start; i < n; ++i;)
403         {
404             char c = s[i];
405             switch (state)
406             {
407                 case 0:
408                 {
409                     if (c >= '0' && c <= '9')
410                     {
411                         x = 10 * x + (cast<int>(c) - cast<int>('0'));
412                     }
413                     else if (c == '.')
414                     {
415                         state = 1;
416                     }

419                         state = 2;
420                     }
421                     else
422                     {

427                 case 1:

430                     {
431                         x = x + (cast<int>(c) - cast<int>('0')) / d;
432                         d = 10 * d;
433                     }
434                     else if (c == 'e' || c == 'E')
435                     {
436                         state = 2;
437                     }
438                     else

440                         return false;

443                 }
444                 case 2:

449                     }

453                         state = 3;
454                     }
455                     else if (c >= '0' && c <= '9')
456                     {
457                         exponent = cast<int>(c) - cast<int>('0');
458                         state = 3;

462                         return false;
463                     }

465                 }

471                     }
472                     else
473                     {
474                         return false;
475                     }
476                     break;
477                 }
478             }
479         }
480         if (negative)
