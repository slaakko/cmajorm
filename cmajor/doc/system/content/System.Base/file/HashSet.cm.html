 6 using System;
 7 using System.Concepts;
 8 
 9 namespace System.Collections
10 {
11     public class HashSet<T, H = Hasher<T>, C = EqualTo<T>>
12         where T is Semiregular and HashFunction<H, T> and C is Relation and C.Domain is T
13     {
14         public typedef T ValueType;
15         public typedef T KeyType;
16         public typedef H HashFun;
17         public typedef C Compare;
18         public typedef TableType.ConstIterator ConstIterator;
19         public typedef TableType.Iterator Iterator;
20         private typedef HashSet<ValueType, HashFun, Compare> Self;
21         private typedef Hashtable<KeyType, ValueType, Identity<ValueType>, HashFun, Compare> TableType;
22 
23         public inline nothrow Iterator Begin()
24         {
25             return table.Begin();
26         }
27         public inline nothrow ConstIterator Begin() const
28         {
29             return table.CBegin();
30         }
31         public inline nothrow ConstIterator CBegin() const
32         {
33             return table.CBegin();
34         }
35         public inline nothrow Iterator End()
36         {
37             return table.End();
38         }
39         public inline nothrow ConstIterator End() const
40         {
41             return table.CEnd();
42         }
43         public inline nothrow ConstIterator CEnd() const
44         {
45             return table.CEnd();
46         }
47         public inline nothrow long Count() const
48         {
49             return table.Count();
50         }
51         public inline nothrow bool IsEmpty() const
52         {
53             return table.IsEmpty();
54         }
55         public inline nothrow void Clear()
56         {
57             table.Clear();
58         }
59         public inline nothrow Iterator Find(const KeyType& key)
60         {
61             return table.Find(key);
62         }
63         public inline nothrow ConstIterator Find(const KeyType& key) const
64         {
65             return table.CFind(key);
66         }
67         public inline nothrow ConstIterator CFind(const KeyType& key) const
68         {
69             return table.CFind(key);
70         }
71         public inline Pair<Iterator, bool> Insert(const ValueType& value)
72         {
73             return table.Insert(value);
74         }
75         public inline nothrow void Remove(const KeyType& key)
76         {
77             table.Remove(key);
78         }
79         public inline nothrow void Remove(Iterator pos)
80         {
81             table.Remove(pos);
82         }
83         private TableType table;
84     }
85 
86     public nothrow bool operator==<T, H, C>(const HashSet<T, H, C>& left, const HashSet<T, H, C>& right) where T is Semiregular and HashFunction<H, T> and C is Relation and C.Domain is T
87     {
88         if (left.Count() != right.Count()) return false;
89         for (const T& value : left)
90         {
91             if (right.CFind(value) == right.CEnd()) return false;
92         }
93         return true;
94     }
