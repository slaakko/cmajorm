  6 using System;

  9 namespace System.Collections

 11     public class HashMap<K, T, H = Hasher<K>, C = EqualTo<K>>
 12         where K is Semiregular and T is Semiregular and HashFunction<H, K> and C is Relation and C.Domain is K
 13     {
 14         public typedef K KeyType;
 15         public typedef T MappedType;
 16         public typedef Pair<KeyType, MappedType> ValueType;
 17         public typedef H HashFun;
 18         public typedef C Compare;
 19         public typedef TableType.ConstIterator ConstIterator;
 20         public typedef TableType.Iterator Iterator;
 21         private typedef HashMap<KeyType, MappedType, HashFun, Compare> Self;
 22         private typedef Hashtable<KeyType, ValueType, SelectFirst<KeyType, MappedType>, HashFun, Compare> TableType;
 23 
 24         public inline nothrow Iterator Begin()
 25         {
 26             return table.Begin();
 27         }
 28         public inline nothrow ConstIterator Begin() const
 29         {
 30             return table.CBegin();
 31         }
 32         public inline nothrow ConstIterator CBegin() const
 33         {
 34             return table.CBegin();
 35         }
 36         public inline nothrow Iterator End()
 37         {
 38             return table.End();
 39         }
 40         public inline nothrow ConstIterator End() const
 41         {
 42             return table.CEnd();
 43         }
 44         public inline nothrow ConstIterator CEnd() const
 45         {
 46             return table.CEnd();
 47         }
 48         public inline nothrow long Count() const
 49         {

 53         {

 55         }

 57         {
 58             table.Clear();
 59         }
 60         public inline nothrow Iterator Find(const KeyType& key)
 61         {

 63         }
 64         public inline nothrow ConstIterator Find(const KeyType& key) const
 65         {

 69         {

 71         }
 72         public inline MappedType& operator[](const KeyType& key)
 73         {
 74             Pair<Iterator, bool> ib = Insert(ValueType(key, MappedType()));
 75             Iterator i = ib.first;
 76             return i->second;
 77         }
 78         public inline Pair<Iterator, bool> Insert(const ValueType& value)
 79         {
 80             return table.Insert(value);
 81         }
 82         public inline nothrow void Remove(const KeyType& key)
 83         {
 84             table.Remove(key);
 85         }
 86         public inline nothrow void Remove(Iterator pos)
 87         {
 88             table.Remove(pos);
 89         }
 90         private TableType table;
 91     }
 92 
 93     public nothrow bool operator==<K, T, H, C>(const HashMap<K, T, H, C>& left, const HashMap<K, T, H, C>& right)
 94         where K is Semiregular and T is Semiregular and HashFunction<H, K> and C is Relation and C.Domain is K
 95     {
 96         if (left.Count() != right.Count()) return false;
 97         for (const Pair<K, T>& p : left)
 98         {
 99             HashMap<K, T, H, C>.ConstIterator i = right.CFind(p.first);
100             if (i == right.CEnd()) return false;
101             if (i->second != p.second) return false;
102         }
103         return true;
104     }
