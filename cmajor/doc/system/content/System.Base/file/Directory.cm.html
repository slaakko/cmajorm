 6 using System;
 7 using System.Collections;
 8 
 9 namespace System.IO
10 {
11     
12     public static class Directory
13     {
14         public static nothrow bool Exists(const string& directoryPath)
15         {
16             return RtDirectoryExists(directoryPath.Chars());
17         }
18         public static nothrow void CreateDirectories(const string& directoryPath)
19         {
20             RtCreateDirectories(directoryPath.Chars());
21         }
22         public static List<string> GetFiles(const string& directoryPath)
23         {
24             List<string> files;
25             int directoryIterationHandle = RtBeginIterateDirectory(directoryPath.Chars());
26             const char* filePath = RtGetNextFilePath(directoryIterationHandle);
27             while (filePath != null)
28             {
29                 files.Add(filePath);
30                 filePath = RtGetNextFilePath(directoryIterationHandle);
31             }
32             RtEndIterateDirectory(directoryIterationHandle);
33             return files;
34         }
35     }
36 }
