 6 namespace System
 7 {
 8     public class Exception
 9     {
10         public nothrow Exception(const string& message_) : message(message_), stackTrace(RtGetStackTrace())
11         {
12             RtDisposeStackTrace();
13         }
14         public virtual default ~Exception();
15         public inline nothrow const string& Message() const
16         {
17             return message;
18         }
19         public inline nothrow const string& StackTrace() const
20         {
21             return stackTrace;
22         }
23         public virtual nothrow string ToString() const
24         {
25             string s = typename(*this);
26             s.Append(": ").Append(message).Append("\nCALL STACK:\n").Append(stackTrace);
27             return s;
28         }
29         private string message;
30         private string stackTrace;
31     }
32     
33     public class ExceptionPtr
34     {
35         public nothrow ExceptionPtr() : exception(null), exceptionClassId(0u)
36         {
37         }
38         public nothrow ExceptionPtr(void* exception_, ulong exceptionClassId_) : exception(exception_), exceptionClassId(exceptionClassId_)
39         {
40         }
41         public inline nothrow void* Exception() const
42         {
43             return exception;
44         }
45         public inline nothrow ulong ExceptionClassId() const
46         {
47             return exceptionClassId;
48         }
49         private void* exception;
50         private ulong exceptionClassId;
51     }
52     
53     public nothrow ExceptionPtr CaptureCurrentException()
54     {
55         void* exception;
56         ulong exceptionClassId;
57         RtCaptureException(exception, exceptionClassId);
58         return ExceptionPtr(exception, exceptionClassId);
59     }
60     
61     public void ThrowCapturedException(const ExceptionPtr& capturedException)
62     {
63         RtThrowCapturedException(capturedException.Exception(), capturedException.ExceptionClassId());
64     }
65 }
