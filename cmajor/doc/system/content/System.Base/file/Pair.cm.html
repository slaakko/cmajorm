 6 using System.Concepts;
 7 
 8 namespace System
 9 {
10     public class Pair<T, U> where T is Semiregular and U is Semiregular
11     {
12         public nothrow Pair() : first(), second() {}
13         public Pair(const T& first_, const U& second_) : first(first_), second(second_) {}
14         public T first;
15         public U second;
16     }
17 
18     public inline nothrow bool operator<<T, U>(const Pair<T, U>& left, const Pair<T, U>& right) where T is TotallyOrdered and U is TotallyOrdered
19     {
20         if (left.first < right.first) return true;
21         else if (left.first > right.first) return false;
22         else return left.second < right.second;
23     }
24 
25     public inline nothrow bool operator==<T, U>(const Pair<T, U>& left, const Pair<T, U>& right) where T is Regular and U is Regular
26     {
27         return left.first == right.first && left.second == right.second;
28     }
29 
30     public Pair<T, U> MakePair<T, U>(const T& first, const U& second) where T is Semiregular and U is Semiregular
31     {
32         return Pair<T, U>(first, second);
33     }
34 
35     public class SelectFirst<T, U> : UnaryFun<Pair<T, U>, T> where T is Semiregular and U is Semiregular
36     {
37         public inline nothrow const T& operator()(const Pair<T, U>& p) const
38         {
39             return p.first;
40         }
41     }
42 
43     public class SelectSecond<T, U> : UnaryFun<Pair<T, U>, U> where T is Semiregular and U is Semiregular
44     {
45         public inline nothrow const U& operator()(const Pair<T, U>& p) const
46         {
47             return p.second;
48         }
49     }
