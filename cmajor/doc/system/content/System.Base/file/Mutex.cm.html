 6 using System;
 7 using System.Concepts;
 8 
 9 namespace System.Threading
10 {
11     public class Mutex
12     {
13         public nothrow Mutex() : mutexId(RtAllocateMutex())
14         {
15         }
16         public ~Mutex()
17         {
18             RtFreeMutex(mutexId);
19         }
20         suppress Mutex(const Mutex&);
21         suppress void operator=(Mutex&);
22         suppress Mutex(Mutex&&);
23         suppress void operator=(Mutex&&);
24         public nothrow void Lock()
25         {
26             RtLockMutex(mutexId);
27         }
28         public nothrow void Unlock()
29         {
30             RtUnlockMutex(mutexId);
31         }
32         private int mutexId;
33     }
34     
35     public class RecursiveMutex
36     {
37         public nothrow RecursiveMutex() : mutexId(RtAllocateRecursiveMutex())
38         {
39         }
40         public ~RecursiveMutex()
41         {
42             RtFreeRecursiveMutex(mutexId);
43         }
44         suppress RecursiveMutex(const RecursiveMutex&);
45         suppress void operator=(RecursiveMutex&);
46         suppress RecursiveMutex(RecursiveMutex&&);
47         suppress void operator=(RecursiveMutex&&);
48         public nothrow void Lock()
49         {
50             RtLockRecursiveMutex(mutexId);
51         }
52         public nothrow void Unlock()
53         {
54             RtUnlockRecursiveMutex(mutexId);
55         }
56         private int mutexId;
57     }
58     
59     public concept Lockable<T>
60     {
61         void T.Lock();
62         void T.Unlock();
63     }
64     
65     public class LockGuard<Mtx> where Mtx is Lockable
66     {
67         private typedef LockGuard<Mtx> Self;
68         public nothrow LockGuard(Mtx& mtx_) : mtx(mtx_)
69         {
70             mtx.Lock();
71         }
72         public ~LockGuard()
73         {
74             mtx.Unlock();
75         }
76         suppress LockGuard(const Self&);
77         suppress void operator=(const Self&);
78         suppress LockGuard(Self&&);
79         suppress void operator=(Self&&);
80         private Mtx& mtx;
81     }
82 }
