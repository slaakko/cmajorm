  6 using System;
  7 
  8 namespace System.Collections
  9 {
 10     public class Bitset
 11     {
 12         public nothrow Bitset() : numBits(0), bits()
 13         {
 14             try
 15             {
 16                 bits.Add(0u);
 17             }
 18             catch (const Exception& ex)
 19             {
 20                 RtExit(1);
 21             }
 22         }
 23         public Bitset(long numBits_) : numBits(0), bits()
 24         {
 25             Resize(numBits_);
 26         }
 27         public Bitset(const string& bits_) : numBits(0), bits()
 28         {
 29             Resize(bits_.Length());
 30             for (long i = 0; i < numBits; ++i;)
 31             {
 32                 if (bits_[i] == '1')
 33                 {
 34                     Set(i);
 35                 }
 36             }
 37         }
 38         public default Bitset(const Bitset&);
 39         public default void operator=(const Bitset&);
 40         public default nothrow Bitset(Bitset&&);
 41         public default nothrow void operator=(Bitset&&);
 42         public default ~Bitset();
 43         public nothrow void Clear()
 44         {
 45             numBits = 0;
 46             bits.Clear();
 47         }
 48         public inline nothrow long Count() const
 49         {
 50             return numBits;
 51         }
 52         public void Resize(long numBits_)
 53         {
 54             numBits = numBits_;
 55             #assert(numBits >= 0);
 56             if (numBits > 0)
 57             {
 58                 bits.Resize(1 + NumFullBlocks());
 59             }
 60             else
 61             {
 62                 bits.Resize(1);
 63             }
 64         }
 65         public nothrow void Set()
 66         {
 67             long n = bits.Count();
 68             for (long i = 0; i < n; ++i;)
 69             {
 70                 bits[i] = allOne;
 71             }
 72         }
 73         public nothrow void Reset()
 74         {
 75             long n = bits.Count();
 76             for (long i = 0; i < n; ++i;)
 77             {
 78                 bits[i] = cast<ulong>(0u);
 79             }
 80         }
 81         public inline nothrow void Set(long pos)
 82         {
 83             #assert(pos >= 0 && pos < numBits);
 84             ulong& b = bits[pos / blockSize];
 85             b = b | (1u << (cast<ulong>(pos) & blockMask));
 86         }
 87         public inline nothrow void Reset(long pos)
 88         {
 89             #assert(pos >= 0 && pos < numBits);
 90             ulong& b = bits[pos / blockSize];
 91             b = b & ~(1u << (cast<ulong>(pos) & blockMask));
 92         }
 93         public nothrow void Set(long pos, bool bit)
 94         {
 95             if (bit)
 96             {
 97                 Set(pos);
 98             }
 99             else
100             {
101                 Reset(pos);
102             }
103         }
104         public nothrow void Flip()
105         {
106             long n = bits.Count();
107             for (long i = 0; i < n; ++i;)
108             {
109                 bits[i] = ~(bits[i]);
110             }
111         }
112         public inline nothrow void Flip(long pos)
113         {
114             #assert(pos >= 0 && pos < numBits);
115             ulong& b = bits[pos / blockSize];
116             b = b ^ (1u << (cast<ulong>(pos) & blockMask));
117         }
118         public inline nothrow bool operator[](long index) const
119         {
120             return Test(index);
121         }
122         public inline nothrow bool Test(long pos) const
123         {
124             #assert(pos >= 0 && pos < numBits);
125             ulong b = bits[pos / blockSize];
126             return (b & (1u << (cast<ulong>(pos) & blockMask))) != 0u;
127         }
128         public nothrow bool All() const
129         {
130             long n = NumFullBlocks();
131             for (long i = 0; i < n; ++i;)
132             {
133                 if (bits[i] != allOne)
134                 {
135                     return false;
136                 }
137             }
138             for (long i = LastBlockStartIndex(); i < numBits; ++i;)
139             {
140                 if (!Test(i))
141                 {
142                     return false;
143                 }
144             }
145             return true;
146         }
147         public nothrow bool Any() const
148         {
149             long n = NumFullBlocks();
150             for (long i = 0; i < n; ++i;)
151             {
152                 if (bits[i] != 0u)
153                 {
154                     return true;
155                 }
156             }
157             for (long i = LastBlockStartIndex(); i < numBits; ++i;)
158             {
159                 if (Test(i))
160                 {
161                     return true;
162                 }
163             }
164             return false;
165         }
166         public nothrow bool None() const
167         {
168             long n = NumFullBlocks();
169             for (long i = 0; i < n; ++i;)
170             {
171                 if (bits[i] != 0u)
172                 {
173                     return false;
174                 }
175             }
176             for (long i = LastBlockStartIndex(); i < numBits; ++i;)
177             {
178                 if (Test(i))
179                 {
180                     return false;
181                 }
182             }
183             return true;
184         }
185         public nothrow string ToString() const
186         {
187             string s;
188             s.Reserve(numBits);
189             for (long i = 0; i < numBits; ++i;)
190             {
191                 if (Test(i))
192                 {
193                     s.Append('1');
194                 }
195                 else
196                 {
197                     s.Append('0');
198                 }
199             }
200             return s;
201         }
202         public inline nothrow long NumBits() const
203         {
204             return numBits;
205         }
206         public inline nothrow List<ulong>& Bits() const
207         {
208             return bits;
209         }
210         public inline nothrow long NumFullBlocks() const
211         {
212             if (numBits == 0)
213             {
214                 return 0;
215             }
216             return (numBits - 1) / blockSize;
217         }
218         public inline nothrow long LastBlockStartIndex() const
219         {
220             return NumFullBlocks() * blockSize;
221         }
222         private const long blockSize = 64;
223         private const ulong blockMask = 63u;
224         private const ulong allOne = cast<ulong>(-1);
225         private long numBits;
226         private List<ulong> bits;
227     }
228 
229     public bool operator==(const Bitset& left, const Bitset& right)
230     {
231         long numBits = left.NumBits();
232         if (numBits != right.NumBits())
233         {
234             return false;
235         }
236         long n = left.NumFullBlocks();
237         for (long i = 0; i < n; ++i;)
238         {
239             if (left.Bits()[i] != right.Bits()[i])
240             {
241                 return false;
242             }
243         }
244         for (long i = left.LastBlockStartIndex(); i < numBits; ++i;)
245         {
246             if (left.Test(i) != right.Test(i))
247             {
248                 return false;
249             }
250         }
251         return true;
252     }
