  8 using System;
  9 
 10 namespace System.Collections
 11 {
 12     public class Bitset
 13     {
 14         public nothrow Bitset() : numBits(0), bits()
 15         {
 16             try
 17             {
 18                 bits.Add(0u);
 19             }
 20             catch (const Exception& ex)
 21             {
 22                 RtExit(1);
 23             }
 24         }
 25         public Bitset(long numBits_) : numBits(0), bits()
 26         {
 27             Resize(numBits_);
 28         }
 29         public Bitset(const string& bits_) : numBits(0), bits()
 30         {
 31             Resize(bits_.Length());
 32             for (long i = 0; i < numBits; ++i;)
 33             {
 34                 if (bits_[i] == '1')
 35                 {
 36                     Set(i);
 37                 }
 38             }
 39         }
 40         public default Bitset(const Bitset&);
 41         public default void operator=(const Bitset&);
 42         public default nothrow Bitset(Bitset&&);
 43         public default nothrow void operator=(Bitset&&);
 44         public default ~Bitset();
 45         public nothrow void Clear()
 46         {
 47             numBits = 0;
 48             bits.Clear();
 49         }
 50         public inline nothrow long Count() const
 51         {
 52             return numBits;
 53         }
 54         public void Resize(long numBits_)
 55         {
 56             numBits = numBits_;
 57             #assert(numBits >= 0);
 58             if (numBits > 0)
 59             {
 60                 bits.Resize(1 + NumFullBlocks());
 61             }
 62             else
 63             {
 64                 bits.Resize(1);
 65             }
 66         }
 67         public nothrow void Set()
 68         {
 69             long n = bits.Count();
 70             for (long i = 0; i < n; ++i;)
 71             {
 72                 bits[i] = allOne;
 73             }
 74         }
 75         public nothrow void Reset()
 76         {
 77             long n = bits.Count();
 78             for (long i = 0; i < n; ++i;)
 79             {
 80                 bits[i] = cast<ulong>(0u);
 81             }
 82         }
 83         public inline nothrow void Set(long pos)
 84         {
 85             #assert(pos >= 0 && pos < numBits);
 86             ulong& b = bits[pos / blockSize];
 87             b = b | (1u << (cast<ulong>(pos) & blockMask));
 88         }
 89         public inline nothrow void Reset(long pos)
 90         {
 91             #assert(pos >= 0 && pos < numBits);
 92             ulong& b = bits[pos / blockSize];
 93             b = b & ~(1u << (cast<ulong>(pos) & blockMask));
 94         }
 95         public nothrow void Set(long pos, bool bit)
 96         {
 97             if (bit)
 98             {
 99                 Set(pos);
100             }
101             else
102             {
103                 Reset(pos);
104             }
105         }
106         public nothrow void Flip()
107         {
108             long n = bits.Count();
109             for (long i = 0; i < n; ++i;)
110             {
111                 bits[i] = ~(bits[i]);
112             }
113         }
114         public inline nothrow void Flip(long pos)
115         {
116             #assert(pos >= 0 && pos < numBits);
117             ulong& b = bits[pos / blockSize];
118             b = b ^ (1u << (cast<ulong>(pos) & blockMask));
119         }
120         public inline nothrow bool operator[](long index) const
121         {
122             return Test(index);
123         }
124         public inline nothrow bool Test(long pos) const
125         {
126             #assert(pos >= 0 && pos < numBits);
127             ulong b = bits[pos / blockSize];
128             return (b & (1u << (cast<ulong>(pos) & blockMask))) != 0u;
129         }
130         public nothrow bool All() const
131         {
132             long n = NumFullBlocks();
133             for (long i = 0; i < n; ++i;)
134             {
135                 if (bits[i] != allOne)
136                 {
137                     return false;
138                 }
139             }
140             for (long i = LastBlockStartIndex(); i < numBits; ++i;)
141             {
142                 if (!Test(i))
143                 {
144                     return false;
145                 }
146             }
147             return true;
148         }
149         public nothrow bool Any() const
150         {
151             long n = NumFullBlocks();
152             for (long i = 0; i < n; ++i;)
153             {
154                 if (bits[i] != 0u)
155                 {
156                     return true;
157                 }
158             }
159             for (long i = LastBlockStartIndex(); i < numBits; ++i;)
160             {
161                 if (Test(i))
162                 {
163                     return true;
164                 }
165             }
166             return false;
167         }
168         public nothrow bool None() const
169         {
170             long n = NumFullBlocks();
171             for (long i = 0; i < n; ++i;)
172             {
173                 if (bits[i] != 0u)
174                 {
175                     return false;
176                 }
177             }
178             for (long i = LastBlockStartIndex(); i < numBits; ++i;)
179             {
180                 if (Test(i))
181                 {
182                     return false;
183                 }
184             }
185             return true;
186         }
187         public nothrow string ToString() const
188         {
189             string s;
190             s.Reserve(numBits);
191             for (long i = 0; i < numBits; ++i;)
192             {
193                 if (Test(i))
194                 {
195                     s.Append('1');
196                 }
197                 else
198                 {
199                     s.Append('0');
200                 }
201             }
202             return s;
203         }
204         public inline nothrow long NumBits() const
205         {
206             return numBits;
207         }
208         public inline nothrow List<ulong>& Bits() const
209         {
210             return bits;
211         }
212         public inline nothrow long NumFullBlocks() const
213         {
214             if (numBits == 0)
215             {
216                 return 0;
217             }
218             return (numBits - 1) / blockSize;
219         }
220         public inline nothrow long LastBlockStartIndex() const
221         {
222             return NumFullBlocks() * blockSize;
223         }
224         private const long blockSize = 64;
225         private const ulong blockMask = 63u;
226         private const ulong allOne = cast<ulong>(-1);
227         private long numBits;
228         private List<ulong> bits;
229     }
230 
231     public bool operator==(const Bitset& left, const Bitset& right)
232     {
233         long numBits = left.NumBits();
234         if (numBits != right.NumBits())
235         {
236             return false;
237         }
238         long n = left.NumFullBlocks();
239         for (long i = 0; i < n; ++i;)
240         {
241             if (left.Bits()[i] != right.Bits()[i])
242             {
243                 return false;
244             }
245         }
246         for (long i = left.LastBlockStartIndex(); i < numBits; ++i;)
247         {
248             if (left.Test(i) != right.Test(i))
249             {
250                 return false;
251             }
252         }
253         return true;
254     }
