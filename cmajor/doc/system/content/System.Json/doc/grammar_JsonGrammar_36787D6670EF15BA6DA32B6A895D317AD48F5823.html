JsonGrammar Grammar

Value → String | Number | Object | Array | keyword("true") | keyword("false") | keyword("null")
Object → '{' (Field % ',')? '}'
Field → (String ':' Value)
Array → '[' (Value % ',')? ']'
String → token('\"' ([^\"\\] | Escape)* '\"')
Escape → token('\\' ('\"' | '\\' | '/' | 'b' | 'f' | 'n' | 'r' | 't' | 'u' (hexdigit hexdigit hexdigit hexdigit)))
Number → token('-'? digit+ ('.' digit+)? ([eE] [+-]? digit+)?)
