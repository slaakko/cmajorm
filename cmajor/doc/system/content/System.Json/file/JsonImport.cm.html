396     public void FromJson<T>(JsonValue* value, UniquePtr<T>& field)
397         where T is JsonConstructible
398     {
399         if (value is JsonNull*)
400         {
401             field.Reset();
402         }
403         else if (value is JsonObject*)
404         {
405             JsonObject* object = cast<JsonObject*>(value);
406             JsonValue* classValue = object->GetField("class");
407             if (classValue != null)
408             {
409                 if (classValue is JsonString*)
410                 {
411                     JsonString* jsonClassString = cast<JsonString*>(classValue);
412                     ustring jsonClassName = jsonClassString->Value();
413                     JsonValue* objectValue = object->GetField("object");
414                     if (objectValue != null)
415                     {
416                         void* jsonClassValue = JsonClassFactory.Instance().CreateJsonClass(jsonClassName, objectValue);
417                         field.Reset(cast<T*>(jsonClassValue));
418                     }
419                     else
420                     {
421                         throw JsonImportException("JSON 'object' field not found");
422                     }
423                 }
424                 else
425                 {
426                     throw JsonImportException("JSON string expected");
427                 }
428             }
429             else
430             {
431                 throw JsonImportException("JSON 'class' field not found");
432             }
433         }
434         else
435         {
436             throw JsonImportException("null or JSON object expected");
437         }
438     }
