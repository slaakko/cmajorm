102         public override string ToString() const
103         {
104             wstring s =  w'"';
105             for (uchar c : value)
106             {
107                 s.Append(JsonCharStr(c));
108             }
109             s.Append( w'"');
110             return ToUtf8(s);
111         }
112         private wstring JsonCharStr(uchar c)
113         {
114             wstring result;
115             ustring s(c);
116             wstring t = ToUtf16(s);
117             for (wchar x : t)
118             {
119                 switch (x)
120                 {
121                     case '"': result.Append( w"\\\"");break;
122                     case '\\': result.Append( w"\\\\");break;
123                     case '/': result.Append( w"\\/");break;
124                     case '\b': result.Append( w"\\b");break;
125                     case '\f': result.Append( w"\\f");break;
126                     case '\n': result.Append( w"\\n");break;
127                     case '\r': result.Append( w"\\r");break;
128                     case '\t': result.Append( w"\\t");break;
129                     default:
130                     {
131                         if (cast<ushort>(x) >= 32u && cast<ushort>(x) <= 126u)
132                         {
133                             result.Append(x);
134                         }
135                         else
136                         {
137                             result.Append( w"\\u").Append(ToUtf16(ToHexString(cast<ushort>(x))));
138                         }
139                         break;
140                     }
141                 }
142             }
143             return result;
144         }
145         private ustring value;
146     }

148     public class JsonObject : JsonValue
149     {
150         public JsonObject() : fieldValues(), fieldMap()
151         {
152         }
153         public void AddField(const string& fieldName, UniquePtr<JsonValue>&& fieldValue)
154         {
155             AddField(ToUtf32(fieldName), fieldValue);
156         }
157         public void AddField(const wstring& fieldName, UniquePtr<JsonValue>&& fieldValue)
158         {
159             AddField(ToUtf32(fieldName), fieldValue);
160         }
161         public void AddField(const ustring& fieldName, UniquePtr<JsonValue>&& fieldValue)
162         {
163             fieldMap[fieldName] = fieldValue.Get();
164             fieldValues.Add(fieldValue);
165         }
166         public JsonValue* GetField(const string& fieldName) const
167         {
168             return GetField(ToUtf32(fieldName));
169         }
170         public JsonValue* GetField(const wstring& fieldName) const
171         {
172             return GetField(ToUtf32(fieldName));
173         }
174         public JsonValue* GetField(const ustring& fieldName) const
175         {
176             Map<ustring, JsonValue*>.ConstIterator i = fieldMap.CFind(fieldName);
177             if (i != fieldMap.CEnd())
178             {
179                 return i->second;
180             }
181             return null;
182         }
183         public nothrow const Map<ustring, JsonValue*>& Fields() const
184         {
185             return fieldMap;
186         }
187         public nothrow Map<ustring, JsonValue*>& Fields()
188         {
189             return fieldMap;
190         }
191         public override string ToString() const
192         {
193             string s = "{";
194             bool first = true;
195             for (const Pair<ustring, JsonValue*>& p : fieldMap)
196             {
197                 JsonString fs = p.first;
198                 JsonValue* fv = p.second;
199                 if (first)
200                 {
201                     first = false;
202                 }
203                 else
204                 {
205                     s.Append(", ");
206                 }
207                 s.Append(fs.ToString()).Append(":").Append(fv->ToString());
208             }
209             s.Append("}");
210             return s;
211         }
212         private List<UniquePtr<JsonValue>> fieldValues;
213         private Map<ustring, JsonValue*> fieldMap;
214     }
