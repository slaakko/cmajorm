 47     public nothrow Element* GetXmlFieldElement(const string& fieldName, Element* fromElement)
 48     {
 49         UniquePtr<XPathObject> result = Evaluate(ToUtf32(fieldName), fromElement);
 50         if (!result.IsNull())
 51         {
 52             if (result.Get() is XPathNodeSet*)
 53             {
 54                 XPathNodeSet* nodeSet = cast<XPathNodeSet*>(result.Get());
 55                 if (nodeSet->Length() == 1)
 56                 {
 57                     Node* node = (*nodeSet)[0];
 58                     if (node is Element*)
 59                     {
 60                         Element* fieldElement = cast<Element*>(node);
 61                         return fieldElement;
 62                     }
 63                 }
 64             }
 65         }
 66         return null;
 67     }

316     public void FromXml<T>(Element* parentElement, const string& fieldName, List<T>& list)
317         where T is XmlImportableScalarType
318     {
319         list.Clear();
320         Element* element = GetXmlFieldElement(fieldName, parentElement);
321         UniquePtr<XPathObject> result = Evaluate(u"item", element);
322         if (!result.IsNull())
323         {
324             if (result.Get() is XPathNodeSet*)
325             {
326                 XPathNodeSet* nodeSet = cast<XPathNodeSet*>(result.Get());
327                 int n = nodeSet->Length();
328                 for (int i = 0; i < n; ++i;)
329                 {
330                     Node* node = (*nodeSet)[i];
331                     if (node is Element*)
332                     {
333                         Element* element = cast<Element*>(node);
334                         T value;
335                         FromXml(element, "value", value);
336                         list.Add(Rvalue(value));
337                     }
338                 }
339             }
340         }
