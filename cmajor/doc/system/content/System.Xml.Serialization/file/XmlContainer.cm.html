  6 using System;
  7 using System.Collections;
  8 
  9 namespace System.Xml.Serialization
 10 {
 11     public class OwningXmlContainer : XmlContainer
 12     {
 13         public override nothrow bool TakeOwnership() const
 14         {
 15             return true;
 16         }
 17     }
 18 
 19     public class XmlContainer
 20     {
 21         public nothrow XmlContainer()
 22         {
 23         }
 24         public virtual ~XmlContainer()
 25         {
 26             for (const Pair<void*, XmlSerializableProxy*>& p : objectProxyMap)
 27             {
 28                 delete p.second;
 29             }
 30         }
 31         public virtual nothrow bool TakeOwnership() const
 32         {
 33             return false;
 34         }
 35         public virtual nothrow void SetRootObjectId(const Uuid& objectId)
 36         {
 37         }
 38         public nothrow long Count() const
 39         {
 40             return objectProxyMap.Count();
 41         }
 42         public nothrow bool IsEmpty() const
 43         {
 44             return objectProxyMap.IsEmpty();
 45         }
 46         public void Add(XmlSerializable xmlSerializable)
 47         {
 48             Add(xmlSerializable, TakeOwnership());
 49         }
 50         public void Add(XmlSerializable xmlSerializable, bool takeOwnership)
 51         {
 52             XmlSerializableProxy* prev = Get(xmlSerializable.ObjectId());
 53             if (prev != null)
 54             {
 55                 throw XmlSerializationException("object '" + ToString(prev->ObjectId()) + "' already added to XML container");
 56             }
 57             XmlSerializableProxy* proxy = new XmlSerializableProxy(xmlSerializable, takeOwnership);
 58             idProxyMap[proxy->ObjectId()] = proxy;
 59             objectProxyMap[proxy->Object()] = proxy;
 60             proxy->SetContainer(this);
 61         }
 62         public nothrow XmlSerializableProxy* Get(const Uuid& objectId) const
 63         {
 64             HashMap<Uuid, XmlSerializableProxy*>.ConstIterator it = idProxyMap.CFind(objectId);
 65             if (it != idProxyMap.CEnd())
 66             {
 67                 return it->second;
 68             }
 69             else
 70             {
 71                 return null;
 72             }
 73         }
 74         public void Remove(const Uuid& objectId)
 75         {
 76             XmlSerializableProxy* proxy = Get(objectId);
 77             if (proxy != null)
 78             {
 79                 objectProxyMap.Remove(proxy->Object());
 80                 idProxyMap.Remove(objectId);
 81                 delete proxy;
 82             }
 83         }
 84         public nothrow XmlSerializableProxy* GetProxy(void* object)
 85         {
 86             HashMap<void*, XmlSerializableProxy*>.ConstIterator it = objectProxyMap.CFind(object);
 87             if (it != objectProxyMap.CEnd())
 88             {
 89                 XmlSerializableProxy* proxy = it->second;
 90                 return proxy;
 91             }
 92             else
 93             {
 94                 return null;
 95             }
 96         }
 97         public List<XmlSerializableProxy*> GetProxies() const
 98         {
 99             List<XmlSerializableProxy*> proxies;
100             for (const Pair<void*, XmlSerializableProxy*>& p : objectProxyMap)
101             {
102                 if (p.second != null)
103                 {
104                     proxies.Add(p.second);
105                 }
106             }
107             return proxies;
108         }
109         public void AddToBundle(XmlBundle& bundle, XmlSerializable intf, HashSet<Uuid>& addedSet)
110         {
111             if (intf.ObjectId() == Uuid())
112             {
113                 return;
114             }
115             if (addedSet.CFind(intf.ObjectId()) != addedSet.CEnd())
116             {
117                 return;
118             }
119             bundle.Add(intf);
120             addedSet.Insert(intf.ObjectId());
121             List<XmlPtrBase*> ptrs = intf.GetPtrs();
122             for (XmlPtrBase* ptr : ptrs)
123             {
124                 if (ptr->IncludeInBundle() && ptr->TargetObjectId() != Uuid())
125                 {
126                     XmlSerializableProxy* target = Get(ptr->TargetObjectId());
127                     if (target != null)
128                     {
129                         AddToBundle(bundle, target->Interface(), addedSet);
130                     }
131                 }
132             }
133         }
134         public XmlBundle CreateBundle(void* object)
135         {
136             HashSet<Uuid> addedSet;
137             XmlBundle bundle;
138             XmlSerializableProxy* proxy = GetProxy(object);
139             if (proxy != null)
140             {
141                 bundle.SetRootObjectId(proxy->ObjectId());
142                 XmlSerializable intf = proxy->Interface();
143                 AddToBundle(bundle, intf, addedSet);
144             }
145             return bundle;
146         }
147         public HashMap<Uuid, XmlSerializableProxy*> idProxyMap;
148         public HashMap<void*, XmlSerializableProxy*> objectProxyMap;
149     }

151     public void Add<T>(T* object, XmlContainer* container)
152     {
153         XmlSerializable xmlSerializable(*object);
154         container->Add(xmlSerializable);
155     }
156 
157     public void AddOrReplace<T>(T* object, XmlContainer* container)
158     {
159         XmlSerializable xmlSerializable(*object);
160         container->Remove(xmlSerializable.ObjectId());
161         container->Add(xmlSerializable);
162     }
163 
164     public void RemoveFromContainer<T>(T* object)
165     {
166         XmlSerializable xmlSerializable(*object);
167         XmlContainer* container = xmlSerializable.Container();
168         if (container != null)
169         {
170             container->Remove(xmlSerializable.ObjectId());
171         }
172     }
