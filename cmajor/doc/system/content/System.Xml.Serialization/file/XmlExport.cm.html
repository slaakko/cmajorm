 61     public UniquePtr<Element> ToXml(const System.Uuid& value, const string& fieldName)
 62     {
 63         UniquePtr<Element> element(new Element(ToUtf32(fieldName)));
 64         element->SetAttribute(u"value", ToUtf32(ToString(value)));
 65         return element;
 66     }

 68     public UniquePtr<Element> ToXml<T>(const T& object, const string& fieldName)
 69         where T is XmlExportableClassType
 70     {
 71         UniquePtr<Element> element = object.ToXml(fieldName);
 72         XmlSerializable intf = object;
 73         int classId = intf.ClassId();
 74         element->SetAttribute(u"classId", ToUtf32(ToString(classId)));
 75         return element;
 76     }

 78     public UniquePtr<Element> ToXml<T>(const UniquePtr<T>& objectPtr, const string& fieldName)
 79         where T is XmlExportableClassType
 80     {
 81         if (objectPtr.IsNull())
 82         {
 83             UniquePtr<Element> element(new Element(ToUtf32(fieldName)));
 84             element->SetAttribute(u"value", u"null");
 85             return element;
 86         }
 87         else
 88         {
 89             UniquePtr<Element> element = objectPtr->ToXml(fieldName);
 90             T* p = objectPtr.Get();
 91             XmlSerializable intf = *p;
 92             int classId = intf.ClassId();
 93             element->SetAttribute(u"classId", ToUtf32(ToString(classId)));
 94             return element;
 95         }
 96     }

152     public UniquePtr<Element> ToXml<T>(const List<T>& list, const string& fieldName)
153         where T is XmlExportable
154     {
155         UniquePtr<Element> element(new Element(ToUtf32(fieldName)));
156         for (const T& item : list)
157         {
158             element->AppendChild(UniquePtr<Node>(ToXml(item, "item").Release()));
159         }
160         return element;
161     }
