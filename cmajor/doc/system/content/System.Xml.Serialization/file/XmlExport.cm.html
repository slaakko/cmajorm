 55     public UniquePtr<Element> ToXml<T>(const T& object, const string& fieldName)
 56         where T is XmlExportableClassType
 57     {
 58         UniquePtr<Element> element = object.ToXml(fieldName);
 59         element->SetAttribute(u"className", ToUtf32(typename(object)));
 60         return element;
 61     }

 63     public UniquePtr<Element> ToXml<T>(const UniquePtr<T>& objectPtr, const string& fieldName)
 64         where T is XmlExportableClassType
 65     {
 66         if (objectPtr.IsNull())
 67         {
 68             UniquePtr<Element> element(new Element(ToUtf32(fieldName)));
 69             element->SetAttribute(u"value", u"null");
 70             return element;
 71         }
 72         else
 73         {
 74             UniquePtr<Element> element = objectPtr->ToXml(fieldName);
 75             T* p = objectPtr.Get();
 76             element->SetAttribute(u"className", ToUtf32(typename(*p)));
 77             return element;
 78         }
 79     }

 99     public UniquePtr<Element> ToXml<T>(const List<T>& list, const string& fieldName)
100         where T is XmlExportable
101     {
102         UniquePtr<Element> element(new Element(ToUtf32(fieldName)));
103         for (const T& item : list)
104         {
105             element->AppendChild(UniquePtr<Node>(ToXml(item, "item").Release()));
106         }
107         return element;
108     }
