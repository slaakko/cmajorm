 25         public nothrow const Uuid& RootObjectId() const
 26         {
 27             return rootObjectId;
 28         }
 29         public nothrow void SetRootObjectId(const Uuid& rootObjectId_)
 30         {
 31             rootObjectId = rootObjectId_;
 32         }
 33         public nothrow XmlSerializableProxy* Root() const
 34         {
 35             if (rootObjectId != Uuid())
 36             {
 37                 return Get(rootObjectId);
 38             }
 39             else
 40             {
 41                 return null;
 42             }
 43         }
 44         public UniquePtr<System.Dom.Document> ToXmlDocument()
 45         {
 46             UniquePtr<System.Dom.Element> rootElement(new System.Dom.Element(u"xmlBundle"));
 47             rootElement->SetAttribute(u"rootObjectId", ToUtf32(ToString(rootObjectId)));
 48             List<XmlSerializableProxy*> proxies = GetProxies();
 49             for (XmlSerializableProxy* proxy : proxies)
 50             {
 51                 if (proxy->Object() != null)
 52                 {
 53                     UniquePtr<System.Dom.Element> serializableElement = proxy->ToXml("object");
 54                     rootElement->AppendChild(UniquePtr<System.Dom.Node>(serializableElement.Release()));
 55                 }
 56             }
 57             UniquePtr<System.Dom.Document> document(new System.Dom.Document());
 58             document->AppendChild(UniquePtr<System.Dom.Node>(rootElement.Release()));
 59             return document;
 60         }
 61         public string ToXmlString() const
 62         {
 63             return ToXmlString(1);
 64         }
 65         public string ToXmlString(int indent) const
 66         {
 67             UniquePtr<System.Dom.Document> document = ToXmlDocument();
 68             StringWriter stringWriter;
 69             CodeFormatter formatter(stringWriter);
 70             formatter.SetIndentSize(indent);
 71             document->Write(formatter);
 72             return stringWriter.GetString();
 73         }
 74         public void ResolveXmlPtrs()
 75         {
 76             List<XmlSerializableProxy*> proxies = GetProxies();
 77             for (XmlSerializableProxy* proxy : proxies)
 78             {
 79                 List<XmlPtrBase*> ptrs = proxy->GetPtrs();
 80                 for (XmlPtrBase* ptr : ptrs)
 81                 {
 82                     ptr->Resolve(this);
 83                 }
 84             }
 85         }
 86         private Uuid rootObjectId;
 87     }

 94     public UniquePtr<XmlBundle> ToXmlBundle(const string& xmlStr, const string& systemId, bool createOwningBundle)
 95     {
 96         UniquePtr<XmlBundle> bundle;
 97         if (createOwningBundle)
 98         {
 99             bundle.Reset(new OwningXmlBundle());
100         }
101         else
102         {
103             bundle.Reset(new XmlBundle());
104         }
105         ustring content = ToUtf32(xmlStr);
106         UniquePtr<System.Dom.Document> document = System.Dom.ParseDocument(content, systemId);
107         UniquePtr<System.XPath.XPathObject> bundleObject = System.XPath.Evaluate(u"xmlBundle", document.Get());
108         if (!bundleObject.IsNull())
109         {
110             if (bundleObject.Get() is System.XPath.XPathNodeSet*)
111             {
112                 System.XPath.XPathNodeSet* nodeSet = cast<System.XPath.XPathNodeSet*>(bundleObject.Get());
113                 int length = nodeSet->Length();
114                 if (length == 1)
115                 {
116                     System.Dom.Node* node = (*nodeSet)[0];
117                     if (node is System.Dom.Element*)
118                     {
119                         System.Dom.Element* rootElement = cast<System.Dom.Element*>(node);
120                         ustring rootObjectIdAttr = rootElement->GetAttribute(u"rootObjectId");
121                         if (!rootObjectIdAttr.IsEmpty())
122                         {
123                             Uuid rootObjectId = ParseUuid(ToUtf8(rootObjectIdAttr));
124                             bundle->SetRootObjectId(rootObjectId);
125                         }
126                     }
127                     UniquePtr<System.XPath.XPathObject> object = System.XPath.Evaluate(u"xmlBundle/object", document.Get());
128                     if (!object.IsNull())
129                     {
130                         if (object.Get() is System.XPath.XPathNodeSet*)
131                         {
132                             System.XPath.XPathNodeSet* nodeSet = cast<System.XPath.XPathNodeSet*>(object.Get());
133                             int n = nodeSet->Length();
134                             for (int i = 0; i < n; ++i;)
135                             {
136                                 System.Dom.Node* node = (*nodeSet)[i];
137                                 if (node is System.Dom.Element*)
138                                 {
139                                     System.Dom.Element* element = cast<System.Dom.Element*>(node);
140                                     ustring classIdAttr = element->GetAttribute(u"classId");
141                                     if (!classIdAttr.IsEmpty())
142                                     {
143                                         int classId = ParseInt(ToUtf8(classIdAttr));
144                                         XmlSerializable xmlSerializable = XmlClassRegistry.Instance().CreateXmlSerializable(classId);
145                                         xmlSerializable.FromXml(element);
146                                         bundle->Add(xmlSerializable);
147                                     }
148                                 }
149                             }
150                         }
151                     }
152                     bundle->ResolveXmlPtrs();
153                 }
154                 else
155                 {
156                     throw XmlSerializationException("single 'xmlBundle' element expected");
157                 }
158             }
159         }
160         return bundle;
161     }

163     public nothrow T* ReleaseRoot<T>(XmlBundle* bundle)
164     {
165         XmlSerializableProxy* root = bundle->Root();
166         if (root != null)
167         {
168             root->ResetOwner();
169             return XmlCast<T>(root);
170         }
171         else
172         {
173             return null;
174         }
