  6 using System;
  7 using System.Collections;
  8 
  9 namespace System.Xml.Serialization
 10 {
 11     public interface XmlSerializable
 12     {
 13         voidDestroyObject();
 14         UuidObjectId();
 15         voidSetObjectId( const Uuid& objectId);
 16         intClassId();
 17         stringClassName();
 18         XmlContainer*Container();
 19         voidSetContainer( XmlContainer* container);
 20         System.Dom.Element*ToXml( const string& elementName);
 21         voidFromXml( System.Dom.Element* element);
 22         List<XmlPtrBase*>GetPtrs();
 23     }
 24 
 25     public class XmlSerializableProxy
 26     {
 27         public nothrow XmlSerializableProxy(XmlSerializable intf_, bool isOwner_) : intf(intf_), isOwner(isOwner_)
 28         {
 29             if (intf.ObjectId() == Uuid())
 30             {
 31                 intf.SetObjectId(Uuid.Random());
 32             }
 33         }
 34         public ~XmlSerializableProxy()
 35         {
 36             if (isOwner)
 37             {
 38                 intf.DestroyObject();
 39             }
 40         }
 41         public nothrow void ResetOwner()
 42         {
 43             isOwner = false;
 44         }
 45         public nothrow XmlSerializable Interface() const
 46         {
 47             return intf;
 48         }
 49         public nothrow XmlContainer* Container() const
 50         {
 51             return intf.Container();
 52         }
 53         public nothrow void SetContainer(XmlContainer* container)
 54         {
 55             if (Container() == null)
 56             {
 57                 intf.SetContainer(container);
 58             }
 59         }
 60         public nothrow void ResetContainer()
 61         {
 62             intf.SetContainer(null);
 63         }
 64         public nothrow void* Object() const
 65         {
 66             return intf.GetObjectPtrFromInterface();
 67         }
 68         public nothrow Uuid ObjectId() const
 69         {
 70             return intf.ObjectId();
 71         }
 72         public nothrow int ClassId() const
 73         {
 74             return intf.ClassId();
 75         }
 76         public nothrow string ClassName() const
 77         {
 78             return intf.ClassName();
 79         }
 80         public System.Dom.Element* ToXml(const string& elementName)
 81         {
 82             return intf.ToXml(elementName);
 83         }
 84         public void FromXml(System.Dom.Element* element)
 85         {
 86             intf.FromXml(element);
 87         }
 88         public List<XmlPtrBase*> GetPtrs() const
 89         {
 90             return intf.GetPtrs();
 91         }
 92         private XmlSerializable intf;
 93         private bool isOwner;
 94     }
 95 
 96     public nothrow T* XmlCast<T>(XmlSerializableProxy* proxy)
 97     {
 98         string className = T.StaticClassName();
 99         string serializableClassName = proxy->ClassName();
100         if (className == serializableClassName)
101         {
102             return cast<T*>(proxy->Object());
103         }
104         else
105         {
106             return null;
107         }
108     }
109 }
