  6 using System;
  7 using System.Collections;
  8 
  9 namespace System.Xml.Serialization
 10 {
 11     public interface XmlSerializable
 12     {
 13         void DestroyObject();
 14         Uuid ObjectId();
 15         void SetObjectId(const Uuid& objectId);
 16         int ClassId();
 17         string ClassName();
 18         XmlContainer* Container();
 19         void SetContainer(XmlContainer* container);
 20         bool IsOwned();
 21         void SetOwned();
 22         void ResetOwned();
 23         System.Dom.Element* ToXml(const string& elementName, XmlSerializationContext& ctx);
 24         void FromXml(System.Dom.Element* element);
 25         List<XmlPtrBase*> GetPtrs();
 26     }
 27 
 28     public class XmlSerializableProxy
 29     {
 30         public nothrow XmlSerializableProxy(XmlSerializable intf_) : intf(intf_)
 31         {
 32             if (intf.ObjectId() == Uuid())
 33             {
 34                 intf.SetObjectId(Uuid.Random());
 35             }
 36         }
 37         public void DestroyObject()
 38         {
 39             intf.DestroyObject();
 40         }
 41         public nothrow XmlSerializable Interface() const
 42         {
 43             return intf;
 44         }
 45         public nothrow XmlContainer* Container() const
 46         {
 47             return intf.Container();
 48         }
 49         public nothrow void SetContainer(XmlContainer* container)
 50         {
 51             if (Container() == null)
 52             {
 53                 intf.SetContainer(container);
 54             }
 55         }
 56         public nothrow void ResetContainer()
 57         {
 58             intf.SetContainer(null);
 59         }
 60         public nothrow bool IsOwned() const
 61         {
 62             return intf.IsOwned();
 63         }
 64         public nothrow void SetOwned()
 65         {
 66             intf.SetOwned();
 67         }
 68         public nothrow void ResetOwned()
 69         {
 70             intf.ResetOwned();
 71         }
 72         public nothrow void* Object() const
 73         {
 74             return intf.GetObjectPtrFromInterface();
 75         }
 76         public nothrow Uuid ObjectId() const
 77         {
 78             return intf.ObjectId();
 79         }
 80         public nothrow int ClassId() const
 81         {
 82             return intf.ClassId();
 83         }
 84         public nothrow string ClassName() const
 85         {
 86             return intf.ClassName();
 87         }
 88         public System.Dom.Element* ToXml(const string& elementName)
 89         {
 90             XmlSerializationContext ctx;
 91             return ToXml(elementName, ctx);
 92         }
 93         public System.Dom.Element* ToXml(const string& elementName, XmlSerializationContext& ctx)
 94         {
 95             return intf.ToXml(elementName, ctx);
 96         }
 97         public void FromXml(System.Dom.Element* element)
 98         {
 99             intf.FromXml(element);
100         }
101         public List<XmlPtrBase*> GetPtrs() const
102         {
103             return intf.GetPtrs();
104         }
105         private XmlSerializable intf;
106     }
107 
108     public nothrow T* XmlCast<T>(XmlSerializableProxy* proxy)
109     {
110         string className = T.StaticClassName();
111         string serializableClassName = proxy->ClassName();
112         if (className == serializableClassName)
113         {
114             return cast<T*>(proxy->Object());
115         }
116         else
117         {
118             return null;
119         }
120     }
