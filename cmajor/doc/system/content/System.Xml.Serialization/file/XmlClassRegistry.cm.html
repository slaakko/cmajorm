 6 using System;
 7 using System.Collections;
 8 
 9 namespace System.Xml.Serialization
10 {
11     public abstract class XmlFactory
12     {
13         public nothrow XmlFactory(const string& className_) : className(className_)
14         {
15         }
16         public abstract void* Create(System.Dom.Element* element);
17         public nothrow const string& ClassName() const
18         {
19             return className;
20         }
21         private string className;
22     }
23  
24     public class XmlClassFactory<T> : XmlFactory
25     {
26         private typedef T XmlSerializableClassType;
27         public nothrow XmlClassFactory(const string& className) : base(className)
28         {
29         }
30         public override void* Create(System.Dom.Element* element)
31         {
32             return new XmlSerializableClassType(element);
33         }
34     }
35 
36     public class XmlClassRegistry
37     {
38         static XmlClassRegistry() : instance(new XmlClassRegistry())
39         {
40         }
41         public static XmlClassRegistry& Instance()
42         {
43             return *instance;
44         }
45         public nothrow void Register(XmlFactory* factory)
46         {
47             factories.Add(UniquePtr<XmlFactory>(factory));
48             factoryMap[factory->ClassName()] = factory;
49         }
50         public void* Create(System.Dom.Element* element)
51         {
52             ustring classNameAttr = element->GetAttribute(u"className");
53             if (classNameAttr.IsEmpty())
54             {
55                 throw Exception("element '" + ToUtf8(element->Name()) + "' does not contain a 'className' attribute");
56             }
57             string className = ToUtf8(classNameAttr);
58             Map<string, XmlFactory*>.ConstIterator it = factoryMap.CFind(className);
59             if (it != factoryMap.CEnd())
60             {
61                 XmlFactory* factory = it->second;
62                 return factory->Create(element);
63             }
64             throw Exception("class '" + className + "' not registered to XML class registry");
65         }
66         private static UniquePtr<XmlClassRegistry> instance;
67         private List<UniquePtr<XmlFactory>> factories;
68         private Map<string, XmlFactory*> factoryMap;
69     }
70 
71     public void RegisterXmlClass<T>()
72     {
73         T defaultValue;
74         string className = typename(defaultValue);
75         XmlClassRegistry.Instance().Register(new XmlClassFactory<T>(className));
76     }
77 }
