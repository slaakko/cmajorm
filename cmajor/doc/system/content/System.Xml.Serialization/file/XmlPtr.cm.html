  6 using System;
  7 using System.Collections;

  9 namespace System.Xml.Serialization
 10 {
 11     // XmlPtrBase contains an object id (UUID) of an object it points to.
 12     // The Resolve member function retrieves a proxy object from an XML container,
 13     // obtains an object pointer from the proxy and calls the SetPtr member function
 14     // of the derived class.
 15     //
 16     // Each derived class implements the SetPtr member function that sets the pointer contained by the derived class.
 17 
 18     public abstract class XmlPtrBase
 19     {
 20         public nothrow XmlPtrBase() : targetObjectId(Uuid()), includeInBundle(true)
 21         {
 22         }
 23         public virtual ~XmlPtrBase()
 24         {
 25         }
 26         public nothrow const Uuid& TargetObjectId() const
 27         {
 28             return targetObjectId;
 29         }
 30         public nothrow void SetTargetObjectId(const Uuid& targetObjectId_)
 31         {
 32             targetObjectId = targetObjectId_;
 33         }
 34         public virtual nothrow bool IncludeInBundle() const
 35         {
 36             return includeInBundle;
 37         }
 38         public nothrow void SetIncludeInBundle(bool includeInBundle_)
 39         {
 40              includeInBundle = includeInBundle_;
 41         }
 42         public nothrow void Resolve(XmlContainer* container)
 43         {
 44             XmlSerializableProxy* proxy = container->Get(targetObjectId);
 45             if (proxy != null)
 46             {
 47                 SetPtr(proxy->Object());
 48             }
 49         }
 50         public abstract nothrow void* GetPtr() const;
 51         public abstract nothrow void SetPtr(void* p);
 52         private Uuid targetObjectId;
 53         private bool includeInBundle;
 54     }

 56     // XmlPtr<T> is like an ordinary pointer with no ownership semantics,
 57     // but it derives from the XmlPtrBase that contains an object id for the pointed-to object.
 58 
 59     public class XmlPtr<T> : XmlPtrBase
 60     {
 61         private typedef XmlPtr<T> Self;
 62 
 63         public nothrow XmlPtr() : base(), ptr(null)
 64         {
 65         }
 66         public explicit nothrow XmlPtr(T* ptr_) : base(), ptr(ptr_)
 67         {
 68         }
 69         public nothrow XmlPtr(const Self& that) : ptr(that.ptr)
 70         {
 71             SetId();
 72         }
 73         public nothrow XmlPtr(Self&& that) : ptr(that.ptr)
 74         {
 75             that.ptr = null;
 76             SetId();
 77         }
 78         public nothrow void operator=(const Self& that)
 79         {
 80             ptr = that.ptr;
 81             SetId();
 82         }
 83         public nothrow void operator=(Self&& that)
 84         {
 85             Swap(ptr, that.ptr);
 86             SetId();
 87             that.SetId();
 88         }
 89         public nothrow void operator=(T* ptr_)
 90         {
 91             ptr = ptr_;
 92             SetId();
 93         }
 94         public nothrow void Reset(T* ptr_)
 95         {
 96             ptr = ptr_;
 97             SetId();
 98         }
 99         public nothrow void Reset()
100         {
101             ptr = null;
102             SetId();
103         }
104         public inline nothrow T* Get() const
105         {
106             return ptr;
107         }
108         public inline nothrow bool IsNull() const
109         {
110             return ptr == null;
111         }
112         public inline nothrow T* operator->()
113         {
114             return ptr;
115         }
116         public inline nothrow const T* operator->() const
117         {
118             return ptr;
119         }
120         public inline nothrow T& operator*()
121         {
122             return *ptr;
123         }
124         public inline nothrow const T& operator*() const
125         {
126             return *ptr;
127         }
128         public nothrow void Swap(Self& that)
129         {
130             Swap(ptr, that.ptr);
131             SetId();
132             that.SetId();
133         }
134         public override nothrow void* GetPtr() const
135         {
136             return ptr;
137         }
138         public override nothrow void SetPtr(void* p)
139         {
140             ptr = cast<T*>(p);
141         }
142         private nothrow void SetId()
143         {
144             if (!IsNull())
145             {
146                 XmlSerializable intf = *Get();
147                 SetTargetObjectId(intf.ObjectId());
148             }
149             else
150             {
151                 SetTargetObjectId(Uuid());
152             }
153         }
154         private T* ptr;
155     }
156 
157     public nothrow bool operator==<T>(const XmlPtr<T>& left, const XmlPtr<T>& right)
158     {
159         return left.Get() == right.Get();
160     }
161 
162     public nothrow bool operator<<T>(const XmlPtr<T>& left, const XmlPtr<T>& right)
163     {
164         return left.Get() < right.Get();
165     }

167     // UniqueXmlPtr<T> is like a UniquePtr<T> but in addition it derives from the XmlPtrBase,
168     // that contains an object id for the pointed-to object.
169 
170     public class UniqueXmlPtr<T> : XmlPtrBase
171     {
172         private typedef UniqueXmlPtr<T> Self;
173 
174         public nothrow UniqueXmlPtr() : ptr()
175         {
176         }
177         public explicit nothrow UniqueXmlPtr(T* ptr_) : ptr(ptr_)
178         {
179             SetId();
180         }
181         suppress UniqueXmlPtr(const Self&);
182         public nothrow UniqueXmlPtr(Self&& that) : ptr(Rvalue(that.ptr))
183         {
184             SetId();
185             that.SetId();
186         }
187         public nothrow void operator=(T* ptr_)
188         {
189             ptr = ptr_;
190             SetId();
191         }
192         suppress void operator=(const Self&);
193         public nothrow void operator=(Self&& that)
194         {
195             ptr = Rvalue(that.ptr);
196             SetId();
197             that.SetId();
198         }
199         public nothrow void Reset()
200         {
201             ptr.Reset();
202             SetId();
203         }
204         public nothrow void Reset(T* ptr_)
205         {
206             ptr.Reset(ptr_);
207             SetId();
208         }
209         public inline nothrow T* Get()
210         {
211             return ptr.Get();
212         }
213         public inline nothrow T* Release()
214         {
215             return ptr.Release();
216         }
217         public inline nothrow bool IsNull() const
218         {
219             return ptr.IsNull();
220         }
221         public inline nothrow T* operator->()
222         {
223             return ptr.Get();
224         }
225         public inline nothrow const T* operator->() const
226         {
227             return ptr.Get();
228         }
229         public inline nothrow T& operator*()
230         {
231             return *ptr;
232         }
233         public inline nothrow const T& operator*() const
234         {
235             return *ptr;
236         }
237         public nothrow void Swap(Self& that)
238         {
239             Swap(ptr, that.ptr);
240             SetId();
241             that.SetId();
242         }
243         public override nothrow void* GetPtr() const
244         {
245             return ptr.Get();
246         }
247         public override nothrow void SetPtr(void* p)
248         {
249             ptr.Reset(cast<T*>(p));
250         }
251         private nothrow void SetId()
252         {
253             if (!IsNull())
254             {
255                 XmlSerializable intf = *Get();
256                 SetTargetObjectId(intf.ObjectId());
257             }
258             else
259             {
260                 SetTargetObjectId(Uuid());
261             }
262         }
263         private UniquePtr<T> ptr;
264     }
265 
266     public nothrow bool operator==<T>(const UniqueXmlPtr<T>& left, const UniqueXmlPtr<T>& right)
267     {
268         return left.Get() == right.Get();
269     }
