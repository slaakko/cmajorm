HttpGrammar Grammar

statusLine → (httpVersion SP statusCode SP reasonPhrase)
httpVersion → ("HTTP" '/' DIGIT+ '.' DIGIT+)
statusCode → (DIGIT DIGIT DIGIT)
reasonPhrase → ((TEXT − (CR | LF))*)
charset → httpToken
contentCoding → httpToken
transferCoding → "chunked" | transferExtension
commaSeparatedFieldValue → generalFieldValue % ','
generalFieldValue → (httpToken (';' generalParameter)*)
generalParameter → (attribute ('=' attrval)?)
chunkHeader → (hexdigit+) chunkExtensions?
chunkExtensions → (';' (chunkExtName ('=' chunkExtVal)?))*
chunkExtName → httpToken
chunkExtVal → httpToken | quotedString
transferExtension → httpToken (';' parameter)*
parameter → (attribute '=' attrval)
attribute → httpToken
attrval → httpToken | quotedString
mediaType → (type '/' subtype) (';' parameter)*
type → httpToken
subtype → httpToken
httpHeader → (fieldName ':' fieldValue)
fieldName → httpToken
fieldValue → (anychar*)
dateValue → httpDate
httpDate → rfc1123Date | rfc850Date | asctimeDate
rfc1123Date → (wkday ',' SP date1 SP time SP "GMT")
rfc850Date → (weekday "," SP date2 SP time SP "GMT")
asctimeDate → (wkday SP date3 SP time SP Year)
date1 → (Day SP month SP Year)
date2 → (Day '-' month '-' Year2)
date3 → (month SP (Day | Day1))
Year → (DIGIT DIGIT DIGIT DIGIT)
Year2 → (DIGIT DIGIT)
Day → (DIGIT DIGIT)
Day1 → SP DIGIT
wkday → "Mon" | "Tue" | "Wed" | "Thu" | "Fri" | "Sat" | "Sun"
weekday → "Monday" | "Tuesday" | "Wednesday" | "Thursday" | "Friday" | "Saturday" | "Sunday"
month → "Jan" | "Feb" | "Mar" | "Apr" | "May" | "Jun" | "Jul" | "Aug" | "Sep" | "Oct" | "Nov" | "Dec"
time → (Hour ':' Min ':' Sec)
Hour → (DIGIT DIGIT)
Min → (DIGIT DIGIT)
Sec → (DIGIT DIGIT)
deltaSeconds → (DIGIT+)
httpToken → ((CHAR − (CTL | sep))+)
sep → [()<>@,;:\\\"/[]?={}] | SP | HT
comment → '(' (ctext | quotedPair | comment)* ')'
ctext → (TEXT − [()])
quotedString → '\"' (qdtext | quotedPair)* '\"'
qdtext → (TEXT − '\"')
quotedPair → '\\' CHAR
OCTET → range(0,255)
CHAR → range(0,127)
UPALPHA → [A-Z]
LOALPHA → [a-z]
ALPHA → UPALPHA | LOALPHA
DIGIT → [0-9]
CTL → range(0,31) | '\x0000007F'
CR → '\r'
LF → '\n'
SP → ' '
HT → '\t'
QUOTE → '\"'
CRLF → CR LF
LWS → CRLF? (SP | HT)+
TEXT → (OCTET − CTL) | CR | LF | SP | HT
HEX → [A-Fa-f0-9]
