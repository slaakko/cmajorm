146         public string ToString() const
147         {
148             string value = fieldValue;
149             for (const Pair<string, string>& p : parameters)
150             {
151                 value.Append(';').Append(p.first);
152                 if (!p.second.IsEmpty())
153                 {
154                     value.Append('=');
155                     string paramValue = p.second;
156                     if (paramValue.Find('"') != -1 || paramValue.Find(' ') != -1)
157                     {
158                         paramValue = MakeStringLiteral(paramValue);
159                     }
160                     value.Append(paramValue);
161                 }
162             }
163             return value;
164         }

166         {
167             fieldValue = fieldValue_;
168         }
169         public void SetParameter(const string& paramName, const string& paramValue)
170         {
171             parameters[ToLower(paramName)] = paramValue;
172         }
173         public string GetParameter(const string& paramName) const
174         {
175             string pn = ToLower(paramName);
176             Map<string, string>.ConstIterator it = parameters.CFind(pn);
177             if (it != parameters.CEnd())
178             {
179                 return it->second;
180             }
181             return string();
182         }

551     public class HttpRequest
552     {
553         public HttpRequest(const UriReference& absoluteUri_, HttpHeaderCollection& headers_) : 
554             this(UniquePtr<HttpMethod>(new HttpGetMethod()), absoluteUri_, UniquePtr<HttpVersion>(new Http_1_1_Version()), headers_)
555         {
556         }
557         public HttpRequest(UniquePtr<HttpMethod>&& method_, const UriReference& absoluteUri_, UniquePtr<HttpVersion>&& httpVersion_, HttpHeaderCollection& headers_) : 
558             method(method_), absoluteUri(absoluteUri_), httpVersion(httpVersion_), headers(headers_)
559         {
560         }
561         public void Write(StreamWriter& writer)
562         {
563             Write(writer, null);
564         }
565         public void Write(StreamWriter& writer, StreamWriter* log)
566         {
567             string requestLine = method->MethodName();
568             UriReference requestUri;
569             requestUri.SetPath(absoluteUri.Path());
570             requestLine.Append(' ');
571             requestLine.Append(requestUri.ToString());
572             requestLine.Append(' ');
573             requestLine.Append(httpVersion->Version());
574             if (log != null)
575             {
576                 log->WriteLine("REQUEST:");
577                 log->WriteLine(requestLine);
578             }
579             writer.Write(requestLine);
580             writer.Write("\r\n");
581             HttpHeader* hostHeader = headers.GetHeader("host");
582             if (hostHeader == null)
583             {
584                 headers.Add(UniquePtr<HttpHeader>(new HttpHostHeader(absoluteUri.GetAuthority().Host(), absoluteUri.GetAuthority().Port())));
585             }
586             HttpHeader* userAgentHeader = headers.GetHeader("user-agent");
587             if (userAgentHeader == null)
588             {
589                 headers.Add(UniquePtr<HttpHeader>(new HttpUserAgentHeader()));
590             }
591             for (const UniquePtr<HttpHeader>& header : headers.Headers())
592             {
593                 string headerStr = header->ToString();
594                 if (log != null)
595                 {
596                     log->WriteLine(headerStr);
597                 }
598                 writer.Write(headerStr);
599                 writer.Write("\r\n");
600             }
601             writer.Write("\r\n");
602             writer.ContainedStream()->Flush();
603         }
604         private UniquePtr<HttpMethod> method;
605         private UriReference absoluteUri;
606         private UniquePtr<HttpVersion> httpVersion;
607         private HttpHeaderCollection& headers;
608     }
