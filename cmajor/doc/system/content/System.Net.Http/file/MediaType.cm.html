177     public class MimeType
178     {
179         public MimeType() : mediaType(new AnyMediaType()), subtype(new AnySubtype())
180         {
181         }
182         public MimeType(UniquePtr<MediaType>&& mediaType_) : mediaType(mediaType_), subtype(new AnySubtype())
183         {
184         }
185         public MimeType(UniquePtr<MediaType>&& mediaType_, UniquePtr<Subtype>&& subtype_) : mediaType(mediaType_), subtype(subtype_)
186         {
187         }
188         public inline nothrow MediaType* GetMediaType() const
189         {
190             return mediaType.Get();
191         }
192         public inline nothrow Subtype* GetSubtype() const
193         {
194             return subtype.Get();
195         }
196         public void SetValue(const ustring& mediaType_, const ustring& subtype_)
197         {
198             mediaType.Reset(new MediaType(mediaType_));
199             subtype.Reset(new Subtype(subtype_));
200         }
201         public inline nothrow const Map<ustring, ustring>& Parameters() const
202         {
203             return parameters;
204         }
205         public ustring GetParameter(const ustring& paramName) const
206         {
207             Map<ustring, ustring>.ConstIterator it = parameters.CFind(paramName);
208             if (it != parameters.CEnd())
209             {
210                 return it->second;
211             }
212             return ustring();
213         }
214         public void SetParameter(const ustring& paramName, const ustring& paramValue)
215         {
216             parameters[paramName] = paramValue;
217         }
218         public ustring ToString() const
219         {
220             if (parameters.IsEmpty())
221             {
222                 return mediaType->Name() +  u"/" + subtype->Name();
223             }
224             else
225             {
226                 ustring s = mediaType->Name() +  u"/" + subtype->Name();
227                 for (const Pair<ustring, ustring>& p : parameters)
228                 {
229                     ustring paramValue = p.second;
230                     if (paramValue.Find( u'"') != -1)
231                     {
232                         paramValue = ToUtf32(MakeStringLiteral(ToUtf8(paramValue)));
233                     }
234                     s.Append( u';').Append(p.first).Append(u'=').Append(paramValue);
235                 }
236                 return s;
237             }
238         }
239         private UniquePtr<MediaType> mediaType;
240         private UniquePtr<Subtype> subtype;
241         private Map<ustring, ustring> parameters;
242     }

244     public class MediaRange
245     {
246         public MediaRange() : mimeType(), quality(1)
247         {
248         }
249         public MediaRange(MimeType&& mimeType_) : mimeType(mimeType_), quality(1)
250         {
251         }
252         public Map<ustring, ustring>& Params()
253         {
254             return params;
255         }
256         public HttpFieldValue ToHttpFieldValue() const
257         {
258             if (quality == 1 && params.IsEmpty())
259             {
260                 return HttpFieldValue(mimeType.ToString());
261             }
262             else
263             {
264                 HttpFieldValue fieldValue(mimeType.ToString());
265                 if (quality != 1)
266                 {
267                     fieldValue.SetParameter( u"q", ToUtf32(ToString(quality, 3)));
268                 }
269                 for (const Pair<ustring, ustring>& p : params)
270                 {
271                     fieldValue.SetParameter(p.first, p.second);
272                 }
273                 return fieldValue;
274             }
275         }
276         private MimeType mimeType;
277         private double quality;
278         private Map<ustring, ustring> params;
279     }
