 6 using System;
 7 using System.Collections;
 8 
 9 namespace System.Net.Http
10 {
11     public class HttpParser
12     {
13         static HttpParser() : instance(new HttpParser())
14         {
15         }
16         public static HttpParser& Instance()
17         {
18             return *instance;
19         }
20         public HttpHeader ParseHeader(const ustring& header)
21         {
22             return headerGrammar->Parse(header.Chars(), header.Chars() + header.Length(), 0, "");
23         }
24         public HttpStatus ParseStatus(const ustring& statusLine)
25         {
26             HttpStatus httpStatus;
27             statusLineGrammar->Parse(statusLine.Chars(), statusLine.Chars() + statusLine.Length(), 0, "", &httpStatus);
28             return httpStatus;
29         }
30         public void ParseChunkHeader(const ustring& chunkHeader, ulong* chunkSize, ChunkExtensionAdder* adder)
31         {
32             chunkHeaderGrammar->Parse(chunkHeader.Chars(), chunkHeader.Chars() + chunkHeader.Length(), 0, "", chunkSize, adder);
33         }
34         public void ParseFieldValue(const ustring& fieldValue, List<HttpFieldValue>* values)
35         {
36             fieldValueGrammar->Parse(fieldValue.Chars(), fieldValue.Chars() + fieldValue.Length(), 0, "", values);
37         }
38         public MimeType ParseMediaType(const ustring& fieldValue)
39         {
40             MimeType mediaType;
41             mediaTypeGrammar->Parse(fieldValue.Chars(), fieldValue.Chars() + fieldValue.Length(), 0, "", &mediaType);
42             return mediaType;
43         }
44         public DateTime ParseDate(const ustring& fieldValue)
45         {
46             DateTime date;
47             dateGrammar->Parse(fieldValue.Chars(), fieldValue.Chars() + fieldValue.Length(), 0, "", &date);
48             return date;
49         }
50         private HttpParser() : 
51             httpGrammar(new HttpGrammar()), 
52             headerGrammar(new HttpHeaderGrammar(httpGrammar->Domain())), 
53             statusLineGrammar(new HttpStatusLineGrammar(httpGrammar->Domain())), 
54             chunkHeaderGrammar(new HttpChunkHeaderGrammar(httpGrammar->Domain())), 
55             fieldValueGrammar(new HttpFieldValueGrammar(httpGrammar->Domain())), 
56             mediaTypeGrammar(new HttpMediaTypeGrammar(httpGrammar->Domain())), 
57             dateGrammar(new HttpDateGrammar(httpGrammar->Domain()))
58         {
59         }
60         private static UniquePtr<HttpParser> instance;
61         private HttpGrammar* httpGrammar;
62         private HttpHeaderGrammar* headerGrammar;
63         private HttpStatusLineGrammar* statusLineGrammar;
64         private HttpChunkHeaderGrammar* chunkHeaderGrammar;
65         private HttpFieldValueGrammar* fieldValueGrammar;
66         private HttpMediaTypeGrammar* mediaTypeGrammar;
67         private HttpDateGrammar* dateGrammar;
68     }
69 }
